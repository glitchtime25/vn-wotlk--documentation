193  // Function to search for new tubber in range
194  void DoFindNewTubber()
195  {
196  GameObjectList lTubbersInRange;
197  GetGameObjectListWithEntryInGrid(lTubbersInRange, m_creature, GO_BLUELEAF_TUBBER, 40.0f);
198 
199  if (lTubbersInRange.empty())
200  return;
201 
202  lTubbersInRange.sort(ObjectDistanceOrder(m_creature));
203  GameObject* pNearestTubber = nullptr;
204 
205  // Always need to find new ones
206  for (GameObjectList::const_iterator itr = lTubbersInRange.begin(); itr != lTubbersInRange.end(); ++itr)
207  {
208  if (!(*itr)->IsSpawned() && (*itr)->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_INTERACT_COND) && (*itr)->IsWithinLOSInMap(m_creature))
209  {
210  pNearestTubber = *itr;
211  break;
212  }
213  }
214 
215  if (!pNearestTubber)
216  return;
217  m_targetTubberGuid = pNearestTubber->GetObjectGuid();
218 
219  float fX, fY, fZ;
220  pNearestTubber->GetContactPoint(m_creature, fX, fY, fZ);
221  m_creature->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
222  m_bIsMovementActive = true;
223  }
224 
225  void UpdateAI(const uint32 uiDiff) override
226  {
227  if (!m_bIsMovementActive)
228  ScriptedPetAI::UpdateAI(uiDiff);
229  }
