246  bool TileAssembler::calculateTransformedBound(ModelSpawn& spawn)
247  {
248  std::string modelFilename = iSrcDir + "/" + spawn.name;
249  ModelPosition modelPosition;
250  modelPosition.iDir = spawn.iRot;
251  modelPosition.iScale = spawn.iScale;
252  modelPosition.init();
253 
254  WorldModel_Raw raw_model;
255  if (!raw_model.Read(modelFilename.c_str()))
256  return false;
257 
258  uint32 groups = raw_model.groupsArray.size();
259  if (groups != 1)
260  printf("Warning: '%s' does not seem to be a M2 model!\n", modelFilename.c_str());
261 
262  AABox modelBound;
263  bool boundEmpty = true;
264  for (uint32 g = 0; g < groups; ++g) // should be only one for M2 files...
265  {
266  std::vector<Vector3>& vertices = raw_model.groupsArray[g].vertexArray;
267 
268  if (vertices.empty())
269  {
270  std::cout << "error: model '" << spawn.name << "' has no geometry!" << std::endl;
271  continue;
272  }
273 
274  uint32 nvectors = vertices.size();
275  for (uint32 i = 0; i < nvectors; ++i)
276  {
277  Vector3 v = modelPosition.transform(vertices[i]);
278  if (boundEmpty)
279  modelBound = AABox(v, v), boundEmpty = false;
280  else
281  modelBound.merge(v);
282  }
283  }
284  spawn.iBound = modelBound + spawn.iPos;
285  spawn.flags |= MOD_HAS_BOUND;
286  return true;
287  }
