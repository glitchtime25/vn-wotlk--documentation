97  m_auiEncounter[uiType] = uiData;
98  break;
99  default:
100  return;
101  }
102 
103  if (uiData == DONE)
104  {
105  OUT_SAVE_INST_DATA;
106

108  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2];
109

220 void instance_sethekk_halls::Update(uint32 diff)
221 {

232  m_anzuTimer = 20000;

234  }

237  Creature* target = GetSingleCreatureFromStorage(NPC_INVIS_RAVEN_GOD_TARGET, true);
238  target->CastSpell(nullptr, SPELL_ULTRIS_DESTROYED, TRIGGERED_NONE);
239  m_anzuTimer = 9000;
240  break;
241  }
242  case 2:
243  {
244  Creature * target = GetSingleCreatureFromStorage(NPC_INVIS_RAVEN_GOD_TARGET, true);
245  target->CastSpell(nullptr, SPELL_RED_LIGHTNING_BOLT, TRIGGERED_NONE);
246  m_anzuTimer = 1000;

248  }
249  case 3: // done

251  break;
252  }
253  ++m_anzuStage;
254  }
255  else m_anzuTimer -= diff;

259 bool ProcessEventId_event_spell_summon_raven_god(uint32 /*uiEventId*/, Object* source, Object* /*pTarget*/, bool bIsStart)
260 {
261  if (bIsStart && source->GetTypeId() == TYPEID_PLAYER)
262  {
263  Player* player = static_cast<Player*>(source);
264  if (instance_sethekk_halls* instance = (instance_sethekk_halls*)(player->GetInstanceData()))
265  {
266  // This should be checked by despawning the Raven Claw Go; However it's better to double check the condition
267  if (instance->GetData(TYPE_ANZU) == DONE || instance->GetData(TYPE_ANZU) == IN_PROGRESS)
268  return true;
269 
270  // Don't summon him twice
271  if (instance->GetSingleCreatureFromStorage(NPC_ANZU, true))
272  return true;
273 
274  instance->StartAnzuIntro(player);
275  }
276  }
277  return false;
278 }
