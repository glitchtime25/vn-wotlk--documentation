51  void UpdateAI(const uint32 diff) override
52  {
53  // Always decrease our global cooldown first
54  if (GlobalCooldown > diff)
55  GlobalCooldown -= diff;
56  else GlobalCooldown = 0;
57 
58  // Buff timer (only buff when we are alive and not in combat
59  if (!m_creature->IsInCombat() && m_creature->IsAlive())
60  {
61  if (BuffTimer < diff)
62  {
63  // Find a spell that targets friendly and applies an aura (these are generally buffs)
64  SpellEntry const* info = SelectSpell(m_creature, -1, -1, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_AURA);
65 
66  if (info && !GlobalCooldown)
67  {
68  // Cast the buff spell
69  m_creature->CastSpell(m_creature, info, TRIGGERED_NONE);
70 
71  // Set our global cooldown
72  GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
73 
74  // Set our timer to 10 minutes before rebuff
75  BuffTimer = 600000;
76  }// Try agian in 30 seconds
77  else BuffTimer = 30000;
78  }
79  else BuffTimer -= diff;
80  }
81 
82  // Return since we have no target
83  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
84  return;
85 
86  // Return if we already cast a spell
87  if (m_creature->IsNonMeleeSpellCasted(false))
88  return;
89 
90  // If we are within range melee the target
91  if (m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
92  {
93  // Make sure our attack is ready
94  if (m_creature->isAttackReady())
95  {
96  bool Healing = false;
97  SpellEntry const* info = nullptr;
98 
99  // Select a healing spell if less than 30% hp
100  if (m_creature->GetHealthPercent() < 30.0f)
101  info = SelectSpell(m_creature, -1, -1, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
102 
103  // No healing spell available, select a hostile spell
104  if (info) Healing = true;
105  else info = SelectSpell(m_creature->GetVictim(), -1, -1, SELECT_TARGET_ANY_ENEMY, 0, 0, 0, 0, SELECT_EFFECT_DONTCARE);
106 
107  // 50% chance if elite or higher, 20% chance if not, to replace our white hit with a spell
108  if (info && (rand() % (m_creature->GetCreatureInfo()->Rank > 1 ? 2 : 5) == 0) && !GlobalCooldown)
109  {
110  // Cast the spell
111  if (Healing)
112  m_creature->CastSpell(m_creature, info, TRIGGERED_NONE);
113  else
114  m_creature->CastSpell(m_creature->GetVictim(), info, TRIGGERED_NONE);
115 
116  // Set our global cooldown
117  GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
118  }
119  else m_creature->AttackerStateUpdate(m_creature->GetVictim());
120 
121  m_creature->resetAttackTimer();
122  }
123  }
124  else
125  {
126  bool Healing = false;
127  SpellEntry const* info = nullptr;
128 
129  // Select a healing spell if less than 30% hp ONLY 33% of the time
130  if (m_creature->GetHealthPercent() < 30.0f && !urand(0, 2))
131  info = SelectSpell(m_creature, -1, -1, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
132 
133  // No healing spell available, See if we can cast a ranged spell (Range must be greater than ATTACK_DISTANCE)
134  if (info) Healing = true;
135  else info = SelectSpell(m_creature->GetVictim(), -1, -1, SELECT_TARGET_ANY_ENEMY, 0, 0, ATTACK_DISTANCE, 0, SELECT_EFFECT_DONTCARE);
136 
137  // Found a spell, check if we arn't on cooldown
138  if (info && !GlobalCooldown)
139  {
140  // If we are currently moving stop us and set the movement generator
141  if (!IsSelfRooted)
142  {
143  IsSelfRooted = true;
144  }
145 
146  // Cast spell
147  if (Healing)
148  m_creature->CastSpell(m_creature, info, TRIGGERED_NONE);
149  else
150  m_creature->CastSpell(m_creature->GetVictim(), info, TRIGGERED_NONE);
151 
152  // Set our global cooldown
153  GlobalCooldown = GENERIC_CREATURE_COOLDOWN;
154  }// If no spells available and we arn't moving run to target
155  else if (IsSelfRooted)
156  {
157  // Cancel our current spell and then allow movement agian
158  m_creature->InterruptNonMeleeSpells(false);
159  IsSelfRooted = false;
160  }
161  }
162  }
