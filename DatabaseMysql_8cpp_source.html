77 bool MySQLConnection::Initialize(const char* infoString)
78 {
79  MYSQL* mysqlInit = mysql_init(nullptr);
80  if (!mysqlInit)
81  {
82  sLog.outError("Could not initialize Mysql connection");
83  return false;
84  }
85 
86  Tokens tokens = StrSplit(infoString, ";");
87 
88  std::string host, port_or_socket, user, password, database;
89  int port;
90  char const* unix_socket;
91 
92  Tokens::iterator iter = tokens.begin();
93 
94  if (iter != tokens.end())
95  host = *iter++;
96  if (iter != tokens.end())
97  port_or_socket = *iter++;
98  if (iter != tokens.end())
99  user = *iter++;
100  if (iter != tokens.end())
101  password = *iter++;
102  if (iter != tokens.end())
103  database = *iter++;
104 
105  mysql_options(mysqlInit, MYSQL_SET_CHARSET_NAME, "utf8");
106 #ifdef _WIN32
107  if (host == ".") // named pipe use option (Windows)
108  {
109  unsigned int opt = MYSQL_PROTOCOL_PIPE;
110  mysql_options(mysqlInit, MYSQL_OPT_PROTOCOL, (char const*)&opt);
111  port = 0;
112  unix_socket = nullptr;
113  }
114  else // generic case
115  {
116  port = atoi(port_or_socket.c_str());
117  unix_socket = nullptr;
118  }
119 #else
120  if (host == ".") // socket use option (Unix/Linux)
121  {
122  unsigned int opt = MYSQL_PROTOCOL_SOCKET;
123  mysql_options(mysqlInit, MYSQL_OPT_PROTOCOL, (char const*)&opt);
124  host = "localhost";
125  port = 0;
126  unix_socket = port_or_socket.c_str();
127  }
128  else // generic case
129  {
130  port = atoi(port_or_socket.c_str());
131  unix_socket = nullptr;
132  }
133 #endif
134 
135  mMysql = mysql_real_connect(mysqlInit, host.c_str(), user.c_str(),
136  password.c_str(), database.c_str(), port, unix_socket, 0);
137 
138  if (!mMysql)
139  {
140  sLog.outError("Could not connect to MySQL database at %s: %s\n",
141  host.c_str(), mysql_error(mysqlInit));
142  mysql_close(mysqlInit);
143  return false;
144  }
145 
146  DETAIL_LOG("Connected to MySQL database %s@%s:%s/%s", user.c_str(), host.c_str(), port_or_socket.c_str(), database.c_str());
147  sLog.outString("MySQL client library: %s", mysql_get_client_info());
148  sLog.outString("MySQL server ver: %s ", mysql_get_server_info(mMysql));
149 
150  /*----------SET AUTOCOMMIT ON---------*/
151  // It seems mysql 5.0.x have enabled this feature
152  // by default. In crash case you can lose data!!!
153  // So better to turn this off
154  // ---
155  // This is wrong since mangos use transactions,
156  // autocommit is turned of during it.
157  // Setting it to on makes atomic updates work
158  // ---
159  // LEAVE 'AUTOCOMMIT' MODE ALWAYS ENABLED!!!
160  // W/O IT EVEN 'SELECT' QUERIES WOULD REQUIRE TO BE WRAPPED INTO 'START TRANSACTION'<>'COMMIT' CLAUSES!!!
161  if (!mysql_autocommit(mMysql, 1))
162  DETAIL_LOG("AUTOCOMMIT SUCCESSFULLY SET TO 1");
163  else
164  DETAIL_LOG("AUTOCOMMIT NOT SET TO 1");
165  /*-------------------------------------*/
166 
167  // set connection properties to UTF8 to properly handle locales for different
168  // server configs - core sends data in UTF8, so MySQL must expect UTF8 too
169  Execute("SET NAMES `utf8`");
170  Execute("SET CHARACTER SET `utf8`");
171 
172  return true;
173 }

175 bool MySQLConnection::_Query(const char* sql, MYSQL_RES** pResult, MYSQL_FIELD** pFields, uint64* pRowCount, uint32* pFieldCount)
176 {
177  if (!mMysql)
178  return false;
179 
180  uint32 _s = WorldTimer::getMSTime();
181 
182  if (mysql_query(mMysql, sql))
183  {
184  sLog.outErrorDb("SQL: %s", sql);
185  sLog.outErrorDb("query ERROR: %s", mysql_error(mMysql));
186  return false;
187  }
188  DEBUG_FILTER_LOG(LOG_FILTER_SQL_TEXT, "[%u ms] SQL: %s", WorldTimer::getMSTimeDiff(_s, WorldTimer::getMSTime()), sql);
189 
190  *pResult = mysql_store_result(mMysql);
191  *pRowCount = mysql_affected_rows(mMysql);
192  *pFieldCount = mysql_field_count(mMysql);
193 
194  if (!*pResult)
195  return false;
196 
197  if (!*pRowCount)
198  {
199  mysql_free_result(*pResult);
200  return false;
201  }
202 
203  *pFields = mysql_fetch_fields(*pResult);
204  return true;

316 bool MySqlPreparedStatement::prepare()
317 {
318  if (isPrepared())
319  return true;
320 
321  // remove old binds
322  RemoveBinds();
323 
324  // create statement object
325  m_stmt = mysql_stmt_init(m_pMySQLConn);
326  if (!m_stmt)
327  {
328  sLog.outError("SQL: mysql_stmt_init() failed ");
329  return false;
330  }
331 
332  // prepare statement
333  if (mysql_stmt_prepare(m_stmt, m_szFmt.c_str(), m_szFmt.length()))
334  {
335  sLog.outError("SQL: mysql_stmt_prepare() failed for '%s'", m_szFmt.c_str());
336  sLog.outError("SQL ERROR: %s", mysql_stmt_error(m_stmt));
337  return false;
338  }
339 
340  /* Get the parameter count from the statement */
341  m_nParams = mysql_stmt_param_count(m_stmt);
342 
343  /* Fetch result set meta information */
344  m_pResultMetadata = mysql_stmt_result_metadata(m_stmt);
345  // if we do not have result metadata
346  if (!m_pResultMetadata && strnicmp(m_szFmt.c_str(), "select", 6) == 0)
347  {
348  sLog.outError("SQL: no meta information for '%s'", m_szFmt.c_str());
349  sLog.outError("SQL ERROR: %s", mysql_stmt_error(m_stmt));
350  return false;
351  }
352 
353  // bind input buffers
354  if (m_nParams)
355  {
356  m_pInputArgs = new MYSQL_BIND[m_nParams];
357  memset(m_pInputArgs, 0, sizeof(MYSQL_BIND) * m_nParams);
358  }
359 
360  // check if we have a statement which returns result sets
361  if (m_pResultMetadata)
362  {
363  // our statement is query
364  m_bIsQuery = true;
365  /* Get total columns in the query */
366  m_nColumns = mysql_num_fields(m_pResultMetadata);
367 
368  // bind output buffers
369  }
370 
371  m_bPrepared = true;
372  return true;
373 }

375 void MySqlPreparedStatement::bind(const SqlStmtParameters& holder)
376 {
377  if (!isPrepared())
378  {
379  MANGOS_ASSERT(false);
380  return;
381  }
382 
383  // finalize adding params
384  if (!m_pInputArgs)
385  return;
386 
387  // verify if we bound all needed input parameters
388  if (m_nParams != holder.boundParams())
389  {
390  MANGOS_ASSERT(false);
391  return;
392  }
393 
394  unsigned int nIndex = 0;
395  SqlStmtParameters::ParameterContainer const& _args = holder.params();
396 
397  SqlStmtParameters::ParameterContainer::const_iterator iter_last = _args.end();
398  for (SqlStmtParameters::ParameterContainer::const_iterator iter = _args.begin(); iter != iter_last; ++iter)
399  {
400  // bind parameter
401  addParam(nIndex++, (*iter));
402  }
403 
404  // bind input arguments
405  if (mysql_stmt_bind_param(m_stmt, m_pInputArgs))
406  {
407  sLog.outError("SQL ERROR: mysql_stmt_bind_param() failed\n");
408  sLog.outError("SQL ERROR: %s", mysql_stmt_error(m_stmt));
409  }
410 }

412 void MySqlPreparedStatement::addParam(unsigned int nIndex, const SqlStmtFieldData& data)
413 {

464 enum_field_types MySqlPreparedStatement::ToMySQLType(const SqlStmtFieldData& data, bool& bUnsigned)
465 {
466  bUnsigned = 0;
467  enum_field_types dataType = MYSQL_TYPE_NULL;
468 
469  switch (data.type())
470  {
471  case FIELD_NONE: dataType = MYSQL_TYPE_NULL; break;
472  // MySQL does not support MYSQL_TYPE_BIT as input type
473  case FIELD_BOOL: // dataType = MYSQL_TYPE_BIT; bUnsigned = 1; break;
474  case FIELD_UI8: dataType = MYSQL_TYPE_TINY; bUnsigned = 1; break;
475  case FIELD_I8: dataType = MYSQL_TYPE_TINY; break;
476  case FIELD_I16: dataType = MYSQL_TYPE_SHORT; break;
477  case FIELD_UI16: dataType = MYSQL_TYPE_SHORT; bUnsigned = 1; break;
478  case FIELD_I32: dataType = MYSQL_TYPE_LONG; break;
479  case FIELD_UI32: dataType = MYSQL_TYPE_LONG; bUnsigned = 1; break;
480  case FIELD_I64: dataType = MYSQL_TYPE_LONGLONG; break;
481  case FIELD_UI64: dataType = MYSQL_TYPE_LONGLONG; bUnsigned = 1; break;
482  case FIELD_FLOAT: dataType = MYSQL_TYPE_FLOAT; break;
483  case FIELD_DOUBLE: dataType = MYSQL_TYPE_DOUBLE; break;
484  case FIELD_STRING: dataType = MYSQL_TYPE_STRING; break;
485  }
486 
487  return dataType;
