140  void UpdateAI(const uint32 uiDiff) override
141  {
142  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
143  return;
144 
145  switch (m_uiPhase)
146  {
147  case PHASE_GROUND:
148 
149  if (m_uiFlameBreathTimer < uiDiff)
150  {
151  if (DoCastSpellIfCan(m_creature, SPELL_FLAME_BREATH) == CAST_OK)
152  m_uiFlameBreathTimer = urand(20000, 25000);
153  }
154  else
155  m_uiFlameBreathTimer -= uiDiff;
156 
157  if (m_uiEnrageTimer < uiDiff)
158  {
159  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
160  {
161  DoScriptText(EMOTE_ENRAGE, m_creature);
162  m_uiEnrageTimer = urand(20000, 25000);
163  }
164  }
165  else
166  m_uiEnrageTimer -= uiDiff;
167 
168  if (m_uiPhaseSwitchTimer < uiDiff)
169  {
170  m_uiPhaseSwitchTimer = 0;
171  m_uiPhase = PHASE_TRANSITION;
172 
173  SetCombatMovement(false);
174  m_creature->SetByteValue(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_FLY_ANIM);
175  m_creature->SetLevitate(true);
176 
177  m_creature->SetWalk(false);
178  m_creature->GetMotionMaster()->Clear();
179  m_creature->GetMotionMaster()->MovePoint(POINT_AIR, aAirPositions[0], aAirPositions[1], aAirPositions[2]);
180  }
181  else
182  m_uiPhaseSwitchTimer -= uiDiff;
183 
184  DoMeleeAttackIfReady();
185 
186  break;
187  case PHASE_AIR:
188  if (m_uiPhaseSwitchTimer)
189  {
190  if (m_uiPhaseSwitchTimer <= uiDiff)
191  {
192  m_uiPhase = PHASE_TRANSITION;
193  m_uiPhaseSwitchTimer = 0;
194 
195  float fX, fY, fZ;
196  m_creature->GetRespawnCoord(fX, fY, fZ);
197  m_creature->SetWalk(false);
198  m_creature->GetMotionMaster()->Clear();
199  m_creature->GetMotionMaster()->MovePoint(POINT_GROUND, fX, fY, fZ);
200  }
201  else
202  m_uiPhaseSwitchTimer -= uiDiff;
203  }
204  break;
205  case PHASE_TRANSITION:
206  // nothing here
207  break;
208  }
209  }
