151 void instance_sunken_temple::SetData(uint32 uiType, uint32 uiData)
152 {
153  switch (uiType)
154  {
155  case TYPE_ATALARION:
156  if (uiData == SPECIAL)
157  DoSpawnAtalarionIfCan();
158  m_auiEncounter[uiType] = uiData;
159  break;
160  case TYPE_PROTECTORS:
161  if (uiData == DONE)
162  {
163  --m_uiProtectorsRemaining;
164  if (!m_uiProtectorsRemaining)
165  {
166  m_auiEncounter[uiType] = uiData;
167  DoUseDoorOrButton(GO_JAMMALAN_BARRIER);
168  // Intro yell
169  DoOrSimulateScriptTextForThisInstance(SAY_JAMMALAN_INTRO, NPC_JAMMALAN);
170 
171  if (Creature* jammalan = GetSingleCreatureFromStorage(NPC_JAMMALAN))
172  jammalan->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
173 
174  if (Creature* ogom = GetSingleCreatureFromStorage(NPC_OGOM))
175  ogom->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
176  }
177  }
178  break;
179  case TYPE_JAMMALAN:
180  if (uiData == DONE)
181  {
182  if (Creature* pEranikus = GetSingleCreatureFromStorage(NPC_SHADE_OF_ERANIKUS))
183  pEranikus->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
184  }
185  m_auiEncounter[uiType] = uiData;
186  break;
187  case TYPE_MALFURION:
188  if (uiData == IN_PROGRESS)
189  m_auiEncounter[uiType] = uiData;
190  break;
191  case TYPE_AVATAR:
192  if (uiData == SPECIAL)
193  {
194  Creature* shadeofHakkar = GetSingleCreatureFromStorage(NPC_SHADE_OF_HAKKAR);
195  if (!shadeofHakkar)
196  return;
197 
198  if (shadeofHakkar->HasAura(SPELL_SUPPRESSION) && !m_suppressionTimer)
199  {
200  m_suppressionTimer = 20 * IN_MILLISECONDS;
201  return; // Return to avoid handling one of the flames dousing
202  }
203 
204  ++m_uiFlameCounter;
205 
206  switch (m_uiFlameCounter)
207  {
208  // Yells on each flame
209  // TODO It might be possible that these yells should be ordered randomly, however this is the seen state
210  case 1: DoScriptText(SAY_AVATAR_BRAZIER_1, shadeofHakkar); break;
211  case 2: DoScriptText(SAY_AVATAR_BRAZIER_2, shadeofHakkar); break;
212  case 3: DoScriptText(SAY_AVATAR_BRAZIER_3, shadeofHakkar); break;
213  // Summon the avatar when all flames are used
214  case MAX_FLAMES:
215  DoScriptText(SAY_AVATAR_BRAZIER_4, shadeofHakkar);
216  shadeofHakkar->CastSpell(shadeofHakkar, SPELL_SUMMON_AVATAR, TRIGGERED_OLD_TRIGGERED);
217  m_uiAvatarSummonTimer = 0;
218  m_uiSupressorTimer = 0;
219  break;
220  }
221 
222  // Summon the suppressors only after the flames are doused
223  // Summon timer is confusing random; timers were: 13, 39 and 52 secs;
224  if (m_uiFlameCounter != MAX_FLAMES)
225  m_uiSupressorTimer = urand(15000, 45000);
226 
227  return;
228  }
229 
230  // Prevent double processing
231  if (m_auiEncounter[uiType] == uiData)
232  return;
233 
234  if (uiData == IN_PROGRESS)
235  {
236  m_uiSupressorTimer = 0;
237  DoUpdateFlamesFlags(false);
238 
239  // Summon timer; use a small delay
240  m_uiAvatarSummonTimer = 3000;
241  m_bIsFirstHakkarWave = true;
242 
243  // Summon the shade
244  Player* pPlayer = GetPlayerInMap();
245  if (!pPlayer)
246  return;
247 
248  if (Creature* pShade = pPlayer->SummonCreature(NPC_SHADE_OF_HAKKAR, aSunkenTempleLocation[1].m_fX, aSunkenTempleLocation[1].m_fY, aSunkenTempleLocation[1].m_fZ, aSunkenTempleLocation[1].m_fO, TEMPSPAWN_MANUAL_DESPAWN, 0))
249  {
250  m_npcEntryGuidStore[NPC_SHADE_OF_HAKKAR] = pShade->GetObjectGuid();
251  pShade->SetRespawnDelay(DAY);
252  }
253 
254  // Respawn evil circle visuals
255  for (auto evilCircleGuid : m_evilCircleGuidList)
256  DoRespawnGameObject(evilCircleGuid, 30 * MINUTE);
257  }
258  else if (uiData == FAIL)
259  {
260  // In case of wipe during the summoning ritual the shade is despawned
261  // The trash mobs stay in place, they are not despawned
262 
263  // Despawn the shade and the avatar if needed
264  if (Creature* pShade = GetSingleCreatureFromStorage(NPC_SHADE_OF_HAKKAR))
265  pShade->ForcedDespawn();
266 
267  // Reset flames - remove no interact flag
268  DoUpdateFlamesFlags(false);
269  m_uiFlameCounter = 0;
270 
271  // Despawn the evil circle visuals
272  for (auto evilCircleGuid : m_evilCircleGuidList)
273  {
274  if (GameObject* evilCircle = instance->GetGameObject(evilCircleGuid))
275  {
276  evilCircle->SetForcedDespawn();
277  evilCircle->SetLootState(GO_JUST_DEACTIVATED);
278  }
279  }
280  }
281 
282  // Use combat doors
283  DoUseDoorOrButton(GO_HAKKAR_DOOR_1);
284  DoUseDoorOrButton(GO_HAKKAR_DOOR_2);
285 
286  m_auiEncounter[uiType] = uiData;
287 
288  break;
289  }
290 
291  if (uiData == DONE)
292  {
293  OUT_SAVE_INST_DATA;
294 
295  std::ostringstream saveStream;
296 
297  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
298  << m_auiEncounter[3] << " " << m_auiEncounter[4];
299 
300  m_strInstData = saveStream.str();
301 
302  SaveToDB();
303  OUT_SAVE_INST_DATA_COMPLETE;
304  }

327 bool instance_sunken_temple::ProcessStatueEvent(uint32 uiEventId)
328 {
329  bool bEventStatus = false;
330 
331  // Check if the statues are activated correctly
332  // Increase the counter when the correct statue is activated
333  for (uint8 i = 0; i < MAX_STATUES; ++i)
334  {
335  if (uiEventId == m_aAtalaiStatueEvents[i] && m_uiStatueCounter == i)
336  {
337  // Right Statue activated
338  ++m_uiStatueCounter;
339  bEventStatus = true;
340  break;
341  }
342  }
343 
344  if (!bEventStatus)
345  return false;
346 
347  // Check if all statues are active
348  if (m_uiStatueCounter == MAX_STATUES)
349  SetData(TYPE_ATALARION, SPECIAL);
350 
351  return true;

391 void instance_sunken_temple::Update(uint32 uiDiff)
392 {
393  if (m_auiEncounter[TYPE_AVATAR] != IN_PROGRESS)
394  return;
395 
396  if (m_suppressionTimer)
397  {
398  if (m_suppressionTimer < uiDiff)
399  {
400  // At the end of the suppression timer, if Shade of Hakkar still has suppression aura : reset event
401  if (Creature* shadeOfHakkar = GetSingleCreatureFromStorage(NPC_SHADE_OF_HAKKAR))
402  {
403  if (shadeOfHakkar->HasAura(SPELL_SUPPRESSION))
404  {
405  DoScriptText(SAY_AVATAR_DESPAWN, shadeOfHakkar, nullptr);
406  SetData(TYPE_AVATAR, FAIL);
407  }
408  }
409  m_suppressionTimer = 0;
410  }
411  else
412  m_suppressionTimer -= uiDiff;
413  }
414 
415  // Summon random mobs around the circles
416  if (m_uiAvatarSummonTimer)
417  {
418  if (m_uiAvatarSummonTimer <= uiDiff)
419  {
420  Creature* pShade = GetSingleCreatureFromStorage(NPC_SHADE_OF_HAKKAR);
421  if (!pShade)
422  return;
423 
424  // If no summon circles are spawned then return
425  if (m_evilCircleGuidList.empty())
426  return;
427 
428  if (m_bIsFirstHakkarWave) // First wave summoned
429  {
430  // Summon at all circles
431  for (GuidVector::const_iterator itr = m_evilCircleGuidList.begin(); itr != m_evilCircleGuidList.end(); ++itr)
432  {
433  if (GameObject* pCircle = instance->GetGameObject(*itr))
434  pShade->SummonCreature(NPC_HAKKARI_MINION, pCircle->GetPositionX(), pCircle->GetPositionY(), pCircle->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
435  }
436 
437  // Summon Bloodkeeper at random circle
438  if (GameObject* pCircle = instance->GetGameObject(m_evilCircleGuidList[urand(0, m_evilCircleGuidList.size() - 1)]))
439  pShade->SummonCreature(NPC_BLOODKEEPER, pCircle->GetPositionX(), pCircle->GetPositionY(), pCircle->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
440 
441  m_bCanSummonBloodkeeper = false;
442  m_bIsFirstHakkarWave = false;
443  m_uiAvatarSummonTimer = 50000;
444  }
445  else // Later wave
446  {
447  uint32 uiRoll = urand(0, 99);
448  uint8 uiMaxSummons = uiRoll < 75 ? 1 : uiRoll < 95 ? 2 : 3;
449 
450  if (m_bCanSummonBloodkeeper && roll_chance_i(30))
451  {
452  // Summon a Bloodkeeper
453  if (GameObject* pCircle = instance->GetGameObject(m_evilCircleGuidList[urand(0, m_evilCircleGuidList.size() - 1)]))
454  pShade->SummonCreature(NPC_BLOODKEEPER, pCircle->GetPositionX(), pCircle->GetPositionY(), pCircle->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
455 
456  m_bCanSummonBloodkeeper = false;
457  --uiMaxSummons;
458  }
459 
460  for (uint8 i = 0; i < uiMaxSummons; ++i)
461  {
462  if (GameObject* pCircle = instance->GetGameObject(m_evilCircleGuidList[urand(0, m_evilCircleGuidList.size() - 1)]))
463  pShade->SummonCreature(NPC_HAKKARI_MINION, pCircle->GetPositionX(), pCircle->GetPositionY(), pCircle->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
464  }
465  m_uiAvatarSummonTimer = urand(3000, 15000);
466  }
467  }
468  else
469  m_uiAvatarSummonTimer -= uiDiff;
470  }
471 
472  // Summon nightmare suppressor after flame used
473  if (m_uiSupressorTimer)
474  {
475  if (m_uiSupressorTimer <= uiDiff)
476  {
477  Creature* pShade = GetSingleCreatureFromStorage(NPC_SHADE_OF_HAKKAR);
478  if (!pShade)
479  {
480  // Something went very wrong!
481  return;
482  }
483 
484  // Summon npc at random door; movement and script handled in DB
485  uint8 uiSummonLoc = urand(0, 1);
486  if (Creature* suppressor = pShade->SummonCreature(NPC_SUPPRESSOR, aHakkariDoorLocations[uiSummonLoc].m_fX, aHakkariDoorLocations[uiSummonLoc].m_fY, aHakkariDoorLocations[uiSummonLoc].m_fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0))
487  {
488  suppressor->GetMotionMaster()->MoveWaypoint(uiSummonLoc);
489  uint32 sayTextId;
490  switch (urand(0, 3))
491  {
492  case 0:
493  sayTextId = SAY_SUPPRESSOR_1;
494  break;
495  case 1:
496  sayTextId = SAY_SUPPRESSOR_2;
497  break;
498  case 2:
499  sayTextId = SAY_SUPPRESSOR_3;
500  break;
501  default:
502  sayTextId = SAY_SUPPRESSOR_4;
503  break;
504  }
505  DoScriptText(sayTextId, suppressor, nullptr);
506  }
507 
508  // This timer is finished now
509  m_uiSupressorTimer = 0;
510  }
511  else
512  m_uiSupressorTimer -= uiDiff;
513  }
