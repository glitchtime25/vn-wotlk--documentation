283  void DoSpawn(uint32 spawnId)
284  {
285  Position spawnPos;
286  uint32 spawnScriptTextId = 0;
287  uint32 formationCreatureEntry = 0;
288  uint32 despawnTimer = 0;
289  TempSpawnType spawnType = TEMPSPAWN_DEAD_DESPAWN;
290 
291  switch (spawnId)
292  {
293  case NPC_DREADKNIGHT:
294  case NPC_FELGUARD_LIEUTENANT:
295  spawnPos = spawnBossPortalPos;
296  break;
297  case NPC_LORD_MARSHAL_RAYNOR:
298  spawnPos = spawnReinforcementPos;
299  despawnTimer = 3 * MINUTE * IN_MILLISECONDS;
300  spawnType = TEMPSPAWN_TIMED_OR_DEAD_DESPAWN;
301  formationCreatureEntry = NPC_STORMWIND_MARSHAL;
302  spawnScriptTextId = LMR_SPAWN;
303  break;
304  case NPC_JUSTINIUS_THE_HARBINGER:
305  spawnPos = spawnReinforcementPos;
306  despawnTimer = 3 * MINUTE * IN_MILLISECONDS;
307  spawnType = TEMPSPAWN_TIMED_OR_DEAD_DESPAWN;
308  formationCreatureEntry = NPC_AZUREMYST_VINDICATOR;
309  spawnScriptTextId = JTH_SPAWN;
310  break;
311  case NPC_MELGROMM_HIGHMOUNTAIN:
312  spawnPos = spawnReinforcementPos;
313  despawnTimer = 3 * MINUTE * IN_MILLISECONDS;
314  spawnType = TEMPSPAWN_TIMED_OR_DEAD_DESPAWN;
315  formationCreatureEntry = NPC_THUNDER_BLUFF_HUNTSMAN;
316  spawnScriptTextId = MH_SPAWN;
317  break;
318  default:
319  spawnPos = urand(0, 1) ? spawnPortalPos : spawnPortalPos1;
320  break;
321  }
322 
323  if (Creature* pProudwell = GetSingleCreatureFromStorage(NPC_AGENT_PROUDWELL, true))
324  {
325  if (Creature* summon = pProudwell->SummonCreature(spawnId, spawnPos.x, spawnPos.y, spawnPos.z, spawnPos.o, spawnType, despawnTimer))
326  {
327  if (spawnScriptTextId)
328  DoScriptText(spawnScriptTextId, summon);
329 
330  if (formationCreatureEntry)
331  {
332  if (Creature* add = summon->SummonCreature(formationCreatureEntry, summon->GetPositionX(), summon->GetPositionY(), summon->GetPositionZ(), 0.0f, spawnType, despawnTimer))
333  add->GetMotionMaster()->MoveFollow(summon, 5.f, 220.f * float(M_PI) / 180.0f, true);
334 
335  if (Creature* add = summon->SummonCreature(formationCreatureEntry, summon->GetPositionX(), summon->GetPositionY(), summon->GetPositionZ(), 0.0f, spawnType, despawnTimer))
336  add->GetMotionMaster()->MoveFollow(summon, 5.f, 140.f * float(M_PI) / 180.0f, true);
337  }
338  }
339  }

362  creature->GetMotionMaster()->MoveWaypoint();

365  m_brewfestEvent.StartDarkIronAttackEvent();
366  else if (event_id == GAME_EVENT_BREWFEST_KEG_TAPPING && activate)
