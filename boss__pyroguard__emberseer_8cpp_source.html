99  // Wrapper to handle the transformation
100  void DoHandleEmberseerGrowing()
101  {
102  ++m_uiGrowingStacks;
103 
104  if (m_uiGrowingStacks == MAX_GROWING_STACKS * 0.5f)
105  DoScriptText(EMOTE_NEAR, m_creature);
106  else if (m_uiGrowingStacks == MAX_GROWING_STACKS)
107  {
108  DoScriptText(EMOTE_FULL, m_creature);
109  DoScriptText(SAY_FREE, m_creature);
110 
111  // Note: the spell order needs further research
112  DoCastSpellIfCan(m_creature, SPELL_FULL_STRENGHT, CAST_TRIGGERED);
113  DoCastSpellIfCan(m_creature, SPELL_BONUS_DAMAGE, CAST_TRIGGERED);
114  DoCastSpellIfCan(m_creature, SPELL_TRANSFORM, CAST_TRIGGERED);
115 
116  // activate all runes
117  if (m_pInstance)
118  {
119  m_pInstance->DoUseEmberseerRunes();
120  // Redundant check: if for some reason the event isn't set in progress until this point - avoid using the altar again when the boss is fully grown
121  m_pInstance->SetData(TYPE_EMBERSEER, IN_PROGRESS);
122  }
123 
124  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
125  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
126  }
127  }

129  void UpdateAI(const uint32 uiDiff) override
130  {
131  // Cast Encage spell on OOC timer
132  if (m_uiEncageTimer)
133  {
134  if (m_uiEncageTimer <= uiDiff)
135  {
136  if (!m_pInstance)
137  {
138  script_error_log("Instance Blackrock Spire: ERROR Failed to load instance data for this instace.");
139  return;
140  }
141 
142  GuidList m_lIncarceratorsGuid;
143  m_pInstance->GetIncarceratorGUIDList(m_lIncarceratorsGuid);
144 
145  for (GuidList::const_iterator itr = m_lIncarceratorsGuid.begin(); itr != m_lIncarceratorsGuid.end(); ++itr)
146  {
147  if (Creature* pIncarcerator = m_creature->GetMap()->GetCreature(*itr))
148  pIncarcerator->CastSpell(m_creature, SPELL_ENCAGE_EMBERSEER, TRIGGERED_NONE);
149  }
150 
151  m_uiEncageTimer = 0;
152  }
153  else
154  m_uiEncageTimer -= uiDiff;
155  }
156 
157  // Return since we have no target
158  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
159  return;
160 
161  // FireNova Timer
162  if (m_uiFireNovaTimer < uiDiff)
163  {
164  if (DoCastSpellIfCan(m_creature, SPELL_FIRENOVA) == CAST_OK)
165  m_uiFireNovaTimer = 6000;
166  }
167  else
168  m_uiFireNovaTimer -= uiDiff;
169 
170  // FlameBuffet Timer
171  if (m_uiFlameBuffetTimer < uiDiff)
172  {
173  if (DoCastSpellIfCan(m_creature, SPELL_FLAMEBUFFET) == CAST_OK)
174  m_uiFlameBuffetTimer = 14000;
175  }
176  else
177  m_uiFlameBuffetTimer -= uiDiff;
178 
179  // PyroBlast Timer
180  if (m_uiPyroBlastTimer < uiDiff)
181  {
182  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
183  {
184  if (DoCastSpellIfCan(pTarget, SPELL_PYROBLAST) == CAST_OK)
185  m_uiPyroBlastTimer = 15000;
186  }
187  }
188  else
189  m_uiPyroBlastTimer -= uiDiff;
190 
191  DoMeleeAttackIfReady();
192  }
193 };
