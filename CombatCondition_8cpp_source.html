129 bool CombatConditionMgr::Meets(Unit const* self, int32 Id, float range)
130 {
131  auto combatConditions = self->GetMap()->GetMapDataContainer().GetCombatConditions();
132  auto itr = combatConditions->find(Id);
133  if (combatConditions->end() == itr)
134  return false;
135 
136  CombatConditionEntry& entry = itr->second;
137  if (entry.WorldStateConditionId)
138  {
139  if (!m_worldStateExpressionMgr.Meets(self->GetMap(), entry.WorldStateConditionId))
140  return false;
141  }
142 
143  if (entry.SelfConditionId)
144  if (!m_unitConditionMgr.Meets(self, self, entry.SelfConditionId))
145  return false;
146 
147  if (entry.TargetConditionId)
148  if (!self->GetTarget() || !m_unitConditionMgr.Meets(self->GetTarget(), self, entry.SelfConditionId))
149  return false;
150 
151  if (entry.FriendConditionId[0])
152  {
153  uint32 eligibleUnits[2];
154  memset(eligibleUnits, 0, sizeof(eligibleUnits));
155 
156  UnitList friendlies;
157  MaNGOS::AnyFriendlyUnitInObjectRangeCheck u_check(self, range);
158  MaNGOS::UnitListSearcher<MaNGOS::AnyFriendlyUnitInObjectRangeCheck> searcher(friendlies, u_check);
159  Cell::VisitGridObjects(self, searcher, range);
160 
161  for (Unit* friendly : friendlies)
162  for (uint32 i = 0; i < 2; ++i)
163  if (entry.FriendConditionId[i] && m_unitConditionMgr.Meets(friendly, self, entry.FriendConditionId[i]))
164  ++eligibleUnits[i];
165 
166  switch (ConditionLogic(entry.FriendConditionLogic))
167  {
168  case ConditionLogic::NONE:
169  if (!ConditionEntry::CheckOp(ConditionOperation(entry.FriendConditionOp[0]), eligibleUnits[0], entry.FriendConditionCount[0]))
170  return false;
171  break;
172  case ConditionLogic::AND:
173  case ConditionLogic::OR:
174  case ConditionLogic::XOR:
175  bool results[2];
176  for (uint32 i = 0; i < 2; ++i)
177  results[i] = ConditionEntry::CheckOp(ConditionOperation(entry.FriendConditionOp[i]), eligibleUnits[i], entry.FriendConditionCount[i]);
178 
179  switch (ConditionLogic(entry.FriendConditionLogic))
180  {
181  case ConditionLogic::AND: if (!results[0] || !results[1]) return false; break;
182  case ConditionLogic::OR: if (!results[0] && !results[1]) return false; break;
183  case ConditionLogic::XOR: if (results[0] == results[1]) return false; break;
184  }
185  break;
186  }
187  }
188 
189  if (entry.EnemyConditionId[0])
190  {
191  uint32 eligibleUnits[2];
192  memset(eligibleUnits, 0, sizeof(eligibleUnits));
193  if (!self->CanHaveThreatList())
194  {
195  for (auto const& ref : self->getHostileRefManager())
196  if (Unit* victim = ref.getSource()->getOwner())
197  for (uint32 i = 0; i < 2; ++i)
198  if (entry.EnemyConditionId[i] && m_unitConditionMgr.Meets(victim, self, entry.EnemyConditionId[i]))
199  ++eligibleUnits[i];
200  }
201  else
202  {
203  for (auto ref : self->getThreatManager().getThreatList())
204  if (Unit* victim = ref->getTarget())
205  for (uint32 i = 0; i < 2; ++i)
206  if (entry.EnemyConditionId[i] && m_unitConditionMgr.Meets(victim, self, entry.EnemyConditionId[i]))
207  ++eligibleUnits[i];
208  }
209 
210  switch (ConditionLogic(entry.EnemyConditionLogic))
211  {
212  case ConditionLogic::NONE:
213  if (!ConditionEntry::CheckOp(ConditionOperation(entry.EnemyConditionOp[0]), eligibleUnits[0], entry.EnemyConditionCount[0]))
214  return false;
215  break;
216  case ConditionLogic::AND:
217  case ConditionLogic::OR:
218  case ConditionLogic::XOR:
219  bool results[2];
220  for (uint32 i = 0; i < 2; ++i)
221  results[i] = ConditionEntry::CheckOp(ConditionOperation(entry.EnemyConditionOp[i]), eligibleUnits[i], entry.EnemyConditionCount[i]);
222 
223  switch (ConditionLogic(entry.EnemyConditionLogic))
224  {
225  case ConditionLogic::AND: if (!results[0] || !results[1]) return false; break;
226  case ConditionLogic::OR: if (!results[0] && !results[1]) return false; break;
227  case ConditionLogic::XOR: if (results[0] == results[1]) return false; break;
228  }
229  break;
230  }
231  }
232 
233  return true;
