495  }
496  else
497  m_uiTransformEmote_Timer -= uiDiff;
498  }
499 
500  if (m_uiTransform_Timer < uiDiff)
501  {
502  m_bTransform = false;
503  Transform();
504  }
505  else
506  m_uiTransform_Timer -= uiDiff;
507  }
508 
510  if (m_uiDespawn_Timer)
511  {
512  if (m_uiDespawn_Timer <= uiDiff)
513  {
514  if (m_creature->IsAlive() && !m_creature->IsInCombat())
515  DemonDespawn(false);
516  }
517  else
518  m_uiDespawn_Timer -= uiDiff;
519  }
520 
521  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
522  return;
523 
524  if (m_creature->getThreatManager().getThreatList().size() > 1 /*|| pHunter->IsDead()*/)
525  DemonDespawn();
526 
527  if (m_uiDemonic_Frenzy_Timer < uiDiff)
528  {
529  if (DoCastSpellIfCan(m_creature, SPELL_DEMONIC_FRENZY) == CAST_OK)
530  {
531  DoScriptText(EMOTE_FRENZY, m_creature);
532  m_uiDemonic_Frenzy_Timer = 15000;
533  }
534  }
535  else

537 
538  DoMeleeAttackIfReady();
