319  void UpdateEscortAI(const uint32 uiDiff) override
320  {
321  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
322  return;
323 
324  // special script for snake fight
325  if (m_uiActivateMummiesTimer)
326  {
327  if (m_uiActivateMummiesTimer <= uiDiff)
328  {
329  if (Creature* pTecahuna = m_creature->GetMap()->GetCreature(m_tecahunaGuid))
330  {
331  // activate 2 mummies at each turn
332  for (uint8 i = 0; i < 2; ++i)
333  {
334  if (Creature* pBunny = m_creature->GetMap()->GetCreature(m_lImmolationBunnyGuids.front()))
335  {
336  pTecahuna->CastSpell(pBunny, SPELL_TECAHUNA_SPIRIT_BEAM, TRIGGERED_OLD_TRIGGERED);
337  pBunny->CastSpell(pBunny, SPELL_SUMMON_DRAKKARI_KING, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
338  pBunny->RemoveAurasDueToSpell(SPELL_BUNNY_IMMOLATION);
339  m_lImmolationBunnyGuids.remove(m_lImmolationBunnyGuids.front());
340  }
341  }
342  }
343 
344  // set timer based on the remaining mummies
345  if (m_lImmolationBunnyGuids.empty())
346  m_uiActivateMummiesTimer = 0;
347  else
348  m_uiActivateMummiesTimer = urand(5000, 10000);
349  }
350  else
351  m_uiActivateMummiesTimer -= uiDiff;
352  }
353 
354  DoMeleeAttackIfReady();
355  }
