111  void WaypointReached(uint32 pointId) override
112  {
113  switch (pointId)
114  {
115  case 4:
116  {
117  // Start summoning the three waves (everything else is handled in DoSendWave())
118  m_wave = 0;
119  ResetTimer(1, 1 * IN_MILLISECONDS);
120  m_creature->HandleEmote(EMOTE_STATE_USESTANDING_NOSHEATHE);
121  break;
122  }
123  case 5:
124  {
125  // Take her gun
126  SetEquipmentSlots(false, EQUIP_NO_CHANGE, EQUIP_NO_CHANGE, EQUIP_ID_RIFLE);
127  m_creature->SetSheath(SHEATH_STATE_RANGED);
128  m_creature->HandleEmote(EMOTE_STATE_STAND);
129  break;
130  }
131  case 7:
132  {
133  // Stay put and wait for the Defias to come down
134  SetRun(false);
135  DoScriptText(YELL_DAPHNE_START, m_creature);
136  SetCombatMovement(false);
137  SetEscortPaused(true);
138  break;
139  }
140  case 8:
141  {
142  // Cheer after defeating the Defias
143  DoScriptText(SAY_DS_DOWN_3, m_creature);
144  m_creature->HandleEmote(EMOTE_ONESHOT_CHEER);
145  break;
146  }
147  case 9:
148  {
149  DoScriptText(SAY_DS_PROLOGUE, m_creature);
150  SetCombatMovement(true);
151  break;
152  }
153  case 12:
154  {
155  SetEquipmentSlots(true);
156  m_creature->SetSheath(SHEATH_STATE_UNARMED);
157  m_creature->HandleEmote(EMOTE_STATE_USESTANDING);
158  break;
159  }
160  case 13:
161  {
162  m_creature->HandleEmote(EMOTE_STATE_STAND);
163  break;
164  }
165  case 16:
166  {
167  if (Player* player = GetPlayerForEscort())
168  player->RewardPlayerAndGroupAtEventExplored(QUEST_TOME_VALOR, m_creature);
169  break;
170  }
171  case 17:
172  {
173  DoEndEscort();
174  break;
175  }
176  default:
177  break;
178  }
179  }

181  void DoSendWave()
182  {
183  if (m_wave < 0 || m_wave > 2)
184  return;
185 
186  ++m_wave;
187 
188  uint8 firstWpOffset = 5;
189  uint8 numberOfAdds = m_wave + 2;
190 
191  for (int counter = 0; counter < numberOfAdds; counter++)
192  {
193  if (Creature* add = m_creature->SummonCreature(NPC_DEFIAS_RAIDER, RaiderCoords[counter][0], RaiderCoords[counter][1], RaiderCoords[counter][2], 0, TEMPSPAWN_TIMED_OOC_DESPAWN, 30000, false, true))
194  {
195  add->GetMotionMaster()->MovePoint(counter, RaiderCoords[firstWpOffset + counter][0], RaiderCoords[firstWpOffset + counter][1], RaiderCoords[firstWpOffset + counter][2]);
196  if (!m_introIsDone)
197  {
198  DoScriptText(YELL_DEFIAS_START, add);
199  m_introIsDone = true;
200  }
201  // Leader of each wave does roar emote on spawn
202  if (counter == 0)
203  add->HandleEmote(EMOTE_ONESHOT_ROAR);
204  }
205  }
206 
207  // Keep summoning waves on a timer
208  if (m_wave < 3)
209  ResetTimer(1, 50 * IN_MILLISECONDS);
210  else
211  DisableTimer(1);
212  }
