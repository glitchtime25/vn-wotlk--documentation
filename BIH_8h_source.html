121  template<typename RayCallback>
122  void intersectRay(const Ray& r, RayCallback& intersectCallback, float& maxDist, bool stopAtFirst = false, bool ignoreM2Model = false) const
123  {
124  float intervalMin = -1.f;
125  float intervalMax = -1.f;
126  Vector3 org = r.origin();
127  Vector3 dir = r.direction();
128  Vector3 invDir;
129  for (int i = 0; i < 3; ++i)
130  {
131  invDir[i] = 1.f / dir[i];
132  if (G3D::fuzzyNe(dir[i], 0.0f))
133  {
134  float t1 = (bounds.low()[i] - org[i]) * invDir[i];
135  float t2 = (bounds.high()[i] - org[i]) * invDir[i];
136  if (t1 > t2)
137  std::swap(t1, t2);
138  if (t1 > intervalMin)
139  intervalMin = t1;
140  if (t2 < intervalMax || intervalMax < 0.f)
141  intervalMax = t2;
142  // intervalMax can only become smaller for other axis,
143  // and intervalMin only larger respectively, so stop early
144  if (intervalMax <= 0 || intervalMin >= maxDist)
145  return;
146  }
147  }
148 
149  if (intervalMin > intervalMax)
150  return;
151  intervalMin = std::max(intervalMin, 0.f);
152  intervalMax = std::min(intervalMax, maxDist);
153 
154  uint32 offsetFront[3];
155  uint32 offsetBack[3];
156  uint32 offsetFront3[3];
157  uint32 offsetBack3[3];
158  // compute custom offsets from direction sign bit
159 
160  for (int i = 0; i < 3; ++i)
161  {
162  offsetFront[i] = floatToRawIntBits(dir[i]) >> 31;
163  offsetBack[i] = offsetFront[i] ^ 1;
164  offsetFront3[i] = offsetFront[i] * 3;
165  offsetBack3[i] = offsetBack[i] * 3;
166 
167  // avoid always adding 1 during the inner loop
168  ++offsetFront[i];
169  ++offsetBack[i];
170  }
171 
172  StackNode stack[MAX_STACK_SIZE];
173  int stackPos = 0;
174  int node = 0;
175 
176  while (true)
177  {
178  while (true)
179  {
180  uint32 tn = tree[node];
181  uint32 axis = (tn & (3 << 30)) >> 30;
182  const bool BVH2 = (tn & (1 << 29)) != 0;
183  int offset = tn & ~(7 << 29);
184  if (!BVH2)
185  {
186  if (axis < 3)
187  {
188  // "normal" interior node
189  float tf = (intBitsToFloat(tree[node + offsetFront[axis]]) - org[axis]) * invDir[axis];
190  float tb = (intBitsToFloat(tree[node + offsetBack[axis]]) - org[axis]) * invDir[axis];
191  // ray passes between clip zones
192  if (tf < intervalMin && tb > intervalMax)
193  break;
194  int back = offset + offsetBack3[axis];
195  node = back;
196  // ray passes through far node only
197  if (tf < intervalMin)
198  {
199  intervalMin = (tb >= intervalMin) ? tb : intervalMin;
200  continue;
201  }
202  node = offset + offsetFront3[axis]; // front
203  // ray passes through near node only
204  if (tb > intervalMax)
205  {
206  intervalMax = (tf <= intervalMax) ? tf : intervalMax;
207  continue;
208  }
209  // ray passes through both nodes
210  // push back node
211  stack[stackPos].node = back;
212  stack[stackPos].tnear = (tb >= intervalMin) ? tb : intervalMin;
213  stack[stackPos].tfar = intervalMax;
214  ++stackPos;
215  // update ray interval for front node
216  intervalMax = (tf <= intervalMax) ? tf : intervalMax;
217  }
218  else
219  {
220  // leaf - test some objects
221  int n = tree[node + 1];
222  while (n > 0)
223  {
224  bool hit = intersectCallback(r, objects[offset], maxDist, stopAtFirst, ignoreM2Model);
225  if (stopAtFirst && hit) return;
226  --n;
227  ++offset;
228  }
229  break;
230  }
231  }
232  else
233  {
234  if (axis > 2)
235  return; // should not happen
236  float tf = (intBitsToFloat(tree[node + offsetFront[axis]]) - org[axis]) * invDir[axis];
237  float tb = (intBitsToFloat(tree[node + offsetBack[axis]]) - org[axis]) * invDir[axis];
238  node = offset;
239  intervalMin = (tf >= intervalMin) ? tf : intervalMin;
240  intervalMax = (tb <= intervalMax) ? tb : intervalMax;
241  if (intervalMin > intervalMax)
242  break;
243  }
244  } // traversal loop
245  do
246  {
247  // stack is empty?
248  if (stackPos == 0)
249  return;
250  // move back up the stack
251  --stackPos;
252  intervalMin = stack[stackPos].tnear;
253  if (maxDist < intervalMin)
254  continue;
255  node = stack[stackPos].node;
256  intervalMax = stack[stackPos].tfar;
257  break;
258  } while (true);
259  }
260  }

264  {
265  if (!bounds.contains(p))
266  return;
267 
268  StackNode stack[MAX_STACK_SIZE];
269  int stackPos = 0;
270  int node = 0;
271 
272  while (true)
273  {
274  while (true)
275  {
276  uint32 tn = tree[node];
277  uint32 axis = (tn & (3 << 30)) >> 30;
278  const bool BVH2 = (tn & (1 << 29)) != 0;
279  int offset = tn & ~(7 << 29);
280  if (!BVH2)
281  {
282  if (axis < 3)
283  {
284  // "normal" interior node
285  float tl = intBitsToFloat(tree[node + 1]);
286  float tr = intBitsToFloat(tree[node + 2]);
287  // point is between clip zones
288  if (tl < p[axis] && tr > p[axis])
289  break;
290  int right = offset + 3;
291  node = right;
292  // point is in right node only
293  if (tl < p[axis])
294  {
295  continue;
296  }
297  node = offset; // left
298  // point is in left node only
299  if (tr > p[axis])
300  {
301  continue;
302  }
303  // point is in both nodes
304  // push back right node
305  stack[stackPos].node = right;
306  ++stackPos;
307  }
308  else
309  {
310  // leaf - test some objects
311  int n = tree[node + 1];
312  while (n > 0)
313  {
314  intersectCallback(p, objects[offset]); // !!!
315  --n;
316  ++offset;
317  }
318  break;
319  }
320  }
321  else // BVH2 node (empty space cut off left and right)
322  {
323  if (axis > 2)
324  return; // should not happen
325  float tl = intBitsToFloat(tree[node + 1]);
326  float tr = intBitsToFloat(tree[node + 2]);
327  node = offset;
328  if (tl > p[axis] || tr < p[axis])
329  break;
330  }
331  } // traversal loop
332 
333  // stack is empty?
334  if (stackPos == 0)
335  return;
336  // move back up the stack
337  --stackPos;
338  node = stack[stackPos].node;
339  }
340  }
