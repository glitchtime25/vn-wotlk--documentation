152  void UpdateAI(const uint32 uiDiff) override
153  {
154  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
155  return;
156 
157  switch (m_uiPhase)
158  {
159  case PHASE_SKELETAL:
160  // Phase switching at 50% (was in older patch versions multiple times, but from 335 on only once)
161  if (m_creature->GetHealthPercent() < 50)
162  {
163  if (DoCastSpellIfCan(m_creature, SPELL_DECAY_FLESH, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
164  {
165  DoScriptText(urand(0, 1) ? SAY_FLESH_1 : SAY_FLESH_2, m_creature);
166  m_uiPhase = PHASE_TAKE_FLESH;
167 
168  return; // return here, as there is nothing more to be done in this phase
169  }
170  }
171 
172  // No break here, the last phase is exactly like the first, but he doesn't change anymore
173  case PHASE_SKELETAL_END:
174  if (m_uiCurseLifeTimer < uiDiff)
175  {
176  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
177  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_CURSE_OF_LIFE : SPELL_CURSE_OF_LIFE_H) == CAST_OK)
178  m_uiCurseLifeTimer = urand(12000, 18000);
179  }
180  else
181  m_uiCurseLifeTimer -= uiDiff;
182 
183  if (m_uiRainFireTimer < uiDiff)
184  {
185  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
186  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_RAIN_OF_FIRE : SPELL_RAIN_OF_FIRE_H) == CAST_OK)
187  m_uiRainFireTimer = urand(22000, 29000);
188  }
189  else
190  m_uiRainFireTimer -= uiDiff;
191 
192  if (m_uiShadowVolleyTimer < uiDiff)
193  {
194  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SHADOW_VOLLEY : SPELL_SHADOW_VOLLEY_H) == CAST_OK)
195  m_uiShadowVolleyTimer = urand(6000, 12000);
196  }
197  else
198  m_uiShadowVolleyTimer -= uiDiff;
199 
200  DoMeleeAttackIfReady();
201  break;
202 
203  case PHASE_FLESH:
204  // This is not entirely clear if this _might_ also be triggered HP-dependend
205  if (m_uiReturnFleshTimer < uiDiff)
206  {
207  if (DoCastSpellIfCan(m_creature, SPELL_RETURN_FLESH, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
208  {
209  DoScriptText(urand(0, 1) ? SAY_SKELETON_1 : SAY_SKELETON_2, m_creature);
210  m_uiReturnFleshTimer = 26000;
211  m_uiPhase = PHASE_RETURN_FLESH;
212 
213  return; // return here, as there is nothing more to be done in this phase
214  }
215  }
216  else
217  m_uiReturnFleshTimer -= uiDiff;
218 
219  if (m_uiPoisonCloudTimer < uiDiff)
220  {
221  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
222  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_POISON_CLOUD : SPELL_POISON_CLOUD_H) == CAST_OK)
223  m_uiPoisonCloudTimer = urand(7000, 12000);
224  }
225  else
226  m_uiPoisonCloudTimer -= uiDiff;
227 
228  if (m_uiLightningBreathTimer < uiDiff)
229  {
230  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_LIGHTNING_BREATH : SPELL_LIGHTNING_BREATH_H) == CAST_OK)
231  m_uiLightningBreathTimer = urand(5000, 8000);
232  }
233  else
234  m_uiLightningBreathTimer -= uiDiff;
235 
236  if (m_uiEyeBeamTimer < uiDiff)
237  {
238  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_EYE_BEAM : SPELL_EYE_BEAM_H) == CAST_OK)
239  m_uiEyeBeamTimer = urand(12000, 15000);
240  }
241  else
242  m_uiEyeBeamTimer -= uiDiff;
243 
244  DoMeleeAttackIfReady();
245  break;
246 
247  case PHASE_TAKE_FLESH:
248  // Turn players into skeletons
249  if (DoCastSpellIfCan(m_creature, SPELL_GIFT_OF_THARONJA) == CAST_OK)
250  {
251  // Change modell - might be UpdateEntry
252  if (CreatureInfo const* pCreatureInfo = GetCreatureTemplateStore(NPC_THARONJA_FLESH))
253  {
254  uint32 uiDisplayId = Creature::ChooseDisplayId(pCreatureInfo);
255  m_creature->SetDisplayId(uiDisplayId);
256  }
257 
258  m_uiPhase = PHASE_FLESH;
259  }
260  break;
261 
262  case PHASE_RETURN_FLESH:
263  // Turn players into normal
264  if (DoCastSpellIfCan(m_creature, SPELL_CLEAR_GIFT_OF_THARONJA) == CAST_OK)
265  {
266  // Change modell - might be UpdateEntry
267  if (CreatureInfo const* pCreatureInfo = GetCreatureTemplateStore(NPC_THARONJA_SKELETAL))
268  {
269  uint32 uiDisplayId = Creature::ChooseDisplayId(pCreatureInfo);
270  m_creature->SetDisplayId(uiDisplayId);
271  }
272 
273  m_uiPhase = PHASE_SKELETAL_END;
274  }
275  break;
276  }
277  }
