210 bool DynamicMapTree::getObjectHitPos(const uint32 phasemask, const Vector3& pPos1, const Vector3& pPos2, Vector3& pResultHitPos, float pModifyDist) const
211 {
212  float maxDist = (pPos2 - pPos1).magnitude();
213  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too:
214  MANGOS_ASSERT(maxDist < std::numeric_limits<float>::max());
215  // prevent NaN values which can cause BIH intersection to enter infinite loop
216  if (maxDist < 1e-10f)
217  {
218  pResultHitPos = pPos2;
219  return false;
220  }
221  Vector3 dir = (pPos2 - pPos1) / maxDist; // direction with length of 1
222  G3D::Ray ray(pPos1, dir);
223  float dist = maxDist;
224  if (getIntersectionTime(phasemask, ray, pPos2, dist))
225  {
226  pResultHitPos = pPos1 + dir * dist;
227  if (pModifyDist < 0)
228  {
229  if ((pResultHitPos - pPos1).magnitude() > -pModifyDist)
230  {
231  pResultHitPos = pResultHitPos + dir * pModifyDist;
232  }
233  else
234  {
235  pResultHitPos = pPos1;
236  }
237  }
238  else
239  {
240  pResultHitPos = pResultHitPos + dir * pModifyDist;
241  }
242  return true;
243  }
244  pResultHitPos = pPos2;
245  return false;
246 }
