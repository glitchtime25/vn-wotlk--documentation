188  void SwitchPhase()
189  {
190  if (m_phase1)
191  {
192  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 2, nullptr, SELECT_FLAG_PLAYER))
193  {
194  m_felRageTarget = target->GetObjectGuid();
195  // Buff self
196  DoCastSpellIfCan(nullptr, SPELL_FEL_RAGE);
197  DoScriptText(urand(0, 1) ? SAY_SPECIAL1 : SAY_SPECIAL2, m_creature);
198 
199  // Debuff player
200  DoCastSpellIfCan(target, SPELL_FEL_RAGE_PLAYER_1, CAST_TRIGGERED);
201  DoCastSpellIfCan(target, SPELL_FEL_RAGE_PLAYER_2, CAST_TRIGGERED);
202  DoCastSpellIfCan(target, SPELL_FEL_RAGE_PLAYER_3, CAST_TRIGGERED);
203  DoCastSpellIfCan(target, SPELL_FEL_RAGE_PLAYER_4, CAST_TRIGGERED);
204  // Allow player to taunt Gurtogg
205  target->CastSpell(m_creature, SPELL_TAUNT_GURTOGG, TRIGGERED_OLD_TRIGGERED);
206 
207  // Don't allow others to generate threat
208  DoCastSpellIfCan(m_creature, SPELL_INSIGNIFIGANCE, CAST_TRIGGERED);
209 
210  m_creature->RemoveAurasDueToSpell(SPELL_ACIDIC_WOUND);
211  m_creature->CastSpell(nullptr, SPELL_FEL_GEYSER, TRIGGERED_OLD_TRIGGERED);
212  m_creature->CastSpell(target, SPELL_FEL_GEYSER_SUMMON, TRIGGERED_OLD_TRIGGERED);
213 
214  // Reset timers
215  m_phase1 = false;
216  DisableCombatAction(GURTOGG_ACTION_BLOODBOIL);
217  DisableCombatAction(GURTOGG_ACTION_BEWILDERING_STRIKE);
218  ResetCombatAction(GURTOGG_ACTION_ARCING_SMASH, 10000);
219  ResetCombatAction(GURTOGG_ACTION_FEL_ACID, urand(12000, 15000));
220  ResetCombatAction(GURTOGG_ACTION_CHARGE, urand(2000, 5000));
221  }
222  }
223  else
224  {
225  // Reset timers
226  // TODO: recheck phase end timers
227  m_phase1 = true;
228  DoCastSpellIfCan(nullptr, SPELL_ACIDIC_WOUND, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
229  m_creature->CastSpell(nullptr, SPELL_FEL_GEYSER, TRIGGERED_OLD_TRIGGERED);
230  ResetCombatAction(GURTOGG_ACTION_BLOODBOIL, 10000);
231  ResetCombatAction(GURTOGG_ACTION_ARCING_SMASH, 10000);
232  ResetCombatAction(GURTOGG_ACTION_FEL_ACID, GetInitialActionTimer(GURTOGG_ACTION_FEL_ACID));
233  ResetCombatAction(GURTOGG_ACTION_BEWILDERING_STRIKE, 28000);
234  ResetCombatAction(GURTOGG_ACTION_EJECT, 30000);
235  DisableCombatAction(GURTOGG_ACTION_CHARGE);
236  if (Unit* felRageTarget = m_creature->GetMap()->GetCreature(m_felRageTarget))
237  m_creature->getThreatManager().modifyThreatPercent(felRageTarget, -100);
238  }
239  }

241  void ExecuteAction(uint32 action) override
242  {
243  switch (action)
244  {
245  case GURTOGG_ACTION_CHANGE_PHASE:
246  {
247  SwitchPhase();
248  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
249  return;
250  }
251  case GURTOGG_ACTION_BERSERK:
252  {
253  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
254  {
255  DoScriptText(urand(0, 1) ? SAY_ENRAGE1 : SAY_ENRAGE2, m_creature);
256  DisableCombatAction(action);
257  }
258  return;
259  }
260  case GURTOGG_ACTION_BLOODBOIL:
261  {
262  if (DoCastSpellIfCan(nullptr, SPELL_BLOODBOIL) == CAST_OK)
263  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
264  return;
265  }
266  case GURTOGG_ACTION_ARCING_SMASH:
267  {
268  if (DoCastSpellIfCan(nullptr, m_phase1 ? SPELL_ARCING_SMASH_1 : SPELL_ARCING_SMASH_2) == CAST_OK)
269  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
270  return;
271  }
272  case GURTOGG_ACTION_FEL_ACID:
273  {
274  uint32 exclusionCount = 0;
275  uint32 flags = SELECT_FLAG_PLAYER | SELECT_FLAG_IN_MELEE_RANGE;
276  if (m_creature->getThreatManager().getThreatList().size() >= 2) // when at least 2 players skip 2 topaggro and main tank
277  {
278  flags |= SELECT_FLAG_SKIP_TANK;
279  exclusionCount = 2;
280  }
281  Unit* target = m_phase1 ? m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, exclusionCount, SPELL_FEL_ACID_1, flags) : m_creature->GetVictim();
282  if (target)
283  {
284  if (DoCastSpellIfCan(target, m_phase1 ? SPELL_FEL_ACID_1 : SPELL_FEL_ACID_2) == CAST_OK)
285  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
286  }
287  return;
288  }
289  case GURTOGG_ACTION_BEWILDERING_STRIKE:
290  {
291  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BEWILDERING_STRIKE) == CAST_OK)
292  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
293  return;
294  }
295  case GURTOGG_ACTION_EJECT:
296  {
297  if (DoCastSpellIfCan(m_creature->GetVictim(), m_phase1 ? SPELL_EJECT_1 : SPELL_EJECT_2) == CAST_OK)
298  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
299  return;
300  }
301  case GURTOGG_ACTION_CHARGE:
302  {
303  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CHARGE) == CAST_OK)
304  ResetCombatAction(action, GetSubsequentActionTimer(GurtoggActions(action)));
305  return;
306  }
307  }
308  }
