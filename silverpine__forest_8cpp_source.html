61  void WaypointReached(uint32 i) override
62  {
63  Player* pPlayer = GetPlayerForEscort();
64 
65  if (!pPlayer)
66  return;
67 
68  switch (i)
69  {
70  case 1:
71  DoScriptText(SAY_START_2, m_creature, pPlayer);
72  break;
73  case 14:
74  DoScriptText(SAY_END, m_creature, pPlayer);
75  pPlayer->RewardPlayerAndGroupAtEventExplored(QUEST_ERLAND, m_creature);
76  break;
77  case 15:
78  if (Creature* pRane = GetClosestCreatureWithEntry(m_creature, NPC_RANE, 45.0f))
79  DoScriptText(SAY_RANE, pRane, m_creature);
80  break;
81  case 16:
82  DoScriptText(SAY_RANE_REPLY, m_creature);
83  break;
84  case 17:
85  DoScriptText(SAY_CHECK_NEXT, m_creature);
86  break;
87  case 25:
88  DoScriptText(SAY_QUINN, m_creature);
89  break;
90  case 26:
91  if (Creature* pQuinn = GetClosestCreatureWithEntry(m_creature, NPC_QUINN, 45.0f))
92  DoScriptText(SAY_QUINN_REPLY, pQuinn, m_creature);
93  break;
94  case 27:
95  DoScriptText(SAY_BYE, m_creature);
96  break;
97  }
98  }
