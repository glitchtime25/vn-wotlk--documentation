101 {
102  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
103  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
104  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
105  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
106  {
107  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
108  {
109  if (PlayerbotAI::ORDERS_TANK & m_ai.GetCombatOrder())
110  {
111  if (meleeReach)
112  {
113  // Set everyone's UpdateAI() waiting to 2 seconds
114  m_ai.SetGroupIgnoreUpdateTime(2);
115  // Clear their TEMP_WAIT_TANKAGGRO flag
116  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
117  // Start attacking, force target on current target
118  m_ai.Attack(m_ai.GetCurrentTarget());
119 
120  // While everyone else is waiting 2 second, we need to build up aggro, so don't return
121  }
122  else
123  {
124  // TODO: add check if target is ranged
125  return RETURN_NO_ACTION_OK; // wait for target to get nearer
126  }
127  }
128  else if (PlayerbotAI::ORDERS_HEAL & m_ai.GetCombatOrder())
129  return _DoNextPVECombatManeuverHeal();
130  else
131  return RETURN_NO_ACTION_OK; // wait it out
132  }
133  else
134  {
135  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
136  }
137  }
138 
139  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
140  {
141  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
142  return RETURN_NO_ACTION_OK; // wait it out
143  else

145  }
