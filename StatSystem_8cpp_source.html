32 bool Player::UpdateStats(Stats stat)
33 {
34  if (stat > STAT_SPIRIT)
35  return false;
36 
37  // value = ((base_value * base_pct) + total_value) * total_pct
38  float value = GetTotalStatValue(stat);
39 
40  int32 oldValue = GetStat(stat);
41  SetStat(stat, int32(value));
42 
43  if (oldValue != value)
44  if (stat == STAT_STAMINA || stat == STAT_INTELLECT)
45  if (Pet* pet = GetPet())
46  pet->UpdateScalingAuras();
47 
48  switch (stat)
49  {
50  case STAT_STRENGTH:
51  UpdateShieldBlockValue();
52  break;
53  case STAT_AGILITY:
54  UpdateArmor();
55  UpdateAllCritPercentages();
56  UpdateDodgePercentage();
57  break;
58  case STAT_STAMINA: UpdateMaxHealth(); break;
59  case STAT_INTELLECT:
60  UpdateMaxPower(POWER_MANA);
61  UpdateAllSpellCritChances();
62  UpdateArmor(); // SPELL_AURA_MOD_RESISTANCE_OF_INTELLECT_PERCENT, only armor currently
63  break;
64 
65  case STAT_SPIRIT:
66  break;
67 
68  default:
69  break;
70  }
71  // Need update (exist AP from stat auras)
72  UpdateAttackPowerAndDamage();
73  UpdateAttackPowerAndDamage(true);
74 
75  UpdateSpellHealingBonus();
76  UpdateSpellDamageBonus();
77  UpdateManaRegen();
78 
79  // Update ratings in exist SPELL_AURA_MOD_RATING_FROM_STAT and only depends from stat
80  uint32 mask = 0;
81  AuraList const& modRatingFromStat = GetAurasByType(SPELL_AURA_MOD_RATING_FROM_STAT);
82  for (auto i : modRatingFromStat)
83  if (Stats(i->GetMiscBValue()) == stat)
84  mask |= i->GetMiscValue();
85  if (mask)
86  {
87  for (uint32 rating = 0; rating < MAX_COMBAT_RATING; ++rating)
88  if (mask & (1 << rating))
89  ApplyRatingMod(CombatRating(rating), 0, true);
90  }
91  return true;
92 }

255 void Player::UpdateAttackPowerAndDamage(bool ranged)
256 {
257  float val2 = 0.0f;
258  float level = float(GetLevel());
259 
260  UnitMods unitMod = ranged ? UNIT_MOD_ATTACK_POWER_RANGED : UNIT_MOD_ATTACK_POWER;
261 
262  uint16 index = UNIT_FIELD_ATTACK_POWER;
263  uint16 index_mod = UNIT_FIELD_ATTACK_POWER_MODS;
264  uint16 index_mult = UNIT_FIELD_ATTACK_POWER_MULTIPLIER;
265 
266  if (ranged)
267  {
268  index = UNIT_FIELD_RANGED_ATTACK_POWER;
269  index_mod = UNIT_FIELD_RANGED_ATTACK_POWER_MODS;
270  index_mult = UNIT_FIELD_RANGED_ATTACK_POWER_MULTIPLIER;
271 
272  switch (getClass())
273  {
274  case CLASS_HUNTER: val2 = level * 2.0f + GetStat(STAT_AGILITY) - 10.0f; break;
275  case CLASS_ROGUE: val2 = level + GetStat(STAT_AGILITY) - 10.0f; break;
276  case CLASS_WARRIOR: val2 = level + GetStat(STAT_AGILITY) - 10.0f; break;
277  case CLASS_DRUID:
278  switch (GetShapeshiftForm())
279  {
280  case FORM_CAT:
281  case FORM_BEAR:
282  case FORM_DIREBEAR:
283  val2 = 0.0f; break;
284  default:
285  val2 = GetStat(STAT_AGILITY) - 10.0f; break;
286  }
287  break;
288  default: val2 = GetStat(STAT_AGILITY) - 10.0f; break;
289  }
290  }
291  else
292  {
293  switch (getClass())
294  {
295  case CLASS_WARRIOR: val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f; break;
296  case CLASS_PALADIN: val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f; break;
297  case CLASS_DEATH_KNIGHT: val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f; break;
298  case CLASS_ROGUE: val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f; break;
299  case CLASS_HUNTER: val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f; break;
300  case CLASS_SHAMAN: val2 = level * 2.0f + GetStat(STAT_STRENGTH) + GetStat(STAT_AGILITY) - 20.0f; break;
301  case CLASS_DRUID:
302  {
303  val2 = level * 3.0f + GetStat(STAT_STRENGTH) * 2.0f - 20.0f;
304  if (ShapeshiftForm form = GetShapeshiftForm())
305  if (SpellShapeshiftFormEntry const* entry = sSpellShapeshiftFormStore.LookupEntry(form))
306  if (entry->flags1 & SHAPESHIFT_FLAG_AGILITY_ATTACK_BONUS)
307  val2 += GetStat(STAT_AGILITY);
308 
309  switch (GetShapeshiftForm())
310  {
311  case FORM_CAT:
312  case FORM_BEAR:
313  case FORM_DIREBEAR:
314  {
315  uint32 weaponAP = m_baseFeralAP;
316  if (Aura* predatoryStrikes = GetKnownTalentRankAuraById(803, EFFECT_INDEX_1))
317  {
318  uint32 totalWeaponAP = 0;
319  if (Item* weapon = GetWeaponForAttack(BASE_ATTACK))
320  totalWeaponAP = weapon->GetTotalAP() + m_baseFeralAP;
321  weaponAP += totalWeaponAP * predatoryStrikes->GetAmount() / 100;
322  }
323  val2 += weaponAP;
324  break;
325  }
326  case FORM_MOONKIN:
327  val2 += m_baseFeralAP;
328  break;
329  }
330  break;
331  }
332  case CLASS_MAGE: val2 = GetStat(STAT_STRENGTH) - 10.0f; break;
333  case CLASS_PRIEST: val2 = GetStat(STAT_STRENGTH) - 10.0f; break;
334  case CLASS_WARLOCK: val2 = GetStat(STAT_STRENGTH) - 10.0f; break;
335  }
336  }
337 
338  SetModifierValue(unitMod, BASE_VALUE, val2);
339 
340  float base_attPower = GetModifierValue(unitMod, BASE_VALUE) * GetModifierValue(unitMod, BASE_PCT);
341  float attPowerMod = GetModifierValue(unitMod, TOTAL_VALUE);
342 
343  // add dynamic flat mods
344  if (ranged)
345  {
346  if ((getClassMask() & CLASSMASK_WAND_USERS) == 0)
347  {
348  AuraList const& mRAPbyStat = GetAurasByType(SPELL_AURA_MOD_RANGED_ATTACK_POWER_OF_STAT_PERCENT);
349  for (auto i : mRAPbyStat)
350  attPowerMod += int32(GetStat(Stats(i->GetModifier()->m_miscvalue)) * i->GetModifier()->m_amount / 100.0f);
351  }
352  }
353  else
354  {
355  AuraList const& mAPbyStat = GetAurasByType(SPELL_AURA_MOD_ATTACK_POWER_OF_STAT_PERCENT);
356  for (auto i : mAPbyStat)
357  attPowerMod += int32(GetStat(Stats(i->GetModifier()->m_miscvalue)) * i->GetModifier()->m_amount / 100.0f);
358 
359  AuraList const& mAPbyArmor = GetAurasByType(SPELL_AURA_MOD_ATTACK_POWER_OF_ARMOR);
360  for (auto iter : mAPbyArmor)
361  // always: ((*i)->GetModifier()->m_miscvalue == 1 == SPELL_SCHOOL_MASK_NORMAL)
362  attPowerMod += int32(GetArmor() / iter->GetModifier()->m_amount);
363  }
364 
365  float attPowerMultiplier = GetModifierValue(unitMod, TOTAL_PCT) - 1.0f;
366 
367  SetInt32Value(index, (uint32)base_attPower); // UNIT_FIELD_(RANGED)_ATTACK_POWER field
368  SetInt32Value(index_mod, (uint32)attPowerMod); // UNIT_FIELD_(RANGED)_ATTACK_POWER_MODS field
369  SetFloatValue(index_mult, attPowerMultiplier); // UNIT_FIELD_(RANGED)_ATTACK_POWER_MULTIPLIER field
370 
371  // automatically update weapon damage after attack power modification
372  if (ranged)
373  {
374  UpdateDamagePhysical(RANGED_ATTACK);
375 
376  if (Pet* pet = GetPet()) // update pet's AP
377  pet->UpdateScalingAuras();
378  }
379  else
380  {
381  UpdateDamagePhysical(BASE_ATTACK);
382  if (CanDualWield() && hasOffhandWeaponForAttack()) // allow update offhand damage only if player knows DualWield Spec and has equipped offhand weapon
383  UpdateDamagePhysical(OFF_ATTACK);
384  }
385 }

405 void Player::CalculateMinMaxDamage(WeaponAttackType attType, bool normalized, float& min_damage, float& max_damage, uint8 index)
406 {
407  UnitMods unitMod;
408 
409  switch (attType)
410  {
411  case BASE_ATTACK:
412  default:
413  unitMod = UNIT_MOD_DAMAGE_MAINHAND;
414  break;
415  case OFF_ATTACK:
416  unitMod = UNIT_MOD_DAMAGE_OFFHAND;
417  break;
418  case RANGED_ATTACK:
419  unitMod = UNIT_MOD_DAMAGE_RANGED;
420  break;
421  }
422 
423  float att_speed = GetAPMultiplier(attType, normalized);
424 
425  float base_value = GetModifierValue(unitMod, BASE_VALUE) + GetTotalAttackPowerValue(attType) / 14.0f * att_speed;
426  float base_pct = GetModifierValue(unitMod, BASE_PCT);
427  float total_value = GetModifierValue(unitMod, TOTAL_VALUE);
428  float total_pct = GetModifierValue(unitMod, TOTAL_PCT);
429 
430  float weapon_mindamage = GetBaseWeaponDamage(attType, MINDAMAGE, index);
431  float weapon_maxdamage = GetBaseWeaponDamage(attType, MAXDAMAGE, index);
432 
433  if (IsNoWeaponShapeShift()) // check if player is in shapeshift which doesnt use weapon
434  {
435  uint32 lvl = GetLevel();
436  if (lvl > 60)
437  lvl = 60;
438 
439  weapon_mindamage = lvl * 0.85f * att_speed;
440  weapon_maxdamage = lvl * 1.25f * att_speed;
441  }
442  else if (!CanUseEquippedWeapon(attType)) // check if player not in form but still can't use weapon (broken/etc)
443  {
444  weapon_mindamage = BASE_MINDAMAGE;
445  weapon_maxdamage = BASE_MAXDAMAGE;
446  }
447  else
448  {
449  total_value += GetEnchantmentModifier(attType);
450  if (attType == RANGED_ATTACK) // add ammo DPS to ranged damage
451  {
452  auto ammoDps = GetAmmoDPS();
453  weapon_mindamage += ammoDps.first * att_speed;
454  weapon_maxdamage += ammoDps.second * att_speed;
455  }
456 
457  if (index != 0)
458  {
459  base_value = 0.0f;
460  total_value = 0.0f;
461  }
462  }
463 
464  min_damage = ((base_value + weapon_mindamage) * base_pct + total_value) * total_pct;
465  max_damage = ((base_value + weapon_maxdamage) * base_pct + total_value) * total_pct;
466 }
