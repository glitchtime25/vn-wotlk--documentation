60 template<class T, class CONTAINER>
61 inline void
62 Cell::Visit(const CellPair& standing_cell, TypeContainerVisitor<T, CONTAINER>& visitor, Map& m, float x, float y, float radius) const
63 {
64  if (standing_cell.x_coord >= TOTAL_NUMBER_OF_CELLS_PER_MAP || standing_cell.y_coord >= TOTAL_NUMBER_OF_CELLS_PER_MAP)
65  return;
66 
67  // no jokes here... Actually placing ASSERT() here was good idea, but
68  // we had some problems with DynamicObjects, which pass radius = 0.0f (DB issue?)
69  // maybe it is better to just return when radius <= 0.0f?
70  if (radius <= 0.0f)
71  {
72  m.Visit(*this, visitor);
73  return;
74  }
75  // lets limit the upper value for search radius
76  if (radius > MAX_VISIBILITY_DISTANCE)
77  radius = MAX_VISIBILITY_DISTANCE;
78 
79  // lets calculate object coord offsets from cell borders.
80  CellArea area = Cell::CalculateCellArea(x, y, radius);
81  // if radius fits inside standing cell
82  if (!area)
83  {
84  m.Visit(*this, visitor);
85  return;
86  }
87 
88  CellPair& begin_cell = area.low_bound;
89  CellPair& end_cell = area.high_bound;
90  // visit all cells, found in CalculateCellArea()
91  // if radius is known to reach cell area more than 4x4 then we should call optimized VisitCircle
92  // currently this technique works with MAX_NUMBER_OF_CELLS 16 and higher, with lower values
93  // there are nothing to optimize because SIZE_OF_GRID_CELL is too big...
94  if (((end_cell.x_coord - begin_cell.x_coord) > 4) && ((end_cell.y_coord - begin_cell.y_coord) > 4))
95  {
96  VisitCircle(visitor, m, begin_cell, end_cell);
97  return;
98  }
99 
100  // ALWAYS visit standing cell first!!! Since we deal with small radiuses
101  // it is very essential to call visitor for standing cell firstly...
102  m.Visit(*this, visitor);
103 
104  // loop the cell range
105  for (uint32 i = begin_cell.x_coord; i <= end_cell.x_coord; ++i)
106  {
107  for (uint32 j = begin_cell.y_coord; j <= end_cell.y_coord; ++j)
108  {
109  CellPair cell_pair(i, j);
110  // lets skip standing cell since we already visited it
111  if (cell_pair != standing_cell)
112  {
113  Cell r_zone(cell_pair);
114  r_zone.data.Part.nocreate = data.Part.nocreate;
115  m.Visit(r_zone, visitor);
116  }
117  }
118  }
119 }

121 template<class T, class CONTAINER>
122 inline void
123 Cell::VisitCircle(TypeContainerVisitor<T, CONTAINER>& visitor, Map& m, const CellPair& begin_cell, const CellPair& end_cell) const
124 {
125  // here is an algorithm for 'filling' circum-squared octagon
126  uint32 x_shift = (uint32)ceilf((end_cell.x_coord - begin_cell.x_coord) * 0.3f - 0.5f);
127  // lets calculate x_start/x_end coords for central strip...
128  const uint32 x_start = begin_cell.x_coord + x_shift;
129  const uint32 x_end = end_cell.x_coord - x_shift;
130 
131  // visit central strip with constant width...
132  for (uint32 x = x_start; x <= x_end; ++x)
133  {
134  for (uint32 y = begin_cell.y_coord; y <= end_cell.y_coord; ++y)
135  {
136  CellPair cell_pair(x, y);
137  Cell r_zone(cell_pair);
138  r_zone.data.Part.nocreate = data.Part.nocreate;
139  m.Visit(r_zone, visitor);
140  }
141  }
142 
143  // if x_shift == 0 then we have too small cell area, which were already
144  // visited at previous step, so just return from procedure...
145  if (x_shift == 0)
146  return;
147 
148  uint32 y_start = end_cell.y_coord;
149  uint32 y_end = begin_cell.y_coord;
150  // now we are visiting borders of an octagon...
151  for (uint32 step = 1; step <= (x_start - begin_cell.x_coord); ++step)
152  {
153  // each step reduces strip height by 2 cells...
154  y_end += 1;
155  y_start -= 1;
156  for (uint32 y = y_start; y >= y_end; --y)
157  {
158  // we visit cells symmetrically from both sides, heading from center to sides and from up to bottom
159  // e.g. filling 2 trapezoids after filling central cell strip...
160  CellPair cell_pair_left(x_start - step, y);
161  Cell r_zone_left(cell_pair_left);
162  r_zone_left.data.Part.nocreate = data.Part.nocreate;
163  m.Visit(r_zone_left, visitor);
164 
165  // right trapezoid cell visit
166  CellPair cell_pair_right(x_end + step, y);
167  Cell r_zone_right(cell_pair_right);
168  r_zone_right.data.Part.nocreate = data.Part.nocreate;
169  m.Visit(r_zone_right, visitor);
170  }
171  }
172 }
