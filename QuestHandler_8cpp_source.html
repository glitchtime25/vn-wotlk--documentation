38 void WorldSession::HandleQuestgiverStatusQueryOpcode(WorldPacket& recv_data)
39 {
40  ObjectGuid guid;
41  recv_data >> guid;
42  uint8 dialogStatus = DIALOG_STATUS_NONE;
43 
44  Object* questgiver = _player->GetObjectByTypeMask(guid, TYPEMASK_CREATURE_OR_GAMEOBJECT);
45 
46  if (!questgiver)
47  {
48  DETAIL_LOG("Error in CMSG_QUESTGIVER_STATUS_QUERY, called for not found questgiver %s", guid.GetString().c_str());
49  return;
50  }
51 
52  DEBUG_LOG("WORLD: Received opcode CMSG_QUESTGIVER_STATUS_QUERY - for %s to %s", _player->GetGuidStr().c_str(), guid.GetString().c_str());
53 
54  switch (questgiver->GetTypeId())
55  {
56  case TYPEID_UNIT:
57  {
58  Creature* cr_questgiver = (Creature*)questgiver;
59 
60  if (_player->CanInteract(static_cast<Unit*>(questgiver))) // not show quest status to enemies
61  {
62  dialogStatus = sScriptDevAIMgr.GetDialogStatus(_player, cr_questgiver);
63 
64  if (dialogStatus == DIALOG_STATUS_UNDEFINED)
65  dialogStatus = getDialogStatus(_player, cr_questgiver, DIALOG_STATUS_NONE);
66  }
67  break;
68  }
69  case TYPEID_GAMEOBJECT:
70  {
71  GameObject* go_questgiver = (GameObject*)questgiver;
72 
73  if (_player->CanInteract(go_questgiver))
74  {
75  dialogStatus = sScriptDevAIMgr.GetDialogStatus(_player, go_questgiver);
76 
77  if (dialogStatus == DIALOG_STATUS_UNDEFINED)
78  dialogStatus = getDialogStatus(_player, go_questgiver, DIALOG_STATUS_NONE);
79  }
80  break;
81  }
82  default:
83  sLog.outError("QuestGiver called for unexpected type %u", questgiver->GetTypeId());
84  break;
85  }
86 
87  // inform client about status of quest
88  _player->GetPlayerMenu()->SendQuestGiverStatus(dialogStatus, guid);
89 }

116 void WorldSession::HandleQuestgiverAcceptQuestOpcode(WorldPacket& recv_data)
117 {
118  ObjectGuid guid;
119  uint32 quest;
120  uint32 unk1;
121  recv_data >> guid >> quest >> unk1;
122 
123  if (!CanInteractWithQuestGiver(guid, "CMSG_QUESTGIVER_ACCEPT_QUEST"))
124  return;
125 
126  DEBUG_LOG("WORLD: Received opcode CMSG_QUESTGIVER_ACCEPT_QUEST - for %s to %s, quest = %u, unk1 = %u", _player->GetGuidStr().c_str(), guid.GetString().c_str(), quest, unk1);
127 
128  Object* pObject = _player->GetObjectByTypeMask(guid, TYPEMASK_CREATURE_GAMEOBJECT_PLAYER_OR_ITEM);
129 
130  // no or incorrect quest giver (player himself is questgiver for SPELL_EFFECT_QUEST_OFFER)
131  if (!pObject
132  || (pObject->GetTypeId() != TYPEID_PLAYER && !pObject->HasQuest(quest))
133  || (pObject->GetTypeId() == TYPEID_PLAYER && pObject != _player && !((Player*)pObject)->CanShareQuest(quest))
134  )
135  {
136  _player->GetPlayerMenu()->CloseGossip();
137  _player->ClearDividerGuid();
138  return;
139  }
140 
141  Quest const* qInfo = sObjectMgr.GetQuestTemplate(quest);
142  if (qInfo)
143  {
144  // prevent cheating
145  if (!GetPlayer()->CanTakeQuest(qInfo, true))
146  {
147  _player->GetPlayerMenu()->CloseGossip();
148  _player->ClearDividerGuid();
149  return;
150  }
151 
152  if (Player* pPlayer = ObjectAccessor::FindPlayer(_player->GetDividerGuid()))
153  {
154  pPlayer->SendPushToPartyResponse(_player, QUEST_PARTY_MSG_ACCEPT_QUEST);
155  _player->ClearDividerGuid();
156  }
157 
158  if (_player->CanAddQuest(qInfo, true))
159  {
160  _player->AddQuest(qInfo, pObject); // pObject (if it item) can be destroyed at call
161 
162  if (qInfo->HasQuestFlag(QUEST_FLAGS_PARTY_ACCEPT))
163  {
164  if (Group* pGroup = _player->GetGroup())
165  {
166  for (GroupReference* itr = pGroup->GetFirstMember(); itr != nullptr; itr = itr->next())
167  {
168  Player* pPlayer = itr->getSource();
169 
170  if (!pPlayer || pPlayer == _player || !pPlayer->IsInWorld()) // not self
171  continue;
172 
173  if (pPlayer->CanTakeQuest(qInfo, true))
174  {
175  pPlayer->SetDividerGuid(_player->GetObjectGuid());
176 
177  // need confirmation that any gossip window will close
178  pPlayer->GetPlayerMenu()->CloseGossip();
179 
180  _player->SendQuestConfirmAccept(qInfo, pPlayer);
181  }
182  }
183  }
184  }
185 
186  if (_player->CanCompleteQuest(quest))
187  _player->CompleteQuest(quest);
188 
189  _player->GetAchievementMgr().StartTimedAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_COMPLETE_QUEST, quest);
190 
191  _player->GetPlayerMenu()->CloseGossip();
192 
193  if (qInfo->GetSrcSpell() > 0)
194  _player->CastSpell(_player, qInfo->GetSrcSpell(), TRIGGERED_OLD_TRIGGERED);
195 
196  return;
197  }
198  }
199 
200  _player->GetPlayerMenu()->CloseGossip();
201 }

224 void WorldSession::HandleQuestQueryOpcode(WorldPacket& recv_data)
225 {
226  uint32 quest;
227  recv_data >> quest;
228  DEBUG_LOG("WORLD: Received opcode CMSG_QUEST_QUERY quest = %u", quest);
229 
230  Quest const* pQuest = sObjectMgr.GetQuestTemplate(quest);
231  if (pQuest)
232  {
233  std::string ObjectiveText[QUEST_OBJECTIVES_COUNT];
234  std::string Title = pQuest->GetTitle();
235  std::string Details = pQuest->GetDetails();
236  std::string Objectives = pQuest->GetObjectives();
237  std::string EndText = pQuest->GetEndText();
238  std::string CompletedText = pQuest->GetCompletedText();
239 
240  for (int i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
241  ObjectiveText[i] = pQuest->ObjectiveText[i];
242 
243  int loc_idx = GetSessionDbLocaleIndex();
244  if (loc_idx >= 0)
245  {
246  if (QuestLocale const* ql = sObjectMgr.GetQuestLocale(pQuest->GetQuestId()))
247  {
248  if (ql->Title.size() > (size_t)loc_idx && !ql->Title[loc_idx].empty())
249  Title = ql->Title[loc_idx];
250  if (ql->Details.size() > (size_t)loc_idx && !ql->Details[loc_idx].empty())
251  Details = ql->Details[loc_idx];
252  if (ql->Objectives.size() > (size_t)loc_idx && !ql->Objectives[loc_idx].empty())
253  Objectives = ql->Objectives[loc_idx];
254  if (ql->EndText.size() > (size_t)loc_idx && !ql->EndText[loc_idx].empty())
255  EndText = ql->EndText[loc_idx];
256  if (ql->CompletedText.size() > (size_t)loc_idx && !ql->CompletedText[loc_idx].empty())
257  CompletedText = ql->CompletedText[loc_idx];
258 
259  for (int i = 0; i < QUEST_OBJECTIVES_COUNT; ++i)
260  if (ql->ObjectiveText[i].size() > (size_t)loc_idx && !ql->ObjectiveText[i][loc_idx].empty())
261  ObjectiveText[i] = ql->ObjectiveText[i][loc_idx];
262  }
263  }
264 
265  WorldPacket data(SMSG_QUEST_QUERY_RESPONSE, 100); // guess size
266 
267  data << uint32(pQuest->GetQuestId()); // quest id
268  data << uint32(pQuest->GetQuestMethod()); // Accepted values: 0, 1 or 2. 0==IsAutoComplete() (skip objectives/details)
269  data << int32(pQuest->GetQuestLevel()); // may be -1, static data, in other cases must be used dynamic level: Player::GetQuestLevelForPlayer (0 is not known, but assuming this is no longer valid for quest intended for client)
270  data << uint32(pQuest->GetMinLevel()); // min required level to obtain (added for 3.3). Assumed allowed (database) range is -1 to 255 (still using uint32, since negative value would not be of any known use for client)
271  data << uint32(pQuest->GetZoneOrSort()); // zone or sort to display in quest log
272 
273  data << uint32(pQuest->GetType()); // quest type
274  data << uint32(pQuest->GetSuggestedPlayers()); // suggested players count
275 
276  data << uint32(pQuest->GetRepObjectiveFaction()); // shown in quest log as part of quest objective
277  data << uint32(pQuest->GetRepObjectiveValue()); // shown in quest log as part of quest objective
278 
279  data << uint32(0); // RequiredOpositeRepFaction
280  data << uint32(0); // RequiredOpositeRepValue, required faction value with another (oposite) faction (objective)
281 
282  data << uint32(pQuest->GetNextQuestInChain()); // client will request this quest from NPC, if not 0
283  data << uint32(pQuest->GetRewXPId()); // column index in QuestXP.dbc (row based on quest level)
284 
285  if (pQuest->HasQuestFlag(QUEST_FLAGS_HIDDEN_REWARDS))
286  data << uint32(0); // Hide money rewarded
287  else
288  data << uint32(pQuest->GetRewOrReqMoney()); // reward money (below max lvl)
289 
290  data << uint32(pQuest->GetRewMoneyMaxLevel()); // used in XP calculation at client
291  data << uint32(pQuest->GetRewSpell()); // reward spell, this spell will display (icon) (casted if RewSpellCast==0)
292  data << uint32(pQuest->GetRewSpellCast()); // casted spell
293 
294  // rewarded honor points
295  data << uint32(pQuest->GetRewHonorAddition());
296  data << float(pQuest->GetRewHonorMultiplier()); // new reward honor (multiplied by ~62 at client side)
297 
298  data << uint32(pQuest->GetSrcItemId()); // source item id
299  data << uint32(pQuest->GetQuestFlags()); // quest flags
300  data << uint32(pQuest->GetCharTitleId()); // CharTitleId, new 2.4.0, player gets this title (id from CharTitles)
301  data << uint32(pQuest->GetPlayersSlain()); // players slain
302  data << uint32(pQuest->GetBonusTalents()); // bonus talents
303  data << uint32(0); // bonus arena points
304  data << uint32(0); // rew rep show mask?
305 
306  int iI;
307 
308  if (pQuest->HasQuestFlag(QUEST_FLAGS_HIDDEN_REWARDS))
309  {
310  for (iI = 0; iI < QUEST_REWARDS_COUNT; ++iI)
311  data << uint32(0) << uint32(0);
312  for (iI = 0; iI < QUEST_REWARD_CHOICES_COUNT; ++iI)
313  data << uint32(0) << uint32(0);
314  }
315  else
316  {
317  for (iI = 0; iI < QUEST_REWARDS_COUNT; ++iI)
318  {
319  data << uint32(pQuest->RewItemId[iI]);
320  data << uint32(pQuest->RewItemCount[iI]);
321  }
322  for (iI = 0; iI < QUEST_REWARD_CHOICES_COUNT; ++iI)
323  {
324  data << uint32(pQuest->RewChoiceItemId[iI]);
325  data << uint32(pQuest->RewChoiceItemCount[iI]);
326  }
327  }
328 
329  for (unsigned int i : pQuest->RewRepFaction) // reward factions ids
330  data << uint32(i);
331 
332  for (int i : pQuest->RewRepValueId) // columnid in QuestFactionReward.dbc (if negative, from second row)
333  data << int32(i);
334 
335  for (int i = 0; i < QUEST_REPUTATIONS_COUNT; ++i) // reward reputation override. No bonus is expected given
336  data << int32(0);
337  // data << int32(pQuest->RewRepValue[i]); // current field for store of rep value, can be reused to implement "override value"
338 
339  data << pQuest->GetPointMapId();
340  data << pQuest->GetPointX();
341  data << pQuest->GetPointY();
342  data << pQuest->GetPointOpt();
343 
344  data << Title;
345  data << Objectives;
346  data << Details;
347  data << EndText;
348  data << CompletedText; // display in quest objectives window once all objectives are completed
349 
350  for (iI = 0; iI < QUEST_OBJECTIVES_COUNT; ++iI)
351  {
352  if (pQuest->ReqCreatureOrGOId[iI] < 0)
353  {
354  // client expected gameobject template id in form (id|0x80000000)
355  data << uint32((pQuest->ReqCreatureOrGOId[iI] * (-1)) | 0x80000000);
356  }
357  else
358  {
359  data << uint32(pQuest->ReqCreatureOrGOId[iI]);
360  }
361  data << uint32(pQuest->ReqCreatureOrGOCount[iI]);
362  data << uint32(pQuest->ReqSourceId[iI]);
363  data << uint32(0); // req source count?
364  }
365 
366  for (iI = 0; iI < QUEST_ITEM_OBJECTIVES_COUNT; ++iI)
367  {
368  data << uint32(pQuest->ReqItemId[iI]);
369  data << uint32(pQuest->ReqItemCount[iI]);
370  }
371 
372  for (iI = 0; iI < QUEST_OBJECTIVES_COUNT; ++iI)
373  data << ObjectiveText[iI];
374 
375  SendPacket(data);
376 
377  DEBUG_LOG("WORLD: Sent SMSG_QUEST_QUERY_RESPONSE questid=%u", pQuest->GetQuestId());
378  }
379 }

390  return;
391  }

404 
405  Quest const* pQuest = sObjectMgr.GetQuestTemplate(quest);

675 uint32 WorldSession::getDialogStatus(const Player* pPlayer, const Object* questgiver, uint32 defstatus) const
676 {
677  MANGOS_ASSERT(defstatus != DIALOG_STATUS_UNDEFINED);
678 
679  uint32 dialogStatus = defstatus;
680 
681  QuestRelationsMapBounds rbounds; // QuestRelations (quest-giver)
682  QuestRelationsMapBounds irbounds; // InvolvedRelations (quest-finisher)
683 
684  switch (questgiver->GetTypeId())
685  {
686  case TYPEID_UNIT:
687  {
688  rbounds = sObjectMgr.GetCreatureQuestRelationsMapBounds(questgiver->GetEntry());
689  irbounds = sObjectMgr.GetCreatureQuestInvolvedRelationsMapBounds(questgiver->GetEntry());
690  break;
691  }
692  case TYPEID_GAMEOBJECT:
693  {
694  rbounds = sObjectMgr.GetGOQuestRelationsMapBounds(questgiver->GetEntry());
695  irbounds = sObjectMgr.GetGOQuestInvolvedRelationsMapBounds(questgiver->GetEntry());
696  break;
697  }
698  default:
699  // it's impossible, but check ^)
700  sLog.outError("Warning: GetDialogStatus called for unexpected type %u", questgiver->GetTypeId());
701  return DIALOG_STATUS_NONE;
702  }
703 
704  // Check markings for quest-finisher
705  for (QuestRelationsMap::const_iterator itr = irbounds.first; itr != irbounds.second; ++itr)
706  {
707  uint32 dialogStatusNew = DIALOG_STATUS_NONE;
708  uint32 quest_id = itr->second;
709  Quest const* pQuest = sObjectMgr.GetQuestTemplate(quest_id);
710 
711  if (!pQuest || !pQuest->IsActive())
712  continue;
713 
714  QuestStatus status = pPlayer->GetQuestStatus(quest_id);
715 
716  if (status == QUEST_STATUS_COMPLETE && !pPlayer->GetQuestRewardStatus(quest_id))
717  dialogStatusNew = pQuest->IsRepeatable() && !pQuest->IsDailyOrWeekly() ? DIALOG_STATUS_REWARD_REP : DIALOG_STATUS_REWARD;
718  else if (pQuest->IsAutoComplete() && pPlayer->CanTakeQuest(pQuest, false))
719  dialogStatusNew = pQuest->IsRepeatable() ? pQuest->IsDailyOrWeekly() ? DIALOG_STATUS_AVAILABLE_REP : DIALOG_STATUS_REWARD_REP : DIALOG_STATUS_AVAILABLE;
720  else if (status == QUEST_STATUS_INCOMPLETE)
721  dialogStatusNew = DIALOG_STATUS_INCOMPLETE;
722 
723  if (dialogStatusNew > dialogStatus)
724  dialogStatus = dialogStatusNew;
725  }
726 
727  // check markings for quest-giver
728  for (QuestRelationsMap::const_iterator itr = rbounds.first; itr != rbounds.second; ++itr)
729  {
730  uint32 dialogStatusNew = DIALOG_STATUS_NONE;
731  uint32 quest_id = itr->second;
732  Quest const* pQuest = sObjectMgr.GetQuestTemplate(quest_id);
733 
734  if (!pQuest || !pQuest->IsActive())
735  continue;
736 
737  QuestStatus status = pPlayer->GetQuestStatus(quest_id);
738 
739  if (status == QUEST_STATUS_NONE) // For all other cases the mark is handled either at some place else, or with involved-relations already
740  {
741  if (pPlayer->CanSeeStartQuest(pQuest))
742  {
743  if (pPlayer->SatisfyQuestLevel(pQuest, false))
744  {
745  int32 lowLevelDiff = sWorld.getConfig(CONFIG_INT32_QUEST_LOW_LEVEL_HIDE_DIFF);
746 
747  if (pQuest->IsAutoComplete())
748  {
749  dialogStatusNew = DIALOG_STATUS_REWARD_REP;
750  }
751  else if (lowLevelDiff < 0 || pPlayer->GetLevel() <= pPlayer->GetQuestLevelForPlayer(pQuest) + uint32(lowLevelDiff))
752  {
753  if (pQuest->HasQuestFlag(QUEST_FLAGS_DAILY) || pQuest->HasQuestFlag(QUEST_FLAGS_WEEKLY))
754  dialogStatusNew = DIALOG_STATUS_AVAILABLE_REP;
755  else
756  dialogStatusNew = DIALOG_STATUS_AVAILABLE;
757  }
758  else // player level much higher then quest-level
759  dialogStatusNew = DIALOG_STATUS_LOW_LEVEL_AVAILABLE;
760  }
761  else
762  dialogStatusNew = DIALOG_STATUS_UNAVAILABLE;
763  }
764  }
765 
766  if (dialogStatusNew > dialogStatus)
767  dialogStatus = dialogStatusNew;
768  }
769 
770  return dialogStatus;
771 }
