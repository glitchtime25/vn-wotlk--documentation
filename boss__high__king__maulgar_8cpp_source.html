85  boss_high_king_maulgarAI(Creature* creature) : CombatAI(creature, MAULGAR_ACTION_MAX), m_instance(static_cast<ScriptedInstance*>(creature->GetInstanceData()))
86  {

94  }

129  void EventCouncilDeath()
130  {
131  switch (++m_uiCouncilDeathCount)
132  {
133  case 1: DoScriptText(SAY_OGRE_DEATH1, m_creature); break;
134  case 2: DoScriptText(SAY_OGRE_DEATH2, m_creature); break;
135  case 3: DoScriptText(SAY_OGRE_DEATH3, m_creature); break;
136  case 4: DoScriptText(SAY_OGRE_DEATH4, m_creature); break;
137  }
138  }

140  void ExecuteAction(uint32 action) override
141  {
142  switch (action)
143  {
144  case MAULGAR_PHASE_2:
145  if (m_creature->GetHealthPercent() < 50.0f)
146  {
147  if (DoCastSpellIfCan(nullptr, SPELL_FLURRY) == CAST_OK)
148  {
149  DoScriptText(SAY_ENRAGE, m_creature);
150  SetActionReadyStatus(action, false);
151  ResetCombatAction(MAULGAR_FEAR, urand(10000, 25000));
152  ResetCombatAction(MAULGAR_CHARGE, 2000);
153  }
154  }
155  break;
156  case MAULGAR_ARCING_SMASH:
157  if (DoCastSpellIfCan(nullptr, SPELL_ARCING_SMASH) == CAST_OK)
158  ResetCombatAction(action, urand(8000, 12000));
159  break;
160  case MAULGAR_MIGHTY_BLOW:
161  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MIGHTY_BLOW) == CAST_OK)
162  ResetCombatAction(action, urand(20000, 35000));
163  break;
164  case MAULGAR_WHIRLWIND:
165  if (DoCastSpellIfCan(nullptr, SPELL_WHIRLWIND) == CAST_OK)
166  ResetCombatAction(action, urand(30000, 40000));
167  break;
168  case MAULGAR_CHARGE:
169  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_CHARGE, SELECT_FLAG_PLAYER))
170  if (DoCastSpellIfCan(target, SPELL_CHARGE) == CAST_OK)
171  ResetCombatAction(action, urand(14000, 20000));
172  break;
173  case MAULGAR_FEAR:
174  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FEAR) == CAST_OK)
175  ResetCombatAction(action, urand(20000, 35000));
176  break;
177  }
178  }
179 };
