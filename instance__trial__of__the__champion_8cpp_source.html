321 void instance_trial_of_the_champion::SetData(uint32 uiType, uint32 uiData)
322 {
323  switch (uiType)
324  {
325  case TYPE_GRAND_CHAMPIONS:
326  // no double count
327  if (m_auiEncounter[uiType] == uiData)
328  return;
329 
330  m_auiEncounter[uiType] = uiData;
331  DoUseDoorOrButton(GO_NORTH_GATE);
332  if (uiData == DONE)
333  {
334  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_CHAMPIONS_LOOT : GO_CHAMPIONS_LOOT_H, 30 * MINUTE);
335 
336  // start delayed dialogue
337  StartNextDialogueText(NPC_ARELAS_BRIGHTSTAR);
338 
339  // move the herald back to center
340  if (Creature* pHerald = GetSingleCreatureFromStorage(m_uiHeraldEntry))
341  {
342  pHerald->GetMotionMaster()->Clear();
343  pHerald->GetMotionMaster()->MovePoint(0, aHeraldPositions[2][0], aHeraldPositions[2][1], aHeraldPositions[2][2]);
344  pHerald->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
345  }
346 
347  DoSendChampionsToExit();
348  }
349  break;
350  case TYPE_ARGENT_CHAMPION:
351  m_auiEncounter[uiType] = uiData;
352  DoUseDoorOrButton(GO_NORTH_GATE);
353  if (uiData == DONE)
354  {
355  if (m_uiGrandChampionEntry == NPC_EADRIC)
356  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_EADRIC_LOOT : GO_EADRIC_LOOT_H, 30 * MINUTE);
357  else if (m_uiGrandChampionEntry == NPC_PALETRESS)
358  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_PALETRESS_LOOT : GO_PALETRESS_LOOT_H, 30 * MINUTE);
359 
360  // start event epilog
361  StartNextDialogueText(POINT_ID_MOUNT);
362 
363  // move the herald back to center
364  if (Creature* pHerald = GetSingleCreatureFromStorage(m_uiHeraldEntry))
365  {
366  pHerald->GetMotionMaster()->Clear();
367  pHerald->GetMotionMaster()->MovePoint(0, aHeraldPositions[2][0], aHeraldPositions[2][1], aHeraldPositions[2][2]);
368  pHerald->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
369  }
370  }
371  break;
372  case TYPE_BLACK_KNIGHT:
373  DoUseDoorOrButton(GO_NORTH_GATE);
374  if (uiData == DONE)
375  StartNextDialogueText(SPELL_SPECTATOR_FORCE_CHEER);
376  else if (uiData == IN_PROGRESS)
377  m_bHadWorseAchiev = true;
378  m_auiEncounter[uiType] = uiData;
379  break;
380  case TYPE_ARENA_CHALLENGE:
381  m_auiEncounter[uiType] = uiData;
382  if (uiData == IN_PROGRESS)
383  {
384  // start arena challenge
385  m_uiArenaStage = 0;
386  DoSendNextArenaWave();
387  }
388  else if (uiData == DONE)
389  {
390  // count the champions that reach the exit
391  ++m_uiChampionsCount;
392 
393  if (m_uiChampionsCount == MAX_CHAMPIONS_ARENA)
394  {
395  // start grand champions challenge (without mount)
396  m_uiChampionsCount = 0;
397  m_uiChampionsTimer = 5000;
398 
399  // despawn vehicle mounts
400  for (GuidList::const_iterator itr = m_lArenaMountsGuids.begin(); itr != m_lArenaMountsGuids.end(); ++itr)
401  {
402  if (Creature* pMount = instance->GetCreature(*itr))
403  pMount->ForcedDespawn();
404  }
405  m_lArenaMountsGuids.clear();
406  }
407  }
408  else if (uiData == FAIL)
409  {
410  DoCleanupArenaOnWipe();
411  SetData(TYPE_ARENA_CHALLENGE, NOT_STARTED);
412  }
413  else if (uiData == SPECIAL)
414  DoSendChampionsToExit();
415  return;
416  default:
417  script_error_log("Instance Trial of The Champion: ERROR SetData = %u for type %u does not exist/not implemented.", uiType, uiData);
418  return;
419  }
420 
421  if (uiData == DONE)
422  {
423  OUT_SAVE_INST_DATA;
424 
425  std::ostringstream saveStream;
426  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2];
427 
428  m_strInstData = saveStream.str();
429 
430  SaveToDB();
431  OUT_SAVE_INST_DATA_COMPLETE;
432  }
