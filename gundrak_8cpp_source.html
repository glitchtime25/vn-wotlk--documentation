193 void instance_gundrak::SetData(uint32 type, uint32 data)
194 {
195  debug_log("SD2: Instance Gundrak: SetData received for type %u with data %u", type, data);
196 
197  switch (type)
198  {
199  case TYPE_SLADRAN:
200  m_auiEncounter[TYPE_SLADRAN] = data;
201  if (data == DONE)
202  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_ALTAR_OF_SLADRAN))
203  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
204  if (data == FAIL)
205  m_uisWhySnakesAchievPlayers.clear();
206  if (data == SPECIAL)
207  m_mAltarInProgress.insert(TypeTimerPair(TYPE_SLADRAN, TIMER_VISUAL_ALTAR));
208  break;
209  case TYPE_MOORABI:
210  m_auiEncounter[TYPE_MOORABI] = data;
211  if (data == DONE)
212  {
213  if (!instance->IsRegularDifficulty())
214  DoUseDoorOrButton(GO_ECK_DOOR);
215  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_ALTAR_OF_MOORABI))
216  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
217  }
218  if (data == IN_PROGRESS)
219  SetLessRabiAchievementCriteria(true);
220  if (data == SPECIAL)
221  m_mAltarInProgress.insert(TypeTimerPair(TYPE_MOORABI, TIMER_VISUAL_ALTAR));
222  break;
223  case TYPE_COLOSSUS:
224  m_auiEncounter[TYPE_COLOSSUS] = data;
225  if (data == DONE)
226  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_ALTAR_OF_COLOSSUS))
227  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
228  if (data == FAIL)
229  {
230  for (auto m_sColossusMojosGuid : m_sColossusMojosGuids)
231  {
232  if (Creature* pMojo = instance->GetCreature(m_sColossusMojosGuid))
233  pMojo->Respawn();
234  }
235  }
236  if (data == SPECIAL)
237  m_mAltarInProgress.insert(TypeTimerPair(TYPE_COLOSSUS, TIMER_VISUAL_ALTAR));
238  break;
239  case TYPE_GALDARAH:
240  m_auiEncounter[TYPE_GALDARAH] = data;
241  DoUseDoorOrButton(GO_GALDARAH_DOOR);
242  if (data == DONE)
243  {
244  DoUseDoorOrButton(GO_EXIT_DOOR_L);
245  DoUseDoorOrButton(GO_EXIT_DOOR_R);
246  }
247  if (data == FAIL)
248  m_uisShareLoveAchievPlayers.clear();
249  break;
250  case TYPE_ECK:
251  m_auiEncounter[TYPE_ECK] = data;
252  if (data == DONE)
253  DoUseDoorOrButton(GO_ECK_UNDERWATER_DOOR);
254  break;
255  case TYPE_ACHIEV_WHY_SNAKES:
256  // insert the players who failed the achiev and haven't been already inserted in the set
257  if (m_uisWhySnakesAchievPlayers.find(data) == m_uisWhySnakesAchievPlayers.end())
258  m_uisWhySnakesAchievPlayers.insert(data);
259  break;
260  case TYPE_ACHIEV_SHARE_LOVE:
261  // insert players who got stampeled and haven't been already inserted in the set
262  if (m_uisShareLoveAchievPlayers.find(data) == m_uisShareLoveAchievPlayers.end())
263  m_uisShareLoveAchievPlayers.insert(data);
264  break;
265  default:
266  script_error_log("Instance Gundrak: ERROR SetData = %u for type %u does not exist/not implemented.", type, data);
267  return;
268  }
269 
270  if (data == DONE || data == SPECIAL) // Save activated altars, too
271  {
272  OUT_SAVE_INST_DATA;
273 
274  std::ostringstream saveStream;
275  saveStream << m_auiEncounter[TYPE_SLADRAN] << " " << m_auiEncounter[TYPE_MOORABI] << " " << m_auiEncounter[TYPE_COLOSSUS] << " " << m_auiEncounter[TYPE_GALDARAH] << " "
276  << m_auiEncounter[TYPE_ECK];
277 
278  m_strInstData = saveStream.str();
279 
280  SaveToDB();
281  OUT_SAVE_INST_DATA_COMPLETE;
282  }

344 void instance_gundrak::DoAltarVisualEffect(uint8 uiType)
345 {
346  // Sort the lists if not yet done
347  if (!m_luiStalkerGUIDs.empty())
348  {
349  float fHeight = 10.0f; // A bit higher than the altar is needed
350  if (GameObject* pCollusAltar = GetSingleGameObjectFromStorage(GO_ALTAR_OF_COLOSSUS))
351  fHeight += pCollusAltar->GetPositionZ();
352 
353  CreatureList lStalkerTargets, lStalkerCasters;
354  for (GuidList::const_iterator itr = m_luiStalkerGUIDs.begin(); itr != m_luiStalkerGUIDs.end(); ++itr)
355  {
356  if (Creature* pStalker = instance->GetCreature(*itr))
357  {
358  if (pStalker->GetPositionZ() > fHeight)
359  lStalkerTargets.push_back(pStalker);
360  else
361  lStalkerCasters.push_back(pStalker);
362  }
363  }
364  m_luiStalkerGUIDs.clear();
365 
366  lStalkerTargets.sort(sortFromEastToWest);
367  lStalkerCasters.sort(sortFromEastToWest);
368 
369  for (CreatureList::const_iterator itr = lStalkerTargets.begin(); itr != lStalkerTargets.end(); ++itr)
370  m_vStalkerTargetGuids.push_back((*itr)->GetObjectGuid());
371  for (CreatureList::const_iterator itr = lStalkerCasters.begin(); itr != lStalkerCasters.end(); ++itr)
372  m_vStalkerCasterGuids.push_back((*itr)->GetObjectGuid());
373  }
374 
375  // Verify that the DB has enough trigger spawned
376  if (m_vStalkerTargetGuids.size() < 3 || m_vStalkerCasterGuids.size() < 3)
377  return;
378 
379  // Get the Index from the bosses
380  uint8 uiIndex = 0;
381  switch (uiType)
382  {
383  case TYPE_SLADRAN: uiIndex = 0; break;
384  case TYPE_COLOSSUS: uiIndex = 1; break;
385  case TYPE_MOORABI: uiIndex = 2; break;
386  default:
387  return;
388  }
389 
390  Creature* pTarget = instance->GetCreature(m_vStalkerTargetGuids[uiIndex]);
391  Creature* pCaster = instance->GetCreature(m_vStalkerCasterGuids[uiIndex]);
392 
393  if (!pTarget || !pCaster)
394  return;
395 
396  uint32 auiFireBeamSpells[3] = {SPELL_BEAM_SNAKE, SPELL_BEAM_ELEMENTAL, SPELL_BEAM_MAMMOTH};
397 
398  // Cast from Caster to Target
399  pCaster->CastSpell(pTarget, auiFireBeamSpells[uiIndex], TRIGGERED_NONE);
400 }

402 void instance_gundrak::Update(uint32 diff)
403 {
404  // Possible multible altars used at the same time, process their timers
405  if (!m_mAltarInProgress.empty())
406  {
407  for (TypeTimerMap::iterator itr = m_mAltarInProgress.begin(); itr != m_mAltarInProgress.end();)
408  {
409  if (itr->second < diff)
410  {
411  // Do Visual Effect
412  DoAltarVisualEffect(itr->first);
413  // Set Timer for Beam-Duration
414  m_mBeamInProgress.insert(TypeTimerPair(itr->first, TIMER_VISUAL_BEAM));
415  // Remove this timer, as processed
416  m_mAltarInProgress.erase(itr++);
417  }
418  else
419  {
420  itr->second -= diff;
421  ++itr;
422  }
423  }
424  }
425 
426  // Possible multible beams used at the same time, process their timers
427  if (!m_mBeamInProgress.empty())
428  {
429  for (TypeTimerMap::iterator itr = m_mBeamInProgress.begin(); itr != m_mBeamInProgress.end();)
430  {
431  if (itr->second < diff)
432  {
433  // Use Key
434  switch (itr->first)
435  {
436  case TYPE_SLADRAN: DoUseOpenableObject(GO_SNAKE_KEY, false); break;
437  case TYPE_MOORABI: DoUseOpenableObject(GO_MAMMOTH_KEY, false); break;
438  case TYPE_COLOSSUS: DoUseOpenableObject(GO_TROLL_KEY, false); break;
439  }
440  // Set Timer for Beam-Duration
441  m_mKeyInProgress.insert(TypeTimerPair(itr->first, TIMER_VISUAL_KEY));
442  m_mBeamInProgress.erase(itr++);
443  }
444  else
445  {
446  itr->second -= diff;
447  ++itr;
448  }
449  }
450  }
451 
452  // Activate Bridge if all Three Encounters are used
453  if (!m_mKeyInProgress.empty())
454  {
455  for (TypeTimerMap::iterator itr = m_mKeyInProgress.begin(); itr != m_mKeyInProgress.end();)
456  {
457  if (itr->second < diff)
458  {
459  // Activate Bridge (and all other Keys) if we are on the last Key, and all other keys are already set
460  if (m_auiEncounter[0] == SPECIAL && m_auiEncounter[1] == SPECIAL && m_auiEncounter[2] == SPECIAL
461  && m_mAltarInProgress.empty() && m_mBeamInProgress.empty() && m_mKeyInProgress.size() == 1)
462  {
463  DoUseDoorOrButton(GO_COLLISION);
464  DoUseDoorOrButton(GO_RHINO_KEY, 0, true);
465 
466  // The already closed keys cannot be done with DoUseDoorOrButton
467  if (GameObject* pTrollKey = GetSingleGameObjectFromStorage(GO_TROLL_KEY))
468  pTrollKey->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
469  if (GameObject* pMammothKey = GetSingleGameObjectFromStorage(GO_MAMMOTH_KEY))
470  pMammothKey->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
471  if (GameObject* pSnakeKey = GetSingleGameObjectFromStorage(GO_SNAKE_KEY))
472  pSnakeKey->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
473 
474  // GO_BRIDGE is type 35 (TRAP_DOOR) and needs to be handled directly - creates a path for players
475  DoUseOpenableObject(GO_BRIDGE, false);
476  }
477  // Remove this timer, as processed
478  m_mKeyInProgress.erase(itr++);
479  }
480  else
481  {
482  itr->second -= diff;
483  ++itr;
484  }
485  }
486  }
