180 void instance_pinnacle::SetData(uint32 uiType, uint32 uiData)
181 {
182  switch (uiType)
183  {
184  case TYPE_SVALA:
185  if (uiData == IN_PROGRESS || uiData == FAIL)
186  SetSpecialAchievementCriteria(TYPE_ACHIEV_INCREDIBLE_HULK, false);
187  else if (uiData == SPECIAL)
188  StartNextDialogueText(NPC_ARTHAS_IMAGE);
189  m_auiEncounter[uiType] = uiData;
190  break;
191  case TYPE_GORTOK:
192  if (uiData == IN_PROGRESS)
193  {
194  if (Creature* pOrb = instance->GetCreature(m_gortokEventTriggerGuid))
195  {
196  pOrb->SetLevitate(true);
197  pOrb->CastSpell(pOrb, SPELL_ORB_VISUAL, TRIGGERED_OLD_TRIGGERED);
198  pOrb->GetMotionMaster()->MovePoint(0, aOrbPositions[0][0], aOrbPositions[0][1], aOrbPositions[0][2]);
199 
200  m_uiGortokOrbTimer = 2000;
201  }
202  }
203  else if (uiData == FAIL)
204  {
205  if (Creature* pOrb = instance->GetCreature(m_gortokEventTriggerGuid))
206  {
207  if (!pOrb->IsAlive())
208  pOrb->Respawn();
209  else
210  pOrb->RemoveAllAuras();
211 
212  // For some reasone the Orb doesn't evade automatically
213  pOrb->GetMotionMaster()->MoveTargetedHome();
214  }
215 
216  for (unsigned int aGortokMiniBosse : aGortokMiniBosses)
217  {
218  // Reset each miniboss
219  if (Creature* pTemp = GetSingleCreatureFromStorage(aGortokMiniBosse))
220  {
221  if (!pTemp->IsAlive())
222  pTemp->Respawn();
223 
224  pTemp->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
225  }
226  }
227 
228  m_uiGortokOrbPhase = 0;
229  }
230  m_auiEncounter[uiType] = uiData;
231  break;
232  case TYPE_SKADI:
233  // Don't process the event twice
234  if (m_auiEncounter[uiType] == uiData)
235  return;
236  switch (uiData)
237  {
238  case DONE:
239  DoUseDoorOrButton(GO_DOOR_SKADI);
240  m_uiGauntletCheckTimer = 0;
241  break;
242  case SPECIAL:
243  // Prepare achievements
244  SetSpecialAchievementCriteria(TYPE_ACHIEV_LOVE_SKADI, true);
245  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_SKADI_ID);
246 
247  m_uiGauntletCheckTimer = 7000;
248  m_auiEncounter[uiType] = uiData;
249  return;
250  case FAIL:
251  {
252  // Handle event reset
253  if (Creature* pGrauf = GetSingleCreatureFromStorage(NPC_GRAUF))
254  pGrauf->ForcedDespawn();
255  if (Creature* pSkadi = GetSingleCreatureFromStorage(NPC_SKADI))
256  pSkadi->ForcedDespawn();
257 
258  m_uiGauntletCheckTimer = 0;
259  m_uiSkadiResetTimer = 30000;
260 
261  // remove periodic gauntlet spell on all players
262  Map::PlayerList const& lPlayers = instance->GetPlayers();
263  for (const auto& lPlayer : lPlayers)
264  {
265  if (Player* pPlayer = lPlayer.getSource())
266  pPlayer->RemoveAurasDueToSpell(SPELL_GAUNTLET_PERIODIC);
267  }
268  }
269  // no break;
270  case NOT_STARTED:
271  // Despawn all summons
272  for (GuidList::const_iterator itr = m_lskadiGauntletMobsList.begin(); itr != m_lskadiGauntletMobsList.end(); ++itr)
273  {
274  if (Creature* pYmirjar = instance->GetCreature(*itr))
275  pYmirjar->ForcedDespawn();
276  }
277  // no break;
278  case IN_PROGRESS:
279 
280  // Remove the summon aura on phase 2 or fail
281  if (Creature* pTrigger = instance->GetCreature(m_skadiMobsTriggerGuid))
282  pTrigger->RemoveAllAuras();
283  break;
284  }
285  m_auiEncounter[uiType] = uiData;
286  break;
287  case TYPE_YMIRON:
288  if (uiData == DONE)
289  DoUseDoorOrButton(GO_DOOR_YMIRON);
290  else if (uiData == IN_PROGRESS)
291  SetSpecialAchievementCriteria(TYPE_ACHIEV_KINGS_BANE, true);
292  else if (uiData == SPECIAL)
293  SetSpecialAchievementCriteria(TYPE_ACHIEV_KINGS_BANE, false);
294  m_auiEncounter[uiType] = uiData;
295  break;
296  default:
297  script_error_log("Instance Pinnacle: SetData = %u for type %u does not exist/not implemented.", uiType, uiData);
298  return;
299  }
300 
301  // Saving also SPECIAL for this instance
302  if (uiData == DONE || uiData == SPECIAL)
303  {
304  OUT_SAVE_INST_DATA;
305 
306  std::ostringstream saveStream;
307  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
308 
309  m_strInstData = saveStream.str();
310 
311  SaveToDB();
312  OUT_SAVE_INST_DATA_COMPLETE;
313  }
314 }

462 void instance_pinnacle::Update(uint32 const uiDiff)
463 {
464  DialogueUpdate(uiDiff);
465 
466  if (m_uiGortokOrbTimer)
467  {
468  if (m_uiGortokOrbTimer <= uiDiff)
469  {
470  if (!m_uiGortokOrbPhase)
471  {
472  if (Creature* pOrb = instance->GetCreature(m_gortokEventTriggerGuid))
473  pOrb->GetMotionMaster()->MovePoint(0, aOrbPositions[1][0], aOrbPositions[1][1], aOrbPositions[1][2]);
474 
475  m_uiGortokOrbTimer = 18000;
476  }
477  // Awaken Gortok if this is the last phase
478  else
479  {
480  uint8 uiMaxOrbPhase = instance->IsRegularDifficulty() ? 3 : 5;
481  uint32 uiSpellId = m_uiGortokOrbPhase == uiMaxOrbPhase ? SPELL_AWAKEN_GORTOK : SPELL_AWAKEN_SUBBOSS;
482 
483  if (Creature* pOrb = instance->GetCreature(m_gortokEventTriggerGuid))
484  {
485  pOrb->CastSpell(pOrb, uiSpellId, TRIGGERED_NONE);
486 
487  if (m_uiGortokOrbPhase == uiMaxOrbPhase)
488  pOrb->ForcedDespawn(10000);
489  }
490 
491  m_uiGortokOrbTimer = 0;
492  }
493  ++m_uiGortokOrbPhase;
494  }
495  else
496  m_uiGortokOrbTimer -= uiDiff;
497  }
498 
499  if (m_uiGauntletCheckTimer)
500  {
501  if (m_uiGauntletCheckTimer <= uiDiff)
502  {
503  if (Creature* pTrigger = GetSingleCreatureFromStorage(NPC_WORLD_TRIGGER_LARGE))
504  pTrigger->CastSpell(pTrigger, SPELL_GAUNTLET_RESET_CHECK, TRIGGERED_NONE);
505 
506  m_uiGauntletCheckTimer = 7000;
507  }
508  else
509  m_uiGauntletCheckTimer -= uiDiff;
510  }
511 
512  // Reset Skadi encounter if fail
513  if (m_uiSkadiResetTimer)
514  {
515  if (m_uiSkadiResetTimer <= uiDiff)
516  {
517  if (Creature* pGrauf = GetSingleCreatureFromStorage(NPC_GRAUF))
518  pGrauf->Respawn();
519 
520  if (Creature* pSkadi = GetSingleCreatureFromStorage(NPC_SKADI))
521  pSkadi->Respawn();
522 
523  m_uiSkadiResetTimer = 0;
524  }
525  else
526  m_uiSkadiResetTimer -= uiDiff;
527  }
528 }
