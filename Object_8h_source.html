225  void Update(TimePoint const& now)
226  {
227  auto spellCDItr = m_spellIdMap.begin();
228  while (spellCDItr != m_spellIdMap.end())
229  {
230  auto& cd = spellCDItr->second;
231  if (cd->IsSpellCDExpired(now) && cd->IsCatCDExpired(now)) // will not remove permanent CD
232  spellCDItr = erase(spellCDItr);
233  else
234  {
235  if (cd->m_category && cd->IsCatCDExpired(now))
236  {
237  m_categoryMap.erase(cd->m_category);
238  cd->m_category = 0;
239  }
240  ++spellCDItr;
241  }
242  }
243  }
244 
245  bool AddCooldown(TimePoint clockNow, uint32 spellId, uint32 duration, uint32 spellCategory = 0, uint32 categoryDuration = 0, uint32 itemId = 0, bool onHold = false)
246  {
247  RemoveBySpellId(spellId);
248  auto resultItr = m_spellIdMap.emplace(spellId, std::move(std::unique_ptr<CooldownData>(new CooldownData(clockNow, spellId, duration, spellCategory, categoryDuration, itemId, onHold))));
249  // do not overwrite one permanent category cooldown with another permanent category cooldown
250  if (resultItr.second && spellCategory && categoryDuration)
251  {
252  auto catItr = FindByCategory(spellCategory);
253  if (!onHold || catItr == m_spellIdMap.end() || !catItr->second->IsPermanent())
254  {
255  // we must keep original category cd owner for sake of client sync
256  if (catItr != m_spellIdMap.end())
257  {
258  catItr->second->SetCatCDExpireTime(std::chrono::milliseconds(categoryDuration) + clockNow);
259  catItr->second->m_typePermanent = false;
260  resultItr.first->second->m_category = 0;
261  }
262  else
263  m_categoryMap.emplace(spellCategory, resultItr.first);
264  }
265  else
266  resultItr.first->second->m_category = 0;
267  }
268 
269  return resultItr.second;
270  }
271 
272  void RemoveBySpellId(uint32 spellId)
273  {
274  auto spellCDItr = m_spellIdMap.find(spellId);
275  if (spellCDItr != m_spellIdMap.end())
276  {
277  auto& cdData = spellCDItr->second;
278  if (cdData->m_category)
279  {
280  auto catCDItr = m_categoryMap.find(cdData->m_category);
281  if (catCDItr != m_categoryMap.end())
282  m_categoryMap.erase(catCDItr);
283  }
284  m_spellIdMap.erase(spellCDItr);
285  }
286  }
287 
288  void RemoveByCategory(uint32 category)
289  {
290  auto spellCDItr = m_categoryMap.find(category);
291  if (spellCDItr != m_categoryMap.end())
292  {
293  spellCDItr->second->second->m_category = 0;
294  m_categoryMap.erase(spellCDItr);
295  }
296  }
297 
298  Iterator erase(ConstIterator spellCDItr)
299  {
300  auto& cdData = spellCDItr->second;
301  if (cdData->m_category)
302  {
303  auto catCDItr = m_categoryMap.find(cdData->m_category);
304  if (catCDItr != m_categoryMap.end())
305  m_categoryMap.erase(catCDItr);
306  }
307  return m_spellIdMap.erase(spellCDItr);
308  }

310  ConstIterator FindBySpellId(uint32 id) const { return m_spellIdMap.find(id); }

312  ConstIterator FindByCategory(uint32 category) const
313  {

315  return itr != m_categoryMap.end() ? itr->second : end();
