57 struct boss_anetheronAI : public CombatAI

59  boss_anetheronAI(Creature* creature) : CombatAI(creature, ANETHERON_ACTION_MAX), m_instance(static_cast<ScriptedInstance*>(creature->GetInstanceData()))
60  {
61  AddCombatAction(ANETHERON_ACTION_CARRION_SWARM, 20000, 28000);
62  AddCombatAction(ANETHERON_ACTION_SLEEP, 25000, 32000);
63  AddCombatAction(ANETHERON_ACTION_INFERNO, 30000, 48000);
64  AddCombatAction(ANETHERON_ACTION_ENRAGE, 600000u);
65  AddCustomAction(ANETHERON_ACTION_INFERNAL_STUN, true, [&]()
66  {
67  // Last summoned infernal
68  if (Creature* infernal = m_creature->GetMap()->GetCreature(m_infernals[m_infernals.size() - 1]))
69  {
70  UnitAI* ai = infernal->AI();
71  ai->SetReactState(REACT_AGGRESSIVE);
72  ai->DoCastSpellIfCan(nullptr, SPELL_INFERNAL_STUN);
73  infernal->SetInCombatWithZone();
74  }
75  });
76  AddOnKillText(SAY_KILL1, SAY_KILL2, SAY_KILL3);
77  Reset();
78  }

95  uint32 GetSubsequentActionTimer(const uint32 action) const

151  void ExecuteAction(uint32 action) override
152  {
153  switch (action)
154  {
155  case ANETHERON_ACTION_CARRION_SWARM:
156  {
157  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_CARRION_SWARM, SELECT_FLAG_PLAYER))
158  {
159  if (DoCastSpellIfCan(target, SPELL_CARRION_SWARM) == CAST_OK)
160  {
161  DoScriptText(urand(0, 1) ? SAY_SWARM1 : SAY_SWARM2, m_creature);
162  ResetCombatAction(action, GetSubsequentActionTimer(action));
163  }
164  }
165  break;
166  }
167  case ANETHERON_ACTION_SLEEP:
168  {
169  if (DoCastSpellIfCan(nullptr, SPELL_SLEEP) == CAST_OK)
170  {
171  DoScriptText(urand(0, 1) ? SAY_SLEEP1 : SAY_SLEEP2, m_creature);
172  ResetCombatAction(action, GetSubsequentActionTimer(action));
173  }
174  break;
175  }
176  case ANETHERON_ACTION_INFERNO:
177  {
178  // cooldown done on cooldown added
179  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_INFERNO, SELECT_FLAG_PLAYER))
180  if (DoCastSpellIfCan(target, SPELL_INFERNO) == CAST_OK)
181  DoScriptText(urand(0, 1) ? SAY_INFERNO1 : SAY_INFERNO2, m_creature);
182  break;
183  }
184  case ANETHERON_ACTION_ENRAGE:
185  {
186  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
187  ResetCombatAction(action, GetSubsequentActionTimer(action));
188  break;
189  }
190  }
191  }
