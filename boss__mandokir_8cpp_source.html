155  void ReceiveAIEvent(AIEventType eventType, Unit* /*sender*/, Unit* invoker, uint32 /*miscValue*/) override
156  {
157  if (eventType == AI_EVENT_CUSTOM_A)
158  {
159  DoScriptText(SAY_WATCH, m_creature, invoker);
160  DoScriptText(SAY_WATCH_WHISPER, m_creature, invoker);
161  m_targetThreat = m_creature->getThreatManager().getThreat(invoker);
162  }
163  else if (eventType == AI_EVENT_CUSTOM_B)
164  {
165  Player* watchTarget = dynamic_cast<Player*>(invoker);
166 
167  // If threat is higher that previously saved, mandokir will act
168  if (watchTarget && watchTarget->IsAlive() && m_creature->getThreatManager().getThreat(watchTarget) > m_targetThreat)
169  {
170  if (!m_creature->IsWithinLOSInMap(watchTarget))
171  m_creature->CastSpell(watchTarget, SPELL_SUMMON_PLAYER, TRIGGERED_OLD_TRIGGERED);
172 
173  m_creature->CastSpell(watchTarget, SPELL_CHARGE_GAZE, TRIGGERED_OLD_TRIGGERED);
174  }
175  }
176  }

178  void KilledUnit(Unit* victim) override
179  {

182  ++m_uiKillCount;
183 
184  if (m_uiKillCount == 3)
185  {
186  DoScriptText(SAY_DING_KILL, m_creature);

191  {
192  if (pJindo->IsAlive())
193  DoScriptText(SAY_GRATS_JINDO, pJindo);
194  }
195  }
196 
197  DoCastSpellIfCan(nullptr, SPELL_LEVEL_UP, CAST_TRIGGERED);
198  m_uiKillCount = 0;
199  }
200 
201  if (m_creature->IsInCombat())
202  {
203  if (Creature* spirit = GetClosestCreatureWithEntry(victim, NPC_CHAINED_SPIRIT, 50.0f))
204  spirit->CastSpell(victim, SPELL_REVIVE, TRIGGERED_NONE);
205  }
206  }
