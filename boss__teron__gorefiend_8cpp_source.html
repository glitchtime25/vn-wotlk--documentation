119  uint32 GetInitialActionTimer(GorefiendActions id)
120  {
121  switch (id)
122  {
123  case GOREFIEND_ACTION_DOOM_BLOSSOM: return urand(5000, 10000);
124  case GOREFIEND_ACTION_INCINERATE: return urand(20000, 30000);;

206  void ExecuteAction(uint32 action) override
207  {
208  switch (action)
209  {
210  case GOREFIEND_ACTION_BERSERK:
211  {
212  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
213  {
214  DoScriptText(EMOTE_BERSERK, m_creature);
215  DisableCombatAction(action);
216  }
217  return;
218  }
219  case GOREFIEND_ACTION_DOOM_BLOSSOM:
220  {
221  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_DOOM_BLOSSOM) == CAST_OK)
222  {
223  if (urand(0, 1))
224  DoScriptText(urand(0, 1) ? SAY_SPELL1 : SAY_SPELL2, m_creature);
225 
226  ResetCombatAction(action, GetSubsequentActionTimer(GorefiendActions(action)));
227  }
228  return;
229  }
230  case GOREFIEND_ACTION_INCINERATE:
231  {
232  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_SKIP_TANK);
233  if (DoCastSpellIfCan(target, SPELL_INCINERATE) == CAST_OK)
234  ResetCombatAction(action, GetSubsequentActionTimer(GorefiendActions(action)));
235  return;
236  }
237  case GOREFIEND_ACTION_SHADOW_OF_DEATH:
238  {
239  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SHADOW_OF_DEATH, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA | SELECT_FLAG_SKIP_TANK))
240  {
241  if (DoCastSpellIfCan(target, SPELL_SHADOW_OF_DEATH) == CAST_OK)
242  {
243  DoScriptText(urand(0, 1) ? SAY_SPECIAL1 : SAY_SPECIAL2, m_creature);
244  return;
245  }
246  }
247  return;
248  }
249  case GOREFIEND_ACTION_CRUSHING_SHADOWS:
250  {
251  if (DoCastSpellIfCan(nullptr, SPELL_CRUSHING_SHADOWS) == CAST_OK)
252  {
253  if (urand(0, 1))
254  DoScriptText(urand(0, 1) ? SAY_SPECIAL1 : SAY_SPECIAL2, m_creature);
255 
256  ResetCombatAction(action, GetSubsequentActionTimer(GorefiendActions(action)));
257  return;
258  }
259  return;
260  }
261  }
262  }
