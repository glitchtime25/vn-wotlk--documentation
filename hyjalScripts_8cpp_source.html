54  void ExecuteAction(uint32 action) override
55  {
56  switch (action)
57  {
58  case JAINA_20:
59  if (m_creature->GetHealthPercent() <= 20.f)
60  {
61  SetActionReadyStatus(action, false);
62  DoScriptText(SAY_CALL_FOR_HELP_EMOTE, m_creature);
63  DoCallForHelp(30.f);
64  }
65  break;
66  case JAINA_BLIZZARD:
67  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
68  if (DoCastSpellIfCan(target, SPELL_BLIZZARD) == CAST_OK)
69  ResetCombatAction(action, urand(15000, 35000));
70  break;
71  case JAINA_PYROBLAST:
72  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
73  if (DoCastSpellIfCan(target, SPELL_PYROBLAST) == CAST_OK)
74  ResetCombatAction(action, urand(2000, 9000));
75  break;
76  case JAINA_SUMMON_ELEMENTALS:
77  if (m_creature->CountGuardiansWithEntry(NPC_WATER_ELEMENTAL) == 0)
78  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_ELEMENTALS) == CAST_OK)
79  ResetCombatAction(action, urand(15000, 45000));
80  break;
81  default: hyjalAI::ExecuteAction(action); break;
82  }

119  player->CLOSE_GOSSIP_MENU();
120  break;
121  }
122  }
123  }

150  bool CanUseFeralSpiritsAgain()
151  {
152  uint32 count = 0;
153 
154  for (auto spiritGuid : m_feralSpirits)
155  if (Creature* spirit = m_creature->GetMap()->GetAnyTypeCreature(spiritGuid))
156  if (spirit->IsAlive())
157  count++;
158 
159  if (count == 0)
160  m_feralSpirits.clear();
161 
162  return count == 0;
163  }
