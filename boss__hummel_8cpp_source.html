114  void ExecuteAction(uint32 action) override
115  {
116  switch (action)
117  {
118  case HUMMEL_ACTION_PERFUME_SPRAY:
119  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ALLURING_PERFUME_SPRAY) == CAST_OK)
120  ResetCombatAction(action, 5000);
121  break;
122  case HUMMEL_ACTION_CHAIN_REACTION:
123  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CHAIN_REACTION) == CAST_OK)
124  {
125  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SUMMON_TABLE, CAST_TRIGGERED);
126  ResetCombatAction(action, 25000);
127  }
128  break;
129  case HUMMEL_ACTION_SUMMON_ADD:
130  if (m_instance) // TODO: These are meant to come from different sides
131  {
132  if (Creature* generator = m_instance->GetSingleCreatureFromStorage(NPC_APOTHECARY_GENERATOR))
133  generator->CastSpell(generator, SPELL_SUMMON_VALENTINE_ADD, TRIGGERED_OLD_TRIGGERED);
134 
135  ResetCombatAction(action, 15000);
136  }
137  break;
138  }
