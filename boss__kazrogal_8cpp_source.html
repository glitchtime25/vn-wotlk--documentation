69  uint32 GetInitialActionTimer(const uint32 action) const
70  {
71  switch (action)
72  {

74  case KAZROGAL_ACTION_WAR_STOMP: return urand(12000, 18000);
75  case KAZROGAL_ACTION_CRIPPLE: return 15000;

81  uint32 GetSubsequentActionTimer(const uint32 action) const
82  {
83  switch (action)

86  case KAZROGAL_ACTION_WAR_STOMP: return urand(15000, 30000);

107  void ExecuteAction(uint32 action) override
108  {
109  switch (action)
110  {
111  case KAZROGAL_ACTION_MALEVOLENT_CLEAVE:
112  {
113  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MALEVOLENT_CLEAVE) == CAST_OK)
114  ResetCombatAction(action, GetSubsequentActionTimer(action));
115  break;
116  }
117  case KAZROGAL_ACTION_WAR_STOMP:
118  {
119  if (DoCastSpellIfCan(nullptr, SPELL_WAR_STOMP) == CAST_OK)
120  ResetCombatAction(action, GetSubsequentActionTimer(action));
121  break;
122  }
123  case KAZROGAL_ACTION_CRIPPLE:
124  {
125  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_CRIPPLE, SELECT_FLAG_PLAYER))
126  if (DoCastSpellIfCan(target, SPELL_CRIPPLE) == CAST_OK)
127  ResetCombatAction(action, GetSubsequentActionTimer(action));
128  break;
129  }
130  case KAZROGAL_ACTION_MARK_OF_KAZROGAL:
131  {
132  if (DoCastSpellIfCan(nullptr, SPELL_MARK_OF_KAZROGAL) == CAST_OK)
133  {
134  DoScriptText(urand(0, 1) ? SAY_MARK1 : SAY_MARK2, m_creature);
135  ++m_markOfKazrogalCounter;
136  ResetCombatAction(action, GetSubsequentActionTimer(action));
137  }
138  break;
139  }
140  }
141  }
