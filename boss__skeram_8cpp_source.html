183  if (boss_skeramAI* skeramAI = dynamic_cast<boss_skeramAI*>(prophet->AI()))
184  teleportSpellID = skeramAI->GetAvailableTeleport();

186  }
187  else

189 
190  // Teleport, reset thread (and restore visibility if needed)
191  DoCastSpellIfCan(nullptr, teleportSpellID);
192  DoResetThreat();
193  ClearMark();
194  m_creature->RemoveAllAuras();
195 
196  if (attacking)
197  StopAttacking();
198 
199  ResetTimer(SKERAM_BLINK_DELAY, 2000);
200  }

244  void ExecuteAction(uint32 action) override
245  {
246  switch (action)
247  {
248  case SKERAM_SPLIT:
249  {
250  // Summon images at 75%, 50% and 25%
251  if (m_creature->GetHealthPercent() < m_hpCheck)
252  {
253  DespawnGuids(m_images);
254  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_IMAGES) == CAST_OK)
255  {
256  m_hpCheck -= 25.0f;
257  StopAttacking();
258  // Teleport shortly after the images are summoned and set invisible to clear the selection (Workaround alert!!!)
259  ResetTimer(SKERAM_INIT_IMAGES, 1500);
260  }
261  }
262  break;
263  }
264  case SKERAM_ARCANE_EXPLOSION:
265  {
266  // Arcane Explosion is done if more than a set number of people are in melee range
267  PlayerList meleePlayerList;
268  float meleeRange = m_creature->GetCombinedCombatReach(m_creature->GetVictim(), true);
269  GetPlayerListWithEntryInWorld(meleePlayerList, m_creature, meleeRange);
270  if (meleePlayerList.size() >= m_maxMeleeAllowed)
271  {
272  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_EXPLOSION) == CAST_OK)
273  ResetCombatAction(action, urand(8, 18) * IN_MILLISECONDS);
274  }
275  else // Recheck in 1 second
276  ResetCombatAction(action, 1 * IN_MILLISECONDS);
277  break;
278  }
279  case SKERAM_TRUE_FULFILMENT:
280  {
281  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 1, SPELL_TRUE_FULFILLMENT, SELECT_FLAG_PLAYER))
282  if (DoCastSpellIfCan(target, SPELL_TRUE_FULFILLMENT) == CAST_OK)
283  ResetCombatAction(action, urand(20, 30) * IN_MILLISECONDS);
284  break;
285  }
286  case SKERAM_BLINK:
287  {
288  std::shuffle(m_teleports.begin(), m_teleports.end(), *GetRandomGenerator()); // Shuffle the teleport spells to ensure that boss and images have a different location assigned randomly
289  m_teleportCounter = 1;
290  DoTeleport(true);
291  for (auto guid : m_images)
292  if (Creature* image = m_creature->GetMap()->GetCreature(guid))
293  static_cast<boss_skeramAI*>(image->AI())->DoTeleport(true);
294  ResetCombatAction(action, urand(10, 30) * IN_MILLISECONDS);
295  break;
296  }
297  case SKERAM_EARTH_SHOCK:
298  {
299  uint32 timer = 500;
300  // If victim exists we have a target in melee range
301  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
302  m_rangeCheckState = -1;
303  // Spam Waterbolt spell when not tanked
304  else
305  {
306  ++m_rangeCheckState;
307  if (m_rangeCheckState > 1)
308  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_EARTH_SHOCK) == CAST_OK)
309  timer = 2500;
310  }
311  ResetCombatAction(action, timer);
312  break;
313  }
314  }
315  }
