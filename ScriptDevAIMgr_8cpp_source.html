103 
104  if (iTextEntry >= 0)
105  {
106  script_error_log("DoOrSimulateScriptTextForMap with source entry %u for map %u attempts to process text entry %i, but text entry must be negative.", uiCreatureEntry, pMap->GetId(), iTextEntry);

114  return;
115  }
116 
117  MangosStringLocale const* pData = GetMangosStringData(iTextEntry);
118  if (!pData)
119  {
120  script_error_log("DoOrSimulateScriptTextForMap with source entry %u for map %u could not find text entry %i.", uiCreatureEntry, pMap->GetId(), iTextEntry);
121  return;
122  }
123 
124  debug_log("SD2: DoOrSimulateScriptTextForMap: text entry=%i, Sound=%u, Type=%u, Language=%u, Emote=%u",
125  iTextEntry, pData->SoundId, pData->Type, pData->LanguageId, pData->Emote);
126 
127  if (pData->Type != CHAT_TYPE_ZONE_YELL && pData->Type != CHAT_TYPE_ZONE_EMOTE)
128  {
129  script_error_log("DoSimulateScriptTextForMap entry %i has not supported chat type %u.", iTextEntry, pData->Type);
130  return;
131  }
132 
133  if (pData->SoundId)
134  pMap->PlayDirectSoundToMap(pData->SoundId);
135 
136  ChatMsg chatMsg = (pData->Type == CHAT_TYPE_ZONE_EMOTE ? CHAT_MSG_MONSTER_EMOTE : CHAT_MSG_MONSTER_YELL);
137 
138  if (pCreatureSource) // If provided pointer for sayer, use direct version
139  pMap->MonsterYellToMap(pCreatureSource->GetObjectGuid(), iTextEntry, chatMsg, pData->LanguageId, pTarget);
140  else // Simulate yell

534 void ScriptDevAIMgr::LoadScriptNames()
535 {
536  m_scriptNames.push_back("");
537  QueryResult* result = WorldDatabase.Query(
538  "SELECT DISTINCT(ScriptName) FROM creature_template WHERE ScriptName <> '' "
539  "UNION "
540  "SELECT DISTINCT(ScriptName) FROM gameobject_template WHERE ScriptName <> '' "
541  "UNION "
542  "SELECT DISTINCT(ScriptName) FROM item_template WHERE ScriptName <> '' "
543  "UNION "
544  "SELECT DISTINCT(ScriptName) FROM scripted_areatrigger WHERE ScriptName <> '' "
545  "UNION "
546  "SELECT DISTINCT(ScriptName) FROM scripted_event_id WHERE ScriptName <> '' "
547  "UNION "
548  "SELECT DISTINCT(ScriptName) FROM instance_template WHERE ScriptName <> '' "
549  "UNION "
550  "SELECT DISTINCT(ScriptName) FROM world_template WHERE ScriptName <> ''");
551 
552  if (!result)
553  {
554  BarGoLink bar(1);
555  bar.step();
556  sLog.outErrorDb(">> Loaded empty set of Script Names!");
557  sLog.outString();
558  return;
559  }
560 
561  BarGoLink bar(result->GetRowCount());
562  uint32 count = 0;
563 
564  do
565  {
566  bar.step();
567  m_scriptNames.push_back((*result)[0].GetString());
568  ++count;
569  }
570  while (result->NextRow());
571  delete result;
572 
573  std::sort(m_scriptNames.begin(), m_scriptNames.end());
574 
575  sLog.outString(">> Loaded %d Script Names", count);
576  sLog.outString();
577 }

590 uint32 ScriptDevAIMgr::GetScriptId(const char* name) const
591 {
592  // use binary search to find the script name in the sorted vector
593  // assume "" is the first element
594  if (!name)
595  return 0;
596 
597  ScriptNameMap::const_iterator itr = std::lower_bound(m_scriptNames.begin(), m_scriptNames.end(), name);
598 
599  if (itr == m_scriptNames.end() || *itr != name)
600  return 0;
601 
602  return uint32(itr - m_scriptNames.begin());
603 }
