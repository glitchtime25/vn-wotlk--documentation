99  uint32 GetSubsequentActionTimer(const uint32 action) const
100  {

112  void ExecuteAction(uint32 action) override
113  {
114  switch (action)
115  {
116  case MUSELEK_TRAP_ONE:
117  {
118  if (m_creature->GetHealthPercent() < 70.f)
119  {
120  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_THROW_FREEZING_TRAP, TRIGGERED_NONE);
121  DoCastSpellIfCan(nullptr, SPELL_HUNTERS_MARK);
122  DistanceYourself();
123  ResetTimer(MUSELEK_ACTION_DETERRENCE, urand(15000, 20000));
124  SetActionReadyStatus(action, false);
125  }
126  break;
127  }
128  case MUSELEK_TRAP_TWO:
129  {
130  if (m_creature->GetHealthPercent() < 30.f)
131  {
132  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_THROW_FREEZING_TRAP, TRIGGERED_NONE);
133  DoCastSpellIfCan(nullptr, SPELL_HUNTERS_MARK);
134  DistanceYourself();
135  SetActionReadyStatus(action, false);
136  }
137  break;
138  }
139  case MUSELEK_AIMED_SHOT:
140  {
141  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_AIMED_SHOT) == CAST_OK)
142  SetActionReadyStatus(action, false);
143  break;
144  }
145  case MUSELEK_ACTION_KNOCK_AWAY:
146  {
147  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_KNOCK_AWAY) == CAST_OK)
148  ResetCombatAction(action, GetSubsequentActionTimer(action));
149  return;
150  }
151  case MUSELEK_ACTION_RAPTOR_STRIKE:
152  {
153  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_RAPTOR_STRIKE) == CAST_OK)
154  ResetCombatAction(action, GetSubsequentActionTimer(action));
155  return;
156  }
157  case MUSELEK_ACTION_BEAR_COMMAND:
158  {
159  if (Creature* claw = m_instance->GetSingleCreatureFromStorage(NPC_CLAW))
160  if (claw->IsAlive() && claw->GetEntry() != NPC_CLAW_DRUID_FORM)
161  {
162  uint8 claw_spell = urand(0, 2);
163 
164  switch (claw_spell)
165  {
166  case 0:
167  claw->AI()->DoCastSpellIfCan(claw, SPELL_MAUL);
168  break;
169  case 1:
170  claw->AI()->DoCastSpellIfCan(claw, SPELL_ROAR);
171  break;
172  default:
173  break;
174  }
175 
176  claw->AI()->DoCastSpellIfCan(claw, SPELL_FRENZY);
177  DoScriptText(SAY_COMMAND, m_creature);
178  }
179 
180  ResetCombatAction(action, GetSubsequentActionTimer(action));
181  return;
182  }
183  case MUSELEK_ACTION_RANGED_ATTACK:
184  {
185  if (!GetCurrentRangedMode())
186  return;
187 
188  uint32 RangedSpell = urand(0, 4) ? SPELL_SHOOT : SPELL_MULTI_SHOT; // 66% shoot, 33% multishot
189  if (DoCastSpellIfCan(m_creature->GetVictim(), RangedSpell) == CAST_OK)
190  ResetCombatAction(action, GetSubsequentActionTimer(action));
191  return;
192  }
193  case MUSELEK_ACTION_DETERRENCE:
194  {
195  if (DoCastSpellIfCan(nullptr, SPELL_DETERRENCE) == CAST_OK)
196  ResetCombatAction(action, GetSubsequentActionTimer(action));
197  return;
198  }
199  }
200  }
