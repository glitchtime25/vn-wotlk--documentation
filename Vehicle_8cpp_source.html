327 void VehicleInfo::SwitchSeat(Unit* passenger, uint8 seat)
328 {
329  MANGOS_ASSERT(passenger);
330 
331  DEBUG_LOG("VehicleInfo::SwitchSeat: passenger: %s try to switch to seat %u", passenger->GetGuidStr().c_str(), seat);
332 
333  // Switching seats is not possible
334  if (m_vehicleEntry->m_flags & VEHICLE_FLAG_DISABLE_SWITCH)
335  return;
336 
337  PassengerMap::const_iterator itr = m_passengers.find(passenger);
338  MANGOS_ASSERT(itr != m_passengers.end());
339 
340  // We are already boarded to this seat
341  if (itr->second->GetTransportSeat() == seat)
342  return;
343 
344  // Check if it's a valid seat
345  if (!IsSeatAvailableFor(passenger, seat))
346  return;
347 
348  VehicleSeatEntry const* seatEntry = GetSeatEntry(itr->second->GetTransportSeat());
349  MANGOS_ASSERT(seatEntry);
350 
351  // Switching seats is only allowed if this flag is set
352  if (seatEntry->CanSwitchFromSeat())
353  return;
354 
355  // Remove passenger modifications of the old seat
356  RemoveSeatMods(passenger, seatEntry->m_flags);
357 
358  // Set to new seat
359  itr->second->SetTransportSeat(seat);
360 
361  Movement::MoveSplineInit init(*passenger);
362  init.MoveTo(0.0f, 0.0f, 0.0f); // ToDo: Set correct local coords
363  //if (oldorientation != neworientation) (?)
364  //init.SetFacing(0.0f); // local orientation ? ToDo: Set proper orientation!
365  // It seems that Seat switching is sent without SplineFlag BoardVehicle
366  init.Launch();
367 
368  // Get seatEntry of new seat
369  seatEntry = GetSeatEntry(seat);
370  MANGOS_ASSERT(seatEntry);
371 
372  // Apply passenger modifications of the new seat
373  ApplySeatMods(passenger, seatEntry->m_flags);
374 }

384 {

386 
387  PassengerMap::const_iterator itr = m_passengers.find(passenger);
388  if (itr == m_passengers.end())
389  return;
390

392 
393  VehicleSeatEntry const* seatEntry = GetSeatEntry(itr->second->GetTransportSeat());
394  MANGOS_ASSERT(seatEntry);
395 
396  UnBoardPassenger(passenger); // Use TransportBase to remove the passenger from storage list
397 
398  // Remove passenger modifications
399  RemoveSeatMods(passenger, seatEntry->m_flags);
400 
401  if (!changeVehicle) // Send expected unboarding packages
402  {
403  if (GenericTransport* transport = m_owner->GetTransport())
404  transport->AddPassenger(passenger);
405  else
406  {
407  // Update movementInfo
408  passenger->m_movementInfo.RemoveMovementFlag(MOVEFLAG_ONTRANSPORT);
409  passenger->m_movementInfo.ClearTransportData();
410  }
411 
412  if (passenger->GetTypeId() == TYPEID_PLAYER)
413  {
414  Player* pPlayer = (Player*)passenger;
415  pPlayer->ResummonPetTemporaryUnSummonedIfAny();
416  pPlayer->SetFallInformation(0, pPlayer->GetPositionZ());
417 
418  // SMSG_PET_DISMISS_SOUND (?)
419  }
420 
421  if (passenger->hasUnitState(UNIT_STAT_ROOT) && !passenger->HasAuraType(SPELL_AURA_MOD_ROOT))
422  passenger->SetImmobilizedState(false);
423 
424  Movement::MoveSplineInit init(*passenger);
425 
426  Position exitPos = m_owner->GetPosition(m_owner->GetTransport());
427  exitPos.o = passenger->GetOrientation();
428 
429  if (VehicleSeatParameters const* params = sObjectMgr.GetVehicleSeatParameters(seatEntry->m_ID))

439  exitPos.y = params->exitParamY;
440  exitPos.z = params->exitParamZ;

442  }
443  }
444 
445  init.MoveTo(exitPos.x, exitPos.y, exitPos.z, false, true);
446  init.SetFacing(exitPos.o);
447  init.SetExitVehicle();
448  init.Launch();
449 
450  // Despawn if passenger was accessory
451  if (passenger->GetTypeId() == TYPEID_UNIT && m_accessoryGuids.find(passenger->GetObjectGuid()) != m_accessoryGuids.end())
452  {
453  Creature* cPassenger = static_cast<Creature*>(passenger);
454  m_accessoryGuids.erase(passenger->GetObjectGuid());
455 
456  // Note: Most of the creature accessories have to stay on the map if unboarded; despawn events are handled by Creauture_Linking_Template if needed
457  }
458  }
459 
460  // Some creature vehicles get despawned after passenger unboarding
461  // Condition only applies for player passengers; creature passengers are controlled by script
462  if (m_owner->IsUnit() && passenger->IsPlayer())
463  {
464  // only for flyable vehicles
465  if (passenger->IsFlying())
466  static_cast<Unit*>(m_owner)->CastSpell(passenger, SPELL_VEHICLE_PARACHUTE, TRIGGERED_OLD_TRIGGERED);
467 
468  // TODO: Guesswork, but seems to be fairly near correct
469  // Only if the passenger was on control seat? Also depending on some flags
470  if ((seatEntry->m_flags & SEAT_FLAG_CAN_CONTROL) &&
471  !(m_vehicleEntry->m_flags & (VEHICLE_FLAG_UNK4 | VEHICLE_FLAG_NOT_DISMISSED)))
472  {
473  if (((Creature*)m_owner)->IsTemporarySummon())
474  ((Creature*)m_owner)->ForcedDespawn(1000);
475  }
476  }

484 bool VehicleInfo::CanBoard(Unit* passenger) const
485 {
486  if (!passenger)
487  return false;
488 
489  // Passenger is this vehicle
490  if (passenger == m_owner)
491  return false;
492 
493  // Passenger is already on this vehicle (in this case switching seats is required)
494  if (passenger->IsBoarded() && passenger->GetTransportInfo()->GetTransport() == m_owner)
495  return false;
496 
497  // Prevent circular boarding: passenger (could only be vehicle) must not have m_owner on board
498  if (passenger->IsVehicle() && passenger->GetVehicleInfo()->HasOnBoard(m_owner))
499  return false;
500 
501  // Check if we have at least one empty seat
502  if (!GetEmptySeats())
503  return false;
504 
505  // Passenger is already boarded
506  if (m_passengers.find(passenger) != m_passengers.end())
507  return false;
508 
509  // Check for empty player seats
510  if (passenger->GetTypeId() == TYPEID_PLAYER)
511  return (GetEmptySeatsMask() & m_playerSeats) != 0;
512 
513  // Check for empty creature seats
514  return (GetEmptySeatsMask() & m_creatureSeats) != 0;
515 }

587 bool VehicleInfo::GetUsableSeatFor(Unit* passenger, uint8& seat, bool reset, bool next) const
588 {
589  MANGOS_ASSERT(passenger);
590 
591  uint8 possibleSeats = (passenger->GetTypeId() == TYPEID_PLAYER) ? (GetEmptySeatsMask() & m_playerSeats) : (GetEmptySeatsMask() & m_creatureSeats);
592 
593  // No usable seats available
594  if (!possibleSeats)
595  return false;
596 
597  // Start with 0
598  if (reset)
599  seat = 0;
600 
601  if (next)
602  {
603  for (uint32 i = 0; i < MAX_VEHICLE_SEAT; ++i, seat = (seat + 1) % MAX_VEHICLE_SEAT)
604  if (possibleSeats & (1 << seat))
605  return true;
606  }
607  else
608  {
609  for (uint32 i = 0; i < MAX_VEHICLE_SEAT; ++i, seat = (seat + MAX_VEHICLE_SEAT - 1) % MAX_VEHICLE_SEAT)
610  if (possibleSeats & (1 << seat))
611  return true;
612  }
613 
614  return false;
615 }
