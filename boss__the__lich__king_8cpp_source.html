389  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
390  return;
391 
392  // Berserk
393  if (m_uiBerserkTimer)
394  {
395  if (m_uiBerserkTimer < uiDiff)
396  {
397  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
398  {
399  DoScriptText(SAY_ENRAGE, m_creature);
400  m_uiBerserkTimer = 0;
401  }
402  }
403  else
404  m_uiBerserkTimer -= uiDiff;
405  }
406  }
407 
408  switch (m_uiPhase)
409  {

412  // wait until set in combat
