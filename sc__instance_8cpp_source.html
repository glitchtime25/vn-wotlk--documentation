366 void DialogueHelper::DoNextDialogueStep()
367 {
368  // Last Dialogue Entry done?
369  if ((m_currentEntry && !m_currentEntry->textEntry) || (m_currentEntryTwoSide && !m_currentEntryTwoSide->textEntry))
370  {
371  m_timer = 0;
372  return;
373  }
374 
375  // Get Text, SpeakerEntry and Timer
376  int32 iTextEntry = 0;
377  uint32 uiSpeakerEntry = 0;
378 
379  if (m_dialogueArray) // One Side
380  {
381  uiSpeakerEntry = m_currentEntry->sayerEntry;
382  iTextEntry = m_currentEntry->textEntry;
383 
384  m_timer = m_currentEntry->timer;
385  }
386  else // Two Sides
387  {
388  // Second Entries can be 0, if they are the entry from first side will be taken
389  uiSpeakerEntry = !m_isFirstSide && m_currentEntryTwoSide->sayerEntryAlt ? m_currentEntryTwoSide->sayerEntryAlt : m_currentEntryTwoSide->sayerEntry;
390  iTextEntry = !m_isFirstSide && m_currentEntryTwoSide->textEntryAlt ? m_currentEntryTwoSide->textEntryAlt : m_currentEntryTwoSide->textEntry;
391 
392  m_timer = m_currentEntryTwoSide->timer;
393  }
394 
395  // Simulate Case
396  if (uiSpeakerEntry && iTextEntry < 0)
397  {
398  // Use Speaker if directly provided
399  Creature* pSpeaker = GetSpeakerByEntry(uiSpeakerEntry);
400  if (m_instance && !pSpeaker) // Get Speaker from instance
401  {
402  if (m_canSimulate) // Simulate case
403  m_instance->DoOrSimulateScriptTextForThisInstance(iTextEntry, uiSpeakerEntry);
404  else
405  pSpeaker = m_instance->GetSingleCreatureFromStorage(uiSpeakerEntry);
406  }
407 
408  if (pSpeaker)
409  {
410  // Use target if provided
411  if (Unit* pTarget = GetDialogueTarget())
412  DoScriptText(iTextEntry, pSpeaker, pTarget);
413  else
414  DoScriptText(iTextEntry, pSpeaker);
415  }
416  }
417 
418  JustDidDialogueStep(m_dialogueArray ? m_currentEntry->textEntry : m_currentEntryTwoSide->textEntry);
419 
420  // Increment position
421  if (m_dialogueArray)
422  ++m_currentEntry;
423  else
424  ++m_currentEntryTwoSide;
425 }
