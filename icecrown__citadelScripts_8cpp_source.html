105 bool GossipSelect_go_icc_teleporter(Player* pPlayer, GameObject* pGo, uint32 uiSender, uint32 uiAction)
106 {
107  instance_icecrown_citadel* pInstance = (instance_icecrown_citadel*)pPlayer->GetInstanceData();

109  return true;
110 
111  // There needs to be displayed a msg when in Combat, it is likely that this is to be handled by core and spell can-cast check

113  if (pPlayer->IsInCombat())
114  return true;
115

118  // Lights Hammer
119  case GOSSIP_ACTION_INFO_DEF:
120  pGo->CastSpell(pPlayer, pPlayer, SPELL_TELE_LIGHTS_HAMMER, TRIGGERED_OLD_TRIGGERED);

130  // Deathbringer's Rise

132  pGo->CastSpell(pPlayer, pPlayer, SPELL_TELE_DEATHBRINGERS_RISE, TRIGGERED_OLD_TRIGGERED);
133  break;
134  // Upper Spire
135  case GOSSIP_ACTION_INFO_DEF + 4:
136  pGo->CastSpell(pPlayer, pPlayer, SPELL_TELE_UPPER_SPIRE, TRIGGERED_OLD_TRIGGERED);
137  break;
138  // Sindragosa's Lair

140  pGo->CastSpell(pPlayer, pPlayer, SPELL_TELE_SINDRAGOSAS_LAIR, TRIGGERED_OLD_TRIGGERED);

190 bool AreaTrigger_at_lights_hammer(Player* pPlayer, AreaTriggerEntry const* pAt)
191 {
192  if (pPlayer->IsGameMaster() || pPlayer->IsDead())
193  return false;
194 
195  // search for the first set of Nerubar Broodkeepers and lower them to the ground
196  if (pAt->id == AT_LIGHTS_HAMMER_INTRO_1)
197  {
198  CreatureList lKeepersInRange;
199  GetCreatureListWithEntryInGrid(lKeepersInRange, pPlayer, NPC_NERUBAR_BROODKEEPER, 150.0f);
200 
201  for (const auto& creature : lKeepersInRange)
202  {
203  if (creature->GetPositionZ() >= 75.0f && creature->GetPositionZ() < 82.0f)
204  {
205  creature->CastSpell(creature, SPELL_WEB_BEAM, TRIGGERED_OLD_TRIGGERED);
206  creature->SetWalk(false);
207  creature->GetMotionMaster()->MoveWaypoint();
208  creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
209  }
210  }
211  }
212  // search for the second set of Nerubar Broodkeepers and lower them to the ground
213  else if (pAt->id == AT_LIGHTS_HAMMER_INTRO_2)
214  {
215  CreatureList lKeepersInRange;
216  GetCreatureListWithEntryInGrid(lKeepersInRange, pPlayer, NPC_NERUBAR_BROODKEEPER, 150.0f);
217 
218  for (const auto& creature : lKeepersInRange)
219  {
220  if (creature->GetPositionZ() >= 85.0f)
221  {
222  creature->CastSpell(creature, SPELL_WEB_BEAM, TRIGGERED_OLD_TRIGGERED);

394  }
395  m_uiSummonTimer = urand(2000, 5000);
396  }
397  else
398  m_uiSummonTimer -= uiDiff;
399  }
400 
401  // event can last max 5 min
402  if (m_uiEventTimer)
403  {
404  if (m_uiEventTimer <= uiDiff)
405  {
406  bool bEventFailed = true;
407 
408  // check withing all players in map if any are still alive and in LoS
409  Map::PlayerList const& pAllPlayers = m_pInstance->instance->GetPlayers();

411  if (!pAllPlayers.isEmpty())
