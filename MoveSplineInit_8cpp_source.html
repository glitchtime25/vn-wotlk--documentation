31  int32 MoveSplineInit::Launch()
32  {
33  // show path in the client if need
34  if (unit.HaveDebugFlag(CMDEBUGFLAG_WP_PATH))
35  {
36  uint32 counter = 0;
37  for (auto pt : args.path)
38  {
39  TempSpawnSettings settings;
40  settings.spawner = &unit;
41  settings.entry = VISUAL_WAYPOINT;
42  settings.x = pt.x; settings.y = pt.y; settings.z = pt.z; settings.ori = 0.0f;
43  settings.activeObject = true;
44  settings.despawnTime = 30 * IN_MILLISECONDS;
45  settings.spawnType = TEMPSPAWN_TIMED_DESPAWN;
46  settings.spawnDataEntry = 2;
47 
48  settings.tempSpawnMovegen = true;
49  settings.waypointId = counter++;
50 
51  WorldObject::SummonCreature(settings, unit.GetMap(), unit.GetPhaseMask());
52  }
53  }
54  MoveSpline& move_spline = *unit.movespline;
55  TransportInfo* transportInfo = unit.GetTransportInfo();
56  // TODO: merge these two together
57  GenericTransport* transport = unit.GetTransport();
58 
59  Location real_position(unit.GetPositionX(), unit.GetPositionY(), unit.GetPositionZ(), unit.GetOrientation());
60 
61  // If boarded use current local position
62  if (transportInfo)
63  transportInfo->GetLocalPosition(real_position.x, real_position.y, real_position.z, real_position.orientation);
64  if (transport)
65  transport->CalculatePassengerOffset(real_position.x, real_position.y, real_position.z, &real_position.orientation);
66 
67  // there is a big chance that current position is unknown if current state is not finalized, need compute it
68  // this also allows calculate spline position and update map position in much greater intervals
69  if (!move_spline.Finalized() && !transportInfo)
70  real_position = move_spline.ComputePosition();
71 
72  bool pathEmpty = false;
73  if (args.path.empty())
74  {
75  // should i do the things that user should do?
76  MoveTo(real_position);
77  pathEmpty = true;
78  }
79 
80  // corrent first vertex
81  args.path[0] = real_position;
82  args.initialOrientation = real_position.orientation;
83 
84  args.flags.enter_cycle = args.flags.cyclic;
85  uint32 moveFlags = unit.m_movementInfo.GetMovementFlags();
86  if (args.flags.walkmode)
87  moveFlags |= MOVEFLAG_WALK_MODE;
88  else
89  moveFlags &= ~MOVEFLAG_WALK_MODE;
90 
91  if (!pathEmpty) // should not broadcast these flags on orientation change only
92  moveFlags |= (MOVEFLAG_SPLINE_ENABLED | MOVEFLAG_FORWARD);
93 
94  if (args.velocity == 0.f) // ignore swim speed and flight speed because its not used in generic scripting - always possible to override
95  {
96  args.velocity = unit.GetSpeed(MovementInfo::GetSpeedType(MovementFlags(moveFlags & ~(MOVEFLAG_FLYING | MOVEFLAG_SWIMMING))));
97  if (args.slowed != 0.f) // when set always > 0.5
98  args.velocity *= args.slowed;
99  }
100 
101  if (!args.Validate(&unit))
102  return 0;
103 
104  bool removedRoot = false;
105  if (args.flags.hasFlag(MoveSplineFlag::eFlags::BoardVehicle | MoveSplineFlag::eFlags::ExitVehicle))
106  {
107  if (moveFlags & MOVEFLAG_ROOT)
108  removedRoot = true;
109  moveFlags &= ~MOVEFLAG_ROOT;
110  }
111 
112  if (moveFlags & MOVEFLAG_ROOT && !pathEmpty)
113  {
114  sLog.outCustomLog("Invalid movement during root. Entry: %u IsImmobilized %s, moveflags %u", unit.GetEntry(), unit.IsImmobilizedState() ? "true" : "false", moveFlags);
115  sLog.traceLog();
116  return 0;
117  }
118 
119  args.splineId = splineCounter++;
120 
121  unit.m_movementInfo.SetMovementFlags(MovementFlags(moveFlags));
122 
123  move_spline.Initialize(args);
124 
125  WorldPacket data(SMSG_MONSTER_MOVE, 64);
126  data << unit.GetPackGUID();
127 
128  if (transportInfo || transport)
129  {
130  data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);
131  if (transportInfo)
132  {
133  data << transportInfo->GetTransportGuid().WriteAsPacked();
134  data << int8(transportInfo->GetTransportSeat());
135  }
136  else if (transport)
137  {
138  data << transport->GetPackGUID();
139  data << int8(-1);
140  }
141  }
142 
143  PacketBuilder::WriteMonsterMove(move_spline, data);
144  unit.SendMessageToAllWhoSeeMe(data, true);
145 
146  if (removedRoot)
147  unit.m_movementInfo.AddMovementFlag(MOVEFLAG_ROOT);
148 
149  return move_spline.Duration();
150  }

153  {

173  // this also allows calculate spline position and update map position in much greater intervals
174  if (!move_spline.Finalized())

190  move_spline.Initialize(args);
191 
192  WorldPacket data(SMSG_MONSTER_MOVE, 64);
193  data << unit.GetPackGUID();
194

197  data.SetOpcode(SMSG_MONSTER_MOVE_TRANSPORT);

201  data << int8(transportInfo->GetTransportSeat());
202  }

205  data << transport->GetPackGUID();
206  data << int8(-1);
207  }
208  }
209
