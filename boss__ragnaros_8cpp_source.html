213  void HandlePhaseTransition()
214  {
215  switch (m_phase)
216  {
217  case PHASE_EMERGED:
218  {
219  // Submerge and attack again after 90 secs
220  DoCastSpellIfCan(m_creature, SPELL_RAGNA_SUBMERGE_VISUAL, CAST_TRIGGERED);
221  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
222 
223  // Say dependend if first time or not
224  DoScriptText(!m_bHasSubmergedOnce ? SAY_REINFORCEMENTS_1 : SAY_REINFORCEMENTS_2, m_creature);
225  m_bHasSubmergedOnce = true;
226 
227  // Summon 8 elementals around the boss
228  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_SONS_FLAME) == CAST_OK)
229  m_uiAddCount = NB_ADDS_IN_SUBMERGE;
230 
231  ResetCombatAction(RAGNAROS_PHASE_TRANSITION, 1000);
232  m_phase = PHASE_SUBMERGING;
233  DisableCombatAction(RAGNAROS_WRATH_OF_RAGNAROS);
234  DisableCombatAction(RAGNAROS_MIGHT_OF_RAGNAROS);
235  DisableCombatAction(RAGNAROS_MAGMA_BLAST);
236  DisableCombatAction(RAGNAROS_LAVA_BURST);
237  SetMeleeEnabled(false);
238  break;
239  }
240  case PHASE_SUBMERGING:
241  {
242  m_creature->HandleEmote(EMOTE_STATE_SUBMERGED);
243  DoCastSpellIfCan(nullptr, SPELL_RAGNA_SUBMERGE, CAST_TRIGGERED);
244  DoCastSpellIfCan(nullptr, SPELL_SUBMERGE_EFFECT, CAST_TRIGGERED);
245  ResetCombatAction(RAGNAROS_PHASE_TRANSITION, 90 * IN_MILLISECONDS);
246  m_phase = PHASE_SUBMERGED;
247  break;
248  }
249  case PHASE_SUBMERGED:
250  {
251  // Become emerged again
252  m_creature->HandleEmote(EMOTE_STATE_STAND);
253  m_creature->RemoveAurasDueToSpell(SPELL_RAGNA_SUBMERGE);
254  m_creature->RemoveAurasDueToSpell(SPELL_SUBMERGE_EFFECT);
255  m_creature->RemoveAurasDueToSpell(SPELL_RAGNA_SUBMERGE_VISUAL);
256  ResetCombatAction(RAGNAROS_PHASE_TRANSITION, 500);
257  m_phase = PHASE_EMERGING;
258  break;
259  }
260  case PHASE_EMERGING:
261  {
262  DoCastSpellIfCan(m_creature, SPELL_RAGNA_EMERGE);
263  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
264  m_phase = PHASE_EMERGED;
265  ResetCombatAction(RAGNAROS_PHASE_TRANSITION, 3 * MINUTE * IN_MILLISECONDS);
266  ResetCombatAction(RAGNAROS_WRATH_OF_RAGNAROS, 20000);
267  ResetCombatAction(RAGNAROS_MIGHT_OF_RAGNAROS, urand(11000, 30000));
268  ResetCombatAction(RAGNAROS_MAGMA_BLAST, 3000);
269  ResetCombatAction(RAGNAROS_LAVA_BURST, urand(5000, 25000));
270  SetMeleeEnabled(true);
271  break;
272  }
273  }
274  }

276  void ExecuteAction(uint32 action) override
277  {
278  switch (action)
279  {
280  case RAGNAROS_PHASE_TRANSITION:
281  {
282  HandlePhaseTransition();
283  break;
284  }
285  case RAGNAROS_WRATH_OF_RAGNAROS:
286  {
287  if (DoCastSpellIfCan(nullptr, SPELL_WRATH_OF_RAGNAROS) == CAST_OK)
288  {
289  DoScriptText(SAY_WRATH, m_creature);
290  ResetCombatAction(action, 25000);
291  }
292  break;
293  }
294  case RAGNAROS_MIGHT_OF_RAGNAROS:
295  {
296  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MIGHT_OF_RAGNAROS, SELECT_FLAG_POWER_MANA))
297  {
298  if (DoCastSpellIfCan(target, SPELL_MIGHT_OF_RAGNAROS) == CAST_OK)
299  {
300  DoScriptText(SAY_HAMMER, m_creature);
301  ResetCombatAction(action, urand(11000, 30000));
302  }
303  }
304  break;
305  }
306  case RAGNAROS_MAGMA_BLAST:
307  {
308  uint32 timer = 500;
309  // If victim exists we have a target in melee range
310  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
311  m_rangeCheckState = -1;
312  // Spam Waterbolt spell when not tanked
313  else
314  {
315  ++m_rangeCheckState;
316  if (m_rangeCheckState > 1)
317  {
318  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MAGMA_BLAST, SELECT_FLAG_PLAYER))
319  {
320  if (DoCastSpellIfCan(target, SPELL_MAGMA_BLAST) == CAST_OK)
321  {
322  //if (!m_bHasYelledMagmaBurst)
323  //{
324  // DoScriptText(SAY_MAGMABURST, m_creature);
325  // m_bHasYelledMagmaBurst = true;
326  //}
327  timer = 2500;
328  }
329  }
330  }
331  }
332  ResetCombatAction(action, timer);
333  break;
334  }
335  case RAGNAROS_LAVA_BURST:
336  {
337  if (DoCastSpellIfCan(nullptr, SPELL_LAVA_BURST) == CAST_OK)
338  ResetCombatAction(action, urand(5000, 25000)); // Upper value is guess work based on videos. It could be up to 45 secs
339  break;
340  }
341  }
342  }
