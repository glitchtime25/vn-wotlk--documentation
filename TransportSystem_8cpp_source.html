53 // Update every now and then (after some change of transporter's position)
54 // This is used to calculate global positions (which don't have to be exact, they are only required for some server-side calculations
55 void TransportBase::Update(uint32 diff)
56 {
57  if (m_updatePositionsTimer < diff)
58  {
59  if (fabs(m_owner->GetPositionX() - m_lastPosition.x) +
60  fabs(m_owner->GetPositionY() - m_lastPosition.y) +
61  fabs(m_owner->GetPositionZ() - m_lastPosition.z) > 1.0f ||
62  MapManager::NormalizeOrientation(m_owner->GetOrientation() - m_lastPosition.o) > 0.01f)
63  UpdateGlobalPositions();
64 
65  m_updatePositionsTimer = 500;
66  }
67  else
68  m_updatePositionsTimer -= diff;
69 }

71 // Update the global positions of all passengers
72 void TransportBase::UpdateGlobalPositions()
73 {
74  Position pos(m_owner->GetPositionX(), m_owner->GetPositionY(),
75  m_owner->GetPositionZ(), m_owner->GetOrientation());
76 
77  // Calculate new direction multipliers
78  if (MapManager::NormalizeOrientation(pos.o - m_lastPosition.o) > 0.01f)
79  {
80  m_sinO = sin(pos.o);
81  m_cosO = cos(pos.o);
82  }
83 
84  // Update global positions
85  for (PassengerMap::const_iterator itr = m_passengers.begin(); itr != m_passengers.end(); ++itr)
86  UpdateGlobalPositionOf(itr->first, itr->second->GetLocalPositionX(), itr->second->GetLocalPositionY(),
87  itr->second->GetLocalPositionZ(), itr->second->GetLocalOrientation());
88 
89  m_lastPosition = pos;
90 }

139 // Helper function to check if a unit is boarded onto this transporter (or a transporter boarded onto this) recursively
140 bool TransportBase::HasOnBoard(WorldObject const* passenger) const
141 {
142  MANGOS_ASSERT(passenger);
143 
144  // For efficiency we go down from the (possible) passenger until we reached our owner, or until we reached no passenger
145  // Note, this will not catch, if self and passenger are boarded onto the same transporter (as it should not)
146  while (passenger->IsBoarded())
147  {
148  // pasenger is boarded onto this
149  if (passenger->GetTransportInfo()->GetTransport() == m_owner)
150  return true;
151  passenger = passenger->GetTransportInfo()->GetTransport();
152  }
153 
154  return false;
155 }

168 void TransportBase::UnBoardPassenger(WorldObject* passenger)
169 {
170  PassengerMap::iterator itr = m_passengers.find(passenger);
171 
172  if (itr == m_passengers.end())
173  return;
174 
175  // Set passengers transportInfo to nullptr
176  passenger->SetTransportInfo(nullptr);
177 
178  // Delete transportInfo
179  delete itr->second;
180 
181  // Unboard finally
182  m_passengers.erase(itr);
183 }
