185  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);

196  void JustSummoned(Creature* summoned) override
197  {
198  if (summoned->GetEntry() == NPC_ELEMENTAL)
199  {
200  // If this is the second summon, then set the health to half
201  if (!m_firstEmerge)
202  {
203  if (boss_drakkari_elementalAI* elementalAI = dynamic_cast<boss_drakkari_elementalAI*>(summoned->AI()))
204  elementalAI->DoPrepareSecondEmerge();
205  }
206 
207  m_firstEmerge = false;
208  if (m_creature->GetVictim())
209  summoned->AI()->AttackStart(m_creature->GetVictim());
210  }
211  }

213  void DoEmergeElemental()
214  {
215  // Avoid casting the merge spell twice
216  if (m_creature->HasAura(SPELL_FREEZE_ANIM))
217  return;
218 
219  if (DoCastSpellIfCan(m_creature, SPELL_EMERGE, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
220  {
221  SetCombatMovement(false);
222  m_creature->GetMotionMaster()->MoveIdle();
223  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
224  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
225  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_STUNNED);
226  DoCastSpellIfCan(nullptr, SPELL_FREEZE_ANIM, CAST_TRIGGERED);
227  }
228  }

236  ResetTimer(COLOSSUS_START_COMBAT, 1s);

239  void JustPreventedDeath(Unit* /*attacker*/) override

324  {
325  CombatAI::EnterEvadeMode();
326  return;

328
