40 void HomeMovementGenerator<Creature>::_setTargetLocation(Creature& owner)
41 {
42  if (owner.hasUnitState(UNIT_STAT_NOT_MOVE))
43  return;
44 
45  Position pos;
46  // at apply we can select more nice return points base at current movegen
47  if (owner.GetMotionMaster()->empty() || !owner.GetMotionMaster()->top()->GetResetPosition(owner, pos.x, pos.y, pos.z, pos.o))
48  owner.GetCombatStartPosition(pos);
49 
50  if (pos.IsEmpty())
51  owner.GetRespawnCoord(pos.x, pos.y, pos.z, &pos.o);
52  if (owner.GetDistance(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ(), DIST_CALC_NONE, owner.GetTransport()) > 150.f * 150.f)
53  {
54  if (!owner.IsInWorld() || !owner.GetMap()->IsDungeon())
55  {
56  arrived = true;
57  Finalize(owner);
58  owner.SetRespawnDelay(5, true);
59  owner.ForcedDespawn(1);
60  return;
61  }
62  }
63 
64  PathFinder path(&owner);
65 
66  Position curPos = owner.GetPosition(owner.GetTransport());
67  // source and target pos must be local coords
68  path.calculate(G3D::Vector3(curPos.GetPositionX(), curPos.GetPositionY(), curPos.GetPositionZ()), G3D::Vector3(pos.GetPositionX(), pos.GetPositionY(), pos.GetPositionZ()), true);
69 
70  Movement::MoveSplineInit init(owner);
71  init.MovebyPath(path.getPath());
72  init.SetWalk(!runHome);
73  init.SetFacing(pos.GetPositionO());
74  if (path.getPathType() & (PATHFIND_NOPATH | PATHFIND_SHORTCUT))
75  init.SetVelocity(400.f);

77 
78  arrived = false;
