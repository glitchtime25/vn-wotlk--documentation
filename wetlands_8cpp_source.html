134  void DamageTaken(Unit* /*dealer*/, uint32& damage, DamageEffectType /*damagetype*/, SpellEntry const* /*spellInfo*/) override
135  {
136  if (!HasEscortState(STATE_ESCORT_ESCORTING))
137  return;
138 
139  if (m_creature->GetHealthPercent() < 20.0f || damage > m_creature->GetHealth())
140  {
141  if (Pet* pFriend = m_creature->FindGuardianWithEntry(NPC_SLIMS_FRIEND))
142  {
143  DoScriptText(SAY_FRIEND_DEFEAT, pFriend);
144  pFriend->ForcedDespawn(1000);
145  }
146 
147  // set escort on pause and evade
148  damage = std::min(damage, m_creature->GetHealth() - 1);
149  m_bEventComplete = true;
150 
151  SetEscortPaused(true);
152  EnterEvadeMode();
153  }
154  }

217  return true;
218  }
