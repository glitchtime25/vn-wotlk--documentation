146  void JustSummoned(Creature* summoned) override
147  {

150  if (summoned->GetEntry() == NPC_FRENZIED_BAT)

182  // Wrapper to despawn the bomb riders on evade / death
183  void DoDespawnBombRiders()
184  {
185  if (m_lBombRiderGuidsList.empty())
186  return;
187 
188  for (GuidList::const_iterator itr = m_lBombRiderGuidsList.begin(); itr != m_lBombRiderGuidsList.end(); ++itr)
189  {
190  if (Creature* pRider = m_creature->GetMap()->GetCreature(*itr))
191  pRider->ForcedDespawn();

195  void ExecuteAction(uint32 action) override
196  {
197  switch (action)
198  {
199  case JEKLIK_PHASE_2:
200  // Phase Switch at 50%
201  if (m_creature->GetHealthPercent() < 50.0f)
202  {
203  if (DoCastSpellIfCan(nullptr, SPELL_TRANSFORM_VISUAL) == CAST_OK)
204  {
205  m_creature->RemoveAurasDueToSpell(SPELL_BAT_FORM);
206  m_creature->SetLevitate(false);
207  DoResetThreat();
208  DisableCombatAction(action);
209  m_creature->SetSpellList(SPELL_LIST_PHASE_2);
210  }
211  }
212  break;
213  case JEKLIK_PHASE_BATS:
214  if (m_creature->GetHealthPercent() < 35.0f)
215  {
216  // TODO: check if more positions exist
217  const std::vector<Position> positions =
218  {
219  {-12298.03f, -1368.506f, 145.3976f, 4.799655f}, // waits 15 sec
220  {-12301.69f, -1371.292f, 145.0924f, 4.747295f},
221  };
222 
223  Position const& pos1 = positions[0];
224  Creature* bat = m_creature->SummonCreature(NPC_BAT_RIDER, pos1.x, pos1.y, pos1.z, 0, TEMPSPAWN_DEAD_DESPAWN, 0);
225  bat->GetMotionMaster()->MoveIdle();
226  m_delayedBat = bat->GetObjectGuid();
227  ResetTimer(JEKLIK_DELAYED_BAT, 15000);
228 
229  Position const& pos2 = positions[1];
230  bat = m_creature->SummonCreature(NPC_BAT_RIDER, pos2.x, pos2.y, pos2.z, 0, TEMPSPAWN_DEAD_DESPAWN, 0);
231  bat->CastSpell(nullptr, SPELL_LIQUID_FIRE, TRIGGERED_OLD_TRIGGERED);
232  bat->GetMotionMaster()->MovePath(1);
233 
234  DoScriptText(SAY_RAIN_FIRE, m_creature);
235  DisableCombatAction(action);
236  }
237  break;
238  }
239  }
