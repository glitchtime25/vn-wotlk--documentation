300 bool AchievementCriteriaRequirement::Meets(uint32 criteria_id, Player const* source, Unit const* target, uint32 miscvalue1 /*= 0*/) const
301 {
302  switch (requirementType)
303  {
304  case ACHIEVEMENT_CRITERIA_REQUIRE_NONE:
305  return true;
306  case ACHIEVEMENT_CRITERIA_REQUIRE_T_CREATURE:
307  if (!target || target->GetTypeId() != TYPEID_UNIT)
308  return false;
309  return target->GetEntry() == creature.id;
310  case ACHIEVEMENT_CRITERIA_REQUIRE_T_PLAYER_CLASS_RACE:
311  if (!target || target->GetTypeId() != TYPEID_PLAYER)
312  return false;
313  if (classRace.class_id && classRace.class_id != ((Player*)target)->getClass())
314  return false;
315  if (classRace.race_id && classRace.race_id != ((Player*)target)->getRace())
316  return false;
317  return true;
318  case ACHIEVEMENT_CRITERIA_REQUIRE_T_PLAYER_LESS_HEALTH:
319  if (!target || target->GetTypeId() != TYPEID_PLAYER)
320  return false;
321  return target->GetHealth() * 100 <= health.percent * target->GetMaxHealth();
322  case ACHIEVEMENT_CRITERIA_REQUIRE_T_PLAYER_DEAD:
323  if (!target || target->GetTypeId() != TYPEID_PLAYER || target->IsAlive() || ((Player*)target)->GetDeathTimer() == 0)
324  return false;
325  // flag set == must be same team, not set == different team
326  return (((Player*)target)->GetTeam() == source->GetTeam()) == (player_dead.own_team_flag != 0);
327  case ACHIEVEMENT_CRITERIA_REQUIRE_S_AURA:
328  return source->HasAura(aura.spell_id, SpellEffectIndex(aura.effect_idx));
329  case ACHIEVEMENT_CRITERIA_REQUIRE_S_AREA:
330  {
331  uint32 zone_id, area_id;
332  source->GetZoneAndAreaId(zone_id, area_id);
333  return area.id == zone_id || area.id == area_id;
334  }
335  case ACHIEVEMENT_CRITERIA_REQUIRE_T_AURA:
336  return target && target->HasAura(aura.spell_id, SpellEffectIndex(aura.effect_idx));
337  case ACHIEVEMENT_CRITERIA_REQUIRE_VALUE:
338  return miscvalue1 >= value.minvalue;
339  case ACHIEVEMENT_CRITERIA_REQUIRE_T_LEVEL:
340  if (!target)
341  return false;
342  return target->GetLevel() >= level.minlevel;
343  case ACHIEVEMENT_CRITERIA_REQUIRE_T_GENDER:
344  if (!target)
345  return false;
346  return target->getGender() == gender.gender;
347  case ACHIEVEMENT_CRITERIA_REQUIRE_DISABLED:
348  return false; // always fail
349  case ACHIEVEMENT_CRITERIA_REQUIRE_MAP_DIFFICULTY:
350  return source->GetMap()->GetSpawnMode() == difficulty.difficulty;
351  case ACHIEVEMENT_CRITERIA_REQUIRE_MAP_PLAYER_COUNT:
352  return source->GetMap()->GetPlayersCountExceptGMs() <= map_players.maxcount;
353  case ACHIEVEMENT_CRITERIA_REQUIRE_T_TEAM:
354  if (!target || target->GetTypeId() != TYPEID_PLAYER)
355  return false;
356  return ((Player*)target)->GetTeam() == Team(team.team);
357  case ACHIEVEMENT_CRITERIA_REQUIRE_S_DRUNK:
358  return (uint32)Player::GetDrunkenstateByValue(source->GetDrunkValue()) >= drunk.state;
359  case ACHIEVEMENT_CRITERIA_REQUIRE_HOLIDAY:
360  return sGameEventMgr.IsActiveHoliday(HolidayIds(holiday.id));
361  case ACHIEVEMENT_CRITERIA_REQUIRE_INSTANCE_SCRIPT:
362  {
363  if (!source->IsInWorld())
364  return false;
365  InstanceData* data = source->GetInstanceData();
366  if (!data)
367  {
368  sLog.outErrorDb("Achievement system call ACHIEVEMENT_CRITERIA_REQUIRE_INSTANCE_SCRIPT (%u) for achievement criteria %u for map %u but map not have instance script",
369  ACHIEVEMENT_CRITERIA_REQUIRE_INSTANCE_SCRIPT, criteria_id, source->GetMapId());
370  return false;
371  }
372  return data->CheckAchievementCriteriaMeet(criteria_id, source, target, miscvalue1);
373  }
374  case ACHIEVEMENT_CRITERIA_REQUIRE_S_EQUIPPED_ITEM_LVL:
375  {
376  Item* item = source->GetItemByPos(INVENTORY_SLOT_BAG_0, miscvalue1);
377  if (!item)
378  return false;
379  ItemPrototype const* proto = item->GetProto();
380  return proto->ItemLevel >= equipped_item.item_level && proto->Quality >= equipped_item.item_quality;
381  }
382  case ACHIEVEMENT_CRITERIA_REQUIRE_NTH_BIRTHDAY:
383  {
384  time_t birthday_start = time_t(sWorld.getConfig(CONFIG_UINT32_BIRTHDAY_TIME));
385 
386  tm birthday_tm = *localtime(&birthday_start);
387 
388  // exactly N birthday
389  birthday_tm.tm_year += birthday_login.nth_birthday;
390 
391  time_t birthday = mktime(&birthday_tm);
392 
393  time_t now = sWorld.GetGameTime();
394  return now <= birthday + DAY && now >= birthday;
395  }
396  case ACHIEVEMENT_CRITERIA_REQUIRE_KNOWN_TITLE:
397  {
398  if (CharTitlesEntry const* titleInfo = sCharTitlesStore.LookupEntry(known_title.title_id))
399  return source && source->HasTitle(titleInfo->bit_index);
400 
401  return false;
402  }
403  case ACHIEVEMENT_CRITERIA_REQUIRE_PVP_SCRIPT:
404  {
405  if (!source->IsInWorld())
406  return false;
407 
408  if (OutdoorPvP* outdoorPvP = sOutdoorPvPMgr.GetScript(source->GetCachedZoneId()))
409  return outdoorPvP->CheckAchievementCriteriaMeet(criteria_id, source, target, miscvalue1);
410  else if (source->InBattleGround())
411  {
412  if (BattleGround* bg = source->GetBattleGround())
413  return bg->CheckAchievementCriteriaMeet(criteria_id, source, target, miscvalue1);
414  }
415 
416  return false;
417  }
418  case ACHIEVEMENT_CRITERIA_REQUIRE_KILL_CREATURE_TYPE:
419  {
420  if (creatureType.customCond == 1) // Total Npc Kills
421  return miscvalue1 > 0;
422  else if (creatureType.customCond == 2) // Kill an NPC that yields XP
423  return target && !target->IsTrivialForTarget(source);
424 
425  return miscvalue1 == creatureType.creatureType;
426  }
427  case ACHIEVEMENT_CRITERIA_REQUIRE_MAP_ID:
428  return source->GetMapId() == mapId.mapId;
429  case ACHIEVEMENT_CRITERIA_REQUIRE_WORLDSTATE_CONDITION:
430  return IsConditionSatisfied(worldStateCondition.conditionEntry, nullptr, source->GetMap(), nullptr, CONDITION_FROM_WORLDSTATE);
431  }
432  return false;
433 }
