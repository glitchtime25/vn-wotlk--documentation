26 bool Timer::UpdateTimer(const uint32 diff, bool combat)
27 {
28  if (disabled)
29  return false;
30 
31  if (combatSetting != TIMER_ALWAYS && combatSetting != combat)
32  return false;
33 
34  if (timer <= diff)
35  {
36  timer = 0;
37  disabled = true;
38  return true;
39  }
40  else timer -= diff;
41 
42  return false;
43 }
