174 void instance_arcatraz::SetData(uint32 uiType, uint32 uiData)
175 {
176  switch (uiType)
177  {
178  case TYPE_ENTRANCE:
179  case TYPE_ZEREKETH:
180  m_auiEncounter[uiType] = uiData;
181  break;
182 
183  case TYPE_DALLIAH:
184  if (uiData == IN_PROGRESS)
185  {
186  // Soccothares taunts after Dalliah gets aggro
187  if (GetData(TYPE_SOCCOTHRATES) != DONE)
188  StartNextDialogueText(TYPE_DALLIAH);
189  }
190  if (uiData == DONE)
191  {
192  DoUseDoorOrButton(GO_CORE_SECURITY_FIELD_BETA);
193 
194  // Soccothares taunts after Dalliah dies
195  if (GetData(TYPE_SOCCOTHRATES) != DONE)
196  StartNextDialogueText(TYPE_SOCCOTHRATES);
197  }
198  m_auiEncounter[uiType] = uiData;
199  break;
200 
201  case TYPE_SOCCOTHRATES:
202  if (uiData != IN_PROGRESS)
203  {
204  GuidVector felfireVector; // at aggro felfire mobs always teleport to respawn location
205  GetCreatureGuidVectorFromStorage(NPC_WRATH_SCRYER_FELFIRE, felfireVector);
206  for (ObjectGuid& guid : felfireVector)
207  {
208  if (Creature* creature = instance->GetCreature(guid))
209  {
210  creature->RemoveAllDynObjects();
211  creature->CombatStop();
212  }
213  }
214  }
215  if (uiData == DONE)
216  DoUseDoorOrButton(GO_CORE_SECURITY_FIELD_ALPHA);
217  m_auiEncounter[uiType] = uiData;
218  break;
219 
220  case TYPE_HARBINGERSKYRISS:
221  if (uiData == FAIL)
222  {
223  SetData(TYPE_WARDEN_1, NOT_STARTED);
224  SetData(TYPE_WARDEN_2, NOT_STARTED);
225  SetData(TYPE_WARDEN_3, NOT_STARTED);
226  SetData(TYPE_WARDEN_4, NOT_STARTED);
227  SetData(TYPE_WARDEN_5, NOT_STARTED);
228 
229  // Reset event in 1 min
230  if (Creature* pMellichar = GetSingleCreatureFromStorage(NPC_MELLICHAR))
231  pMellichar->ForcedDespawn();
232  m_uiResetDelayTimer = 60000;
233 
234  // Despawn all the summons manually
235  for (GuidList::const_iterator itr = m_lSkyrissEventMobsGuidList.begin(); itr != m_lSkyrissEventMobsGuidList.end(); ++itr)
236  {
237  if (Creature* pTemp = instance->GetCreature(*itr))
238  pTemp->ForcedDespawn();
239  }
240 
241  // Reset these objects, because they doesn't reset automatically
242  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_POD_BETA))
243  pGo->ResetDoorOrButton();
244  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_POD_OMEGA))
245  pGo->ResetDoorOrButton();
246  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_SEAL_SPHERE))
247  pGo->ResetDoorOrButton();
248  }
249  if (uiData == IN_PROGRESS)
250  {
251  StartNextDialogueText(YELL_MELLICHAR_INTRO1);
252  DoUseDoorOrButton(GO_SEAL_SPHERE);
253  }
254  if (uiData == DONE)
255  {
256  if (Creature* pMillhouse = GetSingleCreatureFromStorage(NPC_MILLHOUSE))
257  {
258  DoScriptText(SAY_MILLHOUSE_COMPLETE, pMillhouse);
259  if (!instance->IsRegularDifficulty())
260  {
261  Map::PlayerList const& PlayerList = instance->GetPlayers();
262 
263  for (const auto& itr : PlayerList)
264  {
265  Player* pPlayer = itr.getSource();
266  if (pPlayer && pPlayer->GetQuestStatus(QUEST_TRIAL_OF_THE_NAARU_TENACITY) == QUEST_STATUS_INCOMPLETE)
267  pPlayer->AreaExploredOrEventHappens(QUEST_TRIAL_OF_THE_NAARU_TENACITY);
268  }
269  }
270  }
271  }
272  m_auiEncounter[3] = uiData;
273  break;
274 
275  case TYPE_WARDEN_1:
276  if (uiData == IN_PROGRESS)
277  DoUseDoorOrButton(GO_POD_ALPHA);
278  if (uiData == DONE)
279  StartNextDialogueText(YELL_MELLICHAR_RELEASE2);
280  m_auiEncounter[uiType] = uiData;
281  break;
282 
283  case TYPE_WARDEN_2:
284  if (uiData == IN_PROGRESS)
285  DoUseDoorOrButton(GO_POD_BETA);
286  if (uiData == DONE)
287  StartNextDialogueText(YELL_MELLICHAR_RELEASE3);
288  m_auiEncounter[uiType] = uiData;
289  break;
290 
291  case TYPE_WARDEN_3:
292  if (uiData == IN_PROGRESS)
293  DoUseDoorOrButton(GO_POD_DELTA);
294  if (uiData == DONE)
295  StartNextDialogueText(YELL_MELLICHAR_RELEASE4);
296  m_auiEncounter[uiType] = uiData;
297  break;
298 
299  case TYPE_WARDEN_4:
300  if (uiData == IN_PROGRESS)
301  DoUseDoorOrButton(GO_POD_GAMMA);
302  if (uiData == DONE)
303  StartNextDialogueText(YELL_MELLICHAR_RELEASE5);
304  m_auiEncounter[uiType] = uiData;
305  break;
306 
307  case TYPE_WARDEN_5:
308  if (uiData == IN_PROGRESS)
309  DoUseDoorOrButton(GO_POD_OMEGA);
310  m_auiEncounter[uiType] = uiData;
311  break;
312  }
313

317

468 void instance_arcatraz::Update(uint32 uiDiff)
469 {
470  DialogueUpdate(uiDiff);
471 
472  if (m_uiResetDelayTimer)
473  {
474  if (m_uiResetDelayTimer <= uiDiff)
475  {
476  if (Creature* pMellichar = GetSingleCreatureFromStorage(NPC_MELLICHAR))
477  pMellichar->Respawn();
478  m_uiResetDelayTimer = 0;
479  }
480  else
481  m_uiResetDelayTimer -= uiDiff;
482  }
483 
484  if (m_uiEntranceEventTimer)
485  {
486  if (m_uiEntranceEventTimer <= uiDiff)
487  {
488  Player* pPlayer = GetPlayerInMap();
489  if (!pPlayer)
490  return;
491 
492  uint32 uiEntry = urand(0, 10) ? NPC_PROTEAN_HORROR : NPC_PROTEAN_NIGHTMARE;
493 
494  // Protean Horrors stop spawning once 4 warders are killed
495  // Protean Nightmares stop spawning once 3 defenders are killed
496  if ((uiEntry == NPC_PROTEAN_HORROR && m_uiKilledWarders == MAX_WARDERS) || (uiEntry == NPC_PROTEAN_NIGHTMARE && m_uiKilledDefenders == MAX_DEFENDERS))
497  {
498  m_uiEntranceEventTimer = urand(0, 10) ? urand(2000, 3500) : urand(5000, 7000);
499  return;
500  }
501 
502  // Summon and move the intro creatures into combat positions
503  if (Creature* pTemp = pPlayer->SummonCreature(uiEntry, aEntranceSpawnLoc[0], aEntranceSpawnLoc[1], aEntranceSpawnLoc[2], aEntranceSpawnLoc[3], TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 30000))
504  {
505  pTemp->GetMotionMaster()->MoveWaypoint(1);
506  }
507  m_uiEntranceEventTimer = urand(0, 10) ? urand(2000, 3500) : urand(5000, 7000);
508  }
509  else
510  m_uiEntranceEventTimer -= uiDiff;
511  }
512 }
