37  if (unitLvl > targetLvl)

39  const uint32 diff = (unitLvl - targetLvl);

103  inline uint32 BaseGain(uint32 unit_level, uint32 mob_level, ContentLevels content)
104  {
105  uint32 nBaseExp = unit_level * 5;
106  switch (content)
107  {
108  case CONTENT_1_60: nBaseExp += 45; break;
109  case CONTENT_61_70: nBaseExp += 235; break;
110  case CONTENT_71_80: nBaseExp += 580; break;
111  default:
112  sLog.outError("BaseGain: Unsupported content level %u", content);
113  nBaseExp += 45; break;
114  }
115 
116  if (mob_level >= unit_level)
117  {
118  uint32 nLevelDiff = mob_level - unit_level;
119  if (nLevelDiff > 4)
120  nLevelDiff = 4;
121  return nBaseExp * (1.0f + (0.05f * nLevelDiff));
122  }
123 
124  if (!IsTrivialLevelDifference(unit_level, mob_level))
125  {
126  uint32 ZD = GetZeroDifference(unit_level);
127  uint32 nLevelDiff = unit_level - mob_level;
128  return nBaseExp * (1.0f - (float(nLevelDiff) / ZD));
129  }
130  return 0;
131  }
