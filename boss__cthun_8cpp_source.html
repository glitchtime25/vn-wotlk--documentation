203  {

207  // The first three casts of "Eye Beam" use a specific spell that always targets Eye of C'Thun's victim
208  if (m_eyeBeamCount < 3)

211  {

213  {
214  ResetCombatAction(action, urand(2, 3) * IN_MILLISECONDS);
215  ++m_eyeBeamCount;
216  }
217  }
218  }
219  // After the first three casts, random player is targeted with a second specific spell
220  else if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
221  {
222  if (DoCastSpellIfCan(target, SPELL_EYE_BEAM) == CAST_OK)
223  ResetCombatAction(action, urand(2, 3) * IN_MILLISECONDS);
224  }
225  break;
226  case EYECTHUN_DARKGLARE:
227  // Cast the rotation spell
228  if (DoCastSpellIfCan(m_creature, SPELL_ROTATE_TRIGGER) == CAST_OK)
229  {
230  // We ignore targets during Dark Glare to control Eye of C'Thun's orientation ourselves in spells
231  // So we need custom AI management for this sub phase
232  SetCombatScriptStatus(true);
233  m_creature->SetTarget(nullptr);
234 
235  // Switch to Dark Glare phase for 40 seconds
236  ResetTimer(EYECTHUN_END_DARKGLARE, 40 * IN_MILLISECONDS);
237  }
238  break;

240  break;

586  // Area trigger located at C'Thun feet (foot?)
