58 void ScriptedPetAI::UpdateAI(const uint32 diff)
59 {
60  if (!m_creature->IsAlive()) // should not be needed, IsAlive is checked in mangos before calling UpdateAI
61  return;
62 
63  // UpdateAllies() is done in the generic PetAI in Mangos, but we can't do this from script side.
64  // Unclear what side effects this has, but is something to be resolved from Mangos.
65 
66  if (m_creature->GetVictim()) // in combat
67  {
68  if (!m_creature->CanAttack(m_creature->GetVictim()))

71  // doesn't normally reach this, because of how petAi is designed in Mangos. CombatStop

73  ResetPetCombat();
74  return;
75  }
76 
77  // update when in combat
78  UpdatePetAI(diff);
79  }
80  else if (m_creature->GetCharmInfo())
81  {

83  m_creature->CombatStop(true, true);
84

86 
87  if (!owner)
88  return;
89

91  {
92  // Not correct in all cases.

94  AttackStart(owner->getAttackerForHelper());
95  }
96  else if (m_creature->GetCharmInfo()->HasCommandState(COMMAND_FOLLOW))
97  {
98  // not following, so start follow
99  if (!m_creature->hasUnitState(UNIT_STAT_FOLLOW))
100  m_creature->GetMotionMaster()->MoveFollow(owner, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
101 
102  // update when not in combat
103  UpdatePetOOCAI(diff);
104  }
105  }
