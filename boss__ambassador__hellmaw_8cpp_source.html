116  void UpdateAI(const uint32 uiDiff) override
117  {
118  if (m_uiBanishTimer)
119  {
120  if (m_uiBanishTimer <= uiDiff)
121  {
122  if (!m_pInstance)
123  return;
124 
125  // Check for banish
126  if (m_pInstance->IsHellmawUnbanished())
127  {
128  m_creature->RemoveAurasDueToSpell(SPELL_BANISH);
129  SetReactState(REACT_AGGRESSIVE);
130  m_creature->GetMotionMaster()->MoveWaypoint();
131  m_uiBanishTimer = 0;
132  }
133  }
134  else
135  m_uiBanishTimer -= uiDiff;
136  }
137 
138  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
139  return;
140 
141  if (m_uiCorrosiveAcidTimer < uiDiff)
142  {
143  if (DoCastSpellIfCan(m_creature, SPELL_CORROSIVE_ACID) == CAST_OK)
144  m_uiCorrosiveAcidTimer = urand(23000, 35000);
145  }
146  else
147  m_uiCorrosiveAcidTimer -= uiDiff;
148 
149  if (m_uiFearTimer < uiDiff)
150  {
151  if (DoCastSpellIfCan(m_creature, SPELL_FEAR) == CAST_OK)
152  m_uiFearTimer = urand(20000, 38000);
153  }
154  else
155  m_uiFearTimer -= uiDiff;
156 
157  if (!m_bIsRegularMode && !m_bIsEnraged)
158  {
159  if (m_uiEnrageTimer < uiDiff)
160  {
161  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
162  m_bIsEnraged = true;
163  }
164  else
165  m_uiEnrageTimer -= uiDiff;
166  }
167 
168  DoMeleeAttackIfReady();
169  }
