48 void CheckDbscript(WaypointNode& node, uint32 entry, uint32 point, std::set<uint32>& movementScriptSet, std::string const& tablename)
49 {
50  auto creatureMovementScripts = sScriptMgr.GetScriptMap(SCRIPT_TYPE_CREATURE_MOVEMENT);
51  auto relayScripts = sScriptMgr.GetScriptMap(SCRIPT_TYPE_RELAY);
52  auto iter = creatureMovementScripts->second.find(node.script_id);
53  if (iter == creatureMovementScripts->second.end())
54  {
55  sLog.outErrorDb("Table %s for entry %u, point %u have script_id %u that does not exist in `dbscripts_on_creature_movement`, ignoring", tablename.data(), entry, point, node.script_id);
56  return;
57  }
58  else if (!node.delay)
59  {
60  auto& data = *iter;
61  bool delay = false;
62  for (auto& item : data.second)
63  {
64  ScriptInfo const& scriptInfo = *item.second.get();
65  if (scriptInfo.delay != 0)
66  break;
67 
68  if (scriptInfo.command == SCRIPT_COMMAND_DESPAWN_SELF && scriptInfo.delay == 0 && scriptInfo.despawn.despawnDelay == 0)
69  {
70  delay = true;
71  sLog.outErrorDb("Table %s entry %u point %u has no delay and no delay despawn script. Adding delay to point.", tablename.data(), entry, point);
72  break;
73  }
74  else if (scriptInfo.command == SCRIPT_COMMAND_MOVEMENT)
75  {
76  delay = true;
77  sLog.outErrorDb("Table %s entry %u point %u has no delay but changes movegen. Adding delay to point.", tablename.data(), entry, point);
78  break;
79  }
80  else if (scriptInfo.command == SCRIPT_COMMAND_SET_RUN)
81  {
82  delay = true;
83  sLog.outErrorDb("Table %s entry %u point %u has no delay but changes run state. Adding delay to point.", tablename.data(), entry, point);
84  break;
85  }
86  else if (scriptInfo.command == SCRIPT_COMMAND_START_RELAY_SCRIPT)
87  {
88  auto iter = relayScripts->second.find(scriptInfo.relayScript.relayId);
89  if (iter == relayScripts->second.end())
90  {
91  if (scriptInfo.relayScript.templateId)
92  {
93  ScriptMgr::ScriptTemplateVector scriptTemplate;
94  sScriptMgr.GetScriptRelayTemplate(scriptInfo.relayScript.templateId, scriptTemplate);
95  for (auto& item : scriptTemplate)
96  {
97  auto iter = relayScripts->second.find(item.first);
98  if (iter != relayScripts->second.end())
99  {
100  auto& data = *iter;
101  for (auto& item : data.second)
102  {
103  if (scriptInfo.delay != 0)
104  break;
105  if (scriptInfo.command == SCRIPT_COMMAND_DESPAWN_SELF && scriptInfo.delay == 0 && scriptInfo.despawn.despawnDelay == 0)
106  {
107  delay = true;
108  sLog.outErrorDb("Table %s entry %u point %u has no delay and no delay despawn script. Adding delay to point.", tablename.data(), entry, point);
109  break;
110  }
111  else if (scriptInfo.command == SCRIPT_COMMAND_MOVEMENT)
112  {
113  delay = true;
114  sLog.outErrorDb("Table %s entry %u point %u has no delay but changes movegen. Adding delay to point.", tablename.data(), entry, point);
115  break;
116  }
117  }
118  if (delay)
119  break;
120  }
121  }
122  }
123  else
124  continue;
125  }
126  else
127  {
128  auto& data = *iter;
129  for (auto& item : data.second)
130  {
131  if (scriptInfo.delay != 0)
132  break;
133  if (scriptInfo.command == SCRIPT_COMMAND_DESPAWN_SELF && scriptInfo.delay == 0 && scriptInfo.despawn.despawnDelay == 0)
134  {
135  delay = true;
136  sLog.outErrorDb("Table %s entry %u point %u has no delay and no delay despawn script. Adding delay to point.", tablename.data(), entry, point);
137  break;
138  }
139  else if (scriptInfo.command == SCRIPT_COMMAND_MOVEMENT)
140  {
141  delay = true;
142  sLog.outErrorDb("Table %s entry %u point %u has no delay but changes movegen. Adding delay to point.", tablename.data(), entry, point);
143  break;
144  }
145  }
146  }
147  if (delay)
148  break;
149  }
150  }
151  if (delay)
152  node.delay = 1000;
153  }
154 
155  movementScriptSet.erase(node.script_id);

531 WaypointNode const* WaypointManager::AddNode(uint32 entry, uint32 dbGuid, uint32 pathId, uint32& pointId, WaypointPathOrigin wpDest, float x, float y, float z, float orientation)
532 {
533  // Support only normal movement tables
534  if (wpDest != PATH_FROM_GUID && wpDest != PATH_FROM_ENTRY)
535  return nullptr;
536 
537  // Prepare information
538  char const* const table = waypointOriginTables[wpDest];
539  char const* const key_field = waypointKeyColumn[wpDest];
540  WaypointPathMap* wpMap = getMapForPathType(wpDest);
541 
542  WaypointPath& path = (*wpMap)[wpDest == PATH_FROM_GUID ? dbGuid : ((entry << 8) + pathId)];
543 
544  if (pointId == 0 && !path.empty()) // Start with highest waypoint
545  pointId = path.rbegin()->first + 1;
546  else if (pointId == 0)
547  pointId = 1;
548 
549  uint32 nextPoint = pointId;
550  WaypointNode temp = WaypointNode(x, y, z, 100, 0, 0);
551  WaypointPath::iterator find = path.find(nextPoint);
552  if (find != path.end()) // Point already exists
553  {
554  do // Move points along until a free spot is found
555  {
556  std::swap(temp, find->second);
557  ++find;
558  ++nextPoint;
559  }
560  while (find != path.end() && find->first == nextPoint);
561  // After this, we have:
562  // pointId, pointId+1, ..., nextPoint [ Can be == path.end ]]
563  }
564 
565  // Insert new or remaining
566  path[nextPoint] = temp;
567 
568  uint32 key = wpDest == PATH_FROM_GUID ? dbGuid : entry;
569 
570  // Update original waypoints
571  for (WaypointPath::reverse_iterator rItr = path.rbegin(); rItr != path.rend() && rItr->first > pointId; ++rItr)
572  {
573  if (rItr->first <= nextPoint)
574  {
575  if (wpDest == PATH_FROM_ENTRY)
576  WorldDatabase.PExecuteLog("UPDATE %s SET Point=Point+1 WHERE %s=%u AND Point=%u AND PathId=%u", table, key_field, key, rItr->first - 1, pathId);
577  else
578  WorldDatabase.PExecuteLog("UPDATE %s SET Point=Point+1 WHERE %s=%u AND Point=%u", table, key_field, key, rItr->first - 1);
579  }
580  }
581  // Insert new Point to database
582  if (wpDest == PATH_FROM_ENTRY)
583  WorldDatabase.PExecuteLog("INSERT INTO %s (%s,PathId,Point,PositionX,PositionY,PositionZ,Orientation) VALUES (%u,%u,%u, %f,%f,%f, %f)", table, key_field, key, pathId, pointId, x, y, z, orientation);
584  else
585  WorldDatabase.PExecuteLog("INSERT INTO %s (%s,Point,PositionX,PositionY,PositionZ,Orientation) VALUES (%u,%u, %f,%f,%f, %f)", table, key_field, key, pointId, x, y, z, orientation);
586 
587  return &path[pointId];
588 }

590 void WaypointManager::DeleteNode(uint32 entry, uint32 dbGuid, uint32 point, uint32 pathId, WaypointPathOrigin wpOrigin)
591 {
592  // Support only normal movement tables
593  if (wpOrigin == PATH_FROM_EXTERNAL)
594  return;
595 
596  WaypointPath* path = GetPathFromOrigin(entry, dbGuid, pathId, wpOrigin);
597  if (!path)
598  return;
599 
600  char const* const table = waypointOriginTables[wpOrigin];
601  char const* const key_field = waypointKeyColumn[wpOrigin];

605  else if (wpOrigin == PATH_FROM_WAYPOINT_PATH)
606  key = pathId;

611  else

613 
614  path->erase(point);
