195  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
196  m_creature->AI()->AttackStart(target);
197  ResetCombatAction(NEFARIAN_SHADOW_FLAME, 12000u);
198  ResetCombatAction(NEFARIAN_BELLOWING_ROAR, 30000u);
199  ResetCombatAction(NEFARIAN_VEIL_OF_SHADOW, 15000u);

205  void ExecuteAction(uint32 action) override
206  {
207  switch (action)
208  {
209  case NEFARIAN_PHASE_3:
210  {
211  // Phase3 begins when we are below 20% health
212  if (m_creature->GetHealthPercent() < 20.0f)
213  {
214  if (DoCastSpellIfCan(nullptr, SPELL_RAISE_DRAKONID) == CAST_OK)
215  {
216  SetActionReadyStatus(action, false);
217  DoScriptText(SAY_RAISE_SKELETONS, m_creature);
218  }
219  }
220  break;
221  }
222  case NEFARIAN_LOW_HP_YELL:
223  {
224  if (m_creature->GetHealthPercent() < 5.0f)
225  {
226  SetActionReadyStatus(action, false);
227  DoScriptText(SAY_XHEALTH, m_creature);
228  }
229  break;
230  }
231  case NEFARIAN_SHADOW_FLAME:
232  {
233  if (DoCastSpellIfCan(nullptr, SPELL_SHADOWFLAME) == CAST_OK)
234  ResetCombatAction(action, 12000);
235  break;
236  }
237  case NEFARIAN_BELLOWING_ROAR:
238  {
239  if (DoCastSpellIfCan(nullptr, SPELL_BELLOWING_ROAR) == CAST_OK)
240  ResetCombatAction(action, 30000);
241  break;
242  }
243  case NEFARIAN_VEIL_OF_SHADOW:
244  {
245  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_VEIL_OF_SHADOW) == CAST_OK)
246  ResetCombatAction(action, 15000);
247  break;
248  }
249  case NEFARIAN_CLEAVE:
250  {
251  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
252  ResetCombatAction(action, 7000);
253  break;
254  }
255  case NEFARIAN_TAIL_LASH:
256  {
257  if (DoCastSpellIfCan(nullptr, SPELL_TAIL_LASH) == CAST_OK)
258  ResetCombatAction(action, 10000);
259  break;
260  }
261  case NEFARIAN_CLASS_CALL:
262  {
263  // Cast a random class call based on what classes are currently on the hostile list
264  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), SELECT_FLAG_PLAYER))
265  {
266  uint32 spellId = 0;
267  int32 textId = -1;
268  switch (target->getClass())
269  {
270  case CLASS_WARRIOR: spellId = SPELL_WARRIOR; textId = SAY_WARRIOR; break;
271  case CLASS_PALADIN: spellId = SPELL_PALADIN; textId = SAY_PALADIN; break;
272  case CLASS_HUNTER: spellId = SPELL_HUNTER; textId = SAY_HUNTER; break;
273  case CLASS_ROGUE: spellId = SPELL_ROGUE; textId = SAY_ROGUE; break;
274  case CLASS_PRIEST: spellId = SPELL_PRIEST; textId = SAY_PRIEST; break;
275  case CLASS_SHAMAN: spellId = SPELL_SHAMAN; textId = SAY_SHAMAN; break;
276  case CLASS_MAGE: spellId = SPELL_MAGE; textId = SAY_MAGE; break;
277  case CLASS_WARLOCK: spellId = SPELL_WARLOCK; textId = SAY_WARLOCK; break;
278  case CLASS_DRUID: spellId = SPELL_DRUID; textId = SAY_DRUID; break;
279  }
280  if (DoCastSpellIfCan(nullptr, spellId) == CAST_OK)
281  {
282  DoScriptText(textId, m_creature);
283  ResetCombatAction(action, urand(35000, 40000));
284  }
285  }
286  break;
287  }
288  }
289  }
290 };
