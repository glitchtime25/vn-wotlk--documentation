107  void ExecuteAction(uint32 action) override
108  {
109  switch (action)
110  {
111  case VOID_REAVER_BERSERK:
112  {
113  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
114  DisableCombatAction(action);
115  break;
116  }
117  case VOID_REAVER_POUNDING:
118  {
119  if (DoCastSpellIfCan(m_creature, SPELL_POUNDING) == CAST_OK)
120  {
121  DoScriptText(urand(0, 1) ? SAY_POUNDING1 : SAY_POUNDING2, m_creature);
122  ResetCombatAction(action, 14000);
123  }
124  break;
125  }
126  case VOID_REAVER_ARCANE_ORB:
127  {
128  // Search only for players which are not within 18 yards of the boss
129  std::vector<Unit*> suitableTargets;
130  ThreatList const& threatList = m_creature->getThreatManager().getThreatList();
131 
132  for (auto itr : threatList)
133  {
134  if (Unit* target = m_creature->GetMap()->GetUnit(itr->getUnitGuid()))
135  {
136  if (target->GetTypeId() == TYPEID_PLAYER && !target->IsWithinDist3d(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ(), 18.0f))
137  suitableTargets.push_back(target);
138  }
139  }
140 
141  if (suitableTargets.empty())
142  {
143  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
144  {
145  DoCastSpellIfCan(target, SPELL_ARCANE_ORB_MISSILE, CAST_TRIGGERED);
146  ResetCombatAction(action, 3000);
147  }
148  }
149  else
150  {
151  Unit* target = suitableTargets[urand(0, suitableTargets.size() - 1)];
152 
153  if (target)
154  DoCastSpellIfCan(target, SPELL_ARCANE_ORB_MISSILE, CAST_TRIGGERED);
155 
156  ResetCombatAction(action, 3000);
157  }
158  break;
159  }
160  case VOID_REAVER_KNOCK_AWAY:
161  {
162  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_KNOCK_AWAY) == CAST_OK)
163  ResetCombatAction(action, 30000);
164  break;
165  }
166  }
167  }
