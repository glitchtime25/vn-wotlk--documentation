205  {
206  // East
207  case TYPE_ZEVRIM:

218  break;
219  case TYPE_IRONBARK:

221  break;
222  case TYPE_ALZZIN: // This Encounter is expected to be handled within Acid (reason handling at 50% hp)
223  if (uiData == DONE)
224  {
225  if (!m_bWallDestroyed)
226  {
227  DoUseDoorOrButton(GO_CRUMBLE_WALL);

229  }
230

232

235  for (GuidList::const_iterator itr = m_lFelvineShardGUIDs.begin(); itr != m_lFelvineShardGUIDs.end(); ++itr)
236  DoRespawnGameObject(*itr);
237  }

239  else if (uiData == SPECIAL && !m_bWallDestroyed)

241  DoUseDoorOrButton(GO_CRUMBLE_WALL);
242  m_bWallDestroyed = true;
243  }
244  m_auiEncounter[uiType] = uiData;
245  break;
246 
247  // West
248  case TYPE_WARPWOOD:
249  if (uiData == DONE)
250  DoUseDoorOrButton(GO_WARPWOOD_DOOR);
251  m_auiEncounter[uiType] = uiData;
252  break;
253  case TYPE_IMMOLTHAR:
254  if (uiData == DONE)
255  {
256  if (Creature* pPrince = GetSingleCreatureFromStorage(NPC_PRINCE_TORTHELDRIN))
257  {
258  DoScriptText(SAY_FREE_IMMOLTHAR, pPrince);
259  pPrince->SetFactionTemporary(FACTION_HOSTILE, TEMPFACTION_RESTORE_RESPAWN | TEMPFACTION_TOGGLE_IMMUNE_TO_PLAYER);
260  // Despawn Chest-Aura
261  if (GameObject* pChestAura = GetSingleGameObjectFromStorage(GO_PRINCES_CHEST_AURA))
262  pChestAura->Use(pPrince);
263  }
264  }
265  m_auiEncounter[uiType] = uiData;
266  break;
267  case TYPE_PRINCE:
268  m_auiEncounter[uiType] = uiData;
269  break;
270  case TYPE_PYLON_1:
271  case TYPE_PYLON_2:
272  case TYPE_PYLON_3:
273  case TYPE_PYLON_4:
274  case TYPE_PYLON_5:
275  m_auiEncounter[uiType] = uiData;
276  if (uiData == DONE)
277  {
278  DoUseDoorOrButton(m_aCrystalGeneratorGuid[uiType - TYPE_PYLON_1]);
279  if (CheckAllGeneratorsDestroyed())
280  ProcessForceFieldOpening();
281  }
282  break;
283  case TYPE_DREADSTEED:
284  m_auiEncounter[uiType] = uiData;
285  if (uiData == FAIL)
286  {
287  if (Creature* pRitual = GetSingleCreatureFromStorage(NPC_WARLOCK_DUMMY_INFERNAL))
288  pRitual->ForcedDespawn();
289 
290  for (auto portal : m_lDreadsteedPortalsGUIDs)
291  instance->GetGameObject(portal)->SetLootState(GO_JUST_DEACTIVATED);
292 
293  for (auto symbol : m_lRitualSymbolGUIDs)
294  instance->GetGameObject(symbol)->SetLootState(GO_JUST_DEACTIVATED);

301  pGo->SetLootState(GO_JUST_DEACTIVATED);
302  pGo->SetRespawnDelay(1);
303  }
304  }
305 
306  if (GameObject *pGo = GetSingleGameObjectFromStorage(GO_WARLOCK_RITUAL_CIRCLE))
307  pGo->SetLootState(GO_JUST_DEACTIVATED);
308  }
309  break;
310  // North
311  case TYPE_KING_GORDOK:
312  m_auiEncounter[uiType] = uiData;
313  if (uiData == DONE)
314  {
315  // change faction to certian ogres
316  if (Creature* pOgre = GetSingleCreatureFromStorage(NPC_CAPTAIN_KROMCRUSH))

318  if (pOgre->IsAlive())

320  pOgre->SetFactionTemporary(FACTION_FRIENDLY, TEMPFACTION_RESTORE_RESPAWN);

504  + (GetData(TYPE_KROMCRUSH) != DONE ? 1 : 0) + (GetData(TYPE_CHORUSH) != DONE ? 1 : 0) + (GetData(TYPE_STOMPER_KREEG) != DONE ? 1 : 0);

508  }
509 
510  script_error_log("instance_dire_maul::CheckConditionCriteriaMeet called with unsupported Id %u. Called with param plr %s, src %s, condition source type %u",
511  uiInstanceConditionId, pPlayer ? pPlayer->GetGuidStr().c_str() : "NULL", pConditionSource ? pConditionSource->GetGuidStr().c_str() : "NULL", conditionSourceType);

525 void instance_dire_maul::ProcessForceFieldOpening()
526 {

528  DoUseDoorOrButton(GO_FORCEFIELD);
529 
530  // Let the summoners attack Immol'Thar
531  Creature* pImmolThar = GetSingleCreatureFromStorage(NPC_IMMOLTHAR);
532  if (!pImmolThar || pImmolThar->IsDead())
533  return;
534 
535  bool bHasYelled = false;
536  for (GuidList::const_iterator itr = m_luiHighborneSummonerGUIDs.begin(); itr != m_luiHighborneSummonerGUIDs.end(); ++itr)
537  {
538  Creature* pSummoner = instance->GetCreature(*itr);
539 
540  if (!bHasYelled && pSummoner)
541  {
542  DoScriptText(SAY_KILL_IMMOLTHAR, pSummoner);
543  bHasYelled = true;
544  }
545 
546  if (!pSummoner || pSummoner->IsDead())
547  continue;
548 
549  pSummoner->AI()->AttackStart(pImmolThar);
550  }
551  m_luiHighborneSummonerGUIDs.clear();

554 void instance_dire_maul::SortPylonGuards()
555 {
556  if (!m_lGeneratorGuardGUIDs.empty())
557  {
558  for (uint8 i = 0; i < MAX_GENERATORS; ++i)
559  {
560  GameObject* pGenerator = instance->GetGameObject(m_aCrystalGeneratorGuid[i]);
561  // Skip non-existing or finished generators
562  if (!pGenerator || GetData(TYPE_PYLON_1 + i) == DONE)
563  continue;
564 
565  // Sort all remaining (alive) NPCs to unfinished generators
566  for (GuidList::iterator itr = m_lGeneratorGuardGUIDs.begin(); itr != m_lGeneratorGuardGUIDs.end();)
567  {
