40 {
41  void OnHit(Spell* spell, SpellMissInfo missInfo) const override
42  {
43  if (missInfo != SPELL_MISS_NONE)
44  return;
45 
46  uint32 rage = spell->GetCaster()->GetPower(POWER_RAGE);
47 
48  // up to max 30 rage cost
49  if (rage > 300)
50  rage = 300;
51  // Sudden Death
52  if (spell->GetCaster()->HasAura(52437))
53  {
54  Unit::AuraList const& auras = spell->GetCaster()->GetAurasByType(SPELL_AURA_PROC_TRIGGER_SPELL);
55  for (auto aura : auras)
56  {
57  // Only Sudden Death have this SpellIconID with SPELL_AURA_PROC_TRIGGER_SPELL
58  if (aura->GetSpellProto()->SpellIconID == 1989)
59  {
60  // saved rage top stored in next affect
61  uint32 lastrage = aura->GetSpellProto()->CalculateSimpleValue(EFFECT_INDEX_1) * 10;
62  if (lastrage < rage)
63  rage -= lastrage;
64  break;
65  }
66  }
67  }
68  spell->GetCaster()->SetPower(POWER_RAGE, rage);
69  }
