166  void UpdateAI(const uint32 uiDiff) override
167  {
168  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
169  return;
170 
171  switch (m_uiPhase)
172  {
173  case PHASE_NORMAL:
174 
175  // Coldflame
176  if (m_uiColdflameTimer < uiDiff)
177  {
178  if (DoCastSpellIfCan(m_creature, SPELL_COLDFLAME) == CAST_OK)
179  m_uiColdflameTimer = 5000;
180  }
181  else
182  m_uiColdflameTimer -= uiDiff;
183 
184  // Bone Storm
185  if (m_uiBoneStormTimer < uiDiff)
186  {
187  if (DoCastSpellIfCan(m_creature, SPELL_BONE_STORM) == CAST_OK)
188  {
189  // ToDo: research if we need to increase the speed here
190  DoScriptText(SAY_BONE_STORM, m_creature);
191  m_uiPhase = PHASE_BONE_STORM_CHARGE;
192  SetCombatMovement(false);
193  m_creature->GetMotionMaster()->MoveIdle();
194  m_uiBoneStormTimer = 90000;
195  }
196  }
197  else
198  m_uiBoneStormTimer -= uiDiff;
199 
200  // Bone Slice
201  if (m_uiBoneSliceTimer < uiDiff)
202  {
203  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BONE_SLICE) == CAST_OK)
204  m_uiBoneSliceTimer = 1000;
205  }
206  else
207  m_uiBoneSliceTimer -= uiDiff;
208 
209  DoMeleeAttackIfReady();
210 
211  break;
212  case PHASE_BONE_STORM_CHARGE:
213 
214  // next charge to random enemy
215  if (m_uiBoneStormChargeTimer < uiDiff)
216  {
217  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
218  {
219  float fX, fY, fZ;
220  pTarget->GetPosition(fX, fY, fZ);
221  m_creature->SetWalk(false);
222  m_creature->GetMotionMaster()->Clear();
223  m_creature->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
224  m_uiBoneStormChargeTimer = 3000;
225  m_uiPhase = PHASE_BONE_STORM_CHARGING;
226  }
227  }
228  else
229  m_uiBoneStormChargeTimer -= uiDiff;
230 
231  break;
232  case PHASE_BONE_STORM_CHARGING:
233  // waiting to arrive at target position
234  break;
235  case PHASE_BONE_STORM_COLDFLAME:
236 
237  if (m_uiBoneStormColdflameTimer < uiDiff)
238  {
239  if (DoCastSpellIfCan(m_creature, SPELL_COLDFLAME_STORM) == CAST_OK)
240  {
241  // When the max cold flame charges are reached, remove Bone storm aura
242  if (m_uiChargesCount == m_uiMaxCharges)
243  {
244  m_creature->RemoveAurasDueToSpell(SPELL_BONE_STORM);
245  m_uiBoneStormTimer = 60000;
246  m_uiBoneSliceTimer = 10000;
247  SetCombatMovement(true);
248  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
249  m_uiChargesCount = 0;
250  m_uiPhase = PHASE_NORMAL;
251  }
252  else
253  m_uiPhase = PHASE_BONE_STORM_CHARGE;
254 
255  m_uiBoneStormColdflameTimer = 1000;
256  }
257  }
258  else
259  m_uiBoneStormColdflameTimer -= uiDiff;
260 
261  break;
262  }
263 
264  // Bone spike - different spells for the normal phase or storm phase
265  // On heroic mode Bone spike is also cast during the Bone Storm phase
266  if (m_bIsHeroicMode || m_uiPhase == PHASE_NORMAL)
267  {
268  if (m_uiBoneSpikeTimer < uiDiff)
269  {
270  if (DoCastSpellIfCan(m_creature, m_uiPhase == PHASE_NORMAL ? SPELL_BONE_SPIKE : SPELL_BONE_SPIKE_STORM) == CAST_OK)
271  {
272  switch (urand(0, 2))
273  {
274  case 0: DoScriptText(SAY_BONE_SPIKE_1, m_creature); break;
275  case 1: DoScriptText(SAY_BONE_SPIKE_2, m_creature); break;
276  case 2: DoScriptText(SAY_BONE_SPIKE_3, m_creature); break;
277  }
278  m_uiBoneSpikeTimer = 18000;
279  }
280  }
281  else
282  m_uiBoneSpikeTimer -= uiDiff;
283  }
284 
285  // Berserk
286  if (m_uiBerserkTimer)
287  {
288  if (m_uiBerserkTimer <= uiDiff)
289  {
290  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK))
291  {
292  DoScriptText(SAY_BERSERK, m_creature);
293  m_uiBerserkTimer = 0;
294  }
295  }
296  else
297  m_uiBerserkTimer -= uiDiff;
298  }
299  }
