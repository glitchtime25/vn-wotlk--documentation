42 uint32 GameEventMgr::NextCheck(uint16 entry) const
43 {
44  time_t currenttime = time(nullptr);
45 
46  // outdated event: we return max
47  if (currenttime > m_gameEvents[entry].end)
48  return max_ge_check_delay;
49 
50  // never started event, we return delay before start
51  if (m_gameEvents[entry].start > currenttime)
52  return uint32(m_gameEvents[entry].start - currenttime);
53 
54  uint32 delay;
55  // in event, we return the end of it
56  if ((((currenttime - m_gameEvents[entry].start) % (m_gameEvents[entry].occurence * 60)) < (m_gameEvents[entry].length * 60)))
57  // we return the delay before it ends
58  delay = (m_gameEvents[entry].length * MINUTE) - ((currenttime - m_gameEvents[entry].start) % (m_gameEvents[entry].occurence * MINUTE));
59  else // not in window, we return the delay before next start
60  delay = (m_gameEvents[entry].occurence * MINUTE) - ((currenttime - m_gameEvents[entry].start) % (m_gameEvents[entry].occurence * MINUTE));
61  // In case the end is before next check
62  if (m_gameEvents[entry].end < time_t(currenttime + delay))
63  return uint32(m_gameEvents[entry].end - currenttime);
64  return delay;
65 }
