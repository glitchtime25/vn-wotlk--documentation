153 bool ProcessEventId_event_antalarion_statue_activation(uint32 uiEventId, Object* pSource, Object* pTarget, bool /*bIsStart*/)
154 {
155  if (pSource->GetTypeId() == TYPEID_PLAYER && pTarget->GetTypeId() == TYPEID_GAMEOBJECT)
156  {
157  if (instance_sunken_temple* pInstance = (instance_sunken_temple*)((Player*)pSource)->GetInstanceData())
158  {
159  // return if event completed
160  if (pInstance->GetData(TYPE_ATALARION) != NOT_STARTED)
161  return true;
162 
163  // Send the event id to process
164  if (pInstance->ProcessStatueEvent(uiEventId))
165  {
166  // Activate the green light if the correct statue is activated
167  if (GameObject* pLight = GetClosestGameObjectWithEntry((GameObject*)pTarget, GO_ATALAI_LIGHT, INTERACTION_DISTANCE))
168  pInstance->DoRespawnGameObject(pLight->GetObjectGuid(), 30 * MINUTE);
169  }
170  else
171  {
172  // If the wrong statue was activated, then trigger trap
173  // We don't know actually which trap goes to which statue so we need to search for each
174  if (GameObject* pTrap = GetClosestGameObjectWithEntry((GameObject*)pTarget, GO_ATALAI_TRAP_1, INTERACTION_DISTANCE))
175  pTrap->Use((Unit*)pSource);
176  else if (GameObject* pTrap2 = GetClosestGameObjectWithEntry((GameObject*)pTarget, GO_ATALAI_TRAP_2, INTERACTION_DISTANCE))
177  pTrap2->Use((Unit*)pSource);
178  else if (GameObject* pTrap3 = GetClosestGameObjectWithEntry((GameObject*)pTarget, GO_ATALAI_TRAP_3, INTERACTION_DISTANCE))
179  pTrap3->Use((Unit*)pSource);
180  }
181 
182  return true;
183  }
184  }
185  return false;
186 }
