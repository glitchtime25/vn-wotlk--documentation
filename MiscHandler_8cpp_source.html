88 
89  recv_data >> guild_name; // guild name, case sensitive...
90 
91  recv_data >> racemask; // race mask
92  recv_data >> classmask; // class mask

94 
95  if (zones_count > 10)

98  // GM ticket hook shift+click to read

100  return;

102  for (uint32 i = 0; i < zones_count; ++i)
103  {
104  uint32 temp;
105  recv_data >> temp; // zone id, 0 if zone is unknown...
106  zoneids[i] = temp;

108  }

110  recv_data >> str_count; // user entered strings count, client limit=4 (checked on 2.0.10)
111

113  return; // can't be received from real client or broken packet
114 
115  DEBUG_LOG("Minlvl %u, maxlvl %u, name %s, guild %s, racemask %u, classmask %u, zones %u, strings %u", level_min, level_max, player_name.c_str(), guild_name.c_str(), racemask, classmask, zones_count, str_count);
116

119  {
120  std::string temp;
121  recv_data >> temp; // user entered string, it used as universal search pattern(guild+player name)?

124  continue;

126  wstrToLower(str[i]);

129  }

132  std::wstring wguild_name;

136  wstrToLower(wguild_name);
137

139  // update it to show GMs with characters after 100 level

141  level_max = STRONG_MAX_LEVEL;

144  uint32 security = GetSecurity();

149  uint32 displaycount = 0;

153  data << uint32(displaycount); // placeholder, count of players displayed

155  // TODO: Guard Player map

157  for (HashMapHolder<Player>::MapType::const_iterator itr = m.begin(); itr != m.end(); ++itr)

159  Player* pl = itr->second;
160 
161  if (security == SEC_PLAYER)

163  // player can see member of other team only if CONFIG_BOOL_ALLOW_TWO_SIDE_WHO_LIST
164  if (pl->GetTeam() != team && !allowTwoSideWhoList)
165  continue;

172  // do not process players which are not in world
173  if (!pl->IsInWorld())
174  continue;
175 
176  // check if target is globally visible for player

178  continue;
179

182  if (lvl < level_min || lvl > level_max)
183  continue;
184

188  continue;
189 
190  // check if race matches racemask

192  if (!(racemask & (1 << race)))

196  uint8 gender = pl->getGender();

202  {
203  z_show = true;
204  break;
205  }
206 
207  z_show = false;
208  }
209  if (!z_show)
210  continue;
211 
212  std::string pname = pl->GetName();
213  std::wstring wpname;
214  if (!Utf8toWStr(pname, wpname))
215  continue;
216  wstrToLower(wpname);
217 
218  if (!(wplayer_name.empty() || wpname.find(wplayer_name) != std::wstring::npos))
219  continue;
220

222  std::wstring wgname;
223  if (!Utf8toWStr(gname, wgname))
224  continue;
225  wstrToLower(wgname);
226 
227  if (!(wguild_name.empty() || wgname.find(wguild_name) != std::wstring::npos))
228  continue;

232  aname = areaEntry->area_name[GetSessionDbcLocale()];
233 
234  bool s_show = true;

236  {
237  if (!str[i].empty())
238  {
239  if (wgname.find(str[i]) != std::wstring::npos ||
240  wpname.find(str[i]) != std::wstring::npos ||
241  Utf8FitTo(aname, str[i]))
242  {
243  s_show = true;
244  break;
245  }
246  s_show = false;
247  }
248  }
249  if (!s_show)
250  continue;
251 
252  // 49 is maximum player count sent to client
253  if (++matchcount > 49)
254  continue;
255 
256  ++displaycount;
257 
258  data << pname; // player name
259  data << gname; // guild name
260  data << uint32(lvl); // player level
261  data << uint32(class_); // player class
262  data << uint32(race); // player race
263  data << uint8(gender); // player gender
264  data << uint32(pzoneid); // player zone id
265  }
266 
267  if (sWorld.getConfig(CONFIG_UINT32_MAX_WHOLIST_RETURNS) && matchcount > sWorld.getConfig(CONFIG_UINT32_MAX_WHOLIST_RETURNS))
268  matchcount = sWorld.getConfig(CONFIG_UINT32_MAX_WHOLIST_RETURNS);
269 
270  data.put(0, displaycount); // insert right count, count displayed
271  data.put(4, matchcount); // insert right count, count of matches
272 
273  SendPacket(data);
274  DEBUG_LOG("WORLD: Send SMSG_WHO Message");
275 }
