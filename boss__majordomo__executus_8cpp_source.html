215  void SummonedCreatureJustDied(Creature* summoned) override
216  {
217  if (summoned->GetEntry() == NPC_FLAMEWAKER_HEALER || summoned->GetEntry() == NPC_FLAMEWAKER_ELITE)
218  {
219  m_addsKilled += 1;
220 
221  // Yell if only one add is alive and buff it
222  if (m_addsKilled == m_luiMajordomoAddsGUIDs.size() - 1)
223  {
224  DoScriptText(SAY_LAST_ADD, m_creature);
225  DoCastSpellIfCan(nullptr, SPELL_CHAMPION);
226  }
227  // All adds are killed, retreat
228  else if (m_addsKilled == m_luiMajordomoAddsGUIDs.size())
229  {
230  m_bHasEncounterFinished = true;
231  m_creature->GetMotionMaster()->MoveTargetedHome();
232  SetCombatScriptStatus(true);
233  return;
234  }
235 
236  // If 4 adds (half of them) are dead, make all remaining healers immune to polymorph via aura
237  if (m_addsKilled >= MAX_MAJORDOMO_ADDS / 2)
238  DoCastSpellIfCan(nullptr, SPELL_IMMUNE_POLY);
239 
240  // Buff the remaining adds
241  DoCastSpellIfCan(nullptr, SPELL_ENCOURAGEMENT);
242  }

258  void HandleOutro()
259  {
260  uint32 timer = 0;
261  // Handling of his combat-end speech and Ragnaros summoning
262  switch (m_speechStage)
263  {
264  // Majordomo retreat event
265  case 1:
266  DoScriptText(SAY_DEFEAT_1, m_creature);
267  timer = 7500;
268  ++m_speechStage;
269  break;
270  case 2:
271  DoScriptText(SAY_DEFEAT_2, m_creature);
272  timer = 8000;
273  ++m_speechStage;
274  break;
275  case 3:
276  DoScriptText(SAY_DEFEAT_3, m_creature);
277  timer = 21500;
278  ++m_speechStage;
279  break;
280  case 4:
281  DoCastSpellIfCan(m_creature, SPELL_TELEPORT_SELF);
282  // TODO - when should they be unsummoned?
283  // TODO - also unclear how this should be handled, as of range issues
284  timer = 900;
285  ++m_speechStage;
286  break;
287  case 5:
288  // Majordomo is away now, remove his adds
289  UnsummonMajordomoAdds();
290  m_speechStage = 0;
291  break;
292 
293  // Ragnaros Summon Event
294  case 10:
295  DoScriptText(SAY_SUMMON_1, m_creature);
296  if (GameObject* lavaSplash = m_instance->GetSingleGameObjectFromStorage(GO_LAVA_SPLASH))
297  {
298  lavaSplash->SetRespawnTime(900);
299  lavaSplash->Refresh();
300  }
301  if (GameObject* lavaSteam = m_instance->GetSingleGameObjectFromStorage(GO_LAVA_STEAM))
302  {
303  lavaSteam->SetRespawnTime(900);
304  lavaSteam->Refresh();
305  }
306  ++m_speechStage;
307  timer = 1000;
308  break;
309  case 11:
310  DoCastSpellIfCan(nullptr, SPELL_SUMMON_RAGNAROS);
311  // TODO - Move along, this expects to be handled with mmaps
312  m_creature->GetMotionMaster()->MovePoint(1, 830.9636f, -814.7055f, -228.9733f);
313  ++m_speechStage;
314  timer = 11500;
315  break;
316  case 12:
317  // Reset orientation
318  if (GameObject* pLavaSteam = m_instance->GetSingleGameObjectFromStorage(GO_LAVA_STEAM))
319  m_creature->SetFacingToObject(pLavaSteam);
320  DoScriptText(SAY_SUMMON_MAJ, m_creature);
321  ++m_speechStage;
322  timer = 8000;
323  break;
324  case 13:
325  // Summon Ragnaros and make sure it faces Majordomo Executus
326  if (m_instance)
327  if (GameObject* pGo = m_instance->GetSingleGameObjectFromStorage(GO_LAVA_STEAM))
328  m_creature->SummonCreature(NPC_RAGNAROS, 838.3082f, -831.4665f, -232.1853f, 2.199115f, TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 2 * HOUR * IN_MILLISECONDS);
329  ++m_speechStage;
330  timer = 8700;
331  break;
332  case 14:
333  if (Creature* ragnaros = m_creature->GetMap()->GetCreature(m_ragnarosGuid))
334  {
335  ragnaros->HandleEmote(EMOTE_ONESHOT_ROAR);
336  DoScriptText(SAY_ARRIVAL1_RAG, ragnaros);
337  }
338  ++m_speechStage;
339  timer = 11700;
340  break;
341  case 15:
342  DoScriptText(SAY_ARRIVAL2_MAJ, m_creature);
343  ++m_speechStage;
344  timer = 8700;
345  break;
346  case 16:
347  if (Creature* ragnaros = m_creature->GetMap()->GetCreature(m_ragnarosGuid))
348  DoScriptText(SAY_ARRIVAL3_RAG, ragnaros);
349  ++m_speechStage;
350  timer = 16500;
351  break;
352  case 17:
353  if (Creature* ragnaros = m_creature->GetMap()->GetCreature(m_ragnarosGuid))
354  ragnaros->CastSpell(m_creature, SPELL_ELEMENTAL_FIRE, TRIGGERED_NONE);
355  // Rest of summoning speech is handled by Ragnaros, as Majordomo will be dead
356  m_speechStage = 0;
357  break;
358  }
359  if (timer)
360  ResetTimer(MAJORDOMO_OUTRO, timer);
361  }

363  void ExecuteAction(uint32 action) override
364  {
365  switch (action)
366  {
367  case MAJORDOMO_REFLECTION_SHIELD:
368  {
369  if (DoCastSpellIfCan(nullptr, (urand(0, 1) ? SPELL_DAMAGE_REFLECTION : SPELL_MAGIC_REFLECTION)) == CAST_OK)
370  ResetCombatAction(action, 30000);
371  break;
372  }
373  case MAJORDOMO_TELEPORT_RANDOM:
374  {
375  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_TELEPORT_TARGET) == CAST_OK)
376  {
377  DoCastSpellIfCan(nullptr, SPELL_HATE_TO_ZERO);
378  ResetCombatAction(action, urand(25000, 30000));
379  }
380  break;
381  }
382  case MAJORDOMO_TELEPORT_TARGET:
383  {
384  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
385  {
386  if (DoCastSpellIfCan(target, SPELL_TELEPORT_RANDOM) == CAST_OK)
387  {
388  DoCastSpellIfCan(nullptr, SPELL_HATE_TO_ZERO);
389  ResetCombatAction(action, urand(25000, 30000));
390  }
391  }
392  break;
393  }
394  case MAJORDOMO_AEGIS:
395  {
396  if (DoCastSpellIfCan(nullptr, SPELL_AEGIS, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT) == CAST_OK)
397  ResetCombatAction(action, urand(25000, 30000));
398  break;
399  }
400  }
