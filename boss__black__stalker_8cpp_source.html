85  uint32 GetSubsequentActionTimer(const uint32 action) const
86  {
87  switch (action)
88  {
89  case BLACK_STALKER_ACTION_LEVITATE: return 20500;
90  case BLACK_STALKER_ACTION_STATIC_CHARGE: return urand(14000, 16000);
91  case BLACK_STALKER_ACTION_CHAIN_LIGHTNING: return urand(12000, 29500);
92  case BLACK_STALKER_ACTION_SUMMON_SPORE_STRIDER: return urand(20000, 30000);
93  default: return 0; // never occurs but for compiler
94  }

97  void ExecuteAction(uint32 action) override
98  {
99  switch (action)
100  {
101  case BLACK_STALKER_ACTION_LEVITATE:
102  {
103  if (m_creature->getThreatManager().getThreatList().size() > 1)
104  if (DoCastSpellIfCan(nullptr, SPELL_LEVITATE) == CAST_OK)
105  ResetCombatAction(action, GetSubsequentActionTimer(action));
106  break;
107  }
108  case BLACK_STALKER_ACTION_STATIC_CHARGE:
109  {
110  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_STATIC_CHARGE, SELECT_FLAG_PLAYER))
111  if (DoCastSpellIfCan(target, SPELL_STATIC_CHARGE) == CAST_OK)
112  ResetCombatAction(action, GetSubsequentActionTimer(action));
113  break;
114  }
115  case BLACK_STALKER_ACTION_CHAIN_LIGHTNING:
116  {
117  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_CHAIN_LIGHTNING, SELECT_FLAG_PLAYER))
118  if (DoCastSpellIfCan(target, SPELL_CHAIN_LIGHTNING) == CAST_OK)
119  ResetCombatAction(action, GetSubsequentActionTimer(action));
120  break;
121  }
122  case BLACK_STALKER_ACTION_SUMMON_SPORE_STRIDER:
123  {
124  DoCastSpellIfCan(nullptr, SPELL_SUMMON_SPORE_STRIDER, CAST_TRIGGERED);
125  DoCastSpellIfCan(nullptr, SPELL_SUMMON_SPORE_STRIDER, CAST_TRIGGERED);
126  DoCastSpellIfCan(nullptr, SPELL_SUMMON_SPORE_STRIDER, CAST_TRIGGERED);
127  ResetCombatAction(action, GetSubsequentActionTimer(action));
128  break;
129  }
130  }
131  }
