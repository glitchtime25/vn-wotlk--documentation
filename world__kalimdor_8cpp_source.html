243  bool GhostOPlasmEventStep(GhostOPlasmEvent& eventData)
244  {
245  if (eventData.despawnTimer > 180000)
246  {
247  for (auto guid : eventData.summonedMagrami)
248  {
249  if (Creature * magrami = instance->GetCreature(guid))
250  {
251  if (magrami->IsAlive()) // dont despawn corpses with loot
252  {
253  magrami->CastSpell(nullptr, SPELL_SPIRIT_SPAWN_OUT, TRIGGERED_OLD_TRIGGERED);
254  magrami->ForcedDespawn(1000);
255  }
256  }
257  }
258 
259  if (GameObject* go = instance->GetGameObject(eventData.guid))
260  go->AddObjectToRemoveList(); // TODO: Establish rules for despawning temporary GOs that were used in their lifetime (buttons for example)
261 
262  return false;
263  }
264 
265  if (GameObject* go = instance->GetGameObject(eventData.guid))
266  {
267  if (eventData.despawnTimer / 15000 >= eventData.phaseCounter)
268  {
269  float x, y, z;
270  go->GetPosition(x, y, z); // do some urand radius shenanigans to spawn it further and make it walk to go using doing X and Y yourself and using function in MAP to get proper Z
271  uint32 random = urand(0, 35);
272  float xR = x + random, yR = y + (40 - random), zR = z;
273  instance->GetHeightInRange(go->GetPhaseMask(), xR, yR, zR);
274  Creature* creature = go->SummonCreature(NPC_MAGRAMI_SPECTRE, xR, yR, zR, 0, TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 180000); // add more timed logic here
275  instance->GetReachableRandomPointOnGround(go->GetPhaseMask(), x, y, z, 10.0f); // get position to which spectre will walk
276  eventData.phaseCounter++;
277  eventData.summonedMagrami.push_back(creature->GetObjectGuid());
278  creature->GetMotionMaster()->MovePoint(1, x, y, z);
279  }
280  return true;
281  }
282  return false;
283  }

285  void Update(uint32 diff) override
286  {
287  if (!m_vGOEvents.empty())
288  {
289  for (auto iter = m_vGOEvents.begin(); iter != m_vGOEvents.end();)
290  {
291  iter->despawnTimer += diff;
292  if (!GhostOPlasmEventStep((*iter)))
293  iter = m_vGOEvents.erase(iter);
294  else
295  ++iter;
296  }
297  }
298 
299  if (GetData(TYPE_OMEN) == DONE)
300  {
301  // Timer before Omen event reset (OOC)
302  if (m_uiOmenResetTimer < diff)
303  {
304  if (Creature* pOmen = GetSingleCreatureFromStorage(NPC_OMEN))
305  {
306  // Return is Omen is in fight
307  if (pOmen->IsInCombat())
308  return;
309  pOmen->ForcedDespawn();
310  }
311  m_encounter[TYPE_OMEN] = NOT_STARTED;
312  m_uiOmenResetTimer = 0;
313  m_uiRocketsCounter = 0;
314  m_uiOmenMoonlightTimer = 0;
315  b_isOmenSpellCreditDone = false;
316  }
317  else
318  m_uiOmenResetTimer -= diff;
319 
320  // Spell summoning GO traps for buff and quest credit for Omen
321  if (!b_isOmenSpellCreditDone)
322  {
323  if (m_uiOmenMoonlightTimer < diff)
324  {
325  if (Creature* pOmen = GetSingleCreatureFromStorage(NPC_OMEN))
326  {
327  pOmen->CastSpell(pOmen, SPELL_OMEN_MOONLIGHT, TRIGGERED_OLD_TRIGGERED);
328  b_isOmenSpellCreditDone = true;
329  }
330  m_uiOmenMoonlightTimer = 0;
331  }
332  else
333  m_uiOmenMoonlightTimer -= diff;
334  }
335  }
336  // Used for Hive Tower area trigger in Silithus
337  if (m_uiDronesTimer)
338  {
339  if (m_uiDronesTimer <= diff)
340  {
341  SetData(TYPE_HIVE, NOT_STARTED);
342  m_uiDronesTimer = 0;
343  }
344  else
345  m_uiDronesTimer -= diff;
346  }
347 
348  if (sGameEventMgr.IsActiveHoliday(HOLIDAY_BREWFEST))
349  m_brewfestEvent.Update(diff);
350  }

352  void SetData(uint32 uiType, uint32 uiData)
353  {
354  switch (uiType)
355  {
356  case TYPE_OMEN:
357  {
358  switch (uiData)
359  {
360  case NOT_STARTED:
361  // Count another rocket cluster launched
362  m_uiRocketsCounter++;
363  if (m_uiRocketsCounter < MAX_ROCKETS)
364  {
365  // 25% chance of spawning Minions of Omen (guessed), only if not already spawned (encounter is set to FAIL in that case)
366  if (GetData(TYPE_OMEN) == NOT_STARTED && urand(0, 1) < 1)
367  SetData(TYPE_OMEN, SPECIAL); // This will notify the GO to summon Omen's minions
368  }
369  // Set the event in motion and notify the GO to summon Omen
370  else if (GetData(TYPE_OMEN) != IN_PROGRESS && GetData(TYPE_OMEN) != DONE) // Check that Omen is not already spawned and event is reset
371  SetData(TYPE_OMEN, IN_PROGRESS);
372 
373  return; // Don't store NOT_STARTED data unless explicitly told so: we use it to count rockets
374  case SPECIAL:
375  if (GameObject* pRocketCluster = GetSingleGameObjectFromStorage(GO_ROCKET_CLUSTER))
376  {
377  for (uint8 i = POS_IDX_MINION_OMEN_START; i <= POS_IDX_MINION_OMEN_STOP; i++)
378  pRocketCluster->SummonCreature(NPC_MINION_OMEN, aSpawnLocations[i][0], aSpawnLocations[i][1], aSpawnLocations[i][2], aSpawnLocations[i][3], TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 15 * MINUTE * IN_MILLISECONDS);
379  }
380  break;
381  case IN_PROGRESS:
382  if (GameObject* pRocketCluster = GetSingleGameObjectFromStorage(GO_ROCKET_CLUSTER))
383  {
384  if (Creature* pOmen = pRocketCluster->SummonCreature(NPC_OMEN, aSpawnLocations[POS_IDX_OMEN_SPAWN][0], aSpawnLocations[POS_IDX_OMEN_SPAWN][1], aSpawnLocations[POS_IDX_OMEN_SPAWN][2], aSpawnLocations[POS_IDX_OMEN_SPAWN][3], TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 15 * MINUTE * IN_MILLISECONDS))
385  {
386  // Moving him to the lake bank
387  pOmen->SetWalk(true);
388  pOmen->GetMotionMaster()->MovePoint(1, aSpawnLocations[POS_IDX_OMEN_MOVE][0], aSpawnLocations[POS_IDX_OMEN_MOVE][1], aSpawnLocations[POS_IDX_OMEN_MOVE][2]);
389  m_uiOmenResetTimer = 15 * MINUTE * IN_MILLISECONDS; // Reset after 15 minutes if not engaged or defeated
390  }
391  }
392  break;
393  case DONE:
394  m_uiOmenMoonlightTimer = 5 * IN_MILLISECONDS; // Timer before casting the end quest spell
395  m_uiOmenResetTimer = 5 * MINUTE * IN_MILLISECONDS; // Prevent another summoning of Omen for 5 minutes (based on spell duration)
396  break;
397  }
398  }
399  case TYPE_TETHYR:
400  {
401  switch (uiData)
402  {
403  case NOT_STARTED:
404  instance->GetVariableManager().SetVariable(WORLD_STATE_TETHYR_SHOW, 0);
405  instance->GetVariableManager().SetVariable(WORLD_STATE_TETHYR_COUNT, 0);
406  instance->SetZoneWeather(0, AREAID_THERAMORE_ISLE, 0, 0.f);
407  break;
408  case SPECIAL: // Archer slain
409  instance->GetVariableManager().SetVariable(WORLD_STATE_TETHYR_COUNT, instance->GetVariableManager().GetVariable(WORLD_STATE_TETHYR_COUNT) - 1);
410  break;
411  case IN_PROGRESS:
412  if (m_encounter[uiType] != IN_PROGRESS)
413  instance->GetVariableManager().SetVariable(WORLD_STATE_TETHYR_COUNT, 12);
414  instance->GetVariableManager().SetVariable(WORLD_STATE_TETHYR_SHOW, 1);
415  instance->SetZoneWeather(0, AREAID_THERAMORE_ISLE, 3, 0.5f);
416  break;
417  }
418  }
419  case TYPE_HIVE:
420  {
421  if (uiData == IN_PROGRESS)
422  m_uiDronesTimer = 5 * MINUTE * IN_MILLISECONDS;
423  break;
424  }
425  case TYPE_FREEDOM_CREATURES:
426  {
427  if (uiData == IN_PROGRESS)
428  m_freedSpriteDarter = 0;
429  else if (uiData == SPECIAL)
430  {
431  ++m_freedSpriteDarter;
432  if (m_freedSpriteDarter >= 6)
433  uiData = DONE;
434  }
435  }
436  case TYPE_GONG_TIME:
437  // TODO: Handle initial first gong only
438  break;
439  default:
440  if (uiType >= TYPE_SHADE_OF_THE_HORSEMAN_ATTACK_PHASE && uiType <= TYPE_SHADE_OF_THE_HORSEMAN_MAX)
441  return m_shadeData.HandleSetData(uiType, uiData);
442  break;
443  }
444  m_encounter[uiType] = uiData;
445  }
