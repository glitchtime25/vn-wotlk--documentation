64 enum CastFlags
65 {
66  CAST_INTERRUPT_PREVIOUS = 0x01, // Interrupt any spell casting
67  CAST_TRIGGERED = 0x02, // Triggered (this makes spell cost zero mana and have no cast time)
68  CAST_FORCE_CAST = 0x04, // Forces cast even if creature is out of mana or out of range
69  CAST_NO_MELEE_IF_OOM = 0x08, // Prevents creature from entering melee if out of mana or out of range
70  CAST_FORCE_TARGET_SELF = 0x10, // Forces the target to cast this spell on itself
71  CAST_AURA_NOT_PRESENT = 0x20, // Only casts the spell if the target does not have an aura from the spell
72  CAST_IGNORE_UNSELECTABLE_TARGET = 0x40, // Can target UNIT_FLAG_UNINTERACTIBLE - Needed in some scripts
73  CAST_SWITCH_CASTER_TARGET = 0x80, // Switches target and caster for spell cast
74  CAST_MAIN_SPELL = 0x100, // Marks main spell - DEPRECATED - use creature_spell_list instead
75  CAST_PLAYER_ONLY = 0x200, // Selects only player targets - substitution for EAI not having more params
76  CAST_DISTANCE_YOURSELF = 0x400, // If spell with this cast flag hits main aggro target, caster distances himself - EAI only - DEPRECATED - use creature_spell_list instead
77  CAST_TARGET_CASTING = 0x800, // Selects only targets that are casting - EAI only
78  CAST_ONLY_XYZ = 0x1000, // Targets only coords of target and not unit
79 };
