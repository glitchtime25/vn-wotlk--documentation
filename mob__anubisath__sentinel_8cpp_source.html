110  void Aggro(Unit* who) override
111  {
112  // Find all buddies
113  if (m_assistList.empty())
114  {
115  CreatureList assistList;
116  GetCreatureListWithEntryInGrid(assistList, m_creature, m_creature->GetEntry(), 80.0f);
117 
118  for (auto& iter : assistList)
119  m_assistList.push_back(iter->GetObjectGuid());
120 
121  if (m_assistList.size() != MAX_BUDDY)
122  script_error_log("npc_anubisath_sentinel for %s found too few/too many buddies, expected %u.", m_creature->GetGuidStr().c_str(), MAX_BUDDY);
123  }
124 
125  if (!m_myAbility)
126  {
127  std::shuffle(m_abilities.begin(), m_abilities.end(), *GetRandomGenerator()); // shuffle the abilities, they will be set to the current creature and its siblings
128  SetAbility();
129  InitSentinelsNear(who);
130  }

138  void SetAbility(bool forced=false, uint32 abilityId=0)
139  {
140  if (m_myAbility) // Do not set ability if already force set by the first Anubisath Sentinel to get aggro
141  return;
142 
143  // This Anubisath Sentinel is the first one to set his ability as it is the one that got initial aggro
144  // Pick the first ability in the (randomised) list of abilities
145  if (!forced)
146  m_myAbility = m_abilities[0];
147  // This Anubisath Sentinel is forced to set his ability by the first sentinel to get aggro
148  // Check that the requested ability is valid
149  else if (std::find(m_abilities.begin(), m_abilities.end(), abilityId) != m_abilities.end())
150  m_myAbility = abilityId;
151  else
152  {
153  script_error_log("npc_anubisath_sentinel for %s: request illegal spell ID as ability %u.", m_creature->GetGuidStr().c_str(), abilityId);
154  return;
155  }
156 
157  DoCastSpellIfCan(m_creature, m_myAbility, CAST_TRIGGERED);
158  }

160  void DoTransferAbility()
161  {
162  bool hasDoneEmote = false;
163  for (auto guid : m_assistList)
164  {
165  if (Creature* buddy = m_creature->GetMap()->GetCreature(guid))
166  {
167  // Don't transfer ability to dead buddies
168  if (guid == m_creature->GetObjectGuid() || !buddy->IsAlive())
169  continue;
170 
171  // Only emote on the first ability transfert
172  if (!hasDoneEmote)
173  {
174  DoScriptText(EMOTE_SHARE_POWERS, m_creature);
175  hasDoneEmote = true;
176  }
177  m_creature->CastCustomSpell(buddy, SPELL_TRANSFER_POWER, &m_myAbility, nullptr, nullptr, TRIGGERED_OLD_TRIGGERED, nullptr);
178  }
179  }
180  }

182  void InitSentinelsNear(Unit* target)
183  {
184  if (!m_assistList.empty())
185  {
186  int8 buddyCount = 1;
187  for (auto guid : m_assistList)
188  {
189  // Don't init ourselves
190  if (guid == m_creature->GetObjectGuid())
191  continue;
192 
193  if (Creature* buddy = m_creature->GetMap()->GetCreature(guid))
194  {
195  if (buddy->IsAlive())
196  {
197  npc_anubisath_sentinelAI* buddyAI = static_cast<npc_anubisath_sentinelAI*>(buddy->AI());
198  buddyAI->SetAbility(true, m_abilities[buddyCount]);
199  ++buddyCount;
200  buddy->AI()->AttackStart(target);
201  }
202  }
203  }
204  }
205  }

286  void Aggro(Unit* /*who*/) override
287  {
288  // At each aggro Anubisath Defender randomises its four abilities
289  m_creature->RemoveAllAuras();
290  // Pick between the two reflect spells
291  uint32 reflectSpellId = (urand(0, 1) ? SPELL_FIRE_ARCANE_REFLECT : SPELL_SHADOW_FROST_REFLECT);
292  DoCastSpellIfCan(m_creature, reflectSpellId, CAST_TRIGGERED);
293  // Pick between Meteor and Plague
294  if (urand(0, 1))
295  ResetCombatAction(ANUBDEFENDER_METEOR, urand(6, 10) * IN_MILLISECONDS);
296  else
297  ResetCombatAction(ANUBDEFENDER_PLAGUE, urand(6, 10) * IN_MILLISECONDS);
298  // Pick between Shadow Storm and Thunderclap
299  if (urand(0, 1))
300  ResetCombatAction(ANUBDEFENDER_THUNDERCLAP, urand(5, 8) * IN_MILLISECONDS);
301  else
302  ResetCombatAction(ANUBDEFENDER_SHADOW_STORM, urand(5, 8) * IN_MILLISECONDS);
303  // The ability at 10% is randomised when the event triggers
304  }

306  void ExecuteAction(uint32 action) override
307  {
308  switch (action)
309  {
310  case ANUBDEFENDER_HP_CHECK:
311  {
312  if (m_creature->GetHealthPercent() < 10.0f)
313  {
314  if (urand(0, 1))

318  DoScriptText(EMOTE_GENERIC_FRENZY, m_creature);

327  }

336  }
337  break;
