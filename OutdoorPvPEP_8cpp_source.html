59 void OutdoorPvPEP::HandlePlayerEnterZone(Player* player, bool isMainZone)
60 {
61  OutdoorPvP::HandlePlayerEnterZone(player, isMainZone);
62 
63  // remove the buff from the player first; Sometimes on relog players still have the aura
64  for (auto plaguelandsTowerBuff : plaguelandsTowerBuffs)
65  player->RemoveAurasDueToSpell(player->GetTeam() == ALLIANCE ? plaguelandsTowerBuff.spellIdAlliance : plaguelandsTowerBuff.spellIdHorde);
66 
67  // buff the player
68  switch (player->GetTeam())
69  {
70  case ALLIANCE:
71  if (m_towersAlliance > 0)
72  player->CastSpell(player, plaguelandsTowerBuffs[m_towersAlliance - 1].spellIdAlliance, TRIGGERED_OLD_TRIGGERED);
73  break;
74  case HORDE:
75  if (m_towersHorde > 0)
76  player->CastSpell(player, plaguelandsTowerBuffs[m_towersHorde - 1].spellIdHorde, TRIGGERED_OLD_TRIGGERED);
77  break;
78  default:
79  break;
80  }
81 }

183 // process the capture events
184 bool OutdoorPvPEP::HandleEvent(uint32 eventId, Object* source, Object* /*target*/)
185 {
186  if (!source->IsGameObject())
187  return false;
188 
189  GameObject* go = static_cast<GameObject*>(source);
190  for (uint8 i = 0; i < MAX_EP_TOWERS; ++i)
191  {
192  if (plaguelandsBanners[i] == go->GetEntry())
193  {
194  for (uint8 j = 0; j < 4; ++j)
195  {
196  if (plaguelandsTowerEvents[i][j].eventEntry == eventId)
197  {
198  // prevent processing if the owner did not change (happens if progress event is called after contest event)
199  if (plaguelandsTowerEvents[i][j].team != m_towerOwner[i])
200  {
201  if (plaguelandsTowerEvents[i][j].defenseMessage)
202  sWorld.SendDefenseMessageBroadcastText(ZONE_ID_EASTERN_PLAGUELANDS, plaguelandsTowerEvents[i][j].defenseMessage);
203 
204  return ProcessCaptureEvent(go, i, plaguelandsTowerEvents[i][j].team, plaguelandsTowerEvents[i][j].worldState);
205  }
206  // no need to iterate other events or towers
207  return false;
208  }
209  }
210  // no need to iterate other towers
211  return false;
212  }
213  }
214 
215  return false;
216 }

218 bool OutdoorPvPEP::ProcessCaptureEvent(GameObject* go, uint32 towerId, Team team, uint32 newWorldState)
219 {
220  // Remove existing buff for both teams before applying new one: this prevent auras stacking
221  if (m_towersAlliance != 0)
222  BuffTeam(ALLIANCE, plaguelandsTowerBuffs[m_towersAlliance - 1].spellIdAlliance, true);
223  if (m_towersHorde != 0)
224  BuffTeam(HORDE, plaguelandsTowerBuffs[m_towersHorde - 1].spellIdHorde, true);
225 
226  // Update world state and banners for captured tower
227  if (team == ALLIANCE)
228  {
229  // update banner
230  for (GuidList::const_iterator itr = m_towerBanners[towerId].begin(); itr != m_towerBanners[towerId].end(); ++itr)
231  SetBannerVisual(go, (*itr), CAPTURE_ARTKIT_ALLIANCE, CAPTURE_ANIM_ALLIANCE);
232 
233  // update counter
234  ++m_towersAlliance;
235  SendUpdateWorldState(WORLD_STATE_EP_TOWER_COUNT_ALLIANCE, m_towersAlliance);
236  }
237  else if (team == HORDE)
238  {
239  // update banner
240  for (GuidList::const_iterator itr = m_towerBanners[towerId].begin(); itr != m_towerBanners[towerId].end(); ++itr)
241  SetBannerVisual(go, (*itr), CAPTURE_ARTKIT_HORDE, CAPTURE_ANIM_HORDE);
242 
243  // update counter
244  ++m_towersHorde;
245  SendUpdateWorldState(WORLD_STATE_EP_TOWER_COUNT_HORDE, m_towersHorde);
246  }
247  else
248  {
249  // update banner
250  for (GuidList::const_iterator itr = m_towerBanners[towerId].begin(); itr != m_towerBanners[towerId].end(); ++itr)
251  SetBannerVisual(go, (*itr), CAPTURE_ARTKIT_NEUTRAL, CAPTURE_ANIM_NEUTRAL);
252 
253  if (m_towerOwner[towerId] == ALLIANCE)
254  {
255  // update counter
256  --m_towersAlliance;
257  SendUpdateWorldState(WORLD_STATE_EP_TOWER_COUNT_ALLIANCE, m_towersAlliance);
258  }
259  else
260  {
261  // update counter
262  --m_towersHorde;
263  SendUpdateWorldState(WORLD_STATE_EP_TOWER_COUNT_HORDE, m_towersHorde);
264  }
265  }
266 
267  // Update Echoes of Lordaeron aura for capturing team and opposite team (if needed)
268  // Apply auras with updated capture towers values
269  if (m_towersAlliance != 0)
270  BuffTeam(ALLIANCE, plaguelandsTowerBuffs[m_towersAlliance - 1].spellIdAlliance);
271  if (m_towersHorde != 0)
272  BuffTeam(HORDE, plaguelandsTowerBuffs[m_towersHorde - 1].spellIdHorde);
273 
274  // Update reward for captured tower
275  bool eventHandled = true;
276 
277  if (team != TEAM_NONE)
278  {
279  // update capture point owner before rewards are applied
280  m_towerOwner[towerId] = team;
281 
282  // apply rewards of changed tower
283  switch (towerId)
284  {
285  case TOWER_ID_NORTHPASS:
286  RespawnGO(go, team == ALLIANCE ? m_lordaeronShrineAlliance : m_lordaeronShrineHorde, true);
287  break;
288  case TOWER_ID_CROWNGUARD:
289  SetGraveYardLinkTeam(GRAVEYARD_ID_EASTERN_PLAGUE, GRAVEYARD_ZONE_EASTERN_PLAGUE, team, 0);
290  break;
291  case TOWER_ID_EASTWALL:
292  // Return false - allow the DB to handle summons
293  if (m_towerOwner[TOWER_ID_NORTHPASS] != team)
294  eventHandled = false;
295  break;
296  case TOWER_ID_PLAGUEWOOD:
297  // Return false - allow the DB to handle summons
298  eventHandled = false;
299  break;
300  }
301  }
302  else
303  {
304  // remove rewards of changed tower
305  switch (towerId)
306  {
307  case TOWER_ID_NORTHPASS:
308  RespawnGO(go, m_towerOwner[TOWER_ID_NORTHPASS] == ALLIANCE ? m_lordaeronShrineAlliance : m_lordaeronShrineHorde, false);
309  break;
310  case TOWER_ID_CROWNGUARD:
311  SetGraveYardLinkTeam(GRAVEYARD_ID_EASTERN_PLAGUE, GRAVEYARD_ZONE_EASTERN_PLAGUE, TEAM_INVALID, 0);
312  break;
313  case TOWER_ID_EASTWALL:
314  UnsummonSoldiers(go);
315  break;
316  case TOWER_ID_PLAGUEWOOD:
317  UnsummonFlightMaster(go);
318  break;
319  }
320 
321  // update capture point owner after rewards have been removed
322  m_towerOwner[towerId] = team;
323  }
324 
325  // update tower state
326  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_REMOVE);
327  m_towerWorldState[towerId] = newWorldState;
328  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_ADD);
329 
330  // there are some events which required further DB script
331  return eventHandled;
332 }
