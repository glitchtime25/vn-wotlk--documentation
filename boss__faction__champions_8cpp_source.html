348  void DamageTaken(Unit* pDealer, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
349  {
350  uint32 uiStep = m_uiThrowAIEventStep != 100 ? m_uiThrowAIEventStep : 0;
351  if (uiStep < CRUSADER_HEALTH_STEPS)
352  {
353  // Throw at 90%, 50% and 10% health
354  float fHealthSteps[CRUSADER_HEALTH_STEPS] = { 90.0f, 50.0f, 10.0f };
355  float fNewHealthPercent = (m_creature->GetHealth() - uiDamage) * 100.0f / m_creature->GetMaxHealth();
356  AIEventType sendEvent[CRUSADER_HEALTH_STEPS] = { AI_EVENT_LOST_SOME_HEALTH, AI_EVENT_LOST_HEALTH, AI_EVENT_CRITICAL_HEALTH };
357 
358  if (fNewHealthPercent > fHealthSteps[uiStep])
359  return; // Not reached the next mark
360 
361  // search for highest reached mark (with actual event attached)
362  for (uint32 i = CRUSADER_HEALTH_STEPS - 1; i > uiStep; --i)
363  {
364  if (fNewHealthPercent < fHealthSteps[i])
365  {
366  uiStep = i;
367  break;
368  }
369  }
370 
371  // send event around and to self
372  SendAIEventAround(sendEvent[uiStep], pDealer, 0, CRUSADER_AIEVENT_THROW_RADIUS);
373  SendAIEvent(sendEvent[uiStep], pDealer, m_creature);
374  m_uiThrowAIEventStep = uiStep + 1;
375  }
376  }

395  bool CanUseSpecialAbility(uint32 uiSpellId, uint32 uiTargetType, SelectFlags selectFlag, uint32 uiMaxHpPct)
396  {
397  Unit* pTarget = nullptr;
398 
399  switch (uiTargetType)
400  {
401  case TARGET_TYPE_SELF:
402  pTarget = m_creature;
403  break;
404  case TARGET_TYPE_VICTIM:
405  pTarget = m_creature->GetVictim();
406  break;
407  case TARGET_TYPE_RANDOM:
408  pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(uiSpellId), selectFlag);
409  break;
410  case TARGET_TYPE_FRIENDLY:
411  // calculate HP deficit compared to caster health; this might not give very accurate results
412  pTarget = DoSelectLowestHpFriendly(40.0f, uint32(m_creature->GetMaxHealth() * (100 - uiMaxHpPct) * 0.01f));
413  break;
414  }
415 
416  if (pTarget)
417  {
418  if (DoCastSpellIfCan(pTarget, uiSpellId) == CAST_OK)
419  return true;
420  }
421 
422  return false;
423  }

428  void UpdateAI(const uint32 uiDiff) override
429  {
430  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
431  return;
432 
433  // Call specific virtual function
434  if (!UpdateCrusaderAI(uiDiff))
435  return;
436 
437  if (m_uiAbilityTimer < uiDiff)
438  {
439  uint8 uiIndex = urand(0, m_uiMaxAbilities - 1);
440  uint32 uiMinHealth = m_pAbilityArray[uiIndex].m_uiMinHealth;
441  uint8 uiTargetType = m_pAbilityArray[uiIndex].m_uiTargetType;
442 
443  SelectFlags spellSelectFlag = m_pAbilityArray[uiIndex].m_selectFlag;
444 
445  // check timers and health condition
446  // only cast spells that have timers expired
447  // also check for health percentage for self cast spells
448  if (m_uiSpellTimer[uiIndex] || (uiTargetType == TARGET_TYPE_SELF && uiMinHealth && m_creature->GetHealthPercent() > uiMinHealth))
449  {
450  m_uiAbilityTimer = 2000;
451  return;
452  }
453  uint32 uiSpellId = m_pAbilityArray[uiIndex].m_uiSpellId;
454 
455  // special case for heroism / bloodlust
456  if (uiSpellId == SPELL_HEROISM && m_pInstance && m_pInstance->GetPlayerTeam() == ALLIANCE)
457  uiSpellId = SPELL_BLOODLUST;
458 
459  if (CanUseSpecialAbility(uiSpellId, uiTargetType, spellSelectFlag, uiMinHealth))
460  {
461  m_uiSpellTimer[uiIndex] = m_pAbilityArray[uiIndex].m_uiCooldown;
462  m_uiAbilityTimer = urand(2000, 6000);
463  }
464  else
465  m_uiAbilityTimer = 2000;
466  }
467  else
468  m_uiAbilityTimer -= uiDiff;
469 
470  // spell cooldown
471  for (uint8 i = 0; i < m_uiMaxAbilities; ++i)
472  {
473  if (m_uiSpellTimer[i])
474  {
475  if (m_uiSpellTimer[i] <= uiDiff)
476  m_uiSpellTimer[i] = 0;
477  else
478  m_uiSpellTimer[i] -= uiDiff;
479  }
480  }
481 
482  // Change target
483  if (m_uiResetThreatTimer < uiDiff)
484  {
485  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
486  {
487  DoResetThreat();
488  AttackStart(pTarget);
489  m_uiResetThreatTimer = urand(5000, 15000);
490  }
491  }
492  else
493  m_uiResetThreatTimer -= uiDiff;
494 
495  // CC check for PVP trinket
496  if (m_uiIsCCTimer < uiDiff)
497  {
498  if (m_creature->isFrozen() || m_creature->IsCrowdControlled())
499  {
500  // Pvp trinket only in heroic mode
501  if (m_pInstance && m_pInstance->IsHeroicDifficulty() && !m_uiTrinketCooldownTimer)
502  {
503  if (DoCastSpellIfCan(m_creature, SPELL_PVP_TRINKET, CAST_TRIGGERED) == CAST_OK)
504  m_uiTrinketCooldownTimer = 120000;
505  }
506 
507  SendAIEventAround(AI_EVENT_GOT_CCED, nullptr, 0, CRUSADER_AIEVENT_THROW_RADIUS);
508  SendAIEvent(AI_EVENT_GOT_CCED, nullptr, m_creature);
509  m_uiIsCCTimer = 5000;
510  }
511  else
512  m_uiIsCCTimer = 2000;
513  }
514  else
515  m_uiIsCCTimer -= uiDiff;
516 
517  // trinket cooldown
518  if (m_uiTrinketCooldownTimer)
519  {
520  if (m_uiTrinketCooldownTimer <= uiDiff)
521  m_uiTrinketCooldownTimer = 0;
522  else
523  m_uiTrinketCooldownTimer -= uiDiff;
524  }
525 
526  DoMeleeAttackIfReady();
527  }
