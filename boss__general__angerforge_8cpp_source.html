75  void UpdateAI(const uint32 uiDiff) override
76  {
77  // Return since we have no target
78  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
79  return;
80 
81  // Sunder_Armor-Timer
82  if (m_uiSunderArmorTimer < uiDiff)
83  {
84  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SUNDER_ARMOR) == CAST_OK)
85  m_uiSunderArmorTimer = urand(5 * IN_MILLISECONDS, 15 * IN_MILLISECONDS);
86  }
87  else
88  m_uiSunderArmorTimer -= uiDiff;
89 
90  // Alarm-Timer
91  if (m_creature->GetHealthPercent() < 30.0f)
92  {
93  if (m_uiAlarmTimer < uiDiff)
94  {
95  DoScriptText(EMOTE_ALARM, m_creature);
96 
97  for (uint8 i = 0; i < NPC_NORMAL_AMOUNT; i++)
98  SummonAdd(NPC_ANVILRAGE_RESERVIST);
99  for (uint8 i = 0; i < NPC_ELITE_AMOUNT; i++)
100  SummonAdd(NPC_ANVILRAGE_MEDIC);
101 
102  m_uiAlarmTimer = 3 * MINUTE * IN_MILLISECONDS;
103  }
104  else
105  m_uiAlarmTimer -= uiDiff;
106  }
107 
108  DoMeleeAttackIfReady();
109  }
