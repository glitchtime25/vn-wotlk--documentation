140  void ReceiveAIEvent(AIEventType eventType, Unit* sender, Unit* /*invoker*/, uint32 /*miscValue*/) override
141  {
142  // allow events only from the following entries
143  switch (sender->GetEntry())
144  {
145  case NPC_ICHORON:
146  case NPC_ICHOR_GLOBULE:
147  case NPC_SWIRLING:
148  break;
149  default:
150  return;
151  }
152 
153  // handle water globules
154  if (eventType == AI_EVENT_CUSTOM_A)
155  {
156  DoScriptText(EMOTE_BUBBLE, m_creature);
157  DoScriptText(SAY_SHATTERING, m_creature);
158 
159  DoCastSpellIfCan(m_creature, SPELL_BURST, CAST_TRIGGERED);
160  DoCastSpellIfCan(m_creature, SPELL_SPLATTER, CAST_TRIGGERED);
161  DoCastSpellIfCan(m_creature, SPELL_DRAINED, CAST_TRIGGERED);
162 
163  // reduce health and set unselectable
164  m_creature->SetHealthPercent(m_creature->GetHealthPercent() - 30.0f);
165  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
166 
167  m_creature->SetFeignDeath(true);
168 
169  DisableCombatAction(ICHORON_WATER_BLAST);
170  DisableCombatAction(ICHORON_WATER_BOLT_VOLLEY);
171  }
172  // a merg has occurred; end split
173  else if (eventType == AI_EVENT_CUSTOM_B)
174  {
175  m_creature->SetHealthPercent(m_creature->GetHealthPercent() + 3.0f);
176 
177  // remove one stack from the shrink aura
178  if (Aura* shrinkAura = m_creature->GetAura(SPELL_SHRINK, EFFECT_INDEX_0))
179  shrinkAura->GetHolder()->ModStackAmount(-1, m_creature);
180 
181  EndSplitPhase();
182 
183  // fail achievement
184  if (m_instance)
185  m_instance->SetData(TYPE_ICHORON, SPECIAL);
186  }
187  // end split at the end of the Drained aura
188  else if (eventType == AI_EVENT_CUSTOM_C)
189  EndSplitPhase();
190  }

208  // method to end the split phase
209  void EndSplitPhase()
210  {
211  if (m_creature->HasAura(SPELL_DRAINED))
212  {
213  m_creature->RemoveAurasDueToSpell(SPELL_DRAINED);
214  DoScriptText(SAY_SHIELD, m_creature);
215  }
216 
217  m_creature->SetFeignDeath(false);
218 
219  // apply shield if HP is high enough
220  if (m_creature->GetHealthPercent() > 35.0f)
221  DoCastSpellIfCan(m_creature, SPELL_PROTECTIVE_BUBBLE, CAST_AURA_NOT_PRESENT);
222 
223  // reset timers
224  ResetCombatAction(ICHORON_WATER_BLAST, urand(8000, 14000));
225  ResetCombatAction(ICHORON_WATER_BOLT_VOLLEY, urand(7000, 12000));
226 
227  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
228  }

230  void ExecuteAction(uint32 action) override
231  {
232  switch (action)
233  {
234  case ICHORON_WATER_BLAST:
235  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
236  {
237  if (DoCastSpellIfCan(pTarget, m_isRegularMode ? SPELL_WATER_BLAST : SPELL_WATER_BLAST_H) == CAST_OK)
238  ResetCombatAction(action, urand(8000, 14000));
239  }
240  break;
241  case ICHORON_WATER_BOLT_VOLLEY:
242  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_WATER_BOLT_VOLLEY : SPELL_WATER_BOLT_VOLLEY_H) == CAST_OK)
243  ResetCombatAction(action, urand(7000, 12000));
244  break;
245  case ICHORON_FRENZY:
246  if (m_creature->GetHealthPercent() < 25.0f)
247  {
248  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_FRENZY : SPELL_FRENZY_H) == CAST_OK)
249  {
250  DoScriptText(SAY_ENRAGE, m_creature);
251  SetActionReadyStatus(action, false);
252  }
253  }
254  break;
255  }
256  }
