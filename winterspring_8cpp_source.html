194  // Called when Ranshalla starts to channel on a torch / altar
195  void DoChannelTorchSpell(bool bIsAltarWaypoint = false)
196  {
197  // Check if we are using the fire or the altar and remove the no_interact flag
198  if (bIsAltarWaypoint)
199  {
200  if (GameObject* pGo = GetClosestGameObjectWithEntry(m_creature, GO_ELUNE_ALTAR, 10.0f))
201  {
202  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
203  m_creature->SetFacingToObject(pGo);
204  m_altarGuid = pGo->GetObjectGuid();
205  }
206  }
207  else
208  {
209  if (GameObject* pGo = GetClosestGameObjectWithEntry(m_creature, GO_ELUNE_FIRE, 10.0f))
210  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
211  }
212 
213  // Yell and set escort to pause
214  switch (urand(0, 2))
215  {
216  case 0: DoScriptText(SAY_REACH_TORCH_1, m_creature); break;
217  case 1: DoScriptText(SAY_REACH_TORCH_2, m_creature); break;
218  case 2: DoScriptText(SAY_REACH_TORCH_3, m_creature); break;
219  }
220 
221  DoScriptText(EMOTE_CHANT_SPELL, m_creature);
222  DoCastSpellIfCan(m_creature, SPELL_LIGHT_TORCH);
223  SetEscortPaused(true);
224  }

251  void WaypointReached(uint32 uiPointId) override
252  {
253  switch (uiPointId)
254  {
255  case 3:
256  DoScriptText(SAY_ENTER_OWL_THICKET, m_creature);
257  break;
258  case 10: // Cavern 1
259  case 15: // Cavern 2
260  case 20: // Cavern 3
261  case 25: // Cavern 4
262  case 36: // Cavern 5
263  m_uiCurrentWaypoint = uiPointId;
264  DoChannelTorchSpell();
265  break;
266  case 39:
267  m_uiCurrentWaypoint = uiPointId;
268  StartNextDialogueText(SAY_REACH_ALTAR_1);
269  SetEscortPaused(true);
270  break;
271  case 41:
272  {
273  // Search for all nearest lights and respawn them
274  GameObjectList m_lEluneLights;
275  GetGameObjectListWithEntryInGrid(m_lEluneLights, m_creature, GO_ELUNE_LIGHT, 20.0f);
276  for (GameObjectList::const_iterator itr = m_lEluneLights.begin(); itr != m_lEluneLights.end(); ++itr)
277  {
278  if ((*itr)->IsSpawned())
279  continue;
280 
281  (*itr)->SetRespawnTime(115);
282  (*itr)->Refresh();
283  }
284 
285  if (GameObject* pAltar = m_creature->GetMap()->GetGameObject(m_altarGuid))
286  m_creature->SetFacingToObject(pAltar);
287  break;
288  }
289  case 42:
290  // Summon the 2 priestess
291  SetEscortPaused(true);
292  DoSummonPriestess();
293  DoScriptText(SAY_RANSHALLA_ALTAR_2, m_creature);
294  break;
295  case 44:
296  // Stop the escort and turn towards the altar
297  SetEscortPaused(true);
298  if (GameObject* pAltar = m_creature->GetMap()->GetGameObject(m_altarGuid))
299  m_creature->SetFacingToObject(pAltar);
300  break;
301  }
302  }

310  DoChannelTorchSpell(true);

312  case SAY_RANSHALLA_ALTAR_6:

315  case SAY_PRIESTESS_ALTAR_8:
316  // make the gem and its aura respawn

321

323  pGem->Refresh();
324  }
325  if (GameObject* pAura = GetClosestGameObjectWithEntry(m_creature, GO_ELUNE_AURA, 10.0f))
326  {
327  if (pAura->IsSpawned())
328  break;
329 
330  pAura->SetRespawnTime(90);
331  pAura->Refresh();
332  }
333  break;
334  case SAY_PRIESTESS_ALTAR_9:
335  // move near the escort npc
336  if (Creature* pPriestess = m_creature->GetMap()->GetCreature(m_firstPriestessGuid))
337  pPriestess->GetMotionMaster()->MovePoint(0, aWingThicketLocations[6].x, aWingThicketLocations[6].y, aWingThicketLocations[6].z);
338  break;
339  case SAY_PRIESTESS_ALTAR_13:
340  // summon the Guardian of Elune
341  if (Creature* pGuard = m_creature->SummonCreature(NPC_GUARDIAN_ELUNE, aWingThicketLocations[2].x, aWingThicketLocations[2].y, aWingThicketLocations[2].z, aWingThicketLocations[2].o, TEMPSPAWN_CORPSE_DESPAWN, 0))

346  // summon the Voice of Elune
347  if (GameObject* pAltar = m_creature->GetMap()->GetGameObject(m_altarGuid))
348  {
349  if (Creature* pVoice = m_creature->SummonCreature(NPC_VOICE_ELUNE, pAltar->GetPositionX(), pAltar->GetPositionY(), pAltar->GetPositionZ(), 0, TEMPSPAWN_TIMED_DESPAWN, 30000))
350  m_voiceEluneGuid = pVoice->GetObjectGuid();
351  }
352  break;
353  case SAY_VOICE_ALTAR_15:
354  // move near the escort npc and continue dialogue
355  if (Creature* pPriestess = m_creature->GetMap()->GetCreature(m_secondPriestessGuid))
356  {
357  DoScriptText(SAY_PRIESTESS_ALTAR_14, pPriestess);
358  pPriestess->GetMotionMaster()->MovePoint(0, aWingThicketLocations[7].x, aWingThicketLocations[7].y, aWingThicketLocations[7].z);
359  }
360  // make the voice of elune cast Bind Wildkin
361  if (Creature* pGuard = m_creature->GetMap()->GetCreature(m_guardEluneGuid))
362  pGuard->CastSpell(pGuard, SPELL_BIND_WILDKIN, TRIGGERED_NONE);
363  break;
364  case SAY_PRIESTESS_ALTAR_19:
365  // make the voice of elune leave
366  if (Creature* pGuard = m_creature->GetMap()->GetCreature(m_guardEluneGuid))
367  {
368  pGuard->GetMotionMaster()->MovePoint(0, aWingThicketLocations[2].x, aWingThicketLocations[2].y, aWingThicketLocations[2].z);
369  pGuard->ForcedDespawn(4000);
370  }
371  break;
372  case SAY_PRIESTESS_ALTAR_20:
373  // make the first priestess leave
374  if (Creature* pPriestess = m_creature->GetMap()->GetCreature(m_firstPriestessGuid))
375  {
376  pPriestess->GetMotionMaster()->MovePoint(0, aWingThicketLocations[0].x, aWingThicketLocations[0].y, aWingThicketLocations[0].z);
377  pPriestess->ForcedDespawn(4000);
378  }
379  break;
380  case SAY_PRIESTESS_ALTAR_21:
381  // make the second priestess leave
382  if (Creature* pPriestess = m_creature->GetMap()->GetCreature(m_secondPriestessGuid))
383  {
384  pPriestess->GetMotionMaster()->MovePoint(0, aWingThicketLocations[1].x, aWingThicketLocations[1].y, aWingThicketLocations[1].z);
385  pPriestess->ForcedDespawn(4000);
386  }
387  break;
388  case DATA_EVENT_END:
389  // Turn towards the player
390  if (Player* pPlayer = GetPlayerForEscort())
391  {
392  m_creature->SetFacingToObject(pPlayer);
393  DoScriptText(SAY_QUEST_END_1, m_creature, pPlayer);

403  m_creature->ForcedDespawn(1 * MINUTE * IN_MILLISECONDS);

405  }

422  void UpdateEscortAI(const uint32 uiDiff) override
423  {
424  DialogueUpdate(uiDiff);
425 
426  if (m_uiDelayTimer)
427  {
428  if (m_uiDelayTimer <= uiDiff)
429  {
430  m_creature->InterruptNonMeleeSpells(false);
431  SetEscortPaused(false);
432  // Spell Ranshalla Waiting applies a root aura that triggers an internal timer in the waypoint movement manager
433  // We need to manually set the movement manager to the next waypoint in order to reset the timer unless the NPC will wait 3 min before moving again
434  m_creature->GetMotionMaster()->SetNextWaypoint(m_uiCurrentWaypoint + 1);
435  m_uiDelayTimer = 0;
436  }
437  else
438  m_uiDelayTimer -= uiDiff;
439  }
440 
441  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
442  return;
443 
444  DoMeleeAttackIfReady();
445  }

672  return;

675  DemonDespawn();
