96  void ExecuteAction(uint32 action) override
97  {
98  switch (action)
99  {
100  case AZUREGOS_ENRAGE_HP_CHECK:
101  {
102  if (m_creature->GetHealthPercent() < 25.0f)
103  {
104  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
105  DisableCombatAction(action);
106  }
107  break;
108  }
109  case AZUREGOS_MANASTORM:
110  {
111  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0))
112  {
113  if (DoCastSpellIfCan(pTarget, SPELL_MANA_STORM) == CAST_OK)
114  ResetCombatAction(action, urand(18, 35) * IN_MILLISECONDS);
115  }
116  break;
117  }
118  case AZUREGOS_TELEPORT:
119  {
120  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_VACUUM) == CAST_OK)
121  {
122  DoScriptText(SAY_TELEPORT, m_creature);
123  ResetCombatAction(action, urand(20, 30) * IN_MILLISECONDS);
124  }
125  break;
126  }
127  case AZUREGOS_CHILL:
128  {
129  if (DoCastSpellIfCan(m_creature, SPELL_CHILL) == CAST_OK)
130  ResetCombatAction(action, urand(13, 25) * IN_MILLISECONDS);
131  break;
132  }
133  case AZUREGOS_FROSTBREATH:
134  {
135  if (DoCastSpellIfCan(m_creature, SPELL_FROST_BREATH) == CAST_OK)
136  ResetCombatAction(action, urand(10, 25) * IN_MILLISECONDS);
137  break;
138  }
139  case AZUREGOS_REFLECTMAGIC:
140  {
141  if (DoCastSpellIfCan(m_creature, SPELL_REFLECT) == CAST_OK)
142  ResetCombatAction(action, urand(20, 35) * IN_MILLISECONDS);
143  break;
144  }
145  case AZUREGOS_CLEAVE:
146  {
147  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
148  ResetCombatAction(action, 7 * IN_MILLISECONDS);
149  break;
150  }
151  default:
152  break;
153  }
154  }
155 };
