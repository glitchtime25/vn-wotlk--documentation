252  void ExecuteAction(uint32 action) override
253  {
254  switch (action)
255  {
256  case INGVAR_ACTION_CLEAVE:
257  if (!m_bIsResurrected)
258  {
259  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
260  ResetCombatAction(action, 10000);
261  }
262  else
263  {
264  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_WOE_STRIKE : SPELL_WOE_STRIKE_H) == CAST_OK)
265  ResetCombatAction(action, 10000);
266  }
267  break;
268  case INGVAR_ACTION_SMASH:
269  if (!m_bIsResurrected)
270  {
271  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_SMASH : SPELL_SMASH_H) == CAST_OK)
272  ResetCombatAction(action, urand(10000, 15000));
273  }
274  else
275  {
276  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_DARK_SMASH : SPELL_DARK_SMASH_H) == CAST_OK)
277  ResetCombatAction(action, urand(10000, 15000));
278  }
279  break;
280  case INGVAR_ACTION_ROAR:
281  if (!m_bIsResurrected)
282  {
283  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_STAGGERING_ROAR : SPELL_STAGGERING_ROAR_H) == CAST_OK)
284  {
285  DoScriptText(EMOTE_ROAR, m_creature);
286  ResetCombatAction(action, urand(30000, 35000));
287  }
288  }
289  else
290  {
291  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_DREADFUL_ROAR : SPELL_DREADFUL_ROAR_H) == CAST_OK)
292  {
293  DoScriptText(EMOTE_ROAR, m_creature);
294  ResetCombatAction(action, urand(30000, 35000));
295  }
296  }
297  break;
298  case INGVAR_ACTION_ENRAGE:
299  if (!m_bIsResurrected)
300  {
301  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_ENRAGE : SPELL_ENRAGE_H) == CAST_OK)
302  ResetCombatAction(action, 10000);
303  }
304  else
305  {
306  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_AXE) == CAST_OK)
307  {
308  // change equipment temporarely
309  SetEquipmentSlots(false, EQUIP_UNEQUIP);
310 
311  DisableCombatAction(action);
312  DisableCombatAction(INGVAR_ACTION_SMASH);
313  DisableCombatAction(INGVAR_ACTION_CLEAVE);
314  }
315  }
316  break;
317  }
318  }

365  if (!m_pInstance)
366  return;
367 
368  switch (m_uiResurrectPhase)
369  {
370  case 0:
371  DoCastSpellIfCan(m_creature, SPELL_SCOURGE_RES_CHANNEL);
372  if (Creature* pIngvar = m_pInstance->GetSingleCreatureFromStorage(NPC_INGVAR))
373  {
374  if (pIngvar->HasAura(SPELL_SUMMON_BANSHEE))
375  pIngvar->RemoveAurasDueToSpell(SPELL_SUMMON_BANSHEE);
376  }
377  m_uiResurrectTimer = 3000;
378  break;
379  case 1:
380  if (Creature* pIngvar = m_pInstance->GetSingleCreatureFromStorage(NPC_INGVAR))
381  pIngvar->CastSpell(pIngvar, SPELL_SCOURGE_RES_SUMMON, TRIGGERED_OLD_TRIGGERED);
382  m_uiResurrectTimer = 5000;
383  break;
384  case 2:
385  if (Creature* pIngvar = m_pInstance->GetSingleCreatureFromStorage(NPC_INGVAR))
386  pIngvar->CastSpell(pIngvar, SPELL_SCOURGE_RES_HEAL, TRIGGERED_NONE);
387  m_uiResurrectTimer = 3000;
388  break;
389  case 3:
390  // inform Ingvar about transformation
391  if (Creature* pIngvar = m_pInstance->GetSingleCreatureFromStorage(NPC_INGVAR))
392  SendAIEvent(AI_EVENT_CUSTOM_B, m_creature, pIngvar);
393 
394  // despawn the creature
395  m_creature->GetMotionMaster()->MovePoint(2, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 50);
396  m_creature->ForcedDespawn(5000);
397  m_uiResurrectTimer = 0;
398  break;
399  }
400 
401  ++m_uiResurrectPhase;
402  }
403  else
404  m_uiResurrectTimer -= uiDiff;
405  }
406  }
