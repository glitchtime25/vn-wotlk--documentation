103 bool SRP6::CalculateVerifier(const std::string& rI, const char* salt)
104 {
105  if (s.SetHexStr(salt) == 0 || s.isZero())
106  return false;
107 
108  BigNumber I;
109  I.SetHexStr(rI.c_str());
110 
111  // in case of leading zeros in the rI hash, restore them
112  uint8 mDigest[Sha1Hash::GetLength()];
113  memset(mDigest, 0, Sha1Hash::GetLength());
114  if (I.GetNumBytes() <= Sha1Hash::GetLength())
115  {
116  auto const vect_I = I.AsByteArray();
117  memcpy(mDigest, vect_I.data(), vect_I.size());
118  }
119 
120  std::reverse(mDigest, mDigest + Sha1Hash::GetLength());
121 
122  Sha1Hash sha;
123  sha.UpdateData(s.AsByteArray());
124  sha.UpdateData(mDigest, Sha1Hash::GetLength());
125  sha.Finalize();
126  BigNumber x;
127  x.SetBinary(sha.GetDigest(), Sha1Hash::GetLength());
128  v = g.ModExp(x, N);
129 
130  return true;
