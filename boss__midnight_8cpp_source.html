104  {

107  case MIDNIGHT_PHASE_2:
108  {
109  // Spawn Attumen on 95% hp
110  if (m_creature->GetHealthPercent() < 95.0f)
111  {
112  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_ATTUMEN) == CAST_OK)
113  SetActionReadyStatus(action, false);
114  }
115  break;
116  }
117  case MIDNIGHT_PHASE_3:
118  {
119  // Spawn Attumen mounted at 25%
120  if (m_creature->GetHealthPercent() < 25.0f)
121  {
122  if (DoCastSpellIfCan(nullptr, SPELL_MOUNT, CAST_TRIGGERED) == CAST_OK)
123  SetActionReadyStatus(action, false);
124  }
125  break;

272  void ExecuteAction(uint32 action) override
273  {
274  switch (action)
275  {
276  case ATTUMEN_MOUNT:
277  {
278  if (m_creature->GetHealthPercent() > 25.0f)
279  return;
280 
281  if (Creature* midnight = m_instance->GetSingleCreatureFromStorage(NPC_MIDNIGHT))
282  {
283  midnight->CastSpell(m_creature, SPELL_MOUNT, TRIGGERED_OLD_TRIGGERED);
284  SetActionReadyStatus(action, false);
285  }
286  break;
287  }
288  case ATTUMEN_ACTION_CLEAVE:
289  {
290  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SHADOWCLEAVE);
291  ResetCombatAction(action, GetSubsequentActionTimer(action));
292  return;
293  }
294  case ATTUMEN_ACTION_CURSE:
295  {
296  DoCastSpellIfCan(m_creature, SPELL_INTANGIBLE_PRESENCE);
297  ResetCombatAction(action, GetSubsequentActionTimer(action));
298  return;
299  }
300  case ATTUMEN_ACTION_YELL:
301  {
302  DoScriptText(urand(0, 1) ? SAY_RANDOM_1 : SAY_RANDOM_2, m_creature);
303  ResetCombatAction(action, GetSubsequentActionTimer(action));
304  return;
305  }
306  case ATTUMEN_ACTION_KNOCKDOWN:
307  {
308  // Cast knockdown when mounted, otherwise uppercut
309  DoCastSpellIfCan(m_creature->GetVictim(), m_creature->GetEntry() == NPC_ATTUMEN_MOUNTED ? SPELL_KNOCKDOWN : SPELL_UPPERCUT);
310  ResetCombatAction(action, GetSubsequentActionTimer(action));
311  return;
312  }
313  case ATTUMEN_ACTION_CHARGE:
314  {
315  // Sanity check - If creature is mounted then cast charge
316  if (m_creature->GetEntry() == NPC_ATTUMEN_MOUNTED)
317  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_CHARGE, SELECT_FLAG_IN_LOS | SELECT_FLAG_PLAYER))
318  if (DoCastSpellIfCan(target, SPELL_CHARGE) == CAST_OK)
319  ResetCombatAction(action, GetSubsequentActionTimer(action));
320  return;
321  }
322  }
323  }

384 struct MidnightMount : public SpellScript
385 {
386  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const
387  {
388  if (effIdx == EFFECT_INDEX_0)
389  {
390  Unit* target = spell->GetUnitTarget();
391  Unit* caster = spell->GetCaster();
392  if (!target || !target->IsCreature() || !caster->IsCreature())
393  return;
394 
395  // Avoid possible DB errors
396  if (caster->GetEntry() == NPC_MIDNIGHT && target->GetEntry() == NPC_ATTUMEN)

398  // Prepare for mount
399  if (boss_midnightAI* midnightAI = dynamic_cast<boss_midnightAI*>(caster->AI()))
400  midnightAI->DoPrepareMount(static_cast<Creature*>(target));
401 
402  if (boss_attumenAI* attumenAI = dynamic_cast<boss_attumenAI*>(target->AI()))
403  attumenAI->SetActionReadyStatus(ATTUMEN_MOUNT, false);
404  }
405  }
