87 CombatManeuverReturns PlayerbotMageAI::DoFirstCombatManeuver(Unit* pTarget)
88 {
89  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
90  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
91  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
92  {
93  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
94  {
95  return RETURN_NO_ACTION_OK; // wait it out
96  }
97  else
98  {
99  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
100  }
101  }
102 
103  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
104  {
105  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
106  return RETURN_NO_ACTION_OK; // wait it out
107  else
108  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
109  }
110 
111  switch (m_ai.GetScenarioType())
112  {
113  case PlayerbotAI::SCENARIO_PVP_DUEL:
114  case PlayerbotAI::SCENARIO_PVP_BG:
115  case PlayerbotAI::SCENARIO_PVP_ARENA:
116  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
117  return DoFirstCombatManeuverPVP(pTarget);
118  case PlayerbotAI::SCENARIO_PVE:
119  case PlayerbotAI::SCENARIO_PVE_ELITE:
120  case PlayerbotAI::SCENARIO_PVE_RAID:
121  default:
122  return DoFirstCombatManeuverPVE(pTarget);
123  }
124 }

153 CombatManeuverReturns PlayerbotMageAI::DoNextCombatManeuverPVE(Unit* pTarget)
154 {
155  Unit* pVictim = pTarget->GetVictim();
156  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
157 
158  uint32 spec = m_bot.GetSpec();
159 
160  if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED && !meleeReach)
161  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
162  // if can't shoot OR have no ranged (wand) equipped
163  else if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE && (SHOOT == 0 || !m_bot.GetWeaponForAttack(RANGED_ATTACK, true, true)))
164  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
165 
166  //Used to determine if this bot is highest on threat
167  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);
168 
169  // Remove curse on group members
170  if (m_ai.HasDispelOrder() && DispelPlayer(GetDispelTarget(DISPEL_CURSE)) & RETURN_CONTINUE)
171  return RETURN_CONTINUE;
172 
173  if (newTarget) // TODO: && party has a tank
174  {
175  // Insert instant threat reducing spell (if a mage has one)
176 
177  // Have threat, can't quickly lower it. 3 options remain: Stop attacking, lowlevel damage (wand), keep on keeping on.
178  if (newTarget->GetHealthPercent() > 25)
179  {
180  // If elite, do nothing and pray tank gets aggro off you
181  // TODO: Is there an IsElite function? If so, find it and insert.
182  //if (newTarget->IsElite())
183  // return;
184 
185  // Not an elite. You could insert FEAR here but in any PvE situation that's 90-95% likely
186  // to worsen the situation for the group. ... So please don't.
187  CastSpell(SHOOT, pTarget);
188  return RETURN_CONTINUE;
189  }
190  }
191 
192  switch (spec)
193  {
194  case MAGE_SPEC_FROST:
195  if (ICY_VEINS > 0 && m_ai.In_Reach(&m_bot, ICY_VEINS) && !m_bot.HasAura(ICY_VEINS, EFFECT_INDEX_0) && CastSpell(ICY_VEINS, &m_bot))
196  return RETURN_CONTINUE;
197  if (ICE_BLOCK > 0 && m_ai.In_Reach(&m_bot, ICE_BLOCK) && pVictim == &m_bot && !m_bot.HasAura(ICE_BLOCK, EFFECT_INDEX_0) && CastSpell(ICE_BLOCK, &m_bot))
198  return RETURN_CONTINUE;
199  if (ICE_BARRIER > 0 && m_ai.In_Reach(&m_bot, ICE_BARRIER) && pVictim == &m_bot && !m_bot.HasAura(ICE_BARRIER, EFFECT_INDEX_0) && m_ai.GetHealthPercent() < 50 && CastSpell(ICE_BARRIER, &m_bot))
200  return RETURN_CONTINUE;
201  if (DEEP_FREEZE > 0 && m_ai.In_Reach(pTarget, DEEP_FREEZE) && pTarget->HasAura(AURA_STATE_FROZEN, EFFECT_INDEX_0) && !pTarget->HasAura(DEEP_FREEZE, EFFECT_INDEX_0) && CastSpell(DEEP_FREEZE, pTarget))
202  return RETURN_CONTINUE;
203  if (BLIZZARD > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, BLIZZARD) && m_ai.GetAttackerCount() >= 5 && CastSpell(BLIZZARD, pTarget))
204  {
205  m_ai.SetIgnoreUpdateTime(8);
206  return RETURN_CONTINUE;
207  }
208  if (CONE_OF_COLD > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && meleeReach && !pTarget->HasAura(CONE_OF_COLD, EFFECT_INDEX_0) && CastSpell(CONE_OF_COLD, pTarget))
209  return RETURN_CONTINUE;
210  if (FROSTBOLT > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, FROSTBOLT) && !pTarget->HasAura(FROSTBOLT, EFFECT_INDEX_0) && CastSpell(FROSTBOLT, pTarget))
211  return RETURN_CONTINUE;
212  if (FROST_WARD > 0 && m_ai.In_Reach(&m_bot, FROST_WARD) && !m_bot.HasAura(FROST_WARD, EFFECT_INDEX_0) && CastSpell(FROST_WARD, &m_bot))
213  return RETURN_CONTINUE;
214  if (FROST_NOVA > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && meleeReach && !pTarget->HasAura(FROST_NOVA, EFFECT_INDEX_0) && CastSpell(FROST_NOVA, pTarget))
215  return RETURN_CONTINUE;
216  if (ICE_LANCE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, ICE_LANCE) && CastSpell(ICE_LANCE, pTarget))
217  return RETURN_CONTINUE;
218  if (SUMMON_WATER_ELEMENTAL > 0 && CastSpell(SUMMON_WATER_ELEMENTAL))
219  return RETURN_CONTINUE;
220  if (COLD_SNAP > 0 && m_ai.In_Reach(&m_bot, COLD_SNAP) && CastSpell(COLD_SNAP, &m_bot))
221  return RETURN_CONTINUE;
222 
223  if (FROSTBOLT > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FROST) && m_ai.In_Reach(pTarget, FROSTBOLT))
224  return CastSpell(FROSTBOLT, pTarget);
225  break;
226 
227  case MAGE_SPEC_FIRE:
228  if (FIRE_WARD > 0 && m_ai.In_Reach(&m_bot, FIRE_WARD) && !m_bot.HasAura(FIRE_WARD, EFFECT_INDEX_0) && CastSpell(FIRE_WARD, &m_bot))
229  return RETURN_CONTINUE;
230  if (COMBUSTION > 0 && m_ai.In_Reach(&m_bot, COMBUSTION) && !m_bot.HasAura(COMBUSTION, EFFECT_INDEX_0) && CastSpell(COMBUSTION, &m_bot))
231  return RETURN_CONTINUE;
232  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIREBALL) && CastSpell(FIREBALL, pTarget))
233  return RETURN_CONTINUE;
234  if (FIRE_BLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIRE_BLAST) && CastSpell(FIRE_BLAST, pTarget))
235  return RETURN_CONTINUE;
236  if (FLAMESTRIKE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FLAMESTRIKE) && CastSpell(FLAMESTRIKE, pTarget))
237  return RETURN_CONTINUE;
238  if (SCORCH > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, SCORCH) && CastSpell(SCORCH, pTarget))
239  return RETURN_CONTINUE;
240  if (PYROBLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, PYROBLAST) && !pTarget->HasAura(PYROBLAST, EFFECT_INDEX_0) && CastSpell(PYROBLAST, pTarget))
241  return RETURN_CONTINUE;
242  if (BLAST_WAVE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.GetAttackerCount() >= 3 && meleeReach && CastSpell(BLAST_WAVE, pTarget))
243  return RETURN_CONTINUE;
244  if (DRAGONS_BREATH > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && meleeReach && CastSpell(DRAGONS_BREATH, pTarget))
245  return RETURN_CONTINUE;
246  if (LIVING_BOMB > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, LIVING_BOMB) && !pTarget->HasAura(LIVING_BOMB, EFFECT_INDEX_0) && CastSpell(LIVING_BOMB, pTarget))
247  return RETURN_CONTINUE;
248  if (FROSTFIRE_BOLT > 0 && m_ai.In_Reach(pTarget, FROSTFIRE_BOLT) && !pTarget->HasAura(FROSTFIRE_BOLT, EFFECT_INDEX_0) && CastSpell(FROSTFIRE_BOLT, pTarget))
249  return RETURN_CONTINUE;
250 
251  if (FIREBALL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIREBALL))
252  return CastSpell(FIREBALL, pTarget);
253  break;
254 
255  case MAGE_SPEC_ARCANE:
256  if (ARCANE_POWER > 0 && m_ai.In_Reach(pTarget, ARCANE_POWER) && CastSpell(ARCANE_POWER, pTarget))
257  return RETURN_CONTINUE;
258  if (ARCANE_MISSILES > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, ARCANE_MISSILES) && CastSpell(ARCANE_MISSILES, pTarget))
259  {
260  m_ai.SetIgnoreUpdateTime(3);
261  return RETURN_CONTINUE;
262  }
263  if (ARCANE_EXPLOSION > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.GetAttackerCount() >= 3 && meleeReach && CastSpell(ARCANE_EXPLOSION, pTarget))
264  return RETURN_CONTINUE;
265  if (COUNTERSPELL > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && pTarget->IsNonMeleeSpellCasted(true) && CastSpell(COUNTERSPELL, pTarget))
266  return RETURN_CONTINUE;
267  if (SLOW > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, SLOW) && !pTarget->HasAura(SLOW, EFFECT_INDEX_0) && CastSpell(SLOW, pTarget))
268  return RETURN_CONTINUE;
269  if (ARCANE_BARRAGE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, ARCANE_BARRAGE) && CastSpell(ARCANE_BARRAGE, pTarget))
270  return RETURN_CONTINUE;
271  if (ARCANE_BLAST > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_ARCANE) && m_ai.In_Reach(pTarget, ARCANE_BLAST) && CastSpell(ARCANE_BLAST, pTarget))

285  return CastSpell(FROSTBOLT, pTarget);

287  return CastSpell(FIREBALL, pTarget);

330 void PlayerbotMageAI::DoNonCombatActions()
331 {
332  // Remove curse on group members
333  if (m_ai.HasDispelOrder() && DispelPlayer(GetDispelTarget(DISPEL_CURSE)) & RETURN_CONTINUE)
334  return;
335 
336  // Buff armor
337  if (MOLTEN_ARMOR)
338  {
339  if (m_ai.SelfBuff(MOLTEN_ARMOR) == SPELL_CAST_OK)
340  return;
341  }
342  else if (MAGE_ARMOR)
343  {
344  if (m_ai.SelfBuff(MAGE_ARMOR) == SPELL_CAST_OK)
345  return;
346  }
347  else if (ICE_ARMOR)
348  {
349  if (m_ai.SelfBuff(ICE_ARMOR) == SPELL_CAST_OK)
350  return;
351  }
352  else if (FROST_ARMOR)
353  if (m_ai.SelfBuff(FROST_ARMOR) == SPELL_CAST_OK)
354  return;
355 
356  // buff group
357  // the check for group targets is performed by NeedGroupBuff (if group is found for bots by the function)
358  if (NeedGroupBuff(ARCANE_BRILLIANCE, ARCANE_INTELLECT) && m_ai.HasSpellReagents(ARCANE_BRILLIANCE))
359  {
360  if (Buff(&PlayerbotMageAI::BuffHelper, ARCANE_BRILLIANCE) & RETURN_CONTINUE)
361  return;
362  }
363  else if (Buff(&PlayerbotMageAI::BuffHelper, ARCANE_INTELLECT, JOB_MANAONLY) & RETURN_CONTINUE)
364  return;
365 
366  // if there is space on bag try to conjure some consumables
367  if (m_ai.CanStore())
368  {
369  // TODO: The beauty of a mage is not only its ability to supply itself with water, but to share its water
370  // So, conjure at *least* 1.25 stacks, ready to trade a stack and still have some left for self
371  if (m_ai.FindDrink() == nullptr && CONJURE_WATER && m_ai.CastSpell(CONJURE_WATER, m_bot) == SPELL_CAST_OK)
372  {
373  m_ai.TellMaster("I'm conjuring some water.");
374  m_ai.SetIgnoreUpdateTime(3);
