93 CombatManeuverReturns PlayerbotPriestAI::DoFirstCombatManeuver(Unit* pTarget)
94 {
95  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
96  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
97  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
98  {
99  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
100  {
101  if (PlayerbotAI::ORDERS_HEAL & m_ai.GetCombatOrder())
102  return HealPlayer(GetHealTarget());
103  else
104  return RETURN_NO_ACTION_OK; // wait it out
105  }
106  else
107  {
108  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
109  }
110  }
111 
112  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
113  {
114  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
115  return RETURN_NO_ACTION_OK; // wait it out
116  else
117  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
118  }
119 
120  switch (m_ai.GetScenarioType())
121  {
122  case PlayerbotAI::SCENARIO_PVP_DUEL:
123  case PlayerbotAI::SCENARIO_PVP_BG:
124  case PlayerbotAI::SCENARIO_PVP_ARENA:
125  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
126  return DoFirstCombatManeuverPVP(pTarget);
127  case PlayerbotAI::SCENARIO_PVE:
128  case PlayerbotAI::SCENARIO_PVE_ELITE:
129  case PlayerbotAI::SCENARIO_PVE_RAID:
130  default:
131  return DoFirstCombatManeuverPVE(pTarget);
132  }

172 {
173  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
174  uint32 spec = m_bot.GetSpec();
175 
176  // Define a tank bot will look at
177  // or Main Tank if healer has order to do so
178  Unit* mainTank;
179  if (m_ai.IsMainHealer())
180  mainTank = GetHealTarget(JOB_MAIN_TANK);
181  // Else look at tank from own group if one
182  else
183  mainTank = GetHealTarget(JOB_TANK, true);
184  // Finally get any tank
185  if (!mainTank)
186  mainTank = GetHealTarget(JOB_TANK);
187 
188  if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED && !meleeReach)
189  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
190  // if in melee range OR can't shoot OR have no ranged (wand) equipped
191  else if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE
192  && (SHOOT == 0 || !m_bot.GetWeaponForAttack(RANGED_ATTACK, true, true))
193  && !m_ai.IsHealer())
194  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
195 
196  // Priests will try to buff with Fear Ward
197  if (FEAR_WARD > 0 && m_bot.IsSpellReady(FEAR_WARD))
198  {
199  // Buff tank first
200  if (mainTank)
201  {
202  if (m_ai.In_Reach(mainTank, FEAR_WARD) && !mainTank->HasAura(FEAR_WARD, EFFECT_INDEX_0) && CastSpell(FEAR_WARD, mainTank))
203  return RETURN_CONTINUE;
204  }
205  // Else try to buff master
206  if (m_ai.In_Reach(&m_master, FEAR_WARD) && !m_master.HasAura(FEAR_WARD, EFFECT_INDEX_0) && CastSpell(FEAR_WARD, &m_master))
207  return RETURN_CONTINUE;
208  }
209 
210  //Used to determine if this bot is highest on threat
211  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);
212  if (newTarget && !PlayerbotAI::IsNeutralized(newTarget)) // TODO: && party has a tank
213  {
214  if (newTarget && FADE > 0 && !m_bot.HasAura(FADE, EFFECT_INDEX_0))
215  {
216  if (CastSpell(FADE, &m_bot))
217  {
218  //m_ai.TellMaster("I'm casting fade.");
219  return RETURN_CONTINUE;
220  }
221  else
222  m_ai.TellMaster("I have AGGRO.");
223  }
224 
225  // Heal myself
226  // TODO: move to HealTarget code
227  // TODO: you forgot to check for the 'temporarily immune to PW:S because you only just got it cast on you' effect
228  // - which is different effect from the actual shield.
229  if (m_ai.GetHealthPercent() < 25 && POWER_WORD_SHIELD > 0 && !m_bot.HasAura(POWER_WORD_SHIELD, EFFECT_INDEX_0))
230  {
231  if (CastSpell(POWER_WORD_SHIELD) & RETURN_CONTINUE)
232  {
233  //m_ai.TellMaster("I'm casting PW:S on myself.");
234  return RETURN_CONTINUE;
235  }
236  else if (m_ai.IsHealer()) // Even if any other RETURN_ANY_OK - aside from RETURN_CONTINUE
237  m_ai.TellMaster("Your healer's about TO DIE. HELP ME.");
238  }
239  if (m_ai.GetHealthPercent() < 35 && DESPERATE_PRAYER > 0 && m_ai.In_Reach(&m_bot, DESPERATE_PRAYER) && CastSpell(DESPERATE_PRAYER, &m_bot) & RETURN_CONTINUE)
240  {
241  //m_ai.TellMaster("I'm casting desperate prayer.");
242  return RETURN_CONTINUE;
243  }
244 
245  // Already healed self or tank. If healer, do nothing else to anger mob.
246  if (m_ai.IsHealer())
247  return RETURN_NO_ACTION_OK; // In a sense, mission accomplished.
248 
249  // Have threat, can't quickly lower it. 3 options remain: Stop attacking, lowlevel damage (wand), keep on keeping on.
250  if (newTarget->GetHealthPercent() > 25)
251  {
252  // If elite, do nothing and pray tank gets aggro off you
253  // TODO: Is there an IsElite function? If so, find it and insert.
254  //if (newTarget->IsElite())
255  // return;
256 
257  // Not an elite. You could insert PSYCHIC SCREAM here but in any PvE situation that's 90-95% likely
258  // to worsen the situation for the group. ... So please don't.
259  return CastSpell(SHOOT, pTarget);
260  }
261  }
262 
263  // Dispel magic/disease
264  if (m_ai.HasDispelOrder() && DispelPlayer() & RETURN_CONTINUE)
265  return RETURN_CONTINUE;
266 
267  // Damage tweaking for healers
268  if (m_ai.IsHealer())
269  {
270  // Heal (try to pick a target by on common rules, than heal using each PlayerbotClassAI HealPlayer() method)
271  if (FindTargetAndHeal())
272  return RETURN_CONTINUE;
273  }
274  else
275  {
276  // Is this desirable? Debatable.
277  // ... Certainly could be very detrimental to a shadow priest
278  // TODO: In a group/raid with a healer you'd want this bot to focus on DPS (it's not specced/geared for healing either)
279  if (HealPlayer(&m_bot) & RETURN_CONTINUE)
280  return RETURN_CONTINUE;
281 
282  // No one needs to be healed: do small damage instead
283  // If target is elite and not handled by MT: do nothing
284  if (m_ai.IsElite(pTarget) && mainTank && mainTank->GetVictim() != pTarget)
285  return RETURN_NO_ACTION_OK;
286 
287  // Cast Shadow Word:Pain on current target and keep its up (if mana >= 40% or target HP < 15%)
288  if (SHADOW_WORD_PAIN > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_WORD_PAIN) && !pTarget->HasAura(SHADOW_WORD_PAIN, EFFECT_INDEX_0) &&
289  (pTarget->GetHealthPercent() < 15 || m_ai.GetManaPercent() >= 40) && CastSpell(SHADOW_WORD_PAIN, pTarget))
290  return RETURN_CONTINUE;
291  else // else shoot at it
292  return CastSpell(SHOOT, pTarget);
293  }
294 
295  // Damage Spells
296  switch (spec)
297  {
298  case PRIEST_SPEC_HOLY:
299  if (HOLY_FIRE > 0 && m_ai.In_Reach(pTarget, HOLY_FIRE) && !pTarget->HasAura(HOLY_FIRE, EFFECT_INDEX_0) && CastSpell(HOLY_FIRE, pTarget))

306

308  if (DEVOURING_PLAGUE > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, DEVOURING_PLAGUE) && !pTarget->HasAura(DEVOURING_PLAGUE, EFFECT_INDEX_0) && CastSpell(DEVOURING_PLAGUE, pTarget))
309  return RETURN_CONTINUE;

311  return RETURN_CONTINUE;
312  if (SHADOW_WORD_PAIN > 0 && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_WORD_PAIN) && !pTarget->HasAura(SHADOW_WORD_PAIN, EFFECT_INDEX_0) && CastSpell(SHADOW_WORD_PAIN, pTarget))
313  return RETURN_CONTINUE;

396 CombatManeuverReturns PlayerbotPriestAI::HealPlayer(Player* target)
397 {
398  CombatManeuverReturns r = PlayerbotClassAI::HealPlayer(target);
399  if (r != RETURN_NO_ACTION_OK)
400  return r;
401 
402  uint8 hp = target->GetHealthPercent();
403  uint8 hpSelf = m_ai.GetHealthPercent();
404 
405  if (hp >= 90)
406  return RETURN_NO_ACTION_OK;
407 
408  // If target is out of range (40 yards) and is a tank: move towards it
409  // if bot is not asked to stay
410  // Other classes have to adjust their position to the healers
411  // TODO: This code should be common to all healers and will probably
412  // move to a more suitable place like PlayerbotAI::DoCombatMovement()
413  if ((GetTargetJob(target) == JOB_TANK || GetTargetJob(target) == JOB_MAIN_TANK)
414  && m_bot.GetPlayerbotAI()->GetMovementOrder() != PlayerbotAI::MOVEMENT_STAY
