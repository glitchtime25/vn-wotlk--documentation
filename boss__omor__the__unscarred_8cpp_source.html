124  void HandlePullBack()
125  {
126  if (Player* player = m_creature->GetMap()->GetPlayer(m_playerGuid))
127  {
128  // if unit dosen't have this flag, then no pulling back (script will attempt cast, even if orbital strike was resisted)
129  if (player->HasMovementFlag(MOVEFLAG_FALLING))
130  DoCastSpellIfCan(player, SPELL_SHADOW_WHIP, CAST_INTERRUPT_PREVIOUS);
131  }
132  m_playerGuid.Clear();
133  }

142  void ExecuteAction(uint32 action) override
143  {
144  switch (action)
145  {
146  case OMOR_ENABLE_SHIELD:
147  {
148  if (m_creature->GetHealthPercent() < 20.0f)
149  {
150  SetActionReadyStatus(action, false);
151  ResetCombatAction(OMOR_DEMONIC_SHIELD, 1000);
152  }
153  break;
154  }
155  case OMOR_DEMONIC_SHIELD:
156  {
157  if (DoCastSpellIfCan(nullptr, SPELL_DEMONIC_SHIELD) == CAST_OK)
158  ResetCombatAction(action, 15000);
159  break;
160  }
161  case OMOR_FIENDISH_HOUND:
162  {
163  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_FIENDISH_HOUND) == CAST_OK)
164  ResetCombatAction(action, urand(24100, 26900));
165  break;
166  }
167  case OMOR_TREACHERY:
168  {
169  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_TREACHEROUS_AURA, SELECT_FLAG_PLAYER))
170  {
171  if (DoCastSpellIfCan(target, m_inRegularMode ? SPELL_TREACHEROUS_AURA : SPELL_BANE_OF_TREACHERY_H) == CAST_OK)
172  {
173  ResetCombatAction(action, urand(8000, 16000));
174  DoScriptText(SAY_CURSE, m_creature);
175  }
176  }
177  break;
178  }
179  case OMOR_ORBITAL_STRIKE:
180  {
181  Unit* target = nullptr;
182  if (m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
183  target = m_creature->GetVictim();
184  else
185  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_IN_MELEE_RANGE);
186 
187  if (target)
188  {
189  if (DoCastSpellIfCan(target, SPELL_ORBITAL_STRIKE) == CAST_OK)
190  {
191  ResetCombatAction(action, urand(14000, 16000));
192  m_playerGuid = target->GetObjectGuid();
193  }
194  }
195  break;
196  }
197  case OMOR_SHADOW_BOLT:
198  {
199  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 0, SPELL_SHADOW_BOLT, SELECT_FLAG_PLAYER))
200  {
201  if (!m_creature->CanReachWithMeleeAttack(target))
202  {
203  if (DoCastSpellIfCan(target, m_inRegularMode ? SPELL_SHADOW_BOLT : SPELL_SHADOW_BOLT_H) == CAST_OK)
204  ResetCombatAction(action, GetCurrentRangedMode() ? urand(3000, 4000) : urand(6000, 10000));
205  }
206  }
207  break;
208  }
209  }
210  }
