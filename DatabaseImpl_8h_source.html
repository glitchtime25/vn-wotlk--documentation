29 #define ASYNC_PQUERY_BODY(format, szQuery) \
30  if(!format) return false; \
31  \
32  char szQuery [MAX_QUERY_LEN]; \
33  \
34  { \
35  va_list ap; \
36  \
37  va_start(ap, format); \
38  int res = vsnprintf( szQuery, MAX_QUERY_LEN, format, ap ); \
39  va_end(ap); \
40  \
41  if(res==-1) \
42  { \
43  sLog.outError("SQL Query truncated (and not execute) for format: %s",format); \
44  return false; \
45  } \
46  }
