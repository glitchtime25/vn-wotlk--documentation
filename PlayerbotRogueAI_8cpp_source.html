72 CombatManeuverReturns PlayerbotRogueAI::DoFirstCombatManeuver(Unit* pTarget)
73 {
74  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
75  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
76  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
77  {
78  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
79  {
80  return RETURN_NO_ACTION_OK; // wait it out
81  }
82  else
83  {
84  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
85  }
86  }
87 
88  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
89  {
90  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
91  return RETURN_NO_ACTION_OK; // wait it out
92  else
93  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
94  }
95 
96  switch (m_ai.GetScenarioType())
97  {
98  case PlayerbotAI::SCENARIO_PVP_DUEL:
99  case PlayerbotAI::SCENARIO_PVP_BG:
100  case PlayerbotAI::SCENARIO_PVP_ARENA:
101  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
102  return DoFirstCombatManeuverPVP(pTarget);
103  case PlayerbotAI::SCENARIO_PVE:
104  case PlayerbotAI::SCENARIO_PVE_ELITE:
105  case PlayerbotAI::SCENARIO_PVE_RAID:
106  default:
107  return DoFirstCombatManeuverPVE(pTarget);
108  break;
109  }
110 
111  return RETURN_NO_ACTION_ERROR;
112 }

212  SpellSequence = RogueThreat;

214  SpellSequence = RogueCombat;

233  return RETURN_CONTINUE;
234 
235  // No appropriate action found, remove stealth
236  m_bot.RemoveSpellsCausingAura(SPELL_AURA_MOD_STEALTH);
237  return RETURN_CONTINUE;
238 
239  case RogueThreat:
240  if (GOUGE > 0 && !pTarget->HasAura(GOUGE, EFFECT_INDEX_0) && m_ai.CastSpell(GOUGE, *pTarget) == SPELL_CAST_OK)
241  return RETURN_CONTINUE;
242  if (EVASION > 0 && m_ai.GetHealthPercent() <= 35 && !m_bot.HasAura(EVASION, EFFECT_INDEX_0) && m_ai.CastSpell(EVASION) == SPELL_CAST_OK)

292  if (EXPOSE_ARMOR > 0 && !pTarget->HasAura(EXPOSE_ARMOR, EFFECT_INDEX_0) && m_ai.CastSpell(EXPOSE_ARMOR, *pTarget) == SPELL_CAST_OK) // 25 energy (checked above)

298

311  // default combo action for rogue/druid or if other combo action is unavailable/failed
312  // wait for energy
313  if (m_ai.GetEnergyAmount() < 35 && EVISCERATE)
314  return RETURN_NO_ACTION_OK;

385  if (weapon && weapon->GetEnchantmentId(TEMP_ENCHANTMENT_SLOT) == 0)
