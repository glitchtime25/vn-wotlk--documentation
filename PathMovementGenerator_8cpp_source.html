61 void AbstractPathMovementGenerator::Initialize(Unit& unit)
62 {
63  // Stop any previously dispatched splines no matter the source
64  if (!unit.movespline->Finalized())
65  {
66  if (unit.IsClientControlled())
67  unit.StopMoving(true);
68  else
69  unit.InterruptMoving();
70  }
71 
72  if (m_path.empty())
73  {
74  sLog.outError("AbstractPathMovementGenerator::Initialize Path empty for unit name %s entry %u dbguid %u.", unit.GetName(), unit.GetEntry(), unit.GetDbGuid());
75  return;
76  }
77 
78  // Reload current spline points array and end orientation from path:
79  m_spline.clear();
80  m_spline.push_back(Vector3()); // will be set to current position
81 
82  if (!m_cyclic)
83  {
84  for (auto itr = std::next(m_path.begin(), m_pathIndex); itr != m_path.end(); ++itr)
85  {
86  auto& node = (*itr).second;
87  m_spline.push_back({ node.x, node.y, node.z });
88 
89  if ((node.delay || std::next(itr, 1) == m_path.end()))
90  {
91  m_timer.Reset(node.delay);
92  m_orientation = node.orientation;
93  break;
94  }
95  }
96  }
97  else // start at last point and cycle through it all
98  {
99  uint32 count = 0;
100  for (auto itr = std::next(m_path.begin(), m_pathIndex); itr != m_path.end(); ++itr, ++count)
101  {
102  auto& node = (*itr).second;
103  m_spline.push_back({ node.x, node.y, node.z });
104  }
105  if (count < m_path.size())
106  {
107  for (auto itr = m_path.begin(); itr != m_path.end() && count < m_path.size(); ++itr, ++count)
108  {
109  auto& node = (*itr).second;
110  m_spline.push_back({ node.x, node.y, node.z });
111  }
112  }
113  }
114 
115  m_speedChanged = false;
116 }

140 bool AbstractPathMovementGenerator::Update(Unit& unit, const uint32& diff)
141 {
142  if (!m_spline.empty())
143  {
144  // Foolproofing if current spline was tampered with or ran out of splines in the path:
145  if (unit.movespline->FinalDestination() != m_spline.back())
146  {
147  if (unit.movespline->Finalized()) // stopped movement evidence
148  Initialize(unit);
149  else
150  return false;
151  }
152 
153  if (m_cyclic)
154  {
155  for (; m_pathIndex != unit.movespline->currentPathIdx();)
156  {
157  if (!m_firstCycle || m_pathIndex != 0)
158  MovementInform(unit);
159  m_pathIndex = (m_pathIndex + 1) % (m_path.size() + 1);
160  if (unit.movespline->Finalized())
161  return true;
162  }
163  if (m_pathIndex == m_startPoint)
164  m_firstCycle = false;
165  }
166  else
167  {
168  for (; m_pathIndex < unit.movespline->currentPathIdx(); ++m_pathIndex)
169  MovementInform(unit);
170  }
171 
172  if (!unit.movespline->Finalized())
173  {
174  if (m_speedChanged)
175  Initialize(unit);
176 
177  return true;
178  }
179  }
180 
181  m_timer.Update(diff);
182 
183  // Wait for timer:
184  if (!m_timer.Passed())
185  return true;
186 
187  // Load next spline from the path if still needed
188  if (size_t(m_pathIndex) < m_path.size())
189  {
190  Initialize(unit);
191  return true;
192  }
193  return false;
194 }

272 bool FixedPathMovementGenerator::Move(Unit& unit) const
273 {
274  Movement::MoveSplineInit init(unit);
275  init.MovebyPath(m_spline);
276  if (m_forcedMovement == FORCED_MOVEMENT_WALK)
277  init.SetWalk(true);
278  else if (m_forcedMovement == FORCED_MOVEMENT_RUN)
279  init.SetWalk(false);
280  else
281  init.SetWalk(!unit.hasUnitState(UNIT_STAT_RUNNING));
282  if (m_flying)
283  init.SetFly();
284  else
285  {
286  // non catmullrom
287  if (!unit.movespline->Finalized())
288  unit.UpdateSplinePosition(true);
289  Position pos = unit.GetPosition(unit.GetTransport());
290  init.Path()[0].x = pos.x; init.Path()[0].y = pos.y; init.Path()[0].z = pos.z;
291  if (!init.CheckBounds())
292  ERROR_DB_FILTER_LOG(LOG_FILTER_AI_AND_MOVEGENSS, "Path movement non-catmullrom pathId %u point %u for %s has invalid out of bounds spline. Likely meant to be split into several. (by script action or waittime)", unit.GetMotionMaster()->GetPathId(), m_pathIndex, unit.GetName());
293  }
294  if (m_orientation != 0.f)
295  init.SetFacing(m_orientation);
296  if (m_speed != 0.f)
297  init.SetVelocity(m_speed);
298  if (m_cyclic)
299  init.SetCyclic();
300  init.SetFirstPointId(m_pathIndex);
301  return bool(init.Launch());
302 }

354 bool TaxiMovementGenerator::Update(Unit& unit, const uint32& diff)
355 {
356  if (unit.GetTypeId() != TYPEID_PLAYER)
357  return (AbstractPathMovementGenerator::Update(unit, diff) || Resume(unit));
358 
359  Player& player = static_cast<Player&>(unit);

361  Movement::MoveSpline* spline = player.movespline;

406  uint32 i = 0;
