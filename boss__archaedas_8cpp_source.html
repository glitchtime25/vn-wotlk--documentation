99  void UpdateAI(const uint32 uiDiff) override
100  {
101  // so many things are based in this script on instance data
102  // so if we don't have access to it better do nothing
103  if (!m_pInstance)
104  return;
105 
106  // OOC Intro part triggered by Altar activation
107  if (m_pInstance->GetData(TYPE_ARCHAEDAS) == SPECIAL)
108  {
109  if (m_uiAwakeningTimer <= uiDiff)
110  {
111  switch (m_uiSubevent)
112  {
113  case 0:
114  DoCastSpellIfCan(m_creature, SPELL_ARCHAEDAS_AWAKEN_VISUAL);
115  m_uiAwakeningTimer = 2000;
116  break;
117  case 1:
118  DoScriptText(EMOTE_BREAKS_FREE, m_creature);
119  m_uiAwakeningTimer = 3000;
120  break;
121  case 2:
122  DoScriptText(SAY_AGGRO, m_creature);
123  m_creature->RemoveAurasDueToSpell(SPELL_FREEZE_ANIM);
124  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
125 
126  // Attack player
127  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_pInstance->GetGuid(DATA_EVENT_STARTER)))
128  AttackStart(pPlayer);
129  else
130  EnterEvadeMode();
131  break;
132  default:
133  break;
134  }
135 
136  ++m_uiSubevent;
137  }
138  else
139  m_uiAwakeningTimer -= uiDiff;
140  }
141 
142  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
143  return;
144 
145  // Phase switch
146  if (m_creature->GetHealthPercent() < 100.0f - 33.4f * (float)m_uiHpPhaseCheck)
147  {
148  if (DoCastSpellIfCan(m_creature, m_uiHpPhaseCheck == 1 ? SPELL_AWAKEN_EARTHEN_GUARDIAN : SPELL_AWAKEN_VAULT_WARDER) == CAST_OK)
149  {
150  DoScriptText(m_uiHpPhaseCheck == 1 ? SAY_AWAKE_GUARDIANS : SAY_AWAKE_WARDERS, m_creature);
151  ++m_uiHpPhaseCheck;
152  }
153  }
154 
155  // Awake random Dwarf
156  if (!m_bDwarvesAwaken && m_creature->GetHealthPercent() >= 33.0f)
157  {
158  if (m_uiAwakeDwarfTimer < uiDiff)
159  {
160  if (Creature* pEarthen = m_pInstance->GetClosestDwarfNotInCombat(m_creature))
161  {
162  if (DoCastSpellIfCan(pEarthen, SPELL_AWAKEN_EARTHEN_DWARF) == CAST_OK)
163  m_uiAwakeDwarfTimer = urand(9000, 12000);
164  }
165  else
166  m_bDwarvesAwaken = true;
167  }
168  else
169  m_uiAwakeDwarfTimer -= uiDiff;
170  }
171 
172  if (m_uiTremorTimer < uiDiff)
173  {
174  if (DoCastSpellIfCan(m_creature, SPELL_GROUND_TREMOR) == CAST_OK)
175  m_uiTremorTimer = urand(8000, 17000);
176  }
177  else
178  m_uiTremorTimer -= uiDiff;
179 
180  DoMeleeAttackIfReady();
181  }

189 bool EffectDummyCreature_npc_vault_warder(Unit* /*pCaster*/, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
190 {
191  // always check spellid and effectindex
192  if (uiSpellId == SPELL_AWAKEN_VAULT_WARDER && uiEffIndex == EFFECT_INDEX_0)
193  {
194  if (pCreatureTarget->GetEntry() == NPC_VAULT_WARDER)
195  {
196  pCreatureTarget->RemoveAurasDueToSpell(SPELL_STONED);
197 
198  ScriptedInstance* pInstance = (ScriptedInstance*)pCreatureTarget->GetInstanceData();
199  if (!pInstance)
200  return true;
201 
202  if (Creature* pArchaedas = pInstance->GetSingleCreatureFromStorage(NPC_ARCHAEDAS))
203  pCreatureTarget->AI()->AttackStart(pArchaedas->GetVictim());
204 
205  return true;
206  }
207  }
208 
209  return false;
210 }
