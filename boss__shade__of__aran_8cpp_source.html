285  void HandleDrinking()
286  {
287  uint32 timer = 0;
288  switch (m_uiManaRecoveryStage)
289  {
290  case 0:
291  DoCastSpellIfCan(nullptr, SPELL_CONJURE_WATER);
292  timer = 2000;
293  break;
294  case 1:
295  DoCastSpellIfCan(nullptr, SD_SPELL_DRINK);
296  m_creature->SetStandState(UNIT_STAND_STATE_SIT);
297  timer = 5000;
298  break;
299  case 2:
300  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
301  timer = 1000;
302  break;
303  case 3:
304  if (DoCastSpellIfCan(nullptr, SPELL_PYROBLAST) == CAST_OK)
305  {
306  SetMeleeEnabled(true);
307  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
308  SetCombatMovement(true);
309  SetCombatScriptStatus(false);
310  }
311  break;
312  }
313  ++m_uiManaRecoveryStage;
314  if (timer)
315  ResetTimer(ARAN_DRINKING_STAGES, timer);
316  }

318  void ExecuteAction(uint32 action) override
319  {
320  switch (action)
321  {
322  case ARAN_ACTION_DRINK:
323  if (m_creature->GetPowerType() == POWER_MANA && (m_creature->GetPower(POWER_MANA) * 100 / m_creature->GetMaxPower(POWER_MANA)) < 20)
324  {
325  if (DoCastSpellIfCan(nullptr, SPELL_MASS_POLYMORPH) == CAST_OK)
326  {
327  m_creature->CastSpell(nullptr, SPELL_DISPEL_BLIZZARD, TRIGGERED_OLD_TRIGGERED);
328  DoScriptText(SAY_DRINK, m_creature);
329  SetCombatMovement(false);
330  SetCombatScriptStatus(true);
331  SetMeleeEnabled(false);
332  m_creature->SetTarget(nullptr);
333  m_uiManaRecoveryStage = 0;
334  ResetTimer(ARAN_DRINKING_STAGES, 2000);
335  m_bDrinkInterrupted = false;
336  SetActionReadyStatus(action, false);
337  }
338  }
339  return;
340  case ARAN_ACTION_POTION:
341  {
342  if (m_creature->GetPowerPercent() < 3.f) // always drink when low
343  DoCastSpellIfCan(m_creature, SPELL_MANA_POTION);
344  break;
345  }
346  case ARAN_ACTION_ELEMENTALS:
347  {
348  if (m_creature->GetHealthPercent() > 40.0f)
349  return;
350 
351  DoCastSpellIfCan(nullptr, SPELL_SUMMON_WATER_ELEM_1, CAST_TRIGGERED);
352  DoCastSpellIfCan(nullptr, SPELL_SUMMON_WATER_ELEM_2, CAST_TRIGGERED);
353  DoCastSpellIfCan(nullptr, SPELL_SUMMON_WATER_ELEM_3, CAST_TRIGGERED);
354  DoCastSpellIfCan(nullptr, SPELL_SUMMON_WATER_ELEM_4, CAST_TRIGGERED);
355 
356  DoScriptText(SAY_ELEMENTALS, m_creature);
357 
358  SetActionReadyStatus(action, false);
359  return;
360  }
361  case ARAN_ACTION_BERSERK:
362  {
363  for (ObjectGuid guid : m_instance->GetAranTeleportNPCs())
364  if (Creature* teleport = m_creature->GetMap()->GetCreature(guid))
365  {
366  teleport->UpdateEntry(NPC_SHADOW_OF_ARAN);
367  teleport->AIM_Initialize();
368  teleport->SetInCombatWithZone();
369  }
370 
371  DoScriptText(SAY_TIMEOVER, m_creature);
372  DisableCombatAction(action);
373  return;
374  }
375 
376  case ARAN_ACTION_DRAGONS_BREATH:
377  {
378  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), (SELECT_FLAG_PLAYER | SELECT_FLAG_IN_MELEE_RANGE)))
379  {
380  DoCastSpellIfCan(target, SPELL_DRAGONS_BREATH, CAST_TRIGGERED);
381  DisableCombatAction(action);
382  DelayCombatAction(ARAN_ACTION_SUPERSPELL, 6000); // Duration
383  }
384  return;
385  }
386  case ARAN_ACTION_SUPERSPELL:
387  {
388  uint8 uiAvailableSpell = urand(SUPER_FLAME_WREATH, SUPER_ARCANE_EXPL);
389 
390  // randomize so it won't be the same spell twice in a row
391  while (uiAvailableSpell == m_uiLastSuperSpell)
392  uiAvailableSpell = urand(SUPER_FLAME_WREATH, SUPER_ARCANE_EXPL);
393 
394  m_uiLastSuperSpell = uiAvailableSpell;
395 
396  DoCastSpellIfCan(m_creature, SPELL_BLINK_CENTER, CAST_TRIGGERED);
397 
398  switch (m_uiLastSuperSpell)
399  {
400  case SUPER_ARCANE_EXPL:
401  DoCastSpellIfCan(m_creature, SPELL_MASSIVE_MAGNETIC_PULL, CAST_TRIGGERED);
402  DoScriptText(urand(0, 1) ? SAY_EXPLOSION1 : SAY_EXPLOSION2, m_creature);
403  DoCastSpellIfCan(m_creature, SPELL_MASS_SLOW, CAST_TRIGGERED);
404  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_EXPLOSION) == CAST_OK)
405  DoScriptText(SAY_EXPLOSION_EMOTE, m_creature);
406  break;
407  case SUPER_FLAME_WREATH:
408  if (DoCastSpellIfCan(m_creature, SPELL_FLAME_WREATH) == CAST_OK)
409  {
410  DoScriptText(urand(0, 1) ? SAY_FLAMEWREATH1 : SAY_FLAMEWREATH2, m_creature);
411 #ifdef PRENERF_2_0_3
412  ResetCombatAction(ARAN_ACTION_DRAGONS_BREATH, 27000);
413 #endif
414  }
415  break;
416  case SUPER_BLIZZARD:
417  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_BLIZZARD) == CAST_OK)
418  DoScriptText(urand(0, 1) ? SAY_BLIZZARD1 : SAY_BLIZZARD2, m_creature);
419  break;
420  }
421  ResetCombatAction(action, 30000);
422  return;
423  }
424  case ARAN_ACTION_SECONDARY_SPELL:
425  {
426  CanCastResult spellResult = CAST_FAIL_OTHER;
427 
428  switch (urand(0, 1))
429  {
430  case 0:
431  spellResult = DoCastSpellIfCan(nullptr, SPELL_COUNTERSPELL);
432  break;
433  case 1:
434  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
435  spellResult = DoCastSpellIfCan(target, SPELL_CHAINS_OF_ICE);
436  break;
437  }
438  if (spellResult == CAST_OK)
439  ResetCombatAction(action, urand(5000, 20000));
440  return;
441  }
442  case ARAN_ACTION_PRIMARY_SPELL:
443  {
444  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER);
445  if (!target)
446  break;
447 
448  m_choiceVector.clear();
449 
450  for (uint32 i = 0; i < NORMAL_SPELL_COUNT; ++i)
451  {
452  uint32 spellId = GetNormalSpellId(i);
453  if (m_creature->IsSpellReady(spellId))
454  m_choiceVector.push_back(i);
455  }
456 
457  if (m_choiceVector.size() != 0)
458  {
459  uint32 currentSpellIndex = urand(0, m_choiceVector.size() - 1);
460  uint32 currentSpellId = GetNormalSpellId(currentSpellIndex);
461  if (DoCastSpellIfCan(target, currentSpellId) == CAST_OK)
462  ResetCombatAction(action, 2000);
463  }
464  }
465  }
466  }
