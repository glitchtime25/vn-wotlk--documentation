146 enum ScriptInfoDataFlags
147 {
148  // default: s/b -> t
149  SCRIPT_FLAG_BUDDY_AS_TARGET = 0x001, // s -> b
150  SCRIPT_FLAG_REVERSE_DIRECTION = 0x002, // t* -> s* (* result after previous flag is evaluated)
151  SCRIPT_FLAG_SOURCE_TARGETS_SELF = 0x004, // s* -> s* (* result after previous flag is evaluated)
152  SCRIPT_FLAG_COMMAND_ADDITIONAL = 0x008, // command dependend
153  SCRIPT_FLAG_BUDDY_BY_GUID = 0x010, // take the buddy by guid
154  SCRIPT_FLAG_BUDDY_IS_PET = 0x020, // buddy is a pet
155  SCRIPT_FLAG_BUDDY_IS_DESPAWNED = 0x040, // buddy is dead or despawned
156  SCRIPT_FLAG_BUDDY_BY_POOL = 0x080, // buddy should be part of a pool
157  SCRIPT_FLAG_BUDDY_BY_SPAWN_GROUP = 0x100, // buddy is from spawn group
158  SCRIPT_FLAG_ALL_ELIGIBLE_BUDDIES = 0x200, // multisource/multitarget - will execute for each eligible
159  SCRIPT_FLAG_BUDDY_BY_GO = 0x400, // take the buddy by GO (for commands which can target both creature and GO)
160  SCRIPT_FLAG_BUDDY_BY_STRING_ID = 0x800, // takes buddy from string id - creature or go
