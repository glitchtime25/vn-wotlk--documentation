136  void DespawnGolems()
137  {
138  if (m_lGolemGUIDList.empty())
139  return;
140 
141  for (GuidList::const_iterator itr = m_lGolemGUIDList.begin(); itr != m_lGolemGUIDList.end(); ++itr)
142  {
143  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
144  {
145  if (pTemp->IsAlive())
146  pTemp->ForcedDespawn();
147  }
148  }
149  }

151  void ShatterGolems()
152  {
153  if (m_lGolemGUIDList.empty())
154  return;
155 
156  uint8 m_uiBrittleGolemsCount = 0;
157 
158  for (GuidList::const_iterator itr = m_lGolemGUIDList.begin(); itr != m_lGolemGUIDList.end(); ++itr)
159  {
160  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
161  {
162  // only shatter brittle golems
163  if (pTemp->GetEntry() == NPC_BRITTLE_GOLEM)
164  {
165  pTemp->CastSpell(pTemp, m_bIsRegularMode ? SPELL_SHATTER : SPELL_SHATTER_H, TRIGGERED_OLD_TRIGGERED);
166  ++m_uiBrittleGolemsCount;
167  }
168  }
169  }
170 
171  // If shattered more than 4 golems mark achiev as failed
172  if (m_uiBrittleGolemsCount > MAX_ACHIEV_GOLEMS)
173  {
174  if (m_pInstance)
175  m_pInstance->SetData(TYPE_VOLKHAN, SPECIAL);
176  }
177  }

200  void UpdateAI(const uint32 uiDiff) override
201  {
202  // Return since we have no target
203  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
204  return;
205 
206  // he shatters only one time, at 25%
207  if (m_creature->GetHealthPercent() <= 25.0f && !m_bHasShattered)
208  {
209  // should he stomp even if he has no brittle golem to shatter? <-yes!
210  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SHATTERING_STOMP : SPELL_SHATTERING_STOMP_H) == CAST_OK)
211  {
212  DoScriptText(urand(0, 1) ? SAY_STOMP_1 : SAY_STOMP_2, m_creature);
213  DoScriptText(EMOTE_SHATTER, m_creature);
214  m_uiShatterTimer = 3000;
215  m_bHasShattered = true;
216  }
217  }
218 
219  // Shatter Golems 3 seconds after Shattering Stomp
220  if (m_uiShatterTimer)
221  {
222  if (m_uiShatterTimer <= uiDiff)
223  {
224  ShatterGolems();
225  m_uiShatterTimer = 0;
226  }
227  else
228  m_uiShatterTimer -= uiDiff;
229  }
230 
231  // Summon Golems only when over 25% hp
232  if (m_creature->GetHealthPercent() > 25.0f)
233  {
234  if (m_uiTemperTimer < uiDiff)
235  {
236  DoScriptText(EMOTE_TO_ANVIL, m_creature);
237  DoScriptText(urand(0, 1) ? SAY_FORGE_1 : SAY_FORGE_2, m_creature);
238  SetCombatMovement(false);
239 
240  if (m_pInstance)
241  {
242  if (Creature* pAnvil = m_pInstance->GetSingleCreatureFromStorage(NPC_VOLKHAN_ANVIL))
243  {
244  float fX, fY, fZ;
245  pAnvil->GetContactPoint(m_creature, fX, fY, fZ, INTERACTION_DISTANCE);
246  m_creature->SetWalk(false);
247  m_creature->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
248  }
249  else
250  script_error_log("Npc %u couldn't be found or something really bad happened.", NPC_VOLKHAN_ANVIL);
251  }
252  m_uiTemperTimer = 30000;
253  }
254  else
255  m_uiTemperTimer -= uiDiff;
256  }
257 
258  if (m_uiHeatTimer < uiDiff)
259  {
260  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_HEAT : SPELL_HEAT_H) == CAST_OK)
261  m_uiHeatTimer = urand(10000, 15000);
262  else
263  m_uiHeatTimer = urand(3000, 5000);
264  }
265  else
266  m_uiHeatTimer -= uiDiff;
267 
268  DoMeleeAttackIfReady();
269  }
