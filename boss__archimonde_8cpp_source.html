254  void ExecuteAction(uint32 action) override
255  {
256  switch (action)
257  {
258  case ARCHIMONDE_ACTION_PHASE_2:
259  {
260  if (m_creature->GetHealthPercent() > 10.0f)
261  return;
262 
263  DoScriptText(SAY_EPILOGUE, m_creature);
264  m_creature->PlayDirectSound(10992);
265  m_creature->PlayDirectSound(10843);
266  // move at home position and start outro
267  SetCombatScriptStatus(true);
268  m_creature->GetMotionMaster()->MovePoint(POINT_EPILOGUE, outroPoint[0], outroPoint[1], outroPoint[2], FORCED_MOVEMENT_RUN);
269  SetCombatMovement(false);
270  m_creature->SetActiveObjectState(true);
271  SetMeleeEnabled(false);
272  m_phase = true;
273  return;
274  }
275  case ARCHIMONDE_ACTION_HAND_OF_DEATH:
276  if (DoCastSpellIfCan(nullptr, SPELL_HAND_OF_DEATH) == CAST_OK)
277  {
278  if (m_leashCheckTimer != 0)
279  DoScriptText(SAY_ENRAGE, m_creature);
280  ResetCombatAction(action, GetSubsequentActionTimer(action));
281  EnterEvadeMode();
282  }
283  return;
284  case ARCHIMONDE_ACTION_GRIP_OF_THE_LEGION:
285  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
286  if (DoCastSpellIfCan(target, SPELL_GRIP_OF_THE_LEGION) == CAST_OK)
287  ResetCombatAction(action, GetSubsequentActionTimer(action));
288  return;
289  case ARCHIMONDE_ACTION_AIR_BURST:
290  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK))
291  if (DoCastSpellIfCan(target, SPELL_AIR_BURST) == CAST_OK)
292  DoScriptText(urand(0, 1) ? SAY_AIR_BURST1 : SAY_AIR_BURST2, m_creature);
293  return;
294  case ARCHIMONDE_ACTION_FEAR:
295  if (DoCastSpellIfCan(nullptr, SPELL_FEAR) == CAST_OK)
296  ResetCombatAction(action, GetSubsequentActionTimer(action));
297  return;
298  case ARCHIMONDE_ACTION_DOOMFIRE:
299  if (DoCastSpellIfCan(nullptr, SPELL_DOOMFIRE_STRIKE) == CAST_OK)
300  {
301  DoScriptText(urand(0, 1) ? SAY_DOOMFIRE1 : SAY_DOOMFIRE2, m_creature);
302  ResetCombatAction(action, GetSubsequentActionTimer(action));
303  }
304  return;
305  case ARCHIMONDE_ACTION_FINGER_OF_DEATH_COOLUP:
306  if (Unit* closest = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, nullptr, SELECT_FLAG_PLAYER))
307  {
308  if (!m_creature->CanReachWithMeleeAttack(closest))
309  {
310  if (DoCastSpellIfCan(nullptr, SPELL_FINGER_OF_DEATH_COOLUP) == CAST_OK)
311  {
312  ResetCombatAction(action, GetSubsequentActionTimer(action));
313  SetActionReadyStatus(ARCHIMONDE_ACTION_FINGER_OF_DEATH, true); // when GCD expires, cast this
314  }
315  }
316  }
317  return;
318  case ARCHIMONDE_ACTION_FINGER_OF_DEATH:
319  if (Unit* closest = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, nullptr, SELECT_FLAG_PLAYER))
320  {
321  if (!m_creature->CanReachWithMeleeAttack(closest))
322  {
323  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
324  {
325  if (DoCastSpellIfCan(target, SPELL_FINGER_OF_DEATH) == CAST_OK)
326  {
327  SetActionReadyStatus(action, false);
328  return;
329  }
330  }
331  }
332  else SetActionReadyStatus(action, false); // if someone is tanking again, disable
333  }
334  return;
335  case ARCHIMONDE_ACTION_SOUL_CHARGE:
336  uint32 spellId = 0;
337  if (m_creature->HasAura(SPELL_SOUL_CHARGE_YELLOW_CHARGE))
338  spellId = SPELL_SOUL_CHARGE_YELLOW;
339  else if (m_creature->HasAura(SPELL_SOUL_CHARGE_RED_CHARGE))
340  spellId = SPELL_SOUL_CHARGE_RED;
341  else if (m_creature->HasAura(SPELL_SOUL_CHARGE_GREEN_CHARGE))
342  spellId = SPELL_SOUL_CHARGE_GREEN;
343  ResetCombatAction(action, GetSubsequentActionTimer(action)); // does not seem to reset cd only on spell cast
344  if (spellId)
345  if (DoCastSpellIfCan(nullptr, spellId) == CAST_OK)
346  return;
347  return;
348  }
349  }

351  void HandleOutro(const uint32 diff)
352  {
353  // Spam Finger of Death on players and Wisps
354  if (m_stageCounter)
355  {
356  uint32 spellId;
357  switch (m_stageCounter) // First two always guaranteed
358  {
359  case 0: spellId = SPELL_FINGER_DEATH_SCRIPT; break;
360  case 1: spellId = SPELL_FINGER_DEATH_DUMMY; break;
361  default: spellId = urand(0, 2) ? SPELL_FINGER_DEATH_SCRIPT : SPELL_FINGER_DEATH_DUMMY; break;
362  }
363  if (DoCastSpellIfCan(nullptr, spellId) == CAST_OK)
364  ++m_stageCounter;
365  }
366 
367  if (m_summonWispTimer)
368  {
369  if (m_summonWispTimer <= diff)
370  {
371  for (uint32 i = 0; i < 4; ++i)
372  m_creature->CastSpell(nullptr, SPELL_SUMMON_WISP, TRIGGERED_OLD_TRIGGERED);
373  m_summonWispTimer = 2400;
374  }
375  else
376  m_summonWispTimer -= diff;
377  }
378 
379  if (m_outroTimer)
380  {
381  if (m_outroTimer <= diff)
382  {
383  m_creature->CastSpell(nullptr, SPELL_SUICIDE_ARCHIMONDE, TRIGGERED_NONE);
384  m_creature->CastSpell(nullptr, SPELL_DENOUEMENT_WISP, TRIGGERED_NONE);
385  m_outroTimer = 0;
386  return;
387  }
388  else
389  m_outroTimer -= diff;
390  }
391  }

393  void CheckLeash(const uint32 diff)
394  {
395  if (m_leashCheckTimer)
396  {
397  if (m_leashCheckTimer <= diff)
398  {
399  Position pos;
400  m_creature->GetCombatStartPosition(pos);
401  // Range stuff here
402  if (m_creature->GetDistance2d(pos.GetPositionX(), pos.GetPositionY()) >= 155.f || (pos.GetPositionX() < 5534.752f && pos.GetPositionX() > 5381.371f &&
403  pos.GetPositionY() < -3507.099f && pos.GetPositionY() > -3587.244f))
404  {
405  SetActionReadyStatus(ARCHIMONDE_ACTION_HAND_OF_DEATH, true);
406  m_leashCheckTimer = 0;
407  DoScriptText(SAY_SLAY3, m_creature); // seems to be both used as leash yell and slay yell
408  }
409  else m_leashCheckTimer = 2000;
410  }
411  else m_leashCheckTimer -= diff;
412  }

415  void UpdateAI(const uint32 diff) override
416  {
417  // Intro timer
418  if (m_drainNordrassilTimer)
419  {
420  if (m_drainNordrassilTimer <= diff)
421  {
422  m_creature->RemoveAurasDueToSpell(SPELL_SOUL_CHARGE_YELLOW_CHARGE); // workaround for archi not despawning after wipe
423  m_creature->RemoveAurasDueToSpell(SPELL_SOUL_CHARGE_RED_CHARGE);
424  m_creature->RemoveAurasDueToSpell(SPELL_SOUL_CHARGE_GREEN_CHARGE);
425  if (DoCastSpellIfCan(nullptr, SPELL_DRAIN_TREE) == CAST_OK)
426  {
427  if (!m_hasIntro)
428  {
429  DoScriptText(SAY_INTRO, m_creature);
430  m_hasIntro = true;
431  }
432  m_drainNordrassilTimer = 0;
433  }
434  }
435  else
436  m_drainNordrassilTimer -= diff;
437  }
438 
439  UpdateTimers(diff, m_creature->IsInCombat());
440 
441  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
442  return;
443 
444  if (!m_phase) // Phase 1
445  {
446  CheckLeash(diff);
447  ExecuteActions();
448  DoMeleeAttackIfReady();
449  }
450  else // Phase2
451  HandleOutro(diff);
452  }
