24 bool BuildRawData(const std::string &hexData, std::vector<uint8> &out)
25 {
26  if (!!(hexData.length() % 2))
27  return false;
28 
29  auto curr = hexData;
30 
31  out.resize(hexData.length() / 2);
32 
33  for (auto i = 0; i < out.size(); ++i)
34  {
35  auto const byte = curr.substr(0, 2);
36  curr = curr.substr(2);
37 
38  auto const a = byte[0];
39  auto const b = byte[1];
40 
41  unsigned int an;
42  if (a >= '0' && a <= '9')
43  an = a - '0';
44  else if (a >= 'A' && a <= 'F')
45  an = 10 + a - 'A';
46  else
47  return false;
48 
49  unsigned int bn;
50  if (b >= '0' && b <= '9')
51  bn = b - '0';
52  else if (b >= 'A' && b <= 'F')
53  bn = 10 + b - 'A';
54  else
55  return false;
56 
57  out[i] = static_cast<uint8>(an * 16 + bn);
58  }
59 
60  return true;
61 }

173 
174  case API_CHECK:

183  continue;
184  }
185 
186  scan = new WindowsHookScan(module, proc, &hash[0], offset, length, comment, flags);
187  break;
188  }
189 
190  case FIND_DRIVER_BY_NAME:
191  {
192  auto const driver = fields[2].GetCppString();
193  auto const path = fields[3].GetCppString();
194  auto const wanted = fields[6].GetBool();
195 
196  scan = new WindowsDriverScan(driver, path, wanted, comment, flags);

198  }
199 
200  default:

202  sLog.outError("Unhandled Warden scan type %u id %u. Skipped.", scanType, id);
203  continue;
204  }
205  }
206 
207  if (!scan)
208  {
209  sLog.outError("Failed to allocate Warden scan type %u id %u", scanType, id);
210  continue;
211  }
212 
213  m_scans.emplace_back(std::shared_ptr<const Scan>(scan));
214  } while (result->NextRow());
215 
216  sLog.outBasic(">> %lu Warden scans loaded from world database", uint64(m_scans.size()));

239 std::vector<std::shared_ptr<const Scan>> WardenScanMgr::GetRandomScans(ScanFlags flags) const
240 {
241  std::vector<std::shared_ptr<const Scan>> matches;
242  matches.reserve(m_scans.size());
243 
244  // save those scans which match the requested flags
245  for (auto const &scan : m_scans)
246  {
247  auto const buildMask = scan->flags & (WinAllBuild | MacAllBuild);
248 
249  // does the build part of the flags match? if not, continue
250  if (!(flags & buildMask))
251  continue;
252 
253  // is the scan considered advanced but advanced scans were not requested? if so, continue
254  if ((scan->flags & AdvancedScan) && !(flags & AdvancedScan))
255  continue;
256 
257  // if the scan is an initial-login scan, and that wasn't explicitly requested, do not include it
258  if ((scan->flags & InitialLogin) && !(flags & InitialLogin))
259  continue;
260 
261  // if the scan is not an initial-login scan, and that was explicitly requested, do not include it
262  if (!(scan->flags & InitialLogin) && (flags & InitialLogin))
263  continue;
264 
265  // if the scan is an in-world scan, and that wasn't explicitly requested, do not include it
266  if (!!(scan->flags & InWorld) && !(flags & InWorld))
267  continue;
268 
269  matches.push_back(scan);
270  }
271 
272  // randomize the order of matching scans
273  std::shuffle(matches.begin(), matches.end(), *GetRandomGenerator());
274 
275  // determine how many of the identified scans we can fit into the client's request and response buffers
276  size_t request = 0, reply = 0;
277 
278  // if we have matched too many, shrink. this will probably be shrunk even further, depending on buffer sizes
279  if (matches.size() >= sAnticheatConfig.GetWardenScanCount())
280  matches.resize(sAnticheatConfig.GetWardenScanCount());
281 
282  for (auto i = 0u; i < matches.size(); ++i)
283  {
284  auto const &scan = matches[i];
285 
286  // if by including the current scan, the request or reply would become too large, stop here, shrinking the results to size
287  if (request + scan->requestSize > Warden::MaxRequest || reply + scan->replySize > Warden::MaxReply)
288  {
289  matches.resize(i);
290  break;
291  }
292 
293  request += scan->requestSize;
294  reply += scan->replySize;
295  }
296 
297  return matches;
298 }
