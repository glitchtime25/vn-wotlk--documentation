41 void AuctionHouseBot::Initialize()
42 {
43  if (!m_ahBotCfg.SetSource(m_configFileName))
44  {
45  // set buy/sell chance to 0, this prevents Update() from accessing uninitialized variables
46  m_chanceBuy = 0;
47  m_chanceSell = 0;
48  sLog.outString("AHBot is disabled. Unable to open configuration file(%s).", m_configFileName.c_str());
49  return;
50  }
51  sLog.outString("AHBot using configuration file %s", m_configFileName.c_str());
52 
53  m_chanceSell = GetMinMaxConfig("AuctionHouseBot.Chance.Sell", 0, 100, 10);
54  m_chanceBuy = GetMinMaxConfig("AuctionHouseBot.Chance.Buy", 0, 100, 10);
55 
56  sLog.outString("AHBot selling items: %s", m_chanceSell > 0 ? "Enabled" : "Disabled");
57  sLog.outString("AHBot buying items: %s", m_chanceBuy > 0 ? "Enabled" : "Disabled");
58 
59  if (m_chanceSell > 0 || m_chanceBuy > 0)
60  {
61  // creature loot
62  ParseLootConfig("AuctionHouseBot.Loot.Creature.Normal", m_creatureLootNormalConfig);
63  ParseLootConfig("AuctionHouseBot.Loot.Creature.Elite", m_creatureLootEliteConfig);
64  ParseLootConfig("AuctionHouseBot.Loot.Creature.RareElite", m_creatureLootRareEliteConfig);
65  ParseLootConfig("AuctionHouseBot.Loot.Creature.WorldBoss", m_creatureLootWorldBossConfig);
66  ParseLootConfig("AuctionHouseBot.Loot.Creature.Rare", m_creatureLootRareConfig);
67  FillUintVectorFromQuery("SELECT entry FROM creature_template WHERE `rank` = 0 AND entry IN (SELECT entry FROM creature_loot_template)", m_creatureLootNormalTemplates);
68  FillUintVectorFromQuery("SELECT entry FROM creature_template WHERE `rank` = 1 AND entry IN (SELECT entry FROM creature_loot_template)", m_creatureLootEliteTemplates);
69  FillUintVectorFromQuery("SELECT entry FROM creature_template WHERE `rank` = 2 AND entry IN (SELECT entry FROM creature_loot_template)", m_creatureLootRareEliteTemplates);
70  FillUintVectorFromQuery("SELECT entry FROM creature_template WHERE `rank` = 3 AND entry IN (SELECT entry FROM creature_loot_template)", m_creatureLootWorldBossTemplates);
71  FillUintVectorFromQuery("SELECT entry FROM creature_template WHERE `rank` = 4 AND entry IN (SELECT entry FROM creature_loot_template)", m_creatureLootRareTemplates);
72 
73  // disenchant loot
74  ParseLootConfig("AuctionHouseBot.Loot.Disenchant", m_disenchantLootConfig);
75  FillUintVectorFromQuery("SELECT DISTINCT entry FROM disenchant_loot_template", m_disenchantLootTemplates);
76 
77  // fishing loot
78  ParseLootConfig("AuctionHouseBot.Loot.Fishing", m_fishingLootConfig);
79  FillUintVectorFromQuery("SELECT DISTINCT entry FROM fishing_loot_template", m_fishingLootTemplates);
80 
81  // gameobject loot
82  ParseLootConfig("AuctionHouseBot.Loot.Gameobject", m_gameobjectLootConfig);
83  FillUintVectorFromQuery("SELECT DISTINCT entry FROM gameobject_loot_template WHERE entry IN (SELECT data1 FROM gameobject_template WHERE entry IN (SELECT id FROM gameobject WHERE spawntimesecsmax > 0))", m_gameobjectLootTemplates);
84 
85  // skinning loot
86  ParseLootConfig("AuctionHouseBot.Loot.Skinning", m_skinningLootConfig);
87  FillUintVectorFromQuery("SELECT DISTINCT entry FROM skinning_loot_template", m_skinningLootTemplates);
88 
89  // profession items (different than the loot above, but use similar config)
90  ParseLootConfig("AuctionHouseBot.Items.Profession", m_professionItemsConfig);
91  FillUintVectorFromQuery("SELECT entry FROM item_template WHERE entry IN (SELECT EffectItemType1 FROM spell_template WHERE attributes & 32 AND attributes & 65536)", m_professionItems);
92 
93  // vendor items (used to prevent items being bought from vendor and sold at ah for profit)
94  std::vector<uint32> tmpVector;
95  FillUintVectorFromQuery("SELECT item FROM npc_vendor", tmpVector);
96  std::copy(tmpVector.begin(), tmpVector.end(), std::inserter(m_vendorItems, m_vendorItems.end()));
97 
98  // item value
99  ParseItemValueConfig("AuctionHouseBot.Value.Poor", m_itemValue[ITEM_QUALITY_POOR]);
100  ParseItemValueConfig("AuctionHouseBot.Value.Normal", m_itemValue[ITEM_QUALITY_NORMAL]);
101  ParseItemValueConfig("AuctionHouseBot.Value.Uncommon", m_itemValue[ITEM_QUALITY_UNCOMMON]);
102  ParseItemValueConfig("AuctionHouseBot.Value.Rare", m_itemValue[ITEM_QUALITY_RARE]);
103  ParseItemValueConfig("AuctionHouseBot.Value.Epic", m_itemValue[ITEM_QUALITY_EPIC]);
104  ParseItemValueConfig("AuctionHouseBot.Value.Legendary", m_itemValue[ITEM_QUALITY_LEGENDARY]);
105  ParseItemValueConfig("AuctionHouseBot.Value.Artifact", m_itemValue[ITEM_QUALITY_ARTIFACT]);
106 
107  // item value for items sold by vendors
108  m_vendorValue = m_ahBotCfg.GetBoolDefault("AuctionHouseBot.Value.Vendor", true);
109 
110  // item value variance
111  m_valueVariance = GetMinMaxConfig("AuctionHouseBot.Value.Variance", 0, 100, 10);
112 
113  // auction min/max bid
114  m_auctionBidMin = GetMinMaxConfig("AuctionHouseBot.Bid.Min", 0, 100, 75);
115  m_auctionBidMax = GetMinMaxConfig("AuctionHouseBot.Bid.Max", 0, 100, 90);
116  if (m_auctionBidMin > m_auctionBidMax)
117  {
118  sLog.outError("AHBot error: AuctionHouseBot.Bid.Min must be less or equal to AuctionHouseBot.Bid.Max. Setting Bid.Min equal to Bid.Max.");
119  m_auctionBidMin = m_auctionBidMax;
120  }
121 
122  // auction min/max time
123  m_auctionTimeMin = GetMinMaxConfig("AuctionHouseBot.Time.Min", 1, 72, 2);
124  m_auctionTimeMax = GetMinMaxConfig("AuctionHouseBot.Time.Max", 1, 72, 24);
125  if (m_auctionTimeMin > m_auctionTimeMax)
126  {
127  sLog.outError("AHBot error: AuctionHouseBot.Time.Min must be less or equal to AuctionHouseBot.Time.Max. Setting Time.Min equal to Time.Max.");
128  m_auctionTimeMin = m_auctionTimeMax;
129  }
130 
131  // buy item value
132  m_buyValue = GetMinMaxConfig("AuctionHouseBot.Buy.Value", 0, 200, 90);
133 
134  // overridden items
135  QueryResult* result = CharacterDatabase.PQuery("SELECT item, value, add_chance, min_amount, max_amount FROM ahbot_items");
136  if (result)
137  {
138  do
139  {
140  Field* fields = result->Fetch();
141  uint32 itemId = fields[0].GetUInt32();
142  AuctionHouseBotItemData itemData;
143  itemData.Value = fields[1].GetUInt32();
144  itemData.AddChance = fields[2].GetUInt32();
145  itemData.MinAmount = fields[3].GetUInt32();
146  itemData.MaxAmount = fields[4].GetUInt32();
147  m_itemData[itemId] = itemData;
148  }
149  while (result->NextRow());
150  delete result;
151  }
152  }

155 void AuctionHouseBot::Update()
156 {
157  if (++m_houseAction >= MAX_AUCTION_HOUSE_TYPE * 2)
158  m_houseAction = 0;
159 
160  AuctionHouseType houseType = AuctionHouseType(m_houseAction % MAX_AUCTION_HOUSE_TYPE);
161  AuctionHouseObject* auctionHouse = sAuctionMgr.GetAuctionsMap(houseType);
162  if (m_houseAction < MAX_AUCTION_HOUSE_TYPE && urand(0, 99) < m_chanceSell)
163  {
164  // Sell items
165  std::unordered_map<uint32, uint32> itemMap;
166 
167  AddLootToItemMap(&LootTemplates_Creature, m_creatureLootNormalConfig, m_creatureLootNormalTemplates, itemMap); // normal creature loot
168  AddLootToItemMap(&LootTemplates_Creature, m_creatureLootEliteConfig, m_creatureLootEliteTemplates, itemMap); // elite creature loot
169  AddLootToItemMap(&LootTemplates_Creature, m_creatureLootRareEliteConfig, m_creatureLootRareEliteTemplates, itemMap); // rare elite creature loot
170  AddLootToItemMap(&LootTemplates_Creature, m_creatureLootWorldBossConfig, m_creatureLootWorldBossTemplates, itemMap); // world boss creature loot
171  AddLootToItemMap(&LootTemplates_Creature, m_creatureLootRareConfig, m_creatureLootRareTemplates, itemMap); // rare creature loot
172 
173  AddLootToItemMap(&LootTemplates_Disenchant, m_disenchantLootConfig, m_disenchantLootTemplates, itemMap); // disenchant loot
174  AddLootToItemMap(&LootTemplates_Fishing, m_fishingLootConfig, m_fishingLootTemplates, itemMap); // fishing loot
175  AddLootToItemMap(&LootTemplates_Gameobject, m_gameobjectLootConfig, m_gameobjectLootTemplates, itemMap); // gameobject loot
176  AddLootToItemMap(&LootTemplates_Skinning, m_skinningLootConfig, m_skinningLootTemplates, itemMap); // skinning loot
177 
178  // profession items are a bit different (not looted)
179  if (m_professionItemsConfig[1] > 0 && m_professionItemsConfig[3] > 0 && m_professionItems.size() > 0)
180  {
181  int32 maxTemplates = m_professionItemsConfig[0] < 0 ? urand(0, m_professionItemsConfig[1] - m_professionItemsConfig[0]) + m_professionItemsConfig[0] : urand(m_professionItemsConfig[0], m_professionItemsConfig[1]);
182  if (maxTemplates > 0)
183  {
184  for (int32 templateCounter = 0; templateCounter < maxTemplates; ++templateCounter)
185  {
186  uint32 item = m_professionItems[urand(0, m_professionItems.size() - 1)];
187  ItemPrototype const* prototype = ObjectMgr::GetItemPrototype(item);
188  if (!prototype || prototype->Quality == 0 || urand(0, (1 << (prototype->Quality - 1)) - 1) > 0)
189  continue; // make it decreasingly likely that crafted items of higher quality is added to the auction house (white: 100%, green: 50%, blue: 25%, purple: 12.5%, ...)
190  uint32 count = (uint32) round((uint64)prototype->GetMaxStackSize() * urand(m_professionItemsConfig[2], m_professionItemsConfig[3]) / 100.0);
191  if (count <= 0)
192  count = 1;
193  itemMap[item] += count;
194  }
195  }
196  }
197 
198  // remove items we've overridden (AddChance > 0) and add using given AddChance and stack size
199  for (auto& itemData : m_itemData)
200  {
201  if (itemData.second.AddChance > 0) // replace normal loot sources with custom chance of adding item
202  itemMap[itemData.first] = urand(0, 99) < itemData.second.AddChance ? urand(itemData.second.MinAmount, itemData.second.MaxAmount) : 0;
203  }
204 
205  for (auto& itemEntry : itemMap)
206  {
207  ItemPrototype const* prototype = ObjectMgr::GetItemPrototype(itemEntry.first);
208  if (!prototype || prototype->GetMaxStackSize() == 0)
209  continue; // really shouldn't happen, but better safe than sorry
210  auto iterator = m_itemData.find(prototype->ItemId);
211  if (iterator != m_itemData.end() && iterator->second.Value == 0)
212  continue; // item is blacklisted
213  if (iterator == m_itemData.end() || iterator->second.AddChance == 0)
214  {
215  if (prototype->Bonding == BIND_WHEN_PICKED_UP || prototype->Bonding == BIND_QUEST_ITEM)
216  continue; // no BoP and quest items
217  if (prototype->Flags & ITEM_FLAG_HAS_LOOT)
218  continue; // nor items containing loot
219  if (m_itemValue[prototype->Quality][prototype->Class] == 0)
220  continue; // item class is filtered out
221  }
222 
223  uint32 itemValue = ValueWithVariance(iterator != m_itemData.end() ? iterator->second.Value : CalculateBuyoutPrice(prototype));
224  for (uint32 stackCounter = 0; stackCounter < itemEntry.second; stackCounter += prototype->GetMaxStackSize())
225  {
226  uint32 count = itemEntry.second - stackCounter > prototype->GetMaxStackSize() ? prototype->GetMaxStackSize() : itemEntry.second - stackCounter;
227  uint32 buyoutPrice = itemValue * count;
228  Item* item = Item::CreateItem(itemEntry.first, count);
229  if (buyoutPrice == 0 || !item)
230  continue; // don't put up items we don't know the value of
231  uint32 bidPrice = buyoutPrice * (urand(m_auctionBidMin, m_auctionBidMax)) / 100;
232  if (item)
233  auctionHouse->AddAuction(sAuctionHouseStore.LookupEntry(houseType == AUCTION_HOUSE_ALLIANCE ? 1 : (houseType == AUCTION_HOUSE_HORDE ? 6 : 7)), item, urand(m_auctionTimeMin, m_auctionTimeMax) * HOUR, bidPrice, buyoutPrice);
234  }
235  }
236  } else if (m_houseAction >= MAX_AUCTION_HOUSE_TYPE && urand(0, 99) < m_chanceBuy)
237  {
238  // Buy items
239  AuctionHouseObject::AuctionEntryMapBounds bounds = auctionHouse->GetAuctionsBounds();
240  std::vector<AuctionEntry*> buyoutAuctions;
241  for (AuctionHouseObject::AuctionEntryMap::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
242  {
243  AuctionEntry* auction = itr->second;
244  if (auction->owner == 0 && auction->bid == 0)
245  continue; // ignore bidding/buying auctions that were created by ahbot and not bidded on by player
246  Item* item = sAuctionMgr.GetAItem(auction->itemGuidLow);
247  if (!item)
248  continue; // shouldn't happen, but apparently it does(?)
249  auto prototype = item->GetProto();
250  if (!prototype)
251  continue; // shouldn't happen
252  auto iterator = m_itemData.find(prototype->ItemId);
253  if (iterator != m_itemData.end() && iterator->second.Value == 0)
254  continue; // item is blacklisted
255 
256  uint32 buyItemCheck = ValueWithVariance(iterator != m_itemData.end() ? iterator->second.Value : CalculateBuyoutPrice(prototype));
257  buyItemCheck *= item->GetCount();
258  uint32 bidPrice = auction->bid + auction->GetAuctionOutBid();
259  if (auction->startbid > bidPrice)
260  bidPrice = auction->startbid;
261  if (auction->buyout > 0 && buyItemCheck > auction->buyout)
262  buyoutAuctions.push_back(auction); // can't buyout item here as that modifies the AuctionEntryMap, invalidating the iterator
263  else if (buyItemCheck > bidPrice)
264  auction->UpdateBid(bidPrice);
265  }
266  for (auto auction : buyoutAuctions)
267  auction->UpdateBid(auction->buyout);
268  }
269 }

277 void AuctionHouseBot::Rebuild(bool all)
278 {
279  sLog.outString("AHBot: Rebuilding auction house items");
280  for (uint32 i = 0; i < MAX_AUCTION_HOUSE_TYPE; ++i)
281  {
282  AuctionHouseObject::AuctionEntryMapBounds bounds = sAuctionMgr.GetAuctionsMap(AuctionHouseType(i))->GetAuctionsBounds();
283  for (AuctionHouseObject::AuctionEntryMap::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
284  {
285  AuctionEntry* entry = itr->second;
286  if (!entry->owner)
287  {
288  // ahbot auction
289  if (all || entry->bid == 0) // expire auction if no bid or forced
290  entry->expireTime = sWorld.GetGameTime();
291  }
292  }
293  }
294  // refill auction house with items, simulating typical max amount of items available after some time
295  uint32 updateCounter = ((m_auctionTimeMax - m_auctionTimeMin) / 2 + m_auctionTimeMin) * 90;
296  for (uint32 i = 0; i < updateCounter; ++i)
297  {
298  if (m_houseAction >= MAX_AUCTION_HOUSE_TYPE - 1)
299  m_houseAction = -1; // this prevents AHBot from buying items when refilling
300  Update();
301  }

333 void AuctionHouseBot::SetItemData(uint32 item, AuctionHouseBotItemData& itemData, bool reset)
334 {
335  static SqlStatementID delItem;
336  SqlStatement stmt = CharacterDatabase.CreateStatement(delItem, "DELETE FROM ahbot_items WHERE item = ?");
337  stmt.PExecute(item);
338 
339  if (reset)
340  {
341  m_itemData.erase(item);
342  return;
343  }
344  ItemPrototype const* prototype = ObjectMgr::GetItemPrototype(item);
345  if (!prototype)
346  return;
347 
348  if (itemData.AddChance > 100)
349  itemData.AddChance = 100;
350 
351  if (itemData.MinAmount == 0)
352  itemData.MinAmount = prototype->GetMaxStackSize();
353  if (itemData.MaxAmount == 0)
354  itemData.MaxAmount = prototype->GetMaxStackSize();
355  if (itemData.MaxAmount < itemData.MinAmount)
356  itemData.MaxAmount = itemData.MinAmount;
357 
358  m_itemData[item] = itemData;
359 
360  static SqlStatementID addItem;
361  stmt = CharacterDatabase.CreateStatement(addItem, "INSERT INTO ahbot_items (item, value, add_chance, min_amount, max_amount) VALUES (?, ?, ?, ?, ?)");
362  stmt.addUInt32(item);
363  stmt.addUInt32(itemData.Value);
364  stmt.addUInt32(itemData.AddChance);
365  stmt.addUInt32(itemData.MinAmount);
366  stmt.addUInt32(itemData.MaxAmount);
367  stmt.Execute();

398 void AuctionHouseBot::ParseLootConfig(char const* fieldname, std::vector<int32>& lootConfig)
399 {
400  std::stringstream includeStream(m_ahBotCfg.GetStringDefault(fieldname));
401  std::string temp;
402  lootConfig.clear();
403  while (getline(includeStream, temp, ','))
404  lootConfig.push_back(atoi(temp.c_str()));
405  if (lootConfig.size() > 4)
406  sLog.outError("AHBot error: Too many values specified for field %s (%lu), 4 values required. Additional values ignored.", fieldname, lootConfig.size());
407  else if (lootConfig.size() < 4)
408  {
409  sLog.outError("AHBot error: Too few values specified for field %s (%lu), 4 values required. Setting 0 for remaining values.", fieldname, lootConfig.size());
410  while (lootConfig.size() < 4)
411  lootConfig.push_back(0);
412  }
413  for (uint32 index = 1; index < 4; ++index)
414  {
415  if (lootConfig[index] < 0)
416  {
417  sLog.outError("AHBot error: %s value (%d) for field %s should not be a negative number, setting value to 0.", (index == 1 ? "Second" : (index == 2 ? "Third" : "Fourth")), lootConfig[1], fieldname);
418  lootConfig[index] = 0;
419  }
420  }
421  if (lootConfig[0] > lootConfig[1])
422  {
423  sLog.outError("AHBot error: First value (%d) must be less than or equal to second value (%d) for field %s. Setting first value to second value.", lootConfig[0], lootConfig[1], fieldname);
424  lootConfig[0] = lootConfig[1];
425  }
426  if (lootConfig[2] > lootConfig[3])
427  {
428  sLog.outError("AHBot error: Third value (%d) must be less than or equal to fourth value (%d) for field %s. Setting third value to fourth value.", lootConfig[2], lootConfig[3], fieldname);
429  lootConfig[2] = lootConfig[3];
430  }
431 }

463 void AuctionHouseBot::AddLootToItemMap(LootStore* store, std::vector<int32>& lootConfig, std::vector<uint32>& lootTemplates, std::unordered_map<uint32, uint32>& itemMap)
464 {
465  if (lootConfig[1] <= 0 || lootConfig[3] <= 0 || lootTemplates.size() <= 0)
466  return;
467  int32 maxTemplates = lootConfig[0] < 0 ? urand(0, lootConfig[1] - lootConfig[0]) + lootConfig[0] : urand(lootConfig[0], lootConfig[1]);
468  if (maxTemplates <= 0)
469  return;
470  for (int32 templateCounter = 0; templateCounter < maxTemplates; ++templateCounter)
471  {
472  uint32 lootTemplate = urand(0, lootTemplates.size() - 1);
473  LootTemplate const* lootTable = store->GetLootFor(lootTemplates[lootTemplate]);
474  if (!lootTable)
475  continue;
476  std::unique_ptr<Loot> loot = std::unique_ptr<Loot>(new Loot(LOOT_DEBUG));
477  for (uint32 repeat = urand(lootConfig[2], lootConfig[3]); repeat > 0; --repeat)
478  lootTable->Process(*loot, nullptr, *store, store->IsRatesAllowed());
479 
480  LootItem* lootItem;
481  for (uint32 slot = 0; (lootItem = loot->GetLootItemInSlot(slot)); ++slot)
482  itemMap[lootItem->itemId] += lootItem->count;
483  }

486 uint32 AuctionHouseBot::CalculateBuyoutPrice(ItemPrototype const* prototype)
487 {
488  uint32 buyoutPrice = prototype->BuyPrice;
489  // test whether we have a buy price or if buy price greatly exceed sell price (causes item to be valued too high, notably arrows/shells do this)
490  // if using sell price then price must be multiplied by 4 if white or gray item, 5 if green or better to match expected buy price
491  if (buyoutPrice == 0 || (prototype->SellPrice > 0 && buyoutPrice / prototype->SellPrice > 5))
492  buyoutPrice = prototype->SellPrice * (prototype->Quality <= ITEM_QUALITY_NORMAL ? 4 : 5);
493  // multiply buyoutPrice with item quality price percentage
494  // if item is sold by a vendor and vendor value is forced, then multiply by 100 (setting vendor price)
495  buyoutPrice *= (m_vendorValue && m_vendorItems.find(prototype->ItemId) != m_vendorItems.end() ? 100 : m_itemValue[prototype->Quality][prototype->Class]);
496  buyoutPrice /= 100; // since we multiplied with m_itemValue
497  return buyoutPrice;
498 }
