291  void UpdateAI(const uint32 uiDiff) override
292  {
293  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
294  return;
295 
296  switch (m_Phase)
297  {
298  case PHASE_GROUND:
299 
300  // Switch to underground phase on timer
301  if (m_PhaseSwitchTimer < uiDiff)
302  {
303  if (DoCastSpellIfCan(m_creature, SPELL_SUBMERGE) == CAST_OK)
304  {
305  DoScriptText(SAY_SUBMERGE, m_creature);
306  DoScriptText(EMOTE_BURROW, m_creature);
307  m_PhaseSwitchTimer = 63000;
308  m_Phase = PHASE_SUBMERGING;
309  return;
310  }
311  }
312  else
313  m_PhaseSwitchTimer -= uiDiff;
314 
315  // Switch to phase 3 when below 30%
316  if (m_creature->GetHealthPercent() <= 30.0f)
317  {
318  if (DoCastSpellIfCan(m_creature, SPELL_LEECHING_SWARM) == CAST_OK)
319  {
320  DoScriptText(SAY_LEECHING_SWARM, m_creature);
321  DoScriptText(EMOTE_LEECHING_SWARM, m_creature);
322  m_Phase = PHASE_LEECHING_SWARM;
323  }
324  }
325 
326  // No break - the spells are used in both phase 1 and 3
327  case PHASE_LEECHING_SWARM:
328 
329  if (m_uiFreezingSlashTimer < uiDiff)
330  {
331  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FREEZING_SLASH) == CAST_OK)
332  m_uiFreezingSlashTimer = 20000;
333  }
334  else
335  m_uiFreezingSlashTimer -= uiDiff;
336 
337  if (m_uiPenetratingColdTimer < uiDiff)
338  {
339  if (DoCastSpellIfCan(m_creature, SPELL_PENETRATING_COLD) == CAST_OK)
340  m_uiPenetratingColdTimer = 15000;
341  }
342  else
343  m_uiPenetratingColdTimer -= uiDiff;
344 
345  // The Borrowers are summoned in Ground phase only on normal mode or during Ground and Swarm phase on heroic mode
346  if (m_Phase == PHASE_GROUND || (m_pInstance && m_pInstance->IsHeroicDifficulty()))
347  {
348  if (m_uiBurrowerSummonTimer < uiDiff)
349  {
350  // The number of targets is handled in core, based on difficulty
351  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_NERUBIAN_BURROWER) == CAST_OK)
352  m_uiBurrowerSummonTimer = 45000;
353  }
354  else
355  m_uiBurrowerSummonTimer -= uiDiff;
356  }
357 
358  DoMeleeAttackIfReady();
359 
360  break;
361 
362  case PHASE_UNDERGROUND:
363 
364  // Underground phase is finished
365  if (m_PhaseSwitchTimer < uiDiff)
366  {
367  DoCastSpellIfCan(m_creature, SPELL_EMERGE, CAST_TRIGGERED);
368  DoCastSpellIfCan(m_creature, SPELL_TELEPORT_TO_SPIKE, CAST_TRIGGERED);
369  DoScriptText(EMOTE_EMERGE, m_creature);
370  DoDespawnPursuingSpikes();
371 
372  m_creature->RemoveAurasDueToSpell(SPELL_SUBMERGE);
373  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
374 
375  // Refresh spheres only on normal difficulty
376  if (m_pInstance && !m_pInstance->IsHeroicDifficulty())
377  DoRefreshSpheres();
378 
379  m_PhaseSwitchTimer = 80000;
380  m_Phase = PHASE_GROUND;
381  }
382  else
383  m_PhaseSwitchTimer -= uiDiff;
384 
385  break;
386  case PHASE_SUBMERGING: // Do nothing, but continue berserk timer
387  break;
388  }
389 
390  if (m_uiBerserkTimer)
391  {
392  if (m_uiBerserkTimer <= uiDiff)
393  {
394  if (m_Phase != PHASE_UNDERGROUND)
395  {
396  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
397  {
398  DoScriptText(SAY_BERSERK, m_creature);
399  m_uiBerserkTimer = 0;
400  }
401  }
402  }
403  else
404  m_uiBerserkTimer -= uiDiff;
405  }
406  }
