42 bool QuestRewarded_tbc_attunement_titles(Player* player, Creature* /*creature*/, Quest const* quest)
43 {
44  switch (quest->GetQuestId())
45  {
46  case QUEST_CUDGEL_OF_KARDESH:

48  player->SetTitle(TITLE_CHAMPION_OF_THE_NAARU);
49  return true;
50  case QUEST_DISTRACTION_FOR_AKAMA:
51  if (player->GetQuestStatus(QUEST_VIALS_OF_ETERNITY) == QUEST_STATUS_COMPLETE)
52  player->SetTitle(TITLE_HAND_OF_ADAL);
53  return true;
54  case QUEST_VIALS_OF_ETERNITY:
55  if (player->GetQuestStatus(QUEST_DISTRACTION_FOR_AKAMA) == QUEST_STATUS_COMPLETE)
56  player->SetTitle(TITLE_HAND_OF_ADAL);
57  return true;
58  }
59 
60  return false;
61 }
