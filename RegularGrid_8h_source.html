160  template<typename RayCallback>
161  void intersectRay(const Ray& ray, RayCallback& intersectCallback, float& max_dist, const Vector3& end, bool ignoreM2Model)
162  {
163  Cell cell = Cell::ComputeCell(ray.origin().x, ray.origin().y);
164  if (!cell.isValid())
165  return;
166 
167  Cell last_cell = Cell::ComputeCell(end.x, end.y);
168 
169  if (cell == last_cell)
170  {
171  if (Node* node = nodes[cell.x][cell.y])
172  node->intersectRay(ray, intersectCallback, max_dist, ignoreM2Model);
173  return;
174  }
175 
176  float voxel = (float)CELL_SIZE;
177  float kx_inv = ray.invDirection().x, bx = ray.origin().x;
178  float ky_inv = ray.invDirection().y, by = ray.origin().y;
179 
180  int stepX, stepY;
181  float tMaxX, tMaxY;
182  if (kx_inv >= 0)
183  {
184  stepX = 1;
185  float x_border = (cell.x + 1) * voxel;
186  tMaxX = (x_border - bx) * kx_inv;
187  }
188  else
189  {
190  stepX = -1;
191  float x_border = (cell.x - 1) * voxel;
192  tMaxX = (x_border - bx) * kx_inv;
193  }
194 
195  if (ky_inv >= 0)
196  {
197  stepY = 1;
198  float y_border = (cell.y + 1) * voxel;
199  tMaxY = (y_border - by) * ky_inv;
200  }
201  else
202  {
203  stepY = -1;
204  float y_border = (cell.y - 1) * voxel;
205  tMaxY = (y_border - by) * ky_inv;
206  }
207 
208  //int Cycles = std::max((int)ceilf(max_dist/tMaxX),(int)ceilf(max_dist/tMaxY));
209  //int i = 0;
210 
211  float tDeltaX = voxel * fabs(kx_inv);
212  float tDeltaY = voxel * fabs(ky_inv);
213  do
214  {
215  if (Node* node = nodes[cell.x][cell.y])
216  {
217  //float enterdist = max_dist;
218  node->intersectRay(ray, intersectCallback, max_dist, ignoreM2Model);
219  }
220  if (cell == last_cell)
221  break;
222  if (tMaxX < tMaxY)
223  {
224  tMaxX += tDeltaX;
225  cell.x += stepX;
226  }
227  else
228  {
229  tMaxY += tDeltaY;
230  cell.y += stepY;
231  }
232  //++i;
233  }
234  while (cell.isValid());
