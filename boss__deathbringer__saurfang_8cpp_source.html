237  void ExecuteAction(uint32 action) override
238  {
239  switch (action)
240  {
241  case DEATHBRINGER_BERSERK:
242  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
243  {
244  DoScriptText(SAY_BERSERK, m_creature);
245  DisableCombatAction(action);
246  }
247  break;
248  case DEATHBRINGER_FRENZY:
249  if (m_creature->GetHealthPercent() <= 30.0f)
250  {
251  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
252  {
253  DoScriptText(EMOTE_FRENZY, m_creature);
254  DisableCombatAction(action);
255  }
256  }
257  break;
258  case DEATHBRINGER_RUNE_OF_BLOOD:
259  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_RUNE_OF_BLOOD) == CAST_OK)
260  ResetCombatAction(action, 25000);
261  break;
262  case DEATHBRINGER_BOILING_BLOOD:
263  if (DoCastSpellIfCan(m_creature, SPELL_BOILING_BLOOD) == CAST_OK)
264  ResetCombatAction(action, 15000);
265  break;
266  case DEATHBRINGER_BLOOD_NOVA:
267  if (DoCastSpellIfCan(m_creature, SPELL_BLOOD_NOVA) == CAST_OK)
268  ResetCombatAction(action, 20000);
269  break;
270  case DEATHBRINGER_BLOOD_BEAST:
271  DoScriptText(SAY_BLOODBEASTS, m_creature);
272 
273  DoCastSpellIfCan(m_creature, SPELL_CALL_BLOOD_BEAST_1, CAST_TRIGGERED);
274  DoCastSpellIfCan(m_creature, SPELL_CALL_BLOOD_BEAST_2, CAST_TRIGGERED);
275 
276  if (m_bIs25ManMode)
277  {
278  DoCastSpellIfCan(m_creature, SPELL_CALL_BLOOD_BEAST_3, CAST_TRIGGERED);
279  DoCastSpellIfCan(m_creature, SPELL_CALL_BLOOD_BEAST_4, CAST_TRIGGERED);
280  DoCastSpellIfCan(m_creature, SPELL_CALL_BLOOD_BEAST_5, CAST_TRIGGERED);
281  }
282 
283  ResetCombatAction(action, 40000);
284 
285  if (m_bIsHeroicMode)
286  ResetCombatAction(DEATHBRINGER_SCENT_OF_BLOOD, 7000);
287  break;
288  case DEATHBRINGER_SCENT_OF_BLOOD:
289  if (DoCastSpellIfCan(m_creature, SPELL_SCENT_OF_BLOOD) == CAST_OK)
290  {
291  DoScriptText(EMOTE_SCENT, m_creature);
292  ResetCombatAction(action, 40000);
293  }
294  break;
295  case DEATHBRINGER_MARK_FALLEN_CHAMPION:
296  if (m_creature->GetPower(m_creature->GetPowerType()) == 100)
297  {
298  if (DoCastSpellIfCan(m_creature, SPELL_MARK_FALLEN_CHAMPION) == CAST_OK)
299  {
300  DoScriptText(SAY_FALLENCHAMPION, m_creature);
301  m_creature->RemoveAurasDueToSpell(SPELL_BLOOD_POWER);
302  m_creature->SetPower(m_creature->GetPowerType(), 0);
303 
304  // check for achievement fail
305  ++m_uiAchievSpellCount;
306 
307  if (m_instance)
308  {
309  if (m_uiAchievSpellCount == (m_bIs25ManMode ? 5 : 3))
310  m_instance->SetSpecialAchievementCriteria(TYPE_ACHIEV_MADE_A_MESS, false);
311  }
312  }
313  }
314  break;
315  }
316  }
