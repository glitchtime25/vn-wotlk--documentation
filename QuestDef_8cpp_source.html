188 uint32 Quest::GetXPReward(Player* pPlayer) const
189 {
190  if (pPlayer)
191  {
192  uint32 realXP = 0;
193  uint32 xpMultiplier = 0;
194  int32 baseLevel = 0;
195  int32 playerLevel = pPlayer->GetLevel();
196 
197  // formula can possibly be organized better, using less if's and simplify some.
198 
199  if (QuestLevel != -1)
200  baseLevel = QuestLevel;
201 
202  if (((baseLevel - playerLevel) + 10) * 2 > 10)
203  {
204  baseLevel = playerLevel;
205 
206  if (QuestLevel != -1)
207  baseLevel = QuestLevel;
208 
209  if (((baseLevel - playerLevel) + 10) * 2 <= 10)
210  {
211  if (QuestLevel == -1)
212  baseLevel = playerLevel;
213 
214  xpMultiplier = 2 * (baseLevel - playerLevel) + 20;
215  }
216  else
217  {
218  xpMultiplier = 10;
219  }
220  }
221  else
222  {
223  baseLevel = playerLevel;
224 
225  if (QuestLevel != -1)
226  baseLevel = QuestLevel;
227 
228  if (((baseLevel - playerLevel) + 10) * 2 >= 1)
229  {
230  baseLevel = playerLevel;
231 
232  if (QuestLevel != -1)
233  baseLevel = QuestLevel;
234 
235  if (((baseLevel - playerLevel) + 10) * 2 <= 10)
236  {
237  if (QuestLevel == -1)
238  baseLevel = playerLevel;
239 
240  xpMultiplier = 2 * (baseLevel - playerLevel) + 20;
241  }
242  else
243  {
244  xpMultiplier = 10;
245  }
246  }
247  else
248  {
249  xpMultiplier = 1;
250  }
251  }
252 
253  // not possible to reward XP when baseLevel does not exist in dbc
254  if (const QuestXPLevel* pXPData = sQuestXPLevelStore.LookupEntry(baseLevel))
255  {
256  uint32 rawXP = xpMultiplier * pXPData->xpIndex[RewXPId] / 10;
257 
258  // round values
259  if (rawXP > 1000)
260  realXP = ((rawXP + 25) / 50 * 50);
261  else if (rawXP > 500)
262  realXP = ((rawXP + 12) / 25 * 25);
263  else if (rawXP > 100)
264  realXP = ((rawXP + 5) / 10 * 10);
265  else
266  realXP = ((rawXP + 2) / 5 * 5);
267  }
268 
269  return realXP;
270  }
271 
272  return 0;
273 }

323 uint32 Quest::CalculateRewardHonor(uint32 level) const
324 {
325  if (level > GT_MAX_LEVEL)
326  level = GT_MAX_LEVEL;
327 
328  uint32 honor = 0;
329 
330  if (GetRewHonorAddition() > 0 || GetRewHonorMultiplier() > 0.0f)
331  {
332  // values stored from 0.. for 1...
333  TeamContributionPoints const* tc = sTeamContributionPoints.LookupEntry(level - 1);
334  if (!tc)
335  return 0;
336  uint32 i_honor = uint32(tc->Value * GetRewHonorMultiplier() * 0.1f);
337  honor = i_honor + GetRewHonorAddition();
338  }
339 
340  return honor;
