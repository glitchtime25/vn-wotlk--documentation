194  void UpdateAI(const uint32 uiDiff) override
195  {
196  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
197  return;
198 
199  if (!m_pInstance)
200  return;
201 
202  // Berserk
203  if (m_uiBerserkTimer)
204  {
205  if (m_uiBerserkTimer <= uiDiff)
206  {
207  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
208  {
209  DoScriptText(SAY_BERSERK, m_creature);
210  m_uiBerserkTimer = 0;
211  }
212  }
213  else
214  m_uiBerserkTimer -= uiDiff;
215  }
216 
217  if (m_uiGaseousBlightTimer)
218  {
219  if (m_uiGaseousBlightTimer <= uiDiff)
220  {
221  // two stage event; first trigger all the puddle stalkers around then set the room in gas
222  switch (m_uiGaseousBlightStage)
223  {
224  case 0:
225  if (Creature* pProfessor = m_pInstance->GetSingleCreatureFromStorage(NPC_PROFESSOR_PUTRICIDE))
226  {
227  pProfessor->HandleEmote(EMOTE_ONESHOT_TALK_NOSHEATHE);
228  pProfessor->CastSpell(pProfessor, SPELL_GASEOUS_BLIGHT_INIT, TRIGGERED_OLD_TRIGGERED);
229  DoScriptText((m_pInstance->GetData(TYPE_ROTFACE) == DONE && m_pInstance->IsHeroicDifficulty()) ? SAY_BLIGHT_ROTFACE_DEAD : SAY_BLIGHT, pProfessor);
230  }
231  m_uiGaseousBlightTimer = 1000;
232  break;
233  case 1:
234  if (DoCastSpellIfCan(m_creature, SPELL_GASEOUS_BLIGHT_1) == CAST_OK)
235  m_uiGaseousBlightTimer = 0;
236  break;
237  }
238  ++m_uiGaseousBlightStage;
239  }
240  else
241  m_uiGaseousBlightTimer -= uiDiff;
242  }
243 
244  // Inhale Blight and Pungent Blight
245  if (m_uiInhaleBlightTimer < uiDiff)
246  {
247  SpellAuraHolder* pHolder = m_creature->GetSpellAuraHolder(m_pInstance->Is25ManDifficulty() ? SPELL_INHALED_BLIGHT_25 : SPELL_INHALED_BLIGHT_10);
248 
249  // inhale the gas or if already have 3 stacks - release it
250  if (pHolder && pHolder->GetStackAmount() >= 3)
251  {
252  if (DoCastSpellIfCan(m_creature, SPELL_PUNGENT_BLIGHT) == CAST_OK)
253  {
254  DoScriptText(SAY_PUNGUENT_BLIGHT, m_creature);
255  m_uiInhaleBlightTimer = 38000;
256  }
257  }
258  else
259  {
260  if (DoCastSpellIfCan(m_creature, SPELL_INHALE_BLIGHT) == CAST_OK)
261  m_uiInhaleBlightTimer = 36000;
262  }
263  }
264  else
265  m_uiInhaleBlightTimer -= uiDiff;
266 
267  // Gas Spore
268  if (m_uiGasSporeTimer < uiDiff)
269  {
270  if (DoCastSpellIfCan(m_creature, SPELL_GAS_SPORE) == CAST_OK)
271  {
272  DoScriptText(EMOTE_SPORES, m_creature);
273  m_uiGasSporeTimer = 40000;
274  }
275  }
276  else
277  m_uiGasSporeTimer -= uiDiff;
278 
279  // Vile Gas
280  if (m_uiVileGasTimer < uiDiff)
281  {
282  if (DoCastSpellIfCan(m_creature, SPELL_VILE_GAS) == CAST_OK)
283  m_uiVileGasTimer = 30000;
284  }
285  else
286  m_uiVileGasTimer -= uiDiff;
287 
288  // Heroic spells
289  if (m_pInstance->IsHeroicDifficulty())
290  {
291  if (m_uiMalleableGooTimer < uiDiff)
292  {
293  if (DoCastSpellIfCan(m_creature, SPELL_MALLEABLE_GOO_SUMMON) == CAST_OK)
294  m_uiMalleableGooTimer = 15000;
295  }
296  else
297  m_uiMalleableGooTimer -= uiDiff;
298  }
299 
300  DoMeleeAttackIfReady();
301  }

309 bool EffectScriptEffectCreature_spell_inhale_blight(Unit* pCaster, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
310 {
311  if (uiSpellId == SPELL_INHALE_BLIGHT && uiEffIndex == EFFECT_INDEX_0 && pCreatureTarget->GetEntry() == NPC_FESTERGUT)
312  {
313  instance_icecrown_citadel* pInstance = (instance_icecrown_citadel*)pCreatureTarget->GetInstanceData();
314  if (!pInstance)
315  return false;
316 
317  Creature* pGasStalker = pInstance->GetSingleCreatureFromStorage(NPC_GAS_STALKER);
318  if (!pGasStalker)
319  return false;
320 
321  if (pCreatureTarget->HasAura(SPELL_GASEOUS_BLIGHT_1))
322  {
323  pGasStalker->RemoveAllAurasOnEvade();
324  pCreatureTarget->CastSpell(pCreatureTarget, SPELL_GASEOUS_BLIGHT_2, TRIGGERED_OLD_TRIGGERED);
325  }
326  else if (pCreatureTarget->HasAura(SPELL_GASEOUS_BLIGHT_2))
327  {
328  pGasStalker->RemoveAllAurasOnEvade();
329  pCreatureTarget->CastSpell(pCreatureTarget, SPELL_GASEOUS_BLIGHT_3, TRIGGERED_OLD_TRIGGERED);
330  }
331  else if (pCreatureTarget->HasAura(SPELL_GASEOUS_BLIGHT_3))
332  pGasStalker->RemoveAllAurasOnEvade();
333 
334  return true;
335  }
336 
337  return false;
338 }
