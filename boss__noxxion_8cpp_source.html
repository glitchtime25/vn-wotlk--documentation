56  void UpdateAI(const uint32 diff) override
57  {

59  return;
60 
61  if (m_uiToxicVolleyTimer < diff)

64  m_uiToxicVolleyTimer = 9000;

66  else
67  m_uiToxicVolleyTimer -= diff;
68

76

80  m_uiSummonTimer = 40000;
