33 bool CheckPathToTarget(Unit* pet, Unit* target)
34 {
35  float angle = target->GetAngle(pet);
36  float x, y, z;
37  target->GetNearPoint(pet, x, y, z, pet->GetObjectBoundingRadius(), pet->AI()->GetAttackDistance(), angle, target->IsInWater());
38 
39  PathFinder pathFinder(pet);
40  bool result = pathFinder.calculate(x, y, z);
41 
42  if (pathFinder.getPathType() & PATHFIND_NOPATH)
43  return false; // TODO send nopath
44 
45  float collisionHeight = pet->GetCollisionHeight();
46  auto& path = pathFinder.getPath();
47  for (size_t i = 1; i < path.size(); ++i)
48  {
49  const G3D::Vector3& dataPrevious = path.at(i - 1);
50  const G3D::Vector3& data = path.at(i);
51  if (!pet->GetMap()->IsInLineOfSight(dataPrevious.x, dataPrevious.y, dataPrevious.z + collisionHeight, data.x, data.y, data.z + collisionHeight, pet->GetPhaseMask(), true))
52  return false;
53  }
54 
55  return true;
56 }

63 void WorldSession::HandlePetAction(WorldPacket& recv_data)
64 {
65  ObjectGuid petGuid;
66  uint32 data;
67  ObjectGuid targetGuid;
68  recv_data >> petGuid;
69  recv_data >> data;
70  recv_data >> targetGuid;
71 
72  uint32 spellid = UNIT_ACTION_BUTTON_ACTION(data);
73  uint8 flag = UNIT_ACTION_BUTTON_TYPE(data); // delete = 0x07 CastSpell = C1
74 
75  DETAIL_LOG("HandlePetAction: %s flag is %u, spellid is %u, target %s.", petGuid.GetString().c_str(), uint32(flag), spellid, targetGuid.GetString().c_str());
76 
77  // used also for charmed creature/player
78  Unit* petUnit = _player->GetMap()->GetUnit(petGuid);
79  if (!petUnit)
80  {
81  sLog.outError("HandlePetAction: %s not exist.", petGuid.GetString().c_str());
82  return;
83  }
84 
85  if (_player->GetObjectGuid() != petUnit->GetMasterGuid())
86  {
87  sLog.outError("HandlePetAction: %s isn't controlled by %s.", petGuid.GetString().c_str(), _player->GetGuidStr().c_str());
88  return;
89  }
90 
91  if (!petUnit->IsAlive())
92  {
93  petUnit->SendPetActionFeedback(FEEDBACK_PET_DEAD);
94  return;
95  }
96 
97  CharmInfo* charmInfo = petUnit->GetCharmInfo();
98  if (!charmInfo)
99  {
100  sLog.outError("WorldSession::HandlePetAction: object (GUID: %u TypeId: %u) is considered pet-like but doesn't have a charminfo!", petUnit->GetGUIDLow(), petUnit->GetTypeId());
101  return;
102  }
103 
104  bool isPet = false;
105  if (petUnit->IsCreature())
106  {
107  if (static_cast<Creature*>(petUnit)->IsPet())
108  {
109  isPet = true;
110  if (static_cast<Pet*>(petUnit)->GetModeFlags() & PET_MODE_DISABLE_ACTIONS)
111  return;
112  }
113  }
114 
115  if (!isPet)
116  {
117  if (petUnit->hasUnitState(UNIT_STAT_POSSESSED))
118  {
119  // possess case
120  if (flag != uint8(ACT_COMMAND))
121  {
122  sLog.outError("PetHAndler: unknown PET flag Action %i and spellid %i. For possessed %s", uint32(flag), spellid, petUnit->GetGuidStr().c_str());
123  return;
124  }
125 
126  switch (spellid)
127  {
128  case COMMAND_STAY:
129  case COMMAND_FOLLOW:
130  charmInfo->SetCommandState(CommandStates(spellid));
131  break;
132  case COMMAND_ATTACK:
133  {
134  Unit* targetUnit = targetGuid ? _player->GetMap()->GetUnit(targetGuid) : nullptr;
135 
136  if (!targetUnit)
137  {
138  petUnit->SendPetActionFeedback(FEEDBACK_NOTHING_TO_ATT);
139  break;
140  }
141  if (targetUnit == petUnit || !petUnit->CanAttack(targetUnit))
142  {
143  petUnit->SendPetActionFeedback(FEEDBACK_CANT_ATT_TARGET);
144  break;
145  }
146  // This is true if pet has no target or has target but targets differs.
147  if (petUnit->GetVictim() != targetUnit)
148  petUnit->Attack(targetUnit, true);
149  break;
150  }
151  case COMMAND_DISMISS:
152  _player->BreakCharmOutgoing(petUnit);
153  break;
154  default:
155  sLog.outError("PetHandler: Not allowed action %i and spellid %i. Pet %s owner is %s", uint32(flag), spellid, petUnit->GetGuidStr().c_str(), _player->GetGuidStr().c_str());
156  break;
157  }
158  }
159 
160  if (!petUnit->HasCharmer())
161  return;
162  }
163 
164  GuidSet controlled = _player->GetControlled();
165  if (controlled.empty() || controlled.size() == 1)
166  HandlePetActionHelper(flag, spellid, petUnit, targetGuid);
167  else
168  {
169  for (ObjectGuid guid : controlled)
170  if (Unit* unit = _player->GetMap()->GetUnit(guid))
171  HandlePetActionHelper(flag, spellid, unit, targetGuid);
172  }
173 }

175 void WorldSession::HandlePetActionHelper(uint8 flag, uint32 spellid, Unit* petUnit, ObjectGuid targetGuid)
176 {
177  CharmInfo* charmInfo = petUnit->GetCharmInfo();
178  Pet* pet = nullptr;
179  Creature* creature = nullptr;
180 
181  if (petUnit->IsCreature())
182  {
183  creature = static_cast<Creature*>(petUnit);
184  if (creature->IsPet())
185  pet = static_cast<Pet*>(petUnit);
186  }
187 
188  switch (flag)
189  {
190  case ACT_COMMAND: // 0x07
191  switch (spellid)
192  {
193  case COMMAND_STAY: // flat=1792 // STAY
194  {
195  petUnit->AttackStop(true, true);
196  charmInfo->SetCommandState(COMMAND_STAY);
197  break;
198  }
199  case COMMAND_FOLLOW: // spellid=1792 // FOLLOW
200  {
201  if (!petUnit->hasUnitState(UNIT_STAT_POSSESSED))
202  charmInfo->SetIsRetreating(true);
203 
204  petUnit->AttackStop(true, true);
205  charmInfo->SetCommandState(COMMAND_FOLLOW);
206  break;
207  }
208  case COMMAND_ATTACK: // spellid=1792 // ATTACK
209  {
210  charmInfo->SetIsRetreating();
211  charmInfo->SetSpellOpener();
212 
213  Unit* targetUnit = targetGuid ? _player->GetMap()->GetUnit(targetGuid) : nullptr;
214 
215  if (!targetUnit)
216  {
217  petUnit->SendPetActionFeedback(FEEDBACK_NOTHING_TO_ATT);
218  break;
219  }
220  if (targetUnit == petUnit || !petUnit->CanAttack(targetUnit))
221  {
222  petUnit->SendPetActionFeedback(FEEDBACK_CANT_ATT_TARGET);
223  break;
224  }
225 
226  // This is true if pet has no target or has target but targets differs.
227  if (petUnit->GetVictim() != targetUnit)
228  {
229  if (petUnit->hasUnitState(UNIT_STAT_POSSESSED))
230  {
231  petUnit->AttackStop();
232  petUnit->Attack(targetUnit, true);
233  }
234  else
235  {
236  charmInfo->SetPetLastAttackCommandTime(petUnit->GetMap()->GetCurrentMSTime());
237 
238  // Send pet response regardless of command result as acknowledgement of command being processed
239  if (pet)
240  {
241  // 10% chance to play special warlock pet attack talk, else growl
242  if (pet->getPetType() == SUMMON_PET && roll_chance_i(10))
243  pet->SendPetTalk(uint32(PET_TALK_ATTACK));
244 
245  pet->SendPetAIReaction();
246  }
247 
248  // Ignore command if target habitat is incompatible with pet
249  if (!targetUnit->isInAccessablePlaceFor(petUnit))
250  break;
251 
252  // Ignore command if target is moving home
253  if (targetUnit->GetCombatManager().IsEvadingHome())
254  break;
255 
256  if (!CheckPathToTarget(petUnit, targetUnit))
257  {
258  SendNoPathToOwner(petUnit);
259  break;
260  }
261 
262  petUnit->AttackStop();
263  petUnit->GetMotionMaster()->Clear();
264  petUnit->AI()->AttackStart(targetUnit);
265  }
266  }
267  break;
268  }
269  case COMMAND_DISMISS: // dismiss permanent pet, remove temporary pet, uncharm unit
270  {
271  if (pet)
272  {
273  if (pet->IsDismissDisabled())
274  break;
275 
276  pet->PlayDismissSound();
277  // No action for Hunter pets, Hunters must use their Dismiss Pet spell
278  if (pet->getPetType() != HUNTER_PET)
279  pet->ForcedDespawn();
280  }
281  else
282  {
283  // dismissing a summoned pet is like killing them (this prevents returning a soulshard...)
284  if (creature && creature->IsTemporarySummon())
285  creature->ForcedDespawn();
286  else
287  _player->BreakCharmOutgoing(petUnit);
288  }
289 
290  charmInfo->ResetStayPosition();
291  break;
292  }
293  default:
294  sLog.outError("WORLD: unknown PET flag Action %i and spellid %i.", uint32(flag), spellid);
295  }
296  break;
297  case ACT_REACTION: // 0x6
298  switch (spellid)
299  {
300  case REACT_PASSIVE: // passive
301  {
302  petUnit->AttackStop(true, true);
303  charmInfo->SetSpellOpener();
304  }
305  case REACT_DEFENSIVE: // recovery
306  case REACT_AGGRESSIVE: // activete
307  {
308  petUnit->AI()->SetReactState(ReactStates(spellid));
309  break;
310  }
311  }
312  break;
313  case ACT_DISABLED: // 0x81 spell (disabled), ignore
314  case ACT_PASSIVE: // 0x01
315  case ACT_ENABLED: // 0xC1 spell
316  {
317  charmInfo->SetIsRetreating();
318  charmInfo->SetSpellOpener();
319 
320  Unit* unit_target = targetGuid ? _player->GetMap()->GetUnit(targetGuid) : nullptr;
321 
322  // do not cast unknown spells
323  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellid);
324  if (!spellInfo)
325  {
326  sLog.outError("WORLD: unknown PET spell id %i", spellid);
327  return;
328  }
329 
330  // do not cast not learned spells
331  if (IsPassiveSpell(spellInfo) || !petUnit->HasSpell(spellid))
332  return;
333 
334  if (!petUnit->IsSpellReady(*spellInfo))
335  return;
336 
337  for (unsigned int i : spellInfo->EffectImplicitTargetA)
338  {
339  if (i == TARGET_ENUM_UNITS_ENEMY_AOE_AT_SRC_LOC
340  || i == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DEST_LOC
341  || i == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DYNOBJ_LOC)
342  return;
343  }
344 
345  petUnit->clearUnitState(UNIT_STAT_MOVING);
346 
347  uint32 flags = TRIGGERED_NORMAL_COMBAT_CAST;
348  if (!petUnit->hasUnitState(UNIT_STAT_POSSESSED))
349  flags |= TRIGGERED_PET_CAST;
350 
351  const SpellRangeEntry* sRange = sSpellRangeStore.LookupEntry(spellInfo->rangeIndex);
352 
353  if (!IsSpellRequireTarget(spellInfo))
354  unit_target = nullptr;
355 
356  if (unit_target && !(petUnit->IsWithinDistInMap(unit_target, sRange->maxRange) && petUnit->IsWithinLOSInMap(unit_target, true))
357  && petUnit->CanAttackNow(unit_target))
358  {
359  charmInfo->SetSpellOpener(spellid, sRange->minRange, sRange->maxRange);
360 
361  petUnit->AttackStop();
362 
363  if (!petUnit->hasUnitState(UNIT_STAT_POSSESSED))
364  {
365  if (!CheckPathToTarget(petUnit, unit_target))
366  {
367  SendNoPathToOwner(petUnit);
368  return;
369  }
370 
371  petUnit->AI()->AttackStart(unit_target);
372  // 10% chance to play special warlock pet attack talk, else growl
373  if (pet && pet->getPetType() == SUMMON_PET && pet != unit_target && roll_chance_i(10))
374  petUnit->SendPetTalk((uint32)PET_TALK_ATTACK);
375 
376  petUnit->SendPetAIReaction();
377  }
378  else
379  petUnit->Attack(unit_target, true);
380 
381  return;
382  }
383 
384  Spell* spell = new Spell(petUnit, spellInfo, flags);
385  SpellCastTargets targets;
386  targets.setUnitTarget(unit_target);
387  SpellCastResult result = spell->SpellStart(&targets);
388  charmInfo->SetSpellOpener();
389  if (result == SPELL_CAST_OK)
390  {
391  //10% chance to play special pet attack talk, else growl
392  //actually this only seems to happen on special spells, fire shield for imp, torment for voidwalker, but it's stupid to check every spell
393  if (petUnit->GetTypeId() == TYPEID_UNIT)
394  if (static_cast<Creature*>(petUnit)->IsPet() && (static_cast<Pet*>(petUnit)->getPetType() == SUMMON_PET) && (petUnit != unit_target) && (urand(0, 100) < 10))
395  petUnit->SendPetTalk((uint32)PET_TALK_SPECIAL_SPELL);
396  }
397  break;
398  }
399  default:
400  sLog.outError("WORLD: unknown PET flag Action %i and spellid %i.", uint32(flag), spellid);
401  }
402 }

485 void WorldSession::HandlePetSetAction(WorldPacket& recv_data)
486 {
487  DETAIL_LOG("HandlePetSetAction. CMSG_PET_SET_ACTION");
488 
489  ObjectGuid petGuid;
490 
491  recv_data >> petGuid;
492 
493  Unit* petUnit = _player->GetMap()->GetUnit(petGuid);
494 
495  if (!petUnit || (petUnit != _player->GetPet() && petUnit != _player->GetCharm()))
496  {
497  sLog.outError("HandlePetSetAction: Unknown pet or pet owner.");
498  return;
499  }
500 
501  Creature* petCreature = petUnit->GetTypeId() == TYPEID_UNIT ? static_cast<Creature*>(petUnit) : nullptr;
502  Pet* pet = (petCreature && petCreature->IsPet()) ? static_cast<Pet*>(petUnit) : nullptr;
503 
504  // pet can have action bar disabled
505  if (pet && (pet->GetModeFlags() & PET_MODE_DISABLE_ACTIONS))
506  return;
507 
508  CharmInfo* charmInfo = petUnit->GetCharmInfo();
509  if (!charmInfo)
510  {
511  sLog.outError("WorldSession::HandlePetSetAction: %s is considered pet-like but doesn't have a charminfo!", petUnit->GetGuidStr().c_str());
512  return;
513  }
514 
515  uint8 count = (recv_data.size() == 24) ? 2 : 1;
516 
517  uint32 position[2];
518  uint32 data[2];
519  bool move_command = false;
520 
521  for (uint8 i = 0; i < count; ++i)
522  {
523  recv_data >> position[i];
524  recv_data >> data[i];
525 
526  uint8 act_state = UNIT_ACTION_BUTTON_TYPE(data[i]);
527 
528  // ignore invalid position
529  if (position[i] >= MAX_UNIT_ACTION_BAR_INDEX)
530  return;
531 
532  // in the normal case, command and reaction buttons can only be moved, not removed
533  // at moving count ==2, at removing count == 1
534  // ignore attempt to remove command|reaction buttons (not possible at normal case)
535  if (act_state == ACT_COMMAND || act_state == ACT_REACTION)
536  {
537  if (count == 1)
538  return;
539 
540  move_command = true;
541  }
542  }
543 
544  // check swap (at command->spell swap client remove spell first in another packet, so check only command move correctness)
545  if (move_command)
546  {
547  uint8 act_state_0 = UNIT_ACTION_BUTTON_TYPE(data[0]);
548  if (act_state_0 == ACT_COMMAND || act_state_0 == ACT_REACTION)
549  {
550  uint32 spell_id_0 = UNIT_ACTION_BUTTON_ACTION(data[0]);
551  UnitActionBarEntry const* actionEntry_1 = charmInfo->GetActionBarEntry(position[1]);
552  if (!actionEntry_1 || spell_id_0 != actionEntry_1->GetAction() ||
553  act_state_0 != actionEntry_1->GetType())
554  return;
555  }
556 
557  uint8 act_state_1 = UNIT_ACTION_BUTTON_TYPE(data[1]);
558  if (act_state_1 == ACT_COMMAND || act_state_1 == ACT_REACTION)
559  {
560  uint32 spell_id_1 = UNIT_ACTION_BUTTON_ACTION(data[1]);
561  UnitActionBarEntry const* actionEntry_0 = charmInfo->GetActionBarEntry(position[0]);
562  if (!actionEntry_0 || spell_id_1 != actionEntry_0->GetAction() ||
563  act_state_1 != actionEntry_0->GetType())
564  return;
565  }
566  }
567 
568  for (uint8 i = 0; i < count; ++i)
569  {
570  uint32 spell_id = UNIT_ACTION_BUTTON_ACTION(data[i]);
571  uint8 act_state = UNIT_ACTION_BUTTON_TYPE(data[i]);
572 
573  DETAIL_LOG("Player %s has changed pet spell action. Position: %u, Spell: %u, State: 0x%X", _player->GetName(), position[i], spell_id, uint32(act_state));
574 
575  // if it's act for spell (en/disable/cast) and there is a spell given (0 = remove spell) which pet doesn't know, don't add
576  if (!((act_state == ACT_ENABLED || act_state == ACT_DISABLED || act_state == ACT_PASSIVE) && spell_id && !petUnit->HasSpell(spell_id)))
577  {
578  // sign for autocast
579  if (act_state == ACT_ENABLED && spell_id)
580  {
581  if (petUnit->HasCharmer())
582  charmInfo->ToggleCreatureAutocast(spell_id, true);
583  else if (pet)
584  pet->ToggleAutocast(spell_id, true);
585  }
586  // sign for no/turn off autocast
587  else if (act_state == ACT_DISABLED && spell_id)
588  {
589  if (petUnit->HasCharmer())
590  charmInfo->ToggleCreatureAutocast(spell_id, false);
591  else if (pet)
592  pet->ToggleAutocast(spell_id, false);
593  }
594 
595  charmInfo->SetActionBar(position[i], spell_id, ActiveStates(act_state));
596  }
597  }

600 void WorldSession::HandlePetRename(WorldPacket& recv_data)
601 {
602  DETAIL_LOG("HandlePetRename. CMSG_PET_RENAME");
603 
604  ObjectGuid petGuid;
605  uint8 isdeclined;
606 
607  std::string name;
608  DeclinedName declinedname;
609 
610  recv_data >> petGuid;
611  recv_data >> name;
612  recv_data >> isdeclined;
613 
614  Pet* pet = _player->GetMap()->GetPet(petGuid);
615  // check it!
616  if (!pet || pet->getPetType() != HUNTER_PET ||
617  !pet->HasByteFlag(UNIT_FIELD_BYTES_2, UNIT_BYTES_2_OFFSET_PET_FLAGS, UNIT_CAN_BE_RENAMED) ||
618  pet->GetOwnerGuid() != _player->GetObjectGuid() || !pet->GetCharmInfo())
619  return;
620 
621  PetNameInvalidReason res = ObjectMgr::CheckPetName(name);
622  if (res != PET_NAME_SUCCESS)
623  {
624  SendPetNameInvalid(res, name, nullptr);
625  return;
626  }
627 
628  if (sObjectMgr.IsReservedName(name))
629  {
630  SendPetNameInvalid(PET_NAME_RESERVED, name, nullptr);
631  return;
632  }
633 
634  pet->SetName(name);
635 
636  if (_player->GetGroup())
637  _player->SetGroupUpdateFlag(GROUP_UPDATE_FLAG_PET_NAME);
638 
639  pet->RemoveByteFlag(UNIT_FIELD_BYTES_2, UNIT_BYTES_2_OFFSET_PET_FLAGS, UNIT_CAN_BE_RENAMED);
640 
641  if (isdeclined)
642  {
643  for (auto& i : declinedname.name)
644  {
645  recv_data >> i;
646  }
647 
648  std::wstring wname;
649  Utf8toWStr(name, wname);
650  if (!ObjectMgr::CheckDeclinedNames(GetMainPartOfName(wname, 0), declinedname))
651  {
652  SendPetNameInvalid(PET_NAME_DECLENSION_DOESNT_MATCH_BASE_NAME, name, &declinedname);
653  return;
654  }
655  }
656 
657  CharacterDatabase.BeginTransaction();
658  if (isdeclined)
659  {
