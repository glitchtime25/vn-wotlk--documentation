159  void JustSummoned(Creature* summoned) override
160  {
161  switch (summoned->GetEntry())
162  {
163  case NPC_ASTRAL_FLARE_1:
164  case NPC_ASTRAL_FLARE_2:
165  case NPC_ASTRAL_FLARE_3:
166  case NPC_ASTRAL_FLARE_4:
167  {
168  switch (urand(0, 3))
169  {
170  case 0: DoScriptText(SAY_SUMMON1, m_creature); break;
171  case 1: DoScriptText(SAY_SUMMON2, m_creature); break;
172  }
173 
174  // Flare start with aggro on it's target, should be immune to arcane
175  summoned->AI()->DoCastSpellIfCan(summoned, SPELL_ASTRAL_FLARE_PASSIVE, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
176  summoned->AI()->DoCastSpellIfCan(summoned, SPELL_ASTRAL_FLARE_VISUAL, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
177  summoned->AI()->SetMeleeEnabled(false);
178 
179  summoned->SetInCombatWithZone();
180  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
181  summoned->AddThreat(pTarget, 100001.f);
182  m_sparkGuids.push_back(summoned->GetObjectGuid());
183  break;
184  }
185  }
186  }

188  void ExecuteAction(uint32 action) override
189  {
190  switch (action)
191  {
192  case CURATOR_ACTION_BERSERK:
193  {
194  if (DoCastSpellIfCan(nullptr, SPELL_ARCANE_DECONSTRUCTION) == CAST_OK)
195  DisableCombatAction(action); // once per fight
196  return;
197  }
198  case CURATOR_ACTION_ARCANE_INFUSION:
199  {
200  if (m_creature->GetHealthPercent() < 15.0f)
201  {
202  if (DoCastSpellIfCan(nullptr, SPELL_ARCANE_INFUSION) == CAST_OK)
203  {
204  DoScriptText(SAY_ENRAGE, m_creature);
205  SetActionReadyStatus(action, false); // once per fight
206  m_phase = CURATOR_PHASE_2;
207  }
208  }
209  return;
210  }
211  case CURATOR_ACTION_FLARE:
212  {
213  if (m_phase != CURATOR_PHASE_1)
214  return;
215 
216  // summon Astral Flare
217  uint32 randomSpell = astralFlareSpells[urand(0, 3)];
218  DoCastSpellIfCan(nullptr, randomSpell);
219  ResetCombatAction(action, GetSubsequentActionTimer(action));
220  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(randomSpell);
221  // supposed to happen right away not on next oom
222  if (m_creature->GetPower((Powers)spellInfo->powerType) < Spell::CalculatePowerCost(spellInfo, m_creature))
223  {
224  if (DoCastSpellIfCan(nullptr, SPELL_EVOCATION) == CAST_OK)
225  {
226  DoScriptText(SAY_EVOCATE, m_creature);
227  // m_creature->RemoveAurasDueToSpell(SPELL_ASTRAL_ARMOR);
228  }
229  }
230  return;
231  }
232  case CURATOR_ACTION_HATEFUL_BOLT:
233  {
234  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 1, nullptr, SELECT_FLAG_PLAYER))
235  if (DoCastSpellIfCan(pTarget, SPELL_HATEFUL_BOLT) == CAST_OK)
236  ResetCombatAction(action, GetSubsequentActionTimer(action));
237  return;
238  }
239  }
240  }
