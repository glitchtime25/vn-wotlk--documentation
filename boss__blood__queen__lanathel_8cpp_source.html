149  void MovementInform(uint32 uiMovementType, uint32 uiPointId) override
150  {
151  if (uiMovementType != POINT_MOTION_TYPE)
152  return;
153 
154  if (uiPointId == POINT_CENTER_GROUND)
155  {
156  if (m_uiPhase == PHASE_RUNNING)
157  {
158  if (DoCastSpellIfCan(m_creature, SPELL_INCITE_TERROR) == CAST_OK)
159  {
160  m_uiPhase = PHASE_FLYING;
161 
162  m_creature->SetLevitate(true);
163  m_creature->SetByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_FLY_ANIM);
164 
165  m_creature->GetMotionMaster()->Clear();
166  m_creature->GetMotionMaster()->MovePoint(POINT_CENTER_AIR, aQueenPosition[1][0], aQueenPosition[1][1], aQueenPosition[1][2]);
167  }
168  }
169  else if (m_uiPhase == PHASE_FLYING)
170  {
171  m_uiPhase = PHASE_GROUND;
172  m_uiPhaseTimer = 120000;
173  SetCombatMovement(true);
174 
175  m_creature->SetLevitate(false);
176  m_creature->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_FLY_ANIM);
177 
178  m_creature->GetMotionMaster()->Clear();
179  if (m_creature->GetVictim())
180  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
181  }
182  }
183  else if (uiPointId == POINT_CENTER_AIR)
184  {
185  if (DoCastSpellIfCan(m_creature, SPELL_BLOODBOLT_WHIRL) == CAST_OK)
186  {
187  DoScriptText(SAY_AIR_PHASE, m_creature);
188  m_uiPhase = PHASE_AIR;
189  m_uiPhaseTimer = 7000;
190  }
191  }

194  void UpdateAI(const uint32 uiDiff) override
195  {
196  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
197  return;
198 
199  if (m_uiEnrageTimer)
200  {
201  if (m_uiEnrageTimer <= uiDiff)
202  {
203  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
204  {
205  DoScriptText(SAY_BERSERK, m_creature);
206  m_uiEnrageTimer = 0;
207  }
208  }
209  else
210  m_uiEnrageTimer -= uiDiff;
211  }
212 
213  switch (m_uiPhase)
214  {
215  case PHASE_GROUND:
216  {
217  // Air phase change timer
218  if (m_uiPhaseTimer < uiDiff)
219  {
220  SetCombatMovement(false);
221  m_creature->GetMotionMaster()->Clear();
222  m_creature->GetMotionMaster()->MovePoint(POINT_CENTER_GROUND, aQueenPosition[0][0], aQueenPosition[0][1], aQueenPosition[0][2]);
223 
224  m_uiPhase = PHASE_RUNNING;
225  m_uiPhaseTimer = 0;
226  }
227  else
228  m_uiPhaseTimer -= uiDiff;
229 
230  // Only one bite per fight
231  if (m_uiVampiricBiteTimer)
232  {
233  if (m_uiVampiricBiteTimer <= uiDiff)
234  {
235  if (DoCastSpellIfCan(m_creature, SPELL_VAMPIRIC_BITE) == CAST_OK)
236  {
237  DoScriptText(urand(0, 1) ? SAY_BITE_1 : SAY_BITE_2, m_creature);
238  m_uiVampiricBiteTimer = 0;
239  }
240  }
241  else
242  m_uiVampiricBiteTimer -= uiDiff;
243  }
244 
245  if (m_uiBloodMirrorTimer < uiDiff)
246  {
247  if (DoCastSpellIfCan(m_creature, SPELL_BLOOD_MIRROR) == CAST_OK)
248  m_uiBloodMirrorTimer = 5000;
249  }
250  else
251  m_uiBloodMirrorTimer -= uiDiff;
252 
253  if (m_uiBloodboltTimer < uiDiff)
254  {
255  if (DoCastSpellIfCan(m_creature, SPELL_TWILIGHT_BLOODBOLT) == CAST_OK)
256  m_uiBloodboltTimer = urand(15000, 20000);
257  }
258  else
259  m_uiBloodboltTimer -= uiDiff;
260 
261  if (m_uiPactDarkfallenTimer < uiDiff)
262  {
263  if (DoCastSpellIfCan(m_creature, SPELL_PACT_OF_THE_DARKFALLEN) == CAST_OK)
264  {
265  DoScriptText(SAY_PACT, m_creature);
266  m_uiPactDarkfallenTimer = urand(20000, 25000);
267  }
268  }
269  else
270  m_uiPactDarkfallenTimer -= uiDiff;
271 
272  if (m_uiSwarmingShadowsTimer <= uiDiff)
273  {
274  if (DoCastSpellIfCan(m_creature, SPELL_SWARMING_SHADOWS) == CAST_OK)
275  {
276  DoScriptText(SAY_SHADOWS, m_creature);
277  m_uiSwarmingShadowsTimer = urand(30000, 35000);
278  }
279  }
280  else
281  m_uiSwarmingShadowsTimer -= uiDiff;
282 
283  // Heroic spells
284  if (m_pInstance && m_pInstance->IsHeroicDifficulty())
285  {
286  if (m_uiDeliriousSlashTimer < uiDiff)
287  {
288  if (DoCastSpellIfCan(m_creature, SPELL_DELIRIOUS_SLASH) == CAST_OK)
289  m_uiDeliriousSlashTimer = 15000;
290  }
291  else
292  m_uiDeliriousSlashTimer -= uiDiff;
293  }
294 
295  DoMeleeAttackIfReady();
296 
297  break;
298  }
299  case PHASE_RUNNING:
300  case PHASE_FLYING:
301  {
302  // Nothing here. Wait for arriving at the point
303  break;
304  }
305  case PHASE_AIR:
306  {
307  if (m_uiPhaseTimer < uiDiff)
308  {
309  m_uiPhase = PHASE_FLYING;
310  m_uiPhaseTimer = 0;
311 
312  m_creature->GetMotionMaster()->Clear();
313  m_creature->GetMotionMaster()->MovePoint(POINT_CENTER_GROUND, aQueenPosition[0][0], aQueenPosition[0][1], aQueenPosition[0][2]);
314  }
315  else
316  m_uiPhaseTimer -= uiDiff;
317 
318  break;
319  }
320  }
321  }
