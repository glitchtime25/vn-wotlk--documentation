127  void DamageTaken(Unit* /*pDealer*/, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
128  {
129  if (uiDamage >= m_creature->GetHealth())
130  {
131  uiDamage = 0;
132 
133  if (m_bDefeated)
134  return;
135 
136  if (!m_pInstance)
137  return;
138 
139  // second part of the champions challenge
140  if (m_pInstance->GetData(TYPE_ARENA_CHALLENGE) == DONE)
141  {
142  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
143  m_creature->InterruptNonMeleeSpells(false);
144  m_creature->SetStandState(UNIT_STAND_STATE_KNEEL);
145  m_creature->SetHealth(1);
146 
147  // no movement
148  SetCombatMovement(false);
149  m_creature->GetMotionMaster()->Clear();
150  m_creature->GetMotionMaster()->MoveIdle();
151 
152  // check if the other champions are wounded and set instance data
153  if (m_pInstance->IsArenaChallengeComplete(TYPE_GRAND_CHAMPIONS))
154  m_pInstance->SetData(TYPE_GRAND_CHAMPIONS, DONE);
155  }
156  // first part of the champions challenge (arena encounter)
157  else
158  {
159  // unmount
160  if (Creature* pMount = (Creature*)m_creature->GetTransportInfo()->GetTransport())
161  {
162  pMount->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
163  pMount->ForcedDespawn();
164  }
165 
166  m_creature->SetStandState(UNIT_STAND_STATE_DEAD);
167  m_creature->SetHealth(1);
168 
169  // no movement
170  SetCombatMovement(false);
171  m_creature->GetMotionMaster()->Clear();
172  m_creature->GetMotionMaster()->MoveIdle();
173 
174  m_uiDefeatedTimer = 15000;
175  }
176 
177  m_bDefeated = true;
178  }
179  }

181  void MovementInform(uint32 uiMotionType, uint32 uiPointId) override
182  {
183  if (uiMotionType != POINT_MOTION_TYPE || !m_pInstance)
184  return;
185 
186  switch (uiPointId)
187  {
188  case POINT_ID_MOUNT:
189  {
190  // mount the closest vehicle and start attacking
191  uint32 uiMountEntry = m_pInstance->GetMountEntryForChampion();
192 
193  // search for the vehicle again, just in case the previous one was taken
194  Creature* pMount = m_creature->GetMap()->GetCreature(m_newMountGuid);
195  if (pMount->HasAura(SPELL_RIDE_ARGENT_VEHICLE))
196  pMount = GetClosestCreatureWithEntry(m_creature, uiMountEntry, 60.0f);
197 
198  // if we don't have any mount send an error
199  if (!pMount)
200  {
201  script_error_log("Instance Trial of the Champion: ERROR Failed to get a mount replacement for champion %u.", m_creature->GetEntry());
202  return;
203  }
204 
205  DoCastSpellIfCan(pMount, SPELL_RIDE_ARGENT_VEHICLE, CAST_TRIGGERED);
206 
207  if (m_creature->GetVictim())
208  pMount->AI()->AttackStart(m_creature->GetVictim());
209 
210  m_bDefeated = false;
211  break;
212  }
213  case POINT_ID_EXIT:
214  // mark the first part as complete if required
215  if (m_pInstance->GetData(TYPE_GRAND_CHAMPIONS) != DONE)
216  m_pInstance->SetData(TYPE_ARENA_CHALLENGE, DONE);
217 
218  m_creature->ForcedDespawn();
219  break;
220  }
221  }

244  // function that will make the champion to use the nearby available mount
245  void DoUseNearbyMountIfCan()
246  {
247  if (!m_pInstance)
248  return;
249 
250  // set instance data as special if first part is completed
251  if (m_pInstance->IsArenaChallengeComplete(TYPE_ARENA_CHALLENGE))
252  m_pInstance->SetData(TYPE_ARENA_CHALLENGE, SPECIAL);
253  else
254  {
255  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
256 
257  float fX, fY, fZ;
258  uint32 uiMountEntry = m_pInstance->GetMountEntryForChampion();
259 
260  if (Creature* pMount = GetClosestCreatureWithEntry(m_creature, uiMountEntry, 60.0f))
261  {
262  pMount->GetContactPoint(m_creature, fX, fY, fZ);
263  m_creature->SetWalk(true);
264  m_creature->GetMotionMaster()->Clear();
265  m_creature->GetMotionMaster()->MovePoint(POINT_ID_MOUNT, fX, fY, fZ);
266 
267  m_newMountGuid = pMount->GetObjectGuid();
268  }
269  }

275  void UpdateAI(const uint32 uiDiff) override
276  {
277  // Return since we have no target
278  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
279  return;
280 
281  // timer for other champions check
282  if (m_uiDefeatedTimer)
283  {
284  if (m_uiDefeatedTimer <= uiDiff)
285  {
286  DoUseNearbyMountIfCan();
287  m_uiDefeatedTimer = 0;
288  }
289  else
290  m_uiDefeatedTimer -= uiDiff;
291  }
292 
293  // no combat after defeated
294  if (m_bDefeated)
295  return;
296 
297  if (!m_pInstance)
298  return;
299 
300  // arena battle - on vehicles
301  if (m_pInstance->GetData(TYPE_ARENA_CHALLENGE) == IN_PROGRESS)
302  {
303  if (m_uiShieldBreakerTimer < uiDiff)
304  {
305  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SHIELD_BREAKER) == CAST_OK)
306  m_uiShieldBreakerTimer = urand(2000, 4000);
307  }
308  else
309  m_uiShieldBreakerTimer -= uiDiff;
310 
311  if (m_uiChargeTimer)
312  {
313  if (m_uiChargeTimer <= uiDiff)
314  {
315  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CHARGE) == CAST_OK)
316  {
317  if (m_creature->GetTransportInfo() && m_creature->GetTransportInfo()->IsOnVehicle())
318  {
319  if (Creature* pMount = (Creature*)m_creature->GetTransportInfo()->GetTransport())
320  SendAIEvent(AI_EVENT_CUSTOM_A, m_creature->GetVictim(), pMount);
321  }
322  m_uiChargeTimer = 0;
323  }
324  }
325  else
326  m_uiChargeTimer -= uiDiff;
327  }
328  }
329  // arena challenge complete - start normal battle
330  else if (m_pInstance->GetData(TYPE_ARENA_CHALLENGE) == DONE)
331  {
332  // Call specific virtual function
333  if (!UpdateChampionAI(uiDiff))
334  return;
335 
336  // Change target
337  if (m_uiResetThreatTimer < uiDiff)
338  {
339  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
340  {
341  DoResetThreat();
342  AttackStart(pTarget);
343  m_uiResetThreatTimer = urand(5000, 15000);
344  }
345  }
346  else
347  m_uiResetThreatTimer -= uiDiff;
348 
349  DoMeleeAttackIfReady();
350  }
351  }
352 };
