316  m_creature->CastSpell(m_creature, SPELL_PLUCKY_HUMAN, TRIGGERED_NONE);

322  if (m_creature->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
323  return;
324  m_creature->SetFactionTemporary(FACTION_FRIENDLY, TEMPFACTION_RESTORE_RESPAWN | TEMPFACTION_RESTORE_COMBAT_STOP);
325  m_creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);

328  }

331  void UpdateAI(const uint32 uiDiff) override
332  {
333  if (m_creature->HasFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP))
334  {
335  if (m_uiResetTimer < uiDiff)
336  {
337  if (!m_creature->GetVictim())
338  EnterEvadeMode();
339  else
340  m_creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
341 
342  return;
343  }
344  m_uiResetTimer -= uiDiff;
345  }
346 
347  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
348  return;

350  DoMeleeAttackIfReady();
