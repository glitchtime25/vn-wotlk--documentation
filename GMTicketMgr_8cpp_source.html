39 bool GMTicket::Compare(GMTicket* a, GMTicket* b)
40 {
41  // 1) Open over closed (abandoned are stored in memory for recycling and should never appear in listings)
42  if (a->IsOpen() && !b->IsOpen())
43  return true;
44 
45  // 2) Escalated over normal: should be picked up by another GM asap
46  if (a->IsEscalated() && !b->IsEscalated())
47  return true;
48 
49  // 3) Read over unread: should be dealt with asap
50  if (a->IsSeen() && !b->IsSeen())
51  return true;
52 
53  // Final: FIFO - older tickets over newer ones
54  return (a->GetCreatedAt() < b->GetCreatedAt());
55 }

329 {
330  // For reload case: unload all tickets
331  RemoveAll();
332 
333  // Try to get highest ticket id to start counting from
334  std::unique_ptr<QueryResult> max(CharacterDatabase.Query("SELECT MAX(id) FROM gm_tickets"));
335  if (max)
336  m_lastTicketId = max->Fetch()->GetUInt32();
337 
338  // Load open tickets into memory
339  std::unique_ptr<QueryResult> result(CharacterDatabase.Query(
340  "SELECT "
341  "id, category, state, status, level, author_guid, author_name, locale, mapid, x, y, z, o, text, created, updated, seen, answered, closed, assignee_guid, assignee_name, conclusion, notes"
342  " FROM gm_tickets WHERE state = 0"
343  ));
344 
345  if (!result)
346  {
347  BarGoLink bar(1);
348  bar.step();
349  sLog.outString(">> Loaded 0 GM tickets");
350  sLog.outString();
351  return;
352  }
353 
354  BarGoLink bar(result->GetRowCount());
355 
356  do
357  {
358  bar.step();
359 
360  Field* fields = result->Fetch();
361 
362  GMTicket* ticket = new GMTicket(fields);
363  Add(ticket, true);
364  } while (result->NextRow());
365 
366  m_list.sort(GMTicket::Compare);
367 
368  sLog.outString(">> Loaded " SIZEFMTD " GM tickets", GetTicketCount());
369  sLog.outString();
