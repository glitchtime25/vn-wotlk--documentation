56 void instance_blackwing_lair::OnCreatureCreate(Creature* pCreature)

62  break;
63  case NPC_BLACKWING_LEGIONNAIRE:
64  case NPC_BLACKWING_MAGE:
65  case NPC_DRAGONSPAWN:

68  m_uiDragonspawnCount++;
69  else
70  m_uiBlackwingDefCount++;
71  // Egg room defenders attack players and Razorgore on spawn
72  m_lDefendersGuids.push_back(pCreature->GetObjectGuid());
73  break;
74  // Nefarian encounter
75  case NPC_BLACK_SPAWNER:
76  case NPC_RED_SPAWNER:
77  case NPC_GREEN_SPAWNER:
78  case NPC_BRONZE_SPAWNER:
79  case NPC_BLUE_SPAWNER:
80  m_lDrakonidSpawnerGuids.push_back(pCreature->GetObjectGuid());
81  break;
82  case NPC_BLACK_DRAKONID:
83  case NPC_RED_DRAKONID:
84  case NPC_GREEN_DRAKONID:
85  case NPC_BLUE_DRAKONID:
86  case NPC_BRONZE_DRAKONID:
87  case NPC_CHROMATIC_DRAKONID:
88  m_drakonids.push_back(pCreature->GetObjectGuid());
89  break;
90  case NPC_LORD_VICTOR_NEFARIUS:
91  if (!pCreature->IsTemporarySummon())
92  m_npcEntryGuidStore[pCreature->GetEntry()] = pCreature->GetObjectGuid();
93  break;

95  case NPC_NEFARIANS_TROOPS:

145 void instance_blackwing_lair::SetData(uint32 uiType, uint32 uiData)
146 {
147  // Close de the main gate whenever an event starts (if it is not already open)
148  if (m_bIsMainGateOpen && (uiData == IN_PROGRESS || uiData == SPECIAL))
149  {
150  DoUseDoorOrButton(GO_DOOR_RAZORGORE_ENTER);
151  m_bIsMainGateOpen = false;
152  }
153  // If an encounter is failed or won, open the main gate only if it is currently closed and no other event is in progress
154  else if (!m_bIsMainGateOpen && (uiData == FAIL || uiData == DONE))
155  {
156  bool ShouldKeepGateClosed = false;
157  for (uint8 i = 0; i < TYPE_NEFARIAN; i++)
158  {
159  if (uiType != i && (m_auiEncounter[i] == IN_PROGRESS || m_auiEncounter[i] == SPECIAL))
160  ShouldKeepGateClosed = true;
161  }
162 
163  if (!ShouldKeepGateClosed)
164  {
165  DoUseDoorOrButton(GO_DOOR_RAZORGORE_ENTER);
166  m_bIsMainGateOpen = true;
167  }
168  }
169 
170  switch (uiType)
171  {
172  case TYPE_RAZORGORE:
173  m_auiEncounter[uiType] = uiData;
174  if (uiData == DONE)
175  DoUseDoorOrButton(GO_DOOR_RAZORGORE_EXIT);
176  else if (uiData == FAIL)
177  {
178  m_uiResetTimer = 30000;
179 
180  if (Creature* razorgore = GetSingleCreatureFromStorage(NPC_RAZORGORE))
181  {
182  DoScriptText(SAY_RAZORGORE_DEATH, razorgore);
183  razorgore->CastSpell(nullptr, SPELL_FIREBALL, TRIGGERED_OLD_TRIGGERED);
184  razorgore->ForcedDespawn(5000);
185  }
186 
187  // Reset the Orb of Domination and the eggs
188  DoToggleGameObjectFlags(GO_ORB_OF_DOMINATION, GO_FLAG_NO_INTERACT, true);
189  if (Creature* pOrb = GetSingleCreatureFromStorage(NPC_BLACKWING_ORB_TRIGGER))
190  {
191  if (pOrb->IsAlive())
192  pOrb->AI()->EnterEvadeMode();
193  }
194 
195  // Reset defenders
196  for (GuidList::const_iterator itr = m_lDefendersGuids.begin(); itr != m_lDefendersGuids.end(); ++itr)
197  {
198  if (Creature* pDefender = instance->GetCreature(*itr))
199  pDefender->ForcedDespawn();
200  }
201 
202  m_lUsedEggsGuids.clear();
203  m_lDefendersGuids.clear();
204  m_uiBlackwingDefCount = 0;
205  m_uiDragonspawnCount = 0;
206  }
207  break;
208  case TYPE_VAELASTRASZ:
209  m_auiEncounter[uiType] = uiData;
210  // Prevent the players from running back to the first room; use if the encounter is not special
211  if (uiData != SPECIAL)
212  DoUseDoorOrButton(GO_DOOR_RAZORGORE_EXIT);
213  if (uiData == DONE)
214  DoUseDoorOrButton(GO_DOOR_VAELASTRASZ);
215  break;
216  case TYPE_LASHLAYER:
217  m_auiEncounter[uiType] = uiData;
218  if (uiData == DONE)
219  DoUseDoorOrButton(GO_DOOR_LASHLAYER);
220  break;
221  case TYPE_FIREMAW:
222  case TYPE_EBONROC:
223  case TYPE_FLAMEGOR:
224  m_auiEncounter[uiType] = uiData;
225  break;
226  case TYPE_CHROMAGGUS:
227  m_auiEncounter[uiType] = uiData;
228  if (uiData == DONE)
229  DoUseDoorOrButton(GO_DOOR_CHROMAGGUS_EXIT);
230  break;
231  case TYPE_NEFARIAN:
232  // If epic quest for Scepter of the Shifting Sands is in progress when Nefarian is defeated mark it as complete
233  if (uiData == DONE && GetData(TYPE_QUEST_SCEPTER) == IN_PROGRESS)
234  SetData(TYPE_QUEST_SCEPTER, DONE);
235 
236  // Don't store the same thing twice
237  if (m_auiEncounter[uiType] == uiData)
238  break;
239 
240  if (uiData == SPECIAL)
241  {
242  m_auiEncounter[uiType] = uiData;
243  m_uiNefarianSpawnTimer = 5 * IN_MILLISECONDS; // End of phase 1: spawn Nefarian 5 seconds later
244 
245  // Remove drakonids spawners and Lord Victor Nefarius
246  CleanupNefarianStage(false);
247  break;
248  }
249  if (uiData == FAIL)
250  CleanupNefarianStage(true); // Cleanup the drakonid bones, dead drakonids count and spawners
251  m_auiEncounter[uiType] = uiData;
252  DoUseDoorOrButton(GO_DOOR_NEFARIAN);
253  break;
254  case TYPE_QUEST_SCEPTER:
255  m_auiEncounter[uiType] = uiData;
256  // Start 5 hours timer (various steps are handled in Update()
257  if (uiData == IN_PROGRESS)
258  {
259  m_uiScepterEpicTimer = 2000;
260  m_uiScepterQuestStep = 0;
261  }
262  // Stop timer
263  if (uiData == DONE)
264  m_uiScepterEpicTimer = 0;
265  break;
266  case TYPE_CHROMA_LBREATH:
267  case TYPE_CHROMA_RBREATH:
268  case TYPE_NEFA_LTUNNEL:
269  case TYPE_NEFA_RTUNNEL:
270  m_auiEncounter[uiType] = uiData; // Store the spell IDs/NPC entries of the two breaths/drakonids used by Chromaggus/Nefarian for all the instance lifetime. Breaths/Drakonids are picked randomly in Chromaggus/Nefarian script
271  break;
272  }
273 
274  if (uiData >= DONE)
275  {
276  OUT_SAVE_INST_DATA;
277 
278  std::ostringstream saveStream;
279  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
280  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
281  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
282  << m_auiEncounter[9] << " " << m_auiEncounter[10]<< " " << m_auiEncounter[11] << " "
283  << m_auiEncounter[12];
284 
285  m_strInstData = saveStream.str();
286 
287  SaveToDB();
288  OUT_SAVE_INST_DATA_COMPLETE;
289  }
290 }

404  case NPC_GREEN_DRAKONID:
405  case NPC_BRONZE_DRAKONID:
406  case NPC_RED_DRAKONID:
407  case NPC_BLACK_DRAKONID:
408  case NPC_CHROMATIC_DRAKONID:
409  m_uiDeadDrakonidsCount++;
410  // If the requiered amount of drakonids are killed, start phase 2

462 void instance_blackwing_lair::Update(uint32 uiDiff)
463 {
464  // Scepter of the Shifting Sand epic quest line
465  if (m_uiScepterEpicTimer)
466  {
467  if (m_uiScepterEpicTimer <= uiDiff)
468  {
469  switch (m_uiScepterQuestStep)
470  {
471  case 0: // On quest acceptance
472  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_1, NPC_LORD_VICTOR_NEFARIUS);
473  m_uiScepterEpicTimer = 2 * HOUR * IN_MILLISECONDS;
474  break;
475  case 1: // 2 hours time mark
476  switch (urand(0, 1))
477  {
478  case 0:
479  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_2, NPC_LORD_VICTOR_NEFARIUS);
480  DoOrSimulateScriptTextForThisInstance(EMOTE_REDSHARD_TAUNT_1, NPC_LORD_VICTOR_NEFARIUS);
481  break;
482  case 1:
483  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_3, NPC_LORD_VICTOR_NEFARIUS);
484  break;
485  }
486  m_uiScepterEpicTimer = 2 * HOUR * IN_MILLISECONDS;
487  break;
488  case 2: // 1 hour left
489  switch (urand(0, 1))
490  {
491  case 0:
492  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_4, NPC_LORD_VICTOR_NEFARIUS);
493  break;
494  case 1:
495  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_5, NPC_LORD_VICTOR_NEFARIUS);
496  break;
497  }
498  m_uiScepterEpicTimer = 30 * MINUTE * IN_MILLISECONDS;
499  break;
500  case 3: // 30 min left
501  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_6, NPC_LORD_VICTOR_NEFARIUS);
502  m_uiScepterEpicTimer = 30 * MINUTE * IN_MILLISECONDS;
503  break;
504  case 4: // Failure
505  SetData(TYPE_QUEST_SCEPTER, FAIL);
506  if (GetData(TYPE_NEFARIAN) == NOT_STARTED)
507  {
508  DoOrSimulateScriptTextForThisInstance(EMOTE_REDSHARD_TAUNT_2, NPC_LORD_VICTOR_NEFARIUS);
509  DoOrSimulateScriptTextForThisInstance(YELL_REDSHARD_TAUNT_7, NPC_LORD_VICTOR_NEFARIUS);
510  }
511  default: // Something weird happened: stop timer and fail the event
512  m_uiScepterEpicTimer = 0;
513  SetData(TYPE_QUEST_SCEPTER, FAIL);
514  break;
515  }
516  m_uiScepterQuestStep++;
517  }
518  else
519  m_uiScepterEpicTimer -= uiDiff;
520  }
521  // Spawn Nefarian
522  if (m_uiNefarianSpawnTimer)
523  {
524  if (m_uiNefarianSpawnTimer <= uiDiff)
525  {
526  if (Creature* nefarius = GetSingleCreatureFromStorage(NPC_LORD_VICTOR_NEFARIUS))
527  {
528  nefarius->SummonCreature(NPC_NEFARIAN, aNefarianLocs[2].m_fX, aNefarianLocs[2].m_fY, aNefarianLocs[2].m_fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0, true);
529  // Despawn Nefarius
530  if (!nefarius->IsDespawned() && GetData(TYPE_NEFARIAN) == SPECIAL)
531  {
532  nefarius->CastSpell(nullptr, SPELL_SHADOWBLINK_OUTRO, TRIGGERED_OLD_TRIGGERED);
533  nefarius->AI()->SetCombatScriptStatus(true);
534  nefarius->ForcedDespawn(2500);
535  }
536  }
537  m_uiNefarianSpawnTimer = 0;
538  }
539  else
540  m_uiNefarianSpawnTimer -= uiDiff;
541  }
542 
543  // Reset Razorgore in case of wipe
544  if (m_uiResetTimer)
545  {
546  if (m_uiResetTimer <= uiDiff)
547  {
548  // Respawn Razorgore
549  if (Creature* pRazorgore = GetSingleCreatureFromStorage(NPC_RAZORGORE))
550  {
551  if (!pRazorgore->IsAlive())
552  pRazorgore->Respawn();
553  }
554 
555  // Respawn the Dragon Eggs
556  for (GuidList::const_iterator itr = m_lDragonEggsGuids.begin(); itr != m_lDragonEggsGuids.end(); ++itr)
557  {
558  if (GameObject* pEgg = instance->GetGameObject(*itr))
559  {
560  if (!pEgg->IsSpawned())
561  pEgg->Respawn();
562  }
563  }
564 
565  m_uiResetTimer = 0;
566  }
567  else
568  m_uiResetTimer -= uiDiff;
569  }
570 
571  if (GetData(TYPE_RAZORGORE) != IN_PROGRESS)
572  return;
573 
574  if (m_uiDefenseTimer < uiDiff)
575  {
576  // Randomize generators
577  std::shuffle(m_vGeneratorGuids.begin(), m_vGeneratorGuids.end(), *GetRandomGenerator());
578 
579  // Spawn the defenders
580  for (uint8 i = 0; i < MAX_EGGS_DEFENDERS; ++i)
