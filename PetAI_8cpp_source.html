97 void PetAI::AttackStart(Unit* who)
98 {
99  Pet* pet = (m_unit->GetTypeId() == TYPEID_UNIT && static_cast<Creature*>(m_unit)->IsPet()) ? static_cast<Pet*>(m_unit) : nullptr;
100 
101  if (!who || (pet && pet->GetModeFlags() & PET_MODE_DISABLE_ACTIONS))
102  return;
103 
104  // Do not start attack if target is moving home
105  if (who->GetCombatManager().IsEvadingHome())
106  return;
107 
108  if (m_unit->Attack(who, m_meleeEnabled))
109  {
110  // TMGs call CreatureRelocation which via MoveInLineOfSight can call this function
111  // thus with the following clear the original TMG gets invalidated and crash, doh
112  // hope it doesn't start to leak memory without this :-/
113  // i_pet->Clear();
114  HandleMovementOnAttackStart(who, false);
115 
116  inCombat = true;
117  }

135 void PetAI::UpdateAI(const uint32 diff)
136 {
137  UpdateTimers(diff, m_creature->IsInCombat());
138 
139  Creature* creature = (m_unit->GetTypeId() == TYPEID_UNIT) ? static_cast<Creature*>(m_unit) : nullptr;
140  Pet* pet = (creature && creature->IsPet()) ? static_cast<Pet*>(m_unit) : nullptr;
141 
142  Unit* owner = m_unit->GetMaster();
143  if (!owner)
144  return;
145 
146  if (m_updateAlliesTimer <= diff)
147  // UpdateAllies self set update timer
148  UpdateAllies();
149  else
150  m_updateAlliesTimer -= diff;
151 
152  Unit* victim = (pet && pet->GetModeFlags() & PET_MODE_DISABLE_ACTIONS) ? nullptr : m_unit->GetVictim();
153 
154  // Do not continue attacking if victim is moving home
155  if (victim && victim->GetCombatManager().IsEvadingHome())
156  victim = nullptr;
157 
158  CharmInfo* charminfo = m_unit->GetCharmInfo();
159  MANGOS_ASSERT(charminfo);
160 
161  // Stop auto attack and chase if victim was dropped
162  if (inCombat && (!victim || (victim->IsCrowdControlled() && victim->HasAuraPetShouldAvoidBreaking(pet, charminfo->GetPetLastAttackCommandTime()))))
163  {
164  m_unit->AttackStop(true, true);
165  inCombat = false;
166 
167  MotionMaster* mm = m_unit->GetMotionMaster();
168 
169  if (mm->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
170  mm->MovementExpired();
171  }
172 
173  if (charminfo->GetIsRetreating())
174  {
175  if (!owner->IsWithinDistInMap(m_unit, (PET_FOLLOW_DIST * 2)))
176  {
177  if (!m_unit->hasUnitState(UNIT_STAT_FOLLOW))
178  m_unit->GetMotionMaster()->MoveFollow(owner, m_followDist, m_followAngle);
179 
180  return;
181  }
182  charminfo->SetIsRetreating();
183  }
184 
185  // Stop here if casting spell (No melee and no movement)
186  if (m_unit->IsNonMeleeSpellCasted(false))
187  return;
188  // Auto cast (casted only in combat or persistent spells in any state)
189  else
190  {
191  typedef std::vector<std::pair<Unit*, Spell*> > TargetSpellList;
192  TargetSpellList targetSpellStore;
193  if (charminfo->GetSpellOpener() != 0) // have opener stored
194  {
195  uint32 minRange = charminfo->GetSpellOpenerMinRange();
196  victim = m_unit->GetVictim();
197 
198  if (!victim || (minRange != 0 && m_unit->IsWithinDistInMap(victim, minRange)))
199  charminfo->SetSpellOpener();
200  else if (m_unit->IsWithinDistInMap(victim, charminfo->GetSpellOpenerMaxRange())
201  && m_unit->IsWithinLOSInMap(victim, true))
202  {
203  uint32 spellId = charminfo->GetSpellOpener();
204  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
205  uint32 flags = TRIGGERED_NORMAL_COMBAT_CAST;
206  if (m_creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
207  flags |= TRIGGERED_PET_CAST;
208 
209  Spell* spell = new Spell(m_unit, spellInfo, flags);
210 
211  // Push back stored spell
212  targetSpellStore.push_back(TargetSpellList::value_type(victim, spell));
213 
214  // Clear spell opener
215  charminfo->SetSpellOpener();
216  }
217  }
218  else if (creature)
219  {
220  for (uint8 i = 0; i < creature->GetPetAutoSpellSize(); ++i)
221  {
222  uint32 spellId = creature->GetPetAutoSpellOnPos(i);
223  if (!spellId)
224  continue;
225 
226  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
227  if (!spellInfo)
228  continue;
229 
230  if (!m_unit->IsSpellReady(*spellInfo))
231  continue;
232 
233  // ignore some combinations of combat state and combat/non combat spells
234  if (!inCombat)
235  {
236  // ignore attacking spells, and allow only self/around spells
237  if (!IsPositiveSpell(spellInfo->Id))
238  continue;
239 
240  // non combat spells allowed
241  // only pet spells have IsNonCombatSpell and not fit this requirements:
242  // Consume Shadows, Lesser Invisibility, so ignore checks for its
243  if (!IsNonCombatSpell(spellInfo))
244  {
245  int32 duration = GetSpellDuration(spellInfo);
246  int32 cooldown = GetSpellRecoveryTime(spellInfo);
247 
248  // allow only spell not on cooldown
249  if (cooldown != 0 && duration < cooldown)
250  continue;
251  }
252  }
253  // just ignore non-combat spells
254  else if (IsNonCombatSpell(spellInfo))
255  continue;
256 
257  Spell* spell = new Spell(m_unit, spellInfo, TRIGGERED_NORMAL_COMBAT_CAST);
258 
259  if (inCombat && spell->CanAutoCast(IsOnlySelfTargeting(spellInfo) ? m_unit : victim))
260  {
261  targetSpellStore.push_back(TargetSpellList::value_type(victim, spell));
262  }
263  else if (!IsSpellRequireTarget(spellInfo))
264  {
265  if (!spell->CanAutoCast(m_unit))
266  {
267  delete spell;
268  continue;
269  }
270 
271  targetSpellStore.push_back(TargetSpellList::value_type(nullptr, spell));
272  }
273  else
274  {
275  bool spellUsed = false;
276  for (auto tar : m_AllySet)
277  {
278  Unit* Target = m_unit->GetMap()->GetUnit(tar);
279 
280  // only buff targets that are in combat, unless the spell can only be cast while out of combat
281  if (!Target)
282  continue;
283 
284  if (spell->CanAutoCast(Target))
285  {
286  targetSpellStore.push_back(TargetSpellList::value_type(Target, spell));
287  spellUsed = true;
288  break;
289  }
290  }
291  if (!spellUsed)
292  delete spell;
293  }
294  }
295  }
296 
297  // found units to cast on to
298  if (!targetSpellStore.empty())
299  {
300  uint32 index = urand(0, targetSpellStore.size() - 1);
301 
302  Spell* spell = targetSpellStore[index].second;
303  Unit* target = targetSpellStore[index].first;
304 
305  targetSpellStore.erase(targetSpellStore.begin() + index);
306 
307  SpellCastTargets targets;
308  if (target)
309  {
310  targets.setUnitTarget(target);
311 
312  if (!m_unit->HasInArc(target))
313  m_unit->SetFacingToObject(target);
314  }
315 
316  spell->SpellStart(&targets);
317  }
318 
319  // deleted cached Spell objects
320  for (TargetSpellList::const_iterator itr = targetSpellStore.begin(); itr != targetSpellStore.end(); ++itr)
321  delete itr->second;
322  }
323 
324  // we may get our actions disabled during spell casting, so do entire recheck for victim
325  victim = (pet && pet->GetModeFlags() & PET_MODE_DISABLE_ACTIONS) ? nullptr : m_unit->GetVictim();
326 
327  if (victim)
328  {
329  // i_pet.GetVictim() can't be used for check in case stop fighting, i_pet.GetVictim() clear at Unit death etc.
330  // This is needed for charmed creatures, as once their target was reset other effects can trigger threat
331  if (!m_unit->CanAttack(victim))
332  {
333  DEBUG_FILTER_LOG(LOG_FILTER_AI_AND_MOVEGENSS, "PetAI (guid = %u) is stopping attack.", m_unit->GetGUIDLow());
334  m_unit->CombatStop();
335  inCombat = false;
336 
337  return;
338  }
339 
340  // if pet misses its target, it will also be the first in threat list
341  if ((!creature || m_meleeEnabled)
342  && m_unit->CanReachWithMeleeAttack(victim))
343  {
344  if (!m_unit->HasInArc(victim, 2 * M_PI_F / 3))
345  m_unit->SetFacingToObject(victim);
346 
347  DoMeleeAttackIfReady();
348  }
349  else if (!m_unit->hasUnitState(UNIT_STAT_MOVING))
350  AttackStart(victim);
351  }
352  else if (!m_unit->hasUnitState(UNIT_STAT_CAN_NOT_REACT_OR_LOST_CONTROL))
353  {
354  CharmInfo* charmInfo = m_unit->GetCharmInfo();
355 
356  const bool staying = (charmInfo && charmInfo->HasCommandState(COMMAND_STAY));
357  const bool following = (!staying && charmInfo && charmInfo->HasCommandState(COMMAND_FOLLOW));
358 
359  // If not commanded to stay, try to assist owner first
360  if (!staying && owner->IsInCombat() && !HasReactState(REACT_PASSIVE))
361  {
362  AttackStart(owner->getAttackerForHelper());
363 
364  // If target was acquired, skip non-combat movement handling
365  if (inCombat)
366  return;
367  }
368 
369  // Handle non-combat movement
370  if (!m_unit->hasUnitState(UNIT_STAT_NO_FREE_MOVE | UNIT_STAT_CHASE))
371  {
372  MotionMaster* mm = m_unit->GetMotionMaster();
373 
374  if (staying && !m_unit->hasUnitState(UNIT_STAT_STAY))
375  {
376  // If stay command is set, but we don't have stay pos yet: establish current pos as stay position, adjust orientation
377  if (charminfo->UpdateStayPosition())
378  mm->MoveStay(charminfo->GetStayPosX(), charminfo->GetStayPosY(), charminfo->GetStayPosZ(), charminfo->GetStayPosO());
379  else
380  mm->MoveStay(charminfo->GetStayPosX(), charminfo->GetStayPosY(), charminfo->GetStayPosZ());
381  }
382  else if (following && !m_unit->hasUnitState(UNIT_STAT_FOLLOW))
383  mm->MoveFollow(owner, m_followDist, m_followAngle);
384  }
385  }
386 }

388 void PetAI::UpdateAllies()
389 {
390  Unit* owner = m_unit->GetMaster();
391  Group* group = nullptr;
392 
393  m_updateAlliesTimer = 10 * IN_MILLISECONDS; // update friendly targets every 10 seconds, fewer checks increase performance
394 
395  if (!owner)
396  return;
397  if (owner->GetTypeId() == TYPEID_PLAYER)
398  group = ((Player*)owner)->GetGroup();
399 
400  // only pet and owner/not in group->ok
401  if (m_AllySet.size() == 2 && !group)
402  return;
403  // owner is in group; group members filled in already (no raid -> subgroupcount = whole count)
404  if (group && !group->IsRaidGroup() && m_AllySet.size() == (group->GetMembersCount() + 2))
405  return;
406 
407  m_AllySet.clear();
408  m_AllySet.insert(m_unit->GetObjectGuid()); // add self
409  m_AllySet.insert(owner->GetObjectGuid()); // add owner
410 
411  if (group) // add group
412  {
413  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
414  {
415  Player* target = itr->getSource();
416  if (!target || !group->SameSubGroup((Player*)owner, target))
417  continue;
418 
419  if (target->GetObjectGuid() == owner->GetObjectGuid())
420  continue;
421 
422  m_AllySet.insert(target->GetObjectGuid());
423  }
424  }
