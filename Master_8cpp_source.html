58 class FreezeDetectorRunnable : public MaNGOS::Runnable
59 {
60  public:
61  FreezeDetectorRunnable() { _delaytime = 0; }
62  uint32 m_loops, m_lastchange;
63  uint32 w_loops, w_lastchange;
64  uint32 _delaytime;
65  void SetDelayTime(uint32 t) { _delaytime = t; }
66  void run(void)
67  {
68  if (!_delaytime)
69  return;
70  sLog.outString("Starting up anti-freeze thread (%u seconds max stuck time)...", _delaytime / 1000);
71  m_loops = 0;
72  w_loops = 0;
73  m_lastchange = 0;
74  w_lastchange = 0;
75  while (!World::IsStopped())
76  {
77  MaNGOS::Thread::Sleep(1000);
78 
79  uint32 curtime = WorldTimer::getMSTime();
80 
81  // normal work
82  if (w_loops != World::m_worldLoopCounter)
83  {
84  w_lastchange = curtime;
85  w_loops = World::m_worldLoopCounter;
86  }
87  // possible freeze
88  else if (WorldTimer::getMSTimeDiff(w_lastchange, curtime) > _delaytime)
89  {
90  sLog.outError("World Thread hangs, kicking out server!");
91  *((uint32 volatile*)nullptr) = 0; // bang crash
92  }
93  }
94  sLog.outString("Anti-freeze thread exiting without problems.");
95  }
96 };

479 void Master::_OnSignal(int s)
480 {
481  switch (s)
482  {
483  case SIGINT:
484  World::StopNow(RESTART_EXIT_CODE);
485  break;
486  case SIGTERM:
487 #ifdef _WIN32
488  case SIGBREAK:
489 #endif
490  World::StopNow(SHUTDOWN_EXIT_CODE);
491  break;
492  }
493 
494  // give a 30 sec timeout in case of Master cannot finish properly
495  int32 timeOut = 200;
496  while (!m_canBeKilled && timeOut > 0)
497  {
498  std::this_thread::sleep_for(std::chrono::milliseconds(100));
499  --timeOut;
500  }
501 
502  signal(s, _OnSignal);
503 }
