91  void UpdateAI(const uint32 uiDiff) override
92  {
93  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
94  return;
95 
96  // If we are < 30% hp enrage
97  if (!m_bEnrage && m_creature->GetHealthPercent() <= 30.0f && !m_creature->IsNonMeleeSpellCasted(false))
98  {
99  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
100  {
101  DoScriptText(EMOTE_GENERIC_ENRAGED, m_creature);
102  DoScriptText(SAY_ENRAGE, m_creature);
103  m_bEnrage = true;
104  }
105  }
106 
107  // Cleave
108  if (m_uiCleaveTimer < uiDiff)
109  {
110  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE);
111  m_uiCleaveTimer = urand(7500, 17500);
112  }
113  else
114  m_uiCleaveTimer -= uiDiff;
115 
116  if (m_uiWhirlwindTimer < uiDiff)
117  {
118  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_WHIRLWIND) == CAST_OK)
119  {
120  DoScriptText(SAY_WHIRLWIND, m_creature);
121  m_uiWhirlwindTimer = urand(15000, 25000);
122  }
123  }
124  else
125  m_uiWhirlwindTimer -= uiDiff;
126 
127  DoMeleeAttackIfReady();
128  }
