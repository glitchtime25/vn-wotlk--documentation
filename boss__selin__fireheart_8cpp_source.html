121  // Get the nearest alive crystal for draining
122  bool DoSelectNearestCrystal()
123  {
124  // Wait to finish casting
125  if (m_creature->IsNonMeleeSpellCasted(false))
126  return false;
127 
128  if (Creature* crystal = GetClosestCreatureWithEntry(m_creature, NPC_FEL_CRYSTAL, 60.0f))
129  {
130  m_crystalGuid = crystal->GetObjectGuid();
131  DoScriptText(urand(0, 1) ? SAY_DRAIN_1 : SAY_DRAIN_2, m_creature);
132 
133  float x, y, z;
134  m_creature->GetContactPoint(crystal, x, y, z, INTERACTION_DISTANCE);
135  m_creature->GetMotionMaster()->MovePoint(POINT_CRYSTAL, x, y, z, FORCED_MOVEMENT_RUN, true);
136  SetCombatScriptStatus(true);
137  SetMeleeEnabled(false);
138  m_creature->SetTarget(nullptr);
139 
140  return true;
141  }
142 
143  return false;

199  void ReceiveAIEvent(AIEventType eventType, Unit* /*sender*/, Unit* invoker, uint32 miscValue) override
200  {

218  void ExecuteAction(uint32 action) override
219  {
220  switch (action)
221  {
222  case SELIN_ACTION_FEL_EXPLOSION:
223  {
224  if (DoCastSpellIfCan(nullptr, SPELL_FEL_EXPLOSION) == CAST_OK)
225  ResetCombatAction(action, GetSubsequentActionTimer(action));
226  return;
227  }
228  case SELIN_ACTION_DRAIN_LIFE:
229  {
230  if (m_creature->GetPower(POWER_MANA) >= m_felExplosionCost) // do not cast if enough mana for fel explosion
231  return;
232 
233  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_DRAIN_LIFE, SELECT_FLAG_IN_LOS | SELECT_FLAG_PLAYER))
234  if (DoCastSpellIfCan(target, m_isRegularMode ? SPELL_DRAIN_LIFE : SPELL_DRAIN_LIFE_H) == CAST_OK)
235  ResetCombatAction(action, GetSubsequentActionTimer(action));
236  return;
237  }
238  case SELIN_ACTION_DRAIN_MANA:
239  {
240  if (m_creature->GetPower(POWER_MANA) >= m_felExplosionCost) // do not cast if enough mana for fel explosion
241  return;
242 
243  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_DRAIN_MANA, SELECT_FLAG_IN_LOS | SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
244  if (DoCastSpellIfCan(target, SPELL_DRAIN_MANA) == CAST_OK)
245  ResetCombatAction(action, GetSubsequentActionTimer(action));
246  return;
247  }
248  case SELIN_ACTION_DRAIN_CRYSTAL:
249  {
250  DoSelectNearestCrystal();
251  return;
252  }
253  }
254  }
