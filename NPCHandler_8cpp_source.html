107 static void SendTrainerSpellHelper(WorldPacket& data, TrainerSpell const* tSpell, TrainerSpellState state, float fDiscountMod, bool /*can_learn_primary_prof*/, uint32 reqLevel)
108 {
109  bool primary_prof_first_rank = sSpellMgr.IsPrimaryProfessionFirstRankSpell(tSpell->learnedSpell);
110 
111  SpellChainNode const* chain_node = sSpellMgr.GetSpellChainNode(tSpell->learnedSpell);
112 
113  data << uint32(tSpell->spell); // learned spell (or cast-spell in profession case)
114  data << uint8(state == TRAINER_SPELL_GREEN_DISABLED ? TRAINER_SPELL_GREEN : state);
115  data << uint32(floor(tSpell->spellCost * fDiscountMod));
116 
117  data << uint32(0); // spells don't cost talent points
118  data << uint32(primary_prof_first_rank ? 1 : 0); // must be equal prev. field to have learn button in enabled state
119  data << uint8(reqLevel);
120  data << uint32(tSpell->reqSkill);
121  data << uint32(tSpell->reqSkillValue);
122  if (tSpell->reqAbility[0])
123  data << *tSpell->reqAbility[0];
124  else
125  data << uint32(!tSpell->IsCastable() && chain_node ? (chain_node->prev ? chain_node->prev : chain_node->req) : 0);
126  if (tSpell->reqAbility[1])
127  data << *tSpell->reqAbility[1];
128  else
129  data << uint32(!tSpell->IsCastable() && chain_node && chain_node->prev ? chain_node->req : 0);
130  if (tSpell->reqAbility[2])
131  data << *tSpell->reqAbility[2];
132  else
133  data << uint32(0);
134 }

136 void WorldSession::SendTrainerList(ObjectGuid guid) const
137 {
138  DEBUG_LOG("WORLD: SendTrainerList");
139 
140  Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
141  if (!unit)
142  {
143  DEBUG_LOG("WORLD: SendTrainerList - %s not found or you can't interact with him.", guid.GetString().c_str());
144  return;
145  }
146 
147  // trainer list loaded at check;
148  if (!unit->IsTrainerOf(_player, true))
149  return;
150 
151  CreatureInfo const* ci = unit->GetCreatureInfo();
152  if (!ci)
153  return;
154 
155  TrainerSpellData const* cSpells = unit->GetTrainerSpells();
156  TrainerSpellData const* tSpells = unit->GetTrainerTemplateSpells();
157 
158  if (!cSpells && !tSpells)
159  {
160  DEBUG_LOG("WORLD: SendTrainerList - Training spells not found for %s", guid.GetString().c_str());
161  return;
162  }
163 
164  uint32 maxcount = (cSpells ? cSpells->spellList.size() : 0) + (tSpells ? tSpells->spellList.size() : 0);
165  uint32 trainer_type = cSpells && cSpells->trainerType ? cSpells->trainerType : (tSpells ? tSpells->trainerType : 0);
166 
167  std::string strTitle;
168  if (TrainerGreeting const* data = sObjectMgr.GetTrainerGreetingData(guid.GetEntry()))
169  {
170  strTitle = data->text;
171  int loc_idx = GetSessionDbLocaleIndex();
172  sObjectMgr.GetTrainerGreetingLocales(guid.GetEntry(), loc_idx, &strTitle);
173  }
174  else
175  strTitle = GetMangosString(LANG_NPC_TAINER_HELLO); // Fallback in case no trainer greeting found
176 
177  WorldPacket data(SMSG_TRAINER_LIST, 8 + 4 + 4 + maxcount * 38 + strTitle.size() + 1);
178  data << ObjectGuid(guid);
179  data << uint32(trainer_type);
180 
181  size_t count_pos = data.wpos();
182  data << uint32(maxcount);
183 
184  // reputation discount
185  float fDiscountMod = _player->GetReputationPriceDiscount(unit);
186  bool can_learn_primary_prof = GetPlayer()->GetFreePrimaryProfessionPoints() > 0;
187 
188  uint32 count = 0;
189 
190  if (cSpells)
191  {
192  for (const auto& itr : cSpells->spellList)
193  {
194  TrainerSpell const* tSpell = &itr.second;
195 
196  uint32 reqLevel = 0;
197  if (!_player->IsSpellFitByClassAndRace(tSpell->learnedSpell, &reqLevel))
198  continue;
199 
200  if (tSpell->conditionId && !sObjectMgr.IsConditionSatisfied(tSpell->conditionId, GetPlayer(), unit->GetMap(), unit, CONDITION_FROM_TRAINER))
201  continue;
202 
203  reqLevel = tSpell->isProvidedReqLevel ? tSpell->reqLevel : std::max(reqLevel, tSpell->reqLevel);
204 
205  TrainerSpellState state = _player->GetTrainerSpellState(tSpell, reqLevel);
206 
207  SendTrainerSpellHelper(data, tSpell, state, fDiscountMod, can_learn_primary_prof, reqLevel);
208 
209  ++count;
210  }
211  }
212 
213  if (tSpells)
214  {
215  for (const auto& itr : tSpells->spellList)
216  {
217  TrainerSpell const* tSpell = &itr.second;
218 
219  uint32 reqLevel = 0;
220  if (!_player->IsSpellFitByClassAndRace(tSpell->learnedSpell, &reqLevel))
221  continue;
222 
223  if (tSpell->conditionId && !sObjectMgr.IsConditionSatisfied(tSpell->conditionId, GetPlayer(), unit->GetMap(), unit, CONDITION_FROM_TRAINER))
224  continue;
225 
226  reqLevel = tSpell->isProvidedReqLevel ? tSpell->reqLevel : std::max(reqLevel, tSpell->reqLevel);
227 
228  TrainerSpellState state = _player->GetTrainerSpellState(tSpell, reqLevel);
229 
230  SendTrainerSpellHelper(data, tSpell, state, fDiscountMod, can_learn_primary_prof, reqLevel);
231 
232  ++count;
233  }
234  }
235 
236  data << strTitle;
237 
238  data.put<uint32>(count_pos, count);
239  SendPacket(data);
240 }

242 void WorldSession::HandleTrainerBuySpellOpcode(WorldPacket& recv_data)
243 {
244  ObjectGuid guid;
245  uint32 spellId = 0;
246 
247  recv_data >> guid >> spellId;
248  DEBUG_LOG("WORLD: Received opcode CMSG_TRAINER_BUY_SPELL Trainer: %s, learn spell id is: %u", guid.GetString().c_str(), spellId);
249 
250  Creature* unit = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_TRAINER);
251  if (!unit)
252  {
253  DEBUG_LOG("WORLD: HandleTrainerBuySpellOpcode - %s not found or you can't interact with him.", guid.GetString().c_str());
254  return;
255  }
256 
257  if (!unit->IsTrainerOf(_player, true))
258  return;
259 
260  // check present spell in trainer spell list
261  TrainerSpellData const* cSpells = unit->GetTrainerSpells();
262  TrainerSpellData const* tSpells = unit->GetTrainerTemplateSpells();
263 
264  if (!cSpells && !tSpells)
265  return;
266 
267  // Try find spell in npc_trainer
268  TrainerSpell const* trainer_spell = cSpells ? cSpells->Find(spellId) : nullptr;
269 
270  // Not found, try find in npc_trainer_template
271  if (!trainer_spell && tSpells)
272  trainer_spell = tSpells->Find(spellId);
273 
274  // Not found anywhere, cheating?
275  if (!trainer_spell)
276  return;
277 
278  // can't be learn, cheat? Or double learn with lags...
279  uint32 reqLevel = 0;
280  if (!_player->IsSpellFitByClassAndRace(trainer_spell->learnedSpell, &reqLevel))
281  return;
282 
283  reqLevel = trainer_spell->isProvidedReqLevel ? trainer_spell->reqLevel : std::max(reqLevel, trainer_spell->reqLevel);
284  if (_player->GetTrainerSpellState(trainer_spell, reqLevel) != TRAINER_SPELL_GREEN)
285  return;
286 
287  // apply reputation discount
288  uint32 nSpellCost = uint32(floor(trainer_spell->spellCost * _player->GetReputationPriceDiscount(unit)));
289 
290  // check money requirement
291  if (_player->GetMoney() < nSpellCost)
292  return;
293 
294  _player->ModifyMoney(-int32(nSpellCost));
295 
296  SendPlaySpellVisual(guid, 0xB3); // visual effect on trainer
297 
298  WorldPacket data(SMSG_PLAY_SPELL_IMPACT, 8 + 4); // visual effect on player
299  data << _player->GetObjectGuid();
300  data << uint32(0x016A); // index from SpellVisualKit.dbc
301  SendPacket(data);
302 
303  // learn explicitly or cast explicitly
304  // TODO - Are these spells really cast correctly this way?
305  if (trainer_spell->IsCastable())
306  _player->CastSpell(_player, trainer_spell->spell, TRIGGERED_OLD_TRIGGERED);
307  else
308  _player->learnSpell(spellId, false);
309 
310  data.Initialize(SMSG_TRAINER_BUY_SUCCEEDED, 12);
311  data << ObjectGuid(guid);
312  data << uint32(spellId); // should be same as in packet from client
313  SendPacket(data);
314 }

498 void WorldSession::SendStablePet(ObjectGuid guid) const
499 {
500  DEBUG_LOG("WORLD: Recv MSG_LIST_STABLED_PETS Send.");
501 
502  WorldPacket data(MSG_LIST_STABLED_PETS, 200); // guess size
503  data << guid;
504 
505  Pet* pet = _player->GetPet();
506 
507  size_t wpos = data.wpos();
508  data << uint8(0); // place holder for slot show number
509 
510  data << uint8(GetPlayer()->m_stableSlots);
511 
512  uint8 num = 0; // counter for place holder
513  PetSaveMode firstSlot = PET_SAVE_FIRST_STABLE_SLOT; // have to be changed to PET_SAVE_AS_CURRENT if pet is currently temp unsummoned
514 
515  // not let move dead pet in slot
516  if (pet && pet->IsAlive() && pet->getPetType() == HUNTER_PET)
517  {
518  data << uint32(pet->GetCharmInfo()->GetPetNumber());
519  data << uint32(pet->GetEntry());
520  data << uint32(pet->GetLevel());
521  data << pet->GetName(); // petname
522  data << uint8(1); // 1 = current, 2/3 = in stable (any from 4,5,... create problems with proper show)
523  ++num;
524  }
525  else
526  {
527  if(_player->GetTemporaryUnsummonedPetNumber()) // temporary unsummon - mount
528  firstSlot = PET_SAVE_AS_CURRENT;
529  else
530  {
531  QueryResult* result = CharacterDatabase.PQuery("SELECT owner, id, entry, level, name FROM character_pet WHERE owner = '%u' AND slot = '%u' ORDER BY slot",
532  _player->GetGUIDLow(), uint32(PET_SAVE_NOT_IN_SLOT));
533 
534  if (result) // dismissed pet
535  {
536  do
537  {
538  Field* fields = result->Fetch();
539 
540  data << uint32(fields[1].GetUInt32()); // petnumber
541  data << uint32(fields[2].GetUInt32()); // creature entry
542  data << uint32(fields[3].GetUInt32()); // level
543  data << fields[4].GetString(); // name
544  data << uint8(0x01); // active
545 
546  ++num;
547  } while (result->NextRow());
548 
549  delete result;
550  }
551  }
552  }
553 
554  // 0 1 2 3 4 5
555  QueryResult* result = CharacterDatabase.PQuery("SELECT owner, slot, id, entry, level, name FROM character_pet WHERE owner = '%u' AND slot >= '%u' AND slot <= '%u' ORDER BY slot",
556  _player->GetGUIDLow(), uint32(firstSlot), uint32(PET_SAVE_LAST_STABLE_SLOT));
557 
558  if (result)
559  {
560  do
561  {
562  Field* fields = result->Fetch();
563 
564  data << uint32(fields[2].GetUInt32()); // petnumber
565  data << uint32(fields[3].GetUInt32()); // creature entry
566  data << uint32(fields[4].GetUInt32()); // level
567  data << fields[5].GetString(); // name
568  data << uint8(0x3); // inactive - stabled
569 
570  ++num;
571  }
572  while (result->NextRow());
573 
574  delete result;
575  }
576 
577  data.put<uint8>(wpos, num); // set real data to placeholder
578  SendPacket(data);
579 }
