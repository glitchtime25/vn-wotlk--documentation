151  void DoTauntPeons()
152  {
153  if (m_uiPeonKilledCount >= 4)
154  {
155  m_uiTauntTimer = 0;
156  return;
157  }
158 
159  std::list<Creature*> lFelConverts;
160  GuidVector m_vFelConverts;
161 
162  GetCreatureListWithEntryInGrid(lFelConverts, m_creature, NPC_FEL_ORC_CONVERT, 40.0f);
163 
164  for (Creature* convert : lFelConverts)
165  m_vFelConverts.push_back(convert->GetObjectGuid());
166 
167  if (m_vFelConverts.size() == 0)
168  return;
169 
170  switch (urand(0, 2))
171  {
172  case 0:
173  if (Creature* target = m_creature->GetMap()->GetCreature(m_vFelConverts[urand(0, m_vFelConverts.size() - 1)]))
174  {
175  DoCastSpellIfCan(target, SPELL_SHADOW_SEAR);
176  DoScriptText(SAY_TAUNT_1, m_creature);
177  }
178  break;
179  case 1:
180  DoCastSpellIfCan(m_creature, SPELL_DEATH_COIL_2);
181  DoScriptText(SAY_TAUNT_2, m_creature);
182  break;
183  case 2:
184  //DoCastSpellIfCan(m_creature, SPELL_TARGET_FISSURES);
185  if (Creature* target = m_creature->GetMap()->GetCreature(m_vFelConverts[urand(0, m_vFelConverts.size() - 1)]))
186  {
187  DoCastSpellIfCan(target, SPELL_SHADOW_FISSURE_2);
188  }
189  DoScriptText(SAY_TAUNT_3, m_creature);
190  break;
191  }
192 
193  m_uiTauntTimer = urand(30000,35000);
194  }

277  m_uiCleaveTimer = urand(6000, 8500);
278  }
279  else
280  m_uiCleaveTimer -= uiDiff;
281 
282  if (m_uiShadowFissureTimer < uiDiff)
283  {
284  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
285  DoCastSpellIfCan(pTarget, SPELL_SHADOW_FISSURE);
286  m_uiShadowFissureTimer = urand(7500, 15000);
287  }
288  else
289  m_uiShadowFissureTimer -= uiDiff;
290 
291  if (m_uiDeathCoilTimer < uiDiff)
292  {
293  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
294  DoCastSpellIfCan(pTarget, SPELL_DEATH_COIL);
295  m_uiDeathCoilTimer = urand(15000, 20000);
296  }
297  else
298  m_uiDeathCoilTimer -= uiDiff;
299 
300  if (m_creature->GetHealthPercent() <= 20.0f)
301  m_firstPhase = false;
302 
303  DoMeleeAttackIfReady();
304  }
305  else
306  {
307  if (!m_bSpinOnce)
308  {
309  SetCombatMovement(false);
310  DoCastSpellIfCan(nullptr, SPELL_DARK_SPIN);
311  m_bSpinOnce = true;
312  }
313  }
