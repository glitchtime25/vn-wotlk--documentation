65 // returns priority (0 == cannot start script)
66 uint8 GetSpellStartDBScriptPriority(SpellEntry const* spellinfo, SpellEffectIndex effIdx)
67 {
68  if (spellinfo->Effect[effIdx] == SPELL_EFFECT_SCRIPT_EFFECT)
69  return 10;
70 
71  if (spellinfo->Effect[effIdx] == SPELL_EFFECT_DUMMY)
72  return 9;
73 
74  // NonExisting triggered spells can also start DB-Spell-Scripts
75  if (spellinfo->Effect[effIdx] == SPELL_EFFECT_TRIGGER_SPELL && !sSpellTemplate.LookupEntry<SpellEntry>(spellinfo->EffectTriggerSpell[effIdx]))
76  return 5;
77 
78  // NonExisting trigger missile spells can also start DB-Spell-Scripts
79  if (spellinfo->Effect[effIdx] == SPELL_EFFECT_TRIGGER_MISSILE && !sSpellTemplate.LookupEntry<SpellEntry>(spellinfo->EffectTriggerSpell[effIdx]))
80  return 4;
81 
82  // Can not start script
83  return 0;
84 }

86 // Priorize: SCRIPT_EFFECT before DUMMY before Non-Existing triggered spell, for same priority the first effect with the priority triggers
87 bool ScriptMgr::CanSpellEffectStartDBScript(SpellEntry const* spellinfo, SpellEffectIndex effIdx)
88 {
89  uint8 priority = GetSpellStartDBScriptPriority(spellinfo, effIdx);
90  if (!priority)
91  return false;
92 
93  for (int i = 0; i < MAX_EFFECT_INDEX; ++i)
94  {
95  uint8 currentPriority = GetSpellStartDBScriptPriority(spellinfo, SpellEffectIndex(i));
96  if (currentPriority < priority) // lower priority, continue checking
97  continue;
98  if (currentPriority > priority) // take other index with higher priority
99  return false;
100  if (i < effIdx) // same priority at lower index
101  return false;
102  }
103 
104  return true;
105 }
