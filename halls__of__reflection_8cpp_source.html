80 void instance_halls_of_reflection::OnPlayerEnter(Player* pPlayer)
81 {
82  // send quest start
83  if (Creature* pCreature = GetSingleCreatureFromStorage(NPC_FROSTMOURNE_ALTAR_BUNNY))
84  {
85  if (GetData(TYPE_FROSTMOURNE_INTRO) != DONE)
86  pCreature->CastSpell(pCreature, SPELL_START_HALLS_REFLECTION, TRIGGERED_OLD_TRIGGERED);
87  }
88 
89  if (!m_uiTeam) // very first player to enter
90  {
91  m_uiTeam = pPlayer->GetTeam();
92  SetDialogueSide(m_uiTeam == ALLIANCE);
93 
94  // intro event starts on timer
95  if (GetData(TYPE_FROSTMOURNE_INTRO) != DONE)
96  m_uiSummonDelayTimer = 5000;
97  // last encounter
98  else if (GetData(TYPE_FROSTWORN_GENERAL) == DONE && !GetSingleCreatureFromStorage(NPC_LICH_KING, true))
99  DoSetupEscapeEvent(pPlayer);
100  // mini boss
101  else if (GetData(TYPE_MARWYN) == DONE && !GetSingleCreatureFromStorage(NPC_FROSTSWORN_GENERAL, true))
102  pPlayer->SummonCreature(NPC_FROSTSWORN_GENERAL, afGeneralSpawnLoc[0], afGeneralSpawnLoc[1], afGeneralSpawnLoc[2], afGeneralSpawnLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0, true);
103  }
104 
105  // Quel'delar
106  if (!GetSingleCreatureFromStorage(NPC_UTHER, true) && pPlayer->HasAura(SPELL_QUELDELAR_COMPULSION))
107  {
108  if (pPlayer->GetQuestStatus(m_uiTeam == HORDE ? QUEST_ID_HALLS_REFLECTION_HORDE : QUEST_ID_HALLS_REFLECTION_ALLY) == QUEST_STATUS_INCOMPLETE)
109  {
110  // Don't start Quel'delar event if intro already started
111  Creature* pTemp = GetSingleCreatureFromStorage(m_uiTeam == HORDE ? NPC_SYLVANAS_PART1 : NPC_SYLVANAS_PART2, true);
112  if ((pTemp && pTemp->GetMotionMaster()->getLastReachedWaypoint() <= 2) || !pTemp)
113  {
114  // Set data to avoid starting the instance intro
115  if (GetData(TYPE_QUEL_DELAR) == NOT_STARTED)
116  SetData(TYPE_QUEL_DELAR, SPECIAL);
117  }
118  }
119  }

217  {
218  DoUseDoorOrButton(GO_ICECROWN_DOOR_ENTRANCE);
219 
220  m_uiEventTimer = 40000;
221  m_uiEventStage = 0;
222  }
223  m_auiEncounter[uiType] = uiData;
224  break;
225  case TYPE_MARWYN:
226  if (uiData == DONE)
227  {
228  DoUseDoorOrButton(GO_IMPENETRABLE_DOOR);
229  DoUseDoorOrButton(GO_ICECROWN_DOOR_ENTRANCE);
230  DoUpdateWorldState(WORLD_STATE_SPIRIT_WAVES, 0);
231 
232  // spawn next mini boss
233  if (Player* pPlayer = GetPlayerInMap())
234  pPlayer->SummonCreature(NPC_FROSTSWORN_GENERAL, afGeneralSpawnLoc[0], afGeneralSpawnLoc[1], afGeneralSpawnLoc[2], afGeneralSpawnLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0, true);
235  }
236  else if (uiData == FAIL)
237  DoCleanupFrostmourneEvent();
238  else if (uiData == SPECIAL)
239  {
240  DoUseDoorOrButton(GO_ICECROWN_DOOR_ENTRANCE);
241 
242  m_uiEventTimer = 60000;
243  m_uiEventStage = 5;
244  }
245  m_auiEncounter[uiType] = uiData;
246  break;
247  case TYPE_LICH_KING:
248  if (uiData == DONE)
249  {
250  // Move ship to battle position
251  if (m_uiTeam == ALLIANCE)
252  {
253  if (GenericTransport* gunship = instance->GetTransport(ObjectGuid(HIGHGUID_MO_TRANSPORT, uint32(GO_TRANSPORT_SKYBREAKER))))
254  gunship->SetGoState(GO_STATE_ACTIVE);
255  }
256  else
257  {
258  if (GenericTransport* gunship = instance->GetTransport(ObjectGuid(HIGHGUID_MO_TRANSPORT, uint32(GO_TRANSPORT_OGRIMS_HAMMER))))
259  gunship->SetGoState(GO_STATE_ACTIVE);
260  }
261 
262  // update the shio one more time after the timer expires
263  m_uiShipUpdateTimer = 15000;
264  }
265  else if (uiData == IN_PROGRESS)
266  StartNextDialogueText(SAY_ALLY_INTRO_1);
267  else if (uiData == FAIL)
268  {
269  m_uiEscapeResetTimer = 10000;
270 
271  if (Creature* pCreature = GetSingleCreatureFromStorage(m_uiTeam == ALLIANCE ? NPC_JAINA_PART2 : NPC_SYLVANAS_PART2))
272  pCreature->ForcedDespawn(5000);
273  }
274  m_auiEncounter[uiType] = uiData;
275  break;
276  case TYPE_FROSTWORN_GENERAL:
277  if (uiData == DONE)
278  {
279  // spawn creatures for last encounter
280  if (Player* pPlayer = GetPlayerInMap())
281  DoSetupEscapeEvent(pPlayer);
282  }
283  m_auiEncounter[uiType] = uiData;
284  break;
285  case TYPE_QUEL_DELAR:
286  if (uiData == SPECIAL)
287  StartNextDialogueText(NPC_FROSTMOURNE_ALTAR_BUNNY);
288  m_auiEncounter[uiType] = uiData;
289  break;
290  default:
291  return;
292  }
293 
294  if (uiData == DONE || uiData == FAIL)
295  {
296  OUT_SAVE_INST_DATA;
297 
298  std::ostringstream saveStream;
299  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
300  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5];
301 
302  m_strInstData = saveStream.str();
303 
304  SaveToDB();
305  OUT_SAVE_INST_DATA_COMPLETE;
306  }

427 // Function to send the next spirit wave
428 void instance_halls_of_reflection::DoSendNextSpiritWave()
429 {
430  uint8 m_uiMaxMobs = 0;
431  DoUpdateWorldState(WORLD_STATE_SPIRIT_WAVES_COUNT, m_uiEventStage);
432 
433  switch (m_uiEventStage)
434  {
435  case 1:
436  case 6:
437  // First wave in the series
438  m_uiMaxMobs = 3;
439  DoUpdateWorldState(WORLD_STATE_SPIRIT_WAVES, 1);
440  break;
441  case 2:
442  case 7:
443  // Second wave in the series
444  m_uiMaxMobs = 4;
445  break;
446  case 5:
447  // Falric (wave 5)
448  if (Creature* pFalric = GetSingleCreatureFromStorage(NPC_FALRIC))
449  {
450  pFalric->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
451  if (Player* pPlayer = GetPlayerInMap(true, false))
452  pFalric->AI()->AttackStart(pPlayer);
453 
454  m_uiEventTimer = 0;
455  }
456  return;
457  case 10:
458  // Marwyn (wave 10)
459  if (Creature* pMarwyn = GetSingleCreatureFromStorage(NPC_MARWYN))
460  {
461  pMarwyn->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
462  if (Player* pPlayer = GetPlayerInMap(true, false))
463  pMarwyn->AI()->AttackStart(pPlayer);
464 
465  m_uiEventTimer = 0;
466  }
467  return;
468  default:
469  // Third and fourth wave of the series
470  m_uiMaxMobs = 5;
471  break;
472  }
473 
474  // Activate random spirits
475  for (uint8 i = 0; i < m_uiMaxMobs; ++i)
476  {
477  if (m_lRisenSpiritsGuids.empty())
478  return;
479 
480  GuidList::iterator iter = m_lRisenSpiritsGuids.begin();
481  advance(iter, urand(0, m_lRisenSpiritsGuids.size() - 1));
482 
483  if (Creature* pCreature = instance->GetCreature(*iter))
484  {
485  if (!pCreature->IsAlive())
486  {
487  script_error_log("instance_halls_of_reflection: Error: couldn't find alive creature %u", pCreature->GetEntry());
488  return;
489  }
490 
491  pCreature->CastSpell(pCreature, SPELL_SPIRIT_ACTIVATE_VISUAL, TRIGGERED_NONE);
492  pCreature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
493 
494  m_lActiveSpiritsGuids.push_back(*iter);
495  m_lRisenSpiritsGuids.remove(*iter);
496  }
497  else
498  script_error_log("instance_halls_of_reflection: Error: couldn't find creature %u", pCreature->GetEntry());
499  }
500 
501  m_uiActivateTimer = 3000;
502 }

504 // Function to cleanup the first two encounters events
505 void instance_halls_of_reflection::DoCleanupFrostmourneEvent()
506 {
507  // Cleanup boses
508  if (GetData(TYPE_FALRIC) != DONE)
509  {
510  if (Creature* pFalric = GetSingleCreatureFromStorage(NPC_FALRIC))
511  pFalric->ForcedDespawn();
512  }
513  if (Creature* pMarwyn = GetSingleCreatureFromStorage(NPC_MARWYN))
514  pMarwyn->ForcedDespawn();
515 
516  // Cleanup trash
517  for (GuidList::const_iterator itr = m_lRisenSpiritsGuids.begin(); itr != m_lRisenSpiritsGuids.end(); ++itr)
518  {
519  if (Creature* pCreature = instance->GetCreature(*itr))
520  pCreature->ForcedDespawn();
521  }
522  for (GuidList::const_iterator itr = m_lActiveSpiritsGuids.begin(); itr != m_lActiveSpiritsGuids.end(); ++itr)
523  {
524  if (Creature* pCreature = instance->GetCreature(*itr))
525  pCreature->ForcedDespawn();
526  }
527 
528  // Remove world state and open door
529  DoUpdateWorldState(WORLD_STATE_SPIRIT_WAVES, 0);
530  DoUseDoorOrButton(GO_ICECROWN_DOOR_ENTRANCE);
531 
532  m_lRisenSpiritsGuids.clear();
533  m_lActiveSpiritsGuids.clear();
534 
535  m_uiEventTimer = 0;
536 }

550 void instance_halls_of_reflection::Update(uint32 uiDiff)
551 {
552  DialogueUpdate(uiDiff);
553 
554  // Entrance spawn timer
555  if (m_uiSummonDelayTimer)
556  {
557  if (m_uiSummonDelayTimer <= uiDiff)
558  {
559  Player* pPlayer = GetPlayerInMap();
560  if (!pPlayer)
561  {
562  script_error_log("instance_halls_of_reflection: Error: couldn't find any player in instance");
563  m_uiSummonDelayTimer = 0;
564  return;
565  }
566 
567  // Spawn intro npcs and make the start the movement
