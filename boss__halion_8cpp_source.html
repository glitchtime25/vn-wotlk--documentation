354  }
355  }

357  m_uiMeteorTimer -= uiDiff;
358 
359  // Switch to phase 2
360  if (m_creature->GetHealthPercent() < 75.0f && m_uiPhase == PHASE_PHISYCAL_REALM)
361  {
362  if (DoCastSpellIfCan(m_creature, SPELL_TWILIGHT_PHASING, CAST_INTERRUPT_PREVIOUS) == CAST_OK)

364  DoCastSpellIfCan(m_creature, SPELL_SUMMON_PORTAL, CAST_TRIGGERED);

366

369  {
370  if (Creature* pController = m_pInstance->GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
371  SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, pController);
372 
373  if (Creature* pHalion = m_pInstance->GetSingleCreatureFromStorage(NPC_HALION_TWILIGHT))
374  SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, pHalion);
375  }
376  m_uiPhase = PHASE_TWILIGHT_REALM;
377  }
378  }
379 
380  break;
381  case PHASE_TWILIGHT_REALM:
382  // this AI doesn't handle this phase
383  break;
384  }
385 
386  DoMeleeAttackIfReady();
387  }
