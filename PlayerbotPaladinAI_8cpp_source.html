104 CombatManeuverReturns PlayerbotPaladinAI::DoFirstCombatManeuver(Unit* pTarget)
105 {
106  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
107  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
108  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
109  {
110  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
111  {
112  if (PlayerbotAI::ORDERS_TANK & m_ai.GetCombatOrder())
113  {
114  if (m_bot.CanReachWithMeleeAttack(pTarget))
115  {
116  // Set everyone's UpdateAI() waiting to 2 seconds
117  m_ai.SetGroupIgnoreUpdateTime(2);
118  // Clear their TEMP_WAIT_TANKAGGRO flag
119  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
120  // Start attacking, force target on current target
121  m_ai.Attack(m_ai.GetCurrentTarget());
122 
123  // While everyone else is waiting 2 second, we need to build up aggro, so don't return
124  }
125  else
126  {
127  // TODO: add check if target is ranged
128  return RETURN_NO_ACTION_OK; // wait for target to get nearer
129  }
130  }
131  else if (PlayerbotAI::ORDERS_HEAL & m_ai.GetCombatOrder())
132  return HealPlayer(GetHealTarget());
133  else
134  return RETURN_NO_ACTION_OK; // wait it out
135  }
136  else
137  {
138  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
139  }
140  }
141 
142  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
143  {
144  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
145  return RETURN_NO_ACTION_OK; // wait it out
146  else
147  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
148  }
149 
150  switch (m_ai.GetScenarioType())
151  {
152  case PlayerbotAI::SCENARIO_PVP_DUEL:
153  case PlayerbotAI::SCENARIO_PVP_BG:
154  case PlayerbotAI::SCENARIO_PVP_ARENA:
155  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
156  return DoFirstCombatManeuverPVP(pTarget);
157  case PlayerbotAI::SCENARIO_PVE:
158  case PlayerbotAI::SCENARIO_PVE_ELITE:
159  case PlayerbotAI::SCENARIO_PVE_RAID:
160  default:
161  return DoFirstCombatManeuverPVE(pTarget);
162  }

195 CombatManeuverReturns PlayerbotPaladinAI::DoNextCombatManeuverPVE(Unit* pTarget)
196 {
197  if (!pTarget) return RETURN_NO_ACTION_INVALIDTARGET;
198 
199  // damage spells
200  uint32 spec = m_bot.GetSpec();
201  std::ostringstream out;
202 
203  // Make sure healer stays put, don't even melee (aggro) if in range.
204  if (m_ai.IsHealer() && m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED)
205  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
206  else if (!m_ai.IsHealer() && m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE)
207  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
208 
209  // Emergency check: bot is about to die: use Divine Shield (first)
210  // Use Divine Protection if Divine Shield is not available and bot is not tanking because of the pacify effect
211  // TODO adjust treshold (may be too low)
212  if (m_ai.GetHealthPercent() < 8)
213  {
214  if (DIVINE_SHIELD > 0 && m_bot.IsSpellReady(DIVINE_SHIELD) && !m_bot.HasAura(DIVINE_SHIELD, EFFECT_INDEX_0) && !m_bot.HasAura(DIVINE_PROTECTION, EFFECT_INDEX_0) && !m_bot.HasAura(FORBEARANCE, EFFECT_INDEX_0) && m_ai.CastSpell(DIVINE_SHIELD, m_bot) == SPELL_CAST_OK)
215  return RETURN_CONTINUE;
216 
217  if (DIVINE_PROTECTION > 0 && !(m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TANK) && m_bot.IsSpellReady(DIVINE_PROTECTION) && !m_bot.HasAura(DIVINE_SHIELD, EFFECT_INDEX_0) && !m_bot.HasAura(DIVINE_PROTECTION, EFFECT_INDEX_0) && !m_bot.HasAura(FORBEARANCE, EFFECT_INDEX_0) && m_ai.CastSpell(DIVINE_PROTECTION, m_bot) == SPELL_CAST_OK)
218  return RETURN_CONTINUE;
219  }
220 
221  // Dispel magic/disease/poison
222  if (m_ai.HasDispelOrder() && DispelPlayer() & RETURN_CONTINUE)
223  return RETURN_CONTINUE;
224 
225  // Heal (try to pick a target by on common rules, than heal using each PlayerbotClassAI HealPlayer() method)
226  if (FindTargetAndHeal())
227  return RETURN_CONTINUE;
228 
229  //Used to determine if this bot has highest threat
230  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);
231  if (newTarget && !(m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TANK) && !m_ai.IsNeutralized(newTarget)) // TODO: && party has a tank
232  {
233  // Aggroed by an elite
234  if (m_ai.IsElite(newTarget))
235  {
236  // Try to stun the mob
237  if (HAMMER_OF_JUSTICE > 0 && m_ai.In_Reach(newTarget, HAMMER_OF_JUSTICE) && m_bot.IsSpellReady(HAMMER_OF_JUSTICE) && !newTarget->HasAura(HAMMER_OF_JUSTICE) && m_ai.CastSpell(HAMMER_OF_JUSTICE, *newTarget) == SPELL_CAST_OK)
238  return RETURN_CONTINUE;
239 
240  // Bot has low life: use divine powers to protect him/herself
241  if (m_ai.GetHealthPercent() < 15)
242  {
243  if (DIVINE_SHIELD > 0 && m_bot.IsSpellReady(DIVINE_SHIELD) && !m_bot.HasAura(DIVINE_SHIELD, EFFECT_INDEX_0) && !m_bot.HasAura(DIVINE_PROTECTION, EFFECT_INDEX_0) && !m_bot.HasAura(FORBEARANCE, EFFECT_INDEX_0) && m_ai.CastSpell(DIVINE_SHIELD, m_bot) == SPELL_CAST_OK)
244  return RETURN_CONTINUE;
245 
246  if (DIVINE_PROTECTION > 0 && m_bot.IsSpellReady(DIVINE_PROTECTION) && !m_bot.HasAura(DIVINE_SHIELD, EFFECT_INDEX_0) && !m_bot.HasAura(DIVINE_PROTECTION, EFFECT_INDEX_0) && !m_bot.HasAura(FORBEARANCE, EFFECT_INDEX_0) && m_ai.CastSpell(DIVINE_PROTECTION, m_bot) == SPELL_CAST_OK)
247  return RETURN_CONTINUE;
248  }
249 
250  // Else: do nothing and pray for tank to pick aggro from mob
251  return RETURN_NO_ACTION_OK;
252  }
253  }
254 
255  // Damage rotation
256  switch (spec)
257  {
258  case PALADIN_SPEC_HOLY:
259  if (m_ai.IsHealer())
260  return RETURN_NO_ACTION_OK;
261  // else: DPS (retribution, NEVER protection)
262 
263  case PALADIN_SPEC_RETRIBUTION:
264  if (HAMMER_OF_WRATH > 0 && pTarget->GetHealth() < pTarget->GetMaxHealth() * 0.20 && m_ai.CastSpell(HAMMER_OF_WRATH, *pTarget) == SPELL_CAST_OK)
265  return RETURN_CONTINUE;
266  if (ART_OF_WAR > 0 && EXORCISM > 0 && m_bot.IsSpellReady(EXORCISM) && m_bot.HasAura(ART_OF_WAR, EFFECT_INDEX_0) && m_ai.CastSpell(EXORCISM, *pTarget) == SPELL_CAST_OK)
267  return RETURN_CONTINUE;
268  if (CRUSADER_STRIKE > 0 && m_bot.IsSpellReady(CRUSADER_STRIKE) && m_ai.CastSpell(CRUSADER_STRIKE, *pTarget) == SPELL_CAST_OK)
269  return RETURN_CONTINUE;
270  if (DIVINE_STORM > 0 && /*m_ai.GetAttackerCount() >= 3 && meleeReach*/ m_bot.IsSpellReady(DIVINE_STORM) && m_ai.CastSpell(DIVINE_STORM, *pTarget) == SPELL_CAST_OK)
271  return RETURN_CONTINUE;
272  if (JUDGEMENT_OF_LIGHT > 0 && m_ai.CastSpell(JUDGEMENT_OF_LIGHT, *pTarget) == SPELL_CAST_OK)
273  return RETURN_CONTINUE;
274  if (AVENGING_WRATH > 0 && m_ai.CastSpell(AVENGING_WRATH, m_bot) == SPELL_CAST_OK)
275  return RETURN_CONTINUE;
276 
277  case PALADIN_SPEC_PROTECTION:
278  //Taunt if orders specify
279  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TANK && !newTarget && HAND_OF_RECKONING > 0 && m_bot.IsSpellReady(HAND_OF_RECKONING) && m_ai.CastSpell(HAND_OF_RECKONING, *pTarget) == SPELL_CAST_OK)
280  return RETURN_CONTINUE;
281  if (CONSECRATION > 0 && m_bot.IsSpellReady(CONSECRATION) && m_ai.CastSpell(CONSECRATION, *pTarget) == SPELL_CAST_OK)
282  return RETURN_CONTINUE;
283  if (HOLY_SHIELD > 0 && !m_bot.HasAura(HOLY_SHIELD) && m_ai.CastSpell(HOLY_SHIELD, m_bot) == SPELL_CAST_OK)
284  return RETURN_CONTINUE;
285  if (AVENGERS_SHIELD > 0 && m_bot.IsSpellReady(AVENGERS_SHIELD) && m_ai.CastSpell(AVENGERS_SHIELD, *pTarget) == SPELL_CAST_OK)
286  return RETURN_CONTINUE;
287  if (HAMMER_OF_THE_RIGHTEOUS > 0 && m_bot.IsSpellReady(HAMMER_OF_THE_RIGHTEOUS) && m_ai.CastSpell(HAMMER_OF_THE_RIGHTEOUS, *pTarget) == SPELL_CAST_OK)
288  return RETURN_CONTINUE;
289  if (SHIELD_OF_RIGHTEOUSNESS > 0 && m_bot.IsSpellReady(SHIELD_OF_RIGHTEOUSNESS) && m_ai.CastSpell(SHIELD_OF_RIGHTEOUSNESS, *pTarget) == SPELL_CAST_OK)
290  return RETURN_CONTINUE;
291  if (JUDGEMENT_OF_LIGHT > 0 && m_ai.CastSpell(JUDGEMENT_OF_LIGHT, *pTarget) == SPELL_CAST_OK)
292  return RETURN_CONTINUE;
293  return RETURN_NO_ACTION_OK;
294  }
295 
296  return RETURN_NO_ACTION_OK;
297 }

307 CombatManeuverReturns PlayerbotPaladinAI::HealPlayer(Player* target)
308 {
309  CombatManeuverReturns r = PlayerbotClassAI::HealPlayer(target);
310  if (r != RETURN_NO_ACTION_OK)
311  return r;
312 
313  // If target is out of range (40 yards) and is a tank: move towards it
314  // if bot is not asked to stay
315  // Other classes have to adjust their position to the healers
316  // TODO: This code should be common to all healers and will probably
317  // move to a more suitable place like PlayerbotAI::DoCombatMovement()
318  if ((GetTargetJob(target) == JOB_TANK || GetTargetJob(target) == JOB_MAIN_TANK)
319  && m_bot.GetPlayerbotAI()->GetMovementOrder() != PlayerbotAI::MOVEMENT_STAY
320  && !m_ai.In_Reach(target, FLASH_OF_LIGHT))
321  {
322  m_bot.GetMotionMaster()->MoveFollow(target, 39.0f, m_bot.GetOrientation());
323  return RETURN_CONTINUE;
324  }
325 
326  uint8 hp = target->GetHealthPercent();
327 
328  // Everyone is healthy enough, return OK. MUST correlate to highest value below (should be last HP check)
329  if (hp >= 80)
330  return RETURN_NO_ACTION_OK;
331 
332  if (hp < 25 && m_ai.CastSpell(LAY_ON_HANDS, *target) == SPELL_CAST_OK)
333  return RETURN_CONTINUE;
334 
335  // You probably want to save this for tank/healer trouble
336  if (hp < 30 && HAND_OF_PROTECTION > 0 && !target->HasAura(FORBEARANCE, EFFECT_INDEX_0)
337  && !target->HasAura(HAND_OF_PROTECTION, EFFECT_INDEX_0) && !target->HasAura(DIVINE_PROTECTION, EFFECT_INDEX_0)
338  && !target->HasAura(DIVINE_SHIELD, EFFECT_INDEX_0) && (GetTargetJob(target) & (JOB_HEAL | JOB_TANK))
339  && m_ai.CastSpell(HAND_OF_PROTECTION, *target) == SPELL_CAST_OK)
340  return RETURN_CONTINUE;
341 
342  // Low HP : activate Divine Favor to make next heal a critical heal
343  if (hp < 25 && DIVINE_FAVOR > 0 && !m_bot.HasAura(DIVINE_FAVOR, EFFECT_INDEX_0) && m_bot.IsSpellReady(DIVINE_FAVOR))
344  m_ai.CastSpell(DIVINE_FAVOR, m_bot);
345 
346  if (hp < 40 && HOLY_LIGHT > 0 && m_ai.In_Reach(target, HOLY_LIGHT) && m_ai.CastSpell(HOLY_LIGHT, *target) == SPELL_CAST_OK)
347  return RETURN_CONTINUE;
348 
349  if (hp < 60 && HOLY_SHOCK > 0 && m_ai.In_Reach(target, HOLY_SHOCK) && m_ai.CastSpell(HOLY_SHOCK, *target) == SPELL_CAST_OK)
350  return RETURN_CONTINUE;
351 
352  if (hp < 80 && FLASH_OF_LIGHT > 0 && m_ai.In_Reach(target, FLASH_OF_LIGHT) && m_ai.CastSpell(FLASH_OF_LIGHT, *target) == SPELL_CAST_OK)
353  return RETURN_CONTINUE;
354 
355  return RETURN_NO_ACTION_UNKNOWN;
356 } // end HealTarget
