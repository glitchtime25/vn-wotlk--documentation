209  {

211  switch (m_animationSpawnStage)
212  {
213  case 0:
214  m_creature->SetStandState(UNIT_STAND_STATE_STAND);

216  break;
217  case 1:
218  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
219  DoScriptText(SAY_INTRO, m_creature);
220  timer = 1000;
221  break;
222  case 2:

224  m_creature->SetHover(true);
225  timer = 7500;
226  break;
227  case 3:
228  m_creature->GetMotionMaster()->MovePoint(PHASE_GROUND, 1489.438f, 651.615f, 44.65125f, FORCED_MOVEMENT_RUN);
229  break;
230  case 4:
231  SetCombatScriptStatus(false);

234  DoCastSpellIfCan(nullptr, SPELL_NOXIOUS_FUMES);
235  AttackClosestEnemy();
236  break;

238  ++m_animationSpawnStage;
239  if (timer)
240  ResetTimer(FELMYST_WAYPOINT_DELAY, timer);

291  void MovementInform(uint32 motionType, uint32 pointId) override
292  {
293  if (motionType != POINT_MOTION_TYPE)
294  return;
295 
296  switch (pointId)
297  {
298  case PHASE_GROUND:
299  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
300  m_creature->GetMotionMaster()->Clear(false, true);
301  m_creature->GetMotionMaster()->MovePath(1, PATH_FROM_ENTRY, FORCED_MOVEMENT_RUN, true, 0.f, true);
302  break;
303  case SUBPHASE_VAPOR:
304  // prepare to move to flight trigger
305  ResetTimer(FELMYST_DEMONIC_VAPOR, 5000);
306  m_uiSubPhase = SUBPHASE_VAPOR;
307  break;
308  case SUBPHASE_BREATH_SIDE:
309  ResetTimer(FELMYST_PREBREATH_DELAY, 5000);
310  ResetTimer(FELMYST_DUMMY_NUKE, 1500);
311  break;
312  case SUBPHASE_BREATH_PREPARE:
313  // move across the arena
314  if (!m_instance)
315  return;
316 
317  // After the third breath land and resume phase 1
318  if (m_uiCorruptionCount == 3)
319  {
320  static Position landingPositions[] =
321  {
322  {1469.93f, 557.009f, 22.6317f, 0.f},
323  {1476.77f, 665.094f, 20.6423f, 0.f}
324  };
325  m_uiPhase = PHASE_TRANSITION; // TODO: figure out where to land
326  m_creature->GetMotionMaster()->MovePoint(PHASE_TRANSITION, landingPositions[m_bIsLeftSide], FORCED_MOVEMENT_RUN, 0.f, false);
327  return;
328  }
329 
330  ++m_uiCorruptionCount;
331  ResetTimer(FELMYST_BREATH_DELAY, 4000);
332  DoScriptText(EMOTE_DEEP_BREATH, m_creature);
333  break;
334  case SUBPHASE_BREATH_MOVE:
335  if (!m_instance)
336  return;
337 
338  // remove speed aura
339  m_creature->RemoveAurasDueToSpell(SPELL_SPEED_BURST);
340 
341  // switch sides
342  m_bIsLeftSide = !m_bIsLeftSide;
343  HandleSideMove();
344  break;
345  case PHASE_TRANSITION:
346  HandleTransition();
347  break;
348  case POINT_AGGRO:
349  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
350  m_creature->SetLevitate(false);
351  m_creature->SetHover(false);
352  m_creature->GetMotionMaster()->MoveIdle();
353  ResetTimer(FELMYST_WAYPOINT_DELAY, 3000);
354  m_animationSpawnStage = 4; // should already be 3 but sanity assignment
355  break;
356  }
357  }

460  void ExecuteAction(uint32 action) override
461  {
462  switch (action)
463  {
464  case FELMYST_BERSERK:
465  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
466  {
467  DoScriptText(SAY_BERSERK, m_creature);
468  DisableCombatAction(action);
469  }
470  break;
471  case FELMYST_PHASE_CHANGE:
472  {
473  if (!CanExecuteCombatAction() || !m_creature->IsSpellReady(SPELL_ENCAPSULATE_CHANNEL))
474  return;
475 
476  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
477 
478  SetCombatMovement(false);
479  m_creature->SetLevitate(true);
480  m_creature->SetHover(true);
481  DoScriptText(SAY_TAKEOFF, m_creature);
482  SetMeleeEnabled(false);
483  m_creature->SetTarget(nullptr);
484  m_creature->SetByteFlag(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_FLY_ANIM);
485  m_creature->SendForcedObjectUpdate();
486  ResetTimer(FELMYST_LIFTOFF_DELAY, 5000);
487  SetDeathPrevention(true);
488  SetCombatScriptStatus(true);
489 
490  m_uiPhase = PHASE_TRANSITION;
491  m_uiSubPhase = SUBPHASE_VAPOR;
492  m_uiDemonicVaporCount = 0;
493  break;
494  }
495  case FELMYST_ENCAPSULATE:
496  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_ENCAPSULATE_CHANNEL, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IMMUNE))
497  DoCastSpellIfCan(target, SPELL_ENCAPSULATE_CHANNEL);
498  break;
499  case FELMYST_CORROSION:
500  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CORROSION) == CAST_OK)
501  {
502  DoScriptText(SAY_BREATH, m_creature);
503  ResetCombatAction(action, urand(32000, 40000));
504  }
505  break;
506  case FELMYST_CLEAVE:
507  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
508  ResetCombatAction(action, urand(15000, 29000));
509  break;
510  case FELMYST_GAS_NOVA:
511  if (DoCastSpellIfCan(nullptr, SPELL_GAS_NOVA) == CAST_OK)
512  ResetCombatAction(action, urand(22000, 27000));
513  break;
514  }
515  }
