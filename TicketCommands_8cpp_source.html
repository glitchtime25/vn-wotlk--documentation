27 inline static bool HandleTicketMgrCommandResult(ChatHandler &ch, uint32 ticketId, GMTicketMgr::CommandResult result)
28 {
29  MangosStrings error;
30 
31  switch (result)
32  {
33  case GMTicketMgr::COMMAND_RESULT_TICKET_NOT_FOUND: error = LANG_COMMAND_TICKET_NOT_FOUND; break;
34  case GMTicketMgr::COMMAND_RESULT_PLAYER_OFFLINE: error = LANG_COMMAND_TICKET_PLAYER_OFFLINE; break;
35  case GMTicketMgr::COMMAND_RESULT_TICKET_UNSEEN: error = LANG_COMMAND_TICKET_UNSEEN; break;
36  case GMTicketMgr::COMMAND_RESULT_ACCESS_DENIED: error = LANG_COMMAND_TICKET_ACCESS_DENIED; break;
37  case GMTicketMgr::COMMAND_RESULT_TICKET_NOT_ASSIGNED: error = LANG_COMMAND_TICKET_NOT_ASSIGNED; break;
38  case GMTicketMgr::COMMAND_RESULT_SUCCESS: return true;
39  }
40 
41  ch.PSendSysMessage(error, GMTicket::GetIdTag(ticketId).c_str());
42  return false;
43 }

308 bool ChatHandler::HandleTicketsCommand(char* args)
309 {
310  // Detect cases when used as an alias for "ticket list"
311  {
312  if (!args || !*args)
313  return HandleTicketsListCommand(args);
314 
315  std::string string = args;
316  char* input = const_cast<char*>(string.c_str());
317 
318  uint32 arg;
319 
320  bool listing = ExtractUInt32(&input, arg);
321 
322  if (!listing)
323  {
324  if (char* literal = ExtractLiteralArg(&input))
325  {
326  std::string online = literal;
327  strToLower(online);
328  listing = (strcmp(online.c_str(), "online") == 0);
329  }
330  else
331  listing = true;
332  }
333 
334  const bool result = (listing ? HandleTicketsListCommand(args) : false);
335 
336  if (listing)
337  return result;
338  }
339 
340  bool value;
341 
342  if (!ExtractOnOff(&args, value))
343  {
344  SendSysMessage(LANG_USE_BOL);
345  SetSentErrorMessage(true);
346  return false;
347  }
348 
349  Player* player = m_session->GetPlayer();
350 
351  player->SetAcceptTicket(value);
352  sTicketMgr.ShowMOTD(*player);
353 
354  return true;
355 }
