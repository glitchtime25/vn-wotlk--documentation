261  void UpdateAI(const uint32 uiDiff) override
262  {
263  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
264  return;
265 
266  if (m_uiSpiritTransformTimer)
267  {
268  if (m_uiSpiritTransformTimer <= uiDiff)
269  {
270  if (Creature* pSpirit = m_creature->GetMap()->GetCreature(m_uiCurrentSpiritGuid))
271  {
272  pSpirit->CastSpell(pSpirit, aYmironBoatsSpirits[m_vuiBoatPhases[m_uiBoats]].uiSpiritSpell, TRIGGERED_OLD_TRIGGERED);
273  pSpirit->CastSpell(pSpirit, SPELL_EMERGE_STATE, TRIGGERED_OLD_TRIGGERED);
274  }
275  m_uiSpiritTransformTimer = 0;
276  }
277  else
278  m_uiSpiritTransformTimer -= uiDiff;
279  }
280 
281  if (m_uiCombatResumeTimer)
282  {
283  // This should be done on aura 48307 remove, but because of lack of core support, we'll handle it on normal timer
284  if (m_uiCombatResumeTimer <= uiDiff)
285  {
286  DoChannelSpiritYmiron();
287  m_uiCombatResumeTimer = 0;
288  }
289  else
290  m_uiCombatResumeTimer -= uiDiff;
291  }
292 
293  // Don't attack while channeling on the boats
294  if (m_bIsChannelingSpirit)
295  return;
296 
297  if (m_uiBaneTimer < uiDiff)
298  {
299  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_BANE : SPELL_BANE_H) == CAST_OK)
300  m_uiBaneTimer = urand(20000, 25000);
301  }
302  else
303  m_uiBaneTimer -= uiDiff;
304 
305  if (m_uiFetidRotTimer < uiDiff)
306  {
307  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_FETID_ROT : SPELL_FETID_ROT_H) == CAST_OK)
308  m_uiFetidRotTimer = urand(10000, 15000);
309  }
310  else
311  m_uiFetidRotTimer -= uiDiff;
312 
313  if (m_uiDarkSlashTimer < uiDiff)
314  {
315  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_DARK_SLASH) == CAST_OK)
316  m_uiDarkSlashTimer = urand(30000, 35000);
317  }
318  else
319  m_uiDarkSlashTimer -= uiDiff;
320 
321  // Check the spirit phases (also don't allow him to change phase if below 1%)
322  if (m_creature->GetHealthPercent() < 100 - m_fHealthCheck && m_creature->GetHealthPercent() > 1.0f)
323  {
324  // change phase
325  DoCastSpellIfCan(m_creature, SPELL_SCREAMS_OF_THE_DEAD, CAST_INTERRUPT_PREVIOUS);
326 
327  // make the current spirit die (burn)
328  if (Creature* pSpirit = m_creature->GetMap()->GetCreature(m_uiCurrentSpiritGuid))
329  {
330  pSpirit->InterruptNonMeleeSpells(false);
331  pSpirit->CastSpell(pSpirit, SPELL_SPIRIT_DIES, TRIGGERED_NONE);
332  }
333 
334  // Get a close point to the spirits and move near them
335  if (m_pInstance)
336  {
337  if (Creature* pSpirit = m_pInstance->GetSingleCreatureFromStorage(aYmironBoatsSpirits[m_vuiBoatPhases[m_uiBoats]].uiSpiritTarget))
338  {
339  float fX, fY, fZ;
340  m_uiCurrentSpiritGuid = pSpirit->GetObjectGuid();
341  pSpirit->GetContactPoint(m_creature, fX, fY, fZ, INTERACTION_DISTANCE);
342  m_creature->SetWalk(false);
343  m_creature->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
344  }
345  }
346 
347  SetCombatMovement(false);
348  m_bIsChannelingSpirit = true;
349  m_fHealthCheck += m_bIsRegularMode ? 33.3f : 20.0f;
350  }
351 
352  switch (m_uiPhase)
353  {
354  case PHASE_BJORN:
355 
356  if (m_uiSpiritFountTimer)
357  {
358  if (m_uiSpiritFountTimer <= uiDiff)
359  {
360  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_SPIRIT_FOUNT) == CAST_OK)
361  m_uiSpiritFountTimer = 0;
362  }
363  else
364  m_uiSpiritFountTimer -= uiDiff;
365  }
366 
367  break;
368  case PHASE_HALDOR:
369 
370  if (m_uiSpiritStrikeTimer < uiDiff)
371  {
372  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_SPIRIT_STRIKE : SPELL_SPIRIT_STRIKE_H) == CAST_OK)
373  m_uiSpiritStrikeTimer = 5000;
374  }
375  else
376  m_uiSpiritStrikeTimer -= uiDiff;
377 
378  break;
379  case PHASE_RANULF:
380 
381  if (m_uiSpiritBurstTimer < uiDiff)
382  {
383  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SPIRIT_BURST : SPELL_SPIRIT_BURST_H) == CAST_OK)
384  m_uiSpiritBurstTimer = 10000;
385  }
386  else
387  m_uiSpiritBurstTimer -= uiDiff;
388 
389  break;
390  case PHASE_TORGYN:
391 
392  if (m_uiAvengingSpiritsTimer < uiDiff)
393  {
394  if (DoCastSpellIfCan(m_creature, SPELL_AVENGING_SPIRITS) == CAST_OK)
395  m_uiAvengingSpiritsTimer = 15000;
396  }
397  else
398  m_uiAvengingSpiritsTimer -= uiDiff;
399 
400  break;
401  }
402 
403  DoMeleeAttackIfReady();
404  }
