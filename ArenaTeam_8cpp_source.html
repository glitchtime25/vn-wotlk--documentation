103 bool ArenaTeam::AddMember(ObjectGuid playerGuid)
104 {
105  std::string plName;
106  uint8 plClass;
107 
108  // arena team is full (can't have more than type * 2 players!)
109  if (GetMembersSize() >= GetMaxMembersSize())
110  return false;
111 
112  Player* pl = sObjectMgr.GetPlayer(playerGuid);
113  if (pl)
114  {
115  if (pl->GetArenaTeamId(GetSlot()))
116  {
117  sLog.outError("Arena::AddMember() : player already in this sized team");
118  return false;
119  }
120 
121  plClass = pl->getClass();
122  plName = pl->GetName();
123  }
124  else
125  {
126  // 0 1
127  QueryResult* result = CharacterDatabase.PQuery("SELECT name, class FROM characters WHERE guid='%u'", playerGuid.GetCounter());
128  if (!result)
129  return false;
130 
131  plName = (*result)[0].GetCppString();
132  plClass = (*result)[1].GetUInt8();
133  delete result;
134 
135  // check if player already in arenateam of that size
136  if (Player::GetArenaTeamIdFromDB(playerGuid, GetType()) != 0)
137  {
138  sLog.outError("Arena::AddMember() : player %s already in this sized team", playerGuid.GetString().c_str());
139  return false;
140  }
141  }
142 
143  // remove all player signs from another petitions
144  // this will be prevent attempt joining player to many arenateams and corrupt arena team data integrity
145  Player::RemovePetitionsAndSigns(playerGuid, GetType());
146 
147  ArenaTeamMember newmember;
148  newmember.name = plName;
149  newmember.guid = playerGuid;
150  newmember.Class = plClass;
151  newmember.games_season = 0;
152  newmember.games_week = 0;
153  newmember.wins_season = 0;
154  newmember.wins_week = 0;
155 
156  int32 conf_value = sWorld.getConfig(CONFIG_INT32_ARENA_STARTPERSONALRATING);
157  if (conf_value < 0) // -1 = select by season id
158  {
159  if (sWorld.getConfig(CONFIG_UINT32_ARENA_SEASON_ID) >= 6)
160  {
161  if (m_stats.rating < 1000)
162  newmember.personal_rating = 0;
163  else
164  newmember.personal_rating = 1000;
165  }
166  else
167  {
168  newmember.personal_rating = 1500;
169  }
170  }
171  else
172  newmember.personal_rating = uint32(conf_value);
173 
174  m_members.push_back(newmember);
175 
176  CharacterDatabase.PExecute("INSERT INTO arena_team_member (arenateamid, guid, personal_rating) VALUES ('%u', '%u', '%u')", m_TeamId, newmember.guid.GetCounter(), newmember.personal_rating);
177 
178  if (pl)
179  {
180  pl->SetInArenaTeam(m_TeamId, GetSlot(), GetType());
181  pl->SetArenaTeamIdInvited(0);
182  pl->SetArenaTeamInfoField(GetSlot(), ARENA_TEAM_PERSONAL_RATING, newmember.personal_rating);
183 
184  // hide promote/remove buttons
185  if (m_CaptainGuid != playerGuid)
186  pl->SetArenaTeamInfoField(GetSlot(), ARENA_TEAM_MEMBER, 1);
187  }
188  return true;
189 }

222 bool ArenaTeam::LoadMembersFromDB(QueryResult* arenaTeamMembersResult)
223 {
224  if (!arenaTeamMembersResult)
225  return false;
226 
227  bool captainPresentInTeam = false;
228 
229  do
230  {
231  Field* fields = arenaTeamMembersResult->Fetch();
232  // prevent crash if db records are broken, when all members in result are already processed and current team hasn't got any members
233  if (!fields)
234  break;
235  uint32 arenaTeamId = fields[0].GetUInt32();
236  if (arenaTeamId < m_TeamId)
237  {
238  // there is in table arena_team_member record which doesn't have arenateamid in arena_team table, report error
239  sLog.outErrorDb("ArenaTeam %u does not exist but it has record in arena_team_member table, deleting it!", arenaTeamId);
240  CharacterDatabase.PExecute("DELETE FROM arena_team_member WHERE arenateamid = '%u'", arenaTeamId);
241  continue;
242  }
243 
244  if (arenaTeamId > m_TeamId)
245  // we loaded all members for this arena_team already, break cycle
246  break;
247 
248  ArenaTeamMember newmember;
249  newmember.guid = ObjectGuid(HIGHGUID_PLAYER, fields[1].GetUInt32());
250  newmember.games_week = fields[2].GetUInt32();
251  newmember.wins_week = fields[3].GetUInt32();
252  newmember.games_season = fields[4].GetUInt32();
253  newmember.wins_season = fields[5].GetUInt32();
254  newmember.personal_rating = fields[6].GetUInt32();
255  newmember.name = fields[7].GetCppString();
256  newmember.Class = fields[8].GetUInt8();
257 
258  // check if member exists in characters table
259  if (newmember.name.empty())
260  {
261  sLog.outErrorDb("ArenaTeam %u has member with empty name - probably player %s doesn't exist, deleting him from memberlist!", arenaTeamId, newmember.guid.GetString().c_str());
262  DelMember(newmember.guid);
263  continue;
264  }
265 
266  // arena team can't be > 2 * arenatype (2 for 2x2, 3 for 3x3, 5 for 5x5)
267  if (GetMembersSize() >= GetMaxMembersSize())
268  return false;
269 
270  if (newmember.guid == GetCaptainGuid())
271  captainPresentInTeam = true;
272 
273  m_members.push_back(newmember);
274  }
275  while (arenaTeamMembersResult->NextRow());
276 
277  if (Empty() || !captainPresentInTeam)
278  {
279  // arena team is empty or captain is not in team, delete from db
280  sLog.outErrorDb("ArenaTeam %u does not have any members or its captain is not in team, disbanding it...", m_TeamId);
281  return false;
282  }
283 
284  return true;
285 }

552 uint32 ArenaTeam::GetPoints(uint32 MemberRating)
553 {
554  // returns how many points would be awarded with this team type with this rating
555  float points;
556 
557  uint32 rating = MemberRating + 150 < m_stats.rating ? MemberRating : m_stats.rating;
558 
559  if (rating <= 1500)
560  {
561  // As of Season 6 and later, all teams below 1500 rating will earn points as if they were a 1500 rated team
562  if (sWorld.getConfig(CONFIG_UINT32_ARENA_SEASON_ID) >= 6)
563  rating = 1500;
564 
565  points = (float)rating * 0.22f + 14.0f;
566  }
567  else
568  points = 1511.26f / (1.0f + 1639.28f * exp(-0.00412f * (float)rating));
569 
570  // type penalties for <5v5 teams
571  if (m_Type == ARENA_TYPE_2v2)
572  points *= 0.76f;
573  else if (m_Type == ARENA_TYPE_3v3)
574  points *= 0.88f;
575 
576  return (uint32) points;
577 }
