119  void ExecuteAction(uint32 action) override
120  {
121  switch (action)
122  {
123  case MOAM_ARCANE_ERUPTION:
124  {
125  if (DoCastSpellIfCan(nullptr, SPELL_ARCANE_ERUPTION) == CAST_OK)
126  {
127  DoScriptText(EMOTE_MANA_FULL, m_creature);
128  ResetCombatAction(action, 5000); // small CD to prevent AI spam
129  }
130  break;
131  }
132  case MOAM_TRAMPLE:
133  {
134  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_TRAMPLE) == CAST_OK)
135  ResetCombatAction(action, 15000);
136  break;
137  }
138  case MOAM_MANA_DRAIN:
139  {
140  if (DoCastSpellIfCan(nullptr, SPELL_DRAIN_MANA) == CAST_OK)
141  ResetCombatAction(action, 6000);
142  break;
143  }
144  case MOAM_MANA_FIENDS:
145  {
146  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_MANAFIENDS) == CAST_OK)
147  ResetCombatAction(action, 90000);
148  break;
149  }
150  case MOAM_ENERGIZE:
151  {
152  if (DoCastSpellIfCan(nullptr, SPELL_ENERGIZE) == CAST_OK)
153  {
154  DoScriptText(EMOTE_ENERGIZING, m_creature);
155  ResetTimer(MOAM_CANCEL_ENERGIZE, 90000);
156  ResetCombatAction(action, 180000);
157  }
158  break;
159  }
160  }
161  }
