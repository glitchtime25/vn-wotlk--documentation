174  void UpdateEscortAI(const uint32 uiDiff) override
175  {
176  if (HasEscortState(STATE_ESCORT_PAUSED))
177  {

179  {

184  m_uiRitualTimer = 1000;

196  m_uiRitualTimer = 20000;
197  break;
198  case 4:

201  break;
202  case 5:
203  DoSummonSpawner(urand(0, 2));
204  DoScriptText(SAY_BELNISTRASZ_2_MIN, m_creature, m_creature);
205  m_uiRitualTimer = 40000;
206  break;
207  case 6:
208  DoSummonSpawner(urand(0, 2));
209  m_uiRitualTimer = 20000;
210  break;
211  case 7:
212  DoScriptText(SAY_BELNISTRASZ_1_MIN, m_creature, m_creature);
213  m_uiRitualTimer = 40000;
214  break;
215  case 8:
216  DoSummonSpawner(urand(0, 2));
217  m_uiRitualTimer = 20000;
218  break;
219  case 9:
220  DoScriptText(SAY_BELNISTRASZ_FINISH, m_creature, m_creature);
221  m_uiRitualTimer = 3000;
222  break;
223  case 10:
224  {
225  if (Player* pPlayer = GetPlayerForEscort())
226  {
227  pPlayer->RewardPlayerAndGroupAtEventExplored(QUEST_EXTINGUISHING_THE_IDOL, m_creature);
228 
229  if (GameObject* pGo = GetClosestGameObjectWithEntry(m_creature, GO_BELNISTRASZ_BRAZIER, 10.0f))
230  {
231  if (!pGo->IsSpawned())
232  {
233  pGo->SetRespawnTime(HOUR * IN_MILLISECONDS);
234  pGo->Refresh();
235  }
236  }
237  }
238 
239  m_creature->RemoveAurasDueToSpell(SPELL_IDOL_SHUTDOWN);
240  SetEscortPaused(false);
241 
242  // Desactivate the fires on the idol now it is extinguished
243  DoCastSpellIfCan(m_creature, SPELL_IDOL_ROOM_SHAKE);
244  GameObjectList lOvenFires;
245  for (auto&& gameObjectEntry : aGOList)
246  GetGameObjectListWithEntryInGrid(lOvenFires, m_creature, gameObjectEntry, 40.0f);
247 
248  for (auto&& gameObject : lOvenFires)
249  gameObject->SetLootState(GO_JUST_DEACTIVATED);
250 
251  break;
252  }
253  }
254 
255  ++m_uiRitualPhase;
256  }
257  else
258  m_uiRitualTimer -= uiDiff;
259 
260  return;
261  }
262 
263  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
264  return;
265 
266  if (m_uiFireballTimer < uiDiff)
267  {
268  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FIREBALL);
269  m_uiFireballTimer = urand(2000, 3000);
270  }
271  else
272  m_uiFireballTimer -= uiDiff;
273 
274  if (m_uiFrostNovaTimer < uiDiff)
275  {
276  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FROST_NOVA);
277  m_uiFrostNovaTimer = urand(10000, 15000);
278  }
279  else
280  m_uiFrostNovaTimer -= uiDiff;
281 
282  DoMeleeAttackIfReady();
283  }
