192 void MapManager::Update(uint32 diff)
193 {
194  i_timer.Update(diff);
195  if (!i_timer.Passed())
196  return;
197 
198  for (auto& map : i_maps)
199  {
200  if (m_updater.activated())
201  m_updater.schedule_update(new MapUpdateWorker(*map.second, (uint32)i_timer.GetCurrent(), m_updater));
202  else
203  map.second->Update((uint32)i_timer.GetCurrent());
204  }
205 
206  if (m_updater.activated())
207  m_updater.wait();
208 
209  // remove all maps which can be unloaded
210  MapMapType::iterator iter = i_maps.begin();
211  while (iter != i_maps.end())
212  {
213  Map* pMap = iter->second;
214  // check if map can be unloaded
215  if (pMap->CanUnload((uint32)i_timer.GetCurrent()))
216  {
217  pMap->UnloadAll(true);
218  delete pMap;
219 
220  i_maps.erase(iter++);
221  }
222  else
223  ++iter;
224  }
225 
226  i_timer.SetCurrent(0);
227 }

298 Map* MapManager::CreateInstance(uint32 id, Player* player)
299 {
300  Map* map = nullptr;
301  Map* pNewMap = nullptr;
302  uint32 NewInstanceId = 0; // instanceId of the resulting map
303  const MapEntry* entry = sMapStore.LookupEntry(id);
304 
305  if (entry->IsBattleGroundOrArena())
306  {
307  // find existing bg map for player
308  NewInstanceId = player->GetBattleGroundId();
309  MANGOS_ASSERT(NewInstanceId);
310  map = FindMap(id, NewInstanceId);
311  MANGOS_ASSERT(map);
312  }
313  else if (DungeonPersistentState* pSave = player->GetBoundInstanceSaveForSelfOrGroup(id))
314  {
315  // solo/perm/group
316  NewInstanceId = pSave->GetInstanceId();
317  map = FindMap(id, NewInstanceId);
318  // it is possible that the save exists but the map doesn't
319  if (!map)
320  {
321  Difficulty difficulty = entry->IsDynamicDifficultyMap() ? (player->GetGroup() ? player->GetGroup()->GetDifficulty(entry->IsRaid()) : player->GetDifficulty(entry->IsRaid())) : pSave->GetDifficulty();
322  pNewMap = CreateDungeonMap(id, NewInstanceId, difficulty, pSave, player->GetTeam());
323  }
324  else if (entry->IsDynamicDifficultyMap() && map->GetDifficulty() != player->GetDifficulty(entry->IsRaid()))
325  map->ChangeMapDifficulty(player->GetDifficulty(entry->IsRaid()));
326  }
327  else
328  {
329  // if no instanceId via group members or instance saves is found
330  // the instance will be created for the first time
331  NewInstanceId = sObjectMgr.GenerateInstanceLowGuid();
332 
333  Difficulty diff = player->GetGroup() ? player->GetGroup()->GetDifficulty(entry->IsRaid()) : player->GetDifficulty(entry->IsRaid());
334  pNewMap = CreateDungeonMap(id, NewInstanceId, diff, nullptr, player->GetTeam());
335  }
336 
337  // add a new map object into the registry
338  if (pNewMap)
339  {
340  i_maps[MapID(id, NewInstanceId)] = pNewMap;
341  map = pNewMap;
342  }
343 
344  return map;
345 }

348 {

358  MANGOS_ASSERT(false);
359  }
360 
361  // some instances only have one difficulty
362  if (!GetMapDifficultyData(id, difficulty))
363  difficulty = DUNGEON_DIFFICULTY_NORMAL;
364 
365  DEBUG_LOG("MapInstanced::CreateDungeonMap: %s map instance %d for %d created with difficulty %d", save ? "" : "new ", InstanceId, id, difficulty);
366 
367  DungeonMap* map = new DungeonMap(id, i_gridCleanUpDelay, InstanceId, difficulty);
368 
369  // Set owner team before initializing
370  map->SetInstanceTeam(ownerTeam);
371 
372  // Dungeons can have saved instance data
373  bool load_data = save != nullptr;
374  map->Initialize(load_data);
375 
376  return map;
