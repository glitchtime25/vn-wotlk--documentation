61 bool GossipHello_npc_saurfang_gunship(Player* pPlayer, Creature* pCreature)
62 {
63  if (pCreature->isQuestGiver())
64  pPlayer->PrepareQuestMenu(pCreature->GetObjectGuid());
65 
66  if (instance_icecrown_citadel* pInstance = static_cast<instance_icecrown_citadel*>(pCreature->GetInstanceData()))
67  {
68  if (pInstance->GetData(TYPE_GUNSHIP_BATTLE) == NOT_STARTED || pInstance->GetData(TYPE_GUNSHIP_BATTLE) == FAIL)
69  {
70  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, GOSSIP_ITEM_ID_START_HORDE, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
71  pPlayer->SEND_GOSSIP_MENU(TEXT_ID_START_HORDE, pCreature->GetObjectGuid());
72  }
73  else
74  pPlayer->SEND_GOSSIP_MENU(pPlayer->GetGossipTextId(pCreature), pCreature->GetObjectGuid());
75  }
76 
77  return true;
78 }

135  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
136  {
137  Unit* caster = spell->GetAffectiveCaster();
138  Unit* target = spell->GetUnitTarget();
139  if (!target || !caster)
140  return;
141 
142  if (effIdx == EFFECT_INDEX_1)
143  {
144  uint32 damage = spell->GetDamage();
145  uint32 energy = caster->GetPower(caster->GetPowerType());
146 
147  // Note: this calculation has to be verified
148  spell->SetDamage(damage + energy * energy * 8);
149  }
150  // remove all power
151  else if (effIdx == EFFECT_INDEX_2)
152  caster->SetPower(caster->GetPowerType(), 0);
153  }
