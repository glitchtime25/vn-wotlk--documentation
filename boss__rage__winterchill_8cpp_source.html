67  uint32 GetInitialActionTimer(const uint32 action) const
68  {
69  switch (action)
70  {
71  case WINTERCHILL_ACTION_FROST_ARMOR: return urand(18000, 24000);
72  case WINTERCHILL_ACTION_ICEBOLT: return urand(5000, 9000);
73  case WINTERCHILL_ACTION_FROST_NOVA: return urand(12000, 17000);
74  case WINTERCHILL_ACTION_DEATH_AND_DECAY: return urand(21000, 28000);
75  case WINTERCHILL_ACTION_ENRAGE: return 600000;
76  default: return 0; // never occurs but for compiler
77  }

106  void ExecuteAction(uint32 action) override
107  {
108  switch (action)
109  {
110  case WINTERCHILL_ACTION_FROST_ARMOR:
111  {
112  if (DoCastSpellIfCan(nullptr, SPELL_FROST_ARMOR) == CAST_OK)
113  ResetCombatAction(action, GetSubsequentActionTimer(action));
114  break;
115  }
116  case WINTERCHILL_ACTION_ICEBOLT:
117  {
118  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_ICEBOLT, SELECT_FLAG_PLAYER))
119  if (DoCastSpellIfCan(target, SPELL_ICEBOLT) == CAST_OK)
120  ResetCombatAction(action, GetSubsequentActionTimer(action));
121  break;
122  }
123  case WINTERCHILL_ACTION_FROST_NOVA:
124  {
125  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_FROST_NOVA, SELECT_FLAG_PLAYER))
126  {
127  if (DoCastSpellIfCan(target, SPELL_FROST_NOVA) == CAST_OK)
128  {
129  DoScriptText(urand(0, 1) ? SAY_FROST_NOVA1 : SAY_FROST_NOVA2, m_creature);
130  ResetCombatAction(action, GetSubsequentActionTimer(action));
131  }
132  }
133  break;
134  }
135  case WINTERCHILL_ACTION_DEATH_AND_DECAY:
136  {
137  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_DEATH_AND_DECAY, SELECT_FLAG_PLAYER))
138  {
139  if (DoCastSpellIfCan(target, SPELL_DEATH_AND_DECAY, CAST_ONLY_XYZ) == CAST_OK)
140  {
141  DoScriptText(urand(0, 1) ? SAY_DND1 : SAY_DND2, m_creature);
142  ResetCombatAction(action, GetSubsequentActionTimer(action));
143  }
144  }
145  break;
146  }
147  case WINTERCHILL_ACTION_ENRAGE:
148  {
149  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
150  {
151  DoScriptText(SAY_ENRAGE, m_creature);
152  ResetCombatAction(action, GetSubsequentActionTimer(action));
153  }
154  break;
155  }
156  }
157  }
