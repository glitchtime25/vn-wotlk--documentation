170  void UpdateAI(const uint32 uiDiff) override
171  {
172  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
173  return;
174 
175  if (GetCombatScriptStatus())
176  {
177  if (m_overrunExecTimer <= uiDiff)
178  {
179  m_creature->CastSpell(nullptr, SPELL_OVERRUN_DAMAGE, TRIGGERED_NONE);
180  m_overrunExecTimer = 250;
181  }
182  else
183  m_overrunExecTimer -= uiDiff;
184  return;
185  }
186 
187  // Spell Enrage, when hp <= 20% gain enrage
188  if (m_creature->GetHealthPercent() <= 20.0f && !m_bHasEnrage)
189  {
190  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
191  {
192  DoScriptText(EMOTE_FRENZY, m_creature);
193  m_bHasEnrage = true;
194  }
195  }
196 
197  // Spell Overrun
198  if (m_uiOverrunTimer < uiDiff)
199  {
200  std::vector<Unit*> targets;
201  m_creature->SelectAttackingTargets(targets, ATTACKING_TARGET_ALL_SUITABLE, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_RANGE_AOE_RANGE, m_overrunParams);
202  while (targets.size() > COUNT_OVERRUN)
203  targets.erase(targets.begin() + urand(0, targets.size() - 1));
204 
205  if (!targets.empty() && DoCastSpellIfCan(nullptr, SPELL_OVERRUN) == CAST_OK)
206  {
207  float angle = m_creature->GetAngle(targets[0]);
208  m_creature->SetFacingTo(angle);
209  m_creature->SetOrientation(angle);
210  m_creature->CastSpell(nullptr, SPELL_OVERRUN_TRIGGER_SPAWN, TRIGGERED_OLD_TRIGGERED); // shouldnt be sent to client
211  DoScriptText(urand(0, 1) ? SAY_OVERRUN_1 : SAY_OVERRUN_2, m_creature);
212  m_uiOverrunTimer = urand(25000, 40000);
213  }
214  }
215  else
216  m_uiOverrunTimer -= uiDiff;
217 
218  // Spell Earthquake
219  if (m_uiQuakeTimer < uiDiff)
220  {
221  if (DoCastSpellIfCan(m_creature, SPELL_EARTHQUAKE) == CAST_OK)
222  {
223  DoScriptText(urand(0, 1) ? SAY_EARTHQUAKE_1 : SAY_EARTHQUAKE_2, m_creature);
224  m_uiQuakeTimer = urand(30000, 55000);
225  }
226  }
227  else
228  m_uiQuakeTimer -= uiDiff;
229 
230  // Spell Chain Lightning
231  if (m_uiChainTimer < uiDiff)
232  {
233  Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
234  if (!pTarget)
235  pTarget = m_creature->GetVictim();
236 
237  if (pTarget)
238  {
239  if (DoCastSpellIfCan(pTarget, SPELL_LIGHTNING_WRATH) == CAST_OK)
240  m_uiChainTimer = urand(7000, 27000);
241  }
242  }
243  else
244  m_uiChainTimer -= uiDiff;
245 
246  // Spell Sunder Armor
247  if (m_uiArmorTimer < uiDiff)
248  {
249  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CRUSH_ARMOR) == CAST_OK)
250  m_uiArmorTimer = urand(10000, 25000);
251  }
252  else
253  m_uiArmorTimer -= uiDiff;
254 
255  DoMeleeAttackIfReady();
256  }
