180  void DamageTaken(Unit* /*pDealer*/, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
181  {
182  if (m_uiPhase == PHASE_GHOST)
183  return;
184 
185  if (uiDamage >= m_creature->GetHealth())
186  {
187  uiDamage = 0;
188 
189  if (m_uiPhase == PHASE_TRANSITION)
190  return;
191 
192  // start transition phase
193  if (m_uiPhase == PHASE_DEATH_KNIGHT)
194  {
195  m_uiNextPhase = PHASE_SKELETON;
196 
197  // ghould explodes at the end of the round
198  if (Creature* pGhoul = m_creature->GetMap()->GetCreature(m_ghoulGuid))
199  {
200  pGhoul->InterruptNonMeleeSpells(true);
201  pGhoul->CastSpell(pGhoul, SPELL_EXPLODE, TRIGGERED_NONE);
202  }
203  }
204  else if (m_uiPhase == PHASE_SKELETON)
205  m_uiNextPhase = PHASE_GHOST;
206 
207  m_creature->InterruptNonMeleeSpells(true);
208  m_creature->SetHealth(0);
209  m_creature->StopMoving();
210  m_creature->ClearComboPointHolders();
211  m_creature->RemoveAllAurasOnDeath();
212  m_creature->ModifyAuraState(AURA_STATE_HEALTHLESS_20_PERCENT, false);
213  m_creature->ModifyAuraState(AURA_STATE_HEALTHLESS_35_PERCENT, false);
214  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
215  m_creature->ClearAllReactives();
216  m_creature->GetMotionMaster()->Clear();
217  m_creature->GetMotionMaster()->MoveIdle();
218  m_creature->SetStandState(UNIT_STAND_STATE_DEAD);
219 
220  DoCastSpellIfCan(m_creature, SPELL_FEIGN_DEATH, CAST_TRIGGERED);
221  DoCastSpellIfCan(m_creature, SPELL_CLEAR_ALL_DEBUFFS, CAST_TRIGGERED);
222  DoCastSpellIfCan(m_creature, SPELL_FULL_HEAL, CAST_TRIGGERED);
223  m_uiPhase = PHASE_TRANSITION;
224  }
225  }

227  void ReceiveAIEvent(AIEventType eventType, Unit* /*pSender*/, Unit* pInvoker, uint32 /*uiMiscValue*/) override
228  {
229  // finish transition
230  if (eventType == AI_EVENT_CUSTOM_A)
231  {
232  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
233  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
234  m_creature->GetMotionMaster()->Clear();
235  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
236  DoResetThreat();
237 
238  m_uiPhase = m_uiNextPhase;
239 
240  if (m_uiPhase == PHASE_SKELETON)
241  {
242  DoScriptText(SAY_PHASE_2, m_creature);
243  m_creature->SetDisplayId(MODEL_ID_SKELETON);
244 
245  DoCastSpellIfCan(m_creature, SPELL_ARMY_OF_THE_DEAD);
246  }
247  else if (m_uiPhase == PHASE_GHOST)
248  {
249  DoScriptText(SAY_PHASE_3, m_creature);
250  m_creature->SetDisplayId(MODEL_ID_GHOST);
251 
252  SetEquipmentSlots(false, EQUIP_UNEQUIP, EQUIP_NO_CHANGE, EQUIP_NO_CHANGE);
253  }
254  }

257  void UpdateAI(const uint32 uiDiff) override
258  {
259  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
260  return;
261 
262  switch (m_uiPhase)
263  {
264  case PHASE_DEATH_KNIGHT:
265 
266  if (m_uiDeathsRespiteTimer < uiDiff)
267  {
268  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
269  {
270  if (DoCastSpellIfCan(pTarget, SPELL_DEATHS_RESPITE) == CAST_OK)
271  m_uiDeathsRespiteTimer = 20000;
272  }
273  }
274  else
275  m_uiDeathsRespiteTimer -= uiDiff;
276 
277  if (m_uiIcyTouchTimer < uiDiff)
278  {
279  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_ICY_TOUCH, SELECT_FLAG_IN_MELEE_RANGE | SELECT_FLAG_IN_LOS))
280  {
281  if (DoCastSpellIfCan(pTarget, SPELL_ICY_TOUCH) == CAST_OK)
282  m_uiIcyTouchTimer = urand(10000, 15000);
283  }
284  }
285  else
286  m_uiIcyTouchTimer -= uiDiff;
287 
288  if (m_uiObliterateTimer < uiDiff)
289  {
290  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_OBLITERATE) == CAST_OK)
291  m_uiObliterateTimer = urand(18000, 25000);
292  }
293  else
294  m_uiObliterateTimer -= uiDiff;
295 
296  if (m_uiPlagueStrikeTimer < uiDiff)
297  {
298  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_PLAGUE_STRIKE) == CAST_OK)
299  m_uiPlagueStrikeTimer = 10000;
300  }
301  else
302  m_uiPlagueStrikeTimer -= uiDiff;
303 
304  break;
305  case PHASE_SKELETON:
306 
307  if (m_uiDesecrationTimer < uiDiff)
308  {
309  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_DESECRATION) == CAST_OK)
310  m_uiDesecrationTimer = 6000;
311  }
312  else
313  m_uiDesecrationTimer -= uiDiff;
314 
315  if (m_uiGhoulExplodeTimer < uiDiff)
316  {
317  if (DoCastSpellIfCan(m_creature, SPELL_GHOUL_EXPLODE) == CAST_OK)
318  m_uiGhoulExplodeTimer = 12000;
319  }
320  else
321  m_uiGhoulExplodeTimer -= uiDiff;
322 
323  break;
324  case PHASE_GHOST:
325 
326  if (m_uiDeathsBiteTimer < uiDiff)
327  {
328  if (DoCastSpellIfCan(m_creature, SPELL_DEATHS_BITE) == CAST_OK)
329  m_uiDeathsBiteTimer = 2000;
330  }
331  else
332  m_uiDeathsBiteTimer -= uiDiff;
333 
334  if (m_uiMarkedDeathTimer < uiDiff)
335  {
336  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
337  {
338  if (DoCastSpellIfCan(pTarget, SPELL_MARKED_FOR_DEATH) == CAST_OK)
339  m_uiMarkedDeathTimer = urand(10000, 15000);
340  }
341  }
342  else
343  m_uiMarkedDeathTimer -= uiDiff;
344  break;
345  }
346 
347  DoMeleeAttackIfReady();
348  }

419  }
420 
421  if (m_uiClawTimer < uiDiff)
422  {
423  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLAW) == CAST_OK)
424  m_uiClawTimer = urand(7000, 14000);
