40 void instance_deadmines::OnPlayerEnter(Player* pPlayer)
41 {
42  // Respawn the Mysterious chest if one of the players who enter the instance has the quest in his log
43  if (pPlayer->GetQuestStatus(QUEST_FORTUNE_AWAITS) == QUEST_STATUS_COMPLETE &&
44  !pPlayer->GetQuestRewardStatus(QUEST_FORTUNE_AWAITS))
45  DoRespawnGameObject(GO_MYSTERIOUS_CHEST, HOUR);
46 
47  if (m_firstEnter)
48  {
49  m_firstEnter = false;
50 
51  if (m_auiEncounter[TYPE_RHAHKZOR] == DONE)
52  SpawnFirstDeadminesPatrol();
53 
54  if (m_auiEncounter[TYPE_SNEED] == DONE)
55  SpawnSecondDeadminesPatrol();
56 
57  if (m_auiEncounter[TYPE_GILNID] == DONE)
58  SpawnThirdDeadminesPatrol();
59  }
60 }

87 void instance_deadmines::OnObjectCreate(GameObject* pGo)
88 {
89  switch (pGo->GetEntry())
90  {
91  case GO_FACTORY_DOOR:
92  if (m_auiEncounter[TYPE_RHAHKZOR] == DONE)
93  pGo->SetGoState(GO_STATE_ACTIVE);
94 
95  break;
96  case GO_MAST_ROOM_DOOR:
97  if (m_auiEncounter[TYPE_SNEED] == DONE)
98  pGo->SetGoState(GO_STATE_ACTIVE);
99 
100  break;
101  case GO_FOUNDRY_DOOR:
102  if (m_auiEncounter[TYPE_GILNID] == DONE)
103  pGo->SetGoState(GO_STATE_ACTIVE);
104 
105  break;
106  case GO_IRON_CLAD_DOOR:
107  if (m_auiEncounter[TYPE_IRON_CLAD_DOOR] == DONE)
108  pGo->SetGoState(GO_STATE_ACTIVE_ALTERNATIVE);
109 
110  break;
111  case GO_DEFIAS_CANNON:
112  case GO_SMITE_CHEST:
113  case GO_MYSTERIOUS_CHEST:
114  break;
115 
116  default:
117  return;
118  }
119 
120  m_goEntryGuidStore[pGo->GetEntry()] = pGo->GetObjectGuid();

154  }
155  case TYPE_GILNID:
156  {
157  if (uiData == DONE)
158  DoUseDoorOrButton(GO_FOUNDRY_DOOR);
159 
160  SpawnThirdDeadminesPatrol();
161  m_auiEncounter[uiType] = uiData;
162  break;
163  }
164  case TYPE_IRON_CLAD_DOOR:
165  {
166  // delayed door animation to sync with Defias Cannon animation
167  if (uiData == DONE)
168  m_uiIronDoorTimer = 500;
169 
170  m_auiEncounter[uiType] = uiData;
171  break;
172  }
173  }
174 
175  if (uiData == DONE)
176  {
177  OUT_SAVE_INST_DATA;
178 
179  std::ostringstream saveStream;
180  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
181 
182  m_strInstData = saveStream.str();
183 
184  SaveToDB();
185  OUT_SAVE_INST_DATA_COMPLETE;
186  }

219 void instance_deadmines::Update(uint32 uiDiff)
220 {
221  if (m_uiIronDoorTimer)
222  {
223  if (m_uiIronDoorTimer <= uiDiff)
224  {
225  switch (m_uiDoorStep)
226  {
227  case 0:
228  DoUseDoorOrButton(GO_IRON_CLAD_DOOR, 0, true);
229 
230  if (Creature* pMrSmite = GetSingleCreatureFromStorage(NPC_MR_SMITE))
231  DoScriptText(INST_SAY_ALARM1, pMrSmite);
232 
233  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_IRON_CLAD_DOOR))
234  {
235  for (uint32 i = 148u; i <= 149; ++i)
236  if (Creature* guardfast = pDoor->GetMap()->GetCreature(GUID_PREFIX + i))
237  guardfast->GetMotionMaster()->MovePoint(0, pDoor->GetPositionX(), pDoor->GetPositionY(), pDoor->GetPositionZ(), FORCED_MOVEMENT_RUN, true);
238 
239  if (Creature* guardlow = pDoor->GetMap()->GetCreature(GUID_PREFIX + 150))
240  guardlow->GetMotionMaster()->MovePoint(0, pDoor->GetPositionX(), pDoor->GetPositionY(), pDoor->GetPositionZ(), FORCED_MOVEMENT_WALK, true);
241  }
242  ++m_uiDoorStep;
243  m_uiIronDoorTimer = 15000;
244  break;
245  case 1:
246  if (Creature* pMrSmite = GetSingleCreatureFromStorage(NPC_MR_SMITE))
247  DoScriptText(INST_SAY_ALARM2, pMrSmite);
248 
249  m_uiDoorStep = 0;
250  m_uiIronDoorTimer = 0;
251  break;
252  }
253  }
254  else
255  m_uiIronDoorTimer -= uiDiff;
256  }
