26 Channel::Channel(const std::string& name, uint32 channel_id/* = 0*/)
27  : m_name(name)
28 {
29  if (ChatChannelsEntry const* builtin = GetChatChannelsEntryFor(name, channel_id))
30  {
31  m_entry = builtin; // built-in channel entry
32  m_announcements = false; // no join/leave announcements by default
33  m_flags = CHANNEL_FLAG_GENERAL; // default for all built-in channels
34 
35  if (builtin->flags & CHANNEL_DBC_FLAG_TRADE) // for trade channel
36  m_flags |= CHANNEL_FLAG_TRADE;
37 
38  if (builtin->flags & CHANNEL_DBC_FLAG_CITY_ONLY2) // for city only channels
39  m_flags |= CHANNEL_FLAG_CITY;
40 
41  if (builtin->flags & CHANNEL_DBC_FLAG_LFG) // for LFG channel
42  m_flags |= CHANNEL_FLAG_LFG;
43  else // for all other channels
44  m_flags |= CHANNEL_FLAG_NOT_LFG;
45 
46  // Custom features:
47  m_realmzone = true; // channel language matches realm zone
48  }
49  else // it's custom channel
50  {
51  m_announcements = true; // join/leave announcements enabled by default
52  m_flags = CHANNEL_FLAG_CUSTOM; // default for all custom channels
53 
54  // Custom features:
55  m_realmzone = sObjectMgr.CheckPublicMessageLanguage(m_name); // channel language matches channel name
56  }
57 }

59 void Channel::Join(Player* player, const char* password)
60 {
61  ObjectGuid guid = player->GetObjectGuid();
62 
63  WorldPacket data;
64  if (IsOn(guid))
65  {
66  if (!IsConstant()) // non send error message for built-in channels
67  {
68  MakePlayerAlreadyMember(data, m_name, guid);
69  SendToOne(data, guid);
70  }
71  return;
72  }
73 
74  if (IsBanned(guid))
75  {
76  MakeBanned(data, m_name);
77  SendToOne(data, guid);
78  return;
79  }
80 
81  bool checkPassword = (m_password.length() > 0);
82 
83  if (const uint32 level = sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_CHANNEL_MODERATION))
84  if (player->GetSession()->GetSecurity() >= level)
85  checkPassword = false;
86 
87  if (checkPassword && strcmp(password, m_password.c_str()))
88  {
89  MakeWrongPassword(data, m_name);
90  SendToOne(data, guid);
91  return;
92  }
93 
94  if (HasFlag(CHANNEL_FLAG_LFG) && sWorld.getConfig(CONFIG_BOOL_CHANNEL_RESTRICTED_LFG))
95  {
96  if (player->GetSession()->GetSecurity() == SEC_PLAYER)
97  {
98  MakeNotInLFG(data, m_name);
99  SendToOne(data, guid);
100  return;
101  }
102  }
103 
104  if (player->GetGuildId() && (GetFlags() == 0x38))
105  return;
106 
107  // join channel
108  player->JoinedChannel(this);
109 
110  const uint32 level = sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_CHANNEL_SILENT_JOIN);
111  const bool silent = (level && player->GetSession()->GetSecurity() >= level);
112 
113  if (m_announcements && !silent)
114  {
115  MakeJoined(data, m_name, guid);
116  SendToAll(data);
117  }
118 
119  data.clear();
120 
121  PlayerInfo& pinfo = m_players[guid];
122  pinfo.player = guid;
123  pinfo.flags = MEMBER_FLAG_NONE;
124 
125  MakeYouJoined(data, m_name, *this);
126  SendToOne(data, guid);
127 
128  JoinNotify(guid);
129 
130  // if no owner first logged will become
131  if (!IsPublic() && !m_ownerGuid)
132  SetOwner(guid, (m_players.size() > 1));
133 
134  // try to auto-convert this channel to static upon reaching treshold
135  SetStatic(true);
136 }

194

196  const bool gm = (level && player->GetSession()->GetSecurity() >= level);
197

199  {
200  WorldPacket data;
201  MakeNotModerator(data, m_name);
202  SendToOne(data, guid);
203  return;
204  }
205 
206  Player* target = sObjectMgr.GetPlayer(targetName);
207  if (!target)
208  {
209  WorldPacket data;
210  MakePlayerNotFound(data, m_name, targetName);
211  SendToOne(data, guid);
212  return;
213  }
214 
215  ObjectGuid targetGuid = target->GetObjectGuid();
216  if (!IsOn(targetGuid))
217  {
218  WorldPacket data;
219  MakePlayerNotFound(data, m_name, targetName);
220  SendToOne(data, guid);
221  return;
222  }
223 
224  bool changeowner = m_ownerGuid == targetGuid;
225 
226  if (!gm && changeowner && guid != m_ownerGuid)
227  {
228  WorldPacket data;
229  MakeNotOwner(data, m_name);
230  SendToOne(data, guid);
231  return;
232  }
233 
234  // kick or ban player
235  WorldPacket data;
236 
237  if (ban && !IsBanned(targetGuid))
238  {
239  m_banned.insert(targetGuid);
240  MakePlayerBanned(data, m_name, targetGuid, guid);
241  }
242  else
243  MakePlayerKicked(data, m_name, targetGuid, guid);
244 
245  SendToAll(data);
246  m_players.erase(targetGuid);
247  target->LeftChannel(this);
248 
249  if (changeowner && !IsPublic())
250  SetOwner(SelectNewOwner(), (m_players.size() > 1));

333 void Channel::SetModeFlags(Player* player, const char* targetName, ChannelMemberFlags flags, bool set)
334 {
335  // Restrict input flags to currently supported by this method
336  flags = ChannelMemberFlags(uint8(flags) & (MEMBER_FLAG_MODERATOR | MEMBER_FLAG_MUTED));
337 
338  if (!flags)
339  return;
340 
341  ObjectGuid guid = player->GetObjectGuid();
342 
343  if (!IsOn(guid))
344  {
345  WorldPacket data;
346  MakeNotMember(data, m_name);
347  SendToOne(data, guid);
348  return;
349  }
350 
351  const uint32 level = sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_CHANNEL_MODERATION);
352  const bool gm = (level && player->GetSession()->GetSecurity() >= level);
353 
354  if (!m_players[guid].IsModerator() && !gm)
355  {
356  WorldPacket data;
357  MakeNotModerator(data, m_name);
358  SendToOne(data, guid);
359  return;
360  }
361 
362  Player* target = sObjectMgr.GetPlayer(targetName);
363  if (!target)
364  {
365  WorldPacket data;
366  MakePlayerNotFound(data, m_name, targetName);
367  SendToOne(data, guid);
368  return;
369  }
370 
371  ObjectGuid targetGuid = target->GetObjectGuid();
372 
373  if ((flags & MEMBER_FLAG_MODERATOR) && guid == m_ownerGuid && targetGuid == m_ownerGuid)
374  return;
375 
376  if (!IsOn(targetGuid))
377  {
378  WorldPacket data;
379  MakePlayerNotFound(data, m_name, targetName);
380  SendToOne(data, guid);
381  return;
382  }
383 
384  const bool gmTarget = (level && target->GetSession()->GetSecurity() >= level);
385 
386  // allow make moderator from another team only if both is GMs
387  // at this moment this only way to show channel post for GM from another team
388  if ((!gm || !gmTarget) && player->GetTeam() != target->GetTeam() && !sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_CHANNEL))
389  {
390  WorldPacket data;
391  MakePlayerNotFound(data, m_name, targetName);
392  SendToOne(data, guid);
393  return;
394  }
395 
396  if (m_ownerGuid == targetGuid && m_ownerGuid != guid)
397  {
398  WorldPacket data;
399  MakeNotOwner(data, m_name);
400  SendToOne(data, guid);
401  return;
402  }

493 void Channel::SendChannelListResponse(Player* player, bool display/*= false*/)
494 {
495  ObjectGuid guid = player->GetObjectGuid();
496 
497  if (!IsOn(guid))
498  {
499  WorldPacket data;
500  MakeNotMember(data, m_name);
501  SendToOne(data, guid);
502  return;
503  }
504 
505  // list players in channel
506  WorldPacket data(SMSG_CHANNEL_LIST, 1 + (GetName().size() + 1) + 1 + 4 + m_players.size() * (8 + 1));
507  data << uint8(display); // response type: 0 - /chatinfo query, 1 - display list query
508  data << GetName(); // channel name
509  data << uint8(GetFlags()); // channel flags?
510 
511  size_t countpos = data.wpos();
512  data << uint32(0); // size of list, placeholder
513 
514  // PLAYER can't see MODERATOR, GAME MASTER, ADMINISTRATOR characters
515  // MODERATOR, GAME MASTER, ADMINISTRATOR can see all
516  const bool visibilityCheck = (player->GetSession()->GetSecurity() == SEC_PLAYER);
517  AccountTypes visibilityThreshold = AccountTypes(sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_IN_WHO_LIST));
518 
519  uint32 count = 0;
520  for (PlayerList::const_iterator i = m_players.begin(); i != m_players.end(); ++i)
521  {
522  if (Player* member = sObjectMgr.GetPlayer(i->first))
523  {
524  if (visibilityCheck && (member->GetSession()->GetSecurity() > visibilityThreshold || !member->IsVisibleGloballyFor(player)))
525  continue;
526 
527  data << ObjectGuid(i->first);
528  data << uint8(i->second.flags); // flags seems to be changed...
529  ++count;
530  }
531  }
532 
533  data.put<uint32>(countpos, count);
534 
535  SendToOne(data, guid);
536 }

608 void Channel::Say(Player* player, const char* text, uint32 lang)
609 {
610  if (!text)
611  return;
612 
613  ObjectGuid guid = player->GetObjectGuid();
614 
615  if (!IsOn(guid))
616  {
617  WorldPacket data;
618  MakeNotMember(data, m_name);
619  SendToOne(data, guid);
620  return;
621  }
622 
623  if (m_players[guid].IsMuted())
624  {
625  WorldPacket data;
626  MakeMuted(data, m_name);
627  SendToOne(data, guid);
628  return;
629  }
630 
631  const bool moderator = m_players[guid].IsModerator();
632 
633  const uint32 level = sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_CHANNEL_MODERATION);
634  const bool gm = (level && player->GetSession()->GetSecurity() >= level);
635 
636  if (m_moderation && !moderator && !gm)
637  {
638  WorldPacket data;
639  MakeNotModerator(data, m_name);
640  SendToOne(data, guid);
641  return;
642  }
643 
644  if (uint32 restriction = sWorld.getConfig(CONFIG_UINT32_CHANNEL_RESTRICTED_LANGUAGE_MODE))
645  {
646  bool restricted = false;
647 
648  switch (restriction)
649  {
650  case 1: restricted = IsConstant(); break;
651  case 2: restricted = (IsPublic() && m_realmzone); break;
652  case 3: restricted = true; break;
653  }
654 
655  if (restricted && !sObjectMgr.CheckPublicMessageLanguage(text))
656  {
657  WorldPacket data;
658  MakeMuted(data, m_name);
659  SendToOne(data, guid);
660  return;
661  }
662  }
663 
664  // send channel message
665  if (sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_CHANNEL))
666  lang = LANG_UNIVERSAL;
667 
668  auto const silenced = player->GetSession()->GetAnticheat()->IsSilenced();
669 
670  WorldPacket data;
671  ChatHandler::BuildChatPacket(data, CHAT_MSG_CHANNEL, text, Language(lang), player->GetChatTag(), guid, player->GetName(), ObjectGuid(), "", m_name.c_str());
672  // if the source is silenced, send only to them
673  if (silenced)
674  player->GetSession()->SendPacket(data);
675  else
676  SendMessage(data, (moderator ? ObjectGuid() : guid));
677 }

1098 bool Channel::SetStatic(bool state, bool command/* = false*/)
1099 {
1100  // Only custom channels can be set to static
1101  if (IsConstant() || !HasFlag(CHANNEL_FLAG_CUSTOM) || m_static == state)
1102  return false;
1103 
1104  // This conversion cannot be performed if password is set - it has to be removed first
1105  if (state && !m_password.empty())
1106  return false;
1107 
1108  // Treshold for auto-conversion
1109  const uint32 treshold = sWorld.getConfig(CONFIG_UINT32_CHANNEL_STATIC_AUTO_TRESHOLD);
1110 
1111  // Detect auto-converion and reaching treshold
1112  if (!command && (!treshold || state != (GetNumPlayers() >= treshold)))
1113  return false;
1114 
1115  // Strip moderator privileges
1116  if (state)
1117  {
1118  for (PlayerList::const_iterator i = m_players.begin(); i != m_players.end(); ++i)
1119  {
1120  if (i->second.IsModerator())
1121  SetModeFlags(i->second.player, MEMBER_FLAG_MODERATOR, false);
1122  }
1123  }
1124 
1125  // Unset/Set channel owner
1126  if (state == bool(m_ownerGuid))
1127  SetOwner(state ? ObjectGuid() : SelectNewOwner());
1128 
1129  // Disable premoderation mode on conversion to static
1130  if (state && m_moderation)
1131  {
1132  m_moderation = false;
1133 
1134  WorldPacket data;
1135  MakeModerationOff(data, m_name, ObjectGuid());
1136  SendToAll(data);
1137  }
1138 
1139  // Disable announcements on conversion to static
1140  if (state && m_announcements)
1141  {
1142  m_announcements = false;
1143 
1144  WorldPacket data;
1145  MakeAnnouncementsOff(data, m_name, ObjectGuid());
1146  SendToAll(data);
1147  }
1148 
1149  m_static = state;
1150 
1151  return true;
1152 }
