55 bool npc_escortAI::AssistPlayerInCombat(Unit* who)
56 {
57  if (!HasEscortState(STATE_ESCORT_ESCORTING))
58  return false;
59 
60  if (!who->GetVictim())
61  return false;
62 
63  // experimental (unknown) flag not present
64  if (!(m_creature->GetCreatureInfo()->CreatureTypeFlags & CREATURE_TYPEFLAGS_CAN_ASSIST))
65  return false;
66 
67  // unit state prevents (similar check is done in CanInitiateAttack which also include checking unit_flags. We skip those here)
68  if (m_creature->hasUnitState(UNIT_STAT_STUNNED | UNIT_STAT_FEIGN_DEATH | UNIT_STAT_CONFUSED | UNIT_STAT_FLEEING))
69  return false;
70 
71  // victim of pWho is not a player
72  if (!who->GetVictim()->GetBeneficiaryPlayer())
73  return false;
74 
75  // never attack friendly
76  if (!m_creature->CanAttackNow(who))
77  return false;
78 
79  // too far away and no free sight?
80  if (m_creature->IsWithinDistInMap(who, MAX_PLAYER_DISTANCE) && m_creature->IsWithinLOSInMap(who))
81  {
82  // already fighting someone?
83  if (!m_creature->GetVictim())
84  {
85  AttackStart(who);
86  return true;
87  }
88  m_creature->EngageInCombatWith(who);
89  return true;
90  }
91 
92  return false;
93 }

129 void npc_escortAI::UpdateAI(const uint32 diff)
130 {
131  // Check if player or any member of his group is within range
132  if (m_questForEscort && HasEscortState(STATE_ESCORT_ESCORTING) && m_playerGuid && !m_creature->GetVictim() && !HasEscortState(STATE_ESCORT_RETURNING))
133  {
134  if (m_playerCheckTimer < diff)
135  {
136  if (!HasEscortState(STATE_ESCORT_PAUSED) && !IsPlayerOrGroupInRange())
137  {
138  // set the quest status as failed
139  FailQuestForPlayerAndGroup();
140 
141  // TODO: i am not sure this is correct, isn't that the creature should continue until it get killed?
142  debug_log("SD2: EscortAI failed because player/group was to far away or not found");
143 
144  if (m_canInstantRespawn)
145  {
146  m_creature->SetDeathState(JUST_DIED);
147  m_creature->Respawn();
148  }
149  else
150  {
151  if (m_creature->IsPet())
152  static_cast<Pet*>(m_creature)->Unsummon(PET_SAVE_AS_DELETED); // we assume escort AI pet to always be non-saved
153  else if (auto group = m_creature->GetCreatureGroup())
154  group->Despawn();
155  else
156  m_creature->ForcedDespawn();
157  }
158 
159  return;
160  }
161 
162  m_playerCheckTimer = 1000;
163  }
164  else
165  m_playerCheckTimer -= diff;
166  }
167 
168  UpdateEscortAI(diff);
169 }

182 void npc_escortAI::MovementInform(uint32 moveType, uint32 pointId)
183 {
184  if (!IsSD2EscortMovement(moveType) || !HasEscortState(STATE_ESCORT_ESCORTING))
185  return;
186 
187  //uint32 pathId = uiMoveType & 0xFF;
188 
189  if (moveType < EXTERNAL_WAYPOINT_MOVE_START)
190  WaypointReached(pointId);
191  else if (moveType < EXTERNAL_WAYPOINT_FINISHED_LAST)
192  WaypointStart(pointId);
193  else // Last WP Reached
194  {
195  if (m_canInstantRespawn)
196  {
197  m_creature->SetDeathState(JUST_DIED);
198  m_creature->Respawn();
199  }
200  else
201  m_creature->ForcedDespawn();
202  }

239 void npc_escortAI::Start(bool run, const Player* player, const Quest* quest, bool instantRespawn, bool canLoopPath, uint32 waypointPath)
240 {
241  if (m_creature->GetVictim())
242  {
243  script_error_log("EscortAI attempt to Start while in combat for %s.", m_creature->GetScriptName().data());
244  return;
245  }
246 
247  if (HasEscortState(STATE_ESCORT_ESCORTING))
248  {
249  script_error_log("EscortAI attempt to Start while already escorting for %s.", m_creature->GetScriptName().data());
250  return;
251  }
252 
253  uint32 pathId = m_waypointPathID;
254  WaypointPathOrigin origin = PATH_FROM_EXTERNAL;
255  if (waypointPath)
256  {
257  pathId = waypointPath;
258  origin = PATH_FROM_WAYPOINT_PATH;
259  }
260 
261  if (!sWaypointMgr.GetPathFromOrigin(m_creature->GetEntry(), m_creature->GetGUIDLow(), pathId, origin))
262  {
263  script_error_log("EscortAI attempt to start escorting for %s, but has no waypoints loaded.", m_creature->GetScriptName().data());
264  return;
265  }
266 
267  if (origin == PATH_FROM_WAYPOINT_PATH)
268  m_currentEscortWaypointPath = pathId;
269 
270  // set variables
271  m_isRunning = run;
272 
273  m_playerGuid = player ? player->GetObjectGuid() : ObjectGuid();
274  m_questForEscort = quest;
275 
276  m_canInstantRespawn = instantRespawn;
277  m_canReturnToStart = canLoopPath;
278 
279  if (m_canReturnToStart && m_canInstantRespawn)
280  debug_log("SD2: EscortAI is set to return home after waypoint end and instant respawn at waypoint end. Creature will never despawn.");
281 
282  // disable npcflags
283  m_creature->SetUInt32Value(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_NONE);
284 
285  AddEscortState(STATE_ESCORT_ESCORTING);
286 
287  // Set initial speed
288  m_creature->SetWalk(!m_isRunning);
289 
290  m_creature->StopMoving();
291 
292  // Start moving along the path with 2500ms delay
293  m_creature->GetMotionMaster()->Clear(false, true);
294  m_creature->GetMotionMaster()->MoveWaypoint(pathId, origin, 2500);
295 
296  JustStartedEscort();
297 }
