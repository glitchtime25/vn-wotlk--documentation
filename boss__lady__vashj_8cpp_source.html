242  void MovementInform(uint32 uiType, uint32 uiPointId) override
243  {
244  if (uiType != POINT_MOTION_TYPE)
245  return;
246 
247  if (uiPointId == POINT_MOVE_CENTER)
248  {
249  // Initialize all the shield generators
250  if (m_instance)
251  {
252  DoScriptText(SAY_PHASE2, m_creature);
253 
254  for (uint32 i = GO_SHIELD_GENERATOR_1; i <= GO_SHIELD_GENERATOR_4; ++i)
255  if (GameObject* shieldGenerator = m_instance->GetSingleGameObjectFromStorage(i))
256  shieldGenerator->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
257 
258  GuidList lShieldGeneratorsGuid;
259  m_instance->GetTinyTriggerGUIDList(lShieldGeneratorsGuid);
260 
261  for (GuidList::const_iterator itr = lShieldGeneratorsGuid.begin(); itr != lShieldGeneratorsGuid.end(); ++itr)
262  {
263  if (Creature* pGenerator = m_creature->GetMap()->GetCreature(*itr))
264  pGenerator->CastSpell(m_creature, SPELL_MAGIC_BARRIER, TRIGGERED_NONE);
265  }
266  }
267 
268  // only on first fight
269  if (m_triggerGuids.empty())
270  {
271  // set creature facing to PI for guid fills (proper North/South/East/West)
272  m_creature->SetOrientation(M_PI_F);
273 
274  // load all outer ring triggers
275  std::list<Creature*> list;
276  GetCreatureListWithEntryInGrid(list, m_creature, NPC_OLD_WORLD_TRIGGER, 150.f);
277  for (Creature* creature : list)
278  {
279  if (creature->GetDistance(m_creature) > 75.f) // outer ring
280  {
281  // needed to get the true angle
282  float creatureAngle = m_creature->GetAngle(creature) * (360 / (M_PI_F * 2));
283 
284  if (creatureAngle > 315.0f || creatureAngle < 45.0f)
285  m_triggerGuidsNorth.push_back(creature->GetObjectGuid());
286  else if (creatureAngle < 135.0f)
287  m_triggerGuidsEast.push_back(creature->GetObjectGuid());
288  else if (creatureAngle < 225.0f)
289  m_triggerGuidsSouth.push_back(creature->GetObjectGuid());
290  else
291  m_triggerGuidsWest.push_back(creature->GetObjectGuid());
292 
293  m_triggerGuidsAll.push_back(creature->GetObjectGuid());
294  }
295  }
296 
297  m_triggerGuids.push_back(m_triggerGuidsNorth);
298  m_triggerGuids.push_back(m_triggerGuidsSouth);
299  m_triggerGuids.push_back(m_triggerGuidsWest);
300  m_triggerGuids.push_back(m_triggerGuidsEast);
301  }
302 
303  m_phase = PHASE_2;
304 
305  m_creature->SetImmobilizedState(true);
306  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PACIFIED);
307 
308  SetCombatScriptStatus(false);
309  }
310  }
