172  DoScriptText(SAY_ESCAPE, m_creature);
173  if (Player* pPlayer = GetPlayerForEscort())
174  m_creature->SetFacingToObject(pPlayer);
175  SetRun();
176  break;
177  case 34:
178  DoScriptText(SAY_FINISH, m_creature);
179  m_creature->SetImmuneToNPC(true);
180  if (Creature* pQuae = GetClosestCreatureWithEntry(m_creature, NPC_QUAE, 10.0f))

183  m_creature->SetFacingToObject(pQuae);
184  }
185  break;
186  case 35:
187  if (Player* pPlayer = GetPlayerForEscort())

212  void UpdateEscortAI(const uint32 uiDiff) override
213  {
214  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
215  return;
216 
217  if (m_uiBearFormTimer < uiDiff)
218  {
219  if (DoCastSpellIfCan(m_creature, SPELL_BEAR_FORM) == CAST_OK)
220  m_uiBearFormTimer = urand(25000, 30000);
221  }
222  else
223  m_uiBearFormTimer -= uiDiff;
224 
225  if (m_uiHealTimer < uiDiff)
226  {
227  if (Unit* pTarget = DoSelectLowestHpFriendly(40.0f))
228  {
229  if (DoCastSpellIfCan(pTarget, SPELL_REJUVENATION) == CAST_OK)
230  m_uiHealTimer = urand(15000, 25000);
231  }
232  }
233  else
234  m_uiHealTimer -= uiDiff;
235 
236  DoMeleeAttackIfReady();
237  }
