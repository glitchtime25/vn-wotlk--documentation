143  // Wrapper to handle the blue beams animation
144  void DoHandleBeamHelpers(bool bReset)

146  if (!m_instance)
147  return;

149  GuidList lBeamHelpersGuid;
150  m_instance->GetBeamHelpersGUIDList(lBeamHelpersGuid);

171  void UpdateAI(const uint32 uiDiff) override
172  {
173  if (m_uiBeamInitTimer)
174  {
175  if (m_uiBeamInitTimer <= uiDiff)
176  {
177  DoHandleBeamHelpers(false);
178  m_uiBeamInitTimer = 0;
179  }
180  else
181  m_uiBeamInitTimer -= uiDiff;
182  }
183 
184  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
185  {
186  // handle elementals on OOC timer
187  if (m_uiElementalTimer < uiDiff)
188  {
189  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_WATER_ELEMENT) == CAST_OK)
190  m_uiElementalTimer = 20000;
191  }
192  else
193  m_uiElementalTimer -= uiDiff;
194 
195  return;
196  }
197 
198  // corrupted form
199  if (m_bCorruptedForm)
200  {
201  if (m_uiVileSludgeTimer < uiDiff)
202  {
203  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
204  {
205  if (DoCastSpellIfCan(pTarget, SPELL_VILE_SLUDGE) == CAST_OK)
206  m_uiVileSludgeTimer = 15000;
207  }
208  }
209  else
210  m_uiVileSludgeTimer -= uiDiff;
211 
212  // Change to clean
213  if (m_uiPosCheckTimer < uiDiff)
214  {
215  if (CheckTransition())
216  {
217  DoScriptText(SAY_SWITCH_TO_CLEAN, m_creature);
218  m_creature->RemoveAurasDueToSpell(SPELL_CORRUPTION);
219  m_uiMarkCount = 0;
220 
221  DoHandleBeamHelpers(false);
222  DoResetThreat();
223  DoSpawnAdds();
224 
225  m_creature->SetMeleeDamageSchool(SPELL_SCHOOL_FROST);
226  m_creature->ApplySpellImmune(nullptr, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, true);
227  m_creature->ApplySpellImmune(nullptr, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, false);
228 
229  m_bCorruptedForm = false;
230  m_uiMarkTimer = 15000;
231  }
232 
233  m_uiPosCheckTimer = 1000;
234  }
235  else
236  m_uiPosCheckTimer -= uiDiff;
237  }
238  // clean form
239  else
240  {
241  if (m_uiWaterTombTimer < uiDiff)
242  {
243  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
244  {
245  if (DoCastSpellIfCan(pTarget, SPELL_WATER_TOMB) == CAST_OK)
246  m_uiWaterTombTimer = 7000;
247  }
248  }
249  else
250  m_uiWaterTombTimer -= uiDiff;
251 
252  // Change to corrupt
253  if (m_uiPosCheckTimer < uiDiff)
254  {
255  if (!CheckTransition())
256  {
257  if (DoCastSpellIfCan(m_creature, SPELL_CORRUPTION) == CAST_OK)
258  {
259  DoScriptText(SAY_SWITCH_TO_CORRUPT, m_creature);
260  m_uiMarkCount = 0;
261 
262  DoHandleBeamHelpers(true);
263  DoResetThreat();
264  DoSpawnAdds();
265 
266  m_creature->SetMeleeDamageSchool(SPELL_SCHOOL_NATURE);
267  m_creature->ApplySpellImmune(nullptr, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_NATURE, true);
268  m_creature->ApplySpellImmune(nullptr, IMMUNITY_SCHOOL, SPELL_SCHOOL_MASK_FROST, false);
269 
270  m_bCorruptedForm = true;
271  m_uiMarkTimer = 15000;
272  }
273  }
274 
275  m_uiPosCheckTimer = 500;
276  }
277  else
278  m_uiPosCheckTimer -= uiDiff;
279  }
280 
281  // Apply mark debuff
282  if (m_uiMarkTimer < uiDiff)
283  {
284  if (DoCastSpellIfCan(m_creature, m_bCorruptedForm ? aMarkCorruption[m_uiMarkCount] : aMarkHydross[m_uiMarkCount]) == CAST_OK)
285  {
286  ++m_uiMarkCount;
287  m_uiMarkTimer = 15000;
288 
289  // limit the mark counter to 6
290  if (m_uiMarkCount == MAX_HYDROSS_MARKS)
291  m_uiMarkCount = MAX_HYDROSS_MARKS - 1;
292  }
293  }
294  else
295  m_uiMarkTimer -= uiDiff;
296 
297  if (m_uiEnrageTimer)
298  {
299  if (m_uiEnrageTimer <= uiDiff)
300  {
301  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
302  m_uiEnrageTimer = 0;
303  }
304  else
305  m_uiEnrageTimer -= uiDiff;
306  }
307 
308  DoMeleeAttackIfReady();
309  }
