211  void UpdateAI(const uint32 uiDiff) override
212  {
213  if (!m_creature->SelectHostileTarget())
214  {
215  if (GetCombatScriptStatus())
216  EnterEvadeMode();
217  return;
218  }
219 
220  if (m_uiAssassinsTimer)
221  {
222  if (m_uiAssassinsTimer <= uiDiff)
223  {
224  SpawnAssassin();
225  m_uiAssassinsTimer = 0;
226  }
227  else
228  m_uiAssassinsTimer -= uiDiff;
229  }
230 
231  if (m_bInBlade)
232  {
233  if (m_uiWaitTimer)
234  {
235  if (m_uiWaitTimer <= uiDiff)
236  {
237  if (m_uiTargetNum == 0)
238  {
239  // stop bladedance
240  m_bInBlade = false;
241  SetCombatScriptStatus(false);
242  SetCombatMovement(true, true);
243  SetMeleeEnabled(true);
244  m_uiWaitTimer = 0;
245  if (!m_bIsRegularMode)
246  m_uiChargeTimer = 500;
247  }
248  else
249  {
250  // move in bladedance
251  SpellCastResult result = SPELL_NOT_FOUND;
252  do
253  {
254  uint32 index = urand(0, m_bladeDanceTargetGuids.size() - 1);
255  if (Unit* target = m_creature->GetMap()->GetCreature(m_bladeDanceTargetGuids[index]))
256  // Until neutral target type TARGET_UNIT stops rolling for hit, need to force it
257  // should not send SMSG_SPELL_START but triggered spell cast bypasses checks which are necessary
258  result = m_creature->CastSpell(target, SPELL_BLADE_DANCE_CHARGE, TRIGGERED_IGNORE_HIT_CALCULATION);
259  m_bladeDanceTargetGuids.erase(m_bladeDanceTargetGuids.begin() + index);
260  --m_uiTargetNum;
261  } while (result != SPELL_CAST_OK && m_uiTargetNum > 0);
262  if (result != SPELL_CAST_OK)
263  {
264  m_uiWaitTimer = 1;
265  m_uiTargetNum = 0;
266  }
267  else
268  m_uiWaitTimer = 0;
269  }
270  }
271  else
272  m_uiWaitTimer -= uiDiff;
273  }
274  }
275  else // !m_bInBlade
276  {
277  if (m_uiBladeDanceTimer < uiDiff)
278  {
279  m_uiWaitTimer = 1;
280  m_bInBlade = true;
281  SetCombatScriptStatus(true);
282  SetCombatMovement(false);
283  SetMeleeEnabled(false);
284  m_uiBladeDanceTimer = 30000;
285  m_bladeDanceTargetGuids.clear();
286  m_creature->CastSpell(nullptr, SPELL_BLADE_DANCE_TARGETING, TRIGGERED_NONE);
287  m_uiTargetNum = std::min(m_bladeDanceTargetGuids.size(), size_t(TARGET_NUM));
288  return;
289  }
290  m_uiBladeDanceTimer -= uiDiff;
291 
292  if (m_uiChargeTimer)
293  {
294  if (m_uiChargeTimer <= uiDiff)
295  {
296  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
297  DoCastSpellIfCan(pTarget, SPELL_CHARGE_H);
298 
299  m_uiChargeTimer = 0;
300  }
301  else
302  m_uiChargeTimer -= uiDiff;
303  }
304 
305  if (m_uiSummonAssistantTimer < uiDiff)
306  {
307  switch (m_uiSummoned)
308  {
309  case 0: m_creature->SummonCreature(NPC_HEARTHEN_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0.0f, TEMPSPAWN_TIMED_OOC_DESPAWN, 20000); break;
310  case 1: m_creature->SummonCreature(NPC_REAVER_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0.0f, TEMPSPAWN_TIMED_OOC_DESPAWN, 20000); break;
311  case 2: m_creature->SummonCreature(NPC_SHARPSHOOTER_GUARD, AddsEntrance[0], AddsEntrance[1], AddsEntrance[2], 0.0f, TEMPSPAWN_TIMED_OOC_DESPAWN, 20000); break;
312  }
313 
314  m_uiSummoned++;
315 
316  if (m_uiSummoned == 3)
317  m_uiSummoned = 0;
318 
319  m_uiSummonAssistantTimer = 20000;
320  }
321  else
322  m_uiSummonAssistantTimer -= uiDiff;
323 
324  DoMeleeAttackIfReady();
325  }
326  }
