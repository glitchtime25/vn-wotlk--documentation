185  // Wrapper to handle the phase transform
186  void DoReuniteSpirits()
187  {
188  uint32 spellId = 0;
189 
190  // Each health level has it's own spell - but they all do the same thing
191  switch (m_phaseCounter)
192  {
193  case 3: spellId = SPELL_TRANSFORM_TO_LYNX_75; break;
194  case 2: spellId = SPELL_TRANSFORM_TO_LYNX_50; break;
195  case 1: spellId = SPELL_TRANSFORM_TO_LYNX_25; break;
196  }
197 
198  if (DoCastSpellIfCan(nullptr, spellId) == CAST_OK)
199  {
200  DoScriptText(SAY_MERGE, m_creature);
201  // Update stats back to the original Halazzi
202  m_creature->UpdateEntry(NPC_HALAZZI);
203 
204  // Despawn the Lynx
205  if (Creature* lynx = m_creature->GetMap()->GetCreature(m_spiritLynxGuid))
206  lynx->ForcedDespawn();
207 
208  // Set the proper health level - workaround for missing server side spell 43538
209  m_creature->SetHealth(m_creature->GetMaxHealth() / 4 * m_phaseCounter);
210  --m_phaseCounter;
211 
212  m_phase = m_phaseCounter > 0 ? PHASE_SINGLE : PHASE_FINAL;
213  HandlePhaseTransition();
214  }
215  }
216 
217  void HandlePhaseTransition()
218  {
219  switch (m_phase)
220  {
221  case PHASE_TOTEM:
222  DisableCombatAction(HALAZZI_ACTION_FRENZY);
223  DisableCombatAction(HALAZZI_ACTION_SABER_LASH);
224  case PHASE_FINAL:
225  ResetCombatAction(HALAZZI_ACTION_SHOCK, 10000);
226  ResetCombatAction(HALAZZI_ACTION_TOTEM, 12000);
227  if (m_phase == PHASE_TOTEM)
228  break;
229  SetDeathPrevention(false);
230  //[[fallthrough]]
231  case PHASE_SINGLE:
232  ResetCombatAction(HALAZZI_ACTION_FRENZY, 16000);
233  ResetCombatAction(HALAZZI_ACTION_SABER_LASH, 20000);
234  if (m_phase == PHASE_FINAL)
235  break;
236  DisableCombatAction(HALAZZI_ACTION_SHOCK);
237  DisableCombatAction(HALAZZI_ACTION_TOTEM);
238  break;
239  }
240  SetActionReadyStatus(HALAZZI_ACTION_TRANSFORM, true);
241  }

243  void ExecuteAction(uint32 action) override
244  {
245  switch (action)
246  {
247  case HALAZZI_ACTION_BERSERK:
248  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
249  {
250  DoScriptText(SAY_BERSERK, m_creature);
251  DisableCombatAction(action);
252  }
253  return;
254  case HALAZZI_ACTION_TRANSFORM:
255  if (m_phase != PHASE_TOTEM)
256  {
257  // Split boss at 75%, 50% and 25%
258  if (m_creature->GetHealthPercent() <= float(25 * m_phaseCounter))
259  {
260  if (DoCastSpellIfCan(nullptr, SPELL_TRANSFIGURE_TO_TROLL) == CAST_OK)
261  {
262  DoScriptText(SAY_SPLIT, m_creature);
263  SetActionReadyStatus(HALAZZI_ACTION_TRANSFORM, false);
264  }
265  }
266  }
267  else if (m_creature->GetHealthPercent() < 20.0f)
268  DoReuniteSpirits();
269  return;
270  case HALAZZI_ACTION_FRENZY:
271  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
272  ResetCombatAction(action, 16000);
273  return;
274  case HALAZZI_ACTION_SABER_LASH:
275  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SABER_LASH) == CAST_OK)
276  {
277  DoScriptText(urand(0, 1) ? SAY_SABERLASH1 : SAY_SABERLASH2, m_creature);
278  ResetCombatAction(action, 20000);
279  }
280  return;
281  case HALAZZI_ACTION_TOTEM:
282  if (DoCastSpellIfCan(nullptr, SPELL_LIGHTNING_TOTEM) == CAST_OK)
283  ResetCombatAction(action, 20000);
284  return;
285  case HALAZZI_ACTION_SHOCK:
286  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
287  if (DoCastSpellIfCan(target, urand(0, 1) ? SPELL_EARTHSHOCK : SPELL_FLAMESHOCK) == CAST_OK)
288  ResetCombatAction(action, urand(10000, 14000));
289  return;
290  }
291  }
