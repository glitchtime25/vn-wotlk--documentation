157  void ExecuteAction(uint32 action) override
158  {
159  switch (action)
160  {
161  case VEXALLUS_ACTION_CHAIN_LIGHTNING:
162  {
163  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_CHAIN_LIGHTNING, SELECT_FLAG_IN_LOS | SELECT_FLAG_PLAYER))
164  DoCastSpellIfCan(target, m_isRegularMode ? SPELL_CHAIN_LIGHTNING : SPELL_CHAIN_LIGHTNING_H);
165 
166  ResetCombatAction(action, GetSubsequentActionTimer(action));
167  return;
168  }
169  case VEXALLUS_ACTION_SHOCK:
170  {
171  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_ARCANE_SHOCK, SELECT_FLAG_IN_LOS | SELECT_FLAG_PLAYER))
172  DoCastSpellIfCan(target, m_isRegularMode ? SPELL_ARCANE_SHOCK : SPELL_ARCANE_SHOCK_H);
173 
174  ResetCombatAction(action, GetSubsequentActionTimer(action));
175  return;
176  }
177  case VEXALLUS_SUMMON_PURE_ENERGY:
178  {
179  // used for check, when Vexallus cast adds 85%, 70%, 55%, 40%, 25%
180  if (m_creature->GetHealthPercent() <= m_intervalHealthAmount)
181  {
182  DoScriptText(SAY_ENERGY, m_creature);
183  DoScriptText(EMOTE_DISCHARGE_ENERGY, m_creature);
184  m_intervalHealthAmount -= 15.0f;
185  if (m_intervalHealthAmount == 10.f)
186  SetActionReadyStatus(action, false);
187 
188  if (m_isRegularMode)
189  DoCastSpellIfCan(nullptr, SPELL_SUMMON_PURE_ENERGY);
190  else
191  {
192  DoCastSpellIfCan(nullptr, SPELL_SUMMON_PURE_ENERGY1_H, CAST_TRIGGERED);
193  DoCastSpellIfCan(nullptr, SPELL_SUMMON_PURE_ENERGY2_H, CAST_TRIGGERED);
194  }
195  }
196  return;
197  }
198  case VEXALLUS_OVERLOAD:
199  if (m_creature->GetHealthPercent() > 20.f)
200  return;
201  DoCastSpellIfCan(nullptr, SPELL_OVERLOAD);
202  SetActionReadyStatus(action, false);
203  DoScriptText(SAY_OVERLOAD, m_creature);
204  DoScriptText(EMOTE_OVERLOAD, m_creature);
205  break;
206  }
207  }
