142 void instance_shadow_labyrinth::OnCreatureDeath(Creature* pCreature)
143 {
144  // unbanish Hellmaw when all Cabal Ritualists are dead
145  if (pCreature->GetEntry() == NPC_CABAL_RITUALIST)
146  {
147  m_sRitualistsAliveGUIDSet.erase(pCreature->GetObjectGuid());
148  // TODO: Make containment beam collapse when a group of ritualists dies
149 
150  if (m_sRitualistsAliveGUIDSet.empty())
151  {
152  if (Creature* hellmaw = GetSingleCreatureFromStorage(NPC_HELLMAW))
153  hellmaw->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, hellmaw, hellmaw);
154 
155  GuidVector containmentVector;

157  for (ObjectGuid& guid : containmentVector)
158  {
159  if (Creature* pBeam = pCreature->GetMap()->GetCreature(guid))
160  pBeam->ForcedDespawn();
161  }
162  }

195  go_screaming_hall_door(GameObject* go) : GameObjectAI(go), m_doorCheckNearbyPlayersTimer(1000), m_doorOpen(false)

201  uint32 m_doorCheckNearbyPlayersTimer;
202  bool m_doorOpen;
203 
204  void UpdateAI(const uint32 diff) override
205  {
206  if (m_doorOpen)
207  return;
208 
209  if (m_doorCheckNearbyPlayersTimer <= diff)
210  {
211  // If player is in 35y range of door, open it if Vorpil boss is done
212  if (m_go->GetInstanceData()->GetData(TYPE_VORPIL) == DONE)
213  {
214  m_go->GetMap()->ExecuteDistWorker(m_go, 35.0f, [&](Player * player)
215  {
216  if (m_doorOpen)
217  return;
218  m_go->Use(player);
219  m_doorOpen = true;
220 
221  if (Creature* pMurmur = m_pInstance->GetSingleCreatureFromStorage(NPC_MURMUR))
222  pMurmur->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, pMurmur, pMurmur);
223  });
224  }
225  m_doorCheckNearbyPlayersTimer = 1000;
226  }
227  else
228  m_doorCheckNearbyPlayersTimer -= diff;
229  }
230 };
