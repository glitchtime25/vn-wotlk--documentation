219 void WorldSession::HandleLfgPlayerLockInfoRequestOpcode(WorldPacket& recv_data)
220 {
221  DEBUG_LOG("CMSG_LFG_GET_PLAYER_INFO %s", GetPlayer()->GetName());
222 
223  // Get Random dungeons that can be done at a certain level and expansion
224  uint8 level = GetPlayer()->GetLevel();
225  LfgDungeonSet const& randomDungeons = sLFGMgr.GetRandomAndSeasonalDungeons(level, GetPlayer()->GetSession()->GetExpansion());
226 
227  // Get player locked Dungeons
228  LfgLockMap const& lock = sLFGMgr.GetLockedDungeons(GetPlayer());
229  uint32 rsize = uint32(randomDungeons.size());
230  uint32 lsize = uint32(lock.size());
231 
232  DEBUG_LOG("SMSG_LFG_PLAYER_INFO %s", GetPlayer()->GetName());
233  WorldPacket data(SMSG_LFG_PLAYER_INFO, 1 + rsize * (4 + 1 + 4 + 4 + 4 + 4 + 1 + 4 + 4 + 4) + 4 + lsize * (1 + 4 + 4 + 4 + 4 + 1 + 4 + 4 + 4));
234 
235  data << uint8(randomDungeons.size()); // Random Dungeon count
236  for (auto itr = randomDungeons.begin(); itr != randomDungeons.end(); ++itr)
237  {
238  data << uint32(*itr); // Dungeon Entry (id + type)
239  LfgReward const* reward = sLFGMgr.GetRandomDungeonReward(*itr, level);
240  Quest const* quest = nullptr;
241  bool done = false;
242  if (reward)
243  {
244  quest = sObjectMgr.GetQuestTemplate(reward->firstQuest);
245  if (quest)
246  {
247  done = !GetPlayer()->CanRewardQuest(quest, false);
248  if (done)
249  quest = sObjectMgr.GetQuestTemplate(reward->otherQuest);
250  }
251  }
252 
253  if (quest)
254  {
255  data << uint8(done);
256  data << uint32(quest->GetRewOrReqMoney());
257  data << uint32(quest->GetXPReward(GetPlayer()));
258  data << uint32(0); // money variance per missing member when queueing - not actually used back then
259  data << uint32(0); // experience variance per missing member when queueing - not actually used back then
260  data << uint8(quest->GetRewItemsCount());
261  if (quest->GetRewItemsCount())
262  {
263  for (uint8 i = 0; i < QUEST_REWARDS_COUNT; ++i)
264  if (uint32 itemId = quest->RewItemId[i])
265  {
266  ItemPrototype const* item = sObjectMgr.GetItemPrototype(itemId);
267  data << uint32(itemId);
268  data << uint32(item ? item->DisplayInfoID : 0);
269  data << uint32(quest->RewItemCount[i]);
270  }
271  }
272  }
273  else
274  {
275  data << uint8(0);
276  data << uint32(0);
277  data << uint32(0);
278  data << uint32(0);
279  data << uint32(0);
280  data << uint8(0);
281  }
282  }
283  BuildPlayerLockDungeonBlock(data, lock);
284  SendPacket(data);
285 }

337 WorldPacket WorldSession::BuildLfgUpdate(LfgUpdateData const& updateData, bool isGroup)
338 {
339  bool join = false;
340  bool queued = false;
341  uint8 size = uint8(updateData.dungeons.size());
342 
343  switch (updateData.updateType)
344  {
345  case LFG_UPDATETYPE_JOIN_QUEUE:
346  if (isGroup)
347  {
348  queued = false;
349  join = updateData.state == LFG_STATE_RAIDBROWSER;
350  }
351  else
352  {
353  queued = true;
354  }
355  break;
356  case LFG_UPDATETYPE_ADDED_TO_QUEUE: // Rolecheck Success
357  queued = updateData.state != LFG_STATE_RAIDBROWSER;
358  [[fallthrough]];
359  case LFG_UPDATETYPE_PROPOSAL_BEGIN:
360  join = true;
361  break;
362  case LFG_UPDATETYPE_UPDATE_STATUS:
363  join = updateData.state != LFG_STATE_ROLECHECK && updateData.state != LFG_STATE_NONE;
364  queued = updateData.state == LFG_STATE_QUEUED;
365  break;
366  default:
367  break;
368  }
369 
370  size_t packetSize = 0;
371  if (isGroup)
372  packetSize = 1 + 1 + (size > 0 ? 1 : 0) * (1 + 1 + 1 + 1 + 1 + size * 4 + updateData.comment.length());
373  else
374  packetSize = 1 + 1 + (size > 0 ? 1 : 0) * (1 + 1 + 1 + 1 + size * 4 + updateData.comment.length());
375 
376  WorldPacket data(isGroup ? SMSG_LFG_UPDATE_PARTY : SMSG_LFG_UPDATE_PLAYER, packetSize);
377  data << uint8(updateData.updateType); // Lfg Update type
378  data << uint8(size > 0); // Is joined in LFG
379  if (size)
380  {
381  if (isGroup)
382  data << uint8(join);
383 
384  data << uint8(queued); // Join the queue
385  data << uint8(0); // NoPartialClear
386  data << uint8(0); // Achievements
387 
388  if (isGroup)
389  {
390  for (uint32 i = 0; i < 3; ++i)
391  data << uint8(updateData.m_roles[i]); // Needs
392  }
393 
394  data << uint8(size);
395  for (auto itr = updateData.dungeons.begin(); itr != updateData.dungeons.end(); ++itr)
396  data << uint32(*itr);
397  data << updateData.comment;
398  }
399 
400  return data;
401 }
