100 CombatManeuverReturns PlayerbotWarlockAI::DoFirstCombatManeuver(Unit* pTarget)
101 {
102  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
103  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
104  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
105  {
106  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
107  {
108  return RETURN_NO_ACTION_OK; // wait it out
109  }
110  else
111  {
112  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
113  }
114  }
115 
116  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
117  {
118  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
119  return RETURN_NO_ACTION_OK; // wait it out
120  else
121  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
122  }
123 
124  switch (m_ai.GetScenarioType())
125  {
126  case PlayerbotAI::SCENARIO_PVP_DUEL:
127  case PlayerbotAI::SCENARIO_PVP_BG:
128  case PlayerbotAI::SCENARIO_PVP_ARENA:
129  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
130  return DoFirstCombatManeuverPVP(pTarget);
131  case PlayerbotAI::SCENARIO_PVE:
132  case PlayerbotAI::SCENARIO_PVE_ELITE:
133  case PlayerbotAI::SCENARIO_PVE_RAID:
134  default:
135  return DoFirstCombatManeuverPVE(pTarget);
136  break;
137  }
138 
139  return RETURN_NO_ACTION_ERROR;
140 }

172 CombatManeuverReturns PlayerbotWarlockAI::DoNextCombatManeuverPVE(Unit* pTarget)
173 {
174  //Unit* pVictim = pTarget->GetVictim();
175  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
176  Pet* pet = m_bot.GetPet();
177  uint32 spec = m_bot.GetSpec();
178  uint8 shardCount = m_bot.GetItemCount(SOUL_SHARD, false, nullptr);
179 
180  //If we have UA it will replace immolate in our rotation
181  uint32 FIRE = (UNSTABLE_AFFLICTION > 0 ? UNSTABLE_AFFLICTION : IMMOLATE);
182 
183  // Voidwalker is near death - sacrifice it for a shield
184  if (pet && pet->GetEntry() == DEMON_VOIDWALKER && SACRIFICE && !m_bot.HasAura(SACRIFICE) && pet->GetHealthPercent() < 10)
185  m_ai.CastPetSpell(SACRIFICE);
186 
187  // Use healthstone
188  if (m_ai.GetHealthPercent() < 30)
189  {
190  Item* healthStone = m_ai.FindConsumable(HEALTHSTONE_DISPLAYID);
191  if (healthStone)
192  m_ai.UseItem(healthStone);
193  }
194 
195  // Voidwalker sacrifice gives shield - but you lose the pet (and it's DPS/tank) - use only as last resort for your own health!
196  if (m_ai.GetHealthPercent() < 20 && pet && pet->GetEntry() == DEMON_VOIDWALKER && SACRIFICE && !m_bot.HasAura(SACRIFICE))
197  m_ai.CastPetSpell(SACRIFICE);
198 
199  if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED && !meleeReach)
200  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
201  // if in melee range OR can't shoot OR have no ranged (wand) equipped
202  else if (m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE && (meleeReach || SHOOT == 0 || !m_bot.GetWeaponForAttack(RANGED_ATTACK, true, true)))
203  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
204 
205  //Used to determine if this bot is highest on threat
206  Unit* newTarget = m_ai.FindAttacker((PlayerbotAI::ATTACKERINFOTYPE)(PlayerbotAI::AIT_VICTIMSELF | PlayerbotAI::AIT_HIGHESTTHREAT), &m_bot);
207  if (newTarget && !PlayerbotAI::IsNeutralized(newTarget)) // TODO: && party has a tank
208  {
209  if (SOULSHATTER > 0 && shardCount > 0 && m_bot.IsSpellReady(SOULSHATTER))
210  if (CastSpell(SOULSHATTER, &m_bot))
211  return RETURN_CONTINUE;
212 
213  // Have threat, can't quickly lower it. 3 options remain: Stop attacking, lowlevel damage (wand), keep on keeping on.
214  if (newTarget->GetHealthPercent() > 25)
215  {
216  // If elite, do nothing and pray tank gets aggro off you
217  // TODO: Is there an IsElite function? If so, find it and insert.
218  //if (newTarget->IsElite())
219  // return;
220 
221  // Not an elite. You could insert FEAR here but in any PvE situation that's 90-95% likely
222  // to worsen the situation for the group. ... So please don't.
223  return CastSpell(SHOOT, pTarget);
224  }
225  }
226 
227  // Create soul shard
228  uint8 freeSpace = m_ai.GetFreeBagSpace();
229  if (DRAIN_SOUL && pTarget->GetHealth() < pTarget->GetMaxHealth() * 0.20 && m_ai.In_Reach(pTarget, DRAIN_SOUL) &&
230  !pTarget->HasAura(DRAIN_SOUL) && (shardCount < MAX_SHARD_COUNT && freeSpace > 0) && CastSpell(DRAIN_SOUL, pTarget))
231  {
232  m_ai.SetIgnoreUpdateTime(15);
233  return RETURN_CONTINUE;
234  }
235 
236 
237  // Damage Spells
238  switch (spec)
239  {
240  case WARLOCK_SPEC_AFFLICTION:
241  if (CURSE_OF_AGONY && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CURSE_OF_AGONY) && !pTarget->HasAura(CURSE_OF_AGONY) && CastSpell(CURSE_OF_AGONY, pTarget))
242  return RETURN_CONTINUE;
243  if (CORRUPTION && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CORRUPTION) && !pTarget->HasAura(CORRUPTION) && CastSpell(CORRUPTION, pTarget))
244  return RETURN_CONTINUE;
245  if (FIRE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIRE) && !pTarget->HasAura(FIRE) && CastSpell(FIRE, pTarget))
246  return RETURN_CONTINUE;
247  if (HAUNT && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, HAUNT) && m_bot.IsSpellReady(HAUNT) && CastSpell(HAUNT, pTarget))
248  return RETURN_CONTINUE;
249  if (SHADOW_BOLT && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_BOLT) && CastSpell(SHADOW_BOLT, pTarget))
250  return RETURN_CONTINUE;
251 
252  return RETURN_NO_ACTION_OK;
253 
254  case WARLOCK_SPEC_DEMONOLOGY:
255  if (pet && DEMONIC_EMPOWERMENT && m_bot.IsSpellReady(DEMONIC_EMPOWERMENT) && CastSpell(DEMONIC_EMPOWERMENT))
256  return RETURN_CONTINUE;
257  if (CURSE_OF_AGONY && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CURSE_OF_AGONY) && !pTarget->HasAura(CURSE_OF_AGONY) && CastSpell(CURSE_OF_AGONY, pTarget))
258  return RETURN_CONTINUE;
259  if (CORRUPTION && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CORRUPTION) && !pTarget->HasAura(CORRUPTION) && CastSpell(CORRUPTION, pTarget))
260  return RETURN_CONTINUE;
261  if (FIRE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIRE) && !pTarget->HasAura(FIRE) && CastSpell(FIRE, pTarget))
262  return RETURN_CONTINUE;
263  if (INCINERATE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, INCINERATE) && pTarget->HasAura(FIRE) && CastSpell(INCINERATE, pTarget))
264  return RETURN_CONTINUE;
265  if (SHADOW_BOLT && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_BOLT) && CastSpell(SHADOW_BOLT, pTarget))
266  return RETURN_CONTINUE;
267 
268  return RETURN_NO_ACTION_OK;
269 
270  case WARLOCK_SPEC_DESTRUCTION:
271  if (CURSE_OF_AGONY && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CURSE_OF_AGONY) && !pTarget->HasAura(CURSE_OF_AGONY) && CastSpell(CURSE_OF_AGONY, pTarget))
272  return RETURN_CONTINUE;
273  if (CORRUPTION && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, CORRUPTION) && !pTarget->HasAura(CORRUPTION) && CastSpell(CORRUPTION, pTarget))
274  return RETURN_CONTINUE;
275  if (FIRE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIRE) && !pTarget->HasAura(FIRE) && CastSpell(FIRE, pTarget))
276  return RETURN_CONTINUE;
277  if (CONFLAGRATE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, CONFLAGRATE) && pTarget->HasAura(FIRE) && m_bot.IsSpellReady(CONFLAGRATE) && CastSpell(CONFLAGRATE, pTarget))
278  return RETURN_CONTINUE;
279  if (CHAOS_BOLT && m_ai.In_Reach(pTarget, CHAOS_BOLT) && m_bot.IsSpellReady(CHAOS_BOLT) && CastSpell(CHAOS_BOLT, pTarget))
280  return RETURN_CONTINUE;
281  if (INCINERATE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, INCINERATE) && pTarget->HasAura(FIRE) && CastSpell(INCINERATE, pTarget))
282  return RETURN_CONTINUE;
283  if (SHADOW_BOLT && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_BOLT) && CastSpell(SHADOW_BOLT, pTarget))
284  return RETURN_CONTINUE;
285 
286  return RETURN_NO_ACTION_OK;
287  }
288 
289  // No spec due to low level OR no spell found yet
290  if (CORRUPTION && m_ai.In_Reach(pTarget, CORRUPTION) && !pTarget->HasAura(CORRUPTION) && CastSpell(CORRUPTION, pTarget))
291  return RETURN_CONTINUE;
292  if (FIRE && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_FIRE) && m_ai.In_Reach(pTarget, FIRE) && !pTarget->HasAura(FIRE) && CastSpell(FIRE, pTarget))
293  return RETURN_CONTINUE;
294  if (SHADOW_BOLT && !PlayerbotAI::IsImmuneToSchool(pTarget, SPELL_SCHOOL_MASK_SHADOW) && m_ai.In_Reach(pTarget, SHADOW_BOLT))
295  return CastSpell(SHADOW_BOLT, pTarget);
296 
297  return RETURN_NO_ACTION_OK;
298 } // end DoNextCombatManeuver

310 void PlayerbotWarlockAI::CheckDemon()
311 {
312  uint32 spec = m_bot.GetSpec();
313  uint8 shardCount = m_bot.GetItemCount(SOUL_SHARD, false, nullptr);
314  Pet* pet = m_bot.GetPet();
315 
316  //Assign demon of choice
317  if (spec == WARLOCK_SPEC_AFFLICTION)
318  m_demonOfChoice = DEMON_FELHUNTER;
319  else if (spec == WARLOCK_SPEC_DEMONOLOGY)
320  m_demonOfChoice = (DEMON_FELGUARD > 0 ? DEMON_FELGUARD : DEMON_SUCCUBUS);
321  else if (spec == WARLOCK_SPEC_DESTRUCTION)
322  m_demonOfChoice = DEMON_IMP;
323 
324  // Summon demon
325  if (!pet || m_isTempImp || pet->GetEntry() != m_demonOfChoice)
326  {
327  uint32 summonSpellId;
328  if (m_demonOfChoice != DEMON_IMP && shardCount > 0)
329  {
330  switch (m_demonOfChoice)
331  {
332  case DEMON_FELGUARD:
333  summonSpellId = SUMMON_FELGUARD;
334  break;
335 
336  case DEMON_FELHUNTER:
337  summonSpellId = SUMMON_FELHUNTER;
338  break;
339 
340  case DEMON_SUCCUBUS:
341  summonSpellId = SUMMON_SUCCUBUS;
342  break;
343 
344  default:
345  summonSpellId = 0;
346  }
347 
348  if (m_ai.CastSpell(summonSpellId) == SPELL_CAST_OK)
349  {
350  //m_ai.TellMaster("Summoning favorite demon...");
351  m_isTempImp = false;
352  return;
353  }
354  }
355  else if (!pet && SUMMON_IMP && m_ai.CastSpell(SUMMON_IMP) == SPELL_CAST_OK)
356  {
357  if (m_demonOfChoice != DEMON_IMP)
358  m_isTempImp = true;
359 
360  //m_ai.TellMaster("Summoning Imp...");
361  return;
362  }
363  }
