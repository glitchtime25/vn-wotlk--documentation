22 bool WorldSession::CheckChatChannelNameAndPassword(std::string& name, std::string& pass)
23 {
24  // check name max length and truncate if needed
25  if (name.length() > 128)
26  utf8limit(name, 128);
27 
28  // strip invisible characters
29  stripLineInvisibleChars(name);
30 
31  // check if password is too long
32  if (pass.length() > 128)
33  {
34  WorldPacket data(SMSG_CHANNEL_NOTIFY, (1 + name.size() + 1));
35  data << uint8(CHAT_WRONG_PASSWORD_NOTICE) << name;
36  SendPacket(data);
37  return false;
38  }
39 
40  // skip remaining checks for higher sec level accounts
41  if (GetSecurity() > SEC_PLAYER)
42  return true;
43 
44  // check for presence of escape sequences
45  if (ChatHandler::HasEscapeSequences(name.c_str()))
46  {
47  sLog.outError("Player %s (GUID: %u) attempted to join a chat channel with name containing escape sequence: \"%s\"", GetPlayer()->GetName(),
48  GetPlayer()->GetGUIDLow(), name.c_str());
49 
50  WorldPacket data(SMSG_CHANNEL_NOTIFY, (1 + name.size() + 1));
51  data << uint8(CHAT_INVALID_NAME_NOTICE) << name;
52  SendPacket(data);
53  return false;
54  }
55  return true;
56 }

58 void WorldSession::HandleJoinChannelOpcode(WorldPacket& recvPacket)
59 {
60  DEBUG_LOG("WORLD: Received opcode %s (%u, 0x%X)", recvPacket.GetOpcodeName(), recvPacket.GetOpcode(), recvPacket.GetOpcode());
61 
62  uint32 channel_id;
63  uint8 unknown1, unknown2;
64  std::string channelname, pass;
65 
66  recvPacket >> channel_id >> unknown1 >> unknown2;
67  recvPacket >> channelname;
68 
69  if (channelname.empty())
70  return;
71 
72  recvPacket >> pass;
73 
74  if (!CheckChatChannelNameAndPassword(channelname, pass))
75  return;
76 
77  if (ChannelMgr* cMgr = channelMgr(_player->GetTeam()))
78  if (Channel* chn = cMgr->GetJoinChannel(channelname, channel_id))
79  chn->Join(_player, pass.c_str());
80 }
