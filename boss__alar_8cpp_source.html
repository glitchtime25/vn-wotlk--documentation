466  {

470  {
471  StartPhaseTransition();
472  SetActionReadyStatus(ALAR_PHASE_2, false);
473  break;
474  }
475  case ALAR_BERSERK:
476  {
477  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
478  ResetCombatAction(ALAR_BERSERK, uint32(10 * MINUTE * IN_MILLISECONDS));
479  break;
480  }
481  case ALAR_PLATFORM_MOVE:
482  {
483 #ifdef PRENERF_2_0_3

485 #else
486  if (!m_firstPlatform && urand(0, 1) == 0)
487 #endif
488  DoFlameQuills();
489  else
490  DoPlatformMove(true);
491  break;
492  }
493  case ALAR_BUFFET:
494  {
495  uint32 timer = 500;
496  // If victim exists we have a target in melee range
497  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
498  m_rangeCheckState = -1;
499  // Spam Waterbolt spell when not tanked
500  else
501  {
502  ++m_rangeCheckState;
503  if (m_rangeCheckState > 1)
504  if (DoCastSpellIfCan(nullptr, SPELL_FLAME_BUFFET) == CAST_OK)
505  timer = 2500;
506  }
507  ResetCombatAction(action, timer);
508  break;
509  }
510  case ALAR_DIVE_BOMB:
511  {
512  ResetCombatAction(ALAR_CHARGE, urand(25000, 30000));
513  SetCombatMovement(false);
514  SetMeleeEnabled(false);
515  SetCombatScriptStatus(true);
516  m_creature->SetTarget(nullptr);
517  m_creature->SetHover(true);

520  SetDeathPrevention(true);
521  ResetCombatAction(action, urand(50000, 60000));
522  break;
523  }
524  case ALAR_FLAME_PATCH:
525  {
526  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), SELECT_FLAG_PLAYER))
527  if (DoCastSpellIfCan(target, SPELL_FLAME_RING) == CAST_OK)
528  ResetCombatAction(action, urand(12000, 17000));
529  break;
530  }
531  case ALAR_MELT_ARMOR:
532  {
533  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MELT_ARMOR) == CAST_OK)
534  ResetCombatAction(action, 60000);

536  }
537  case ALAR_CHARGE:

541  ResetCombatAction(action, urand(25000, 30000));
542  break;
543  }
544  }
