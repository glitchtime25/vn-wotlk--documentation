220 bool QuestRewarded_war_effort(Player* player, GameObject* go, Quest const* quest)
221 {
222  if (quest->GetQuestId() == QUEST_BANG_A_GONG)
223  {
224  if (sWorldState.GetAqPhase() == PHASE_3_GONG_TIME)
225  {
226  SendBgNeutralToPlayer(SAY_BANG_GONG, player);
227  SendBgNeutralToPlayer(SAY_QIRAJI_BREAK, player);
228  if (InstanceData* data = go->GetInstanceData())
229  data->SetData(TYPE_GONG_TIME, 0);
230  sWorldState.HandleWarEffortPhaseTransition(PHASE_4_10_HOUR_WAR);
231  }
232  }
233  return true;
234 }
