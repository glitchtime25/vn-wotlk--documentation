185 bool GossipHello_npc_barnes(Player* pPlayer, Creature* pCreature)
186 {
187  if (ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData())
188  {
189  // Check if opera event is not yet in progress
190  if (pInstance->GetData(TYPE_OPERA) == IN_PROGRESS)
191  return true;
192 
193  // Check for death of Moroes
194  if (pInstance->GetData(TYPE_MOROES) != DONE)
195  pPlayer->SEND_GOSSIP_MENU(TEXT_ID_MOROES_ALIVE, pCreature->GetObjectGuid());
196  else
197  {
198  switch (pInstance->GetData(TYPE_OPERA))
199  {
200  case NOT_STARTED:
201  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, GOSSIP_ITEM_OPERA_1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
202  // for GMs we add the possibility to change the event
203  if (pPlayer->IsGameMaster())
204  {
205  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_OZ", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
206  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_HOOD", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
207  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_RAJ", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 5);
208  }
209  pPlayer->SEND_GOSSIP_MENU(TEXT_ID_OPERA_1, pCreature->GetObjectGuid());
210  break;
211  case FAIL:
212  if (pPlayer->IsGameMaster())
213  {
214  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_OZ", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
215  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_HOOD", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 4);
216  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_DOT, "[GM] Change event to EVENT_RAJ", GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 5);
217  }
218  int32 gossipItemText;
219  uint32 gossipMenuText;
220  switch (pInstance->GetData(TYPE_OPERA_PERFORMANCE))
221  {
222  case OPERA_EVENT_WIZARD_OZ: gossipItemText = GOSSIP_ITEM_OPERA_OZ_WIPE; gossipMenuText = TEXT_ID_OPERA_OZ_WIPE; break;
223  case OPERA_EVENT_RED_RIDING_HOOD: gossipItemText = GOSSIP_ITEM_WOLF_WIPE; gossipMenuText = TEXT_ID_OPERA_WOLF_WIPE; break;
224  default:
225  case OPERA_EVENT_ROMULO_AND_JUL: gossipItemText = GOSSIP_ITEM_JUL_WIPE; gossipMenuText = TEXT_ID_OPERA_JUL_WIPE; break;
226  }
227  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, gossipItemText, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
228  pPlayer->SEND_GOSSIP_MENU(gossipMenuText, pCreature->GetObjectGuid());
229  break;
230  case DONE:
231  pPlayer->SEND_GOSSIP_MENU(TEXT_ID_OPERA_DONE, pCreature->GetObjectGuid());
232  break;
233  }
234 
235  return true;
236  }
237  }
238 
239  return true;
240 }

242 bool GossipSelect_npc_barnes(Player* pPlayer, Creature* pCreature, uint32 /*uiSender*/, uint32 uiAction)
243 {
244  switch (uiAction)
245  {
246  case GOSSIP_ACTION_INFO_DEF+1:

250  case GOSSIP_ACTION_INFO_DEF+2:

254  if (npc_barnesAI* pBarnesAI = dynamic_cast<npc_barnesAI*>(pCreature->AI()))
255  pBarnesAI->Start(false, nullptr, nullptr, true);
256  break;
257  // GM gossip options
258  case GOSSIP_ACTION_INFO_DEF+3:
259  pPlayer->CLOSE_GOSSIP_MENU();
260  if (ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData())
261  {
262  pInstance->SetData(TYPE_OPERA_PERFORMANCE, OPERA_EVENT_WIZARD_OZ);
263  outstring_log("SD2: %s manually set Opera event to EVENT_OZ", pPlayer->GetGuidStr().c_str());
264  }
265  break;
266  case GOSSIP_ACTION_INFO_DEF+4:
267  pPlayer->CLOSE_GOSSIP_MENU();
268  if (ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData())
269  {
270  pInstance->SetData(TYPE_OPERA_PERFORMANCE, OPERA_EVENT_RED_RIDING_HOOD);
271  outstring_log("SD2: %s manually set Opera event to EVENT_HOOD", pPlayer->GetGuidStr().c_str());
272  }

274  case GOSSIP_ACTION_INFO_DEF+5:
275  pPlayer->CLOSE_GOSSIP_MENU();
276  if (ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData())
277  {
278  pInstance->SetData(TYPE_OPERA_PERFORMANCE, OPERA_EVENT_ROMULO_AND_JUL);
