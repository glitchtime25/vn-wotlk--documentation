108  void UpdateAI(const uint32 diff) override
109  {
110  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
111  return;
112 
113  if (m_isRegularMode)
114  {
115  // Damage Shields
116  if (m_reflectiveShieldTimer < diff)
117  {
118  if (m_reflectiveShield)
119  {
120  if (DoCastSpellIfCan(m_creature, SPELL_REFLECTIVE_DAMAGE_SHIELD) == CAST_OK)
121  {
122  m_reflectiveShieldTimer = 20000;
123  m_reflectiveShield = false;
124  }
125  }
126  else
127  {
128  if (DoCastSpellIfCan(m_creature, SPELL_REFLECTIVE_MAGIC_SHIELD) == CAST_OK)
129  {
130  m_reflectiveShieldTimer = 20000;
131  m_reflectiveShield = true;
132  }
133  }
134  }
135  else
136  m_reflectiveShieldTimer -= diff;
137  }
138  else
139  {
140  // Polarity Shift
141  if (m_polarityShiftTimer < diff)
142  {
143  if (DoCastSpellIfCan(m_creature, SPELL_POLARITY_SHIFT) == CAST_OK)
144  m_polarityShiftTimer = urand(27000, 34000);
145 
146  switch (urand(0, 3))
147  {
148  case 0: DoScriptText(SAY_ABILITY_USE_1, m_creature); break;
149  case 1: DoScriptText(SAY_ABILITY_USE_2, m_creature); break;
150  default: break;
151  }
152  }
153  else
154  m_polarityShiftTimer -= diff;
155  }
156 
157  // Berserk
158  if (m_netherChargeTimer < diff)
159  {
160  uint32 spellId;
161  switch (urand(0, 3))
162  {
163  case 0: spellId = SPELL_SUMMON_NETHER_CHARGE_NE; break;
164  case 1: spellId = SPELL_SUMMON_NETHER_CHARGE_NW; break;
165  case 2: spellId = SPELL_SUMMON_NETHER_CHARGE_SE; break;
166  default:
167  case 3: spellId = SPELL_SUMMON_NETHER_CHARGE_SW; break;
168  }
169  m_creature->CastSpell(m_creature, spellId, TRIGGERED_NONE);
170  m_netherChargeTimer = m_isRegularMode ? urand(2000, 15000) : urand(2000, 20000);
171 
172  switch (urand(0, 3))
173  {
174  case 0: DoScriptText(SAY_ABILITY_USE_1, m_creature); break;
175  case 1: DoScriptText(SAY_ABILITY_USE_2, m_creature); break;
176  default: break;
177  }
178  }
179  else
180  m_netherChargeTimer -= diff;
181 
182  // Head Crack
183  if (m_headCrackTimer < diff)
184  {
185  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_HEAD_CRACK) == CAST_OK)
186  m_headCrackTimer = urand(19500, 33500);
187  }
188  else
189  m_headCrackTimer -= diff;
190 
191  // Berserk
192  if (m_berserkTimer < diff)
193  {
194  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
195  m_berserkTimer = 300000;
196  }
197  else
198  m_berserkTimer -= diff;
199 
200  DoMeleeAttackIfReady();
201  }
