176  void HandleIntro()
177  {
178  uint32 timer = 0;
179  switch (m_uiIntroPhase)
180  {
181  case 0:
182  {
183  // Summon Lord Victor Nefarius in front of the Throne
184  m_creature->SummonCreature(NPC_LORD_VICTOR_NEFARIUS, aNefariusSpawnLoc[0], aNefariusSpawnLoc[1], aNefariusSpawnLoc[2], aNefariusSpawnLoc[3], TEMPSPAWN_TIMED_DESPAWN, 25000);
185 
186  bool bHasYelled = false;
187  CreatureList lTechniciansList;
188  // Search for the Blackwing Technicians tormeting Vaelastrasz to make them flee to the next room above the stairs
189  GetCreatureListWithEntryInGrid(lTechniciansList, m_creature, NPC_BLACKWING_TECHNICIAN, 40.0f);
190  for (CreatureList::const_iterator itr = lTechniciansList.begin(); itr != lTechniciansList.end(); ++itr)
191  {
192  // Ignore Blackwing Technicians on upper floors and dead ones
193  if (!((*itr)->IsAlive()) || (*itr)->GetPositionZ() > m_creature->GetPositionZ() + 1)
194  continue;
195 
196  // Each fleeing part and despawn is handled in DB, we only need to make them run
197  (*itr)->SetWalk(false);
198 
199  // The technicians will behave differently depending on they are on the right or left side of
200  // Vaelastrasz. We compare their X position to Vaelastrasz X position to sort them out
201  if ((*itr)->GetPositionX() > m_creature->GetPositionX())
202  {
203  // Left side
204  if (!bHasYelled)
205  {
206  DoScriptText(SAY_TECHNICIAN_RUN, (*itr));
207  bHasYelled = true;
208  }
209  (*itr)->GetMotionMaster()->MoveWaypoint(0);
210  }
211  else
212  // Right side
213  (*itr)->GetMotionMaster()->MoveWaypoint(1);
214  }
215  timer = 1000;
216  break;
217  }
218  case 1:
219  if (Creature* nefarius = m_creature->GetMap()->GetCreature(m_nefariusGuid))
220  {
221  nefarius->CastSpell(m_creature, SPELL_NEFARIUS_CORRUPTION, TRIGGERED_OLD_TRIGGERED);
222  DoScriptText(SAY_NEFARIUS_CORRUPT_1, nefarius);
223  }
224  timer = 14000;
225  break;
226  case 2:
227  if (Creature* nefarius = m_creature->GetMap()->GetCreature(m_nefariusGuid))
228  DoScriptText(SAY_NEFARIUS_CORRUPT_2, nefarius);
229 
230  timer = 2000;
231  break;
232  case 3:
233  if (Creature* nefarius = m_creature->GetMap()->GetCreature(m_nefariusGuid))
234  nefarius->CastSpell(m_creature, SPELL_RED_LIGHTNING, TRIGGERED_NONE);
235 
236  // Set npc flags now
237  m_creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
238  m_creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
239  m_creature->RemoveAurasDueToSpell(SPELL_NEFARIUS_CORRUPTION);
240  break;
241  }
242  ++m_uiIntroPhase;
243  if (timer)
244  ResetTimer(VAEL_INTRO, timer);
245  }

247  void HandleSpeech()
248  {
249  uint32 timer = 0;
250  switch (m_uiSpeechNum)
251  {
252  case 0:
253  // 16 seconds till next line
254  DoScriptText(SAY_LINE_2, m_creature);
255  timer = 16000;
256  ++m_uiSpeechNum;
257  break;
258  case 1:
259  // This one is actually 16 seconds but we only go to 10 seconds because he starts attacking after he says "I must fight this!"
260  DoScriptText(SAY_LINE_3, m_creature);
261  timer = 10000;
262  ++m_uiSpeechNum;
263  break;
264  case 2:
265  m_creature->SetFactionTemporary(FACTION_HOSTILE, TEMPFACTION_RESTORE_RESPAWN);
266  m_creature->SetInCombatWithZone();
267  AttackClosestEnemy();
268  break;
269  }
270  if (timer)
271  ResetTimer(VAEL_SPEECH, timer);
272  }

274  void ExecuteAction(uint32 action) override
275  {
276  switch (action)
277  {
278  case VAEL_LOW_HP_YELL:
279  {
280  if (m_creature->GetHealthPercent() < 15.0f)
281  {
282  DoScriptText(SAY_HALFLIFE, m_creature);
283  SetActionReadyStatus(action, false);
284  }
285  break;
286  }
287  case VAEL_BURNING_ADRENALINE_TANK:
288  {
289  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BURNING_ADRENALINE_TANK) == CAST_OK)
290  ResetCombatAction(action, 45000);
291  break;
292  }
293  case VAEL_BURNING_ADRENALINE_TARGET:
294  {
295  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_BURNING_ADRENALINE, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
296  if (DoCastSpellIfCan(target, SPELL_BURNING_ADRENALINE) == CAST_OK)
297  ResetCombatAction(action, 15000);
298  break;
299  }
300  case VAEL_FLAME_BREATH:
301  {
302  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FLAME_BREATH) == CAST_OK)
303  ResetCombatAction(action, urand(4000, 8000));
304  break;
305  }
306  case VAEL_FIRE_NOVA:
307  {
308  if (DoCastSpellIfCan(nullptr, SPELL_FIRE_NOVA) == CAST_OK)
309  ResetCombatAction(action, 5000);
310  break;
311  }
312  case VAEL_CLEAVE:
313  {
314  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
315  ResetCombatAction(action, 15000);
316  break;
317  }
318  case VAEL_TAIL_SWEEP:
319  {
320  if (DoCastSpellIfCan(nullptr, SPELL_TAIL_SWEEP) == CAST_OK)
321  ResetCombatAction(action, 20000);
322  break;
323  }
324  }
325  }
