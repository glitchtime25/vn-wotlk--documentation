98  {
99  switch (m_type)

101  case FIELD_NONE: return 0;
102  case FIELD_BOOL: // return sizeof(bool);
103  case FIELD_UI8: return sizeof(uint8);
104  case FIELD_UI16: return sizeof(uint16);
105  case FIELD_UI32: return sizeof(uint32);
106  case FIELD_UI64: return sizeof(uint64);
107  case FIELD_I8: return sizeof(int8);
108  case FIELD_I16: return sizeof(int16);
109  case FIELD_I32: return sizeof(int32);
110  case FIELD_I64: return sizeof(int64);
111  case FIELD_FLOAT: return sizeof(float);
112  case FIELD_DOUBLE: return sizeof(double);
113  case FIELD_STRING: return m_szStringData.length();
114 
115  default:
116  throw std::runtime_error("unrecognized type of SqlStmtFieldType obtained");
117  }
