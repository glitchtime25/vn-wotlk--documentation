163  void DoSummonPack(uint8 uiIndex)
164  {
165  for (const auto& i : asSummonInfo)
166  {
167  // This requires order of the array
168  if (i.uiPosition > uiIndex)
169  break;
170  if (i.uiPosition == uiIndex)
171  m_creature->SummonCreature(i.uiEntry, i.fX, i.fY, i.fZ, i.fO, TEMPSPAWN_DEAD_DESPAWN, 0);
172  }
173  }

175  void JustSummoned(Creature* pSummoned) override
176  {
177  switch (pSummoned->GetEntry())
178  {
179  case NPC_CAVERNDEEP_BURROWER:
180  case NPC_CAVERNDEEP_AMBUSHER:
181  {
182  if (GameObject* pDoor = m_pInstance->GetSingleGameObjectFromStorage(m_uiPhase > 20 ? GO_CAVE_IN_NORTH : GO_CAVE_IN_SOUTH))

191  // Movement of Grubbis and Add to be handled by DB waypoints
192  DoScriptText(SAY_GRUBBIS_SPAWN, pSummoned);
193  break;
194  }
