573 struct PolarityShift : public SpellScript
574 {
575  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx ) const override
576  {
577  if (effIdx == EFFECT_INDEX_0)
578  {
579  if (Unit* unitTarget = spell->GetUnitTarget())
580  {
581  unitTarget->RemoveAurasDueToSpell(SPELL_POSITIVE_CHARGE);
582  unitTarget->RemoveAurasDueToSpell(SPELL_NEGATIVE_CHARGE);
583 
584  uint64 scriptValue = spell->GetScriptValue();
585 
586  // 28059 : Positive Charge, 28084 : Negative Charge
587  switch (scriptValue)
588  {
589  case 0: // first target random
590  scriptValue = urand(0, 1) ? SPELL_POSITIVE_CHARGE : SPELL_NEGATIVE_CHARGE;
591  spell->SetScriptValue(scriptValue);
592  unitTarget->CastSpell(unitTarget, scriptValue, TRIGGERED_INSTANT_CAST);
593  break;
594  case SPELL_POSITIVE_CHARGE: // second target the other
595  spell->SetScriptValue(1);
596  unitTarget->CastSpell(unitTarget, SPELL_NEGATIVE_CHARGE, TRIGGERED_INSTANT_CAST);
597  break;
598  case SPELL_NEGATIVE_CHARGE:
599  spell->SetScriptValue(1);
600  unitTarget->CastSpell(unitTarget, SPELL_POSITIVE_CHARGE, TRIGGERED_INSTANT_CAST);
601  break;
602  default: // third and later random
603  unitTarget->CastSpell(unitTarget, urand(0, 1) ? SPELL_POSITIVE_CHARGE : SPELL_NEGATIVE_CHARGE, TRIGGERED_INSTANT_CAST);
604  break;
605  }
606  }
607  }
