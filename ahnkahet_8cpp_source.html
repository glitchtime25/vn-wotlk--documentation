44 void instance_ahnkahet::OnCreatureCreate(Creature* pCreature)

48  case NPC_ELDER_NADOX:

65  else if (pCreature->GetPositionZ() > 20.0f)
66  // Used for Jedoga visual
67  m_lJedogaEventControllersGuidList.push_back(pCreature->GetObjectGuid());
68  else if (pCreature->GetPositionZ() < -16.0f)
69  // Used for Jedoga sacrifice
70  m_jedogaSacrificeController = pCreature->GetObjectGuid();
71  break;
72  case NPC_TWISTED_VISAGE_1:
73  case NPC_TWISTED_VISAGE_2:
74  case NPC_TWISTED_VISAGE_3:
75  case NPC_TWISTED_VISAGE_4:
76  case NPC_TWISTED_VISAGE_5:
77  ++m_uiTwistedVisageCount;

107 void instance_ahnkahet::SetData(uint32 uiType, uint32 uiData)
108 {
109  debug_log("SD2: Instance Ahn'Kahet: SetData received for type %u with data %u", uiType, uiData);
110 
111  switch (uiType)
112  {
113  case TYPE_NADOX:
114  m_auiEncounter[uiType] = uiData;
115  if (uiData == IN_PROGRESS)
116  m_bRespectElders = true;
117  else if (uiData == SPECIAL)
118  m_bRespectElders = false;
119  else if (uiData == DONE)
120  {
121  DoToggleGameObjectFlags(GO_ANCIENT_DEVICE_L, GO_FLAG_NO_INTERACT, false);
122  DoToggleGameObjectFlags(GO_ANCIENT_DEVICE_R, GO_FLAG_NO_INTERACT, false);
123  }
124  break;
125  case TYPE_TALDARAM:
126  if (uiData == SPECIAL)
127  {
128  ++m_uiDevicesActivated;
129 
130  if (m_uiDevicesActivated == 2)
131  {
132  m_auiEncounter[uiType] = uiData;
133  DoUseDoorOrButton(GO_VORTEX);
134 
135  // Lower Taldaram
136  if (Creature* pTaldaram = GetSingleCreatureFromStorage(NPC_TALDARAM))
137  pTaldaram->GetMotionMaster()->MovePoint(1, aTaldaramLandingLoc[0], aTaldaramLandingLoc[1], aTaldaramLandingLoc[2]);
138 
139  // Interrupt the channeling
140  for (GuidList::const_iterator itr = m_lJedogaControllersGuidList.begin(); itr != m_lJedogaControllersGuidList.end(); ++itr)
141  {
142  if (Creature* pTemp = instance->GetCreature(*itr))
143  pTemp->InterruptNonMeleeSpells(false);
144  }
145  }
146  }
147  else if (uiData == DONE)
148  {
149  m_auiEncounter[uiType] = uiData;
150  DoUseDoorOrButton(GO_DOOR_TALDARAM);
151  }
152  break;
153  case TYPE_JEDOGA:
154  m_auiEncounter[uiType] = uiData;
155  if (uiData == IN_PROGRESS)
156  m_bVolunteerWork = true;
157  else if (uiData == SPECIAL)
158  m_bVolunteerWork = false;
159  break;
160  case TYPE_AMANITAR:
161  m_auiEncounter[uiType] = uiData;
162  break;
163  case TYPE_VOLAZJ:
164  m_auiEncounter[uiType] = uiData;
165  if (uiData == IN_PROGRESS)
166  {
167  m_uiTwistedVisageCount = 0;
168  m_lInsanityPlayersGuidList.clear();
169 
170  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_VOLAZJ_ID);
171  }
172  break;
173 
174  default:
175  script_error_log("Instance Ahn'Kahet: ERROR SetData = %u for type %u does not exist/not implemented.", uiType, uiData);
176  break;
177  }
178 
179  // For some encounters Special data needs to be saved
180  if (uiData == DONE || (uiData == SPECIAL && uiType == TYPE_TALDARAM))
181  {
182  OUT_SAVE_INST_DATA;
183 
184  std::ostringstream saveStream;
185  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3]
186  << " " << m_auiEncounter[4];
187 
188  m_strInstData = saveStream.str();
189 
190  SaveToDB();
191  OUT_SAVE_INST_DATA_COMPLETE;
192  }
193 }

195 void instance_ahnkahet::OnCreatureDeath(Creature* pCreature)
196 {
197  switch (pCreature->GetEntry())
198  {
199  case NPC_TWILIGHT_INITIATE:
200  ++m_uiInitiatesKilled;
201 
202  // If all initiates are killed, then land Jedoga and stop the channeling
203  if (m_uiInitiatesKilled == MAX_INITIATES)
204  {
205  if (Creature* pJedoga = GetSingleCreatureFromStorage(NPC_JEDOGA_SHADOWSEEKER))
206  pJedoga->GetMotionMaster()->MovePoint(1, aJedogaLandingLoc[0], aJedogaLandingLoc[1], aJedogaLandingLoc[2]);
207 
208  for (GuidList::const_iterator itr = m_lJedogaEventControllersGuidList.begin(); itr != m_lJedogaEventControllersGuidList.end(); ++itr)
209  {
210  if (Creature* pTemp = instance->GetCreature(*itr))
211  pTemp->InterruptNonMeleeSpells(false);
212  }
213  }
214 
215  break;
216  case NPC_TWISTED_VISAGE_1:
217  case NPC_TWISTED_VISAGE_2:
218  case NPC_TWISTED_VISAGE_3:
219  case NPC_TWISTED_VISAGE_4:
220  case NPC_TWISTED_VISAGE_5:
221  pCreature->CastSpell(pCreature, SPELL_TWISTED_VISAGE_DEATH, TRIGGERED_OLD_TRIGGERED);
222 
223  --m_uiTwistedVisageCount;
224 
225  // When all Twisted Visages were killed or despawned switch back to combat phase
226  if (!m_uiTwistedVisageCount)
227  {
228  // Clear Insanity
229  if (Creature* pVolazj = GetSingleCreatureFromStorage(NPC_HERALD_VOLAZJ))
230  {
231  pVolazj->CastSpell(pVolazj, SPELL_INSANITY_CLEAR, TRIGGERED_OLD_TRIGGERED);
232  pVolazj->RemoveAllAuras();
233  }
234 
235  // Clear insanity manually for now, because the spell won't hit phased players
236  HandleInsanityClear();
237 
238  SetData(TYPE_VOLAZJ, IN_PROGRESS);
239  }
240  else
241  {
242  // Switch Insanity
243  if (Creature* pVolazj = GetSingleCreatureFromStorage(NPC_HERALD_VOLAZJ))
244  pVolazj->CastSpell(pVolazj, SPELL_INSANITY_SWITCH, TRIGGERED_OLD_TRIGGERED);
245 
246  // Handle insanity switch manually, because the boss can't hit phased players
247  if (pCreature->IsTemporarySummon())
248  {
249  // Switch insanity phase for the master player
250  if (Player* pPlayer = instance->GetPlayer(pCreature->GetSpawnerGuid()))
251  HandleInsanitySwitch(pPlayer);
252  }
253  }
254  break;
255  }
256 }

331 void instance_ahnkahet::HandleInsanitySwitch(Player* pPhasedPlayer)
332 {
333  // Get the phase aura id
334  std::list<Aura*> lAuraList = pPhasedPlayer->GetAurasByType(SPELL_AURA_PHASE);
335  if (lAuraList.empty())
336  return;
337 
338  uint32 uiPhaseAura = (*lAuraList.begin())->GetId();
339 
340  PlayerList lSamePhasePlayers;
341  std::vector<Player*> vOtherPhasePlayers;
342 
343  // Sort the insanity players, into those which have same phase and others
344  for (GuidList::const_iterator itr = m_lInsanityPlayersGuidList.begin(); itr != m_lInsanityPlayersGuidList.end(); ++itr)
345  {
346  if (Player* pTemp = instance->GetPlayer(*itr))
347  {
348  if (pTemp->HasAura(uiPhaseAura))
349  lSamePhasePlayers.push_back(pTemp);
350  // Check only for alive players
351  else if (pTemp->IsAlive())
352  vOtherPhasePlayers.push_back(pTemp);
353  }
354  }
355 
356  // This shouldn't happen
357  if (vOtherPhasePlayers.empty())
358  return;
359 
360  // Get the phase aura of the new selected player
361  Player* pNewPlayer = vOtherPhasePlayers[urand(0, vOtherPhasePlayers.size() - 1)];
362 
363  // Get the phase aura id
364  std::list<Aura*> lNewAuraList = pNewPlayer->GetAurasByType(SPELL_AURA_PHASE);
365  if (lNewAuraList.empty())
366  return;
367 
368  uint32 uiNewPhaseAura = (*lNewAuraList.begin())->GetId();
369 
370  // Move the same phase players to the new phase
371  for (PlayerList::const_iterator itr = lSamePhasePlayers.begin(); itr != lSamePhasePlayers.end(); ++itr)
372  (*itr)->CastSpell((*itr), uiNewPhaseAura, TRIGGERED_OLD_TRIGGERED);
373 }
