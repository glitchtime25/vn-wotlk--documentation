243 bool BattleGroundIC::HandleEvent(uint32 eventId, Object* source, Object* target)
244 {
245  if (!source->IsGameObject())
246  return false;
247 
248  GameObject* go = static_cast<GameObject*>(source);

250  DEBUG_LOG("BattleGroundIC: Handle event for gameobject entry %u.", go->GetEntry());
251 
252  for (uint8 i = 0; i < BG_IC_MAX_KEEP_GATES; ++i)
253  {
254  if (eventId == isleAllianceWallsData[i].eventId)
255  {
256  GetBgMap()->GetVariableManager().SetVariable(m_gatesAllianceState[i], WORLD_STATE_REMOVE);

262 
263  SendMessage2ToAll(isleAllianceWallsData[i].message, CHAT_MSG_BG_SYSTEM_HORDE, nullptr);
264 
265  if (!m_isKeepInvaded[TEAM_INDEX_ALLIANCE])
266  {
267  // open the inner keep gates
268  if (GameObject* pGate = GetBgMap()->GetGameObject(m_allianceInnerGate1Guid))
269  pGate->UseDoorOrButton();
270  if (GameObject* pGate = GetBgMap()->GetGameObject(m_allianceInnerGate2Guid))
271  pGate->UseDoorOrButton();
272 
273  // spawn the boss
274  for (const auto& i : iocAllianceKeepSpawns)
275  go->SummonCreature(i.entry, i.x, i.y, i.z, i.o, TEMPSPAWN_DEAD_DESPAWN, 0);
276 
277  // allow the graveyard to be captured
278  if (GameObject* pFlag = GetSingleGameObjectFromStorage(BG_IC_GO_BANNER_ALLIANCE_KEEP_A))
279  pFlag->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
280  }
281 
282  m_isKeepInvaded[TEAM_INDEX_ALLIANCE] = true;
283 
284  // allow script to continue using DB scripts
285  return false;
286  }
287 
288  if (eventId == isleHordeWallsData[i].eventId)
289  {
290  GetBgMap()->GetVariableManager().SetVariable(m_gatesHordeState[i], WORLD_STATE_REMOVE);
291  m_gatesHordeState[i] = isleHordeWallsData[i].stateOpened;
292  GetBgMap()->GetVariableManager().SetVariable(m_gatesHordeState[i], WORLD_STATE_ADD);
293 
294  if (GameObject* pGate = GetBgMap()->GetGameObject(m_keepGatesGuid[TEAM_INDEX_HORDE][i]))
295  pGate->UseDoorOrButton(0, true);
296 
297  SendMessage2ToAll(isleHordeWallsData[i].message, CHAT_MSG_BG_SYSTEM_ALLIANCE, nullptr);
298 
299  if (!m_isKeepInvaded[TEAM_INDEX_HORDE])
300  {
301  // open the inner keep gates
302  if (GameObject* pGate = GetBgMap()->GetGameObject(m_hordeInnerGateGuid))
303  pGate->UseDoorOrButton();
304 
305  // spawn the boss

312  }
313 
314  m_isKeepInvaded[TEAM_INDEX_HORDE] = true;
315 
316  // allow script to continue using DB scripts

318  }
319  }

324 // Called when a player clicks a capturable banner
325 void BattleGroundIC::HandlePlayerClickedOnFlag(Player* player, GameObject* go)
326 {
327  DEBUG_LOG("BattleGroundIC: Handle flag clicked for gameobject entry %u.", go->GetEntry());
328 
329  PvpTeamIndex newOwnerIdx = GetTeamIndexByTeamId(player->GetTeam());
330  uint8 nodeId = 0;
331  uint32 newWorldState = 0;
332  uint32 soundId = 0;
333  uint32 nextFlagEntry = 0;
334  int32 textEntry = 0;
335 
336  // *** Check if status changed from neutral to contested ***
337  for (const auto& i : isleGameObjectNeutralData)
338  {
339  if (go->GetEntry() == i.objectEntry)
340  {
341  UpdatePlayerScore(player, SCORE_BASES_ASSAULTED, 1);
342  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 245);
343 
344  nodeId = i.objectiveId;
345  newWorldState = newOwnerIdx == TEAM_INDEX_ALLIANCE ? i.nextWorldStateAlly : i.nextWorldStateHorde;
346 
347  textEntry = LANG_BG_IC_NODE_ASSAULTED;
348  soundId = newOwnerIdx == TEAM_INDEX_ALLIANCE ? BG_IC_SOUND_NODE_ASSAULTED_ALLIANCE : BG_IC_SOUND_NODE_ASSAULTED_HORDE;
349 
350  nextFlagEntry = newOwnerIdx == TEAM_INDEX_ALLIANCE ? i.nextObjectAlly : i.nextObjectHorde;
351  break;
352  }
353  }
354 
355  if (!nodeId)
356  {
357  // *** Check if status changed from owned to contested ***
358  for (const auto& i : isleGameObjectOwnedData)
359  {
360  if (go->GetEntry() == i.objectEntry)
361  {
362  UpdatePlayerScore(player, SCORE_BASES_ASSAULTED, 1);
363  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 245);
364 
365  nodeId = i.objectiveId;
366  newWorldState = i.nextState;
367 
368  textEntry = LANG_BG_IC_NODE_ASSAULTED;
369  soundId = newOwnerIdx == TEAM_INDEX_ALLIANCE ? BG_IC_SOUND_NODE_ASSAULTED_ALLIANCE : BG_IC_SOUND_NODE_ASSAULTED_HORDE;
370 
371  nextFlagEntry = i.nextObject;
372 
373  // reset the objective
374  DoResetObjective(IsleObjective(nodeId));
375  break;
376  }
377  }
378  }
379 
380  if (!nodeId)
381  {
382  // *** Check if status changed from contested to owned / contested ***
383  for (const auto& i : isleGameObjectContestedData)
384  {
385  if (go->GetEntry() == i.objectEntry)
386  {
387  nodeId = i.objectiveId;
388 
389  UpdatePlayerScore(player, m_isleNode[nodeId].nodeConquerer == newOwnerIdx ? SCORE_BASES_DEFENDED : SCORE_BASES_ASSAULTED, 1);
390  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, m_isleNode[nodeId].nodeConquerer == newOwnerIdx ? 246 : 245);
391 
392  newWorldState = m_isleNode[nodeId].nodeConquerer == newOwnerIdx ? i.nextStateDefend : i.nextStateAssault;
393 
394  textEntry = m_isleNode[nodeId].nodeConquerer == newOwnerIdx ? LANG_BG_IC_NODE_DEFENDED : LANG_BG_IC_NODE_ASSAULTED;
395  soundId = newOwnerIdx == TEAM_INDEX_ALLIANCE ? BG_IC_SOUND_NODE_ASSAULTED_ALLIANCE : BG_IC_SOUND_NODE_ASSAULTED_HORDE;
396 
397  nextFlagEntry = m_isleNode[nodeId].nodeConquerer == newOwnerIdx ? i.nextObjectDefend : i.nextObjectDefend;
398 
399  // re-apply benefits
400  if (m_isleNode[nodeId].nodeConquerer == newOwnerIdx)
401  DoApplyObjectiveBenefits(IsleObjective(nodeId), go);
402  break;
403  }
404  }
405  }
406 
407  // only process the event if needed
408  if (nodeId)
409  {
410  m_isleNode[nodeId].nodeOwner = newOwnerIdx;
411 
412  // update world states
413  GetBgMap()->GetVariableManager().SetVariable(m_isleNode[nodeId].nodeWorldState, WORLD_STATE_REMOVE);
414  m_isleNode[nodeId].nodeWorldState = newWorldState;
415  GetBgMap()->GetVariableManager().SetVariable(m_isleNode[nodeId].nodeWorldState, WORLD_STATE_ADD);
416 
417  // start timer
418  m_isleNode[nodeId].nodeChangeTimer = BG_IC_FLAG_CAPTURING_TIME;
419 
420  // send the zone message and sound
421  ChatMsg chatSystem = newOwnerIdx == TEAM_INDEX_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE;
422  uint32 factionStrig = newOwnerIdx == TEAM_INDEX_ALLIANCE ? LANG_BG_ALLY : LANG_BG_HORDE;
423 
424  SendMessage2ToAll(textEntry, chatSystem, player, isleObjectiveData[nodeId].message, factionStrig);
425  PlaySoundToAll(soundId);
426 
427  // change banners
428  m_isleNode[nodeId].bannerChangeTimer = 3000;
429  m_isleNode[nodeId].currentBannerEntry = nextFlagEntry;
430  m_isleNode[nodeId].oldBannerEntry = go->GetEntry();
431  }

434 void BattleGroundIC::HandleKillUnit(Creature* creature, Player* killer)
435 {
436  DEBUG_LOG("BattleGroundIC: Handle unit kill for creature entry %u.", creature->GetEntry());
437 
438  if (GetStatus() != STATUS_IN_PROGRESS)
439  return;
440 
441  // end battle if boss was killed
442  switch (creature->GetEntry())
443  {
444  case BG_IC_NPC_COMMANDER_WYRMBANE:
445  RewardHonorToTeam(GetBonusHonorFromKill(BG_IC_BONUS_HONOR_BOSS_KILL), HORDE);
446  EndBattleGround(HORDE);
447  break;
448  case BG_IC_NPC_OVERLORD_AGMAR:
449  RewardHonorToTeam(GetBonusHonorFromKill(BG_IC_BONUS_HONOR_BOSS_KILL), ALLIANCE);
450  EndBattleGround(ALLIANCE);
451  break;
452  case BG_IC_VEHICLE_DEMOLISHER:
453  case BG_IC_VEHICLE_CATAPULT:
454  case BG_IC_VEHICLE_GLAIVE_THROWER_A:
455  case BG_IC_VEHICLE_GLAIVE_THROWER_H:
456  creature->CastSpell(killer, BG_IC_SPELL_ACHIEV_DESTROYED_VEHICLE, TRIGGERED_OLD_TRIGGERED);
457  break;
458  case BG_IC_VEHICLE_SIEGE_ENGINE_A:
459  case BG_IC_VEHICLE_SIEGE_ENGINE_H:
460  {
461  // kill credit
