235  void ExecuteAction(uint32 action) override
236  {
237  switch (action)
238  {
239  case TEROKK_COMBAT_ACTION_SPAWN_ACE:
240  if (m_creature->GetHealthPercent() > 50.f)
241  return;
242  m_creature->SummonCreature(NPC_ACE, -3335.223f, 3431.947f, 426.3864f, 0.f, TEMPSPAWN_MANUAL_DESPAWN, 0, true, true);
243  m_creature->SummonCreature(NPC_ACE, -3335.223f, 3431.947f, 426.3864f, 0.f, TEMPSPAWN_MANUAL_DESPAWN, 0, true, true);
244  m_creature->SummonCreature(NPC_ACE, -3335.223f, 3431.947f, 426.3864f, 0.f, TEMPSPAWN_MANUAL_DESPAWN, 0, true, true);
245  SetActionReadyStatus(action, false);
246  return; // not exclusive action
247  case TEROKK_COMBAT_ACTION_PHASE_2:
248  if (m_creature->GetHealthPercent() > 25.f)
249  break;
250  m_phase = true;
251  DoScriptText(SAY_DIVINE_SHIELD, m_creature);
252  if (Creature* ace = m_creature->GetMap()->GetCreature(m_aces[0]))
253  DoScriptText(SAY_FLAMES, ace);
254  SetActionReadyStatus(action, false);
255  return; // not exclusive action
256  case TEROKK_COMBAT_ACTION_DIVINE_SHIELD:
257  if (!m_phase)
258  return;
259  m_creature->RemoveAurasDueToSpell(SPELL_ENRAGE);
260  if (DoCastSpellIfCan(nullptr, SPELL_DIVINE_SHIELD) == CAST_OK)
261  {
262  Creature* stalker = GetClosestCreatureWithEntry(m_creature, NPC_INVISIBLE_STALKER, 70.f);
263  if (stalker)
264  if (Creature* ace = m_creature->GetMap()->GetCreature(m_aces[0]))
265  ace->AI()->DoCastSpellIfCan(stalker, SPELL_SKYGUARD_FLARE);
266  ResetTimer(TEROKK_ACTION_ACE_CAST, 15000);
267  ResetCombatAction(action, GetSubsequentActionTimer(TerokkActions(action)));
268  }
269  return;
270  case TEROKK_COMBAT_ACTION_SHADOW_BOLT_VOLLEY:
271  if (DoCastSpellIfCan(nullptr, SPELL_SHADOW_BOLT_VOLLEY) == CAST_OK)
272  ResetCombatAction(action, GetSubsequentActionTimer(TerokkActions(action)));
273  return;
274  case TEROKK_COMBAT_ACTION_CHOSEN_ONE:
275  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
276  {
277  if (DoCastSpellIfCan(target, SPELL_CHOSEN_ONE) == CAST_OK)
278  {
279  m_creature->CastSpell(nullptr, SPELL_WILL_OF_THE_ARAKKOA_GOD, TRIGGERED_NONE);
280  DoScriptText(SAY_CHOSEN_ONE, m_creature, target);
281  ResetCombatAction(action, GetSubsequentActionTimer(TerokkActions(action)));
282  return;
283  }
284  }
285  return;
286  case TEROKK_COMBAT_ACTION_CLEAVE:
287  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
288  ResetCombatAction(action, GetSubsequentActionTimer(TerokkActions(action)));
289  return;
290  }
291  }
