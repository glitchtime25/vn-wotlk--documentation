100  {
101  switch (action)
102  {
103  case COMMON_RETARGET:
104  {
105  if (m_creature->getThreatManager().getThreatList().size() <= 1)
106  break;
107 
108  std::vector<Unit*> units;
109  DoResetThreat();
110  for (auto& data : m_creature->getThreatManager().getThreatList())
111  {
112  if (data->isValid())
113  if (data->getTarget()->IsPlayer())
114  units.push_back(data->getTarget());
115  }
116  std::shuffle(units.begin(), units.end(), *GetRandomGenerator());
117  uint32 i = 0;
118  for (Unit* unit : units)
119  m_creature->AddThreat(unit, urand(1, (++i) * 100000));
120  ResetCombatAction(action, 10000);
121  break;
122  }
123  case COMMON_MEDALLION:
124  if (m_creature->isFrozen() || m_creature->IsCrowdControlled())
