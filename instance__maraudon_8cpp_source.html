55 void instance_maraudon::SetData(uint32 type, uint32 data)
56 {
57  if (type == TYPE_NOXXION)
58  {
59  // Do nothing if we are receiving the same data
60  if (m_encounter[TYPE_NOXXION] == data)
61  return;
62 
63  m_encounter[type] = data;
64  switch (data)
65  {
66  // If Noxxion encounter is done or spewer attacked, stop spewing larvas
67  case DONE:
68  // Destroy the corruption spewer near Noxxion
69  if (GameObject* go = GetSingleGameObjectFromStorage(GO_CORRUPTION_SPEWER))
70  go->SetLootState(GO_ACTIVATED);
71  // no break, the special part is also executed
72  case SPECIAL:
73  m_spewLarvaTimer = 0;
74  // Destroy the larva spewer if not already done
75  if (GameObject* go = GetSingleGameObjectFromStorage(GO_LARVA_SPEWER))
76  {
77  if (go->GetLootState() != GO_ACTIVATED)
78  go->SetLootState(GO_ACTIVATED);
79  }
80  }
81  }
82 
83  if (data == DONE)
84  {
85  OUT_SAVE_INST_DATA;
86 
87  std::ostringstream saveStream;
88 
89  saveStream << m_encounter[0];
90 
91  m_strInstData = saveStream.str();
92  SaveToDB();
93  OUT_SAVE_INST_DATA_COMPLETE;
94  }
95 }

147 void instance_maraudon::Update(uint32 uiDiff)
148 {
149  // Nicely spew a larva that will go toward Noxxion (pathing is handled by DB)
150  if (m_spewLarvaTimer)
151  {
152  if (m_spewLarvaTimer <= uiDiff)
153  {
154  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_LARVA_SPEWER))
155  pGo->SummonCreature(NPC_SPEWED_LARVA, spawnLocation.m_fX, spawnLocation.m_fY, spawnLocation.m_fZ, spawnLocation.m_fO, TEMPSPAWN_TIMED_DESPAWN, 10 * MINUTE * IN_MILLISECONDS); // Despawn is handled by DBScript we just want to avoid an infinite number of larvas near Noxxion
156  m_spewLarvaTimer = 1 * MINUTE * IN_MILLISECONDS;
157  }
158  else
159  m_spewLarvaTimer -= uiDiff;
160  }
161 }
