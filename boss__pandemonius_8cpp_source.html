85  void UpdateAI(const uint32 uiDiff) override
86  {
87  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
88  return;
89 
90  if (m_uiVoidBlastTimer < uiDiff)
91  {
92  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, m_bIsRegularMode ? SPELL_VOID_BLAST : SPELL_VOID_BLAST_H))
93  DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_VOID_BLAST : SPELL_VOID_BLAST_H);
94 
95  // reset timer and counter when counter has reached the max limit
96  if (m_uiVoidBlastCounter == MAX_VOID_BLASTS)
97  {
98  m_uiVoidBlastTimer = urand(25000, 30000);
99  m_uiVoidBlastCounter = 0;
100  }
101  // cast the void blasts in a row until we reach the max limit
102  else
103  {
104  m_uiVoidBlastTimer = 500;
105  ++m_uiVoidBlastCounter;
106  }
107  }
108  else
109  m_uiVoidBlastTimer -= uiDiff;
110 
111  // use the darkshell only when the boss isn't casting the void blasts
112  if (!m_uiVoidBlastCounter)
113  {
114  if (m_uiDarkShellTimer < uiDiff)
115  {
116  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_DARK_SHELL : SPELL_DARK_SHELL_H) == CAST_OK)
117  {
118  DoScriptText(EMOTE_DARK_SHELL, m_creature);
119  m_uiDarkShellTimer = urand(18000, 30000);
120  }
121  }
122  else
123  m_uiDarkShellTimer -= uiDiff;
124  }
125 
126  DoMeleeAttackIfReady();
127  }
