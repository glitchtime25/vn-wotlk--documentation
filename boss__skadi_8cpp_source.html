280  ResetCombatAction(action, 23000);

283  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))

286  ResetCombatAction(action, urand(10000, 15000));
287  }
288  break;
289  }
