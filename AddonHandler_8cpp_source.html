22 bool FingerprintValid(uint32 fingerprint)
23 {
24  // no bytes can equal 0x00, 0x01, or 0x02
25  const uint8 *b = reinterpret_cast<uint8 *>(&fingerprint);
26  for (auto i = 0u; i < sizeof(fingerprint); ++i)
27  if (b[i] <= 0x02)
28  return false;
29 
30  return true;
31 }

33 bool FingerprintExists(uint32 fingerprint)
34 {

44 uint32 GenerateFingerprint()
45 {
46  do
47  {
48  uint32 fingerprint = 0;
49 
50  // generate four random bytes between 0x03 and 0xFF
51  for (auto i = 0u; i < sizeof(uint32); ++i)
52  fingerprint |= static_cast<uint8>(urand(0x03, 0xFF)) << 8*i;
53 
54  // if the fingerprint already exists, repeat
55  if (FingerprintExists(fingerprint))
56  continue;
57 
58  // if we reach here, we are done
59 
60  return fingerprint;
61  } while (true);
62 }
