28 // This part provides assistance to a player that are attacked by pWho, even if out of normal aggro range
29 // It will cause m_creature to attack pWho that are attacking _any_ player (which has been confirmed may happen also on offi)
30 bool FollowerAI::AssistPlayerInCombat(Unit* who)
31 {
32  if (!HasFollowState(STATE_FOLLOW_INPROGRESS))
33  return false;
34 
35  if (!who->GetVictim())
36  return false;
37 
38  // experimental (unknown) flag not present
39  if (!(m_creature->GetCreatureInfo()->CreatureTypeFlags & CREATURE_TYPEFLAGS_CAN_ASSIST))
40  return false;
41 
42  // unit state prevents (similar check is done in CanInitiateAttack which also include checking unit_flags. We skip those here)
43  if (m_creature->hasUnitState(UNIT_STAT_STUNNED | UNIT_STAT_FEIGN_DEATH | UNIT_STAT_CONFUSED | UNIT_STAT_FLEEING))
44  return false;
45 
46  // victim of pWho is not a player
47  if (!who->GetVictim()->GetBeneficiaryPlayer())
48  return false;
49 
50  // never attack friendly
51  if (!m_creature->CanAttackNow(who))
52  return false;
53 
54  // too far away and no free sight?
55  if (m_creature->IsWithinDistInMap(who, MAX_PLAYER_DISTANCE) && m_creature->IsWithinLOSInMap(who))
56  {
57  // already fighting someone?
58  if (!m_creature->GetVictim())
59  {
60  AttackStart(who);
61  return true;
62  }
63  m_creature->EngageInCombatWith(who);
64  return true;
65  }
66 
67  return false;
68 }

132 void FollowerAI::UpdateAI(const uint32 diff)
133 {
134  if (HasFollowState(STATE_FOLLOW_INPROGRESS) && !m_creature->GetVictim())
135  {
136  if (m_updateFollowTimer < diff)
137  {
138  if (HasFollowState(STATE_FOLLOW_COMPLETE) && !HasFollowState(STATE_FOLLOW_POSTEVENT))
139  {
140  debug_log("SD2: FollowerAI is set completed, despawns.");
141  m_creature->ForcedDespawn();
142  return;
143  }
144 
145  bool isMaxRangeExceeded = true;
146 
147  if (Player* player = GetLeaderForFollower())
148  {
149  if (HasFollowState(STATE_FOLLOW_RETURNING))
150  {
151  debug_log("SD2: FollowerAI is returning to leader.");
152 
153  RemoveFollowState(STATE_FOLLOW_RETURNING);
154  m_creature->GetMotionMaster()->MoveFollow(player, PET_FOLLOW_DIST, PET_FOLLOW_ANGLE);
155  return;
156  }
157 
158  if (Group* pGroup = player->GetGroup())
159  {
160  for (GroupReference* ref = pGroup->GetFirstMember(); ref != nullptr; ref = ref->next())
161  {
162  Player* member = ref->getSource();
163 
164  if (member && m_creature->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE))
165  {
166  isMaxRangeExceeded = false;
167  break;
168  }
169  }
170  }
171  else
172  {
173  if (m_creature->IsWithinDistInMap(player, MAX_PLAYER_DISTANCE))
174  isMaxRangeExceeded = false;
175  }
176  }
177 
178  if (isMaxRangeExceeded)
179  {
180  debug_log("SD2: FollowerAI failed because player/group was to far away or not found");
181  m_creature->ForcedDespawn();
182  return;
183  }
184 
185  m_updateFollowTimer = 1000;
186  }
187  else
188  m_updateFollowTimer -= diff;
189  }
190 
191  UpdateFollowerAI(diff);
192 }

257 Player* FollowerAI::GetLeaderForFollower()
258 {
259  if (Player* leader = m_creature->GetMap()->GetPlayer(m_leaderGuid))
260  {
261  if (leader->IsAlive())
262  return leader;
263  if (Group* group = leader->GetGroup())

267  Player* member = ref->getSource();
268 
269  if (member && member->IsAlive() && m_creature->IsWithinDistInMap(member, MAX_PLAYER_DISTANCE))

272  m_leaderGuid = member->GetObjectGuid();
