216  void HandleEmerge()
217  {
218  m_creature->RemoveAurasDueToSpell(SPELL_AHUNE_SELF_STUN);
219 
220  DoCastSpellIfCan(m_creature, SPELL_BIRTH);
221  DoCastSpellIfCan(m_creature, SPELL_STAND);
222  SpawnBunnies();
223 
224  if (Creature* pCore = m_creature->GetMap()->GetCreature(m_frozenCoreGuid))
225  pCore->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_IMMUNE_TO_PLAYER);
226 
227  ResetCombatAction(AHUNE_FROSTWIND, urand(20000, 25000));
228  ResetCombatAction(AHUNE_HAILSTONE, 1000);
229  ResetCombatAction(AHUNE_COLDWAVE, urand(5000, 10000));
230  ResetCombatAction(AHUNE_COLD_SLAP, 1000);
231  ResetCombatAction(AHUNE_SUBMERGE, 90000);
232 
233  SetMeleeEnabled(true);
234  SetCombatScriptStatus(false);
235 
236  ++m_uiPhaseChangeCount;
237  }

239  void ExecuteAction(uint32 action)
240  {
241  switch (action)
242  {
243  case AHUNE_SUBMERGE:
244  if (DoCastSpellIfCan(m_creature, SPELL_SUBMERGE) == CAST_OK)
245  {
246  DisableCombatAction(AHUNE_SUBMERGE);
247  DisableCombatAction(AHUNE_COLD_SLAP);
248  DisableCombatAction(AHUNE_HAILSTONE);
249  DisableCombatAction(AHUNE_FROSTWIND);
250  DisableCombatAction(AHUNE_COLDWAVE);
251  ResetTimer(AHUNE_EMERGE, 40000);
252 
253  SetMeleeEnabled(false);
254  SetCombatScriptStatus(true);
255  }
256  break;
257  case AHUNE_COLD_SLAP:
258  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, SPELL_COLD_SNAP, SELECT_FLAG_PLAYER))
259  if (DoCastSpellIfCan(target, SPELL_COLD_SNAP) == CAST_OK)
260  ResetCombatAction(action, 3000);
261  break;
262  case AHUNE_HAILSTONE:
263  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_HAILSTONE) == CAST_OK)
264  DisableCombatAction(action);
265  break;
266  case AHUNE_COLDWAVE:
267  {
268  bool ok = false;
269  for (uint8 i = 0; i < 2; ++i)
270  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_COLDWAVE) == CAST_OK)
271  ok = true;
272 
273  if (ok)
274  ResetCombatAction(action, urand(5000, 10000));
275  break;
276  }
277  case AHUNE_FROSTWIND:
278  {
279  bool ok = false;
280  for (uint8 i = 0; i < m_uiPhaseChangeCount; ++i)
281  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_FROSTWIND) == CAST_OK)
282  ok = true;
283 
284  if (ok)
285  ResetCombatAction(action, urand(5000, 10000));
286  break;
287  }
288  }
289  }
