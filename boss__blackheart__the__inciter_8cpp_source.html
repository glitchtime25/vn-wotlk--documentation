153  void HandleInciteEnd()
154  {
155  SetCombatScriptStatus(false);
156  SetCombatMovement(true);
157  m_meleeEnabled = true;
158  if (m_creature->GetVictim())
159  {
160  m_creature->MeleeAttackStart(m_creature->GetVictim());
161  m_creature->SetTarget(m_creature->GetVictim());
162  }
163  }
164 
165  void ExecuteAction(uint32 action) override
166  {
167  switch (action)
168  {
169  case BLACKHEART_INCITE_CHAOS:
170  {
171  if (m_creature->getThreatManager().getThreatList().size() > 1)
172  {
173  if (DoCastSpellIfCan(m_creature, SPELL_INCITE_CHAOS) == CAST_OK)
174  {
175  DoScriptText(SAY_INCITE_CHAOS, m_creature);
176  SetCombatScriptStatus(true);
177  SetCombatMovement(false);
178  m_meleeEnabled = false;
179  DoResetThreat();
180  m_creature->MeleeAttackStop(m_creature->GetVictim());
181  m_creature->SetTarget(nullptr);
182  m_creature->CastSpell(nullptr, SPELL_LAUGH_PERIODIC, TRIGGERED_NONE);
183  ResetCombatAction(action, GetSubsequentActionTimer(BlackheartActions(action)));
184  ResetTimer(BLACKHEART_INCITE_TIMER, 19000);
185  return;
186  }
187  }
188  break;
189  }
190  case BLACKHEART_CHARGE:
191  {
192  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_CHARGE, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IN_MELEE_RANGE))
193  {
194  if (DoCastSpellIfCan(target, SPELL_CHARGE) == CAST_OK)
195  {
196  ResetCombatAction(action, GetSubsequentActionTimer(BlackheartActions(action)));
197  DoResetThreat();
198  }
199  }
200  break;
201  }
202  case BLACKHEART_KNOCKBACK:
203  {
204  if (DoCastSpellIfCan(nullptr, SPELL_WAR_STOMP) == CAST_OK)
205  {
206  ResetCombatAction(action, GetSubsequentActionTimer(BlackheartActions(action)));
207  DoResetThreat();
208  }
209  break;
210  }
211  }
212  }
