140  // Wrapper to handle the drakkari invaders summon
141  void DoSummonDrakkariInvaders()
142  {
143  if (!m_pInstance)
144  return;
145 
146  // check if there are there are at least 2 triggers in the vector
147  if (m_vTriggers.size() < 2)
148  return;
149 
150  if (roll_chance_i(30))
151  {
152  // Summon a troll in the corner and 2 trolls in the air
153  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_pInstance->GetTrollgoreCornerTrigger()))
154  pTrigger->CastSpell(pTrigger, roll_chance_i(20) ? SPELL_SUMMON_INVADER_1 : SPELL_SUMMON_INVADER_2, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
155 
156  // get two random outside triggers
157  uint8 uiMaxTriggers = m_vTriggers.size();
158  uint8 uiPos1 = urand(0, uiMaxTriggers - 1);
159  uint8 uiPos2 = (uiPos1 + urand(1, uiMaxTriggers - 1)) % uiMaxTriggers;
160 
161  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_vTriggers[uiPos1]))
162  pTrigger->CastSpell(pTrigger, roll_chance_i(30) ? SPELL_SUMMON_INVADER_1 : SPELL_SUMMON_INVADER_2, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
163  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_vTriggers[uiPos2]))
164  pTrigger->CastSpell(pTrigger, roll_chance_i(30) ? SPELL_SUMMON_INVADER_1 : SPELL_SUMMON_INVADER_2, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
165  }
166  else
167  {
168  // Summon 3 trolls in the air
169  for (auto m_vTrigger : m_vTriggers)
170  {
171  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_vTrigger))
172  pTrigger->CastSpell(pTrigger, roll_chance_i(30) ? SPELL_SUMMON_INVADER_1 : SPELL_SUMMON_INVADER_2, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
173  }
174  }
175  }

177  void UpdateAI(const uint32 uiDiff) override
178  {
179  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
180  return;
181 
182  if (m_uiCrushTimer < uiDiff)
183  {
184  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CRUSH) == CAST_OK)
185  m_uiCrushTimer = 10000;
186  }
187  else
188  m_uiCrushTimer -= uiDiff;
189 
190  if (m_uiInfectedWoundTimer < uiDiff)
191  {
192  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_INFECTED_WOUND) == CAST_OK)
193  m_uiInfectedWoundTimer = urand(20000, 30000);
194  }
195  else
196  m_uiInfectedWoundTimer -= uiDiff;
197 
198  if (m_uiWaveTimer < uiDiff)
199  {
200  DoSummonDrakkariInvaders();
201  m_uiWaveTimer = 30000;
202  }
203  else
204  m_uiWaveTimer -= uiDiff;
205 
206  if (m_uiConsumeTimer < uiDiff)
207  {
208  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_CONSUME : SPELL_CONSUME_H) == CAST_OK)
209  {
210  DoScriptText(SAY_CONSUME, m_creature);
211  m_uiConsumeTimer = 15000;
212  }
213  }
214  else
215  m_uiConsumeTimer -= uiDiff;
216 
217  if (m_uiCorpseExplodeTimer < uiDiff)
218  {
219  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_CORPSE_EXPLODE : SPELL_CORPSE_EXPLODE_H) == CAST_OK)
220  {
221  DoScriptText(SAY_EXPLODE, m_creature);
222  m_uiCorpseExplodeTimer = 10000;
223  }
224  }
225  else
226  m_uiCorpseExplodeTimer -= uiDiff;
227 
228  DoMeleeAttackIfReady();
229  }
