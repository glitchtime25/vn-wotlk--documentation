99  ResetCombatAction(action, urand(4000, 12000));
100  break;
101  case MAKER_DOMINATION:
102  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
103  if (DoCastSpellIfCan(target, SPELL_DOMINATION) == CAST_OK)
104  ResetCombatAction(action, urand(20000, 25000));
105  break;
106  }
