211  void UpdateAI(const uint32 uiDiff) override
212  {
213  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
214  return;
215 
216  if (!m_pInstance)
217  return;
218 
219  // Slime Spray
220  if (m_uiSlimeSprayTimer < uiDiff)
221  {
222  if (DoCastSpellIfCan(m_creature, SPELL_SLIME_SPRAY_SUMMON) == CAST_OK)
223  m_uiSlimeSprayTimer = urand(20000, 25000);
224  }
225  else
226  m_uiSlimeSprayTimer -= uiDiff;
227 
228  // Mutated Infection - faster with time
229  // implemented this instead of phases
230  if (m_uiInfectionsRate < MAX_MUTATE_INFACTION_STEPS)
231  {
232  if (m_uiMutatedInfectionTimer <= uiDiff)
233  {
234  if (DoCastSpellIfCan(m_creature, uiMutatedInfections[m_uiInfectionsRate], CAST_TRIGGERED) == CAST_OK)
235  {
236  m_creature->RemoveAurasDueToSpell(uiMutatedInfections[m_uiInfectionsRate - 1]);
237  // every next 15 seconds faster
238  m_uiMutatedInfectionTimer = 60000 - m_uiInfectionsRate * 15000;
239  ++m_uiInfectionsRate;
240  }
241  }
242  else
243  m_uiMutatedInfectionTimer -= uiDiff;
244  }
245 
246  // Heroic spells
247  if (m_pInstance->IsHeroicDifficulty())
248  {
249  if (m_uiVileGasTimer < uiDiff)
250  {
251  if (DoCastSpellIfCan(m_creature, SPELL_VILE_TRIGGER) == CAST_OK)
252  m_uiVileGasTimer = 30000;
253  }
254  else
255  m_uiVileGasTimer -= uiDiff;
256  }
257 
258  DoMeleeAttackIfReady();
259  }
260 };

381 
382  if (lTargetsInRange.empty())

385  m_uiUnstableExplosionTimer = 0;
386  return;
387  }
388 
389  // explode all the summoned triggers
390  for (CreatureList::const_iterator itr = lTargetsInRange.begin(); itr != lTargetsInRange.end(); ++itr)

396  m_uiUnstableExplosionTimer = 0;

420 bool EffectScriptEffectCreature_spell_unstable_ooze(Unit* pCaster, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
421 {
422  if (uiSpellId == SPELL_UNSTABLE_OOZE && uiEffIndex == EFFECT_INDEX_2 && pCreatureTarget->GetEntry() == NPC_BIG_OOZE)
423  {
424  // send AI event on 5 stacks of unstable Ooze
425  SpellAuraHolder* pOozeHolder = pCreatureTarget->GetSpellAuraHolder(uiSpellId);
426  if (pOozeHolder)
427  {
428  // Note: stacks are increased after the effect is processed, so we need to use (stacks - 1)
429  switch (pOozeHolder->GetStackAmount())
430  {
431  case 1:
432  DoScriptText(EMOTE_OOZE_GROW_1, pCreatureTarget);
433  break;
434  case 2:
435  DoScriptText(EMOTE_OOZE_GROW_2, pCreatureTarget);
436  break;
437  case 4:
438  pCreatureTarget->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, pCaster, pCreatureTarget);
439  // no break;
440  case 3:
441  DoScriptText(EMOTE_OOZE_GROW_3, pCreatureTarget);
442  break;
443  }
444  }
445 
446  return true;
447  }
448 
449  return false;
450 }
