69 bool VisibilityData::CanDetectInvisibilityOf(WorldObject const* target) const
70 {
71  if (uint32 mask = (GetInvisibilityDetectMask() & target->GetVisibilityData().GetInvisibilityMask()))
72  {
73  for (int32 i = 0; i < INVISIBILITY_TYPE_MAX; ++i)
74  {
75  if (((1 << i) & mask) == 0)
76  continue;
77 
78  // find invisibility level
79  int32 invLevel = GetInvisibilityValue(i);
80 
81  // find invisibility detect level - this is taken from controlling player or self
82  int32 detectLevel = GetInvisibilityDetectValue(i);
83 
84  if (invLevel <= detectLevel)
85  return true;
86  }
87  }
88 
89  return false;
90 }

157 float VisibilityData::GetStealthVisibilityDistance(Unit const* target, bool alert) const
158 {
159  // Starting points
160  int32 detectionValue = 30;
161 
162  // Level difference: 5 point / level, starting from level 1.
163  // There may be spells for this and the starting points too, but
164  // not in the DBCs of the client.
165  detectionValue += int32(target->GetLevelForTarget(dynamic_cast<Unit*>(m_owner)) - 1) * 5;
166 
167  // Apply modifiers
168  detectionValue += target->GetVisibilityData().GetStealthDetectionStrength(STEALTH_UNIT);
169  if (m_owner->GetTypeId() == TYPEID_GAMEOBJECT) // trap case
170  if (Unit* owner = static_cast<GameObject*>(m_owner)->GetOwner())
171  detectionValue -= int32(owner->GetLevelForTarget(dynamic_cast<Unit*>(m_owner)) - 1) * 5;
172 
173  detectionValue -= GetStealthStrength(STEALTH_UNIT);
174 
175  // Calculate max distance
176  float visibilityRange = float(detectionValue) * 0.3f;
177 
178  // If this unit is an NPC then player detect range doesn't apply
179  if (target->GetTypeId() == TYPEID_PLAYER && visibilityRange > MAX_PLAYER_STEALTH_DETECT_RANGE)
180  visibilityRange = MAX_PLAYER_STEALTH_DETECT_RANGE;
181 
182  // When checking for alert state, look 8% further, and then 1.5 yards more than that.
183  if (alert)
184  visibilityRange += (visibilityRange * 0.08f) + 1.5f;
185 
186  return std::max(target->GetCombatReach(), visibilityRange);
187 }
