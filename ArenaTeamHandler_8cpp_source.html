28 void WorldSession::HandleInspectArenaTeamsOpcode(WorldPacket& recv_data)
29 {
30  DEBUG_LOG("MSG_INSPECT_ARENA_TEAMS");
31 
32  ObjectGuid guid;
33  recv_data >> guid;
34  DEBUG_LOG("Inspect Arena stats %s", guid.GetString().c_str());
35 
36  Player* player = sObjectMgr.GetPlayer(guid);
37  if (!player)
38  return;
39 
40  if (!_player->IsWithinDistInMap(player, INSPECT_DISTANCE, false))
41  return;
42 
43  if (_player->CanAttack(player))
44  return;
45 
46  for (uint8 i = 0; i < MAX_ARENA_SLOT; ++i)
47  {
48  if (uint32 a_id = player->GetArenaTeamId(i))
49  {
50  if (ArenaTeam* arenaTeam = sObjectMgr.GetArenaTeamById(a_id))
51  arenaTeam->InspectStats(this, player->GetObjectGuid());
52  }
53  }
54 }

81 void WorldSession::HandleArenaTeamInviteOpcode(WorldPacket& recv_data)
82 {
83  DEBUG_LOG("CMSG_ARENA_TEAM_INVITE");
84 
85  uint32 ArenaTeamId; // arena team id
86  std::string Invitedname;
87 
88  Player* player = nullptr;
89 
90  recv_data >> ArenaTeamId >> Invitedname;
91 
92  if (!Invitedname.empty())
93  {
94  if (!normalizePlayerName(Invitedname))
95  return;
96 
97  player = ObjectAccessor::FindPlayerByName(Invitedname.c_str());
98  }
99 
100  if (!player)
101  {
102  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", Invitedname, ERR_ARENA_TEAM_PLAYER_NOT_FOUND_S);
103  return;
104  }
105 
106  if (player->GetLevel() < sWorld.getConfig(CONFIG_UINT32_MAX_PLAYER_LEVEL))
107  {
108  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", player->GetName(), ERR_ARENA_TEAM_TARGET_TOO_LOW_S);
109  return;
110  }
111 
112  ArenaTeam* arenateam = sObjectMgr.GetArenaTeamById(ArenaTeamId);
113  if (!arenateam)
114  {
115  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_PLAYER_NOT_IN_TEAM);
116  return;
117  }
118 
119  // OK result but not send invite
120  if (player->GetSocial()->HasIgnore(GetPlayer()->GetObjectGuid()))
121  return;
122 
123  if (!sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_GUILD) && player->GetTeam() != GetPlayer()->GetTeam())
124  {
125  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
126  return;
127  }
128 
129  if (player->GetArenaTeamId(arenateam->GetSlot()))
130  {
131  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", player->GetName(), ERR_ALREADY_IN_ARENA_TEAM_S);
132  return;
133  }
134 
135  if (player->GetArenaTeamIdInvited())
136  {
137  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", player->GetName(), ERR_ALREADY_INVITED_TO_ARENA_TEAM_S);
138  return;
139  }
140 
141  if (arenateam->GetMembersSize() >= arenateam->GetMaxMembersSize())
142  {
143  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, arenateam->GetName(), "", ERR_ARENA_TEAM_TOO_MANY_MEMBERS_S);
144  return;
145  }
146 
147  DEBUG_LOG("Player %s Invited %s to Join his ArenaTeam", GetPlayer()->GetName(), Invitedname.c_str());
148 
149  player->SetArenaTeamIdInvited(arenateam->GetId());
150 
151  WorldPacket data(SMSG_ARENA_TEAM_INVITE, (8 + 10));
152  data << GetPlayer()->GetName();
153  data << arenateam->GetName();
154  player->GetSession()->SendPacket(data);
155 
156  DEBUG_LOG("WORLD: Sent SMSG_ARENA_TEAM_INVITE");
157 }

159 void WorldSession::HandleArenaTeamAcceptOpcode(WorldPacket& /*recv_data*/)
160 {
161  DEBUG_LOG("CMSG_ARENA_TEAM_ACCEPT"); // empty opcode
162 
163  ArenaTeam* at = sObjectMgr.GetArenaTeamById(_player->GetArenaTeamIdInvited());
164  if (!at)
165  return;
166 
167  if (_player->GetArenaTeamId(at->GetSlot()))
168  {
169  // already in arena team that size
170  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ALREADY_IN_ARENA_TEAM);
171  return;
172  }
173 
174  if (!sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_GUILD) &&
175  _player->GetTeam() != sObjectMgr.GetPlayerTeamByGUID(at->GetCaptainGuid()))
176  {
177  // not let enemies sign petition
178  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
179  return;
180  }
181 
182  if (!at->AddMember(_player->GetObjectGuid()))
183  {
184  // arena team not found
185  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", "", ERR_ARENA_TEAM_INTERNAL);
186  return;
187  }
188 
189  // event
190  at->BroadcastEvent(ERR_ARENA_TEAM_JOIN_SS, _player->GetObjectGuid(), _player->GetName(), at->GetName().c_str());
191 }

200 void WorldSession::HandleArenaTeamLeaveOpcode(WorldPacket& recv_data)
201 {
202  DEBUG_LOG("CMSG_ARENA_TEAM_LEAVE");
203 
204  uint32 ArenaTeamId; // arena team id
205  recv_data >> ArenaTeamId;
206 
207  ArenaTeam* at = sObjectMgr.GetArenaTeamById(ArenaTeamId);
208  if (!at)
209  return;
210 
211  if (_player->InBattleGroundQueue() || _player->InArena())
212  return;
213 
214  if (_player->GetObjectGuid() == at->GetCaptainGuid() && at->GetMembersSize() > 1)
215  {
216  // check for correctness
217  SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, "", "", ERR_ARENA_TEAM_LEADER_LEAVE_S);
218  return;
219  }
220 
221  // arena team has only one member (=captain)
222  if (_player->GetObjectGuid() == at->GetCaptainGuid())
223  {
224  at->Disband(this);
225  delete at;
226  return;
227  }
228 
229  at->DelMember(_player->GetObjectGuid());
230 
231  // event
232  at->BroadcastEvent(ERR_ARENA_TEAM_LEAVE_SS, _player->GetObjectGuid(), _player->GetName(), at->GetName().c_str());
233 
234  // send you are no longer member of team
235  SendArenaTeamCommandResult(ERR_ARENA_TEAM_QUIT_S, at->GetName(), "", 0);
236 }
