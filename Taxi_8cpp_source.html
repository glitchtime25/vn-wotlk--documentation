64 bool Tracker::Load(std::string& string, DestID &destOrphan)
65 {
66  if (!Clear())
67  return false;
68 
69  // Read a modified format. All old entries were converted to orphaned taxi node records via DB update for failsafe loading.
70 
71  // Original numbers format: "src dest dest dest " variable amount of destinations in the string, dangling destinations possible, trailing space @ the end)
72  // Updated numbers format: "resumenode:src:dest:dest:dest" strictly formatted, introduced in taxi system rewrite
73  // Modified numbers format: "resumenode#pathid#pathid#pathid" strictly formatted, directly uses flights, introduced in an update for redesigned taxi system
74 
75  Tokens tokens = StrSplit(string, "#");
76 
77  std::deque<PathID> numbers;
78 
79  for (auto& token : tokens)
80  numbers.push_back(uint32(std::stoi(token)));
81 
82  // Try to parse as robustly as possible
83  switch (numbers.size())
84  {
85  case 0: // No routes
86  return true;
87  case 1: // Orphaned: probably loading from an outdated format, just dangling destination node
88  destOrphan = numbers.front();
89  return false;
90  default:
91  {
92  // Extract and verify last known flight path node index
93  uint32 nodeResume = numbers.front();
94  numbers.pop_front();
95 
96  // On successful load, prepares the container right away, otherwise does all the housekeeping
97  if (!AddRoutes(numbers, 0.0f, false) || !Prepare(nodeResume))
98  {
99  Clear();
100  return false;
101  }
102  }
103  }

124 bool Tracker::AddRoute(const TaxiPathEntry *entry, float discountMulti /*= 0.0f*/, bool requireModel /*= true*/)
125 {
126  // Cant be altered while taxi ride is prepared
127  if (m_state > TRACKER_STAGING)
128  return false;
129 
130  // Verify input
131  if (!entry)
132  return false;
133 
134  uint32 cost = entry->price;
135 
136  const bool commercial = int32(discountMulti) != 0;
137 
138  // Can't add taxi path without mount display id unless specified otherwise
139  uint32 displayId = sObjectMgr.GetTaxiMountDisplayId(entry->from, m_owner.GetTeam());
140  if (requireModel && !displayId)
141  return false;
142 
143  // Use first route's mount appearance for the entirety of the flight
144  if (!m_displayId)
145  m_displayId = displayId;
146 
147  if (commercial)
148  {
149  cost = uint32(cost * discountMulti);
150  m_cost += cost;
151  // If global preference is set, do not record individual flight costs
152  if (sWorld.getConfig(CONFIG_BOOL_LONG_TAXI_PATHS_PERSISTENCE))
153  cost = 0;
154  }
155 
156  TaxiPathNodeList const& nodes = sTaxiPathNodesByPath[entry->ID];
157  m_routes.push_back(Route(entry->ID, entry->from, entry->to, nodes.front()->index, nodes.back()->index, cost));
158 
159  const size_t count = m_routes.size();
160 
161  // Multi-route taxi path: set shortcut
162  if (count > 1)
163  Trim(m_routes[count - 2], m_routes[count - 1]);
164 
165  return true;
166 }

223 bool Tracker::Prepare(Index nodeResume /*= 0*/)
224 {
225  // Cant be altered while taxi ride is prepared
226  if (m_state > TRACKER_STAGING)
227  return false;
228 
229  // Cant be finalized when no routes added
230  if (m_routes.empty())
231  return false;
232 
233  Route& current = m_routes.front();
234 
235  // If global preference is set, we will charge total cost for the entire ride on first flight at once
236  if (sWorld.getConfig(CONFIG_BOOL_LONG_TAXI_PATHS_PERSISTENCE))
237  current.cost = m_cost;
238 
239  // Sanity check: make sure that resume node for current route hasn't gotten cut off or out of bounds
240  nodeResume = std::min(nodeResume, current.nodeEnd);
241 
242  m_atlas.clear();
243  // Open a new map in the atlas
244  m_atlas.push_back(Map());
245  // Start populating map(s)
246  const TaxiPathNodeEntry* prev = nullptr;
247  for (auto i = m_routes.begin(); i != m_routes.end(); ++i)
248  {
249  const TaxiPathNodeList& nodes = sTaxiPathNodesByPath[(*i).pathID];
250  for (auto j = nodes.begin(); j != nodes.end(); ++j)
251  {
252  // Abide route trimmed routes when building a spline
253  if ((*j)->index > (*i).nodeEnd || (*j)->index < (*i).nodeStart)
254  continue;
255 
256  // Resume node defined: we are loading, additional care is required
257  if (nodeResume && (*i).pathID == current.pathID)
258  {
259  // Internal algorithm sanity check: make sure resume node for current route will end up on the same map
260  // Dont build spline for preceding nodes which are located on another map
261  if ((*j)->index < nodes[nodeResume]->index && (*j)->mapid != nodes[nodeResume]->mapid)
262  continue;
263  // When resume node reached on the route: calculate resume location index on the spline (which will be equal to the current size on push)
264  if ((*j)->index == nodeResume)
265  {
266  m_resumeIndex = m_atlas.back().size();
267  m_locationIndex = m_resumeIndex;
268  }
269  }
270  // Detect level change and insert a new map
271  if ((*j)->index >= (*i).nodeStart && (*j)->index <= (*i).nodeEnd)
272  {
273  if (prev && (*j)->mapid != prev->mapid)
274  {
275  // Detect map change and advance atlas by adding a new map
276  if (sMapStore.LookupEntry((*j)->mapid))
277  {
278  // Bugcheck: latest finished map spline is suspiciously short
279  MANGOS_ASSERT(m_atlas.back().size() > 2);
280  m_atlas.push_back(Map());
281  }
282  else
283  {
284  sLog.outError("TAXI: Malformed flight path node detected and ignored. Node %u on flight path id %u refers non-existent mapid %u",
285  (*j)->index, (*j)->path, (*j)->mapid);
286  continue;
287  }
288  }
289  }
290  // Add entry to the latest Map
291  m_atlas.back().push_back((*j));
292  if ((*j)->index > (*i).nodeStart)
293  prev = (*j);
294  }
295  }
296  // Bugcheck: latest finished map spline is suspiciously short
297  MANGOS_ASSERT(m_atlas.back().size() > 2);
298 
299  m_state = TRACKER_STANDBY;
300  return true;
301 }

319 bool Tracker::UpdateRoute(const TaxiPathNodeEntry* entry, PathID& start, PathID& end)
320 {
321  if (!entry)
322  return false;
323 
324  // Cant be advanced when no taxi flight is being tracked
325  if (m_state != TRACKER_FLIGHT)
326  return false;
327 
328  // Bugcheck: continuity error - either routes or splines are missing
329  MANGOS_ASSERT(!m_atlas.empty() && !m_routes.empty())
330 
331  const Map& map = GetMap();
332 
333  // Advance internal spline node counter only if not on the first waypoint
334  if (m_location)
335  {
336  ++m_locationIndex;
337  m_resumeIndex = m_locationIndex;
338  }
339 
340  m_location = entry;
341 
342  // Bugcheck: internal spline node counter overflow
343  MANGOS_ASSERT(size_t(m_locationIndex) < map.size())
344 
345  if (entry == map.back())
346  {
347  // Acknowledge spline end: nuke current map in the atlas, reset location index
348  // That does not mean, however, that taxi ride was completed right away, as it can be a multimap flight route
349  m_atlas.pop_front();
350  m_location = nullptr;
351  m_locationIndex = 0;
352  m_resumeIndex = 0;
353  }
354 
355  const Route& route = GetRoute();
356 
357  if (entry->index == route.nodeEnd)
358  {
359  // Acknowledge the end of the current route to other systems
360  end = route.pathID;
361  // Nuke the route from the container
362  m_cost -= route.cost;
363  m_routes.pop_front();
364  if (m_routes.empty())
365  {
366  m_state = TRACKER_EMPTY;
367  m_displayId = 0;
368  }
369  return true;
370  }
371  if (entry->index == route.nodeStart)
372  {
373  // Acknowledge the start of the current route to other systems
374  start = route.pathID;
375  return true;
376  }
377 
378  return false;
379 }

381 bool Tracker::Trim(Route& first, Route& second)
382 {
383  if (first.destEnd != second.destStart || first.pathID == second.pathID)
384  return false;
385 
386  // Retail uses serverside pre-definied junction points for all interconnected flight paths.
387  // We have a fallback custom automatic runtime trimmer. The result will not match retail and can be visually unpleasing at times.
388  // For best experience, the taxi shortcuts table needs to be populated with faithful data.
389 
390  // Lookup known shortcuts first
391  uint32 lengthTakeoff = 0;
392  uint32 lengthLanding = 0;
393 
394  TaxiShortcutData data;
395 
396  // Get the landing range of the first route if exists
397  if (sObjectMgr.GetTaxiShortcut(first.pathID, data))
398  lengthLanding = data.lengthLanding;
399 
400  // Get the takeoff range of the second route if exists
401  if (sObjectMgr.GetTaxiShortcut(second.pathID, data))
402  lengthTakeoff = data.lengthTakeoff;
403 
404  // Sanity check for bounds
405  lengthLanding = std::min(lengthLanding, (first.nodeEnd - 1));
406  lengthTakeoff = std::min(lengthTakeoff, (second.nodeEnd - 1));
407 
408  // Both landing and takeoff data is present
409  if (lengthLanding && lengthTakeoff)
410  {
411  first.nodeEnd -= lengthLanding;
412  second.nodeStart += lengthTakeoff;
413  return true;
414  }
415 
416  // Try to trim dynamically if data is not available or incomplete
417  TaxiPathNodeList const& waypoints1 = sTaxiPathNodesByPath[first.pathID];
418  TaxiPathNodeList const& waypoints2 = sTaxiPathNodesByPath[second.pathID];
419  TaxiNodesEntry const* destination = sTaxiNodesStore.LookupEntry(first.destEnd);
420 
421  if (destination && waypoints1 != waypoints2)
422  {
423  // Linear complexity for better performance
424  const TaxiPathNodeEntry* last1 = nullptr;
425  const TaxiPathNodeEntry* last2 = nullptr;
426  const double refdistsq = double(48.0f * 48.0f);
427  auto i1 = (waypoints1.rbegin() + lengthLanding);
428  auto i2 = (waypoints2.begin() + lengthTakeoff);
429  for (; (i1 != waypoints1.rend() && i2 != waypoints2.end()); (++i1, ++i2))
430  {
431  // Verify that both paths are on the same map
432  if ((*i1)->mapid != (*i2)->mapid)
433  break;
434  // Verify that there are no transitions between maps on either end
435  if ((last1 && last1->mapid != (*i1)->mapid) || (last2 && last2->mapid != (*i2)->mapid))
436  break;
437  // Quick check for distance to destination on 2d plane
438  auto distsq = [&destination] (TaxiPathNodeEntry const* node) {
439  return (std::pow((destination->x - node->x), 2) + std::pow((destination->y - node->y), 2));
440  };
441  if (distsq(*i1) > refdistsq && distsq(*i2) > refdistsq)
442  {
443  first.nodeEnd = (*i1)->index;
444  second.nodeStart = (*i2)->index;
445  return true;
446  }
447  last1 = (*i1);
448  last2 = (*i2);
449  }
450  }
451  return false;
452 }
