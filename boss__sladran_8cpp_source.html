96  void JustSummoned(Creature* summoned) override
97  {
98  if (summoned->GetEntry() != NPC_SLADRAN_CONSTRICTOR && summoned->GetEntry() != NPC_SLADRAN_VIPER)
99  return;
100 
101  summoned->SetWalk(false);
102 
103  if (summoned->GetEntry() == NPC_SLADRAN_VIPER)
104  {
105  summoned->SetInCombatWithZone();
106  return;
107  }
108 
109  if (summoned->AI())
110  summoned->AI()->SetReactState(REACT_DEFENSIVE);
111 
112  summoned->SetInCombatWithZone();
113  summoned->AI()->AttackClosestEnemy();
114  }

116  std::chrono::milliseconds GetSubsequentActionTimer(SladranActions action)
117  {
118  switch (action)
119  {
120  case SLADRAN_SUMMON: return m_isRegularMode ? 5s : 3s;
121  default: return 0s;
122  }
123  }
124 
125  void ExecuteAction(uint32 action) override
126  {
127  switch (action)
128  {
129  case SLADRAN_HEALTH_CHECK:
130  {
131  if (m_creature->GetHealthPercent() <= 90.f)
132  {
133  ResetCombatAction(SLADRAN_SUMMON, 0s);
134  DisableCombatAction(SLADRAN_HEALTH_CHECK);
135  }
136  return;
137  }
138  case SLADRAN_SUMMON:
139  {
140  if (!m_instance)
141  return;
142 
143  if (Creature* summonTarget = m_creature->GetMap()->GetCreature(m_instance->SelectRandomSladranTargetGuid()))
144  {
145  if (summonTarget->GetPositionZ() > 132)
146  {
147  // we don't want to get spammed
148  if (!urand(0, 4))
149  DoBroadcastText(SAY_SUMMON_CONSTRICTOR, m_creature);
150 
151  summonTarget->CastSpell(nullptr, SPELL_SUMMON_CONSTRICTOR, TRIGGERED_NONE, nullptr, nullptr, m_creature->GetObjectGuid());
152  }
153  else
154  {
155  // we don't want to get spammed
156  if (!urand(0, 4))
157  DoBroadcastText(SAY_SUMMON_SNAKE, m_creature);
158 
159  summonTarget->CastSpell(nullptr, SPELL_SUMMON_VIPER, TRIGGERED_NONE, nullptr, nullptr, m_creature->GetObjectGuid());
160  }
161  }
162  break;
163  }
164  }
165  ResetCombatAction(action, GetSubsequentActionTimer(SladranActions(action)));
166  }
