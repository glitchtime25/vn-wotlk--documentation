299 bool ProcessEventId_event_go_ulduar_tower(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool /*bIsStart*/)
300 {
301  if (pSource->GetTypeId() == TYPEID_GAMEOBJECT && ((GameObject*)pSource)->GetGoType() == GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING)
302  {
303  instance_ulduar* pInstance = (instance_ulduar*)((GameObject*)pSource)->GetInstanceData();
304  if (!pInstance)
305  return true;
306 
307  // Towers can be deactivated by destroying them. Notify instance data in case they get destroyed.
308  switch (uiEventId)
309  {
310  case EVENT_ID_TOWER_LIFE:
311  pInstance->SetData(TYPE_TOWER_FREYA, FAIL);
312  break;
313  case EVENT_ID_TOWER_FLAME:
314  pInstance->SetData(TYPE_TOWER_MIMIRON, FAIL);
315  break;
316  case EVENT_ID_TOWER_FROST:
317  pInstance->SetData(TYPE_TOWER_HODIR, FAIL);
318  break;
319  case EVENT_ID_TOWER_STORMS:
320  pInstance->SetData(TYPE_TOWER_THORIM, FAIL);
321  break;
322  default:
323  return false;
324  }
325 
326  // despawn all generators in range
327  CreatureList lGenerators;
328  GetCreatureListWithEntryInGrid(lGenerators, (GameObject*)pSource, NPC_GENERATOR_SMALL, 100.0f);
329  for (auto& lGenerator : lGenerators)
330  lGenerator->ForcedDespawn();
331 
332  // allow further DB processing
333  return false;
334  }
335 
336  return false;
337 }

418  {
419  if (DoCastSpellIfCan(m_creature, SPELL_SEPARATION_ANXIETY) == CAST_OK)
420  m_uiCheckBuddyTimer = 5000;

425  m_uiCheckBuddyTimer = 0;
426  }
427  else
428  m_uiCheckBuddyTimer = 1000;
429  }
430  else
431  m_uiCheckBuddyTimer -= uiDiff;
432 
433  // spawn a sphere only if the buddy is stil alive
434  if (m_uiSphereTimer < uiDiff)
435  {
436  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_CHARGED_SPHERE) == CAST_OK)
437  m_uiSphereTimer = urand(20000, 35000);
438  }
439  else
440  m_uiSphereTimer -= uiDiff;
441  }
442

444  {
445  if (DoCastSpellIfCan(m_creature, SPELL_FORKED_LIGHTNING) == CAST_OK)
446  m_uiLightningTimer = urand(10000, 15000);
447  }
448  else
449  m_uiLightningTimer -= uiDiff;
450 
451  DoMeleeAttackIfReady();
452  }
