106 bool RASocket::HandleInput()
107 {
108  auto const minLevel = static_cast<AccountTypes>(sConfig.GetIntDefault("RA.MinLevel", AccountTypes::SEC_ADMINISTRATOR));
109 
110  switch (m_authLevel)
111  {
113  case AuthLevel::None:
114  {
115  m_accountId = sAccountMgr.GetId(m_input);
116 
118  if (!m_accountId)
119  {
120  Send("-No such user.\r\n");
121  sLog.outRALog("User %s does not exist.", m_input.c_str());
122 
123  if (m_secure)
124  return false;
125 
126  Send("\r\n");
127  Send(sObjectMgr.GetMangosStringForDbcLocale(LANG_RA_USER));
128  break;
129  }
130 
131  m_accountLevel = sAccountMgr.GetSecurity(m_accountId);
132 
134  if (m_accountLevel < minLevel)
135  {
136  Send("-Not enough privileges.\r\n");
137  sLog.outRALog("User %s has no privilege.", m_input.c_str());
138 
139  if (m_secure)
140  return false;
141 
142  Send("\r\n");
143  Send(sObjectMgr.GetMangosStringForDbcLocale(LANG_RA_USER));
144  break;
145  }
146 
148  if (m_accountLevel >= SEC_ADMINISTRATOR && !m_restricted)
149  m_accountLevel = SEC_CONSOLE;
150 
151  m_authLevel = AuthLevel::HaveUsername;
152  Send(sObjectMgr.GetMangosStringForDbcLocale(LANG_RA_PASS));
153  break;
154  }
156  case AuthLevel::HaveUsername:
157  {
158  // login+pass ok
159  if (sAccountMgr.CheckPassword(m_accountId, m_input))
160  {
161  m_authLevel = AuthLevel::Authenticated;
162 
163  Send("+Logged in.\r\n");
164  sLog.outRALog("User account %u has logged in.", m_accountId);
165  Send("mangos>");
166  }
167  else
168  {
170  Send("-Wrong pass.\r\n");
171  sLog.outRALog("User account %u has failed to log in.", m_accountId);
172 
173  if (m_secure)
174  return false;
175 
176  Send("\r\n");
177  Send(sObjectMgr.GetMangosStringForDbcLocale(LANG_RA_PASS));
178  }
179  break;
180  }
182  case AuthLevel::Authenticated:
183  {
184  if (m_input.length())
185  {
186  sLog.outRALog("Got '%s' cmd.", m_input.c_str());
187 
188  if (m_input == "quit")
189  return false;
190 
191  sWorld.QueueCliCommand(new CliCommandHolder(m_accountId, m_accountLevel, m_input.c_str(),
192  [this](const char* buffer) { this->Send(buffer); },
193  [this](bool) { this->Send("mangos>"); }));
194  }
195  else
196  Send("mangos>");
197 
198  break;
199  }
200 
201  default:
202  return false;
204  }
205 
206  m_input.clear();
207 
208  return true;
209 }
