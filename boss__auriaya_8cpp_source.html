135  void UpdateAI(const uint32 uiDiff) override
136  {
137  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
138  return;
139 
140  if (m_uiTerrifyingScreechTimer < uiDiff)
141  {
142  if (DoCastSpellIfCan(m_creature, SPELL_TERRIFYING_SCREECH) == CAST_OK)
143  {
144  DoScriptText(EMOTE_SCREECH, m_creature);
145  m_uiTerrifyingScreechTimer = 35000;
146  m_uiSentinelBlastTimer = 2000;
147  }
148  }
149  else
150  m_uiTerrifyingScreechTimer -= uiDiff;
151 
152  if (m_uiSentinelBlastTimer < uiDiff)
153  {
154  // Cast Sentinel blast right after terrifying screech
155  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SENTINEL_BLAST : SPELL_SENTINEL_BLAST_H) == CAST_OK)
156  m_uiSentinelBlastTimer = 35000;
157  }
158  else
159  m_uiSentinelBlastTimer -= uiDiff;
160 
161  if (m_uiDefenderTimer)
162  {
163  if (m_uiDefenderTimer <= uiDiff)
164  {
165  // Summon the feral defender trigger
166  if (DoCastSpellIfCan(m_creature, SPELL_ACTIVATE_FERAL_DEFENDER_TRIGG) == CAST_OK)
167  m_uiDefenderTimer = 0;
168  }
169  else
170  m_uiDefenderTimer -= uiDiff;
171  }
172 
173  if (m_uiSonicScreechTimer < uiDiff)
174  {
175  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SONIC_SCREECH : SPELL_SONIC_SCREECH_H) == CAST_OK)
176  m_uiSonicScreechTimer = 27000;
177  }
178  else
179  m_uiSonicScreechTimer -= uiDiff;
180 
181  if (m_uiSwarmTimer < uiDiff)
182  {
183  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
184  {
185  if (DoCastSpellIfCan(pTarget, SPELL_GUARDIAN_SWARM) == CAST_OK)
186  m_uiSwarmTimer = 37000;
187  }
188  }
189  else
190  m_uiSwarmTimer -= uiDiff;
191 
192  if (m_uiEnrageTimer < uiDiff)
193  {
194  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
195  {
196  DoScriptText(SAY_BERSERK, m_creature);
197  m_uiEnrageTimer = 10 * MINUTE * IN_MILLISECONDS;
198  }
199  }
200  else
201  m_uiEnrageTimer -= uiDiff;
202 
203  DoMeleeAttackIfReady();
204  }

256  void DamageTaken(Unit* /*pDealer*/, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
257  {
258  // If we don't have the feral essence anymore then ignore this
259  if (m_uiKilledCount >= 8) // 9-1 == 8
260  return;
261 
262  if (m_uiReviveDelayTimer) // Already faking
263  {
264  uiDamage = 0;
265  return;
266  }
267 
268  if (uiDamage >= m_creature->GetHealth())
269  {
270  uiDamage = 0;
271 
272  // Set Feign death, remove one aura stack and summon a feral essence
273  DoCastSpellIfCan(m_creature, SPELL_FEIGN_DEATH, CAST_TRIGGERED);
274  DoCastSpellIfCan(m_creature, SPELL_FERAL_ESSENCE_REMOVAL, CAST_TRIGGERED);
275  DoCastSpellIfCan(m_creature, SPELL_SEEPING_FERAL_ESSENCE_SUMMON, CAST_TRIGGERED);
276 
277  // the feign death aura doesn't do everything, so keep the following here as well
278  m_creature->SetHealth(0);
279  m_creature->ClearComboPointHolders();
280  m_creature->ModifyAuraState(AURA_STATE_HEALTHLESS_20_PERCENT, false);
281  m_creature->ModifyAuraState(AURA_STATE_HEALTHLESS_35_PERCENT, false);
282  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
283  m_creature->ClearAllReactives();
284  m_creature->GetMotionMaster()->Clear();
285  m_creature->GetMotionMaster()->MoveIdle();
286  m_creature->SetStandState(UNIT_STAND_STATE_DEAD);
287 
288  m_uiReviveDelayTimer = 30000;
289  ++m_uiKilledCount;
290  }
291  }

300  void UpdateAI(const uint32 uiDiff) override
301  {
302  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
303  return;
304 
305  if (m_uiReviveDelayTimer)
306  {
307  if (m_uiReviveDelayTimer <= uiDiff)
308  {
309  // ToDo: figure out how to enable the ressurrect animation
310  DoCastSpellIfCan(m_creature, SPELL_FULL_HEAL, CAST_TRIGGERED);
311  m_uiReviveDelayTimer = 0;
312  m_uiAttackDelayTimer = 3000;
313  }
314  else
315  m_uiReviveDelayTimer -= uiDiff;
316 
317  // No Further action while faking
318  return;
319  }
320 
321  if (m_uiAttackDelayTimer)
322  {
323  if (m_uiAttackDelayTimer <= uiDiff)
324  {
325  DoResetThreat();
326  m_creature->RemoveAurasDueToSpell(SPELL_FEIGN_DEATH);
327  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
328  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
329 
330  m_uiPounceTimer = urand(9000, 10000);
331  m_uiFeralRushCount = 0;
332  m_uiFeralRushTimer = 1000;
333  m_uiAttackDelayTimer = 0;
334  }
335  else
336  m_uiAttackDelayTimer -= uiDiff;
337 
338  // No Further action while faking
339  return;
340  }
341 
342  if (m_uiPounceTimer < uiDiff)
343  {
344  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
345  {
346  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_FERAL_POUNCE : SPELL_FERAL_POUNCE_H) == CAST_OK)
347  m_uiPounceTimer = urand(13000, 16000);
348  }
349  }
350  else
351  m_uiPounceTimer -= uiDiff;
352 
353  if (m_uiFeralRushTimer < uiDiff)
354  {
355  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_FERAL_RUSH : SPELL_FERAL_RUSH_H) == CAST_OK)
356  {
357  ++m_uiFeralRushCount;
358 
359  if (m_uiFeralRushCount == m_uiMaxFeralRush)
360  {
361  m_uiFeralRushCount = 0;
362  m_uiFeralRushTimer = 12000;
363  }
364  else
365  m_uiFeralRushTimer = 400;
366  }
367  }
368  else
369  m_uiFeralRushTimer -= uiDiff;
370 
371  DoMeleeAttackIfReady();
372  }
