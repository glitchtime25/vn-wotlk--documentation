163  void UpdateAI(const uint32 uiDiff) override
164  {
165  if (m_uiShieldTimer)
166  {
167  if (m_uiShieldTimer <= uiDiff)
168  {
169  if (!m_pInstance)
170  return;
171 
172  // Check for shield first
173  if (m_pInstance->IsShieldBroken())
174  {
175  m_uiShieldTimer = 0;
176  return;
177  }
178 
179  // spell will set the proper immunity flags
180  if (DoCastSpellIfCan(m_creature, SPELL_CENTRIFUGE_SHIELD) == CAST_OK)
181  m_uiShieldTimer = 0;
182  }
183  else
184  m_uiShieldTimer -= uiDiff;
185  }
186 
187  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
188  return;
189 
190  if (m_uiAmplifyMagicTimer < uiDiff)
191  {
192  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
193  {
194  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_AMPLIFY_MAGIC : SPELL_AMPLIFY_MAGIC_H) == CAST_OK)
195  m_uiAmplifyMagicTimer = urand(15000, 20000);
196  }
197  }
198  else
199  m_uiAmplifyMagicTimer -= uiDiff;
200 
201  if (m_uiEnergizeCoresTimer < uiDiff)
202  {
203  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ENERGIZE_CORES : SPELL_ENERGIZE_CORES_H) == CAST_OK)
204  m_uiEnergizeCoresTimer = urand(5000, 7000);
205  }
206  else
207  m_uiEnergizeCoresTimer -= uiDiff;
208 
209  if (m_uiCallCaptainTimer < uiDiff)
210  {
211  // choose a random captain spell
212  uint32 uiSpellId = 0;
213  switch (urand(0, 3))
214  {
215  case 0: uiSpellId = SPELL_CALL_CAPTAIN_1; break;
216  case 1: uiSpellId = SPELL_CALL_CAPTAIN_2; break;
217  case 2: uiSpellId = SPELL_CALL_CAPTAIN_3; break;
218  case 3: uiSpellId = SPELL_CALL_CAPTAIN_4; break;
219  }
220 
221  if (DoCastSpellIfCan(m_creature, uiSpellId) == CAST_OK)
222  {
223  switch (urand(0, 2))
224  {
225  case 0: DoScriptText(SAY_CALL_CAPTAIN_1, m_creature); break;
226  case 1: DoScriptText(SAY_CALL_CAPTAIN_2, m_creature); break;
227  case 2: DoScriptText(SAY_CALL_CAPTAIN_3, m_creature); break;
228  }
229 
230  DoScriptText(EMOTE_CAPTAIN, m_creature);
231  m_uiCallCaptainTimer = urand(13000, 23000);
232  }
233  }
234  else
235  m_uiCallCaptainTimer -= uiDiff;
236 
237  DoMeleeAttackIfReady();
238  }

245 bool ProcessEventId_event_spell_call_captain(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool bIsStart)
246 {
247  if (bIsStart && pSource->GetTypeId() == TYPEID_UNIT)
248  {
249  Creature* pVaros = (Creature*)pSource;
250  if (!pVaros)
251  return false;
252 
253  // each guardian has it's own spawn position
254  for (const auto& i : aVarosCaptainData)
255  {
256  if (uiEventId == i.uiEventId)
257  {
258  if (Creature* pGuardian = pVaros->SummonCreature(NPC_AZURE_RING_CAPTAIN, i.fX, i.fY, i.fZ, i.fO, TEMPSPAWN_DEAD_DESPAWN, 0))
259  pGuardian->GetMotionMaster()->MovePoint(1, i.fDestX, i.fDestY, i.fDestZ, FORCED_MOVEMENT_FLIGHT, false);
260 
261  return true;
262  }
263  }
264  }
265 
266  return false;
267 }
