103  void ReceiveAIEvent(AIEventType eventType, Unit* /*sender*/, Unit* invoker, uint32 /*miscValue*/) override
104  {
105  if (eventType == AI_EVENT_CUSTOM_A)
106  {
107  m_creature->SetFactionTemporary(1693, TEMPFACTION_RESTORE_RESPAWN | TEMPFACTION_RESTORE_COMBAT_STOP);
108  AttackStart(invoker);
109  m_eventStarted = true;
110  DisableCombatAction(BONECHEWER_FRENZY);
111  }
112  else if (eventType == AI_EVENT_CUSTOM_B)
113  {
114  m_creature->getThreatManager().modifyThreatPercent(invoker, -101);
115  m_creature->ClearTemporaryFaction();
116  m_eventStarted = false;
117  ResetCombatAction(BONECHEWER_FRENZY, urand(7000, 14000));
118  }
119  }

121  void ExecuteAction(uint32 action) override
122  {
123  switch (action)
124  {
125  case BONECHEWER_STOP_EVENT:
126  {
127  if (m_eventStarted && m_creature->GetHealthPercent() <= 40.f)
128  {
129  EnterEvadeMode();
130  }
131  break;
132  }
133  case BONECHEWER_ENRAGE:
134  {
135  if (m_creature->GetHealthPercent() > 30.f)
136  break;
137 
138  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
139  SetActionReadyStatus(action, false);
140  break;
141  }
142  case BONECHEWER_FRENZY:
143  {
144  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
145  {
146  ResetCombatAction(action, urand(21000, 28000));

150  }
