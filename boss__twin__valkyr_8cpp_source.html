159  // function that handles the special ability for both twins
160  bool DoCastSpecialAbility()
161  {
162  if (!m_pInstance)
163  return false;
164 
165  // choose the caster; it always alternates
166  Unit* pCaster = nullptr;
167  uint32 uiSpell = 0;
168  uint32 uiShieldSpell = 0;
169 
170  if (m_bIsLightTwin)
171  pCaster = m_creature;
172  else
173  {
174  Creature* pEydis = m_pInstance->GetSingleCreatureFromStorage(NPC_EYDIS);
175  if (!pEydis)
176  return false;
177 
178  pCaster = pEydis;
179  }
180 
181  if (!pCaster)
182  return false;
183 
184  // select and cast ability
185  if (m_bIsVortex)
186  {
187  uiSpell = m_bIsLightTwin ? SPELL_LIGHT_VORTEX : SPELL_DARK_VORTEX;
188  pCaster->CastSpell(pCaster, uiSpell, TRIGGERED_NONE);
189  DoScriptText(m_bIsLightTwin ? SAY_TO_WHITE : SAY_TO_BLACK, pCaster);
190  }
191  else
192  {
193  uiSpell = m_bIsLightTwin ? SPELL_TWINS_PACT_LIGHT : SPELL_TWINS_PACT_DARK;
194  uiShieldSpell = m_bIsLightTwin ? SPELL_SHIELD_OF_LIGHTS : SPELL_SHIELD_OF_DARKNESS;
195  pCaster->CastSpell(pCaster, uiSpell, TRIGGERED_NONE);
196  pCaster->CastSpell(pCaster, uiShieldSpell, TRIGGERED_OLD_TRIGGERED);
197  DoScriptText(SAY_COLORSWITCH, pCaster);
198  }
199 
200  m_bIsVortex = urand(0, 1) != 0;
201  m_bIsLightTwin = !m_bIsLightTwin;
202  return true;
203  }

205  void UpdateAI(const uint32 uiDiff) override
206  {
207  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
208  return;
209 
210  // special ability spell
211  if (m_uiSpecialAbilityTimer < uiDiff)
212  {
213  if (DoCastSpecialAbility())
214  m_uiSpecialAbilityTimer = 45000;
215  }
216  else
217  m_uiSpecialAbilityTimer -= uiDiff;
218 
219  if (m_uiSummonTimer < uiDiff)
220  {
221  DoCastSpellIfCan(m_creature, SPELL_LIGHT_BULLET_SUMMON_TRIGGER, CAST_TRIGGERED);
222  DoCastSpellIfCan(m_creature, SPELL_DARK_BULLET_SUMMON_TRIGGER, CAST_TRIGGERED);
223  m_uiSummonTimer = 30000;
224  }
225  else
226  m_uiSummonTimer -= uiDiff;
227 
228  // berserk spell
229  if (m_uiBerserkTimer)
230  {
231  if (m_uiBerserkTimer <= uiDiff)
232  {
233  // handle berserk for both twins
234  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
235  {
236  if (m_pInstance)
237  {
238  if (Creature* pEydis = m_pInstance->GetSingleCreatureFromStorage(NPC_EYDIS))
239  {
240  pEydis->CastSpell(pEydis, SPELL_BERSERK, TRIGGERED_OLD_TRIGGERED);
241  DoScriptText(SAY_BERSERK, pEydis);
242  }
243  }
244 
245  DoScriptText(SAY_BERSERK, m_creature);
246  m_uiBerserkTimer = 0;
247  }
248  }
249  else
250  m_uiBerserkTimer -= uiDiff;
251  }
252 
253  if (m_uiTwinSpikeTimer < uiDiff)
254  {
255  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_TWIN_SPIKE_LIGHT) == CAST_OK)
256  m_uiTwinSpikeTimer = 10000;
257  }
258  else
259  m_uiTwinSpikeTimer -= uiDiff;
260 
261  // heroic abilities
262  if (m_pInstance && m_pInstance->IsHeroicDifficulty())
263  {
264  if (m_uiTouchTimer < uiDiff)
265  {
266  if (DoCastSpellIfCan(m_creature, SPELL_LIGHT_TOUCH) == CAST_OK)
267  m_uiTouchTimer = 20000;
268  }
269  else
270  m_uiTouchTimer -= uiDiff;
271  }
272 
273  DoMeleeAttackIfReady();
274  }
