194  // Helper function which summons all the Volunteers
195  void DoCallVolunteers()
196  {
197  // The volunteers should be summoned on the bottom of each stair in 2 lines - 7 in the front line and 6 in the back line
198  // However, because this would involve too many hardcoded coordinates we'll summon this on random point near the stairs
199 
200  float fX, fY, fZ;
201  for (uint8 j = 0; j < 2; ++j)
202  {
203  for (uint8 i = 0; i < MAX_VOLUNTEERS_PER_SIDE; ++i)
204  {
205  // In order to get a good movement position we need to handle the coordinates calculation here, inside the iteration.
206  m_creature->GetRandomPoint(aVolunteerPosition[j][0], aVolunteerPosition[j][1], aVolunteerPosition[j][2], 10.0f, fX, fY, fZ);
207  if (Creature* pVolunteer = m_creature->SummonCreature(NPC_TWILIGHT_VOLUNTEER, fX, fY, fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0))
208  {
209  // Adjust coordinates based on the wave number and side
210  float fDist = i < 7 ? 20.0f : 30.0f;
211  float fAngle = 0;
212  if (!j)
213  fAngle = i < 7 ? (i - 2) * (3 * M_PI_F / 35) : (i - 6) * (M_PI_F / 16);
214  else
215  fAngle = i < 7 ? (i - 10) * (3 * M_PI_F / 35) : 3 * M_PI_F / 2 - (i - 6) * (M_PI_F / 16);
216 
217  m_creature->GetNearPoint(m_creature, fX, fY, fZ, 0, fDist, fAngle);
218  pVolunteer->GetMotionMaster()->MovePoint(POINT_ID_PREPARE, fX, fY, fZ);
219  }
220  }
221  }
222 
223  // Summon one more Volunteer for the center position
224  m_creature->GetRandomPoint(aVolunteerPosition[0][0], aVolunteerPosition[0][1], aVolunteerPosition[0][2], 10.0f, fX, fY, fZ);
225  if (Creature* pVolunteer = m_creature->SummonCreature(NPC_TWILIGHT_VOLUNTEER, fX, fY, fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0))
226  {
227  m_creature->GetNearPoint(m_creature, fX, fY, fZ, 0, 20.0f, 7 * M_PI_F / 4);
228  pVolunteer->GetMotionMaster()->MovePoint(POINT_ID_PREPARE, fX, fY, fZ);
229  }
230  }

276  void MovementInform(uint32 uiMoveType, uint32 uiPointId) override
277  {
278  if (uiMoveType != POINT_MOTION_TYPE)
279  return;
280 
281  switch (uiPointId)
282  {
283  // Prepare for combat
284  case POINT_ID_PREPARE:
285 
286  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
287  m_creature->RemoveAurasDueToSpell(SPELL_LIGHTNING_VISUAL);
288  m_creature->RemoveAurasDueToSpell(SPELL_SPHERE_VISUAL);
289  m_creature->SetLevitate(false);
290  break;
291 
292  // Prepare for sacrifice lift off
293  case POINT_ID_SACRIFICE:
294  DoCastSpellIfCan(m_creature, SPELL_HOVER_FALL);
295  m_creature->SetLevitate(true);
296  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LEVITATE, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 10.0f);
297  break;
298 
299  // Call a volunteer to sacrifice
300  case POINT_ID_LEVITATE:
301  if (Creature* pVolunteer = m_creature->GetMap()->GetCreature(SelectRandomVolunteer()))
302  {
303  DoScriptText(urand(0, 1) ? SAY_CALL_SACRIFICE1 : SAY_CALL_SACRIFICE2, m_creature);
304  DoScriptText(SAY_VOLUNTEER_CHOOSEN, pVolunteer);
305 
306  pVolunteer->RemoveAurasDueToSpell(SPELL_VOLUNTEER_SPHERE);
307  pVolunteer->SetStandState(UNIT_STAND_STATE_STAND);
308  pVolunteer->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
309  pVolunteer->CastSpell(pVolunteer, SPELL_PILLAR_LIGHTNING, TRIGGERED_NONE);
310  pVolunteer->SetWalk(true);
311  pVolunteer->GetMotionMaster()->MovePoint(POINT_ID_SACRIFICE, aJedogaLandingLoc[0], aJedogaLandingLoc[1], aJedogaLandingLoc[2]);
312  }
313 
314  // Set visual aura
315  if (m_pInstance)
316  {
317  if (Creature* pTemp = m_creature->GetMap()->GetCreature(m_pInstance->SelectJedogaSacrificeControllerGuid()))
318  pTemp->CastSpell(pTemp, SPELL_SACRIFICE_VISUAL, TRIGGERED_NONE);
319  }
320  break;
321 
322  // Resume combat
323  case POINT_ID_COMBAT:
324  m_creature->RemoveAurasDueToSpell(SPELL_HOVER_FALL);
325  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
326 
327  m_bIsSacrificing = false;
328  SetCombatMovement(true);
329  m_creature->SetLevitate(false);
330  if (m_creature->GetVictim())
331  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
332  break;
333  }
334  }

336  void UpdateAI(const uint32 uiDiff) override
337  {
338  if (m_uiVisualTimer)
339  {
340  if (m_uiVisualTimer <= uiDiff)
341  {
342  GuidList lControllersList;
343  if (m_pInstance)
344  m_pInstance->GetJedogaEventControllersList(lControllersList);
345 
346  for (GuidList::const_iterator itr = lControllersList.begin(); itr != lControllersList.end(); ++itr)
347  {
348  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
349  pTemp->CastSpell(m_creature, SPELL_BEAM_VISUAL, TRIGGERED_NONE);
350  }
351 
352  if (DoCastSpellIfCan(m_creature, SPELL_LIGHTNING_VISUAL) == CAST_OK)
353  m_uiVisualTimer = 0;
354  }
355  else
356  m_uiVisualTimer -= uiDiff;
357  }
358 
359  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
360  return;
361 
362  // Don't use abilities while sacrificing
363  if (m_bIsSacrificing)
364  return;
365 
366  // Note: this was changed in 3.3.2 and now it does this only once
367  if (m_creature->GetHealthPercent() < 50.0f && !m_bSacrifice)
368  {
369  SetCombatMovement(false);
370  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
371  m_creature->GetMotionMaster()->MovePoint(POINT_ID_SACRIFICE, aJedogaLandingLoc[0], aJedogaLandingLoc[1], aJedogaLandingLoc[2]);
372  m_bSacrifice = true;
373  m_bIsSacrificing = true;
374  }
375 
376  if (m_uiThundershockTimer < uiDiff)
377  {
378  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
379  DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_THUNDERSHOCK : SPELL_THUNDERSHOCK_H);
380 
381  m_uiThundershockTimer = 40000;
382  }
383  else
384  m_uiThundershockTimer -= uiDiff;
385 
386  if (m_uiLightningBoltTimer < uiDiff)
387  {
388  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
389  DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_LIGHTNING_BOLT : SPELL_LIGHTNING_BOLT_H);
390 
391  m_uiLightningBoltTimer = 7000;
392  }
393  else
394  m_uiLightningBoltTimer -= uiDiff;
395 
396  if (m_uiCycloneStrikeTimer < uiDiff)
397  {
398  DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_CYCLONE_STRIKE : SPELL_CYCLONE_STRIKE_H);
399  m_uiCycloneStrikeTimer = 15000;
400  }
401  else
402  m_uiCycloneStrikeTimer -= uiDiff;
403 
404  DoMeleeAttackIfReady();
405  }
