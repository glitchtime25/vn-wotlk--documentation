71 void instance_blackfathom_deeps::DoSpawnMobs(uint8 uiWaveIndex)
72 {
73  Creature* pKelris = GetSingleCreatureFromStorage(NPC_KELRIS);
74  if (!pKelris)
75  return;
76 
77  float fX_resp, fY_resp, fZ_resp;
78 
79  pKelris->GetRespawnCoord(fX_resp, fY_resp, fZ_resp);
80 
81  for (auto i : aWaveSummonInformation)
82  {
83  if (i.m_uiWaveIndex != uiWaveIndex)
84  continue;
85 
86  // Summon mobs at positions
87  for (uint8 j = 0; j < MAX_COUNT_POS; ++j)
88  {
89  for (uint8 k = 0; k < i.m_aCountAndPos[j].m_uiCount; ++k)
90  {
91  uint8 uiPos = i.m_aCountAndPos[j].m_uiSummonPosition;
92  float fPosX = aSpawnLocations[uiPos].x;
93  float fPosY = aSpawnLocations[uiPos].y;
94  float fPosZ = aSpawnLocations[uiPos].z;
95  float fPosO = aSpawnLocations[uiPos].o;
96 
97  // Adapt fPosY slightly in case of higher summon-counts
98  if (i.m_aCountAndPos[j].m_uiCount > 1)
99  fPosY = fPosY - INTERACTION_DISTANCE / 2 + k * INTERACTION_DISTANCE / i.m_aCountAndPos[j].m_uiCount;
100 
101  if (Creature* pSummoned = pKelris->SummonCreature(i.m_uiNpcEntry, fPosX, fPosY, fPosZ, fPosO, TEMPSPAWN_DEAD_DESPAWN, 0))
102  {
103  pSummoned->GetMotionMaster()->MovePoint(0, fX_resp, fY_resp, fZ_resp);
104  m_lWaveMobsGuids[uiWaveIndex].push_back(pSummoned->GetGUIDLow());
105  }
106  }
107  }
108  }
109 }

111 void instance_blackfathom_deeps::SetData(uint32 uiType, uint32 uiData)
112 {
113  switch (uiType)
114  {
115  case TYPE_KELRIS: // EventAI must set instance data (1,3) at his death
116  if (m_auiEncounter[0] != DONE && uiData == DONE)
117  m_auiEncounter[0] = uiData;
118  break;
119  case TYPE_SHRINE:
120  m_auiEncounter[1] = uiData;
121  if (uiData == IN_PROGRESS)
122  {
123  m_uiSpawnMobsTimer[m_uiWaveCounter] = 3000;
124  ++m_uiWaveCounter;
125  }
126  else if (uiData == DONE)
127  DoUseDoorOrButton(GO_PORTAL_DOOR);
128  break;
129  case TYPE_AQUANIS:
130  m_auiEncounter[2] = uiData;
131  break;
132  }
133 
134  if (uiData == DONE)
135  {
136  OUT_SAVE_INST_DATA;
137 
138  std::ostringstream saveStream;
139 
140  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2];
141 
142  m_strInstData = saveStream.str();
143  SaveToDB();
144  OUT_SAVE_INST_DATA_COMPLETE;
145  }
146 }

228 void instance_blackfathom_deeps::Update(uint32 uiDiff)
229 {
230  // Only use this function if shrine event is in progress
231  if (m_auiEncounter[1] != IN_PROGRESS)
232  return;
233 
234  for (uint8 i = 0; i < MAX_FIRES; ++i)
235  {
236  if (m_uiSpawnMobsTimer[i])
237  {
238  if (m_uiSpawnMobsTimer[i] <= uiDiff)
239  {
240  DoSpawnMobs(i);
241  m_uiSpawnMobsTimer[i] = 0;
242  }
243  else
244  m_uiSpawnMobsTimer[i] -= uiDiff;
245  }
246  }
