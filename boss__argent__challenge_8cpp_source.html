151  void UpdateAI(const uint32 uiDiff) override
152  {
153  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
154  return;
155 
156  if (m_uiHammerTimer < uiDiff)
157  {
158  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
159  {
160  if (DoCastSpellIfCan(pTarget, SPELL_HAMMER_OF_RIGHTEOUS) == CAST_OK)
161  {
162  DoCastSpellIfCan(m_creature, SPELL_HAMMER_OF_JUSTICE, CAST_TRIGGERED);
163 
164  DoScriptText(EMOTE_EADRIC_HAMMER, m_creature, pTarget);
165  m_uiHammerTimer = 35000;
166  }
167  }
168  }
169  else
170  m_uiHammerTimer -= uiDiff;
171 
172  if (m_uiRadianceTimer < uiDiff)
173  {
174  if (DoCastSpellIfCan(m_creature, SPELL_RADIANCE) == CAST_OK)
175  {
176  DoScriptText(EMOTE_EADRIC_RADIANCE, m_creature);
177  m_uiRadianceTimer = urand(30000, 35000);
178  }
179  }
180  else
181  m_uiRadianceTimer -= uiDiff;
182 
183  DoMeleeAttackIfReady();
184  }
