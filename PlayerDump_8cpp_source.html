92 bool findnth(std::string& str, int n, std::string::size_type& s, std::string::size_type& e)
93 {
94  s = str.find("VALUES ('") + 9;
95  if (s == std::string::npos)
96  return false;
97 
98  do
99  {
100  e = str.find('\'', s);
101  if (e == std::string::npos)
102  return false;
103  }
104  while (str[e - 1] == '\\');
105 
106  for (int i = 1; i < n; ++i)
107  {
108  do
109  {
110  s = e + 4;
111  e = str.find('\'', s);
112  if (e == std::string::npos)
113  return false;
114  }
115  while (str[e - 1] == '\\');
116  }
117  return true;
118 }

207 std::string CreateDumpString(char const* tableName, QueryResult* result)
208 {
209  if (!tableName || !result)
210  return "";
211 
212  std::ostringstream ss;
213  ss << "INSERT INTO " << _TABLE_SIM_ << tableName << _TABLE_SIM_ << " VALUES (";
214  Field* fields = result->Fetch();
215  for (uint32 i = 0; i < result->GetFieldCount(); ++i)
216  {
217  if (i != 0)
218  ss << ", ";
219 
220  if (fields[i].IsNULL())
221  ss << "NULL";
222  else
223  {
224  std::string s = fields[i].GetCppString();
225  CharacterDatabase.escape_string(s);
226 
227  ss << "'" << s << "'";

281 void PlayerDumpWriter::DumpTableContent(std::string& dump, uint32 guid, char const* tableFrom, char const* tableTo, DumpTableType type)
282 {
283  GUIDs const* guids = nullptr;
284  char const* fieldname;
285 
286  switch (type)
287  {
288  case DTT_ITEM: fieldname = "guid"; guids = &items; break;
289  case DTT_ITEM_GIFT: fieldname = "item_guid"; guids = &items; break;
290  case DTT_ITEM_LOOT: fieldname = "guid"; guids = &items; break;
291  case DTT_PET: fieldname = "owner"; break;
292  case DTT_PET_TABLE: fieldname = "guid"; guids = &pets; break;
293  case DTT_PET_DECL: fieldname = "id"; break;
294  case DTT_MAIL: fieldname = "receiver"; break;
295  case DTT_MAIL_ITEM: fieldname = "mail_id"; guids = &mails; break;
296  default: fieldname = "guid"; break;
297  }
298 
299  // for guid set stop if set is empty
300  if (guids && guids->empty())
301  return; // nothing to do
302 
303  // setup for guids case start position
304  GUIDs::const_iterator guids_itr;
305  if (guids)
306  guids_itr = guids->begin();
307 
308  do
309  {
310  std::string wherestr;
311 
312  if (guids) // set case, get next guids string
313  wherestr = GenerateWhereStr(fieldname, *guids, guids_itr);
314  else // not set case, get single guid string
315  wherestr = GenerateWhereStr(fieldname, guid);
316 
317  QueryResult* result = CharacterDatabase.PQuery("SELECT * FROM %s WHERE %s", tableFrom, wherestr.c_str());
318  if (!result)
319  return;
320 
321  do
322  {
323  // collect guids
324  switch (type)
325  {
326  case DTT_INVENTORY:
327  StoreGUID(result, 3, items); break; // item guid collection (character_inventory.item)
328  case DTT_PET:
329  StoreGUID(result, 0, pets); break; // pet petnumber collection (character_pet.id)
330  case DTT_MAIL:
331  StoreGUID(result, 0, mails); // mail id collection (mail.id)
332  case DTT_MAIL_ITEM:
333  StoreGUID(result, 1, items); break; // item guid collection (mail_items.item_guid)
334  default: break;
335  }
336 
337  dump += CreateDumpString(tableTo, result);
338  dump += "\n";
339  }
340  while (result->NextRow());
341 
342  delete result;
343  }
344  while (guids && guids_itr != guids->end()); // not set case iterate single time, set case iterate for all guids
345 }

347 std::string PlayerDumpWriter::GetDump(uint32 guid)
348 {
349  std::string dump;
350 
351  dump += "IMPORTANT NOTE: This sql queries not created for apply directly, use '.pdump load' command in console or client chat instead.\n";
352  dump += "IMPORTANT NOTE: NOT APPLY ITS DIRECTLY to character DB or you will DAMAGE and CORRUPT character DB\n\n";
353 
354  // revision check guard
355  QueryNamedResult* result = CharacterDatabase.QueryNamed("SELECT * FROM character_db_version LIMIT 1");
356  if (result)
357  {
358  QueryFieldNames const& namesMap = result->GetFieldNames();
359  std::string reqName;
360  for (const auto& itr : namesMap)
361  {
362  if (itr.substr(0, 9) == "required_")
363  {
364  reqName = itr;
365  break;
366  }
367  }
368 
369  if (!reqName.empty())
370  {
371  // this will fail at wrong character DB version
372  dump += "UPDATE character_db_version SET " + reqName + " = 1 WHERE FALSE;\n\n";
373  }
374  else
375  sLog.outError("Table 'character_db_version' not have revision guard field, revision guard query not added to pdump.");
376 
377  delete result;
378  }
379  else
380  sLog.outError("Character DB not have 'character_db_version' table, revision guard query not added to pdump.");
381 
382  for (DumpTable* itr = &dumpTables[0]; itr->isValid(); ++itr)
383  DumpTableContent(dump, guid, itr->name, itr->name, itr->type);
384 
385  // TODO: Add instance/group..
386  // TODO: Add a dump level option to skip some non-important tables
387 
388  return dump;
389 }

407 DumpReturn PlayerDumpReader::LoadDump(const std::string& file, uint32 account, std::string name, uint32 guid)

410 
411  // check character count
412  uint32 charcount = sAccountMgr.GetCharactersCount(account);
413  if (charcount >= 10)
414  return DUMP_TOO_MANY_CHARS;
415 
416  FILE* fin = fopen(file.c_str(), "r");
417  if (!fin)
418  return DUMP_FILE_OPEN_ERROR;
419 
420  QueryResult* result;
421  char newguid[20], chraccount[20], newpetid[20], currpetid[20], lastpetid[20];
422 
423  // make sure the same guid doesn't already exist and is safe to use
424  bool incHighest = true;
425  if (guid != 0 && guid < sObjectMgr.m_CharGuids.GetNextAfterMaxUsed())
426  {
427  result = CharacterDatabase.PQuery("SELECT * FROM characters WHERE guid = '%u'", guid);
428  if (result)
429  {
430  guid = sObjectMgr.m_CharGuids.GetNextAfterMaxUsed();
431  delete result;
432  }
433  else incHighest = false;
434  }
435  else
436  guid = sObjectMgr.m_CharGuids.GetNextAfterMaxUsed();
437 
438  // normalize the name if specified and check if it exists
439  if (!normalizePlayerName(name))
440  name.clear();

443  {
444  CharacterDatabase.escape_string(name); // for safe, we use name only for sql quearies anyway
445  result = CharacterDatabase.PQuery("SELECT * FROM characters WHERE name = '%s'", name.c_str());
446  if (result)
447  {

459  snprintf(newpetid, 20, "%u", sObjectMgr.GeneratePetNumber());

464  std::map<uint32, uint32> eqsets;
465  char buf[32000] = "";
466

468  typedef PetIds::value_type PetIdsPair;

470 
471  CharacterDatabase.BeginTransaction();
472  while (!feof(fin))
473  {
474  if (!fgets(buf, 32000, fin))
475  {
476  if (feof(fin)) break;
477  ROLLBACK(DUMP_FILE_BROKEN);
478  }
479 
480  std::string line; line.assign(buf);
481 
482  // skip empty strings

484  if (nw_pos == std::string::npos)
485  continue;
486 
487  // skip NOTE
488  if (line.substr(nw_pos, 15) == "IMPORTANT NOTE:")
489  continue;
490 
491  // add required_ check
492  if (line.substr(nw_pos, 41) == "UPDATE character_db_version SET required_")
493  {
494  if (!CharacterDatabase.Execute(line.c_str()))
495  ROLLBACK(DUMP_FILE_BROKEN);

497  continue;
498  }

502  if (tn.empty())

506  }
507

514  type = dTable->type;

516  }
517  }

613  ROLLBACK(DUMP_FILE_BROKEN); // item_loot.guid update
614  if (!changenth(line, 2, newguid)) // item_Loot.owner_guid update
615  ROLLBACK(DUMP_FILE_BROKEN);
616  break;
617  }

621  snprintf(currpetid, 20, "%s", getnth(line, 1).c_str());
622  if (lastpetid[0] == '\0')
623  snprintf(lastpetid, 20, "%s", currpetid);

629  }

648 
649  // lookup currpetid and match to new inserted pet id
650  std::map<uint32, uint32> :: const_iterator petids_iter = petids.find(atoi(currpetid));
