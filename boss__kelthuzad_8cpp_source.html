289  void SpellHit(Unit* /*caster*/, const SpellEntry* spell) override
290  {
291  // Phase 1 start
292  if (spell->Id == SPELL_CHANNEL_VISUAL)
293  {
294  DoScriptText(SAY_SUMMON_MINIONS, m_creature);
295 
296  // Get all summoning trigger NPCs
297  m_summoningTriggers.clear();
298  CreatureList triggers;
299  GetCreatureListWithEntryInGrid(triggers, m_creature, NPC_WORLD_TRIGGER, 100.0f);
300  for (Creature* trigger : triggers)
301  if (centerRoomKT.GetDistance(trigger->GetPosition()) > (60.f * 60.f) && centerRoomKT.GetDistance(trigger->GetPosition()) < (70.f * 70.f))
302  m_summoningTriggers.push_back(trigger->GetObjectGuid());
303  }
304  // Phase 1 periodic update every 1 second (everything in phase 1 is handled here rather than in UpdateAI())
305  else if (spell->Id == SPELL_CHANNEL_VISUAL_EFFECT)
306  {
307  ++m_summonTicks;
308  switch (m_summonTicks)
309  {
310  case 4:
311  SummonIntroCreatures(SPELL_SUMMON_TYPE_A, 9);
312  break;
313  case 9:
314  SummonIntroCreatures(SPELL_SUMMON_TYPE_B, 3);
315  break;
316  case 12:
317  SummonIntroCreatures(SPELL_SUMMON_TYPE_C, 1);
318  break;
319  case MAX_SUMMON_TICKS:
320  DespawnIntroCreatures(); // Will leave those that for some reason are in combat
321  switch (urand(0, 2))
322  {
323  case 0:
324  DoScriptText(SAY_AGGRO1, m_creature);
325  break;
326  case 1:
327  DoScriptText(SAY_AGGRO2, m_creature);
328  break;
329  case 2:
330  DoScriptText(SAY_AGGRO3, m_creature);
331  break;
332  }
333  break;
334  case MAX_CHANNEL_TICKS:
335  StartPhase2();
336  break;
337  default:
338  UpdateSummoning();
339  // Kel'Thuzad does not enter combat until phase 2, so we check every second if there are still players alive and force him to evade otherwise
340  if (!m_instance->GetPlayerInMap(true, false))
341  EnterEvadeMode();
342  break;
343  }
344  }
345  }

347  // Called every tick (1 second) to check if the periodic summoning auras on Kel'Thuzad need to be updated or removed
348  void UpdateSummoning()
349  {
350  for (auto add : phaseOneAdds)
351  {
352  uint32 summoningAura = 0;
353  std::vector<std::vector<uint32>> spellTimers;
354  switch (add)
355  {
356  case NPC_SOUL_WEAVER:
357  spellTimers = soulWeaverSpellsTimers;
358  break;
359  case NPC_UNSTOPPABLE_ABOM:
360  spellTimers = unstoppableAbominationSpellsTimers;
361  break;
362  case NPC_SOLDIER_FROZEN:
363  spellTimers = soldierFrozenWasteSpellsTimers;
364  break;
365  default:
366  return;
367  }
368 
369  // Iterate the summoning auras timeline for the current add as long as the tick is in range
370  // then exit with the last valid summoning aura
371  for (const auto& spellTimer : spellTimers)
372  {
373  if (spellTimer[0] <= m_summonTicks)
374  summoningAura = spellTimer[1];
375  else
376  break;
377  }
378 
379  // Do not update aura for current add type if summoning aura is already present
380  if (m_creature->HasAura(summoningAura))
381  continue;
382 
383  // Clear all summoning auras for current add before applying the new one (especially needed in case new one is NULL)
384  StopAllSummoningForNPC(add);
385  DoCastSpellIfCan(m_creature, summoningAura, CAST_TRIGGERED);
386  }
387  }

389  // Remove all periodic auras on Kel'Thuzad used to summon a given NPC type
390  void StopAllSummoningForNPC(uint32 entry)
391  {
392  auto iter = summoningSpells.begin();
393  while (iter != summoningSpells.end())
394  {
395  if (iter->first == entry)
396  for (auto aura : iter->second)
397  m_creature->RemoveAurasDueToSpell(aura);
398  ++iter;
399  }

402  // Summon three type of adds in each of the surrounding alcoves
403  bool SummonIntroCreatures(uint32 spellId, uint8 count)
404  {
405  if (!m_instance)
406  return false;
407 
408  float newX, newY, newZ;
409 
410  // Spawn all the adds for phase 1 from each of the trigger NPCs
411  for (auto triggerGuid : m_summoningTriggers)
412  {
413  Creature* trigger = m_creature->GetMap()->GetCreature(triggerGuid);
414  if (!trigger)
415  continue;
416  // "How many NPCs per type" is stored in a vector: {npc_entry:number_of_npcs}
417  for (uint8 i = 0; i < count; ++i)
418  {
419  m_creature->GetRandomPoint(trigger->GetPositionX(), trigger->GetPositionY(), trigger->GetPositionZ(), 12.0f, newX, newY, newZ);
420  trigger->CastSpell(newX, newY, newZ, spellId, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
421  }
422  }
423  return true;
424  }

530  void OnEffectExecute(Spell* spell, SpellEffectIndex /* effIdx */) const override
531  {
532  if (Unit* unitTarget = spell->GetUnitTarget())
533  {
534  uint32 summonId = 0;
535  switch (spell->m_spellInfo->Id)
536  {
537  case 28415: summonId = SPELL_SUMMON_TYPE_A; break;
538  case 28416: summonId = SPELL_SUMMON_TYPE_B; break;
539  case 28417: summonId = SPELL_SUMMON_TYPE_C; break;
540  case 28455: summonId = SPELL_SUMMON_TYPE_D; break;
541  default: break;
542  }
543 
544  if (summonId)
545  unitTarget->CastSpell(unitTarget, summonId, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, spell->GetCaster()->GetObjectGuid(), nullptr);
546 
547  if (!unitTarget->GetMap()->GetInstanceData())
548  return;
549 
550  if (Creature* kelthuzad = static_cast<ScriptedInstance*>(unitTarget->GetMap()->GetInstanceData())->GetSingleCreatureFromStorage(NPC_KELTHUZAD))
551  {
552  if (boss_kelthuzadAI* ai = dynamic_cast<boss_kelthuzadAI*>(kelthuzad->AI()))
553  {
554  auto& summonedNpcs = ai->m_introMobsList[unitTarget->GetObjectGuid()];
555  std::shuffle(summonedNpcs.begin(), summonedNpcs.end(), *GetRandomGenerator());
556  for (auto guid : summonedNpcs)
557  if (Creature* creature = kelthuzad->GetMap()->GetCreature(guid))
