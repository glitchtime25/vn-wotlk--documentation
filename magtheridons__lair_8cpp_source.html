90 void instance_magtheridons_lair::SetData(uint32 uiType, uint32 uiData)
91 {
92  switch (uiType)
93  {
94  case TYPE_MAGTHERIDON_EVENT:
95  switch (uiData)
96  {
97  case FAIL:
98  // Reset channelers
99  for (GuidList::const_iterator itr = m_lChannelerGuidList.begin(); itr != m_lChannelerGuidList.end(); ++itr)
100  {
101  if (Creature* pChanneler = instance->GetCreature(*itr))
102  {
103  if (!pChanneler->IsAlive())
104  pChanneler->Respawn();
105  }
106  }
107 
108  // Reset columns
109  for (GuidList::const_iterator itr = m_lColumnGuidList.begin(); itr != m_lColumnGuidList.end(); ++itr)
110  {
111  if (GameObject* pColumn = instance->GetGameObject(*itr))
112  pColumn->ResetDoorOrButton();
113  }
114 
115  {
116  GuidVector abyssals;
117  GetCreatureGuidVectorFromStorage(NPC_BURNING_ABYSSAL, abyssals);
118  DespawnGuids(abyssals);
119  }
120 
121  // Reset cubes
122  for (GuidList::const_iterator itr = m_lCubeGuidList.begin(); itr != m_lCubeGuidList.end(); ++itr)
123  DoToggleGameObjectFlags(*itr, GO_FLAG_NO_INTERACT, true);
124 
125  // Reset timers and doors
126  SetData(TYPE_CHANNELER_EVENT, NOT_STARTED);
127  m_uiCageBreakTimer = 0;
128  m_uiCageBreakStage = 0;
129 
130  // no break;
131  case DONE:
132  // Reset door on Fail or Done
133  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_DOODAD_HF_MAG_DOOR01))
134  pDoor->ResetDoorOrButton();
135 
136  SetData(TYPE_CHANNELER_EVENT, DONE);
137  break;
138  case IN_PROGRESS:
139  // Set boss in combat
140  if (Creature* pMagtheridon = GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
141  if (pMagtheridon->IsAlive())
142  pMagtheridon->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, pMagtheridon, pMagtheridon);
143  // Enable cubes
144  for (GuidList::const_iterator itr = m_lCubeGuidList.begin(); itr != m_lCubeGuidList.end(); ++itr)
145  DoToggleGameObjectFlags(*itr, GO_FLAG_NO_INTERACT, false);
146  break;
147  case SPECIAL:
148  // Collapse the hall - don't store this value
149  for (GuidList::const_iterator itr = m_lColumnGuidList.begin(); itr != m_lColumnGuidList.end(); ++itr)
150  DoUseDoorOrButton(*itr);
151  // return, don't set encounter as special
152  return;
153  }
154  m_auiEncounter[uiType] = uiData;
155  break;
156  case TYPE_CHANNELER_EVENT:
157  // don't set the same data twice
158  if (m_auiEncounter[1] == uiData)
159  break;
160  // stop the event timer on fail
161  if (uiData == FAIL)
162  {
163  m_uiCageBreakTimer = 0;
164  m_uiCageBreakStage = 0;
165 
166  // Reset door on Fail
167  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_DOODAD_HF_MAG_DOOR01))
168  pDoor->ResetDoorOrButton();
169 
170  // Reset Magtheridon
171  if (Creature* pMagtheridon = GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
172  {
173  if (pMagtheridon->IsAlive())
174  pMagtheridon->AI()->EnterEvadeMode();
175  }
176  }
177  // prepare Magtheridon for release
178  if (uiData == IN_PROGRESS)
179  {
180  if (Creature* pMagtheridon = GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
181  {
182  if (pMagtheridon->IsAlive())
183  {
184  pMagtheridon->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, pMagtheridon, pMagtheridon);
185  m_uiCageBreakTimer = MINUTE * IN_MILLISECONDS;
186  }
187  }
188 
189  // combat door
190  DoUseDoorOrButton(GO_DOODAD_HF_MAG_DOOR01);
191  }
192  m_auiEncounter[uiType] = uiData;
193  break;
194  }
195 
196  // Instance save isn't needed for this one
197 }

207 void instance_magtheridons_lair::Update(uint32 uiDiff)
208 {
209  // Prepare to release Magtheridon
210  if (m_uiCageBreakTimer)
211  {
212  if (m_uiCageBreakTimer <= uiDiff)
213  {
214  switch (m_uiCageBreakStage)
215  {
216  case 0:
217  if (Creature* pMagtheridon = GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
218  {
219  if (pMagtheridon->IsAlive())
220  {
221  DoScriptText(EMOTE_NEARLY_FREE, pMagtheridon);
222  m_uiCageBreakTimer = MINUTE * IN_MILLISECONDS;
223  }
224  }
225  break;
226  case 1:
227  SetData(TYPE_MAGTHERIDON_EVENT, IN_PROGRESS);
228  m_uiCageBreakTimer = 0;
229  break;
230  }
231 
232  ++m_uiCageBreakStage;
233  }
234  else
235  m_uiCageBreakTimer -= uiDiff;
236  }
237 
238  // no yell if event is in progress or finished
239  if (m_auiEncounter[TYPE_CHANNELER_EVENT] == IN_PROGRESS || m_auiEncounter[TYPE_MAGTHERIDON_EVENT] == DONE)
240  return;
241 
242  if (m_uiRandYellTimer < uiDiff)
243  {
244  DoOrSimulateScriptTextForThisInstance(aRandomTaunt[urand(0, 5)], NPC_MAGTHERIDON);
245  m_uiRandYellTimer = 90000;
246  }
247  else
248  m_uiRandYellTimer -= uiDiff;
249 }
