182  void OnPeriodicDummy(Aura* aura) const override
183  {
184  if (KillingSpreeStorage* storage = dynamic_cast<KillingSpreeStorage*>(aura->GetScriptStorage()))
185  {
186  Unit* target = aura->GetTarget();
187  Unit* victim = nullptr;
188  std::vector<Unit*> eligibleUnits;
189  for (ObjectGuid guid : storage->targets)
190  if (Unit* unit = target->GetMap()->GetUnit(guid))
191  if (unit->IsAlive() && target->CanAttackSpell(unit) && target->IsWithinCombatDistInMap(unit, 10.f))
192  eligibleUnits.push_back(unit);
193 
194  if (eligibleUnits.size() > 0)
195  victim = eligibleUnits[urand(0, eligibleUnits.size() - 1)];
196 
197  if (victim)

199  target->CastSpell(victim, 57840, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_HIDE_CAST_IN_COMBAT_LOG);
