104 inline void MaNGOS::PlayerVisitObjectsNotifier::Visit(CreatureMapType& m)
105 {
106  if (!i_player.IsAlive() || i_player.IsTaxiFlying())
107  return;
108 
109  bool playerHasAI = i_player.AI() != nullptr;
110 
111  for (auto& iter : m)
112  {
113  Creature* creature = iter.getSource();
114  if (!creature->IsAlive())
115  continue;
116 
117  UnitVisitObjectsNotifierWorker(creature, &i_player);
118 
119  if (playerHasAI)
120  UnitVisitObjectsNotifierWorker(&i_player, creature);
121  }
122 }

192 inline void MaNGOS::DynamicObjectUpdater::VisitHelper(Unit* target)
193 {
194  if (!target->IsAlive() || target->IsTaxiFlying())
195  return;
196 
197  if (target->GetTypeId() == TYPEID_UNIT && ((Creature*)target)->IsTotem())
198  return;
199 
200  Unit* caster = i_dynobject.GetCaster();
201  float radius = i_dynobject.GetRadius();
202  if (caster->IsPlayerControlled() && !target->IsPlayerControlled())
203  radius += target->GetCombatReach();
204  if (i_dynobject.GetDistance(target, true, DIST_CALC_NONE) > radius * radius)
205  return;
206 
207  // Evade target
208  if (target->GetCombatManager().IsInEvadeMode())
209  return;
210 
211  // Check player targets and remove if in GM mode or GM invisibility (for not self casting case)
212  if (target->GetTypeId() == TYPEID_PLAYER && target != i_check && (((Player*)target)->IsGameMaster() || ((Player*)target)->GetVisibility() == VISIBILITY_OFF))
213  return;
214 
215  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(i_dynobject.GetSpellId());
216  SpellEffectIndex eff_index = i_dynobject.GetEffIndex();
217 
218  SQLMultiStorage::SQLMSIteratorBounds<SpellTargetEntry> bounds = sSpellScriptTargetStorage.getBounds<SpellTargetEntry>(spellInfo->Id);
219  if (bounds.first != bounds.second)
220  {
221  bool found = false;
222  for (SQLMultiStorage::SQLMultiSIterator<SpellTargetEntry> i_spellST = bounds.first; i_spellST != bounds.second; ++i_spellST)
223  {
224  if (i_spellST->CanNotHitWithSpellEffect(eff_index))
225  continue;
226 
227  // only creature entries supported for this target type
228  if (i_spellST->type == SPELL_TARGET_TYPE_GAMEOBJECT || i_spellST->type == SPELL_TARGET_TYPE_GAMEOBJECT_GUID)
229  continue;
230 
231  if (target->GetEntry() == i_spellST->targetEntry)
232  {
233  if (i_spellST->type == SPELL_TARGET_TYPE_DEAD && ((Creature*)target)->IsCorpse())
234  found = true;
235  else if (i_spellST->type == SPELL_TARGET_TYPE_CREATURE && target->IsAlive())
236  found = true;
237 
238  break;
239  }
240  }
241 
242  if (!found)
243  return;
244  }
245  // This condition is only needed due to missing neutral spell type
246  else if (!i_script)
247  {
248  // for player casts use less strict negative and more stricted positive targeting
249  if (i_positive)
250  {
251  if (!caster->CanAssistSpell(target, spellInfo))
252  return;
253  }
254  else
255  {
256  if (!caster->CanAttackSpell(target, spellInfo, true))
257  return;
258  }
259  }
260 
261  if (spellInfo->HasAttribute(SPELL_ATTR_EX3_ONLY_ON_PLAYER) && target->GetTypeId() != TYPEID_PLAYER)
262  return;
263 
264  if (spellInfo->HasAttribute(SPELL_ATTR_EX5_NOT_ON_PLAYER) && target->GetTypeId() == TYPEID_PLAYER)
265  return;
266 
267  if (spellInfo->HasAttribute(SPELL_ATTR_EX5_NOT_ON_PLAYER_CONTROLLED_NPC) && target->IsPlayerControlled() && target->GetTypeId() != TYPEID_PLAYER)
268  return;
269 
270  // Check target immune to spell or aura
271  if (!spellInfo->HasAttribute(SPELL_ATTR_NO_IMMUNITIES)) // confirmed 40657 - Ancient Flames goes through immunity
272  if (target->IsImmuneToSpell(spellInfo, false, (1 << eff_index), caster) || target->IsImmuneToSpellEffect(spellInfo, eff_index, false))
273  return;
274 
275  if (!spellInfo->HasAttribute(SPELL_ATTR_EX2_IGNORE_LINE_OF_SIGHT) && !i_dynobject.IsWithinLOSInMap(target))
276  return;
277 
278  // Apply PersistentAreaAura on target
279  // in case 2 dynobject overlap areas for same spell, same holder is selected, so dynobjects share holder
280  SpellAuraHolder* holder = target->GetSpellAuraHolder(spellInfo->Id, i_dynobject.GetCasterGuid());
281 
282  if (holder)
283  {
284  if (!holder->GetAuraByEffectIndex(eff_index))
285  {
286  PersistentAreaAura* Aur = new PersistentAreaAura(spellInfo, eff_index, &i_dynobject.GetDamage(), &i_dynobject.GetBasePoints(), holder, target, caster);
287  holder->AddAura(Aur, eff_index);
288  target->AddAuraToModList(Aur);
289  Aur->ApplyModifier(true, true);
290  }
291  else if (holder->GetAuraDuration() >= 0 && uint32(holder->GetAuraDuration()) < i_dynobject.GetDuration())
292  {
293  holder->SetAuraDuration(i_dynobject.GetDuration());
294  holder->SendAuraUpdate(false);
295  }
296  }
297  else
298  {
299  holder = CreateSpellAuraHolder(spellInfo, target, caster);
300  PersistentAreaAura* Aur = new PersistentAreaAura(spellInfo, eff_index, &i_dynobject.GetDamage(), &i_dynobject.GetBasePoints(), holder, target, caster);
301  holder->SetAuraDuration(i_dynobject.GetDuration());
302  holder->AddAura(Aur, eff_index);
303  if (!target->AddSpellAuraHolder(holder))
304  delete holder;
305  }
306 
307  if (!i_dynobject.IsAffecting(target))
308  {
309  i_dynobject.AddAffected(target);
310  caster->CasterHitTargetWithSpell(caster, target, spellInfo, false);
311  }
312 }
