154 void PlayerMenu::SendGossipMenu(uint32 TitleTextId, ObjectGuid objectGuid)
155 {
156  WorldPacket data(SMSG_GOSSIP_MESSAGE, (100)); // guess size
157  data << ObjectGuid(objectGuid);
158  data << uint32(mGossipMenu.GetMenuId()); // new 2.4.0
159  data << uint32(TitleTextId);
160  data << uint32(mGossipMenu.MenuItemCount()); // max count 0x20
161 
162  for (uint32 iI = 0; iI < mGossipMenu.MenuItemCount(); ++iI)
163  {
164  GossipMenuItem const& gItem = mGossipMenu.GetItem(iI);
165  data << uint32(iI);
166  data << uint8(gItem.m_gIcon);
167  data << uint8(gItem.m_gCoded); // makes pop up box password
168  data << uint32(gItem.m_gBoxMoney); // money required to open menu, 2.0.3
169  data << gItem.m_gMessage; // text for gossip item, max 0x800
170  data << gItem.m_gBoxMessage; // accept text (related to money) pop up box, 2.0.3, max 0x800
171  }
172 
173  data << uint32(mQuestMenu.MenuItemCount()); // max count 0x20
174 
175  for (uint32 iI = 0; iI < mQuestMenu.MenuItemCount(); ++iI)
176  {
177  QuestMenuItem const& qItem = mQuestMenu.GetItem(iI);
178  uint32 questID = qItem.m_qId;
179  Quest const* pQuest = sObjectMgr.GetQuestTemplate(questID);
180 
181  data << uint32(questID);
182  data << uint32(qItem.m_qIcon);
183  data << int32(pQuest->GetQuestLevel());
184  data << uint32(pQuest->GetQuestFlags()); // 3.3.3 quest flags
185  data << uint8(pQuest->IsAutoComplete() && pQuest->IsRepeatable() && !pQuest->IsDailyOrWeekly() && !pQuest->IsMonthly()); // 3.3.3 icon changes - 0: yellow exclapamtion mark, 1: blue question mark
186 
187  int loc_idx = GetMenuSession()->GetSessionDbLocaleIndex();
188  std::string title = pQuest->GetTitle();
189  sObjectMgr.GetQuestLocaleStrings(questID, loc_idx, &title);
190 
191  data << title; // max 0x200
192  }
193 
194  GetMenuSession()->SendPacket(data);
195  DEBUG_LOG("WORLD: Sent SMSG_GOSSIP_MESSAGE from %s", objectGuid.GetString().c_str());
196 }

367  data << uint32(pQuest->GetQuestFlags()); // 3.3.3 questFlags

571 void PlayerMenu::SendQuestGiverRequestItems(Quest const* pQuest, ObjectGuid npcGUID, bool Completable, bool CloseOnCancel) const
572 {
573  // We can always call to RequestItems, but this packet only goes out if there are actually
574  // items. Otherwise, we'll skip straight to the OfferReward
575 
576  std::string Title = pQuest->GetTitle();
577  std::string RequestItemsText = pQuest->GetRequestItemsText();
578 
579  int loc_idx = GetMenuSession()->GetSessionDbLocaleIndex();
580  if (loc_idx >= 0)
581  {
582  if (QuestLocale const* ql = sObjectMgr.GetQuestLocale(pQuest->GetQuestId()))
583  {
584  if (ql->Title.size() > (size_t)loc_idx && !ql->Title[loc_idx].empty())
585  Title = ql->Title[loc_idx];
586  if (ql->RequestItemsText.size() > (size_t)loc_idx && !ql->RequestItemsText[loc_idx].empty())
587  RequestItemsText = ql->RequestItemsText[loc_idx];
588  }
589  }
590 
591  // Only shown for incomplete quests, or ones that require items.
592  if (RequestItemsText.empty() || ((pQuest->GetReqItemsCount() == 0) && Completable))
593  {
594  SendQuestGiverOfferReward(pQuest, npcGUID, true);
595  return;
596  }
597 
598  WorldPacket data(SMSG_QUESTGIVER_REQUEST_ITEMS, 50); // guess size
599  data << ObjectGuid(npcGUID);
