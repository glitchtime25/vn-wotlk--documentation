162  void SummonedCreatureJustDied(Creature* pSummoned) override
163  {
164  if (!m_pInstance)
165  return;
166 
167  m_lSummonedCreaturesGuids.remove(pSummoned->GetObjectGuid());
168 
169  // when all creatures are cleared, destroy the wall and allow Jaina / Sylvanas to continue
170  if (m_lSummonedCreaturesGuids.empty())
171  {
172  // Make Jaina / Sylvanas start next path
173  if (Creature* pCreature = m_pInstance->GetSingleCreatureFromStorage(m_pInstance->GetPlayerTeam() == ALLIANCE ? NPC_JAINA_PART2 : NPC_SYLVANAS_PART2))
174  {
175  // destroy the wall
176  if (GameObject* pWall = GetClosestGameObjectWithEntry(pCreature, GO_ICE_WALL, 30.0f))
177  pWall->Use(pCreature);
178 
179  // force creature to evade, in case it was attacked by the minions
180  ++m_uiPathId;
181  pCreature->AI()->EnterEvadeMode();
182  pCreature->InterruptNonMeleeSpells(false);
183 
184  // start next WP movement path
