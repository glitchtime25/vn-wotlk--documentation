515  DoUseDoorOrButton(GO_SHIELD_WALL);
516  if (uiData == IN_PROGRESS)
517  {
518  // make sure that the Lightning door is closed when engaged in combat
519  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_LIGHTNING_DOOR))
520  {

528  {
529  DoUseDoorOrButton(GO_XT002_GATE);
530  DoUseDoorOrButton(GO_LIGHTNING_DOOR);
531  }
532  else if (uiData == FAIL)
533  DoCallLeviathanHelp();
534  break;

536  m_auiEncounter[uiType] = uiData;
537  if (uiData == IN_PROGRESS)
538  {

540  SetSpecialAchievementCriteria(TYPE_ACHIEV_SHATTERED, false);

544  if (uiData == IN_PROGRESS)
545  SetSpecialAchievementCriteria(TYPE_ACHIEV_QUICK_SHAVE, true);
546  else if (uiData == FAIL)
