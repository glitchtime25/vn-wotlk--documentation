88 CombatManeuverReturns PlayerbotDeathKnightAI::DoFirstCombatManeuver(Unit* pTarget)
89 {
90  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
91  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
92  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
93  {
94  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
95  {
96  if (PlayerbotAI::ORDERS_TANK & m_ai.GetCombatOrder())
97  {
98  if (m_bot.GetDistance(pTarget, true, DIST_CALC_COMBAT_REACH_WITH_MELEE) <= ATTACK_DISTANCE)
99  {
100  // Set everyone's UpdateAI() waiting to 2 seconds
101  m_ai.SetGroupIgnoreUpdateTime(2);
102  // Clear their TEMP_WAIT_TANKAGGRO flag
103  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
104  // Start attacking, force target on current target
105  m_ai.Attack(m_ai.GetCurrentTarget());
106 
107  // While everyone else is waiting 2 second, we need to build up aggro, so don't return
108  }
109  else
110  {
111  // TODO: add check if target is ranged
112  return RETURN_NO_ACTION_OK; // wait for target to get nearer
113  }
114  }
115  else
116  return RETURN_NO_ACTION_OK; // wait it out
117  }
118  else
119  {
120  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
121  }
122  }
123 
124  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
125  {
126  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
127  return RETURN_NO_ACTION_OK; // wait it out
128  else
129  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
130  }
131 
132  switch (m_ai.GetScenarioType())
133  {
134  case PlayerbotAI::SCENARIO_PVP_DUEL:
135  case PlayerbotAI::SCENARIO_PVP_BG:
136  case PlayerbotAI::SCENARIO_PVP_ARENA:
137  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
138  return DoFirstCombatManeuverPVP(pTarget);
139  case PlayerbotAI::SCENARIO_PVE:
140  case PlayerbotAI::SCENARIO_PVE_ELITE:
141  case PlayerbotAI::SCENARIO_PVE_RAID:
142  default:
143  return DoFirstCombatManeuverPVE(pTarget);
144  break;
145  }
146 
147  return RETURN_NO_ACTION_ERROR;
148 }
