90 bool AreaTrigger_at_pit_of_saron(Player* pPlayer, AreaTriggerEntry const* pAt)
91 {
92  if (pPlayer->IsGameMaster() || !pPlayer->IsAlive())
93  return false;
94 
95  instance_pit_of_saron* pInstance = (instance_pit_of_saron*)pPlayer->GetInstanceData();
96  if (!pInstance)
97  return false;
98 
99  if (pAt->id == AREATRIGGER_ID_TUNNEL_START)
100  {
101  if (pInstance->GetData(TYPE_GARFROST) != DONE || pInstance->GetData(TYPE_KRICK) != DONE ||
102  pInstance->GetData(TYPE_AMBUSH) != NOT_STARTED)
103  return false;
104 
105  pInstance->SetData(TYPE_AMBUSH, IN_PROGRESS);
106  return true;
107  }
108  if (pAt->id == AREATRIGGER_ID_TUNNEL_END)
109  {
110  if (pInstance->GetData(TYPE_AMBUSH) != IN_PROGRESS)
111  return false;
112 
113  pInstance->SetData(TYPE_AMBUSH, DONE);
114  return true;
115  }
116 
117  return false;
118 }
