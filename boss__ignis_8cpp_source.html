85 struct boss_ignisAI : public ScriptedAI
86 {
87  boss_ignisAI(Creature* pCreature) : ScriptedAI(pCreature)
88  {
89  m_pInstance = (instance_ulduar*)pCreature->GetInstanceData();
90  m_bIsRegularMode = pCreature->GetMap()->IsRegularDifficulty();
91  Reset();
92  }
93 
94  instance_ulduar* m_pInstance;
95  bool m_bIsRegularMode;
96 
97  uint32 m_uiBerserkTimer;
98  uint32 m_uiFlameJetsTimer;
99  uint32 m_uiSlagPotTimer;
100  uint32 m_uiScorchTimer;
101  uint32 m_uiConstructTimer;
102 
103  void Reset() override
104  {
105  m_uiBerserkTimer = 10 * MINUTE * IN_MILLISECONDS;
106  m_uiFlameJetsTimer = 20000;
107  m_uiSlagPotTimer = 25000;
108  m_uiScorchTimer = 13000;
109  m_uiConstructTimer = 10000;
110  }
111 
112  void JustDied(Unit* /*pKiller*/) override
113  {
114  if (m_pInstance)
115  m_pInstance->SetData(TYPE_IGNIS, DONE);
116 
117  DoCastSpellIfCan(m_creature, SPELL_KILL_ALL_CONSTRUCTS, CAST_TRIGGERED);
118  DoScriptText(SAY_DEATH, m_creature);
119  }
120

122  {
123  if (pVictim->GetTypeId() != TYPEID_PLAYER)
124  return;

126  DoScriptText(urand(0, 1) ? SAY_SLAY_1 : SAY_SLAY_2, m_creature);
127  }

135  }

142 
143  void JustSummoned(Creature* pSummoned) override
144  {
145  if (pSummoned->GetEntry() == NPC_SCORCH)
146  pSummoned->CastSpell(pSummoned, m_bIsRegularMode ? SPELL_SCORCH_AURA : SPELL_SCORCH_AURA_H, TRIGGERED_OLD_TRIGGERED);
147  }
148 
149  // TODO: Use the vehicle boarding wrappers when they are implemented in core
150  void SpellHit(Unit* pCaster, const SpellEntry* pSpell) override
151  {
152  if (pCaster->GetTypeId() != TYPEID_PLAYER)
153  return;
154 
155  // Handle the case when passenger is loaded to the second seat
156  if (pSpell->Id == SPELL_GRAB_POT)
157  DoCastSpellIfCan(pCaster, SPELL_SLAG_POT, CAST_TRIGGERED);
158  }
159 
160  void UpdateAI(const uint32 uiDiff) override
161  {
162  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
163  return;
164 
165  if (m_uiBerserkTimer)
166  {
167  if (m_uiBerserkTimer <= uiDiff)
168  {
169  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
170  {
171  DoScriptText(SAY_BERSERK, m_creature);
172  m_uiBerserkTimer = 0;
173  }
174  }
175  else
176  m_uiBerserkTimer -= uiDiff;
177  }
178 
179  if (m_uiFlameJetsTimer < uiDiff)
180  {
181  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_FLAME_JETS : SPELL_FLAME_JETS_H) == CAST_OK)
182  {
183  DoScriptText(EMOTE_FLAME_JETS, m_creature);
184  DoScriptText(SAY_FLAME_JETS, m_creature);
185  m_uiFlameJetsTimer = 35000;
186  }
187  }
188  else
189  m_uiFlameJetsTimer -= uiDiff;
190 
191  if (m_uiSlagPotTimer < uiDiff)
192  {
193  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_GRAB, SELECT_FLAG_PLAYER))
194  {
195  if (DoCastSpellIfCan(pTarget, SPELL_GRAB) == CAST_OK)
196  {
197  DoScriptText(urand(0, 1) ? SAY_SLAGPOT_1 : SAY_SLAGPOT_2, m_creature);
198  m_uiSlagPotTimer = 30000;
199  }
200  }
201  }
202  else
203  m_uiSlagPotTimer -= uiDiff;
204 
205  if (m_uiConstructTimer < uiDiff)
206  {
207  if (DoCastSpellIfCan(m_creature, SPELL_ACTIVATE_CONSTRUCT) == CAST_OK)
208  {
209  DoScriptText(SAY_ADDS, m_creature);
210  m_uiConstructTimer = 40000;
211  }
212  }
213  else
214  m_uiConstructTimer -= uiDiff;
215 
216  if (m_uiScorchTimer < uiDiff)
217  {
218  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SCORCH : SPELL_SCORCH_H) == CAST_OK)
219  {
220  DoCastSpellIfCan(m_creature, SPELL_SCORCH_SUMMON, CAST_TRIGGERED);
221  m_uiScorchTimer = 25000;
222  }
223  }
224  else
225  m_uiScorchTimer -= uiDiff;
226 
227  DoMeleeAttackIfReady();
228  }
229 };

269  void DamageTaken(Unit* pDoneBy, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
270  {
271  // ToDo: This may need more research related to spell proc
272  if (m_creature->HasAura(m_bIsRegularMode ? SPELL_BRITTLE : SPELL_BRITTLE_H) && !m_bHasShattered)
273  {
274  if (uiDamage > 5000)
275  {
276  DoCastSpellIfCan(m_creature, SPELL_SHATTER, CAST_TRIGGERED);
277  DoCastSpellIfCan(m_creature, SPELL_STRENGTH_REMOVE, CAST_TRIGGERED);
278 
279  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
280  m_creature->GetMotionMaster()->MoveIdle();
281  m_bHasShattered = true;
282  }
283  }
284  }
285 
286  void SpellHit(Unit* pCaster, const SpellEntry* pSpell) override
287  {
288  if (pSpell->Id == SPELL_HEAT)
289  {
290  if (SpellAuraHolder* pHeatAura = m_creature->GetSpellAuraHolder(SPELL_HEAT))
291  {
292  if (pHeatAura && pHeatAura->GetStackAmount() == MAX_HEAT_STACKS)
293  DoCastSpellIfCan(m_creature, SPELL_MOLTEN);
294  }
295  }
296  }
297 
298  void UpdateAI(const uint32 /*uiDiff*/) override
299  {
300  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
301  return;
302 
303  // stop attacking after shattered
304  if (m_bHasShattered)
305  return;
306 
307  DoMeleeAttackIfReady();
308  }
