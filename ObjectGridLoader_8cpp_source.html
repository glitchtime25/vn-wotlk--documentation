48 void
49 ObjectGridRespawnMover::Visit(CreatureMapType& m)
50 {
51  // creature in unloading grid can have respawn point in another grid
52  // if it will be unloaded then it will not respawn in original grid until unload/load original grid
53  // move to respawn point to prevent this case. For player view in respawn grid this will be normal respawn.
54  for (CreatureMapType::iterator iter = m.begin(), next; iter != m.end(); iter = next)
55  {
56  next = iter; ++next;
57 
58  Creature* c = iter->getSource();
59 
60  MANGOS_ASSERT(!c->IsPet() && "ObjectGridRespawnMover don't must be called for pets");
61 
62  Cell const& cur_cell = c->GetCurrentCell();
63 
64  float resp_x, resp_y, resp_z;
65  c->GetRespawnCoord(resp_x, resp_y, resp_z);
66  CellPair resp_val = MaNGOS::ComputeCellPair(resp_x, resp_y);
67  Cell resp_cell(resp_val);
68 
69  if (cur_cell.DiffGrid(resp_cell))
70  {
71  c->GetMap()->CreatureRespawnRelocation(c);
72  // false result ignored: will be unload with other creatures at grid
73  }
74  }
75 }

114 template <class T>
115 void LoadHelper(CellGuidSet const& guid_set, CellPair& cell, GridRefManager<T>& /*m*/, uint32& count, Map* map, GridType& grid)
116 {
117  BattleGround* bg = map->IsBattleGroundOrArena() ? ((BattleGroundMap*)map)->GetBG() : nullptr;
118 
119  for (uint32 guid : guid_set)
120  {
121  T* obj;
122  if constexpr (std::is_same_v<T, GameObject>)
123  {
124  GameObjectData const* data = sObjectMgr.GetGOData(guid);
125  MANGOS_ASSERT(data);
126  obj = (T*)GameObject::CreateGameObject(data->id);
127  }
128  else
129  obj = new T;
130  // sLog.outString("DEBUG: LoadHelper from table: %s for (guid: %u) Loading",table,guid);
131  if (!obj->LoadFromDB(guid, map, guid, 0))
132  {
133  delete obj;
134  continue;
135  }
136 
137  if (!obj->IsCreature())
138  {
139  grid.AddGridObject(obj);
140 
141  addUnitState(obj, cell);
142  }
143 
144  // if this assert is hit we have a problem somewhere because LoadFromDb should already add to map due to AI
145  MANGOS_ASSERT(obj->IsInWorld());
146 
147  obj->GetViewPoint().Event_AddedToWorld(&grid);
148 
149  if (bg)
150  bg->OnObjectDBLoad(obj);
151 
152  ++count;
153  }
154 }

156 void LoadHelper(CellCorpseSet const& cell_corpses, CellPair& cell, CorpseMapType& /*m*/, uint32& count, Map* map, GridType& grid)
157 {
158  if (cell_corpses.empty())
159  return;
160 
161  for (const auto& cell_corpse : cell_corpses)
162  {
163  if (cell_corpse.second != map->GetInstanceId())
164  continue;
165 
166  uint32 player_lowguid = cell_corpse.first;
167 
168  Corpse* obj = sObjectAccessor.GetCorpseForPlayerGUID(ObjectGuid(HIGHGUID_PLAYER, player_lowguid));
169  if (!obj)
170  continue;
171 
172  grid.AddWorldObject(obj);
173 
174  addUnitState(obj, cell);
175  obj->SetMap(map);
176  obj->AddToWorld();
177  if (obj->isActiveObject())
178  map->AddToActive(obj);
179 
180  ++count;
181  }
182 }

185 ObjectGridLoader::Visit(GameObjectMapType& m)
186 {
187  uint32 x = (i_cell.GridX() * MAX_NUMBER_OF_CELLS) + i_cell.CellX();
188  uint32 y = (i_cell.GridY() * MAX_NUMBER_OF_CELLS) + i_cell.CellY();
189  CellPair cell_pair(x, y);
190  uint32 cell_id = (cell_pair.y_coord * TOTAL_NUMBER_OF_CELLS_PER_MAP) + cell_pair.x_coord;
191 
192  CellObjectGuids const& cell_guids = sObjectMgr.GetCellObjectGuids(i_map->GetId(), i_map->GetSpawnMode(), cell_id);
193 
194  GridType& grid = (*i_map->getNGrid(i_cell.GridX(), i_cell.GridY()))(i_cell.CellX(), i_cell.CellY());
195  LoadHelper(cell_guids.gameobjects, cell_pair, m, i_gameObjects, i_map, grid);
196  LoadHelper(i_map->GetPersistentState()->GetCellObjectGuids(cell_id).gameobjects, cell_pair, m, i_gameObjects, i_map, grid);
