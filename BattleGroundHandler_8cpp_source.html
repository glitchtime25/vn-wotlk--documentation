85 // Sent by client when player wants to join a battleground
86 void WorldSession::HandleBattlemasterJoinOpcode(WorldPacket& recv_data)
87 {
88  ObjectGuid guid;
89  uint32 receivedBgTypeId;
90  uint32 instanceId;
91  uint8 joinAsGroup;
92  bool isPremade = false;
93  Group* grp = nullptr;
94 
95  recv_data >> guid; // battlemaster guid
96  recv_data >> receivedBgTypeId; // battleground type id (DBC id)
97  recv_data >> instanceId; // instance id, 0 if First Available selected
98  recv_data >> joinAsGroup; // join as group
99 
100  if (!sBattlemasterListStore.LookupEntry(receivedBgTypeId))
101  {
102  sLog.outError("Battleground: invalid bgtype (%u) received. possible cheater? player guid %u", receivedBgTypeId, _player->GetGUIDLow());
103  return;
104  }
105 
106  BattleGroundTypeId bgTypeId = BattleGroundTypeId(receivedBgTypeId);
107 
108  DEBUG_LOG("WORLD: Received opcode CMSG_BATTLEMASTER_JOIN from %s", guid.GetString().c_str());
109 
110  // can do this, since it's battleground, not arena
111  BattleGroundQueueTypeId bgQueueTypeId = BattleGroundMgr::BgQueueTypeId(bgTypeId, ARENA_TYPE_NONE);
112  BattleGroundQueueTypeId bgQueueTypeIdRandom = BattleGroundMgr::BgQueueTypeId(BATTLEGROUND_RB, ARENA_TYPE_NONE);
113 
114  // ignore if player is already in BG
115  if (_player->InBattleGround())
116  return;
117 
118  // get bg instance or bg template if instance not found
119  BattleGround* bg = nullptr;
120  if (instanceId)
121  bg = sBattleGroundMgr.GetBattleGroundThroughClientInstance(instanceId, bgTypeId);
122 
123  if (!bg)
124  bg = sBattleGroundMgr.GetBattleGroundTemplate(bgTypeId);
125 
126  if (!bg)
127  {
128  sLog.outError("Battleground: no available bg / template found");
129  return;
130  }
131 
132  // expected bracket entry
133  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->GetLevel());
134  if (!bracketEntry)
135  return;
136 
137  GroupJoinBattlegroundResult err = ERR_GROUP_JOIN_BATTLEGROUND_FAIL;
138 
139  // check queue conditions
140  if (!joinAsGroup)
141  {
142  // check Deserter debuff
143  if (!_player->CanJoinToBattleground())
144  {
145  WorldPacket data;
146  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS);
147  _player->GetSession()->SendPacket(data);
148  return;
149  }
150 
151  // check if already in random queue
152  if (_player->GetBattleGroundQueueIndex(bgQueueTypeIdRandom) < PLAYER_MAX_BATTLEGROUND_QUEUES)
153  {
154  WorldPacket data;
155  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, ERR_IN_RANDOM_BG);
156  _player->GetSession()->SendPacket(data);
157  return;
158  }
159 
160  // check if already in normal queue; cannot start random queue
161  if(_player->InBattleGroundQueue() && bgTypeId == BATTLEGROUND_RB)
162  {
163  WorldPacket data;
164  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, ERR_IN_NON_RANDOM_BG);
165  _player->GetSession()->SendPacket(data);
166  return;
167  }
168 
169  // check if already in queue
170  if (_player->GetBattleGroundQueueIndex(bgQueueTypeId) < PLAYER_MAX_BATTLEGROUND_QUEUES)
171  // player is already in this queue
172  return;
173 
174  // check if has free queue slots
175  if (!_player->HasFreeBattleGroundQueueId())
176  {
177  WorldPacket data;
178  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, ERR_BATTLEGROUND_TOO_MANY_QUEUES);
179  _player->GetSession()->SendPacket(data);
180  return;
181  }
182  }
183  else
184  {
185  grp = _player->GetGroup();
186  // no group found, error
187  if (!grp)
188  return;
189  if (grp->GetLeaderGuid() != _player->GetObjectGuid())
190  return;
191  err = grp->CanJoinBattleGroundQueue(bg, bgQueueTypeId, 0, bg->GetMaxPlayersPerTeam(), false, 0);
192  isPremade = sWorld.getConfig(CONFIG_UINT32_BATTLEGROUND_PREMADE_GROUP_WAIT_FOR_MATCH) &&
193  (grp->GetMembersCount() >= bg->GetMinPlayersPerTeam());
194  }
195  // if we're here, then the conditions to join a bg are met. We can proceed in joining.
196 
197  // _player->GetGroup() was already checked, grp is already initialized
198  BattleGroundQueue& bgQueue = sBattleGroundMgr.m_battleGroundQueues[bgQueueTypeId];
199  if (joinAsGroup)
200  {
201  GroupQueueInfo* queueInfo = nullptr;
202  uint32 avgTime = 0;
203 
204  if (err > ERR_GROUP_JOIN_BATTLEGROUND_FAIL)
205  {
206  DEBUG_LOG("Battleground: the following players are joining as group:");
207  queueInfo = bgQueue.AddGroup(_player, grp, bgTypeId, bracketEntry, ARENA_TYPE_NONE, false, isPremade, instanceId, 0);
208  avgTime = bgQueue.GetAverageQueueWaitTime(queueInfo, bracketEntry->GetBracketId());
209  }
210 
211  for (GroupReference* itr = grp->GetFirstMember(); itr != nullptr; itr = itr->next())
212  {
213  Player* member = itr->getSource();
214  if (!member)
215  continue; // this should never happen
216 
217  WorldPacket data;
218 
219  if (err <= ERR_GROUP_JOIN_BATTLEGROUND_FAIL)
220  {
221  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, err);
222  member->GetSession()->SendPacket(data);
223  continue;
224  }
225 
226  // add to queue
227  uint32 queueSlot = member->AddBattleGroundQueueId(bgQueueTypeId);
228 
229  // send status packet (in queue)
230  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, queueInfo->arenaType, TEAM_NONE);
231  member->GetSession()->SendPacket(data);
232  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, err);
233  member->GetSession()->SendPacket(data);
234  DEBUG_LOG("Battleground: player joined queue for bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, member->GetGUIDLow(), member->GetName());
235  }
236  DEBUG_LOG("Battleground: group end");
237  }
238  else
239  {
240  GroupQueueInfo* queueInfo = bgQueue.AddGroup(_player, nullptr, bgTypeId, bracketEntry, ARENA_TYPE_NONE, false, isPremade, instanceId, 0);
241  uint32 avgTime = bgQueue.GetAverageQueueWaitTime(queueInfo, bracketEntry->GetBracketId());
242  // already checked if queueSlot is valid, now just get it
243  uint32 queueSlot = _player->AddBattleGroundQueueId(bgQueueTypeId);
244 
245  WorldPacket data;
246  // send status packet (in queue)
247  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, queueInfo->arenaType, TEAM_NONE);
248  SendPacket(data);
249  DEBUG_LOG("Battleground: player joined queue for bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, _player->GetGUIDLow(), _player->GetName());
250  }
251 
252  sBattleGroundMgr.ScheduleQueueUpdate(0, ARENA_TYPE_NONE, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
253 }

255 // Sent by client while inside battleground; depends on the battleground type
256 void WorldSession::HandleBattleGroundPlayerPositionsOpcode(WorldPacket& /*recv_data*/)
257 {
258  // empty opcode
259  DEBUG_LOG("WORLD: Received opcode MSG_BATTLEGROUND_PLAYER_POSITIONS");
260 
261  BattleGround* bg = _player->GetBattleGround();
262  if (!bg) // can't be received if player not in battleground
263  return;
264 
265  switch (bg->GetTypeId())
266  {
267  case BATTLEGROUND_WS:
268  {
269  uint32 flagCarrierCount = 0;
270 
271  Player* flagCarrierAlliance = sObjectMgr.GetPlayer(static_cast<BattleGroundWS*>(bg)->GetFlagCarrierGuid(TEAM_INDEX_ALLIANCE));
272  if (flagCarrierAlliance)
273  {
274  if (flagCarrierAlliance->GetTeam() == _player->GetTeam() || static_cast<BattleGroundWS*>(bg)->IsFlagHeldFor45Seconds(flagCarrierAlliance->GetTeam()))
275  ++flagCarrierCount;
276  }
277 
278  Player* flagCarrierHorde = sObjectMgr.GetPlayer(static_cast<BattleGroundWS*>(bg)->GetFlagCarrierGuid(TEAM_INDEX_HORDE));
279  if (flagCarrierHorde)
280  {
281  if (flagCarrierHorde->GetTeam() == _player->GetTeam() || static_cast<BattleGroundWS*>(bg)->IsFlagHeldFor45Seconds(flagCarrierHorde->GetTeam()))
282  ++flagCarrierCount;
283  }
284 
285  WorldPacket data(MSG_BATTLEGROUND_PLAYER_POSITIONS, 4 + 4 + 16 * flagCarrierCount);
286  data << uint32(0); // own team position XYZ
287  data << uint32(flagCarrierCount);
288 
289  if (flagCarrierAlliance)
290  {
291  data << flagCarrierAlliance->GetObjectGuid();
292  data << float(flagCarrierAlliance->GetPositionX());
293  data << float(flagCarrierAlliance->GetPositionY());
294  }
295  if (flagCarrierHorde)
296  {
297  data << flagCarrierHorde->GetObjectGuid();
298  data << float(flagCarrierHorde->GetPositionX());
299  data << float(flagCarrierHorde->GetPositionY());
300  }
301 
302  SendPacket(data);
303  break;
304  }
305  case BATTLEGROUND_EY:
306  {
307  uint32 flagCarrierCount = 0;
308 
309  Player* flagCarrier = sObjectMgr.GetPlayer(((BattleGroundEY*)bg)->GetFlagCarrierGuid());
310  if (flagCarrier)
311  flagCarrierCount = 1;
312 
313  WorldPacket data(MSG_BATTLEGROUND_PLAYER_POSITIONS, 4 + 4 + 16 * flagCarrierCount);
314  data << uint32(0);
315  data << uint32(flagCarrierCount);
316 
317  if (flagCarrier)
318  {
319  data << flagCarrier->GetObjectGuid();
320  data << float(flagCarrier->GetPositionX());
321  data << float(flagCarrier->GetPositionY());
322  }
323 
324  SendPacket(data);
325  break;
326  }
327  case BATTLEGROUND_AB:
328  case BATTLEGROUND_AV:
329  {
330  // for other BG types - send default
331  WorldPacket data(MSG_BATTLEGROUND_PLAYER_POSITIONS, 4 + 4);
332  data << uint32(0);
333  data << uint32(0);
334  SendPacket(data);
335  break;
336  }
337  default:
338  // maybe it is sent also in arena - do nothing
339  break;
340  }
341 }

389 // Sent by client when requesting teleport to the battleground location
390 void WorldSession::HandleBattlefieldPortOpcode(WorldPacket& recv_data)
391 {
392  DEBUG_LOG("WORLD: Received opcode CMSG_BATTLEFIELD_PORT");
393 
394  uint8 type; // arenatype if arena
395  uint8 unk2; // unk, can be 0x0 (may be if was invited?) and 0x1
396  uint32 receivedBgTypeId; // type id from dbc
397  uint16 unk; // 0x1F90 constant?
398  uint8 action; // enter battle 0x1, leave queue 0x0
399 
400  recv_data >> type >> unk2 >> receivedBgTypeId >> unk >> action;
401 
402  if (!sBattlemasterListStore.LookupEntry(receivedBgTypeId))
403  {
404  sLog.outError("BattlegroundHandler: invalid bgtype (%u) received.", receivedBgTypeId);
405  return;
406  }
407 
408  if (type && !IsArenaTypeValid(ArenaType(type)))
409  {
410  sLog.outError("BattlegroundHandler: Invalid CMSG_BATTLEFIELD_PORT received from player (%u), arena type wrong: %u.", _player->GetGUIDLow(), type);
411  return;
412  }
413 
414  if (!_player->InBattleGroundQueue())
415  {
416  sLog.outError("BattlegroundHandler: Invalid CMSG_BATTLEFIELD_PORT received from player (%u), he is not in bg_queue.", _player->GetGUIDLow());
417  return;
418  }
419 
420  // get GroupQueueInfo from BattleGroundQueue
421  BattleGroundTypeId bgTypeId = BattleGroundTypeId(receivedBgTypeId);
422  BattleGroundQueueTypeId bgQueueTypeId = BattleGroundMgr::BgQueueTypeId(bgTypeId, ArenaType(type));
423  BattleGroundQueue& bgQueue = sBattleGroundMgr.m_battleGroundQueues[bgQueueTypeId];
424 
425  // we must use temporary variable, because GroupQueueInfo pointer can be deleted in BattleGroundQueue::RemovePlayer() function
426  GroupQueueInfo queueInfo;
427  if (!bgQueue.GetPlayerGroupInfoData(_player->GetObjectGuid(), &queueInfo))
428  {
429  sLog.outError("BattlegroundHandler: itrplayerstatus not found.");
430  return;
431  }
432 
433  // if action == 1, then instanceId is required
434  if (!queueInfo.isInvitedToBgInstanceGuid && action == 1)
435  {
436  sLog.outError("BattlegroundHandler: instance not found.");
437  return;
438  }
439 
440  BattleGround* bg = sBattleGroundMgr.GetBattleGround(queueInfo.isInvitedToBgInstanceGuid, bgTypeId);
441 
442  // bg template might and must be used in case of leaving queue, when instance is not created yet
443  if (!bg && action == 0)
444  bg = sBattleGroundMgr.GetBattleGroundTemplate(bgTypeId);
445  if (!bg)
446  {
447  sLog.outError("BattlegroundHandler: bg_template not found for type id %u.", bgTypeId);
448  return;
449  }
450 
451  // expected bracket entry
452  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->GetLevel());
453  if (!bracketEntry)
454  return;
455 
456  // some checks if player isn't cheating - it is not exactly cheating, but we cannot allow it
457  if (action == 1 && queueInfo.arenaType == ARENA_TYPE_NONE)
458  {
459  // if player is trying to enter battleground (not arena!) and he has deserter debuff, we must just remove him from queue
460  if (!_player->CanJoinToBattleground())
461  {
462  // send bg command result to show nice message
463  WorldPacket data2;
464  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data2, ERR_GROUP_JOIN_BATTLEGROUND_DESERTERS);
465  _player->GetSession()->SendPacket(data2);
466  action = 0;
467 
468  DEBUG_LOG("Battleground: player %s (%u) has a deserter debuff, do not port him to battleground!", _player->GetName(), _player->GetGUIDLow());
469  }
470  // if player don't match battleground max level, then do not allow him to enter! (this might happen when player leveled up during his waiting in queue
471  if (_player->GetLevel() > bg->GetMaxLevel())
472  {
473  sLog.outError("Battleground: Player %s (%u) has level (%u) higher than maxlevel (%u) of battleground (%u)! Do not port him to battleground!",
474  _player->GetName(), _player->GetGUIDLow(), _player->GetLevel(), bg->GetMaxLevel(), bg->GetTypeId());
475  action = 0;
476  }
477  }
478 
479  uint32 queueSlot = _player->GetBattleGroundQueueIndex(bgQueueTypeId);
480  WorldPacket data;
481 
482  switch (action)
483  {
484  case 1: // port to battleground
485  if (!_player->IsInvitedForBattleGroundQueueType(bgQueueTypeId))
486  return; // cheating?
487 
488  if (!_player->InBattleGround())
489  _player->SetBattleGroundEntryPoint();
490 
491  // resurrect the player
492  if (!_player->IsAlive())
493  {
494  _player->ResurrectPlayer(1.0f);
495  _player->SpawnCorpseBones();
496  }
497 
498  _player->TaxiFlightInterrupt(false);
499 
500  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_IN_PROGRESS, 0, bg->GetStartTime(), bg->GetArenaType(), _player->GetBGTeam());
501  _player->GetSession()->SendPacket(data);
502 
503  // remove battleground queue status from BGmgr
504  bgQueue.RemovePlayer(_player->GetObjectGuid(), false);
505 
506  // this is still needed here if battleground "jumping" shouldn't add deserter debuff
507  // also this is required to prevent stuck at old battleground after SetBattleGroundId set to new
508  if (BattleGround* currentBg = _player->GetBattleGround())
509  currentBg->RemovePlayerAtLeave(_player->GetObjectGuid(), false, true);
510 
511  // set the destination instance id
512  _player->SetBattleGroundId(bg->GetInstanceId(), bgTypeId);
513  // set the destination team
514  _player->SetBGTeam(queueInfo.groupTeam);
515  // bg->HandleBeforeTeleportToBattleGround(_player);
516  sBattleGroundMgr.SendToBattleGround(_player, queueInfo.isInvitedToBgInstanceGuid, bgTypeId);
517  // add only in HandleMoveWorldPortAck()
518  // bg->AddPlayer(_player,team);
519 
520  DEBUG_LOG("Battleground: player %s (%u) joined battle for bg %u, bgtype %u, queue type %u.", _player->GetName(), _player->GetGUIDLow(), bg->GetInstanceId(), bg->GetTypeId(), bgQueueTypeId);
521 
522  break;
523  case 0: // leave queue
524  // if player leaves rated arena match before match start, it is counted as he played but he lost
525  if (queueInfo.isRated && queueInfo.isInvitedToBgInstanceGuid)
526  {
527  ArenaTeam* at = sObjectMgr.GetArenaTeamById(queueInfo.arenaTeamId);
528  if (at)
529  {
530  DEBUG_LOG("UPDATING memberLost's personal arena rating for %s by opponents rating: %u, because he has left queue!", _player->GetGuidStr().c_str(), queueInfo.opponentsTeamRating);
531  at->MemberLost(_player, queueInfo.opponentsTeamRating);
532  at->SaveToDB();
533  }
534  }
535 
536  _player->RemoveBattleGroundQueueId(bgQueueTypeId); // must be called this way, because if you move this call to queue->removeplayer, it causes bugs
537  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_NONE, 0, 0, ARENA_TYPE_NONE, TEAM_NONE);
538  bgQueue.RemovePlayer(_player->GetObjectGuid(), true);
539 
540  // player left queue, we should update it - do not update Arena Queue
541  if (queueInfo.arenaType == ARENA_TYPE_NONE)
542  sBattleGroundMgr.ScheduleQueueUpdate(queueInfo.arenaTeamRating, queueInfo.arenaType, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
543 
544  SendPacket(data);
545 
546  DEBUG_LOG("Battleground: player %s (%u) left queue for bgtype %u, queue type %u.", _player->GetName(), _player->GetGUIDLow(), bg->GetTypeId(), bgQueueTypeId);
547 
548  break;
549  default:
550  sLog.outError("Battleground port: unknown action %u", action);
551  break;
552  }
553 }

577 // Sent by client when requesting battleground status
578 void WorldSession::HandleBattlefieldStatusOpcode(WorldPacket& /*recv_data*/)
579 {
580  // empty opcode
581  DEBUG_LOG("WORLD: Battleground status");
582 
583  WorldPacket data;
584  // we must update all queues here
585  BattleGround* bg;
586  for (uint8 i = 0; i < PLAYER_MAX_BATTLEGROUND_QUEUES; ++i)
587  {
588  BattleGroundQueueTypeId bgQueueTypeId = _player->GetBattleGroundQueueTypeId(i);
589  if (!bgQueueTypeId)
590  continue;

691 // Sent by client when requesting arena join
692 void WorldSession::HandleBattlemasterJoinArena(WorldPacket& recv_data)
693 {
694  DEBUG_LOG("WORLD: CMSG_BATTLEMASTER_JOIN_ARENA");
695  // recv_data.hexlike();
696 
697  ObjectGuid guid; // arena Battlemaster guid
698  uint8 arenaslot; // 2v2, 3v3 or 5v5
699  uint8 asGroup; // asGroup
700  uint8 isRated; // isRated
701 
702  recv_data >> guid >> arenaslot >> asGroup >> isRated;
703 
704  // ignore if we already in BG or BG queue
705  if (_player->InBattleGround())
706  return;
707 
708  Creature* unit = GetPlayer()->GetMap()->GetCreature(guid);
709  if (!unit)
710  return;
711 
712  if (!unit->isBattleMaster()) // it's not battle master
713  return;
714 
715  ArenaType arenatype;
716  uint32 arenaRating = 0;
717 
718  switch (arenaslot)
719  {
720  case 0:
721  arenatype = ARENA_TYPE_2v2;
722  break;
723  case 1:
724  arenatype = ARENA_TYPE_3v3;
725  break;
726  case 2:
727  arenatype = ARENA_TYPE_5v5;
728  break;
729  default:
730  sLog.outError("Unknown arena slot %u at HandleBattlemasterJoinArena()", arenaslot);
731  return;
732  }
733 
734  // check existence
735  BattleGround* bg = sBattleGroundMgr.GetBattleGroundTemplate(BATTLEGROUND_AA);
736  if (!bg)
737  {
738  sLog.outError("Battleground: template bg (all arenas) not found");
739  return;
740  }
741 
742  BattleGroundTypeId bgTypeId = bg->GetTypeId();
743  BattleGroundQueueTypeId bgQueueTypeId = BattleGroundMgr::BgQueueTypeId(bgTypeId, arenatype);
744  PvPDifficultyEntry const* bracketEntry = GetBattlegroundBracketByLevel(bg->GetMapId(), _player->GetLevel());
745  if (!bracketEntry)
746  return;
747 
748  GroupJoinBattlegroundResult err = ERR_GROUP_JOIN_BATTLEGROUND_FAIL;
749 
750  Group* group = nullptr;
751 
752  // check queue conditions
753  if (!asGroup)
754  {
755  // you can't join in this way by client
756  if (isRated)
757  return;
758 
759  // check if already in queue
760  if (_player->GetBattleGroundQueueIndex(bgQueueTypeId) < PLAYER_MAX_BATTLEGROUND_QUEUES)
761  // player is already in this queue
762  return;
763  // check if has free queue slots
764  if (!_player->HasFreeBattleGroundQueueId())
765  return;
766  }
767  else
768  {
769  group = _player->GetGroup();
770  // no group found, error
771  if (!group)
772  return;
773 
774  if (group->GetLeaderGuid() != _player->GetObjectGuid())
775  return;
776 
777  // may be Group::CanJoinBattleGroundQueue should be moved to player class...
778  err = group->CanJoinBattleGroundQueue(bg, bgQueueTypeId, arenatype, arenatype, isRated != 0, arenaslot);
779  }
780 
781  uint32 ateamId = 0;
782 
783  if (isRated)
784  {
785  ateamId = _player->GetArenaTeamId(arenaslot);
786  // check real arena team existence only here (if it was moved to group->CanJoin .. () then we would have to get it twice)
787  ArenaTeam* at = sObjectMgr.GetArenaTeamById(ateamId);
788  if (!at)
789  {
790  _player->GetSession()->SendNotInArenaTeamPacket(arenatype);
791  return;
792  }
793  // get the team rating for queue
794  arenaRating = at->GetRating();
795  // the arena team id must match for everyone in the group
796  // get the personal ratings for queue
797  uint32 avg_pers_rating = 0;
798 
799  for (const auto& citr : group->GetMemberSlots())
800  {
801  ArenaTeamMember const* at_member = at->GetMember(citr.guid);
802  if (!at_member) // group member joining to arena must be in leader arena team
803  return;
804 
805  // calc avg personal rating
806  avg_pers_rating += at_member->personal_rating;
807  }
808 
809  avg_pers_rating /= group->GetMembersCount();
810 
811  // if avg personal rating is more than 150 points below the teams rating, the team will be queued against an opponent matching or similar to the average personal rating
812  if (avg_pers_rating + 150 < arenaRating)
813  arenaRating = avg_pers_rating;
814  }
815 
816  BattleGroundQueue& bgQueue = sBattleGroundMgr.m_battleGroundQueues[bgQueueTypeId];
817  if (asGroup)
818  {
819  uint32 avgTime = 0;
820 
821  if (err > ERR_GROUP_JOIN_BATTLEGROUND_FAIL)
822  {
823  DEBUG_LOG("Battleground: arena join as group start");
824  if (isRated)
825  DEBUG_LOG("Battleground: arena team id %u, leader %s queued with rating %u for type %u", _player->GetArenaTeamId(arenaslot), _player->GetName(), arenaRating, arenatype);
826 
827  // set arena rated type to show correct minimap arena icon
828  bg->SetRated(isRated != 0);
829 
830  GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, group, bgTypeId, bracketEntry, arenatype, isRated != 0, false, 0, arenaRating, ateamId);
831  avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
832  }
833 
834  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
835  {
836  Player* member = itr->getSource();
837  if (!member)
838  continue;
839 
840  WorldPacket data;
841 
842  if (err <= ERR_GROUP_JOIN_BATTLEGROUND_FAIL)
843  {
844  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, err);
845  member->GetSession()->SendPacket(data);
846  continue;
847  }
848 
849  // add to queue
850  uint32 queueSlot = member->AddBattleGroundQueueId(bgQueueTypeId);
851 
852  // send status packet (in queue)
853  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, arenatype, TEAM_NONE);
854  member->GetSession()->SendPacket(data);
855  sBattleGroundMgr.BuildGroupJoinedBattlegroundPacket(data, err);
856  member->GetSession()->SendPacket(data);
857  DEBUG_LOG("Battleground: player joined queue for arena as group bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, member->GetGUIDLow(), member->GetName());
858  }
859  DEBUG_LOG("Battleground: arena join as group end");
860  }
861  else
862  {
863  GroupQueueInfo* ginfo = bgQueue.AddGroup(_player, nullptr, bgTypeId, bracketEntry, arenatype, isRated != 0, false, 0, arenaRating, ateamId);
864  uint32 avgTime = bgQueue.GetAverageQueueWaitTime(ginfo, bracketEntry->GetBracketId());
865  uint32 queueSlot = _player->AddBattleGroundQueueId(bgQueueTypeId);
866 
867  WorldPacket data;
868  // send status packet (in queue)
869  sBattleGroundMgr.BuildBattleGroundStatusPacket(data, bg, queueSlot, STATUS_WAIT_QUEUE, avgTime, 0, arenatype, TEAM_NONE);
870  SendPacket(data);
871  DEBUG_LOG("Battleground: player joined queue for arena, skirmish, bg queue type %u bg type %u: GUID %u, NAME %s", bgQueueTypeId, bgTypeId, _player->GetGUIDLow(), _player->GetName());
872  }
873  sBattleGroundMgr.ScheduleQueueUpdate(arenaRating, arenatype, bgQueueTypeId, bgTypeId, bracketEntry->GetBracketId());
874 }
