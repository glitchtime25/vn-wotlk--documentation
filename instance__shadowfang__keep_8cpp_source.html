73 void instance_shadowfang_keep::OnCreatureCreate(Creature* creature)
74 {
75  switch (creature->GetEntry())
76  {
77  case NPC_ASH:
78  case NPC_ADA:
79  case NPC_FENRUS:
80  case NPC_HUMMEL:
81  case NPC_FRYE:
82  case NPC_BAXTER:
83  case NPC_APOTHECARY_GENERATOR:
84  case NPC_VALENTINE_BOSS_MGR:
85  case NPC_MASTER_NANDOS:
86  break;
87  case NPC_VINCENT:
88  // If Arugal has done the intro, make Vincent dead!
89  if (m_auiEncounter[4] == DONE)
90  creature->SetStandState(UNIT_STAND_STATE_DEAD);
91  else
92  StartNextDialogueText(NPC_VINCENT);
93  break;
94  case NPC_LUPINE_HORROR:
95  case NPC_WOLFGUARD_WORG:
96  case NPC_BLEAK_WORG:
97  case NPC_SLAVERING_WORG:
98  // Only store the wolves/worgs that are static spawn on the top level of the instance
99  if (creature->GetPositionZ() > nandosMovement.fZ && !creature->IsTemporarySummon())
100  m_lNandosWolvesGuids.push_back(creature->GetObjectGuid());
101  break;
102  case NPC_CROWN_APOTHECARY:
103  m_lCrownApothecaryGuids.push_back(creature->GetObjectGuid());
104  return;
105  case NPC_ARUGAL:
106  creature->SetVisibility(VISIBILITY_OFF);
107  break;
108  case NPC_ARCHMAGE_ARUGAL:
109  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
110  break;
111  default:
112  return;
113  }
114  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
115 }

146 void instance_shadowfang_keep::OnCreatureDeath(Creature* creature)
147 {
148  switch (creature->GetEntry())
149  {
150  // Instance data is set to SPECIAL because the encounter depends on multiple bosses
151  case NPC_FRYE:
152  case NPC_BAXTER:
153  SetData(TYPE_APOTHECARY, SPECIAL);
154  break;
155  case NPC_LUPINE_HORROR:
156  case NPC_WOLFGUARD_WORG:
157  case NPC_BLEAK_WORG:
158  case NPC_SLAVERING_WORG:
159  if (m_lNandosWolvesGuids.empty())
160  return;
161 
162  m_lNandosWolvesGuids.remove(creature->GetObjectGuid());
163  // If all wolves are dead, make Wolf Master Nandos engage in combat
164  if (m_lNandosWolvesGuids.empty())
165  {
166  if (Creature* nandos = GetSingleCreatureFromStorage(NPC_MASTER_NANDOS))
167  {
168  if (nandos->IsInCombat()) // Wolf Master Nandos already joined the fight: no need to go further
169  return;
170  DoScriptText(YELL_PACK_DEAD, nandos);
171  nandos->SetWalk(false);
172  nandos->GetMotionMaster()->MovePoint(0, nandosMovement.fX, nandosMovement.fY, nandosMovement.fZ);
173  }
174  }
175  break;
176  default:
177  return;
178  }
179 }

223 void instance_shadowfang_keep::SetData(uint32 type, uint32 data)
224 {
225  switch (type)
226  {
227  case TYPE_FREE_NPC:
228  if (data == DONE)
229  DoUseDoorOrButton(GO_COURTYARD_DOOR);
230  m_auiEncounter[0] = data;
231  break;
232  case TYPE_RETHILGORE:
233  if (data == DONE)
234  DoSpeech();
235  m_auiEncounter[1] = data;
236  break;
237  case TYPE_FENRUS:
238  if (data == DONE)
239  {
240  if (Creature* fenrus = GetSingleCreatureFromStorage(NPC_FENRUS))
241  {
242  fenrus->SummonCreature(NPC_ARCHMAGE_ARUGAL, -136.89f, 2169.17f, 136.58f, 2.794f, TEMPSPAWN_TIMED_DESPAWN, 30000);
243  StartNextDialogueText(NPC_ARCHMAGE_ARUGAL);
244  }
245  }
246  m_auiEncounter[2] = data;
247  break;
248  case TYPE_NANDOS:
249  if (data == DONE)
250  DoUseDoorOrButton(GO_ARUGAL_DOOR);
251  m_auiEncounter[3] = data;
252  break;
253  case TYPE_INTRO:
254  m_auiEncounter[4] = data;
255  break;
256  case TYPE_VOIDWALKER:
257  if (data == DONE)
258  {
259  m_auiEncounter[5]++;
260  if (m_auiEncounter[5] > 3)
261  DoUseDoorOrButton(GO_SORCERER_DOOR);
262  }
263  break;
264  case TYPE_APOTHECARY:
265  // Reset apothecary counter on fail
266  if (data == IN_PROGRESS)
267  m_uiApothecaryDead = 0;
268  else if (data == FAIL)
269  {
270  // despawn bosses and reset on timer
271  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_HUMMEL))
272  pBoss->ForcedDespawn();
273  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_BAXTER))
274  pBoss->ForcedDespawn();
275  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_FRYE))
276  pBoss->ForcedDespawn();
277 
278  m_uiApothecaryResetTimer = 30000;
279  }
280 
281  // count dead apothecaries
282  if (data == SPECIAL)
283  {
284  ++m_uiApothecaryDead;
285 
286  // Set Hummel as lootable only when the others are dead
287  if (m_uiApothecaryDead == MAX_APOTHECARY)
288  {
289  if (Creature* pHummel = GetSingleCreatureFromStorage(NPC_HUMMEL))
290  {
291  pHummel->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
292  pHummel->Suicide();
293  }
294 
295  SetData(TYPE_APOTHECARY, DONE);
296  instance->AwardLFGRewards(SEASONAL_CROWN_CHEMICAL_CO);
297  }
298  }
299  // We don't want to store the SPECIAL data
300  else
301  m_auiEncounter[6] = data;
302  break;
303  }
304 
305  if (data == DONE)
306  {
307  OUT_SAVE_INST_DATA;
308 
309  std::ostringstream saveStream;
310  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3]
311  << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " " << m_auiEncounter[6];
312 
313  m_strInstData = saveStream.str();
314 
315  SaveToDB();
316  OUT_SAVE_INST_DATA_COMPLETE;
317  }
318 }

359 void instance_shadowfang_keep::Update(uint32 uiDiff)
360 {
361  DialogueUpdate(uiDiff);
362 
363  if (m_uiApothecaryResetTimer)
364  {
365  if (m_uiApothecaryResetTimer <= uiDiff)
366  {
367  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_HUMMEL))
368  pBoss->Respawn();
369  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_BAXTER))
370  pBoss->Respawn();
371  if (Creature* pBoss = GetSingleCreatureFromStorage(NPC_FRYE))
372  pBoss->Respawn();
373 
374  for (const auto& guid : m_lCrownApothecaryGuids)
375  if (Creature* pApothecary = instance->GetCreature(guid))
376  pApothecary->Respawn();
377  }
378  else
379  m_uiApothecaryResetTimer -= uiDiff;

384 void instance_shadowfang_keep::JustDidDialogueStep(int32 entry)
385 {
386  switch(entry)
387  {
388  case VINCENT_DEATH:
389  if (Creature* vincent = GetSingleCreatureFromStorage(NPC_VINCENT))
390  vincent->SetStandState(UNIT_STAND_STATE_DEAD);
391  break;
392  case ARUGAL_VISIBLE:
393  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
394  creature->SetVisibility(VISIBILITY_ON);
395  break;
396  case ARUGAL_TELEPORT_IN:
397  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
398  creature->AI()->DoCastSpellIfCan(creature, SPELL_SPAWN);
399  break;
400  case ARUGAL_TURN_TO_VINCENT:
401  if (Creature* creatureA = GetSingleCreatureFromStorage(NPC_ARUGAL))
402  if (Creature* creatureV = GetSingleCreatureFromStorage(NPC_VINCENT))
403  creatureA->SetFacingToObject(creatureV);
404  break;
405  case ARUGAL_EMOTE_POINT:
406  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
407  creature->HandleEmote(EMOTE_ONESHOT_POINT);
408  break;
409  case ARUGAL_EMOTE_EXCLAMATION:
410  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
411  creature->HandleEmote(EMOTE_ONESHOT_EXCLAMATION);
412  break;
413  case ARUGAL_EMOTE_LAUGH:
414  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
415  creature->HandleEmote(EMOTE_ONESHOT_LAUGH);
416  break;
417  case ARUGAL_TELEPORT_OUT:
418  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
419  creature->AI()->DoCastSpellIfCan(creature, SPELL_ARUGAL_TELEPORT);
420  break;
421  case ARUGAL_INTRO_DONE:
422  SetData(TYPE_INTRO, DONE);
423  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARUGAL))
424  {
425  creature->SetVisibility(VISIBILITY_OFF);
426  creature->ForcedDespawn();
427  }
428  break;
429  case YELL_FENRUS:
430  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARCHMAGE_ARUGAL))
431  creature->SetVisibility(VISIBILITY_ON);
432  break;
433  case ARCHMAGE_FIRE:
434  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARCHMAGE_ARUGAL))
435  creature->AI()->DoCastSpellIfCan(creature, SPELL_FIRE);
436  break;
437  case ARCHMAGE_LIGHTNING:
438  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARCHMAGE_ARUGAL))
439  {
440  if (GameObject* lightning = GetSingleGameObjectFromStorage(GO_ARUGAL_FOCUS))
441  lightning->Use(creature);
442  }
443  break;
444  case ARCHMAGE_INVIS:
445  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARCHMAGE_ARUGAL))
446  creature->SetVisibility(VISIBILITY_OFF);
447  break;
448  case ARCHMAGE_VOIDWALKERS:
449  if (Creature* creature = GetSingleCreatureFromStorage(NPC_ARCHMAGE_ARUGAL))
450  creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, creature, creature);
451  break;
452  default:
453  break;
454  }
455 }
