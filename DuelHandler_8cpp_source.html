26 {
27  ObjectGuid guid;
28  recvPacket >> guid;
29 
30  // Check for own duel info first
31  Player* self = GetPlayer();
32  if (!self || !self->duel)
33  return;
34 
35  // Check if we are not accepting our own duel request
36  Player* initiator = self->duel->initiator;
37  if (!initiator || self == initiator)
38  return;
39 
40  // Check for opponent
41  Player* opponent = self->duel->opponent;
42  if (!opponent || self == opponent)
43  return;
44 
45  // Check if duel is starting
46  if (self->duel->startTimer != 0 || opponent->duel->startTimer != 0)
47  return;
48 
49  // Check if duel is in progress
50  if (self->duel->startTime != 0 || opponent->duel->startTime != 0)
51  return;
52 
53  DEBUG_FILTER_LOG(LOG_FILTER_COMBAT, "WORLD: received CMSG_DUEL_ACCEPTED");
54  DEBUG_FILTER_LOG(LOG_FILTER_COMBAT, "Player 1 is: %u (%s)", self->GetGUIDLow(), self->GetName());
55  DEBUG_FILTER_LOG(LOG_FILTER_COMBAT, "Player 2 is: %u (%s)", opponent->GetGUIDLow(), opponent->GetName());
56 
57  time_t now = time(nullptr);
58  self->duel->startTimer = now;
59  opponent->duel->startTimer = now;
60 
61  self->SendDuelCountdown(3000);

63 }
