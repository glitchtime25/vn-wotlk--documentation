153  void ExecuteAction(uint32 action) override
154  {
155  switch (action)
156  {
157  case SHAHRAZ_ACTION_FRENZY:
158  {
159  if (m_creature->GetHealthPercent() < 10.0f)
160  {
161  if (DoCastSpellIfCan(nullptr, SPELL_RANDOM_PERIODIC) == CAST_OK)
162  {
163  DoScriptText(SAY_ENRAGE, m_creature);
164  DisableCombatAction(SHAHRAZ_ACTION_BEAM);
165  SetActionReadyStatus(action, false); // timerless
166  }
167  }
168  return;
169  }
170  case SHAHRAZ_ACTION_BERSERK:
171  {
172  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
173  {
174  DoScriptText(SAY_ENRAGE, m_creature);
175  DisableCombatAction(action);
176  }
177  return;
178  }
179  case SHAHRAZ_ACTION_BEAM:
180  {
181  m_currentBeam = (m_currentBeam + urand(1, 3)) % 4;
182  if (DoCastSpellIfCan(nullptr, aPeriodicBeams[m_currentBeam]) == CAST_OK)
183  {
184  switch (urand(0, 2))
185  {
186  case 0: DoScriptText(SAY_BEAM_1, m_creature); break;
187  case 1: DoScriptText(SAY_BEAM_2, m_creature); break;
188  case 2: DoScriptText(SAY_BEAM_3, m_creature); break;
189  }
190  ResetCombatAction(action, GetSubsequentActionTimer(ShahrazActions(action)));
191  }
192  return;
193  }
194  case SHAHRAZ_ACTION_PRISMATIC_SHIELD:
195  {
196  if (DoCastSpellIfCan(nullptr, aPrismaticAuras[urand(0, 5)]) == CAST_OK)
197  ResetCombatAction(action, GetSubsequentActionTimer(ShahrazActions(action)));
198  return;
199  }
200  case SHAHRAZ_ACTION_FATAL_ATTRACTION:
201  {
202  if (m_creature->getThreatManager().getThreatList().size() < 3)
203  return;
204  if (DoCastSpellIfCan(nullptr, SPELL_FATAL_ATTRACTION) == CAST_OK)
205  {
206  switch (urand(0, 2))
207  {
208  case 0: DoScriptText(SAY_ATTRACTION_1, m_creature); break;
209  case 1: DoScriptText(SAY_ATTRACTION_3, m_creature); break;
210  case 2: DoScriptText(SAY_ATTRACTION_3, m_creature); break;
211  }
212  ResetCombatAction(action, GetSubsequentActionTimer(ShahrazActions(action)));
213  }
214  return;
215  }
216  case SHAHRAZ_ACTION_SHRIEK:
217  {
218  if (DoCastSpellIfCan(nullptr, SPELL_SILENCING_SHRIEK) == CAST_OK)
219  ResetCombatAction(action, GetSubsequentActionTimer(ShahrazActions(action)));
220  return;
221  }
222  }
223  }
