97 void Corpse::SaveToDB()
98 {
99  // bones should not be saved to DB (would be deleted on startup anyway)
100  MANGOS_ASSERT(GetType() != CORPSE_BONES);
101 
102  // prevent DB data inconsistence problems and duplicates
103  CharacterDatabase.BeginTransaction();
104  DeleteFromDB();
105 
106  std::ostringstream ss;
107  ss << "INSERT INTO corpse (guid,player,position_x,position_y,position_z,orientation,map,time,corpse_type,instance,phaseMask) VALUES ("
108  << GetGUIDLow() << ", "
109  << GetOwnerGuid().GetCounter() << ", "
110  << GetPositionX() << ", "
111  << GetPositionY() << ", "
112  << GetPositionZ() << ", "
113  << GetOrientation() << ", "
114  << GetMapId() << ", "
115  << uint64(m_time) << ", "
116  << uint32(GetType()) << ", "
117  << int(GetInstanceId()) << ", "
118  << uint16(GetPhaseMask()) << ")"; // prevent out of range error
119  CharacterDatabase.Execute(ss.str().c_str());
120  CharacterDatabase.CommitTransaction();

149 bool Corpse::LoadFromDB(uint32 lowguid, Field* fields)
150 {
152  // QueryResult *result = CharacterDatabase.Query("SELECT corpse.guid, player, corpse.position_x, corpse.position_y, corpse.position_z, corpse.orientation, corpse.map,"
154  // "time, corpse_type, instance, phaseMask, gender, race, class, playerBytes, playerBytes2, equipmentCache, guildId, playerFlags FROM corpse"
155  uint32 playerLowGuid = fields[1].GetUInt32();
156  float positionX = fields[2].GetFloat();
157  float positionY = fields[3].GetFloat();
158  float positionZ = fields[4].GetFloat();
159  float orientation = fields[5].GetFloat();
160  uint32 mapid = fields[6].GetUInt32();
161 
162  Object::_Create(lowguid, lowguid, 0, HIGHGUID_CORPSE);
163 
164  m_time = time_t(fields[7].GetUInt64());
165  m_type = CorpseType(fields[8].GetUInt32());
166 
167  if (m_type >= MAX_CORPSE_TYPE)
168  {
169  sLog.outError("%s Owner %s have wrong corpse type (%i), not load.", GetGuidStr().c_str(), GetOwnerGuid().GetString().c_str(), m_type);
170  return false;
171  }
172 
173  uint32 instanceid = fields[9].GetUInt32();
174  uint32 phaseMask = fields[10].GetUInt32();
175  uint8 gender = fields[11].GetUInt8();
176  uint8 race = fields[12].GetUInt8();
177  uint8 _class = fields[13].GetUInt8();
178  uint32 playerBytes = fields[14].GetUInt32();
179  uint32 playerBytes2 = fields[15].GetUInt32();
180  uint32 guildId = fields[17].GetUInt32();
181  uint32 playerFlags = fields[18].GetUInt32();
182 
183  ObjectGuid guid = ObjectGuid(HIGHGUID_CORPSE, lowguid);
184  ObjectGuid playerGuid = ObjectGuid(HIGHGUID_PLAYER, playerLowGuid);
185 
186  // overwrite possible wrong/corrupted guid
187  SetGuidValue(OBJECT_FIELD_GUID, guid);
188  SetOwnerGuid(playerGuid);
189 
190  SetObjectScale(DEFAULT_OBJECT_SCALE);
191 
192  PlayerInfo const* info = sObjectMgr.GetPlayerInfo(race, _class);
193  if (!info)
194  {
195  sLog.outError("Player %u has incorrect race/class pair.", GetGUIDLow());
196  return false;
197  }
198  SetUInt32Value(CORPSE_FIELD_DISPLAY_ID, gender == GENDER_FEMALE ? info->displayId_f : info->displayId_m);
199 
200  // Load equipment
201  Tokens data = StrSplit(fields[16].GetCppString(), " ");
202  for (uint8 slot = 0; slot < EQUIPMENT_SLOT_END; ++slot)
203  {
204  uint32 visualbase = slot * 2;
205  uint32 item_id = GetUInt32ValueFromArray(data, visualbase);
206  const ItemPrototype* proto = ObjectMgr::GetItemPrototype(item_id);
207  if (!proto)
208  {
209  SetUInt32Value(CORPSE_FIELD_ITEM + slot, 0);
210  continue;
211  }
212 
213  SetUInt32Value(CORPSE_FIELD_ITEM + slot, proto->DisplayInfoID | (proto->InventoryType << 24));
214  }
215 
216  uint8 skin = (uint8)(playerBytes);
217  uint8 face = (uint8)(playerBytes >> 8);
218  uint8 hairstyle = (uint8)(playerBytes >> 16);
219  uint8 haircolor = (uint8)(playerBytes >> 24);
220  uint8 facialhair = (uint8)(playerBytes2);
221  SetUInt32Value(CORPSE_FIELD_BYTES_1, ((0x00) | (race << 8) | (gender << 16) | (skin << 24)));
222  SetUInt32Value(CORPSE_FIELD_BYTES_2, ((face) | (hairstyle << 8) | (haircolor << 16) | (facialhair << 24)));
223 
224  SetUInt32Value(CORPSE_FIELD_GUILD, guildId);
225 
226  uint32 flags = CORPSE_FLAG_UNK2;
227  if (playerFlags & PLAYER_FLAGS_HIDE_HELM)
228  flags |= CORPSE_FLAG_HIDE_HELM;
229  if (playerFlags & PLAYER_FLAGS_HIDE_CLOAK)
230  flags |= CORPSE_FLAG_HIDE_CLOAK;
231  SetUInt32Value(CORPSE_FIELD_FLAGS, flags);
232 
233  // no need to mark corpse as lootable, because corpses are not saved in battle grounds
234 
235  // place
236  SetLocationInstanceId(instanceid);
237  SetLocationMapId(mapid);
238  SetPhaseMask(phaseMask, false);
239  Relocate(positionX, positionY, positionZ, orientation);
240 
241  if (!IsPositionValid())
242  {
243  sLog.outError("%s Owner %s not created. Suggested coordinates isn't valid (X: %f Y: %f)",
244  GetGuidStr().c_str(), GetOwnerGuid().GetString().c_str(), GetPositionX(), GetPositionY());
245  return false;
246  }
247 
248  m_grid = MaNGOS::ComputeGridPair(GetPositionX(), GetPositionY());
249 
250  return true;
251 }
