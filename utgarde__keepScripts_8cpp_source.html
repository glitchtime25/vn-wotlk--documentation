67  // Method to open / close the corresponding forge based on location
68  void SetMyForge()
69  {
70  if (!m_instance)
71  return;
72 
73  GameObjectList lGOList;
74  uint32 uiGOBellow = 0;
75  uint32 uiGOFire = 0;
76 
77  for (uint8 i = 0; i < MAX_FORGE; ++i)
78  {
79  switch (i)
80  {
81  case 0: uiGOBellow = GO_BELLOW_1; break;
82  case 1: uiGOBellow = GO_BELLOW_2; break;
83  case 2: uiGOBellow = GO_BELLOW_3; break;
84  }
85 
86  if (GameObject* pGOTemp = m_instance->GetSingleGameObjectFromStorage(uiGOBellow))
87  lGOList.push_back(pGOTemp);
88  }
89 
90  if (!lGOList.empty())
91  {
92  if (lGOList.size() != MAX_FORGE)
93  script_error_log("mob_dragonflayer_forge_master expected %u in lGOList, but does not match.", MAX_FORGE);
94 
95  lGOList.sort(ObjectDistanceOrder(m_creature));
96 
97  if (lGOList.front()->GetLootState() == GO_READY)
98  lGOList.front()->UseDoorOrButton(DAY);
99  else if (lGOList.front()->GetLootState() == GO_ACTIVATED)
100  lGOList.front()->ResetDoorOrButton();
101 
102  switch (lGOList.front()->GetEntry())
103  {
104  case GO_BELLOW_1: uiGOFire = GO_FORGEFIRE_1; m_uiForgeEncounterId = TYPE_BELLOW_1; break;
105  case GO_BELLOW_2: uiGOFire = GO_FORGEFIRE_2; m_uiForgeEncounterId = TYPE_BELLOW_2; break;
106  case GO_BELLOW_3: uiGOFire = GO_FORGEFIRE_3; m_uiForgeEncounterId = TYPE_BELLOW_3; break;
107  }
108 
109  if (GameObject* pGOTemp = m_instance->GetSingleGameObjectFromStorage(uiGOFire))
110  {
111  if (pGOTemp->GetLootState() == GO_READY)
112  pGOTemp->UseDoorOrButton(DAY);
113  else if (pGOTemp->GetLootState() == GO_ACTIVATED)
114  pGOTemp->ResetDoorOrButton();
115  }
116  }
117  }
