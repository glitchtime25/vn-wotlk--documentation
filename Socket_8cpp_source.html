93  void Socket::OnRead(const boost::system::error_code& error, size_t length)
94  {
95  if (error)
96  {
97  m_readState = ReadState::Idle;
98  OnError(error);
99  return;
100  }
101 
102  if (IsClosed())
103  {
104  m_readState = ReadState::Idle;
105  return;
106  }
107 
108  m_inBuffer->m_writePosition += length;
109 
110  const size_t available = m_socket.available();
111 
112  // if there is still data to read, increase the buffer size and do so (if necessary)
113  if (available > 0 && (length + available) > m_inBuffer->m_buffer.size())
114  {
115  m_inBuffer->m_buffer.resize(m_inBuffer->m_buffer.size() + available);
116  StartAsyncRead();
117  return;
118  }
119 
120  // we must repeat this in case we have read in multiple messages from the client
121  while (m_inBuffer->m_readPosition < m_inBuffer->m_writePosition)
122  {
123  if (!ProcessIncomingData())
124  {
125  // this errno is set when there is not enough buffer data available to either complete a header, or the packet length
126  // specified in the header goes past what we've read. in this case, we will reset the buffer with the remaining data
127  if (errno == EBADMSG)
128  {
129  const size_t bytesRemaining = m_inBuffer->m_writePosition - m_inBuffer->m_readPosition;
130 
131  ::memmove(&m_inBuffer->m_buffer[0], &m_inBuffer->m_buffer[m_inBuffer->m_readPosition], bytesRemaining);
132 
133  m_inBuffer->m_readPosition = 0;
134  m_inBuffer->m_writePosition = bytesRemaining;
135 
136  StartAsyncRead();
137  }
138  else if (!IsClosed())
139  Close();
140 
141  return;
142  }
143  }
144 
145  // at this point, the packet has been read and successfully processed. reset the buffer.
146  m_inBuffer->m_writePosition = m_inBuffer->m_readPosition = 0;
147 
148  StartAsyncRead();
149  }

254  void Socket::OnWriteComplete(const boost::system::error_code& error, size_t length)
255  {
256  // we must check this before locking the mutex because the connection will be closed,
257  // which leads to a locked mutex being destroyed. not good!
258  if (error)
259  {
260  OnError(error);
261  return;
262  }
263 
264  if (IsClosed())
265  {
266  m_writeState = WriteState::Idle;
267  return;
268  }
269 
270  std::lock_guard<std::mutex> guard(m_mutex);
271 
272  assert(m_writeState == WriteState::Sending);
273  assert(length <= m_outBuffer->m_writePosition);
274 
275  // if there is data left to write, move it to the start of the buffer
276  if (length < m_outBuffer->m_writePosition)
277  {
278  memcpy(&(m_outBuffer->m_buffer[0]), &(m_outBuffer->m_buffer[length]), (m_outBuffer->m_writePosition - length) * sizeof(m_outBuffer->m_buffer[0]));
279  m_outBuffer->m_writePosition -= length;
280  }
281  // if not, reset the write pointer
282  else
283  m_outBuffer->m_writePosition = 0;
284 
285  // if there is data in the secondary buffer, append it to the primary buffer
286  if (m_secondaryOutBuffer->m_writePosition > 0)
287  {
288  // do we have enough space? if not, resize
289  if (m_outBuffer->m_buffer.size() < (m_outBuffer->m_writePosition + m_secondaryOutBuffer->m_writePosition))
290  m_outBuffer->m_buffer.resize(m_outBuffer->m_writePosition + m_secondaryOutBuffer->m_writePosition);
291 
292  memcpy(&(m_outBuffer->m_buffer[m_outBuffer->m_writePosition]), &(m_secondaryOutBuffer->m_buffer[0]), (m_secondaryOutBuffer->m_writePosition) * sizeof(m_secondaryOutBuffer->m_buffer[0]));
293 
294  m_outBuffer->m_writePosition += m_secondaryOutBuffer->m_writePosition;
295  m_secondaryOutBuffer->m_writePosition = 0;
296  }
297 
298  std::shared_ptr<Socket> ptr = shared<Socket>();
299  // if there is any data to write, do so immediately
300  if (m_outBuffer->m_writePosition > 0)
301  m_socket.async_write_some(boost::asio::buffer(m_outBuffer->m_buffer, m_outBuffer->m_writePosition),
302  make_custom_alloc_handler(m_allocator,
303  [ptr](const boost::system::error_code & error, size_t length) { ptr->OnWriteComplete(error, length);}));
304  else
305  m_writeState = WriteState::Idle;
306  }
