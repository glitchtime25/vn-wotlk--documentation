230 void instance_karazhan::SetData(uint32 uiType, uint32 uiData)
231 {
232  switch (uiType)
233  {
234  case TYPE_ATTUMEN:
235  m_auiEncounter[uiType] = uiData;
236  if (uiData == FAIL)
237  {
238  // Respawn Midnight on Fail
239  if (Creature* pMidnight = GetSingleCreatureFromStorage(NPC_MIDNIGHT))
240  {
241  if (!pMidnight->IsAlive())
242  pMidnight->Respawn();
243  }
244  }
245  break;
246  case TYPE_MOROES:
247  case TYPE_MAIDEN:
248  m_auiEncounter[uiType] = uiData;
249  break;
250  case TYPE_OPERA:
251  // Don't store the same data twice
252  if (uiData == m_auiEncounter[uiType])
253  break;
254  m_auiEncounter[uiType] = uiData;
255  if (uiData == IN_PROGRESS)
256  m_uiOzDeathCount = 0;
257  if (uiData == FAIL)
258  {
259  DoUseDoorOrButton(GO_STAGE_DOOR_LEFT);
260  }
261  if (uiData == DONE)
262  {
263  DoUseDoorOrButton(GO_STAGE_DOOR_LEFT);
264  DoUseDoorOrButton(GO_STAGE_DOOR_RIGHT);
265  DoToggleGameObjectFlags(GO_SERVANTS_ACCESS_DOOR, GO_FLAG_LOCKED, false);
266  DoToggleGameObjectFlags(GO_SIDE_ENTRANCE_DOOR, GO_FLAG_LOCKED, false);
267  }
268  // use curtain only for event start or fail
269  else
270  DoUseDoorOrButton(GO_STAGE_CURTAIN);
271  break;
272  case TYPE_CURATOR:
273  case TYPE_TERESTIAN:
274  m_auiEncounter[uiType] = uiData;
275  break;
276  case TYPE_ARAN:
277  if (uiData == FAIL || uiData == DONE)
278  DoToggleGameObjectFlags(GO_PRIVATE_LIBRARY_DOOR, GO_FLAG_LOCKED, false);
279  if (uiData == IN_PROGRESS)
280  DoToggleGameObjectFlags(GO_PRIVATE_LIBRARY_DOOR, GO_FLAG_LOCKED, true);
281  m_auiEncounter[uiType] = uiData;
282  break;
283  case TYPE_NETHERSPITE:
284  m_auiEncounter[uiType] = uiData;
285  DoUseDoorOrButton(GO_MASSIVE_DOOR);
286  break;
287  case TYPE_CHESS:
288  if (uiData == DONE)
289  {
290  static_cast<DungeonMap*>(instance)->PermBindAllPlayers();
291  DoFinishChessEvent();
292  }
293  else if (uiData == FAIL)
294  DoFailChessEvent();
295  else if (uiData == IN_PROGRESS || uiData == SPECIAL)
296  DoPrepareChessEvent();
297  m_auiEncounter[uiType] = uiData;
298  break;
299  case TYPE_MALCHEZZAR:
300  DoUseDoorOrButton(GO_NETHERSPACE_DOOR);
301  m_auiEncounter[uiType] = uiData;
302  break;
303  case TYPE_NIGHTBANE:
304  m_auiEncounter[uiType] = uiData;
305  DoUseDoorOrButton(GO_MASTERS_TERRACE_DOOR_1);
306  DoUseDoorOrButton(GO_MASTERS_TERRACE_DOOR_2);
307  break;
308  // Store the event type for the Opera
309  case TYPE_OPERA_PERFORMANCE:
310  m_uiOperaEvent = uiData;
311  break;
312  }
313 
314  // Also save the opera performance, once it's set
315  if (uiData == DONE || uiType == TYPE_OPERA_PERFORMANCE)
316  {
317  OUT_SAVE_INST_DATA;
318 
319  std::ostringstream saveStream;
320  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
321  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
322  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
323  << m_auiEncounter[9] << " " << m_auiEncounter[10] << " " << m_uiOperaEvent;
324 
325  m_strInstData = saveStream.str();
326 
327  SaveToDB();
328  OUT_SAVE_INST_DATA_COMPLETE;
329  }
330 }

440 
441  // spawn boss when empty

445  m_bBasementBossReady = false;

448  aBasementEnum[uiIndex].fO, TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 2 * HOUR * IN_MILLISECONDS, true))
449  {
450  if (aBasementEnum[uiIndex].emote2)

454  }
455  }
456  }
457  break;
458  }

461 void instance_karazhan::DoPrepareChessEvent()
462 {
463  // Allow all the chess pieces to init start position
464  for (GuidList::const_iterator itr = m_lChessPiecesAlliance.begin(); itr != m_lChessPiecesAlliance.end(); ++itr)
465  {
466  if (Creature* pChessPiece = instance->GetCreature(*itr))
467  {
468  Creature* pSquare = GetClosestCreatureWithEntry(pChessPiece, NPC_SQUARE_BLACK, 2.0f);
469  if (!pSquare)
470  pSquare = GetClosestCreatureWithEntry(pChessPiece, NPC_SQUARE_WHITE, 2.0f);
471  if (!pSquare)
472  {
473  script_error_log("Instance Karazhan: ERROR Failed to properly load the Chess square for %s.", pChessPiece->GetGuidStr().c_str());
474  return;
475  }
476 
477  // send event which will prepare the current square
478  pChessPiece->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, pSquare, pChessPiece);
479  }
480  }
481 
482  for (GuidList::const_iterator itr = m_lChessPiecesHorde.begin(); itr != m_lChessPiecesHorde.end(); ++itr)
483  {
484  if (Creature* pChessPiece = instance->GetCreature(*itr))
485  {
486  Creature* pSquare = GetClosestCreatureWithEntry(pChessPiece, NPC_SQUARE_BLACK, 2.0f);
487  if (!pSquare)
488  pSquare = GetClosestCreatureWithEntry(pChessPiece, NPC_SQUARE_WHITE, 2.0f);
489  if (!pSquare)
490  {
491  script_error_log("Instance Karazhan: ERROR Failed to properly load the Chess square for %s.", pChessPiece->GetGuidStr().c_str());
492  return;
493  }
494 
495  // send event which will prepare the current square
496  pChessPiece->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, pSquare, pChessPiece);
497  }
498  }
499 
500  if (GameObject* chessboard = GetSingleGameObjectFromStorage(GO_CHESSBOARD))
501  if (GameObjectAI* ai = chessboard->AI())
502  ai->ReceiveAIEvent(AI_EVENT_CUSTOM_A, uint32(true));
503 
504  m_uiAllianceStalkerCount = 0;
505  m_uiHordeStalkerCount = 0;
506  m_vHordeStalkers.clear();
507  m_vAllianceStalkers.clear();
508 
509  // sort stalkers depending on side
510  CreatureList lStalkers;
511  for (GuidList::const_iterator itr = m_lChessHordeStalkerList.begin(); itr != m_lChessHordeStalkerList.end(); ++itr)
512  {
513  if (Creature* pTemp = instance->GetCreature(*itr))
514  lStalkers.push_back(pTemp);
515  }
516 
517  if (lStalkers.empty())
518  {
519  script_error_log("Instance Karazhan: ERROR Failed to properly load the horde side stalkers for the Chess Event.");
520  return;
521  }
522 
523  // get the proper statusBar npc
524  Creature* pStatusBar = instance->GetCreature(m_HordeStatusGuid);
525  if (!pStatusBar)
526  return;
527 
528  lStalkers.sort(ObjectDistanceOrder(pStatusBar));
529  for (CreatureList::const_iterator itr = lStalkers.begin(); itr != lStalkers.end(); ++itr)
530  m_vHordeStalkers.push_back((*itr)->GetObjectGuid());
531 
532  lStalkers.clear();
533  for (GuidList::const_iterator itr = m_lChessAllianceStalkerList.begin(); itr != m_lChessAllianceStalkerList.end(); ++itr)
534  {
535  if (Creature* pTemp = instance->GetCreature(*itr))
536  lStalkers.push_back(pTemp);
537  }
538 
539  if (lStalkers.empty())
540  {
541  script_error_log("Instance Karazhan: ERROR Failed to properly load the alliance side stalkers for the Chess Event.");
542  return;
543  }
544 
545  // get the proper statusBar npc
546  pStatusBar = instance->GetCreature(m_AllianceStatusGuid);
547  if (!pStatusBar)
548  return;
549 
550  lStalkers.sort(ObjectDistanceOrder(pStatusBar));
551  for (CreatureList::const_iterator itr = lStalkers.begin(); itr != lStalkers.end(); ++itr)
552  m_vAllianceStalkers.push_back((*itr)->GetObjectGuid());
553 }

555 void instance_karazhan::DoMoveChessPieceToSides(uint32 uiSpellId, uint32 uiFaction, bool bGameEnd /*= false*/)
556 {
557  // assign proper faction variables
558  GuidVector& vStalkers = uiFaction == FACTION_ID_CHESS_ALLIANCE ? m_vAllianceStalkers : m_vHordeStalkers;
559  uint32 uiCount = uiFaction == FACTION_ID_CHESS_ALLIANCE ? m_uiAllianceStalkerCount : m_uiHordeStalkerCount;
560 
561  // get the proper statusBar npc
562  Creature* pStatusBar = instance->GetCreature(uiFaction == FACTION_ID_CHESS_ALLIANCE ? m_AllianceStatusGuid : m_HordeStatusGuid);
563  if (!pStatusBar)
564  return;
565 
566  if (vStalkers.size() < uiCount + 1)
567  return;
568 
569  // handle stalker transformation
570  if (Creature* pStalker = instance->GetCreature(vStalkers[uiCount]))
571  {
572  // need to provide specific target, in order to ensure the logic of the event
573  pStatusBar->CastSpell(pStalker, uiSpellId, TRIGGERED_OLD_TRIGGERED);
574  uiFaction == FACTION_ID_CHESS_ALLIANCE ? ++m_uiAllianceStalkerCount : ++m_uiHordeStalkerCount;
575  }
576 
577  // handle emote on end game
578  if (bGameEnd)
579  {
580  // inverse factions
581  vStalkers.clear();
582  vStalkers = uiFaction == FACTION_ID_CHESS_ALLIANCE ? m_vHordeStalkers : m_vAllianceStalkers;
583 
584  for (GuidVector::const_iterator itr = vStalkers.begin(); itr != vStalkers.end(); ++itr)

739 void instance_karazhan::Update(uint32 uiDiff)
740 {

742  {
743  // respawn all chess pieces and side stalkers on the original position
744  if (m_uiChessResetTimer <= uiDiff)
