132  void JustSummoned(Creature* summoned) override
133  {
134  // store the swarmers for a future attack
135  if (summoned->GetEntry() == NPC_SWARMER)
136  {
137  m_swarmersGuidList.push_back(summoned->GetObjectGuid());
138  uint32 spellId = 0;
139  switch (urand(0, 4))
140  {
141  case 0: spellId = SPELL_SWARMER_TELEPORT_1; break;
142  case 1: spellId = SPELL_SWARMER_TELEPORT_2; break;
143  case 2: spellId = SPELL_SWARMER_TELEPORT_3; break;
144  case 3: spellId = SPELL_SWARMER_TELEPORT_4; break;
145  case 4: spellId = SPELL_SWARMER_TELEPORT_5; break;
146  default: break;
147  }
148  summoned->CastSpell(nullptr, spellId, TRIGGERED_OLD_TRIGGERED);
149  }
150  // move the larva to paralyze target position
151  else if (summoned->GetEntry() == NPC_LARVA)
152  {
153  summoned->SetCorpseDelay(5);
154  summoned->SetWalk(false);
155  if (Unit* target = m_creature->GetMap()->GetUnit(m_paralyzeTarget))
156  {
157  summoned->AI()->AttackStart(target);
158  summoned->FixateTarget(target);
159  }
160  }
161  else if (summoned->GetEntry() == NPC_HORNET)
162  {
163  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
164  summoned->AI()->AttackStart(target);
165  }
166  m_spawns.push_back(summoned->GetObjectGuid());
167  }

196  void ExecuteAction(uint32 action) override
197  {
198  switch (action)
199  {
200  case AYAMISS_FLY_UP:
201  {
202  m_creature->SetLevitate(true);
203  m_creature->SetHover(true);
204  SetCombatScriptStatus(true);
205  m_creature->GetMotionMaster()->MovePoint(POINT_AIR, -9689.292f, 1547.912f, 48.02729f);
206  SetActionReadyStatus(action, false);
207  break;
208  }
209  case AYAMISS_PHASE_2:
210  {
211  if (m_creature->GetHealthPercent() <= 70.0f)
212  {
213  m_phase = PHASE_GROUND;
214  m_creature->SetImmobilizedState(false);
215  SetMeleeEnabled(true);
216  m_creature->SetLevitate(false);
217  m_creature->SetHover(false);
218  DoResetThreat();
219  StartLanding();
220  SetCombatScriptStatus(true);
221  SetActionReadyStatus(action, false);
222 
223  DisableCombatAction(AYAMISS_POISON_STINGER);
224  ResetCombatAction(AYAMISS_LASH, urand(5000, 8000));
225  ResetCombatAction(AYAMISS_THRASH, urand(3000, 6000));
226  }
227  break;
228  }
229  case AYAMISS_FRENZY:
230  {
231  if (m_creature->GetHealthPercent() < 20.0f)
232  {
233  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
234  {
235  DoScriptText(EMOTE_GENERIC_FRENZY, m_creature);
236  SetActionReadyStatus(action, false);
237  }
238  }
239  break;
240  }
241  case AYAMISS_PARALYZE:
242  {
243  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_PARALYZE, SELECT_FLAG_PLAYER);
244  if (!target)
245  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_PARALYZE, SELECT_FLAG_PLAYER);
246 
247  if (DoCastSpellIfCan(target, SPELL_PARALYZE) == CAST_OK)
248  {
249  m_paralyzeTarget = target->GetObjectGuid();
250  ResetCombatAction(action, 15000);
251 
252  // Summon a larva
253  uint32 spellId = urand(0, 1) ? SPELL_SUMMON_LARVA_1 : SPELL_SUMMON_LARVA_2;
254  m_creature->CastSpell(nullptr, spellId, TRIGGERED_OLD_TRIGGERED);
255  }
256  break;
257  }
258  case AYAMISS_STINGER_SPRAY:
259  {
260  if (DoCastSpellIfCan(nullptr, SPELL_STINGER_SPRAY) == CAST_OK)
261  ResetCombatAction(action, urand(15000, 20000));
262  break;
263  }
264  case AYAMISS_POISON_STINGER:
265  {
266  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 0, SPELL_POISON_STINGER, SELECT_FLAG_PLAYER))
267  if (DoCastSpellIfCan(target, SPELL_POISON_STINGER) == CAST_OK)
268  ResetCombatAction(action, m_phase == PHASE_AIR ? urand(2000, 3000) : urand(6000, 12000));
269  break;
270  }
271  case AYAMISS_SUMMON_SWARMER:
272  {
273  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_SWARMER) == CAST_OK)
274  ResetCombatAction(action, 5000);
275  break;
276  }
277  case AYAMISS_SWARMER_ATTACK:
278  {
279  for (auto guid : m_swarmersGuidList)
280  {
281  if (Creature* temp = m_creature->GetMap()->GetCreature(guid))
282  {
283  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
284  temp->AI()->AttackStart(target);
285  }
286  }
287  m_swarmersGuidList.clear();
288  ResetCombatAction(action, 60000);
289  break;
290  }
291  case AYAMISS_LASH:
292  {
293  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_LASH) == CAST_OK)
294  ResetCombatAction(action, urand(8000, 15000));
295  break;
296  }
297  case AYAMISS_THRASH:
298  {
299  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_THRASH) == CAST_OK)
300  ResetCombatAction(action, urand(5000, 7000));
301  break;
302  }
303  }
304  }
