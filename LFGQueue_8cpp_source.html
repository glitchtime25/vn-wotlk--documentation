102 void LFGQueue::Update()
103 {
104  uint32 counter = 1;
105  while (!World::IsStopped())
106  {
107  GetMessager().Execute(this);
108 
109  TimePoint now = sWorld.GetCurrentClockTime();
110  for (auto itr = m_queueData.begin(); itr != m_queueData.end();)
111  {
112  LFGQueueData& queueData = itr->second;
113  if (queueData.m_roleCheckState == LFG_ROLECHECK_INITIALITING && queueData.m_cancelTime < now)
114  {
115  queueData.UpdateRoleCheck(ObjectGuid(), 0, true, true);
116  itr = m_queueData.erase(itr);
117  }
118  else
119  ++itr;
120  }
121 
122  if (IsTestingEnabled()) // in debug pop any queue regardless of eligibility
123  {
124  for (auto& queuedGroupData : m_queueData)
125  {
126  LFGQueueData& queueData = queuedGroupData.second;
127  if (queueData.GetState() == LFG_STATE_QUEUED)
128  {
129  LfgProposal proposal;
130  proposal.id = counter++;
131  queueData.PopQueue(proposal);
132  m_proposals[proposal.id] = proposal;
133  }
134  }
135  }
136  else
137  {
138  // no actual matchmaking for now - enable entering for full groups - TODO: matchmaking
139  for (auto& queuedGroupData : m_queueData)
140  {
141  LFGQueueData& queueData = queuedGroupData.second;
142  // rolecheck makes sure integrity of correct group is held
143  if (queueData.GetState() == LFG_STATE_QUEUED && queueData.m_playerInfoPerGuid.size() == 5)
144  {
145  LfgProposal proposal;
146  proposal.id = counter++;
147  queueData.PopQueue(proposal);
148  m_proposals[proposal.id] = proposal;
149  }
150  }
151  }
152 
153  for (auto& proposalData : m_proposals)
154  proposalData.second.UpdateProposal(*this);
155 
156  for (auto itr = m_queueData.begin(); itr != m_queueData.end();)
157  {
158  if (itr->second.GetState() == LFG_STATE_FAILED)
159  itr = m_queueData.erase(itr);
160  else
161  ++itr;
162  }
163 
164  for (uint32 proposalId : m_proposalsForRemoval)
165  m_proposals.erase(proposalId);
166 
167  std::this_thread::sleep_for(std::chrono::milliseconds(500));
168  }
169 }

205 void LFGQueueData::UpdateRoleCheck(ObjectGuid guid, uint8 roles, bool abort, bool timeout)
206 {
207  LfgPlayerInfoMap check_roles;
208 
209  bool sendRoleChosen = m_roleCheckState != LFG_ROLECHECK_DEFAULT && abort;
210 
211  if (abort)
212  m_roleCheckState = timeout ? LFG_ROLECHECK_MISSING_ROLE : LFG_ROLECHECK_ABORTED;
213  else if (m_playerInfoPerGuid.empty() || !roles) // Player selected no role.
214  m_roleCheckState = LFG_ROLECHECK_NO_ROLE;
215  else
216  {
217  m_playerInfoPerGuid[guid].m_roles = roles;
218 
219  // Check if all players have selected a role
220  auto itrRoles = m_playerInfoPerGuid.begin();
221  while (itrRoles != m_playerInfoPerGuid.end() && itrRoles->second.m_roles != PLAYER_ROLE_NONE)
222  ++itrRoles;
223 
224  if (itrRoles == m_playerInfoPerGuid.end())
225  {
226  // use temporal var to check roles, CheckGroupRoles modifies the roles
227  check_roles = m_playerInfoPerGuid;
228  m_roleCheckState = (m_raid || LFGMgr::CheckGroupRoles(check_roles)) ? LFG_ROLECHECK_FINISHED : LFG_ROLECHECK_WRONG_ROLES;
229  }
230  }
231 
232  LfgDungeonSet dungeons = m_dungeons;
233 
234  LfgJoinResultData joinData = LfgJoinResultData(LFG_JOIN_FAILED, m_roleCheckState);
235 
236  std::vector<WorldPacket> packets;
237  if (sendRoleChosen)
238  packets.emplace_back(WorldSession::BuildLfgRoleChosen(guid, roles));
239 
240  packets.emplace_back(WorldSession::BuildLfgRoleCheckUpdate(*this));
241  switch (m_roleCheckState)
242  {
243  case LFG_ROLECHECK_INITIALITING:
244  break;
245  case LFG_ROLECHECK_FINISHED:
246  packets.emplace_back(WorldSession::BuildLfgUpdate(LfgUpdateData(LFG_UPDATETYPE_ADDED_TO_QUEUE,
247  m_raid ? LFG_STATE_RAIDBROWSER : LFG_STATE_NONE, GetDungeons(), m_comment), true));
248  break;
249  default:
250  packets.emplace_back(WorldSession::BuildLfgJoinResult(joinData));
251  packets.emplace_back(WorldSession::BuildLfgUpdate(LfgUpdateData(LFG_UPDATETYPE_ROLECHECK_FAILED), true));
252  break;
253  }
254 
255  sWorld.GetMessager().AddMessage([groupGuid = m_ownerGuid, packets](World* world)
256  {
257  world->BroadcastToGroup(groupGuid, packets);
258  });
259 
260  if (m_roleCheckState == LFG_ROLECHECK_FINISHED)
261  {
262  m_state = m_raid ? LFG_STATE_RAIDBROWSER : LFG_STATE_QUEUED;
263  m_queueTime = sWorld.GetCurrentClockTime();
264  m_roleCheckState = LFG_ROLECHECK_FINISHED;
265  RecalculateRoles();
266  }
267  else if (m_roleCheckState != LFG_ROLECHECK_INITIALITING)
268  m_state = LFG_STATE_FAILED;
269 }

271 void LFGQueueData::PopQueue(LfgProposal& proposal)
272 {
273  auto itr = m_dungeons.begin();
274  std::advance(itr, urand(0, m_dungeons.size() - 1));
275  uint32 selectedDungeonId = *itr;
276  m_state = LFG_STATE_PROPOSAL;
277  std::vector<WorldPacket> packets;
278  packets.emplace_back(WorldSession::BuildLfgUpdate(LfgUpdateData(LFG_UPDATETYPE_PROPOSAL_BEGIN, GetDungeons(), ""), true));
279 
280  proposal.dungeonId = selectedDungeonId;
281  proposal.state = LFG_PROPOSAL_INITIATING;
282  proposal.group = ObjectGuid(); // only filled when already lfg group
283  proposal.leader = m_leaderGuid;
284  proposal.cancelTime = sWorld.GetCurrentClockTime() + std::chrono::seconds(LFG_TIME_ROLECHECK);
285  proposal.encounters = 0;
286  proposal.isNew = true;
287  proposal.queues.push_back(m_ownerGuid);
288 
289  for (auto& playerData : m_playerInfoPerGuid)
290  {
291  uint8 roles = playerData.second.m_roles; // give one random role out of given ones
292  if (roles & PLAYER_ROLE_TANK)
293  roles = PLAYER_ROLE_TANK;
294  if (roles & PLAYER_ROLE_HEALER)
295  roles = PLAYER_ROLE_HEALER;
296  if (roles & PLAYER_ROLE_DAMAGE)
297  roles = PLAYER_ROLE_DAMAGE;
298  proposal.players[playerData.first] = LfgProposalPlayer(roles, LFG_ANSWER_PENDING, m_ownerGuid.IsGroup() ? m_ownerGuid : ObjectGuid(), m_randomDungeonId);
299  }
300 
301  std::map<ObjectGuid, std::vector<WorldPacket>> personalizedPackets;
302  for (auto& playerData : m_playerInfoPerGuid)
303  {
304  personalizedPackets[playerData.first] = packets;
305  personalizedPackets[playerData.first].emplace_back(WorldSession::BuildLfgUpdateProposal(proposal, m_randomDungeonId, playerData.first));
306  }
307 
308  sWorld.GetMessager().AddMessage([=](World* world)
309  {
310  world->BroadcastPersonalized(personalizedPackets);
311  });
312 }

330 void LfgProposal::UpdateProposal(LFGQueue& queue)
331 {
332  bool end = true;
333  bool fail = false;
334 
335  for (auto& playerData : players)
336  {
337  if (playerData.second.answer == LFG_ANSWER_PENDING)
338  {
339  end = false;
340  break;
341  }
342 
343  if (playerData.second.answer == LFG_ANSWER_DENY)
344  fail = true;
345  }
346 
347  if (end)
348  {
349  if (fail)
350  FailProposal(queue);
351  else
352  AcceptProposal(queue);
353  return;
354  }
355 
356  if (cancelTime < sWorld.GetCurrentClockTime())
357  FailProposal(queue);
358 }

360 void LfgProposal::FailProposal(LFGQueue& queue)
361 {
362  state = LFG_PROPOSAL_FAILED;
363  std::set<ObjectGuid> declinedSelectors;
364  std::set<ObjectGuid> failedSelectors;
365  for (auto& proposalPlayer : players)
366  {
367  LfgProposalPlayer& player = proposalPlayer.second;
368  if (player.answer == 0) // did not select or declined
369  declinedSelectors.insert(player.group ? player.group : proposalPlayer.first);
370  else if (player.answer == -1)
371  failedSelectors.insert(player.group ? player.group : proposalPlayer.first);
372  }
373 
374  LfgUpdateData updateData(declinedSelectors.size() > 0 ? LFG_UPDATETYPE_PROPOSAL_DECLINED : LFG_UPDATETYPE_PROPOSAL_FAILED);
375  std::map<ObjectGuid, std::vector<WorldPacket>> personalizedPackets;
376  WorldPacket proposalFailed = WorldSession::BuildLfgUpdate(updateData, true);
377  for (auto& proposalPlayer : players)
378  personalizedPackets[proposalPlayer.first].push_back(proposalFailed);
379 
380  updateData.updateType = LFG_UPDATETYPE_REMOVED_FROM_QUEUE;
381  std::set<ObjectGuid>& whoToKick = declinedSelectors.size() > 0 ? declinedSelectors : failedSelectors;
382  for (auto& queued : queues)
383  {
384  LFGQueueData& queueData = queue.GetQueueData(queued);
385  if (whoToKick.find(queueData.m_ownerGuid) != whoToKick.end())
386  {
387  // kick from queue
388  for (auto& playerInfo : queueData.m_playerInfoPerGuid)
389  personalizedPackets[playerInfo.first].push_back(WorldSession::BuildLfgUpdate(updateData, queueData.m_ownerGuid.IsGroup()));
390  queueData.SetState(LFG_STATE_FAILED);
391  }
392  else
393  {
394  // continue being queued - did nothing wrong
395  queueData.SetState(LFG_STATE_QUEUED);
396  }
397  }
398 
399  sWorld.GetMessager().AddMessage([personalizedPackets](World* world)
400  {
401  world->BroadcastPersonalized(personalizedPackets);
402  });
403 
404  queue.RemoveProposal(id);
405 }

407 void LfgProposal::AcceptProposal(LFGQueue& queue)
408 {
409  bool sendUpdate = state != LFG_PROPOSAL_SUCCESS;
410  state = LFG_PROPOSAL_SUCCESS;
411  TimePoint joinTime = sWorld.GetCurrentClockTime();
412 
413  std::map<ObjectGuid, std::vector<WorldPacket>> personalizedPackets;
414 
415  std::map<ObjectGuid, ObjectGuid> m_groupPerPlayer;
416  std::map<ObjectGuid, uint8> m_rolesPerPlayer;
417 
418  LfgUpdateData updateData = LfgUpdateData(LFG_UPDATETYPE_GROUP_FOUND);
419  for (LfgProposalPlayerContainer::const_iterator itr = players.begin(); itr != players.end(); ++itr)
420  {
421  ObjectGuid pguid = itr->first;
422  ObjectGuid gguid = itr->second.group;
423  uint32 randomDungeonId = itr->second.group;
424  int32 waitTime = -1;
425 
426  std::vector<WorldPacket>& packets = personalizedPackets[pguid];
427 
428  packets.emplace_back(WorldSession::BuildLfgUpdateProposal(*this, randomDungeonId, pguid));
429 
430  LFGQueueData& queueData = queue.GetQueueData(gguid ? gguid : pguid);
431  waitTime = int32((joinTime - queueData.GetJoinTime()).count());
432  packets.emplace_back(WorldSession::BuildLfgUpdate(updateData, true));
433  updateData.updateType = LFG_UPDATETYPE_REMOVED_FROM_QUEUE;
434  packets.emplace_back(WorldSession::BuildLfgUpdate(updateData, true));
435  packets.emplace_back(WorldSession::BuildLfgUpdate(updateData, false));
436 
437  LFGQueuePlayer& playerInfo = queueData.m_playerInfoPerGuid[pguid];
438  // Update timers
439  uint8 role = playerInfo.m_roles;
440  role &= ~PLAYER_ROLE_LEADER;
441  switch (role)
442  {
443  case PLAYER_ROLE_DAMAGE:
444  queue.UpdateWaitTimeDps(waitTime, dungeonId);
445  break;
446  case PLAYER_ROLE_HEALER:
447  queue.UpdateWaitTimeHealer(waitTime, dungeonId);
448  break;
449  case PLAYER_ROLE_TANK:
450  queue.UpdateWaitTimeTank(waitTime, dungeonId);
451  break;
452  default:
453  queue.UpdateWaitTimeAvg(waitTime, dungeonId);
454  break;
455  }
456 
457  // Store the number of players that were present in group when joining RFD, used for achievement purposes
458  queue.SetPartyMemberCountAtJoin(pguid, queueData.m_playerInfoPerGuid.size());
459 
460  m_groupPerPlayer[pguid] = gguid;
461  m_rolesPerPlayer[pguid] = playerInfo.m_roles;
462  queueData.SetState(LFG_STATE_DUNGEON);
463  }
464 
465  // Remove players/groups from Queue
466  for (GuidList::const_iterator it = queues.begin(); it != queues.end(); ++it)
467  queue.RemoveFromQueue(*it);
468 
469  queue.RemoveProposal(id);
470 
471  sWorld.GetMessager().AddMessage([=](World* world)
472  {
473  world->BroadcastPersonalized(personalizedPackets);
474  });
475 
476  GuidList allPlayers, tankPlayers, healPlayers, dpsPlayers;
477  GuidList playersToTeleport;
478  std::map<ObjectGuid, uint32> randomDungeonPerPlayer;
479  std::map<ObjectGuid, uint32> partyCountPerPlayer;
480 
481  for (LfgProposalPlayerContainer::const_iterator itr = players.begin(); itr != players.end(); ++itr)
482  {
483  ObjectGuid guid = itr->first;
484  if (guid == leader)
485  allPlayers.push_back(guid);
486  else
487  switch (itr->second.role & ~PLAYER_ROLE_LEADER)
488  {
489  case PLAYER_ROLE_TANK:
490  tankPlayers.push_back(guid);
491  break;
492  case PLAYER_ROLE_HEALER:
493  healPlayers.push_back(guid);
494  break;
495  case PLAYER_ROLE_DAMAGE:
496  dpsPlayers.push_back(guid);
497  break;
498  default:
499  MANGOS_ASSERT(false); // "Invalid LFG role %u"
500  break;
501  }
502 
503  if (isNew || m_groupPerPlayer[guid] != group)
504  playersToTeleport.push_back(guid);
505 
506  randomDungeonPerPlayer[guid] = itr->second.randomDungeonId;
507  partyCountPerPlayer[guid] = queue.GetPartyMemberCountAtJoin(guid);
508  }
509 
510  allPlayers.splice(allPlayers.end(), tankPlayers);
511  allPlayers.splice(allPlayers.end(), healPlayers);
512  allPlayers.splice(allPlayers.end(), dpsPlayers);
513 
514  // Set the dungeon difficulty
515  LFGDungeonData const* dungeon = sLFGMgr.GetLFGDungeon(dungeonId);
516  MANGOS_ASSERT(dungeon);
517 
518  sWorld.GetMessager().AddMessage([allPlayers, group = group, dungeon, playersToTeleport, randomDungeonPerPlayer, partyCountPerPlayer](World* world)
519  {
520  Group* grp = group ? sObjectMgr.GetGroupById(group.GetCounter()) : nullptr;
521  for (GuidList::const_iterator it = allPlayers.begin(); it != allPlayers.end(); ++it)
522  {
523  ObjectGuid pguid = (*it);
524  Player* player = ObjectAccessor::FindPlayer(pguid);
525  if (!player)
526  continue;
527 
528  Group* group = player->GetGroup();
529  if (group && group != grp)
530  group->RemoveMember(player->GetObjectGuid(), 0);
531 
532  if (!grp)
533  {
534  grp = new Group();
535  grp->Create(player->GetObjectGuid(), player->GetName());
536  grp->ConvertToLFG();
537  ObjectGuid gguid = grp->GetObjectGuid();
538  sObjectMgr.AddGroup(grp);
539  }
540  else if (group != grp)
541  grp->AddMember(player->GetObjectGuid(), player->GetName());
542 
543  player->GetLfgData().SetCountAtJoin(partyCountPerPlayer.find(pguid)->second);
544  player->GetLfgData().SetDungeon(dungeon->id);
545 
546  // Add the cooldown spell if queued for a random dungeon
547  auto randomItr = randomDungeonPerPlayer.find(player->GetObjectGuid());
548  if (uint32 randomDungeonId = randomItr->second)
549  {
550  LFGDungeonEntry const* dungeonEntry = sLFGDungeonStore.LookupEntry(randomDungeonId);
551  if (dungeonEntry && dungeonEntry->TypeID == LFG_TYPE_RANDOM_DUNGEON)
552  {
553  player->CastSpell(player, LFG_SPELL_DUNGEON_COOLDOWN, TRIGGERED_OLD_TRIGGERED);
554  player->GetLfgData().SetDungeon(randomDungeonId);
555  }
556  }
557  }
558 
559  MANGOS_ASSERT(grp);
560  grp->SetDungeonDifficulty(Difficulty(dungeon->difficulty));
561  ObjectGuid gguid = grp->GetObjectGuid();
562  grp->GetLfgData().SetDungeon(dungeon->id);
563  grp->GetLfgData().SetState(LFG_STATE_DUNGEON);
564 
565  // _SaveToDB(gguid, grp->GetDbStoreId()); - not saving anything for now
566 
567  // Teleport Player
568  for (GuidList::const_iterator it = playersToTeleport.begin(); it != playersToTeleport.end(); ++it)
569  if (Player* player = ObjectAccessor::FindPlayer(*it))
570  sLFGMgr.TeleportPlayer(player, false, false);
571 
572  // Update group info
573  grp->SendUpdate();
574  });
575 }
