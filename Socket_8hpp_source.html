123  // Class to manage the memory to be used for handler-based custom allocation.
124  // It contains a single block of memory which may be returned for allocation
125  // requests. If the memory is in use when an allocation request is made, the
126  // allocator delegates allocation to the global heap.
127  class handler_allocator
128  {
129  private:
130  static const size_t BufferSize = 512;
131 
132  // Storage space used for handler-based custom memory allocation.
133  std::aligned_storage<BufferSize>::type m_buffer;
134 
135  // Whether the handler-based custom allocation storage has been used.
136  bool m_inUse;
137 
138  public:
139  handler_allocator() : m_inUse(false) {}
140 
141  handler_allocator(const handler_allocator&) = delete;
142  handler_allocator& operator=(const handler_allocator&) = delete;
143 
144  void* allocate(std::size_t size)
145  {
146  if (!m_inUse && size <= sizeof(m_buffer))
147  {
148  m_inUse = true;
149  return &m_buffer;
150  }
151 
152  return ::operator new(size);
153  }
154 
155  void deallocate(void* pointer)
156  {
157  if (pointer == &m_buffer)
158  m_inUse = false;
159  else
160  ::operator delete(pointer);
161  }
162  };

178  void operator()(Args&&... args)
179  {
180  m_handler(std::forward<Args>(args)...);
181  }
