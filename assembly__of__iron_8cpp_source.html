249  void MovementInform(uint32 uiMoveType, uint32 uiPointId) override
250  {
251  if (uiMoveType != POINT_MOTION_TYPE || !uiPointId)
252  return;
253 
254  switch (uiPointId)
255  {
256  // After lift up follow a target and set the target change timer
257  case POINT_ID_LIFT_OFF:
258  // TODO: the boss should follow without changing his Z position - missing core feature
259  // Current implementation with move point is wrong
260  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IN_MELEE_RANGE))
261  {
262  DoMoveToTarget(pTarget);
263  m_followTargetGuid = pTarget->GetObjectGuid();
264  }
265  m_uiTendrilsTargetTimer = 5000;
266  m_uiTendrilsFollowTimer = 500;
267  break;
268  // After reached the land remove all the auras and resume basic combat
269  case POINT_ID_LAND:
270  m_creature->SetLevitate(false);
271  SetCombatMovement(true);
272  if (m_creature->GetVictim())
273  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
274 
275  m_creature->RemoveAurasDueToSpell(SPELL_TENDRILS_VISUAL);
276  m_creature->RemoveAurasDueToSpell(m_bIsRegularMode ? SPELL_LIGHTNING_TENDRILS : SPELL_LIGHTNING_TENDRILS_H);
277  break;
278  }
279  }

281  // Wrapper for target movement

283  {
284  if (pTarget)
285  {
286  m_creature->GetMotionMaster()->Clear();
287  m_creature->GetMotionMaster()->MovePoint(0, pTarget->GetPositionX(), pTarget->GetPositionY(), m_creature->GetPositionZ());
288  }

291  void UpdateAI(const uint32 uiDiff) override
292  {
293  // Pre fight visual spell
294  if (m_uiVisualTimer)
295  {
296  if (m_uiVisualTimer <= uiDiff)
297  {
298  if (DoCastSpellIfCan(m_creature, SPELL_LIGHTNING_CHANNEL_PREFIGHT) == CAST_OK)
299  m_uiVisualTimer = 0;
300  }
301  else
302  m_uiVisualTimer -= uiDiff;
303  }
304 
305  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
306  return;
307 
308  switch (m_uiPhase)
309  {
310  case PHASE_CHARGE_TWO:
311 
312  if (m_uiTendrilsTimer < uiDiff)
313  {
314  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_LIGHTNING_TENDRILS : SPELL_LIGHTNING_TENDRILS_H) == CAST_OK)
315  {
316  DoCastSpellIfCan(m_creature, SPELL_TENDRILS_VISUAL, CAST_TRIGGERED);
317  DoScriptText(SAY_BRUNDIR_FLY, m_creature);
318  SetCombatMovement(false);
319  m_creature->SetLevitate(true);
320  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LIFT_OFF, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 15.0f);
321  m_uiTendrilsTimer = 90000;
322  m_uiTendrilsEndTimer = 25000;
323  }
324  }
325  else
326  m_uiTendrilsTimer -= uiDiff;
327 
328  if (m_uiTendrilsEndTimer)
329  {
330  if (m_uiTendrilsEndTimer <= uiDiff)
331  {
332  // Get proper Z position and land
333  float groundZ = m_creature->GetMap()->GetHeight(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ(), false);
334  float fZ = m_creature->GetTerrain()->GetWaterOrGroundLevel(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ(), groundZ);
335  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LAND, m_creature->GetPositionX(), m_creature->GetPositionY(), fZ);
336  m_uiOverloadTimer = 40000;
337  m_uiWhirlTimer = 15000;
338  m_uiChainLightningTimer = 3000;
339  m_uiTendrilsEndTimer = 0;
340  m_uiTendrilsTargetTimer = 0;
341  }
342  else
343  m_uiTendrilsEndTimer -= uiDiff;
344 
345  // Change follow target every 5 seconds
346  if (m_uiTendrilsTargetTimer)
347  {
348  if (m_uiTendrilsTargetTimer <= uiDiff)
349  {
350  // TODO: the boss should follow without changing his Z position - missing core feature
351  // Current implementation with move point is wrong
352  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IN_MELEE_RANGE))
353  {
354  DoMoveToTarget(pTarget);
355  m_followTargetGuid = pTarget->GetObjectGuid();
356  }
357  m_uiTendrilsTargetTimer = 5000;
358  m_uiTendrilsFollowTimer = 500;
359  }
360  else
361  m_uiTendrilsTargetTimer -= uiDiff;
362 
363  // Workaround to follow the target
364  if (m_uiTendrilsFollowTimer < uiDiff)
365  {
366  if (Unit* pTarget = m_creature->GetMap()->GetUnit(m_followTargetGuid))
367  DoMoveToTarget(pTarget);
368  m_uiTendrilsFollowTimer = 500;
369  }
370  else
371  m_uiTendrilsFollowTimer -= uiDiff;
372  }
373 
374  // no other spells during tendrils
375  return;
376  }
377 
378  // no break here; he uses the other spells as well
379  case PHASE_CHARGE_ONE:
380 
381  if (m_uiWhirlTimer < uiDiff)
382  {
383  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_LIGHTNING_WHIRL : SPELL_LIGHTNING_WHIRL_H) == CAST_OK)
384  {
385  DoScriptText(SAY_BRUNDIR_WHIRL, m_creature);
386  m_uiWhirlTimer = 30000;
387  }
388  }
389  else
390  m_uiWhirlTimer -= uiDiff;
391 
392  // no break here; he uses the other spells as well
393  case PHASE_NO_CHARGE:
394 
395  if (m_uiChainLightningTimer < uiDiff)
396  {
397  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
398  {
399  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_CHAIN_LIGHTNING : SPELL_CHAIN_LIGHTNING_H) == CAST_OK)
400  m_uiChainLightningTimer = 2000;
401  }
402  }
403  else
404  m_uiChainLightningTimer -= uiDiff;
405 
406  if (m_uiOverloadTimer < uiDiff)
407  {
408  if (DoCastSpellIfCan(m_creature, SPELL_OVERLOAD) == CAST_OK)
409  m_uiOverloadTimer = 80000;
410  }
411  else
412  m_uiOverloadTimer -= uiDiff;
413 
414  break;
415  }
416 
417  DoMeleeAttackIfReady();
418  }
