287  void ExecuteActions() override
288  {
289  if (!CanExecuteCombatAction())
290  return;
291 
292  for (uint32 i = 0; i < ANZU_COMBAT_ACTION_MAX; ++i)
293  {
294  if (!GetActionReadyStatus(i))
295  continue;
296 
297  switch (i)
298  {
299  case ANZU_ACTION_SPAWN_BROODS:
300  if (m_creature->GetHealthPercent() < m_healthBanishCheck)
301  {
302  DoSummonBroodsOfAnzu();
303  m_healthBroodCheck -= 40.0f;
304  }
305  continue;
306  case ANZU_ACTION_BANISH:
307  if (m_creature->GetHealthPercent() < m_healthBanishCheck)
308  {
309  if (DoCastSpellIfCan(m_creature, SPELL_BANISH_SELF) == CAST_OK)
310  {
311  DoScriptText(SAY_BANISH, m_creature);
312  SetMeleeEnabled(false);
313  m_healthBanishCheck -= 40.0f;
314  ResetTimer(ANZU_BROOD_ATTACK, 15000);
315  return;
316  }
317  }
318  continue;
319  case ANZU_ACTION_FLESH_RIP:
320  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FLESH_RIP) == CAST_OK)
321  {
322  SetActionReadyStatus(i, false);
323  ResetTimer(i, GetSubsequentActionTimer(AnzuActions(i)));
324  return;
325  }
326  continue;
327  case ANZU_ACTION_SCREECH:
328  if (DoCastSpellIfCan(m_creature, SPELL_SCREECH) == CAST_OK)
329  {
330  SetActionReadyStatus(i, false);
331  ResetTimer(i, GetSubsequentActionTimer(AnzuActions(i)));
332  return;
333  }
334  continue;
335  case ANZU_ACTION_SPELL_BOMB:
336  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
337  {
338  if (DoCastSpellIfCan(target, SPELL_SPELL_BOMB) == CAST_OK)
339  {
340  switch (urand(0, 2))
341  {
342  case 0: DoScriptText(SAY_WHISPER_MAGIC_1, m_creature, target); break;
343  case 1: DoScriptText(SAY_WHISPER_MAGIC_2, m_creature, target); break;
344  case 2: DoScriptText(SAY_WHISPER_MAGIC_3, m_creature, target); break;
345  }
346  SetActionReadyStatus(i, false);
347  ResetTimer(i, GetSubsequentActionTimer(AnzuActions(i)));
348  return;
349  }
350  }
351  continue;
352  case ANZU_ACTION_CYCLONE:
353  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
354  {
355  if (DoCastSpellIfCan(target, SPELL_CYCLONE) == CAST_OK)
356  {
357  SetActionReadyStatus(i, false);
358  ResetTimer(i, GetSubsequentActionTimer(AnzuActions(i)));
359  return;
360  }
361  }
362  continue;
363  case ANZU_ACTION_DIVE:
364  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
365  {
366  if (DoCastSpellIfCan(target, SPELL_DIVE) == CAST_OK)
367  {
368  SetActionReadyStatus(i, false);
369  ResetTimer(i, GetSubsequentActionTimer(AnzuActions(i)));
370  return;
371  }
372  }
373  continue;
374  }
375  }
376  }
