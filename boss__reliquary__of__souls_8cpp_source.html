121 {
122  boss_reliquary_of_soulsAI(Creature* creature) : ScriptedAI(creature, 0), m_instance(static_cast<instance_black_temple*>(creature->GetInstanceData()))
123  {
124  AddCustomAction(RELIQUARY_ACTION_SUBMERGE, 0u, [&]
125  {
126  uint32 spellId = 0;
127  switch (m_phase)
128  {
129  case PHASE_1_SUFFERING: spellId = SPELL_SUMMON_ESSENCE_SUFFERING; break;
130  case PHASE_2_DESIRE: spellId = SPELL_SUMMON_ESSENCE_DESIRE; break;
131  case PHASE_3_ANGER: spellId = SPELL_SUMMON_ESSENCE_ANGER; break;
132  default: return;
133  }
134 
135  m_submerged = true;
136  GuidVector& souls = m_instance->GetEnslavedSouls();
137  DespawnGuids(souls);
138  DoCastSpellIfCan(nullptr, spellId);
139  DoCastSpellIfCan(nullptr, SPELL_SUBMERGE_VISUAL);
140  });
141  AddCustomAction(RELIQUARY_ACTION_SUMMON_SOUL, 0u, [&]
142  {
143  DoCastSpellIfCan(nullptr, SPELL_SUMMON_ENSLAVED_SOUL);
144  DoCastSpellIfCan(nullptr, SPELL_SUMMON_ENSLAVED_SOUL);
145  DoCastSpellIfCan(nullptr, SPELL_SUMMON_ENSLAVED_SOUL);
146  m_soulSummonedCount += 3;
147  if (m_soulSummonedCount < MAX_ENSLAVED_SOULS)
148  ResetTimer(RELIQUARY_ACTION_SUMMON_SOUL, 2400);
149  });
150  Reset();
151  }

231  void SummonedMovementInform(Creature* summoned, uint32 moveType, uint32 pointId) override
232  {
233  if (moveType != POINT_MOTION_TYPE || !pointId || m_phase == PHASE_0_NOT_BEGUN)
234  return;
235 
236  // Switch to next phase when the essence gets back
237  switch (summoned->GetEntry())
238  {
239  case NPC_ESSENCE_SUFFERING:
240  DoScriptText(SUFF_SAY_AFTER, summoned);
241  m_phase = PHASE_2_DESIRE;
242  break;
243  case NPC_ESSENCE_DESIRE:
244  DoScriptText(DESI_SAY_AFTER, summoned);
245  m_phase = PHASE_3_ANGER;
246  break;
247  }
248 
249  // Despawn and set animation
250  summoned->ForcedDespawn();
251 
252  m_soulDeathCount = 0;
253  m_soulSummonedCount = 0;
254  // Reset animation
255  m_submerged = false;
256  m_creature->RemoveAurasDueToSpell(SPELL_SUBMERGE_VISUAL);
257  ResetTimer(RELIQUARY_ACTION_SUMMON_SOUL, 1000);
258  ResetTimer(RELIQUARY_ACTION_SUBMERGE, 41000);
259 
260  if (m_instance)
261  if (Creature* trigger = m_instance->GetSingleCreatureFromStorage(NPC_RELIQUARY_COMBAT_TRIGGER))
262  trigger->SetInCombatWithZone();
263  }
264 
265  // Wrapper to count the dead spirits
266  void DoNotifySouldDead()
267  {
268  ++m_soulDeathCount;
269 
270  // Prepare to summon the essence
271  if (m_soulDeathCount == MAX_ENSLAVED_SOULS)
272  ReduceTimer(RELIQUARY_ACTION_SUBMERGE, 4000);
273  }
