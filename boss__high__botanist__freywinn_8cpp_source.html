126  void UpdateAI(const uint32 uiDiff) override
127  {
128  if (!m_creature->SelectHostileTarget())
129  return;
130 
131  if (m_uiTreeFormTimer < uiDiff)
132  {
133  if (DoCastSpellIfCan(nullptr, SPELL_TRANQUILITY) == CAST_OK)
134  {
135  // Note: This should remove only negative auras
136  m_creature->RemoveAllAuras();
137 
138  DoCastSpellIfCan(nullptr, SPELL_TREE_FORM, CAST_TRIGGERED);
139  DoScriptText(urand(0, 1) ? SAY_TREE_1 : SAY_TREE_2, m_creature);
140 
141  SetCombatMovement(false);
142  m_bCanMoveFree = false;
143  m_uiFrayerTimer = 1000;
144  m_uiTreeFormEndTimer = 45000;
145  m_uiTreeFormTimer = 75000;
146  }
147  }
148  else
149  m_uiTreeFormTimer -= uiDiff;
150 
151  // The Frayer is summoned after one second in the tree phase
152  if (m_uiFrayerTimer)
153  {
154  if (m_uiFrayerTimer <= uiDiff)
155  {
156  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_FRAYER, CAST_TRIGGERED) == CAST_OK)
157  m_uiFrayerTimer = 0;
158  }
159  else
160  m_uiFrayerTimer -= uiDiff;
161  }
162 
163  // Tree phase will be removed when the timer expires;
164  if (m_uiTreeFormEndTimer)
165  {
166  if (m_uiTreeFormEndTimer <= uiDiff)
167  {
168  SetCombatMovement(true);
169  m_bCanMoveFree = true;
170  m_uiTreeFormEndTimer = 0;
171  }
172  else
173  m_uiTreeFormEndTimer -= uiDiff;
174  }
175 
176  // Don't do any other actions during tree form
177  if (!m_bCanMoveFree)
178  return;
179 
180  // one random seedling every 5 secs, but not in tree form
181  if (m_uiSummonSeedlingTimer < uiDiff)
182  {
183  DoSummonSeedling();
184  m_uiSummonSeedlingTimer = 6000;
185  }
186  else
187  m_uiSummonSeedlingTimer -= uiDiff;
188 
189  DoMeleeAttackIfReady();
190  }
