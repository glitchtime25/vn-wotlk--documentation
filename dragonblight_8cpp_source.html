89  void UpdateAI(const uint32 uiDiff) override
90  {
91  if (m_bCanPulse)
92  {
93  if (DoCastSpellIfCan(m_creature, m_uiStack > MAX_STACK ? SPELL_DESTRUCTIVE_BARRAGE : SPELL_DESTRUCTIVE_PULSE) == CAST_OK)
94  m_bCanPulse = false;
95  }
96 
97  if (m_uiSummonTimer)
98  {
99  if (m_uiSummonTimer <= uiDiff)
100  {
101  if (m_bFirst)
102  m_uiSummonTimer = 25000;
103  else
104  m_uiSummonTimer = 0;
105 
106  switch (m_uiStack)
107  {
108  case 0:
109  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_SKELETON, CAST_TRIGGERED);
110  break;
111  case 1:
112  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_CONSTRUCT, CAST_TRIGGERED);
113 
114  if (m_bFirst)
115  break;
116 
117  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_CONSTRUCT, CAST_TRIGGERED);
118  break;
119  case 2:
120  if (m_bFirst)
121  break;
122 
123  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_SKELETON, CAST_TRIGGERED);
124  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_SKELETON, CAST_TRIGGERED);
125  DoCastSpellIfCan(m_creature, SPELL_SUMMON_SMOLDERING_CONSTRUCT, CAST_TRIGGERED);
126  break;
127  }
128 
129  m_bFirst = !m_bFirst;
130  }
131  else
132  m_uiSummonTimer -= uiDiff;
133  }
134 
135  if (!m_uiPowerTimer)
136  return;
137 
138  if (m_uiPowerTimer <= uiDiff)
139  {
140  if (m_uiStack > MAX_STACK)
141  {
142  if (DoCastSpellIfCan(m_creature, SPELL_DESTRUCTIVE_WARD_KILL_CREDIT) == CAST_OK)
143  {
144  DoScriptText(SAY_WARD_CHARGED, m_creature, m_creature->GetOwner());
145  m_uiPowerTimer = 0;
146  m_uiSummonTimer = 0;
147  m_bCanPulse = true;
148  }
149  }
150  else if (DoCastSpellIfCan(m_creature, SPELL_DESTRUCTIVE_WARD_POWERUP) == CAST_OK)
151  {
152  DoScriptText(SAY_WARD_POWERUP, m_creature, m_creature->GetOwner());
153 
154  m_uiPowerTimer = 30000;
155  m_uiSummonTimer = 2000;
156 
157  m_bFirst = true;
158  m_bCanPulse = true; // pulse right after each charge
159 
160  ++m_uiStack;
161  }
162  }
163  else
164  m_uiPowerTimer -= uiDiff;
165  }
166 };
