149  void UpdateAI(const uint32 uiDiff) override
150  {
151  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
152  return;
153 
154  if (m_uiPhase == PHASE_GROUND)
155  {
156  if (m_uiLeechingSwarmTimer < uiDiff)
157  {
158  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_LEECHING_SWARM : SPELL_LEECHING_SWARM_H) == CAST_OK)
159  {
160  switch (urand(0, 2))
161  {
162  case 0: DoScriptText(SAY_LOCUST_1, m_creature); break;
163  case 1: DoScriptText(SAY_LOCUST_2, m_creature); break;
164  case 2: DoScriptText(SAY_LOCUST_3, m_creature); break;
165  }
166 
167  m_uiLeechingSwarmTimer = 19000;
168  }
169  }
170  else
171  m_uiLeechingSwarmTimer -= uiDiff;
172 
173  if (m_uiCarrionBeetlesTimer < uiDiff)
174  {
175  if (DoCastSpellIfCan(m_creature, SPELL_CARRION_BEETLES) == CAST_OK)
176  m_uiCarrionBeetlesTimer = 25000;
177  }
178  else
179  m_uiCarrionBeetlesTimer -= uiDiff;
180 
181  if (m_uiPoundTimer < uiDiff)
182  {
183  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_POUND : SPELL_POUND_H) == CAST_OK)
184  m_uiPoundTimer = 16000;
185  }
186  else
187  m_uiPoundTimer -= uiDiff;
188 
189  if (m_creature->GetHealthPercent() < 100 - 25 * m_uiSubmergePhase)
190  {
191  if (DoCastSpellIfCan(m_creature, SPELL_SUBMERGE) == CAST_OK)
192  {
193  DoCastSpellIfCan(m_creature, SPELL_IMPALE_AURA, CAST_TRIGGERED);
194  DoScriptText(urand(0, 1) ? SAY_SUBMERGE_1 : SAY_SUBMERGE_2, m_creature);
195  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SPAWNING | UNIT_FLAG_UNINTERACTIBLE);
196  m_uiPhase = PHASE_SUBMERGED;
197  m_uiSummonTimer = 5000;
198 
199  // Emerge timers aren't the same. They depend on the submerge phase
200  switch (m_uiSubmergePhase)
201  {
202  case 1: m_uiEmergeTimer = 20000; break;
203  case 2: m_uiEmergeTimer = 45000; break;
204  case 3: m_uiEmergeTimer = 50000; break;
205  }
206  ++m_uiSubmergePhase;
207  }
208  }
209 
210  DoMeleeAttackIfReady();
211  }
212  else if (m_uiPhase == PHASE_SUBMERGED)
213  {
214  if (m_uiSummonTimer < uiDiff)
215  {
216  // summon guardian and assassins
217  DoCastSpellIfCan(m_creature, SPELL_SUMMON_ASSASSIN, CAST_TRIGGERED);
218  DoCastSpellIfCan(m_creature, SPELL_SUMMON_GUARDIAN, CAST_TRIGGERED);
219 
220  // venomancer summoned by trigger
221  if (m_pInstance)
222  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_pInstance->GetVenomancerTrigger()))
223  pTrigger->CastSpell(pTrigger, SPELL_SUMMON_VENOMANCER, TRIGGERED_OLD_TRIGGERED);
224 
225  m_uiSummonTimer = 26000;
226  }
227  else
228  m_uiSummonTimer -= uiDiff;
229 
230  // only on the last submerge phase
231  if (m_uiSubmergePhase == 4)
232  {
233  if (m_uiDarterTimer < uiDiff)
234  {
235  // Darter summoned by trigger
236  if (m_pInstance)
237  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(m_pInstance->GetDarterTrigger()))
238  pTrigger->CastSpell(pTrigger, SPELL_SUMMON_DARTER, TRIGGERED_OLD_TRIGGERED);
239 
240  m_uiDarterTimer = urand(10000, 15000);
241  }
242  else
243  m_uiDarterTimer -= uiDiff;
244  }
245 
246  if (m_uiEmergeTimer < uiDiff)
247  {
248  DoCastSpellIfCan(m_creature, SPELL_EMERGE, CAST_INTERRUPT_PREVIOUS);
249  m_creature->RemoveAurasDueToSpell(SPELL_SUBMERGE);
250  m_creature->RemoveAurasDueToSpell(SPELL_IMPALE_AURA);
251  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SPAWNING | UNIT_FLAG_UNINTERACTIBLE);
252  m_uiPhase = PHASE_GROUND;
253  }
254  else
255  m_uiEmergeTimer -= uiDiff;
256  }
257  }
