25  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
26  {
27  Unit* unitTarget = spell->GetUnitTarget();
28  if (!unitTarget || !unitTarget->IsAlive())
29  return;
30 
31  Unit* caster = spell->GetCaster();
32 
33  uint32 spellId1;
34  uint32 spellId2 = 0;
35 
36  // Judgement self add switch
37  switch (spell->m_spellInfo->Id)
38  {
39  case 53407: spellId1 = 20184; break; // Judgement of Justice
40  case 20271: // Judgement of Light
41  case 57774: spellId1 = 20185; break; // Judgement of Light
42  case 53408: spellId1 = 20186; break; // Judgement of Wisdom
43  default:
44  sLog.outError("Unsupported Judgement (seal trigger) spell (Id: %u) in Spell::EffectScriptEffect", spell->m_spellInfo->Id);
45  return;
46  }
47 
48  // offensive seals have aura dummy in 2 effect
49  Unit::AuraList const& m_dummyAuras = caster->GetAurasByType(SPELL_AURA_DUMMY);
50  for (auto m_dummyAura : m_dummyAuras)
51  {
52  // search seal (offensive seals have judgement's aura dummy spell id in 2 effect
53  if (m_dummyAura->GetEffIndex() != EFFECT_INDEX_2 || !IsSealSpell(m_dummyAura->GetSpellProto()))
54  continue;
55  spellId2 = m_dummyAura->GetModifier()->m_amount;
56  SpellEntry const* judge = sSpellTemplate.LookupEntry<SpellEntry>(spellId2);
57  if (!judge)
58  continue;
59  break;
60  }
61 
62  // if there were no offensive seals than there is seal with proc trigger aura
63  if (!spellId2)
64  {
65  Unit::AuraList const& procTriggerAuras = caster->GetAurasByType(SPELL_AURA_PROC_TRIGGER_SPELL);
66  for (auto procTriggerAura : procTriggerAuras)
67  {
68  if (procTriggerAura->GetEffIndex() != EFFECT_INDEX_0 || !IsSealSpell(procTriggerAura->GetSpellProto()))
69  continue;
70  spellId2 = 54158;
71  break;
72  }
73  }
74 
75  if (spellId1)
76  caster->CastSpell(unitTarget, spellId1, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL);
77 
78  if (spellId2)
79  caster->CastSpell(unitTarget, spellId2, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL);
80 
81  if (caster->HasAura(37188)) // improved judgement
82  caster->CastSpell(nullptr, 43838, TRIGGERED_OLD_TRIGGERED);
83 
84  if (caster->HasAura(40470)) // spell_paladin_tier_6_trinket
85  if (roll_chance_f(50.f))
86  caster->CastSpell(unitTarget, 40472, TRIGGERED_OLD_TRIGGERED);
87  }

151  spell->SendCastResult(SPELL_FAILED_TARGET_AFFECTING_COMBAT);
152  return;

154 
155  // not empty (checked), copy
156  Unit::AttackerSet attackers = unitTarget->getAttackers();
157 
158  // selected from list 3
159  size_t size = std::min(size_t(3), attackers.size());
160  for (uint32 i = 0; i < size; ++i)
161  {
162  Unit::AttackerSet::iterator aItr = attackers.begin();
163  std::advance(aItr, urand() % attackers.size());

245  {
246  currentAbsorb = 0;
247  Player* player = dynamic_cast<Player*>(aura->GetCaster());
248  if (!player)
249  return;
250  if (int32(player->GetHealth()) - remainingDamage > (player->GetMaxHealth() * 0.35))
251  return;
252  int32 reduction = 0;
253  int32 healMod = 0;
254  if (Aura* aur = aura->GetHolder()->GetAuraByEffectIndex(EFFECT_INDEX_0))
255  reduction = aur->GetBasePoints();
256  if (Aura* aur = aura->GetHolder()->GetAuraByEffectIndex(EFFECT_INDEX_1))
257  healMod = aur->GetBasePoints();
258  remainingDamage *= (100 - reduction) / 100.f;
259  if (int32(player->GetHealth()) - remainingDamage > 0 || player->HasAura(66233))
260  return;
261  float defenseFactor = std::min(140u, (player->GetDefenseSkillValue() - player->GetLevel() * 5)) / 140.f;
262  defenseFactor = std::max(0.f, defenseFactor);
263  healMod *= defenseFactor;
264  healMod = player->GetMaxHealth() * (healMod / 100.f);
265  remainingDamage = 0;
266  preventedDeath = true;
267  player->CastCustomSpell(nullptr, 66235, &healMod, nullptr, nullptr, TRIGGERED_OLD_TRIGGERED);
268  player->CastSpell(nullptr, 66233, TRIGGERED_OLD_TRIGGERED);
