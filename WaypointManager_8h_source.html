67  WaypointPath* GetDefaultPath(uint32 entry, uint32 lowGuid, WaypointPathOrigin* wpOrigin = nullptr)
68  {
69  WaypointPath* path = GetPath(lowGuid);
70  if (path && wpOrigin)
71  *wpOrigin = PATH_FROM_GUID;
72 
73  // No movement found for guid
74  if (!path)
75  {
76  path = GetPathTemplate(entry);
77  if (path && wpOrigin)
78  *wpOrigin = PATH_FROM_ENTRY;
79 
80  if (!path)
81  {
82  // check movement_template
83  path = GetPathMovementTemplate(entry);
84  if (path)
85  *wpOrigin = PATH_FROM_WAYPOINT_PATH;
86  }
87  }
88 
89  return path;
90  }

92  // Helper function to get a path provided the required information
93  WaypointPath* GetPathFromOrigin(uint32 entry, uint32 lowGuid, int32 pathId, WaypointPathOrigin wpOrigin)
94  {
95  WaypointPathMap* wpMap;
96  uint32 key;
97 
98  switch (wpOrigin)
99  {
100  case PATH_FROM_GUID:
101  key = lowGuid;
102  wpMap = &m_pathMap;
103  break;
104  case PATH_FROM_ENTRY:
105  if (pathId >= 0xFF || pathId < 0)
106  return nullptr;
107  key = (entry << 8) + pathId;
108  wpMap = &m_pathTemplateMap;
109  break;
110  case PATH_FROM_EXTERNAL:
111  if (pathId >= 0xFF || pathId < 0)
112  return nullptr;
113  key = (entry << 8) + pathId;
114  wpMap = &m_externalPathTemplateMap;
115  break;
116  case PATH_FROM_WAYPOINT_PATH:
117  key = pathId;
118  wpMap = &m_pathMovementTemplateMap;
119  break;
120  case PATH_NO_PATH:
121  default:
122  return nullptr;
123  }
124  WaypointPathMap::iterator find = wpMap->find(key);
125  return find != wpMap->end() ? &find->second : nullptr;
126  }
