63 void instance_blood_furnace::OnObjectCreate(GameObject* go)
64 {
65  switch (go->GetEntry())
66  {
67  case GO_DOOR_MAKER_FRONT: // the maker front door
68  break;
69  case GO_DOOR_MAKER_REAR: // the maker rear door
70  if (m_auiEncounter[TYPE_THE_MAKER_EVENT] == DONE)
71  go->SetGoState(GO_STATE_ACTIVE);
72  break;
73  case GO_DOOR_BROGGOK_FRONT: // broggok front door
74  break;
75  case GO_DOOR_BROGGOK_REAR: // broggok rear door
76  if (m_auiEncounter[TYPE_BROGGOK_EVENT] == DONE)
77  go->SetGoState(GO_STATE_ACTIVE);
78  break;
79  case GO_DOOR_KELIDAN_EXIT: // kelidan exit door
80  if (m_auiEncounter[TYPE_KELIDAN_EVENT] == DONE)
81  go->SetGoState(GO_STATE_ACTIVE);
82  break;
83  case GO_DOOR_FINAL_EXIT: // final exit door
84  if (m_auiEncounter[TYPE_KELIDAN_EVENT] == DONE)
85  go->SetGoState(GO_STATE_ACTIVE);
86  break;
87 
88  case GO_PRISON_CELL_BROGGOK_1: m_aBroggokEvent[0].m_cellGuid = go->GetObjectGuid(); return;
89  case GO_PRISON_CELL_BROGGOK_2: m_aBroggokEvent[1].m_cellGuid = go->GetObjectGuid(); return;
90  case GO_PRISON_CELL_BROGGOK_3: m_aBroggokEvent[2].m_cellGuid = go->GetObjectGuid(); return;
91  case GO_PRISON_CELL_BROGGOK_4: m_aBroggokEvent[3].m_cellGuid = go->GetObjectGuid(); return;
92 
93  case GO_PRISON_CELL_DOOR_LEVER:
94  m_lLeverGO = go;
95 
96  if (m_auiEncounter[TYPE_BROGGOK_EVENT] != DONE && m_auiEncounter[TYPE_BROGGOK_EVENT] != IN_PROGRESS)
97  {
98  if (m_lLeverGO)
99  {
100  m_lLeverGO->SetRespawnTime(time(nullptr));
101  m_lLeverGO->Respawn();
102  }
103  }
104  break;
105  case GO_CRACK_1:
106  case GO_CRACK_2:
107  case GO_CRACK_3:
108  case GO_CRACK_4:
109  case GO_CRACK_5:
110  case GO_CRACK_6:
111  case GO_CRACK_7:
112  m_cracks.push_back(go->GetObjectGuid());
113  return;
114  default:
115  return;
116  }
117  m_goEntryGuidStore[go->GetEntry()] = go->GetObjectGuid();
118 }

120 void instance_blood_furnace::SetData(uint32 type, uint32 data)
121 {
122  switch (type)
123  {
124  case TYPE_THE_MAKER_EVENT:
125  if (data == IN_PROGRESS)
126  DoUseDoorOrButton(GO_DOOR_MAKER_FRONT);
127  if (data == FAIL)
128  DoUseDoorOrButton(GO_DOOR_MAKER_FRONT);
129  if (data == DONE)
130  {
131  DoUseDoorOrButton(GO_DOOR_MAKER_FRONT);
132  DoUseDoorOrButton(GO_DOOR_MAKER_REAR);
133  }
134  m_auiEncounter[type] = data;
135  break;
136  case TYPE_BROGGOK_EVENT:
137  if (m_auiEncounter[type] == data)
138  return;
139 
140  // Combat door; the exit door is opened in event
141  DoUseDoorOrButton(GO_DOOR_BROGGOK_FRONT);
142  if (data == IN_PROGRESS)
143  {
144  if (Creature* broggok = GetSingleCreatureFromStorage(NPC_BROGGOK))
145  {
146  broggok->CastSpell(nullptr, SPELL_COMBAT_TRIGGER, TRIGGERED_OLD_TRIGGERED); // cast doesnt show in sniff
147  DoScriptText(SAY_BROGGOK_INTRO, broggok);
148  }
149 
150  if (m_uiBroggokEventPhase <= MAX_ORC_WAVES)
151  {
152  m_uiBroggokEventPhase = 0;
153 
154  if (m_aBroggokEvent[0].m_sSortedOrcGuids.empty())
155  DoSortBroggokOrcs();
156 
157  // open first cage
158  DoNextBroggokEventPhase();
159  }
160  }
161  else if (data == FAIL)
162  {
163  // On wipe we reset only the orcs; if the party wipes at the boss itself then the orcs don't reset
164  if (m_uiBroggokEventPhase <= MAX_ORC_WAVES)
165  {
166  for (auto& i : m_aBroggokEvent)
167  {
168  // Reset Orcs
169  if (!i.m_bIsCellOpened)
170  continue;
171 
172  i.m_uiKilledOrcCount = 0;
173  for (GuidSet::const_iterator itr = i.m_sSortedOrcGuids.begin(); itr != i.m_sSortedOrcGuids.end(); ++itr)
174  {
175  if (Creature* pOrc = instance->GetCreature(*itr))
176  {
177  pOrc->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
178 
179  if (pOrc->IsAlive())
180  pOrc->ForcedDespawn();
181 
182  pOrc->Respawn();
183  }
184  }
185 
186  // Close Door
187  DoUseDoorOrButton(i.m_cellGuid);
188  i.m_bIsCellOpened = false;
189  }
190 
191  if (m_lLeverGO)
192  {
193  m_lLeverGO->SetRespawnTime(time(nullptr));
194  m_lLeverGO->Respawn();
195  }
196  }
197  }
198  if (data == FAIL || data == DONE)
199  if (Creature* trigger = GetSingleCreatureFromStorage(NPC_IN_COMBAT_TRIGGER))
200  trigger->ForcedDespawn();
201  m_auiEncounter[type] = data;
202  break;
203  case TYPE_KELIDAN_EVENT:
204  if (data == DONE)
205  {
206  DoUseDoorOrButton(GO_DOOR_KELIDAN_EXIT);
207  DoUseDoorOrButton(GO_DOOR_FINAL_EXIT);
208  }
209  m_auiEncounter[type] = data;
210  break;
211  default:
212  script_error_log("Instance Blood Furnace SetData with Type %u Data %u, but this is not implemented.", type, data);
213  return;
214  }
215 
216  if (data == DONE)
217  {
218  OUT_SAVE_INST_DATA;
219 
220  std::ostringstream saveStream;
221  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2];
222 
223  m_strInstData = saveStream.str();
224 
225  SaveToDB();
226  OUT_SAVE_INST_DATA_COMPLETE;
227  }

257 void instance_blood_furnace::DoNextBroggokEventPhase()
258 {
259  // Open door to the final boss now and move boss to the center of the room
260  if (m_uiBroggokEventPhase >= MAX_ORC_WAVES)
261  {
262  DoUseDoorOrButton(GO_DOOR_BROGGOK_REAR);
263 
264  if (Creature* pBroggok = GetSingleCreatureFromStorage(NPC_BROGGOK))
265  {
266  pBroggok->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
267  pBroggok->SetWalk(false);
268  pBroggok->GetMotionMaster()->Clear(false, true);
269  pBroggok->GetMotionMaster()->MoveWaypoint();
270  }
271  }
272  else
273  {
274  // Open cage door
275  if (!m_aBroggokEvent[m_uiBroggokEventPhase].m_bIsCellOpened)
276  DoUseDoorOrButton(m_aBroggokEvent[m_uiBroggokEventPhase].m_cellGuid);
277 
278  m_aBroggokEvent[m_uiBroggokEventPhase].m_bIsCellOpened = true;
279 
280  // TODO: add small delay - after gate opened
281  for (auto m_sSortedOrcGuid : m_aBroggokEvent[m_uiBroggokEventPhase].m_sSortedOrcGuids)
282  {
283  if (Creature* pOrc = instance->GetCreature(m_sSortedOrcGuid))
284  {
285  pOrc->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
286  pOrc->SetInCombatWithZone();
287  }
288  }
289  }
290 
291  // Prepare for further handling
292  m_uiBroggokEventTimer = 90 * IN_MILLISECONDS;
293  ++m_uiBroggokEventPhase;
294 }

296 void instance_blood_furnace::OnCreatureEvade(Creature* creature)
297 {
298  if (m_auiEncounter[TYPE_BROGGOK_EVENT] == FAIL)

305  {

314 void instance_blood_furnace::OnCreatureDeath(Creature* creature)
315 {
316  if (m_auiEncounter[TYPE_BROGGOK_EVENT] != IN_PROGRESS)
317  return;
318 
319  if (creature->GetEntry() == NPC_NASCENT_FEL_ORC || creature->GetEntry() == NPC_FEL_ORC_NEOPHYTE)
320  {
321  uint8 uiClearedCells = 0;
322  for (uint8 i = 0; i < std::min<uint32>(m_uiBroggokEventPhase, MAX_ORC_WAVES); ++i)
323  {
324  if (m_aBroggokEvent[i].m_sSortedOrcGuids.size() == m_aBroggokEvent[i].m_uiKilledOrcCount)
325  {
326  ++uiClearedCells;
327  continue;
328  }
329 
330  // Increase kill counter, if we found a mob of this cell
331  if (m_aBroggokEvent[i].m_sSortedOrcGuids.find(creature->GetObjectGuid()) != m_aBroggokEvent[i].m_sSortedOrcGuids.end())
332  m_aBroggokEvent[i].m_uiKilledOrcCount++;
333 
334  if (m_aBroggokEvent[i].m_sSortedOrcGuids.size() == m_aBroggokEvent[i].m_uiKilledOrcCount)
335  ++uiClearedCells;
336  }
337 
338  // Increase phase when all opened cells are cleared
339  if (uiClearedCells == m_uiBroggokEventPhase)
340  DoNextBroggokEventPhase();
341  }

344 void instance_blood_furnace::Update(uint32 uiDiff)
345 {
346  // Broggok Event: For the last wave we don't check the timer; the boss is released only when all mobs die
347  if (m_auiEncounter[TYPE_BROGGOK_EVENT] == IN_PROGRESS)
348  if (m_uiBroggokEventPhase <= MAX_ORC_WAVES)
349  {
350  if (!GetPlayerInMap(true, false))
351  SetData(TYPE_BROGGOK_EVENT, FAIL);
352  else if (!instance->IsRegularDifficulty())
353  {
354  if (m_uiBroggokEventTimer < uiDiff)
355  DoNextBroggokEventPhase();
356  else
357  m_uiBroggokEventTimer -= uiDiff;
358  }
359  }
360 
361  if (m_uiRandYellTimer < uiDiff)
362  {
363  if (Creature* pMagtheridon = GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
364  {
365  DoScriptText(aRandomTaunt[urand(0, 5)], pMagtheridon);
366  m_uiRandYellTimer = 90000;
367  }
368  }
369  else
370  m_uiRandYellTimer -= uiDiff;
371 
372  if (m_crackTimer <= uiDiff)
373  {
374  m_crackTimer = 30000;
375  for (ObjectGuid guid : m_cracks)
376  {
377  if (GameObject* crack = instance->GetGameObject(guid))
378  {
379  if (crack->GetGoState() != GO_STATE_READY)
380  if (urand(0, 5))
381  crack->SetGoState(GO_STATE_READY);
382 
383  if (crack->GetGoState() == GO_STATE_READY)
384  if (urand(0, 2))
385  crack->SendGameObjectCustomAnim(crack->GetObjectGuid(), 2);
386  }
387  }
388  }
389  else
390  m_crackTimer -= uiDiff;
391 }

443 // Sort all nascent orcs & fel orc neophytes in the instance in order to get only those near broggok doors
444 void instance_blood_furnace::DoSortBroggokOrcs()
445 {
446  for (GuidList::const_iterator itr = m_luiNascentOrcGuids.begin(); itr != m_luiNascentOrcGuids.end(); ++itr)
447  {
448  if (Creature* pOrc = instance->GetCreature(*itr))
449  {
450  for (auto& i : m_aBroggokEvent)
451  {
452  if (GameObject* pDoor = instance->GetGameObject(i.m_cellGuid))
453  {
454  Position orcPos = pOrc->GetRespawnPosition();
455  if (pDoor->IsWithinDist3d(orcPos.x, orcPos.y, orcPos.z, 16.0f) && orcPos.GetPositionZ() < 15.0f)
456  {
457  i.m_sSortedOrcGuids.insert(pOrc->GetObjectGuid());
458  if (!pOrc->IsAlive())
459  pOrc->Respawn();
460  break;
461  }
462  }
463  }
464  }
