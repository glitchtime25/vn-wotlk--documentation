100  void ExecuteAction(uint32 action) override
101  {
102  switch (action)
103  {
104  case GROBBULUS_CHECK_MELEE:

107  {
108  ResetCombatAction(GROBBULUS_SLIME_STREAM, 5s);

114  }
115  case GROBBULUS_SLIME_STREAM:
116  {
117  if (!m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
118  DoCastSpellIfCan(nullptr, SPELL_SLIME_STREAM, CAST_TRIGGERED);
119 
120  ResetCombatAction(GROBBULUS_CHECK_MELEE, GetSubsequentActionTimer(GROBBULUS_CHECK_MELEE));
121  DisableCombatAction(action);
122  break;
123  }
124  }

128 struct MutatingInjection : public AuraScript
129 {
130  void OnApply(Aura* aura, bool apply) const override
131  {
132  if (!apply)
133  {
134  Unit* target = aura->GetTarget();
135  if (!target)
136  return;
137 
138  if (aura->GetRemoveMode() == AURA_REMOVE_BY_EXPIRE)
139  // Embalming Cloud
140  target->CastSpell(target, SPELL_EMBALMING_CLOUD, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
141  else // Removed by dispel
142  // Mutagen Explosion
143  target->CastSpell(target, SPELL_MUTAGEN_EXPLOSION, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
144  // Poison Cloud
145  target->CastSpell(target, SPELL_POISON_CLOUD, TRIGGERED_OLD_TRIGGERED, nullptr, aura);
146  }
147  else
148  {
149  Unit* target = aura->GetTarget();
150  Unit* caster = aura->GetCaster();
151  if (!target || caster)
152  return;
153  DoBroadcastText(EMOTE_INJECTION, caster, target);
154  }
155  }
