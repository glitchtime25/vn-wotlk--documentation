224  void ReceiveAIEvent(AIEventType eventType, Unit* sender, Unit* /*invoker*/, uint32 /*miscValue*/) override
225  {
226  if (eventType == AI_EVENT_CUSTOM_A)
227  {
228  // Note: this is called from the Shade, Channeler and Sorcerer script
229  // If the function is changed in the future, please review this.
230  switch (sender->GetEntry())
231  {
232  case NPC_SHADE_OF_AKAMA:
233  m_phase = PHASE_EPILOGUE;
234 
235  m_creature->GetMotionMaster()->MovePoint(PHASE_EPILOGUE, akamaWaypoints[2].x, akamaWaypoints[2].y, akamaWaypoints[2].z);
236  break;
237  case NPC_ASH_CHANNELER:
238  // Move the shade to Akama when all channelers are dead
239  // Note: the boss should be already slowly moving, but this isn't possible because of the missing stack for the speed debuff
240  break;
241  }
242  }
243  else if (eventType == AI_EVENT_CUSTOM_B)
244  {
245  // When the Shade starts to attack Akama, switch to melee phase
246  if (m_phase == PHASE_CHANNEL && sender->GetEntry() == NPC_SHADE_OF_AKAMA)
247  {
248  m_creature->InterruptNonMeleeSpells(false);
249  SetCombatMovement(true);
250  sender->AI()->AttackStart(m_creature);
251  m_phase = PHASE_COMBAT;
252  sender->CastSpell(nullptr, SPELL_THREAT, TRIGGERED_NONE);
253  m_creature->CastSpell(nullptr, SPELL_FIXATE, TRIGGERED_NONE);
254 
255  ResetTimer(AKAMA_ACTION_DESTRUCTIVE_POISON, GetInitialActionTimer(AKAMA_ACTION_DESTRUCTIVE_POISON));
256  ResetTimer(AKAMA_ACTION_CHAIN_LIGHTNING, GetInitialActionTimer(AKAMA_ACTION_CHAIN_LIGHTNING));
257 
258  for (ObjectGuid guid : m_instance->GetGeneratorGuidVector()) // notify generators to despawn sorcerers
259  if (Creature* generator = m_creature->GetMap()->GetCreature(guid))
260  m_creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_C, m_creature, generator);
261 
262  for (ObjectGuid guid : m_instance->GetChannelersGuidList()) // despawn channelers
263  if (Creature* channeler = m_creature->GetMap()->GetCreature(guid))
264  channeler->ForcedDespawn();
265  }
266  }
267  }
