196 World::AddSession_(WorldSession* s)
197 {
198  MANGOS_ASSERT(s);
199 
200  if (FindSession(s->GetAccountId()))
201  {
202  sLog.outError("Trying to add an already existing session");
203  return;
204  }
205 
206  // NOTE - Still there is race condition in WorldSession* being used in the Sockets
207 
210  if (!RemoveSession(s->GetAccountId()))
211  {
212  s->KickPlayer();
213  delete s; // session not added yet in session list, so not listed in queue
214  return;
215  }
216 
217  // decrease session counts only at not reconnection case
218  bool decrease_session = true;
219 
220  // if session already exist, prepare to it deleting at next world update
221  // NOTE - KickPlayer() should be called on "old" in RemoveSession()
222  {
223  SessionMap::const_iterator old = m_sessions.find(s->GetAccountId());
224 
225  if (old != m_sessions.end())
226  {
227  // prevent decrease sessions count if session queued
228  if (RemoveQueuedSession(old->second))
229  decrease_session = false;
230  // do not remove replaced session from queue if listed
231  delete old->second;
232  }
233  }
234 
235  m_sessions[s->GetAccountId()] = s;
236 
237  uint32 Sessions = GetActiveAndQueuedSessionCount();
238  uint32 pLimit = GetPlayerAmountLimit();
239  uint32 QueueSize = GetQueuedSessionCount(); // number of players in the queue
240 
241  // so we don't count the user trying to
242  // login as a session and queue the socket that we are using
243  if (decrease_session)
244  --Sessions;
245 
246  if (m_uniqueSessionCount.find(s->GetAccountId()) == m_uniqueSessionCount.end())
247  m_uniqueSessionCount.insert(s->GetAccountId());
248 
249  if (pLimit > 0 && Sessions >= pLimit && s->GetSecurity() == SEC_PLAYER)
250  {
251  AddQueuedSession(s);
252  UpdateMaxSessionCounters();
253  DETAIL_LOG("PlayerQueue: Account id %u is in Queue Position (%u).", s->GetAccountId(), ++QueueSize);

255  }

257  WorldPacket pkt(SMSG_CLIENTCACHE_VERSION, 4);

260 
261  s->SendTutorialsData();
262 
263  UpdateMaxSessionCounters();
264 
265  // Updates the population
266  if (pLimit > 0)
267  {
268  float popu = float(GetActiveSessionCount()); // updated number of users on the server
269  popu /= pLimit;
270  popu *= 2;

272  static SqlStatementID id;

309 bool World::RemoveQueuedSession(WorldSession* sess)
310 {
311  // sessions count including queued to remove (if removed_session set)
312  uint32 sessions = GetActiveSessionCount();
313 
314  uint32 position = 1;
315  Queue::iterator iter = m_QueuedSessions.begin();
316 
317  // search to remove and count skipped positions
318  bool found = false;
319 
320  for (; iter != m_QueuedSessions.end(); ++iter, ++position)
321  {
322  if (*iter == sess)
323  {
324  sess->SetInQueue(false);
325  iter = m_QueuedSessions.erase(iter);
326  found = true; // removing queued session
327  break;
328  }
329  }
330 
331  // iter point to next socked after removed or end()
332  // position store position of removed socket and then new position next socket after removed
333 
334  // if session not queued then we need decrease sessions count
335  if (!found && sessions)
336  --sessions;
337 
338  // accept first in queue
339  if ((!m_playerLimit || (int32)sessions < m_playerLimit) && !m_QueuedSessions.empty())
340  {
341  WorldSession* pop_sess = m_QueuedSessions.front();
342  pop_sess->SetInQueue(false);
343  pop_sess->SendAuthWaitQue(0);
344 
345  WorldPacket pkt(SMSG_CLIENTCACHE_VERSION, 4);
346  pkt << uint32(getConfig(CONFIG_UINT32_CLIENTCACHE_VERSION));
347  pop_sess->SendPacket(pkt);
348 
349  pop_sess->SendAccountDataTimes(GLOBAL_CACHE_MASK);
350  pop_sess->SendTutorialsData();
351 
352  m_QueuedSessions.pop_front();
353 
354  // update iter to point first queued socket or end() if queue is empty now
355  iter = m_QueuedSessions.begin();
356  position = 1;
357  }
358 
359  // update position from iter to end()
360  // iter point to first not updated socket, position store new position
361  for (; iter != m_QueuedSessions.end(); ++iter, ++position)
362  (*iter)->SendAuthWaitQue(position);
363 
364  return found;
365 }
