163  void UpdateAI(const uint32 diff)
164  {
165  switch (m_phase)
166  {
167  case PHASE_CALLING:
168  if (m_callTimer < diff)
169  {
170  for (auto ventureNpc:m_ventureNpcList)
171  {
172  if (!ventureNpc || !ventureNpc->IsAlive() || ventureNpc->IsInCombat())
173  continue;
174  float x, y, z;
175  m_go->GetContactPoint(ventureNpc, x, y, z, 10.0f);
176  ventureNpc->SetWalk(false);
177  ventureNpc->GetMotionMaster()->MoveIdle();
178  ventureNpc->GetMotionMaster()->MovePoint(1, x, y, z);
179  }
180  m_callTimer = 0;
181  m_phase = PHASE_HOMING;
182  }
183  else
184  m_callTimer -= diff;
185  break;
186  case PHASE_HOMING:
187  if (m_returnTimer < diff)
188  {
189  for (auto ventureNpc:m_ventureNpcList)
190  {
191  if (!ventureNpc || !ventureNpc->IsAlive() || ventureNpc->IsInCombat())
192  continue;
193  ventureNpc->SetWalk(true);
194  ventureNpc->AI()->EnterEvadeMode();
195  }
196  m_returnTimer = 0;
197  m_phase = PHASE_INACTIVE;
198  }
199  else
200  m_returnTimer -= diff;
201  break;
202  default:
203  break;
204  }
205  }
206 };
