205  void ExecuteActions()
206  {
207  if (!CanExecuteCombatAction())
208  return;
209 
210  for (uint32 i = 0; i < KRUUL_ACTION_MAX; ++i)
211  {
212  if (GetActionReadyStatus(i))
213  {
214  switch (i)
215  {
216  case KRUUL_ACTION_SUPREME:
217  {
218  // Enrage - cast shadowbolt volley every second
219  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
220  {
221  DoScriptText(EMOTE_FRENZY, m_creature);
222  DisableCombatAction(KRUUL_ACTION_SUPREME);
223  return;
224  }
225  break;
226  }
227  case KRUUL_ACTION_VOIDBOLT:
228  {
229  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_VOIDBOLT) == CAST_OK)
230  {
231  ResetTimer(i, GetSubsequentActionTimer(i));
232  SetActionReadyStatus(i, false);
233  return;
234  }
235  break;
236  }
237  case KRUUL_ACTION_TWISTED_REFLECTION:
238  {
239  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_TWISTED_REFLECTION, SELECT_FLAG_PLAYER))
240  {
241  if (DoCastSpellIfCan(pTarget, SPELL_TWISTED_REFLECTION) == CAST_OK)
242  {
243  ResetTimer(i, GetSubsequentActionTimer(i));
244  SetActionReadyStatus(i, false);
245  return;
246  }
247  }
248  break;
249  }
250  case KRUUL_ACTION_MARK_OF_KAZZAK:
251  {
252  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MARK_OF_KAZZAK, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
253  {
254  if (DoCastSpellIfCan(pTarget, SPELL_MARK_OF_KAZZAK) == CAST_OK)
255  {
256  ResetTimer(i, GetSubsequentActionTimer(i));
257  SetActionReadyStatus(i, false);
258  return;
259  }
260  break;
261  }
262  }
263  case KRUUL_ACTION_SHADOW_BOLT_VOLLEY:
264  {
265  // Cast shadowbolt volley on timer before Berserk
266  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_VOLLEY, SELECT_FLAG_USE_EFFECT_RADIUS) == CAST_OK)
267  {
268  ResetTimer(i, GetSubsequentActionTimer(i));
269  SetActionReadyStatus(i, false);
270  return;
271  }
272  break;
273  }
274  case KRUUL_ACTION_THUNDERCLAP:
275  {
276  if (DoCastSpellIfCan(nullptr, SPELL_THUNDERCLAP, SELECT_FLAG_USE_EFFECT_RADIUS) == CAST_OK)
277  {
278  ResetTimer(i, GetSubsequentActionTimer(i));
279  SetActionReadyStatus(i, false);
280  return;
281  }
282  break;
283  }
284  case KRUUL_ACTION_CLEAVE:
285  {
286  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
287  {
288  ResetTimer(i, GetSubsequentActionTimer(i));
289  SetActionReadyStatus(i, false);
290  return;
291  }
292  break;
293  }
294  case KRUUL_ACTION_KAZZAKS_ASSAULT:
295  {
296  if (m_creature->GetVictim()->GetTypeId() != TYPEID_PLAYER)
297  {
298  if (DoCastSpellIfCan(m_creature, SPELL_KAZZAKS_ASSAULT) == CAST_OK)
299  {
300  ResetTimer(i, GetSubsequentActionTimer(i));
301  SetActionReadyStatus(i, false);
302  return;
303  }
304  }
305  break;
306  }
307  }
308  }
309  }

312  void UpdateAI(const uint32 diff) override
313  {
314  UpdateTimers(diff, m_creature->IsInCombat());
315 
316  if (m_uiCreateInfernalingSummonerTimer)
317  {
318  if (m_uiCreateInfernalingSummonerTimer <= diff)
319  {
320  DoCastSpellIfCan(m_creature, SPELL_CREATE_INFERNALING_SUMMONER);
321  m_uiCreateInfernalingSummonerTimer = 6 * MINUTE * IN_MILLISECONDS;
322  }
323  else
324  m_uiCreateInfernalingSummonerTimer -= diff;
325  }
326 
327  if (m_uiDespawnTimer)
328  {
329  if (m_uiDespawnTimer <= diff)
330  {
331  if (m_creature->IsAlive())
332  {
333  DoScriptText(SAY_DESPAWN, m_creature);
334  sWorldState.HandleExternalEvent(CUSTOM_EVENT_HIGHLORD_KRUUL_DIED, 0);
335  m_creature->ForcedDespawn(5000);
336  }
337  }
338  else
339  m_uiDespawnTimer -= diff;
340  }
341 
342  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
343  return;
344 
345  ExecuteActions();
346  DoMeleeAttackIfReady();
347  }
