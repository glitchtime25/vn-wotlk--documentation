110 uint32 GetSpellCastTime(SpellEntry const* spellInfo, WorldObject* caster, Spell* spell, bool consume)
111 {
112  if (spell)
113  {
114  // Workaround for custom cast time
115  switch (spellInfo->Id)
116  {
117  case 3366: // Opening - seems to have a settable timer per usage
118  if (Item* item = spell->GetCastItem())
119  {
120  switch (item->GetEntry())
121  {
122  case 31088: // Tainted Core - instant opening
123  return 0;
124  default:
125  break;
126  }
127  }
128  break;
129  case 46546: // Ritual of Summoning
130  return 0;
131  default:
132  break;
133  }
134 
135  // some triggered spells have data only usable for client
136  if (spell->IsTriggeredSpellWithRedundantCastTime())
137  return 0;
138 
139  // spell targeted to non-trading trade slot item instant at trade success apply
140  if (caster->IsPlayer())
141  if (TradeData* my_trade = static_cast<Player*>(caster)->GetTradeData())
142  if (Item* nonTrade = my_trade->GetTraderData()->GetItem(TRADE_SLOT_NONTRADED))
143  if (nonTrade == spell->m_targets.getItemTarget())
144  return 0;
145  }
146 
147  SpellCastTimesEntry const* spellCastTimeEntry = sSpellCastTimesStore.LookupEntry(spellInfo->CastingTimeIndex);
148 
149  // not all spells have cast time index and this is all is passive abilities
150  if (!spellCastTimeEntry)
151  return 0;
152 
153  int32 spellRank = caster && caster->GetTypeId() != TYPEID_GAMEOBJECT ? static_cast<Unit*>(caster)->GetSpellRank(spellInfo) : 0;
154  int32 castTime = spellCastTimeEntry->CastTime + spellCastTimeEntry->CastTimePerLevel * (spellRank / 5 - spellInfo->baseLevel);
155  castTime = std::max(castTime, spellCastTimeEntry->MinCastTime);
156 
157  // Hunter Ranged spells need cast time + 0.5s to reflect tooltips, excluding Auto Shot
158  if (spellInfo->HasAttribute(SPELL_ATTR_USES_RANGED_SLOT) && (!spell || !spell->IsAutoRepeat()))
159  castTime += 500;
160 
161  if (spell)
162  {
163  if (Player* modOwner = caster->GetSpellModOwner())
164  modOwner->ApplySpellMod(spellInfo->Id, SPELLMOD_CASTING_TIME, castTime, consume);
165 
166  if (caster->IsUnit())
167  {
168  Unit* unitCaster = static_cast<Unit*>(caster);
169  if (!spellInfo->HasAttribute(SPELL_ATTR_IS_ABILITY) && !spellInfo->HasAttribute(SPELL_ATTR_IS_TRADESKILL) && !spellInfo->HasAttribute(SPELL_ATTR_EX3_IGNORE_CASTER_MODIFIERS))
170  castTime = int32(castTime * unitCaster->GetFloatValue(UNIT_MOD_CAST_SPEED));
171  else if (spell->IsRangedSpell() && !spell->IsAutoRepeat())
172  castTime = int32(castTime * unitCaster->m_modAttackSpeedPct[RANGED_ATTACK]);
173  }
174  }
175 
176  return (castTime > 0) ? uint32(castTime) : 0;
177 }

179 uint32 GetSpellCastTimeForBonus(SpellEntry const* spellProto, DamageEffectType damagetype)
180 {
181  uint32 CastingTime = (!IsChanneledSpell(spellProto)) ? GetSpellCastTime(spellProto, nullptr) : GetSpellDuration(spellProto);
182 
183  if (CastingTime > 7000) CastingTime = 7000;
184  if (CastingTime < 1500) CastingTime = 1500;
185 
186  if (damagetype == DOT && !IsChanneledSpell(spellProto))
187  CastingTime = 3500;
188 
189  int32 overTime = 0;
190  uint8 effects = 0;
191  bool DirectDamage = false;
192  bool AreaEffect = false;
193 
194  for (uint32 i = 0; i < MAX_EFFECT_INDEX; ++i)
195  if (IsAreaEffectTarget(SpellTarget(spellProto->EffectImplicitTargetA[i])) || IsAreaEffectTarget(SpellTarget(spellProto->EffectImplicitTargetB[i])))
196  AreaEffect = true;
197 
198  for (uint32 i = 0; i < MAX_EFFECT_INDEX; ++i)
199  {
200  switch (spellProto->Effect[i])
201  {
202  case SPELL_EFFECT_SCHOOL_DAMAGE:
203  case SPELL_EFFECT_POWER_DRAIN:
204  case SPELL_EFFECT_HEALTH_LEECH:
205  case SPELL_EFFECT_ENVIRONMENTAL_DAMAGE:
206  case SPELL_EFFECT_POWER_BURN:
207  case SPELL_EFFECT_HEAL:
208  DirectDamage = true;
209  break;
210  case SPELL_EFFECT_APPLY_AURA:
211  switch (spellProto->EffectApplyAuraName[i])
212  {
213  case SPELL_AURA_PERIODIC_DAMAGE:
214  case SPELL_AURA_PERIODIC_HEAL:
215  case SPELL_AURA_PERIODIC_LEECH:
216  if (GetSpellDuration(spellProto))
217  overTime = GetSpellDuration(spellProto);
218  break;
219  // Penalty for additional effects
220  case SPELL_AURA_DUMMY:
221  ++effects;
222  break;
223  case SPELL_AURA_MOD_DECREASE_SPEED:
224  ++effects;
225  break;
226  case SPELL_AURA_MOD_CONFUSE:
227  case SPELL_AURA_MOD_STUN:
228  case SPELL_AURA_MOD_ROOT:
229  // -10% per effect
230  effects += 2;
231  break;
232  default:
233  break;
234  }
235  break;
236  default:
237  break;
238  }
239  }
240 
241  // Combined Spells with Both Over Time and Direct Damage
242  if (overTime > 0 && CastingTime > 0 && DirectDamage)
243  {
244  // mainly for DoTs which are 3500 here otherwise
245  uint32 OriginalCastTime = GetSpellCastTime(spellProto, nullptr);
246  if (OriginalCastTime > 7000) OriginalCastTime = 7000;
247  if (OriginalCastTime < 1500) OriginalCastTime = 1500;
248  // Portion to Over Time
249  float PtOT = (overTime / 15000.0f) / ((overTime / 15000.0f) + (OriginalCastTime / 3500.0f));
250 
251  if (damagetype == DOT)
252  CastingTime = uint32(CastingTime * PtOT);
253  else if (PtOT < 1.0f)
254  CastingTime = uint32(CastingTime * (1 - PtOT));
255  else
256  CastingTime = 0;
257  }
258 
259  // Area Effect Spells receive only half of bonus
260  if (AreaEffect)
261  CastingTime /= 2;
262 
263  // 50% for damage and healing spells for leech spells from damage bonus and 0% from healing
264  for (int j = 0; j < MAX_EFFECT_INDEX; ++j)
265  {
266  if (spellProto->Effect[j] == SPELL_EFFECT_HEALTH_LEECH ||
267  (spellProto->Effect[j] == SPELL_EFFECT_APPLY_AURA && spellProto->EffectApplyAuraName[j] == SPELL_AURA_PERIODIC_LEECH))
268  {
269  CastingTime /= 2;
270  break;
271  }
272  }
273 
274  // -5% of total per any additional effect (multiplicative)
275  for (int i = 0; i < effects; ++i)
276  CastingTime *= 0.95f;
277 
278  return CastingTime;
279 }

281 uint16 GetSpellAuraMaxTicks(SpellEntry const* spellInfo)
282 {
283  int32 DotDuration = GetSpellDuration(spellInfo);
284  if (DotDuration == 0)
285  return 1;
286 
287  // 200% limit
288  if (DotDuration > 30000)
289  DotDuration = 30000;
290 
291  for (int j = 0; j < MAX_EFFECT_INDEX; ++j)
292  {
293  if (spellInfo->Effect[j] == SPELL_EFFECT_APPLY_AURA && (
294  spellInfo->EffectApplyAuraName[j] == SPELL_AURA_PERIODIC_DAMAGE ||
295  spellInfo->EffectApplyAuraName[j] == SPELL_AURA_PERIODIC_HEAL ||
296  spellInfo->EffectApplyAuraName[j] == SPELL_AURA_PERIODIC_LEECH))
297  {
298  if (spellInfo->EffectAmplitude[j] != 0)
299  return DotDuration / spellInfo->EffectAmplitude[j];
300  break;
301  }
302  }
303 
304  return 6;
305 }

319 float CalculateDefaultCoefficient(SpellEntry const* spellProto, DamageEffectType const damagetype)
320 {
321  // Damage over Time spells bonus calculation
322  float DotFactor = 1.0f;
323  if (damagetype == DOT)
324  {
325  if (!IsChanneledSpell(spellProto))
326  DotFactor = GetSpellDuration(spellProto) / 15000.0f;
327 
328  if (uint16 DotTicks = GetSpellAuraMaxTicks(spellProto))
329  DotFactor /= DotTicks;
330  }
331 
332  // Distribute Damage over multiple effects, reduce by AoE
333  float coeff = GetSpellCastTimeForBonus(spellProto, damagetype) / 3500.0f;
334 
335  return coeff * DotFactor;

532  (spellInfo->Effect[EFFECT_INDEX_0] == SPELL_EFFECT_LEARN_SPELL || spellInfo->Effect[EFFECT_INDEX_1] == SPELL_EFFECT_LEARN_SPELL || spellInfo->Effect[EFFECT_INDEX_2] == SPELL_EFFECT_LEARN_SPELL))
533  return SPELL_CAST_OK;

536 
537  if (stanceMask & spellInfo->StancesNot[0]) // can explicitly not be casted in this stance

539 
540  if (stanceMask & spellInfo->Stances[0]) // can explicitly be casted in this stance
541  return SPELL_CAST_OK;

544  if (form > 0)
545  {
546  SpellShapeshiftFormEntry const* shapeInfo = sSpellShapeshiftFormStore.LookupEntry(form);
547  if (!shapeInfo)
548  {
549  sLog.outError("GetErrorAtShapeshiftedCast: unknown shapeshift %u", form);
550  return SPELL_CAST_OK;
551  }
552  actAsShifted = !(shapeInfo->flags1 & 1); // shapeshift acts as normal form for spells
553  }
554

556  {
557  if (spellInfo->HasAttribute(SPELL_ATTR_NOT_SHAPESHIFT)) // not while shapeshifted

566  return SPELL_FAILED_ONLY_SHAPESHIFT;
567  }
568 
569  return SPELL_CAST_OK;
