31 {
32  // DEBUG_LOG("WORLD: CMSG_SPLIT_ITEM");
33  uint8 srcbag, srcslot, dstbag, dstslot;
34  uint32 count;
35 
36  recv_data >> srcbag >> srcslot >> dstbag >> dstslot >> count;
37  // DEBUG_LOG("STORAGE: receive srcbag = %u, srcslot = %u, dstbag = %u, dstslot = %u, count = %u", srcbag, srcslot, dstbag, dstslot, count);
38 
39  uint16 src = ((srcbag << 8) | srcslot);
40  uint16 dst = ((dstbag << 8) | dstslot);
41 
42  if (src == dst)
43  return;
44 
45  if (count == 0)
46  return; // check count - if zero it's fake packet
47 
48  if (!_player->IsValidPos(srcbag, srcslot, true))
49  {
50  _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
51  return;
52  }
53 
54  if (!_player->IsValidPos(dstbag, dstslot, false)) // can be autostore pos
55  {
56  _player->SendEquipError(EQUIP_ERR_ITEM_DOESNT_GO_TO_SLOT, nullptr, nullptr);
57  return;
58  }
59 
60  _player->SplitItem(src, dst, count);
61 }

142 void WorldSession::HandleAutoEquipItemOpcode(WorldPacket& recv_data)
143 {
144  // DEBUG_LOG("WORLD: CMSG_AUTOEQUIP_ITEM");
145  uint8 srcbag, srcslot;
146 
147  recv_data >> srcbag >> srcslot;
148  // DEBUG_LOG("STORAGE: receive srcbag = %u, srcslot = %u", srcbag, srcslot);
149 
150  Item* pSrcItem = _player->GetItemByPos(srcbag, srcslot);
151  if (!pSrcItem)
152  return; // only at cheat
153 
154  uint16 dest;
155  InventoryResult msg = _player->CanEquipItem(NULL_SLOT, dest, pSrcItem, !pSrcItem->IsBag());
156  if (msg != EQUIP_ERR_OK)
157  {
158  _player->SendEquipError(msg, pSrcItem, nullptr);
159  return;
160  }
161 
162  uint16 src = pSrcItem->GetPos();
163  if (dest == src) // prevent equip in same slot, only at cheat
164  return;
165 
166  Item* pDstItem = _player->GetItemByPos(dest);
167  if (!pDstItem) // empty slot, simple case
168  {
169  _player->RemoveItem(srcbag, srcslot, true);
170  _player->EquipItem(dest, pSrcItem, true);
171  _player->AutoUnequipOffhandIfNeed(srcbag);
172  }
173  else // have currently equipped item, not simple case
174  {
175  uint8 dstbag = pDstItem->GetBagSlot();
176  uint8 dstslot = pDstItem->GetSlot();
177 
178  msg = _player->CanUnequipItem(dest, !pSrcItem->IsBag());
179  if (msg != EQUIP_ERR_OK)
180  {
181  _player->SendEquipError(msg, pDstItem, nullptr);
182  return;
183  }
184 
185  // check dest->src move possibility
186  ItemPosCountVec sSrc;
187  uint16 eSrc = 0;
188  if (Player::IsInventoryPos(src))
189  {
190  msg = _player->CanStoreItem(srcbag, srcslot, sSrc, pDstItem, true);
191  if (msg != EQUIP_ERR_OK)
192  msg = _player->CanStoreItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
193  if (msg != EQUIP_ERR_OK)
194  msg = _player->CanStoreItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
195  }
196  else if (Player::IsBankPos(src))
197  {
198  msg = _player->CanBankItem(srcbag, srcslot, sSrc, pDstItem, true);
199  if (msg != EQUIP_ERR_OK)
200  msg = _player->CanBankItem(srcbag, NULL_SLOT, sSrc, pDstItem, true);
201  if (msg != EQUIP_ERR_OK)
202  msg = _player->CanBankItem(NULL_BAG, NULL_SLOT, sSrc, pDstItem, true);
203  }
204  else if (Player::IsEquipmentPos(src))
205  {
206  msg = _player->CanEquipItem(srcslot, eSrc, pDstItem, true);
207  if (msg == EQUIP_ERR_OK)
208  msg = _player->CanUnequipItem(eSrc, true);
209  }
210 
211  if (msg != EQUIP_ERR_OK)
212  {
213  _player->SendEquipError(msg, pDstItem, pSrcItem);
214  return;
215  }
216 
217  // now do moves, remove...
218  _player->RemoveItem(dstbag, dstslot, false);
219  _player->RemoveItem(srcbag, srcslot, false);
220 
221  // add to dest
222  _player->EquipItem(dest, pSrcItem, true);
223 
224  // add to src
225  if (Player::IsInventoryPos(src))
226  _player->StoreItem(sSrc, pDstItem, true);
227  else if (Player::IsBankPos(src))
228  _player->BankItem(sSrc, pDstItem, true);
229  else if (Player::IsEquipmentPos(src))
230  _player->EquipItem(eSrc, pDstItem, true);
231 
232  _player->AutoUnequipOffhandIfNeed(srcbag);
233  }
234 }

236 void WorldSession::HandleDestroyItemOpcode(WorldPacket& recv_data)
237 {
238  // DEBUG_LOG("WORLD: CMSG_DESTROYITEM");
239  uint8 bag, slot, count, data1, data2, data3;
240 
241  recv_data >> bag >> slot >> count >> data1 >> data2 >> data3;
242  // DEBUG_LOG("STORAGE: receive bag = %u, slot = %u, count = %u", bag, slot, count);
243 
244  uint16 pos = (bag << 8) | slot;
245 
246  // prevent drop unequipable items (in combat, for example) and non-empty bags
247  if (Player::IsEquipmentPos(pos) || Player::IsBagPos(pos))
248  {
249  InventoryResult msg = _player->CanUnequipItem(pos, false);
250  if (msg != EQUIP_ERR_OK)
251  {
252  _player->SendEquipError(msg, _player->GetItemByPos(pos), nullptr);
253  return;
254  }
255  }
256 
257  Item* pItem = _player->GetItemByPos(bag, slot);
258  if (!pItem)
259  {
260  _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
261  return;
262  }
263 
264  // checked at client side and not have server side appropriate error output
265  if (pItem->GetProto()->Flags & ITEM_FLAG_NO_USER_DESTROY)
266  {
267  _player->SendEquipError(EQUIP_ERR_CANT_DROP_SOULBOUND, nullptr, nullptr);
268  return;
269  }
270 
271  if (count)
272  {
273  uint32 i_count = count;
274  _player->DestroyItemCount(pItem, i_count, true);
275  }
276  else
277  _player->DestroyItem(bag, slot, true);
278 }

481 void WorldSession::HandleSellItemOpcode(WorldPacket& recv_data)
482 {
483  DEBUG_LOG("WORLD: Received opcode CMSG_SELL_ITEM");
484 
485  ObjectGuid vendorGuid;
486  ObjectGuid itemGuid;
487  uint32 count;
488 
489  recv_data >> vendorGuid;
490  recv_data >> itemGuid;
491  recv_data >> count;
492 
493  if (!itemGuid)
494  return;
495 
496  Creature* pCreature = GetPlayer()->GetNPCIfCanInteractWith(vendorGuid, UNIT_NPC_FLAG_VENDOR);
497  if (!pCreature)
498  {
499  DEBUG_LOG("WORLD: HandleSellItemOpcode - %s not found or you can't interact with him.", vendorGuid.GetString().c_str());
500  _player->SendSellError(SELL_ERR_CANT_FIND_VENDOR, nullptr, itemGuid, 0);
501  return;
502  }
503 
504  Item* pItem = _player->GetItemByGuid(itemGuid);
505 
506  if (!pItem)
507  return;
508 
509  // prevent sell not owner item
510  if (_player->GetObjectGuid() != pItem->GetOwnerGuid())
511  {
512  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
513  return;
514  }
515 
516  // prevent sell non empty bag by drag-and-drop at vendor's item list
517  if (pItem->IsBag() && !((Bag*)pItem)->IsEmpty())
518  {
519  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
520  return;
521  }
522 
523  // prevent sell currently looted item
524  if (_player->GetLootGuid() == pItem->GetObjectGuid())
525  {
526  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
527  return;
528  }
529 
530  // special case at auto sell (sell all)
531  if (count == 0)
532  {
533  count = pItem->GetCount();
534  }
535  else
536  {
537  // prevent sell more items that exist in stack (possible only not from client)
538  if (count > pItem->GetCount())
539  {
540  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
541  return;
542  }
543  }
544 
545  ItemPrototype const* pProto = pItem->GetProto();
546 
547  if (!pProto)
548  {
549  _player->SendSellError(SELL_ERR_CANT_FIND_ITEM, pCreature, itemGuid, 0);
550  return;
551  }
552 
553  if (pProto->SellPrice == 0)
554  {
555  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
556  return;
557  }
558 
559  uint32 money = pProto->SellPrice * count;
560 
561  for (auto i = 0; i < MAX_ITEM_PROTO_SPELLS; ++i)
562  {
563  auto const& spell = pProto->Spells[i];
564 
565  // if spell charges for this item are negative, it means that the item should be destroyed once the charges are consumed.
566  // it also means that the value of this item is relative to how many charges are remaining.
567  if (spell.SpellId != 0 && spell.SpellCharges < 0)
568  {
569  auto const multiplier = static_cast<float>(pItem->GetSpellCharges(i)) / spell.SpellCharges;
570  money *= multiplier;
571  break;
572  }
573  }
574 
575  if (count < pItem->GetCount()) // need split items
576  {
577  Item* pNewItem = pItem->CloneItem(count, _player);
578  if (!pNewItem)
579  {
580  sLog.outError("WORLD: HandleSellItemOpcode - could not create clone of item %u; count = %u", pItem->GetEntry(), count);
581  _player->SendSellError(SELL_ERR_CANT_SELL_ITEM, pCreature, itemGuid, 0);
582  return;
583  }
584 
585  _player->InterruptNonMeleeSpells(false);
586  pItem->SetCount(pItem->GetCount() - count);
587  _player->ItemRemovedQuestCheck(pItem->GetEntry(), count);
588  if (_player->IsInWorld())
589  pItem->SendCreateUpdateToPlayer(_player);
590  pItem->SetState(ITEM_CHANGED, _player);
591 
592  _player->AddItemToBuyBackSlot(pNewItem, money);
593  if (_player->IsInWorld())
594  pNewItem->SendCreateUpdateToPlayer(_player);
595  }
596  else
597  {
598  _player->InterruptNonMeleeSpells(false);
599  _player->ItemRemovedQuestCheck(pItem->GetEntry(), pItem->GetCount());
600  _player->RemoveItem(pItem->GetBagSlot(), pItem->GetSlot(), true);
601  pItem->RemoveFromUpdateQueueOf(_player);
602  _player->AddItemToBuyBackSlot(pItem, money);
603  }
604 
605  _player->ModifyMoney(money);
606  _player->GetAchievementMgr().UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_MONEY_FROM_VENDORS, money);
607 }
