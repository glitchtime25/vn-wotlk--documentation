90 void instance_zulgurub::SetData(uint32 type, uint32 data)
91 {
92  switch (type)
93  {
94  case TYPE_JEKLIK:
95  case TYPE_VENOXIS:
96  case TYPE_THEKAL:
97  m_auiEncounter[type] = data;
98  if (data == DONE)
99  RemoveHakkarPowerStack();
100  break;
101  case TYPE_MARLI:
102  m_auiEncounter[type] = data;
103  if (data == DONE)
104  RemoveHakkarPowerStack();
105  if (data == FAIL)
106  {
107  for (GuidList::const_iterator itr = m_lSpiderEggGUIDList.begin(); itr != m_lSpiderEggGUIDList.end(); ++itr)
108  {
109  if (GameObject* pEgg = instance->GetGameObject(*itr))
110  {
111  // Note: this type of Gameobject needs to be respawned manually
112  pEgg->SetRespawnTime(2 * DAY);
113  pEgg->Respawn();
114  }
115  }
116  }
117  break;
118  case TYPE_ARLOKK:
119  m_auiEncounter[type] = data;
120  if (data == IN_PROGRESS)
121  DoUseDoorOrButton(GO_FORCEFIELD);
122  else if (GameObject* pForcefield = GetSingleGameObjectFromStorage(GO_FORCEFIELD))
123  pForcefield->ResetDoorOrButton();
124  if (data == DONE)
125  RemoveHakkarPowerStack();
126  if (data == FAIL)
127  {
128  // Note: this gameobject should change flags - currently it despawns which isn't correct
129  if (GameObject* pGong = GetSingleGameObjectFromStorage(GO_GONG_OF_BETHEKK))
130  {
131  pGong->SetRespawnTime(2 * DAY);
132  pGong->Respawn();
133  }
134  }
135  break;
136  case TYPE_OHGAN:
137  // Note: SPECIAL instance data is set via ACID!
138  if (data == SPECIAL)
139  {
140  if (Creature* pMandokir = GetSingleCreatureFromStorage(NPC_BLOODLORD_MANDOKIR))
141  {
142  pMandokir->SetWalk(false);
143  pMandokir->GetMotionMaster()->MovePoint(1, aMandokirDownstairsPos[0], aMandokirDownstairsPos[1], aMandokirDownstairsPos[2]);
144  }
145  }
146  m_auiEncounter[type] = data;
147  break;
148  case TYPE_LORKHAN:
149  case TYPE_ZATH:
150  m_auiEncounter[type] = data;
151  break;
152  }
153 
154  if (data == DONE)
155  {
156  OUT_SAVE_INST_DATA;
157 
158  std::ostringstream saveStream;
159  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
160  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
161  << m_auiEncounter[6] << " " << m_auiEncounter[7];
162 
163  m_strInstData = saveStream.str();
164 
165  SaveToDB();
166  OUT_SAVE_INST_DATA_COMPLETE;
167  }
168 }

213 Creature* instance_zulgurub::SelectRandomPantherTrigger(bool isLeft)
214 {
215  GuidList* plTempList = isLeft ? &m_lLeftPantherTriggerGUIDList : &m_lRightPantherTriggerGUIDList;
216  std::vector<Creature*> vTriggers;
217  vTriggers.reserve(plTempList->size());
218 
219  for (GuidList::const_iterator itr = plTempList->begin(); itr != plTempList->end(); ++itr)
220  {
221  if (Creature* pTemp = instance->GetCreature(*itr))
222  vTriggers.push_back(pTemp);
223  }
224 
225  if (vTriggers.empty())
226  return nullptr;
227 
228  return vTriggers[urand(0, vTriggers.size() - 1)];
229 }
