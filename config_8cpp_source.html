147 bool AnticheatConfig::CheckResponse(CheatType cheatType, uint32 tickCount, uint32 totalCount, uint32 &actionMask) const
148 {
149  MANGOS_ASSERT(cheatType < CHEATS_COUNT);
150 
151  actionMask = CHEAT_ACTION_NONE;
152 
153  // too many for this tick? append these actions
154  if (tickCount && _responseRules[cheatType].tickCount &&
155  tickCount >= _responseRules[cheatType].tickCount)
156  actionMask |= _responseRules[cheatType].tickAction;
157 
158  // too many total? append these actions
159  if (totalCount && _responseRules[cheatType].totalCount &&
160  totalCount >= _responseRules[cheatType].totalCount)
161  actionMask |= _responseRules[cheatType].totalAction;
162 
163  return actionMask != CHEAT_ACTION_NONE;
164 }
