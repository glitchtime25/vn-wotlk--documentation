153  void ExecuteAction(uint32 action) override
154  {
155  switch (action)
156  {
157  case SHAFFAR_ACTION_BEACON:
158  if (DoCastSpellIfCan(nullptr, SPELL_ETHEREAL_BEACON) == CAST_OK)
159  {
160  if (!urand(0, 3))
161  DoScriptText(SAY_SUMMON, m_creature);
162 
163  ResetCombatAction(action, GetSubsequentActionTimer(ShaffarActions(action)));
164  return;
165  }
166  break;
167  case SHAFFAR_ACTION_FROSTNOVA:
168  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, SPELL_FROSTNOVA, SELECT_FLAG_PLAYER | SELECT_FLAG_USE_EFFECT_RADIUS))
169  {
170  if (DoCastSpellIfCan(nullptr, SPELL_FROSTNOVA) == CAST_OK)
171  {
172  ResetCombatAction(action, GetSubsequentActionTimer(ShaffarActions(action)));
173  return;
174  }
175  }
176  break;
177  case SHAFFAR_ACTION_BLINK:
178  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()) && DoCastSpellIfCan(nullptr, SPELL_BLINK) == CAST_OK)
179  {
180  ResetCombatAction(action, GetSubsequentActionTimer(ShaffarActions(action)));
181  return;
182  }
183  break;
184  case SHAFFAR_ACTION_PRIMARY_SPELL:
185  {
186  uint32 spell = urand(0, 1) ? SPELL_FIREBALL : SPELL_FROSTBOLT;
187  if (DoCastSpellIfCan(m_creature->GetVictim(), spell) == CAST_OK)
188  ResetCombatAction(action, GetSubsequentActionTimer(ShaffarActions(action)));
189  return;
190  }
191  break;
192  }
193  }
194 };
