320 bool Group::AddMember(ObjectGuid guid, const char* name)
321 {
322  if (!_addMember(guid, name))
323  return false;
324 
325  SendUpdate();
326 
327  if (Player* player = sObjectMgr.GetPlayer(guid))
328  {
329  _updateMembersOnRosterChanged(player);
330 
331  if (!IsLeader(player->GetObjectGuid()) && !IsBattleGroup())
332  {
333  // reset the new member's instances, unless he is currently in one of them
334  // including raid/heroic instances that they are not permanently bound to!
335  player->ResetInstances(INSTANCE_RESET_GROUP_JOIN, false);
336  player->ResetInstances(INSTANCE_RESET_GROUP_JOIN, true);
337 
338  if (player->GetLevel() >= LEVELREQUIREMENT_HEROIC)
339  {
340  if (player->GetDungeonDifficulty() != GetDungeonDifficulty())
341  {
342  player->SetDungeonDifficulty(GetDungeonDifficulty());
343  player->SendDungeonDifficulty(true);
344  }
345  if (player->GetRaidDifficulty() != GetRaidDifficulty())
346  player->SendRaidDifficulty(true, GetRaidDifficulty());
347  }
348  }
349  player->SetGroupUpdateFlag(GROUP_UPDATE_FULL);
350  UpdatePlayerOutOfRange(player);
351 
352  // quest related GO state dependent from raid membership
353  if (IsRaidGroup())
354  player->UpdateForQuestWorldObjects();
355 
356  // Broadcast new player group member fields to rest of the group
357  UpdateData groupData;
358 
359  // Broadcast group members' fields to player
360  for (GroupReference* itr = GetFirstMember(); itr != nullptr; itr = itr->next())
361  {
362  if (itr->getSource() == player)
363  continue;
364 
365  if (Player* member = itr->getSource())
366  {
367  if (player->HasAtClient(member))
368  {
369  UpdateMask updateMask;

374  }
375 
376  if (member->HasAtClient(player))
377  {
378  UpdateMask updateMask;
379  updateMask.SetCount(member->GetValuesCount());
380  member->MarkUpdateFieldsWithFlagForUpdate(updateMask, UF_FLAG_GROUP_ONLY);
381  if (updateMask.HasData())
382  {
383  UpdateData newData;
384  player->BuildValuesUpdateBlockForPlayer(newData, updateMask, member);
385 
386  if (newData.HasData())
387  {
388  WorldPacket newDataPacket = newData.BuildPacket(0);
389  member->SendDirectMessage(newDataPacket);
390  }
391  }
392  }
393  }
394  }
395 
396  if (groupData.HasData())
397  {
398  WorldPacket groupDataPacket = groupData.BuildPacket(0);
399  player->SendDirectMessage(groupDataPacket);
400  }
401  }
402 
403  return true;
404 }

421  {

423  {
424  if (groupMember->GetObjectGuid() == guid)
425  continue;
426 
427  groupMember->RemoveAllGroupBuffsFromCaster(guid);
428  if (player)
429  player->RemoveAllGroupBuffsFromCaster(groupMember->GetObjectGuid());
430  }
431  }
432 
433  // remove member and change leader (if need) only if strong more 2 members _before_ member remove
434  if (GetMembersCount() > GetMembersMinCount())
435  {
436  bool leaderChanged = _removeMember(guid);
437 
438  if (player)
439  {
440  // quest related GO state dependent from raid membership
441  if (IsRaidGroup())
442  player->UpdateForQuestWorldObjects();
443 
444  WorldPacket data;
445 
446  if (method == 1)
447  {
448  data.Initialize(SMSG_GROUP_UNINVITE, 0);
449  player->GetSession()->SendPacket(data);
450  }
451 
452  // we already removed player from group and in player->GetGroup() is his original group!
453  if (Group* group = player->GetGroup())
454  {
455  group->SendUpdate();
456  }
457  else
458  {
459  data.Initialize(SMSG_GROUP_LIST, 1 + 1 + 1 + 1 + 8 + 4 + 4 + 8);
460  data << uint8(0x10) << uint8(0) << uint8(0) << uint8(0);
461  data << uint64(0) << uint32(0) << uint32(0) << uint64(0);
462  player->GetSession()->SendPacket(data);
463  }
464 
465  _homebindIfInstance(player);
466  }
467 
468  if (leaderChanged)
469  {
470  WorldPacket data(SMSG_GROUP_SET_LEADER, (m_leaderName.size() + 1));
471  data << m_leaderName;
472  BroadcastPacket(data, true);
473  }
474 
475  SendUpdate();
476  }
477  // if group before remove <= 2 disband it
478  else
479  Disband(true);
480 
481  return m_memberSlots.size();
482 }
