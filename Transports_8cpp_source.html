39 void MapManager::LoadTransports()
40 {
41  sTransportMgr.LoadTransportTemplates();
42 
43  QueryResult* result = WorldDatabase.Query("SELECT entry, name, period FROM transports");
44 
45  uint32 count = 0;
46 
47  if (!result)
48  {
49  BarGoLink bar(1);
50  bar.step();
51  sLog.outString(">> Loaded %u transports", count);
52  sLog.outString();
53  return;
54  }
55 
56  BarGoLink bar(result->GetRowCount());
57 
58  do
59  {
60  bar.step();
61 
62  Field* fields = result->Fetch();
63 
64  uint32 entry = fields[0].GetUInt32();
65  std::string name = fields[1].GetCppString();
66  uint32 period = fields[2].GetUInt32();
67 
68  const GameObjectInfo* goinfo = ObjectMgr::GetGameObjectInfo(entry);
69 
70  if (!goinfo)
71  {
72  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, gameobject_template missing", entry, name.c_str());
73  continue;
74  }
75 
76  if (goinfo->type != GAMEOBJECT_TYPE_MO_TRANSPORT)
77  {
78  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, gameobject_template type wrong", entry, name.c_str());
79  continue;
80  }
81 
82  TransportTemplate* transportTemplate = sTransportMgr.GetTransportTemplate(entry);
83  if (!transportTemplate)
84  {
85  sLog.outErrorDb("Transport ID:%u, Name: %s, will not be loaded, transport template missing (core generated)", entry, name.c_str());
86  continue;
87  }
88 
89  const MapEntry* pMapInfo = sMapStore.LookupEntry(transportTemplate->keyFrames.begin()->Node->mapid);
90  if (!pMapInfo)
91  continue;
92 
93  transportTemplate->pathTime = period;
94  transportTemplate->keyFrames.back().DepartureTime = period;
95 
96  m_transportsByMap[pMapInfo->MapID].push_back(transportTemplate);
97  m_transportsByEntry[entry].push_back(transportTemplate);
98 
99  ++count;
100  }
101  while (result->NextRow());
102  delete result;
103 
104  // check transport data DB integrity
105  result = WorldDatabase.Query("SELECT gameobject.guid,gameobject.id,transports.name FROM gameobject,transports WHERE gameobject.id = transports.entry");
106  if (result) // wrong data found
107  {
108  do
109  {
110  Field* fields = result->Fetch();
111 
112  uint32 guid = fields[0].GetUInt32();
113  uint32 entry = fields[1].GetUInt32();
114  std::string name = fields[2].GetCppString();
115  sLog.outErrorDb("Transport %u '%s' have record (GUID: %u) in `gameobject`. Transports MUST NOT have any records in `gameobject` or its behavior will be unpredictable/bugged.", entry, name.c_str(), guid);
116  }
117  while (result->NextRow());
118 
119  delete result;
120  }
121 
122  sLog.outString(">> Loaded %u transports", count);
123  sLog.outString();
124 }

282 {

285 
286  bool mapChange = GetMapId() != newMapid;
287 
288  auto& passengers = GetPassengers();
289  for (m_passengerTeleportIterator = passengers.begin(); m_passengerTeleportIterator != passengers.end();)

292 
293  Position pos = obj->m_movementInfo.GetTransportPos();
294 
295  if (obj->IsUnit())
296  {
297  Unit* unit = static_cast<Unit*>(obj);
298  if (unit->GetVehicleInfo())
299  {
300  unit->GetVehicleInfo()->TeleportPassengers(newMapid);
301  }
302  }
303 
304  switch (obj->GetTypeId())
305  {
306  case TYPEID_UNIT:
307  {
308  if (mapChange)
309  {
310  RemovePassenger(obj);
311  if (obj->IsCreature() && !static_cast<Creature*>(obj)->IsPet())
312  obj->AddObjectToRemoveList();
313  }
314  break;
315  }
316  case TYPEID_PLAYER:
317  {
318  Player* player = static_cast<Player*>(obj);
319  if (player->IsDead() && !player->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_GHOST))
320  player->ResurrectPlayer(1.0);
321  player->TeleportTo(newMapid, pos.x, pos.y, pos.z, pos.o, TELE_TO_NOT_LEAVE_TRANSPORT, nullptr, this);
322  break;
323  }
324  case TYPEID_GAMEOBJECT:
325  case TYPEID_DYNAMICOBJECT:
326  {
327  if (mapChange)
328  {
329  RemovePassenger(obj);
330  obj->AddObjectToRemoveList();
331  }
332  break;
333  }
334  }
335  }
336 
337  // we need to create and save new Map object with 'newMapid' because if not done -> lead to invalid Map object reference...
338  // player far teleport would try to create same instance, but we need it NOW for transport...
339  // correct me if I'm wrong O.o
340  if (mapChange)
341  {
342  RemoveModelFromMap();
343  oldMap->RemoveTransport(this);
344  UpdateForMap(oldMap, false);
345  Object::RemoveFromWorld();
346  ResetMap();
347 
348  Map* newMap = sMapMgr.CreateMap(newMapid, this);
349  newMap->GetMessager().AddMessage([transport = this](Map* map)
350  {
351  transport->SetMap(map);
352  transport->Object::AddToWorld();
353  map->AddTransport(transport);
354  transport->AddModelToMap();
355  transport->SpawnPassengers();
356  transport->UpdateForMap(map, true);
357  });
358  }
359  else
360  {
361  UpdateModelPosition();
362  UpdatePassengerPositions(m_passengers);
363  }
364 }

459 void Transport::Update(const uint32 diff)
460 {
461  uint32 const positionUpdateDelay = 50;
462  uint32 const dynChangeTimer = 1000;
463 
464  if (GetKeyFrames().size() <= 1)
465  return;
466 
467  if (AI())
468  AI()->UpdateAI(diff);
469 
470  uint32 transportDiff = diff;
471  if (!GetGOInfo()->moTransport.canBeStopped)
472  {
473  uint32 currentMsTime = GetMap()->GetCurrentMSTime() - m_movementStarted;
474  if (m_pathProgress >= currentMsTime) // map transition and update happened in same tick due to MT
475  return;
476  transportDiff = currentMsTime - m_pathProgress; // override diff for nonstoppable GOs
477  m_pathProgress = currentMsTime;
478  }
479  else
480  {
481  if (m_stopped)
482  m_pathProgress = std::min(m_pathProgress + diff, (m_pathProgress / GetPeriod()) * GetPeriod() + m_currentFrame->DepartureTime);
483  else
484  m_pathProgress += diff;
485  }
486 
487  uint32 pathProgress = m_pathProgress % GetPeriod();
488  while (true)
489  {
490  if (m_currentFrame->IsStopFrame() && GetGOInfo()->moTransport.canBeStopped && m_lastStopIndex != m_currentFrame->Index)
491  {
492  m_lastStopIndex = m_currentFrame->Index;
493  transportDiff = 0;
494  m_stopped = true;
495  if (pathProgress > m_currentFrame->DepartureTime)
496  m_pathProgress = (m_pathProgress / GetPeriod()) * GetPeriod() + m_currentFrame->DepartureTime;
497  SetUInt16Value(GAMEOBJECT_DYNAMIC, 0, GO_DYNFLAG_LO_STOPPED);
498  SetUInt16Value(GAMEOBJECT_DYNAMIC, 1, pathProgress % GetPeriod());
499  m_dynamicChangeTimer.Reset(dynChangeTimer);
500  GameObject::SetGoState(GO_STATE_READY);
501  if (AI())
502  AI()->JustReachedStopPoint();
503  }
504 
505  if (pathProgress >= m_currentFrame->ArriveTime && pathProgress < m_currentFrame->DepartureTime)
506  {
507  SetMoving(false);
508  break; // its a stop frame and we are waiting
509  }
510 
511  if (m_stopped)
512  return;
513 
514  // not waiting anymore
515  SetMoving(true);
516 
517  if (pathProgress >= m_currentFrame->DepartureTime && pathProgress < m_currentFrame->NextArriveTime)
518  break; // found current waypoint
519 
520  DoEventIfAny(*m_currentFrame->Node, true);
521 
522  MoveToNextWayPoint();
523 
524  DoEventIfAny(*m_currentFrame->Node, false);
525 
526  // first check help in case client-server transport coordinates de-synchronization
527  if (m_currentFrame->Node->mapid != GetMapId() || m_currentFrame->IsTeleportFrame())
528  {
529  TeleportTransport(m_nextFrame->Node->mapid, m_nextFrame->Node->x, m_nextFrame->Node->y, m_nextFrame->Node->z, m_nextFrame->InitialOrientation);
530  return;
531  }
532 
533  if (m_currentFrame == GetKeyFrames().begin())
534  DETAIL_FILTER_LOG(LOG_FILTER_TRANSPORT_MOVES, " ************ BEGIN ************** %s", GetName());
535 
536  DETAIL_FILTER_LOG(LOG_FILTER_TRANSPORT_MOVES, "%s moved to %f %f %f %d", GetName(), m_currentFrame->Node->x, m_currentFrame->Node->y, m_currentFrame->Node->z, m_currentFrame->Node->mapid);
537  }
538 
539  // Set position
540  m_positionChangeTimer.Update(transportDiff);
541  if (m_positionChangeTimer.Passed())
542  {
543  m_positionChangeTimer.Reset(positionUpdateDelay);
544  if (IsMoving() && pathProgress)
545  {
546  float t = CalculateSegmentPos(float(pathProgress) * 0.001f);
547  G3D::Vector3 pos, dir;
548  m_currentFrame->Spline->evaluate_percent(m_currentFrame->Index, t, pos);
549  m_currentFrame->Spline->evaluate_derivative(m_currentFrame->Index, t, dir);
550  UpdatePosition(pos.x, pos.y, pos.z, atan2(dir.y, dir.x) + M_PI);
551  }
552  if (IsInWorld())
553  {
554  bool gridActive = GetMap()->IsLoaded(GetPositionX(), GetPositionY());
555  if (!gridActive && !m_staticPassengers.empty())
556  DespawnPassengers();
557  else if (gridActive && m_staticPassengers.empty())
558  SpawnPassengers();
559  }
560  }
561 
562  m_dynamicChangeTimer.Update(transportDiff);
563  if ((GetUInt16Value(GAMEOBJECT_DYNAMIC, 0) & GO_DYNFLAG_LO_STOPPED) == 0 && m_dynamicChangeTimer.Passed())
564  {
565  m_dynamicChangeTimer.Reset(dynChangeTimer);
566  SetUInt16Value(GAMEOBJECT_DYNAMIC, 1, pathProgress % GetPeriod());
567  }
568 }

570 float Transport::CalculateSegmentPos(float now)
571 {
572  KeyFrame const& frame = *m_currentFrame;
573  float const speed = float(m_goInfo->moTransport.moveSpeed);
574  float const accel = float(m_goInfo->moTransport.accelRate);
575  float timeSinceStop = frame.TimeFrom + (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
576  float timeUntilStop = frame.TimeTo - (now - (1.0f / IN_MILLISECONDS) * frame.DepartureTime);
577  float segmentPos, dist;
578  float accelTime = m_transportTemplate.accelTime;
579  float accelDist = m_transportTemplate.accelDist;
580  // calculate from nearest stop, less confusing calculation...
581  if (timeSinceStop < timeUntilStop)
582  {
583  if (timeSinceStop < accelTime)
584  dist = 0.5f * accel * timeSinceStop * timeSinceStop;
585  else
586  dist = accelDist + (timeSinceStop - accelTime) * speed;
587  segmentPos = dist - frame.DistSinceStop;
588  }
589  else
590  {
591  if (timeUntilStop < m_transportTemplate.accelTime)
592  dist = 0.5f * accel * timeUntilStop * timeUntilStop;
593  else
594  dist = accelDist + (timeUntilStop - accelTime) * speed;
595  segmentPos = frame.DistUntilStop - dist;
596  }
597 
598  return segmentPos / frame.NextDistFromPrev;
599 }

617 void ElevatorTransport::Update(const uint32 diff)
618 {
619  if (!m_animationInfo)
620  return;
621 
622  if (!m_stopped)
623  {
624  uint32 timeSinceLastStop = (GetMap()->GetCurrentMSTime() - m_movementStarted) % m_animationInfo->TotalTime;
625  if (!GetGOInfo()->transport.pause)
626  m_pathProgress = timeSinceLastStop;
627  else
628  {
629  if (timeSinceLastStop >= GetGOInfo()->transport.pause) // stopped and needs to be reenabled by script
630  {
631  m_stopped = true;
632  timeSinceLastStop = GetGOInfo()->transport.pause;
633 
634  if (AI())
635  AI()->JustReachedStopPoint();
636  }
637  m_pathProgress = ((m_pathProgress / GetGOInfo()->transport.pause) * GetGOInfo()->transport.pause + timeSinceLastStop) % m_animationInfo->TotalTime;
638  }
639  TransportAnimationEntry const* nodeNext = m_animationInfo->GetNextAnimNode(m_pathProgress);
640  TransportAnimationEntry const* nodePrev = m_animationInfo->GetPrevAnimNode(m_pathProgress);
641  if (nodeNext && nodePrev)

750 void GenericTransport::UpdatePassengerPosition(WorldObject* passenger)
751 {
752  // transport teleported but passenger not yet (can happen for players)
753  if (passenger->IsInWorld() && passenger->GetMap() != GetMap())
754  return;
755 
756  // Do not use Unit::UpdatePosition here, we don't want to remove auras
757  // as if regular movement occurred
758  float x, y, z, o;
759  x = passenger->GetTransOffsetX();
760  y = passenger->GetTransOffsetY();
761  z = passenger->GetTransOffsetZ();
762  o = passenger->GetTransOffsetO();
763  CalculatePassengerPosition(x, y, z, &o);
764  if (!MaNGOS::IsValidMapCoord(x, y, z))
765  {
766  sLog.outError("[TRANSPORTS] Object %s [guid %u] has invalid position on transport.", passenger->GetName(), passenger->GetGUIDLow());
767  return;
768  }
769  switch (passenger->GetTypeId())
770  {
771  case TYPEID_UNIT:
772  {
773  Creature* creature = dynamic_cast<Creature*>(passenger);
774  if (passenger->IsInWorld())
775  GetMap()->CreatureRelocation(creature, x, y, z, o);
776  else
777  passenger->Relocate(x, y, z, o);
778  creature->m_movementInfo.t_time = GetPathProgress();
779  break;
780  }
781  case TYPEID_PLAYER:
782  //relocate only passengers in world and skip any player that might be still logging in/teleporting
783  if (passenger->IsInWorld())
784  GetMap()->PlayerRelocation(dynamic_cast<Player*>(passenger), x, y, z, o);
785  else
786  {
787  passenger->Relocate(x, y, z, o);
788  dynamic_cast<Player*>(passenger)->m_movementInfo.t_guid = GetObjectGuid();
789  }
790  dynamic_cast<Player*>(passenger)->m_movementInfo.t_time = GetPathProgress();
791  break;
792  case TYPEID_GAMEOBJECT:
793  GetMap()->GameObjectRelocation(static_cast<GameObject*>(passenger), x, y, z, o, false);
794  break;
795  case TYPEID_DYNAMICOBJECT:
796  GetMap()->DynamicObjectRelocation(static_cast<DynamicObject*>(passenger), x, y, z, o);
797  break;
798  default:
799  break;
800  }

809 {

811  if (o)
812  *o = MapManager::NormalizeOrientation(transO + *o);
813 
814  x = transX + inx * std::cos(transO) - iny * std::sin(transO);

819 void GenericTransport::CalculatePassengerOffset(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
820 {
821  if (o)
822  *o = MapManager::NormalizeOrientation(*o - transO);
823 
824  z -= transZ;
825  y -= transY; // y = searchedY * std::cos(o) + searchedX * std::sin(o)
826  x -= transX; // x = searchedX * std::cos(o) + searchedY * std::sin(o + pi)
827  float inx = x, iny = y;
828  y = (iny - inx * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
829  x = (inx + iny * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
830 }
