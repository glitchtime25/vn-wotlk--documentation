247  void CheckGroupStatus()
248  {
249  ResetTimer(VOIDWALKER_CHECK_GROUP, 1 * IN_MILLISECONDS);
250 
251  // Do not care about group while fighting
252  if (m_creature->IsInCombat())
253  return;
254 
255  // If a group leader already exists and is alive, do nothing
256  if (m_leaderGuid && !IsLeader())
257  {
258  if (Creature* leader = m_creature->GetMap()->GetCreature(m_leaderGuid))
259  {
260  // Leader is alive and we are following it, do nothing
261  if (leader->IsAlive() && m_creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == FOLLOW_MOTION_TYPE)
262  return;
263  }
264  }
265 
266  // If we are leader and already following path, do nothing
267  if (IsLeader())
268  {
269  if (m_creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
270  return;
271  }
272 
273  // For whatever other reason, re-initialise the group
274  InitVoidWalkersGroup();

277  void InitVoidWalkersGroup()
278  {

282  CreatureList walkersList;
283  GetCreatureListWithEntryInGrid(walkersList, m_creature, m_creature->GetEntry(), 50.0f);
284 
285  uint8 position = 0;
