334 void SpellTargetMgr::Initialize()
335 {
336  for (uint32 i = 0; i <= sSpellTemplate.GetMaxEntry(); ++i)
337  {
338  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(i);
339  if (!spellInfo)
340  continue;
341 
342  SpellTargetingData& data = spellTargetingData[i];
343  // figure out what targeting dynamic effects should use
344  for (uint32 effIdx = 0; effIdx < MAX_EFFECT_INDEX; ++effIdx)
345  {
346  if (!spellInfo->Effect[effIdx])
347  continue;
348 
349  uint32 effect = spellInfo->Effect[effIdx];
350  uint32 targetA = spellInfo->EffectImplicitTargetA[effIdx];
351  uint32 targetB = spellInfo->EffectImplicitTargetB[effIdx];
352  if (SpellEffectInfoTable[effect].requiredTarget != TARGET_TYPE_DYNAMIC && SpellEffectInfoTable[effect].requiredTarget != TARGET_TYPE_DYNAMIC_ANY)
353  data.implicitType[effIdx] = SpellEffectInfoTable[effect].requiredTarget;
354  else
355  {
356  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_LOCK || SpellTargetInfoTable[targetB].type == TARGET_TYPE_LOCK)
357  {
358  data.implicitType[effIdx] = TARGET_TYPE_LOCK;
359  continue;
360  }
361  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_CORPSE || SpellTargetInfoTable[targetB].type == TARGET_TYPE_CORPSE)
362  {
363  data.implicitType[effIdx] = TARGET_TYPE_CORPSE;
364  continue;
365  }
366  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_GAMEOBJECT || SpellTargetInfoTable[targetB].type == TARGET_TYPE_GAMEOBJECT)
367  {
368  data.implicitType[effIdx] = TARGET_TYPE_GAMEOBJECT;
369  continue;
370  }
371  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_LOCATION_TRAJ || SpellTargetInfoTable[targetB].type == TARGET_TYPE_LOCATION_TRAJ)
372  {
373  data.implicitType[effIdx] = TARGET_TYPE_LOCATION_TRAJ;
374  continue;
375  }
376  if ((SpellTargetInfoTable[targetA].type == TARGET_TYPE_LOCATION_DEST && SpellTargetInfoTable[targetB].type == TARGET_TYPE_UNIT)
377  || (SpellTargetInfoTable[targetA].type == TARGET_TYPE_UNIT && SpellTargetInfoTable[targetB].type == TARGET_TYPE_LOCATION_DEST))
378  {
379  data.implicitType[effIdx] = TARGET_TYPE_UNIT_DEST;
380  continue;
381  }
382  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_UNIT || SpellTargetInfoTable[targetB].type == TARGET_TYPE_UNIT)
383  {
384  data.implicitType[effIdx] = TARGET_TYPE_UNIT;
385  continue;
386  }
387  if (SpellTargetInfoTable[targetA].type == TARGET_TYPE_LOCATION_DEST || SpellTargetInfoTable[targetB].type == TARGET_TYPE_LOCATION_DEST ||
388  SpellEffectInfoTable[effect].requiredTarget == TARGET_TYPE_DYNAMIC_ANY) // some effects cant be none
389  {
390  data.implicitType[effIdx] = TARGET_TYPE_LOCATION_DEST;
391  continue;
392  }
393  if ((SpellTargetInfoTable[targetA].type == TARGET_TYPE_LOCATION_SRC || targetA == TARGET_NONE) && targetB == TARGET_NONE)
394  {
395  data.implicitType[effIdx] = TARGET_TYPE_NONE;
396  continue;
397  }
398  sLog.outError("Spell %u effect index %u failed to pick type for dynamic effect targeting type.", i, effIdx);
399  }
400  }
401  // evaluate which targets should be evaluated on execution
402  for (uint32 effIdxSource = 0; effIdxSource < MAX_EFFECT_INDEX; ++effIdxSource)
403  {
404  if (!spellInfo->Effect[effIdxSource])
405  continue;
406 
407  SpellEffectInfo& effectTargetingSource = SpellEffectInfoTable[spellInfo->Effect[effIdxSource]];
408 
409  SpellTargetImplicitType implicitEffectType = data.implicitType[effIdxSource];
410  for (uint8 rightSource = 0; rightSource < 2; ++rightSource)
411  {
412  uint32 targetSource;
413  if (rightSource == 0)
414  {
415  if (data.ignoredTargets[effIdxSource].first) // already evaluated as ignored
416  continue;
417  targetSource = spellInfo->EffectImplicitTargetA[effIdxSource];
418  }
419  else
420  {
421  if (data.ignoredTargets[effIdxSource].second) // already evaluated as ignored
422  continue;
423  targetSource = spellInfo->EffectImplicitTargetB[effIdxSource];
424  }
425  if (!targetSource) // no target found
426  continue;
427  // start from first next target
428  for (uint32 effIdxTarget = effIdxSource; effIdxTarget < MAX_EFFECT_INDEX; ++effIdxTarget)
429  {
430  if (!spellInfo->Effect[effIdxTarget])
431  continue;
432 
433  SpellEffectInfo& effectTargetingTarget = SpellEffectInfoTable[spellInfo->Effect[effIdxTarget]];
434 
435  SpellTargetImplicitType implicitEffectTypeTarget = data.implicitType[effIdxTarget];
436  for (uint8 rightTarget = effIdxSource == effIdxTarget ? rightSource + 1 : 0; rightTarget < 2; ++rightTarget)
437  {
438  uint32 targetTarget;
439  if (rightTarget == 0)
440  {
441  if (data.ignoredTargets[effIdxTarget].first) // already evaluated as ignored
442  continue;
443  targetTarget = spellInfo->EffectImplicitTargetA[effIdxTarget];
444  }
445  else
446  {
447  if (data.ignoredTargets[effIdxTarget].second) // already evaluated as ignored
448  continue;
449  targetTarget = spellInfo->EffectImplicitTargetB[effIdxTarget];
450  }
451  if (!targetTarget) // no target found
452  continue;
453  if (targetSource == targetTarget) // if both target types are the same check if subsequent can be skipped
454  {
455  SpellTargetInfo& info = SpellTargetInfoTable[targetTarget];
456  bool ignore = false;
457  // exception for area auras
458  if (implicitEffectType == TARGET_TYPE_SPECIAL_UNIT && info.type == TARGET_TYPE_UNIT && info.enumerator != TARGET_ENUMERATOR_SINGLE)
459  ignore = true;
460  else if (implicitEffectTypeTarget == TARGET_TYPE_LOCATION_DEST && info.type == TARGET_TYPE_UNIT)
461  ignore = true;
462  else
463  {
464  switch (info.type)
465  {
466  case TARGET_TYPE_LOCATION_SRC:
467  case TARGET_TYPE_LOCATION_DEST:
468  case TARGET_TYPE_LOCATION_TRAJ:
469  case TARGET_TYPE_LOCK:
470  case TARGET_TYPE_CORPSE: ignore = true; break; // always ignore subsequent
471  case TARGET_TYPE_GAMEOBJECT:
472  case TARGET_TYPE_UNIT:
473  case TARGET_TYPE_PLAYER:
474  {
475  // always ignore subsequent
476  if (info.enumerator == TARGET_ENUMERATOR_SINGLE && CanEffectConsumeTarget(info.type, effectTargetingSource.requiredTarget))
477  {
478  ignore = true;
479  break;
480  }
481  // aoe and chain case
482  if (info.enumerator == TARGET_ENUMERATOR_CHAIN) // chain checks chaining
483  // if chain targets are not the same, cant ignore
484  if (spellInfo->EffectChainTarget[effIdxSource] != spellInfo->EffectChainTarget[effIdxTarget])
485  continue;
486  // different radiuses, cant ignore
487  if (spellInfo->EffectRadiusIndex[effIdxSource] != spellInfo->EffectRadiusIndex[effIdxTarget])
488  continue;
489  if (info.filter == TARGET_SCRIPT && effIdxSource != effIdxTarget)
490  {
491  SQLMultiStorage::SQLMSIteratorBounds<SpellTargetEntry> bounds = sSpellScriptTargetStorage.getBounds<SpellTargetEntry>(i);
492  if (!CheckBoundsEqualForEffects(bounds, SpellEffectIndex(effIdxSource), SpellEffectIndex(effIdxTarget)))
493  continue;
494  }
495  ignore = true;
496  break;
497  }
498  default: break;
499  }
500  if (ignore)
501  data.targetMask[effIdxSource][rightSource] |= (1 << effIdxTarget);
502  }
503  if (ignore)
504  {
505  auto& ignored = data.ignoredTargets[effIdxTarget];
506  if (rightTarget == 0)
507  ignored.first = true;
508  else
509  ignored.second = true;
510  }
511  }
512  }
513  }
514  }
515  }
516  for (uint32 effIdx = 0; effIdx < MAX_EFFECT_INDEX; ++effIdx)
517  {
518  if (!spellInfo->Effect[effIdx])
519  continue;
520  SpellTargetImplicitType implicitEffectType = data.implicitType[effIdx];
521  for (uint8 right = 0; right < 2; ++right)
522  {
523  uint32 target;
524  if (right == 0)
525  target = spellInfo->EffectImplicitTargetA[effIdx];
526  else
527  target = spellInfo->EffectImplicitTargetB[effIdx];
528 
529  if (target)
530  {
531  SpellTargetFilterScheme scheme = SCHEME_RANDOM;
532  if (SpellTargetInfoTable[target].enumerator == TARGET_ENUMERATOR_CHAIN)
533  {
534  switch (target)
535  {
536  case TARGET_UNIT_FRIEND_CHAIN_HEAL: scheme = SCHEME_LOWEST_HP_CHAIN; break;
537  case TARGET_UNIT_ENEMY_NEAR_CASTER:
538  case TARGET_UNIT_FRIEND_NEAR_CASTER:
539  case TARGET_UNIT_NEAR_CASTER:
540  case TARGET_UNIT_ENEMY: scheme = SCHEME_CLOSEST_CHAIN; break;
541  case TARGET_UNIT: scheme = SCHEME_RANDOM_CHAIN; break;
542  default: break;
543  }
544  }
545  scheme = GetSpellTargetingFilterScheme(scheme, spellInfo->Id);
546 
547  if (scheme != SCHEME_RANDOM)
548  data.filteringScheme[effIdx][right] = scheme;
549  }
550  }
551  }
552  }
553 }

555 bool SpellTargetMgr::CanEffectBeFilledWithMask(uint32 spellId, uint32 effIdx, uint32 mask)
556 {
557  SpellTargetingData& data = GetSpellTargetingData(spellId);
558  switch (data.implicitType[effIdx])
559  {
560  case TARGET_TYPE_SPECIAL_DEST:
561  case TARGET_TYPE_LOCATION_DEST: return bool(mask & TARGET_FLAG_DEST_LOCATION);
562  case TARGET_TYPE_LOCATION_TRAJ: return bool(mask & (TARGET_FLAG_SOURCE_LOCATION | TARGET_FLAG_DEST_LOCATION));
563  case TARGET_TYPE_GAMEOBJECT: return bool(mask & (TARGET_FLAG_GAMEOBJECT | TARGET_FLAG_LOCKED));
564  case TARGET_TYPE_UNIT_DEST: return bool(mask & (TARGET_FLAG_UNIT_ALLY | TARGET_FLAG_UNIT | TARGET_FLAG_UNIT_ENEMY | TARGET_FLAG_UNIT_DEAD | TARGET_FLAG_DEST_LOCATION));
565  case TARGET_TYPE_PLAYER:
566  case TARGET_TYPE_UNIT: return bool(mask & (TARGET_FLAG_UNIT_ALLY | TARGET_FLAG_UNIT | TARGET_FLAG_UNIT_ENEMY | TARGET_FLAG_UNIT_DEAD));
567  case TARGET_TYPE_CORPSE: return bool(mask & (TARGET_FLAG_CORPSE_ENEMY | TARGET_FLAG_CORPSE_ALLY));
568  case TARGET_TYPE_LOCK: return bool(mask & (TARGET_FLAG_GAMEOBJECT | TARGET_FLAG_LOCKED | TARGET_FLAG_ITEM));
569  case TARGET_TYPE_ITEM: return bool(mask & (TARGET_FLAG_LOCKED | TARGET_FLAG_ITEM));
570  default: return false;
571  }
