100 SpellEntry const* ScriptedAI::SelectSpell(Unit* target, int32 school, int32 mechanic, SelectTarget selectTargets, uint32 powerCostMin, uint32 powerCostMax, float rangeMin, float rangeMax, SelectEffect selectEffects)
101 {
102  // No target so we can't cast
103  if (!target)
104  return nullptr;
105 
106  // Silenced so we can't cast
107  if (m_creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SILENCED))
108  return nullptr;
109 
110  // Using the extended script system we first create a list of viable spells
111  SpellEntry const* spellInfos[4];
112  memset(spellInfos, 0, sizeof(SpellEntry*) * 4);
113 
114  uint32 spellCount = 0;
115 
116  // Check if each spell is viable(set it to null if not)
117  std::vector<uint32> spells = m_unit->GetCharmSpells();
118  for (uint8 i = 0; i < 4; ++i)
119  {
120  SpellEntry const* tempSpellInfo = GetSpellStore()->LookupEntry<SpellEntry>(spells[i]);
121 
122  // This spell doesn't exist
123  if (!tempSpellInfo)
124  continue;
125 
126  // Targets and Effects checked first as most used restrictions
127  // Check the spell targets if specified
128  if (selectTargets && !(SpellSummary[spells[i]].Targets & (1 << (selectTargets - 1))))
129  continue;
130 
131  // Check the type of spell if we are looking for a specific spell type
132  if (selectEffects && !(SpellSummary[spells[i]].Effects & (1 << (selectEffects - 1))))
133  continue;
134 
135  // Check for school if specified
136  if (school >= 0 && tempSpellInfo->SchoolMask & school)
137  continue;
138 
139  // Check for spell mechanic if specified
140  if (mechanic >= 0 && tempSpellInfo->Mechanic != (uint32)mechanic)
141  continue;
142 
143  // Make sure that the spell uses the requested amount of power
144  if (powerCostMin && tempSpellInfo->manaCost < powerCostMin)
145  continue;
146 
147  if (powerCostMax && tempSpellInfo->manaCost > powerCostMax)
148  continue;
149 
150  // Continue if we don't have the mana to actually cast this spell
151  if (tempSpellInfo->manaCost > m_creature->GetPower((Powers)tempSpellInfo->powerType))
152  continue;
153 
154  // Get the Range
155  SpellRangeEntry const* tempRange = GetSpellRangeStore()->LookupEntry(tempSpellInfo->rangeIndex);
156 
157  // Spell has invalid range store so we can't use it
158  if (!tempRange)
159  continue;
160 
161  // Check if the spell meets our range requirements
162  if (rangeMin && tempRange->maxRange < rangeMin)
163  continue;
164 
165  if (rangeMax && tempRange->maxRange > rangeMax)
166  continue;
167 
168  // Check if our target is in range
169  if (m_creature->IsWithinDistInMap(target, tempRange->minRange) || !m_creature->IsWithinDistInMap(target, tempRange->maxRange))
170  continue;
171 
172  // All good so lets add it to the spell list
173  spellInfos[spellCount] = tempSpellInfo;
174  ++spellCount;
175  }
176 
177  // We got our usable spells so now lets randomly pick one
178  if (!spellCount)
179  return nullptr;
180 
181  return spellInfos[urand(0, spellCount - 1)];
182 }

184 bool ScriptedAI::CanCast(Unit* target, SpellEntry const* spellInfo, bool triggered)
185 {
186  // No target so we can't cast
187  if (!target || !spellInfo)
188  return false;
189 
190  // Silenced so we can't cast
191  if (!triggered && m_creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SILENCED))
192  return false;
193 
194  // Check for power
195  if (!triggered && m_creature->GetPower((Powers)spellInfo->powerType) < spellInfo->manaCost)
196  return false;
197 
198  SpellRangeEntry const* tempRange = GetSpellRangeStore()->LookupEntry(spellInfo->rangeIndex);
199 
200  // Spell has invalid range store so we can't use it
201  if (!tempRange)
202  return false;
203 
204  // Unit is out of range of this spell
205  if (!m_creature->IsInRange(target, tempRange->minRange, tempRange->maxRange))
206  return false;
207 
208  return true;
209 }

211 void FillSpellSummary()
212 {
213  SpellSummary = new TSpellSummary[GetSpellStore()->GetMaxEntry()];
214 
215  for (uint32 i = 0; i < GetSpellStore()->GetMaxEntry(); ++i)
216  {
217  SpellSummary[i].Effects = 0;
218  SpellSummary[i].Targets = 0;
219 
220  SpellEntry const* tempSpell = GetSpellStore()->LookupEntry<SpellEntry>(i);
221  // This spell doesn't exist
222  if (!tempSpell)
223  continue;
224 
225  for (uint8 j = 0; j < 3; ++j)
226  {
227  // Spell targets self
228  if (tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_CASTER)
229  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_SELF - 1);
230 
231  // Spell targets a single enemy
232  if (tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_ENEMY ||
233  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_TARGET_POSITION)
234  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_SINGLE_ENEMY - 1);
235 
236  // Spell targets AoE at enemy
237  if (tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_SRC_LOC ||
238  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DEST_LOC ||
239  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_SRC ||
240  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DYNOBJ_LOC)
241  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_AOE_ENEMY - 1);
242 
243  // Spell targets an enemy
244  if (tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_ENEMY ||
245  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_TARGET_POSITION ||
246  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_SRC_LOC ||
247  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DEST_LOC ||
248  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_SRC ||
249  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_ENEMY_AOE_AT_DYNOBJ_LOC)
250  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_ANY_ENEMY - 1);
251 
252  // Spell targets a single friend(or self)
253  if (tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_CASTER ||
254  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_FRIEND ||
255  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_PARTY)
256  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_SINGLE_FRIEND - 1);
257 
258  // Spell targets aoe friends
259  if (tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_PARTY_WITHIN_CASTER_RANGE ||
260  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_FRIEND_AND_PARTY ||
261  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_SRC)
262  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_AOE_FRIEND - 1);
263 
264  // Spell targets any friend(or self)
265  if (tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_CASTER ||
266  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_FRIEND ||
267  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_PARTY ||
268  tempSpell->EffectImplicitTargetA[j] == TARGET_ENUM_UNITS_PARTY_WITHIN_CASTER_RANGE ||
269  tempSpell->EffectImplicitTargetA[j] == TARGET_UNIT_FRIEND_AND_PARTY ||
270  tempSpell->EffectImplicitTargetA[j] == TARGET_LOCATION_CASTER_SRC)
271  SpellSummary[i].Targets |= 1 << (SELECT_TARGET_ANY_FRIEND - 1);
272 
273  // Make sure that this spell includes a damage effect
274  if (tempSpell->Effect[j] == SPELL_EFFECT_SCHOOL_DAMAGE ||
275  tempSpell->Effect[j] == SPELL_EFFECT_INSTAKILL ||
276  tempSpell->Effect[j] == SPELL_EFFECT_ENVIRONMENTAL_DAMAGE ||
277  tempSpell->Effect[j] == SPELL_EFFECT_HEALTH_LEECH)
278  SpellSummary[i].Effects |= 1 << (SELECT_EFFECT_DAMAGE - 1);
279 
280  // Make sure that this spell includes a healing effect (or an apply aura with a periodic heal)
281  if (tempSpell->Effect[j] == SPELL_EFFECT_HEAL ||
282  tempSpell->Effect[j] == SPELL_EFFECT_HEAL_MAX_HEALTH ||
283  tempSpell->Effect[j] == SPELL_EFFECT_HEAL_MECHANICAL ||
284  (tempSpell->Effect[j] == SPELL_EFFECT_APPLY_AURA && tempSpell->EffectApplyAuraName[j] == 8))
285  SpellSummary[i].Effects |= 1 << (SELECT_EFFECT_HEALING - 1);
286 
287  // Make sure that this spell applies an aura
288  if (tempSpell->Effect[j] == SPELL_EFFECT_APPLY_AURA)
289  SpellSummary[i].Effects |= 1 << (SELECT_EFFECT_AURA - 1);
290  }
291  }
