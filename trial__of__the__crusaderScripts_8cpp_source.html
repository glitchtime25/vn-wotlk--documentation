123 bool GossipHello_npc_barrett_ramsey(Player* pPlayer, Creature* pCreature)
124 {
125  ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData();
126  uint8 uiPos = 0;
127  uint32 uiType = 0;
128 
129  for (uint8 i = 0; i < countof(aRamseyInfo); ++i)
130  {
131  if (pCreature->GetEntry() == aRamseyInfo[i].uiEntry)
132  {
133  if (!aRamseyInfo[i].uiOptionId)
134  uiPos = i;
135  else
136  {
137  uiType = aRamseyInfo[i].uiOptionId;
138  break;
139  }
140  }
141  }
142 
143  if (!uiType || !pInstance)
144  return true;
145 
146  if (pInstance->GetData(uiType) == FAIL)
147  {
148  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, aRamseyInfo[uiPos].iWipeGossipItem, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
149  pPlayer->SEND_GOSSIP_MENU(aRamseyInfo[uiPos].uiWipeTextEntry, pCreature->GetObjectGuid());
150  }
151  else
152  {
153  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, aRamseyInfo[uiPos].iGossipItem, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF);
154  pPlayer->SEND_GOSSIP_MENU(aRamseyInfo[uiPos].uiTextEntry, pCreature->GetObjectGuid());
155  }
156 
157  return true;
158 }

160 bool GossipSelect_npc_barrett_ramsey(Player* pPlayer, Creature* pCreature, uint32 /*uiSender*/, uint32 uiAction)
161 {
162  ScriptedInstance* pInstance = (ScriptedInstance*)pCreature->GetInstanceData();
163  if (!pInstance)
164  return true;
165 
166  if (uiAction > GOSSIP_ACTION_INFO_DEF)
167  {
168  // Begin Event
169  uint32 uiType = uiAction - GOSSIP_ACTION_INFO_DEF;
170  if (pInstance->GetData(uiType) == FAIL || pInstance->GetData(uiType) == NOT_STARTED)
171  pInstance->SetData(uiAction - GOSSIP_ACTION_INFO_DEF, SPECIAL);
172 
173  pPlayer->CLOSE_GOSSIP_MENU();
174  pCreature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
175  pCreature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
176  pCreature->GetMotionMaster()->MovePoint(1, aRamsayPositions[1][0], aRamsayPositions[1][1], aRamsayPositions[1][2]);
177 
178  return true;
179  }
180 
181  for (const auto& i : aRamseyInfo)
182  {
183  if (pCreature->GetEntry() == i.uiEntry && i.uiOptionId)
184  {
185  if (pInstance->GetData(i.uiOptionId) == FAIL)
186  {
187  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, i.iWipeGossipItem, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + i.uiOptionId);
188  pPlayer->SEND_GOSSIP_MENU(i.uiWipeTextEntry, pCreature->GetObjectGuid());
189  }
190  else
191  {
192  pPlayer->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, i.iGossipItem, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + i.uiOptionId);
193  pPlayer->SEND_GOSSIP_MENU(i.uiTextEntry, pCreature->GetObjectGuid());
194  }
195 
196  return true;
197  }
198  }
199 
200  return true;
201 }
