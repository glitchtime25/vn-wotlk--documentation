120 void instance_dark_portal::SetData(uint32 uiType, uint32 uiData)
121 {
122  switch (uiType)
123  {
124  case TYPE_MEDIVH:
125  {
126  if (uiData == IN_PROGRESS)
127  {
128  if (Creature* pMedivh = GetSingleCreatureFromStorage(NPC_MEDIVH))
129  {
130  if (pMedivh->IsAlive())
131  {
132  DoScriptText(SAY_MEDIVH_ENTER, pMedivh);
133 
134  if (Creature* pSaat = GetSingleCreatureFromStorage(NPC_SAAT)) // Sa'at is not a questgiver while event is in progress
135  pSaat->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
136  }
137  // If Medivh is not available the do not store the uiData;
138  else
139  return;
140  }
141 
142  UpdateWorldState();
143  m_uiNextPortalTimer = 3000;
144  m_uiSummonCrystalTimer = 1000;
145  m_uiSummonBeamStalkerTimer = 1000;
146  }
147  if (uiData == DONE)
148  {
149  StartPostEvent();
150 
151  if (Creature* pSaat = GetSingleCreatureFromStorage(NPC_SAAT))
152  pSaat->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
153  }
154  if (uiData == FAIL)
155  {
156  DoResetEvent();
157 
158  if (Creature* pSaat = GetSingleCreatureFromStorage(NPC_SAAT))
159  pSaat->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
160  }
161  m_auiEncounter[uiType] = uiData;
162  break;
163  }
164  case TYPE_SHIELD:
165  if (uiData == SPECIAL)
166  {
167  --m_uiWorldStateShieldCount;
168  DoUpdateWorldState(WORLD_STATE_OPENING_THE_DARK_PORTAL_SHIELD_STATE, m_uiWorldStateShieldCount);
169 
170  // Yell at 75%, 50% and 25% shield
171  if (m_uiWorldStateShieldCount < 100 - 25 * m_uiMedivhYellCount)
172  {
173  if (Creature* pMedivh = GetSingleCreatureFromStorage(NPC_MEDIVH))
174  {
175  DoScriptText(uiMedivhWeakYell[m_uiMedivhYellCount - 1], pMedivh);
176  ++m_uiMedivhYellCount;
177  }
178 
179  if (Creature* pDarkPortalDummy = GetSingleCreatureFromStorage(NPC_DARK_PORTAL_DUMMY))
180  pDarkPortalDummy->CastSpell(pDarkPortalDummy, SPELL_CRYSTAL_SHATTER, TRIGGERED_OLD_TRIGGERED);
181  }
182 
183  // Kill the npc when the shield is broken
184  if (!m_uiWorldStateShieldCount)
185  {
186  if (Creature* medivh = GetSingleCreatureFromStorage(NPC_MEDIVH))
187  {
188  if (medivh->IsAlive())
189  medivh->Suicide();
190  }
191  }
192  }
193  m_auiEncounter[uiType] = uiData;
194  return;
195  case TYPE_TIME_RIFT:
196  {
197  // Set the delay to the next time rift from the point the rift despawns
198  if (uiData == DONE)
199  m_uiNextPortalTimer = IsBossTimeRift() ? 125000 : 3000;
200  // Set the delay to the next time rift from the point the rift summons it's guardian
201  // ToDo: research if these timers are correct
202  else if (uiData == SPECIAL)
203  m_uiNextPortalTimer = IsBossTimeRift() ? 0 : m_uiWorldStateRiftCount > 12 ? 90000 : 2 * MINUTE * IN_MILLISECONDS;
204 
205  m_auiEncounter[uiType] = uiData;
206  return;
207  }
208  case TYPE_CHRONO_LORD:
209  case TYPE_TEMPORUS:
210  if (m_auiEncounter[uiType] != DONE) // Keep the DONE-information stored
211  m_auiEncounter[uiType] = uiData;
212  break;
213  case TYPE_AEONUS:
214  if (uiData == DONE)
215  {
216  UpdateWorldState(false);
217  SetData(TYPE_MEDIVH, DONE); // Start Post Event
218  }
219  m_auiEncounter[uiType] = uiData;
220  break;
221  default:
222  return;
223  }
224 
225  if (uiData == DONE)
226  {
227  OUT_SAVE_INST_DATA;
228 
229  std::ostringstream saveStream;
230  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
231  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5];
232 
233  m_strInstData = saveStream.str();
234 
235  SaveToDB();
236  OUT_SAVE_INST_DATA_COMPLETE;
237  }

380  Map::PlayerList const& players = instance->GetPlayers();

384  for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
385  {
386  if (Player* pPlayer = itr->getSource())

490 void instance_dark_portal::Update(uint32 uiDiff)
491 {
492  if (m_uiPostEventTimer && GetData(TYPE_MEDIVH) == DONE)
493  {
494  if (m_uiPostEventTimer <= uiDiff)
495  {
496  ExecutePostEventStep();
497  }
498  else
499  m_uiPostEventTimer -= uiDiff;
500  }
501 
502  if (GetData(TYPE_MEDIVH) != IN_PROGRESS)
503  return;
504 
505  if (m_uiNextPortalTimer)
506  {
507  if (m_uiNextPortalTimer <= uiDiff)
508  {
509  DoUpdateWorldState(WORLD_STATE_OPENING_THE_DARK_PORTAL_RIFT_STATE, ++m_uiWorldStateRiftCount);
510 
511  DoSpawnNextPortal();
512  m_uiNextPortalTimer = 0;
513  }
514  else
515  m_uiNextPortalTimer -= uiDiff;
516  }
517 
518  if (m_uiSummonCrystalTimer)
519  {
520  if (m_uiSummonCrystalTimer <= uiDiff)
521  {
522  if (m_uiSummonedCrystalCount < MAX_CRYSTALS)
523  {
524  if (Creature* pDarkPortalDummy = GetSingleCreatureFromStorage(NPC_DARK_PORTAL_DUMMY))
525  {
526  pDarkPortalDummy->CastSpell(pDarkPortalDummy, SPELL_PORTAL_CRYSTAL, TRIGGERED_OLD_TRIGGERED);
527  m_uiSummonedCrystalCount++;
528  m_uiSummonCrystalTimer = 5000;
529  }
530  }
531  else
532  m_uiSummonCrystalTimer = 0;
