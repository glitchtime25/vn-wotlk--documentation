147 struct SwiftFlightFormPassive : public AuraScript
148 {
149  int32 OnAuraValueCalculate(AuraCalcData& data, int32 value) const override
150  {
151  if (!data.target->IsPlayer() || data.effIdx != EFFECT_INDEX_1)
152  return value;
153  Player* player = static_cast<Player*>(data.target);
154  for (PlayerSpellMap::const_iterator iter = player->GetSpellMap().begin(); iter != player->GetSpellMap().end(); ++iter)
155  {
156  if (iter->second.state != PLAYERSPELL_REMOVED)
157  {
158  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(iter->first);
159  for (int i = 0; i < MAX_EFFECT_INDEX; ++i)
160  {
161  if (spellInfo->EffectApplyAuraName[i] == SPELL_AURA_MOD_FLIGHT_SPEED_MOUNTED)
162  {
163  int32 mountSpeed = spellInfo->CalculateSimpleValue(SpellEffectIndex(i));
164  if (mountSpeed > value)
165  return mountSpeed;
166  }
167  }
168  }
169  }
170  return value;
171  }
