129  void ExecuteAction(uint32 action) override
130  {
131  switch (action)
132  {
133  case KRIKTHIR_ACTION_FRENZY:
134  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
135  {
136  DoScriptText(EMOTE_BOSS_GENERIC_FRENZY, m_creature);
137  DisableCombatAction(action);
138  }
139  break;
140  case KRIKTHIR_ACTION_CURSE_FATIGUE:
141  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_CURSE_OF_FATIGUE : SPELL_CURSE_OF_FATIGUE_H) == CAST_OK)
142  ResetCombatAction(action, 30000);
143  break;
144  case KRIKTHIR_ACTION_MINDFLAY:
145  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_MINDFLAY : SPELL_MINDFLAY_H) == CAST_OK)
146  ResetCombatAction(action, 10000);
147  break;
148  case KIRKTHIR_ACTION_SWARM:
149  if (DoCastSpellIfCan(m_creature, SPELL_SWARM) == CAST_OK)
150  {
151  DoScriptText(urand(0, 1) ? SAY_SWARM_1 : SAY_SWARM_2, m_creature);
152  ResetCombatAction(action, 30000);
153  }
154  break;
155  }
156  }
157 };
