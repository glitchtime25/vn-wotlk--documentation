153  void UpdateAI(const uint32 uiDiff) override
154  {
155  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
156  return;
157 
158  if (m_uiSummonTimer)
159  {
160  if (m_uiSummonTimer <= uiDiff)
161  {
162  // Summon adds and restart chasing the victim
163  DoSummons();
164  DoScriptText(EMOTE_SUMMON, m_creature);
165 
166  if (m_creature->GetVictim())
167  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
168  m_uiSummonTimer = 0;
169  }
170  else
171  m_uiSummonTimer -= uiDiff;
172  }
173 
174  if (m_uiAllergicTimer < uiDiff)
175  {
176  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ALLERGIC_REACTION) == CAST_OK)
177  m_uiAllergicTimer = urand(21000, 32000);
178  }
179  else
180  m_uiAllergicTimer -= uiDiff;
181 
182  if (m_uiTeleportTimer < uiDiff)
183  {
184  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT_SELF) == CAST_OK)
185  {
186  m_creature->GetMotionMaster()->MoveIdle();
187  m_uiTeleportTimer = urand(25000, 33000);
188  m_uiSummonTimer = 4000;
189  }
190  }
191  else
192  m_uiTeleportTimer -= uiDiff;
193 
194  if (m_uiTransformTimer < uiDiff)
195  {
196  DoTransform();
197  m_uiTransformTimer = urand(25000, 40000);
198  }
199  else
200  m_uiTransformTimer -= uiDiff;
201 
202  if (m_uiTrashTimer < uiDiff)
203  {
204  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_TRASH) == CAST_OK)
205  m_uiTrashTimer = urand(10000, 24000);
206  }
207  else
208  m_uiTrashTimer -= uiDiff;
209 
210  DoMeleeAttackIfReady();
211  }
