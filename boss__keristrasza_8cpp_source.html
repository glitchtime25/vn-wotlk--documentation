121  void ExecuteAction(uint32 action) override
122  {
123  switch (action)
124  {
125  case KERISTRASZA_ACTION_TAIL_SWEEP:
126  if (DoCastSpellIfCan(m_creature, SPELL_TAIL_SWEEP) == CAST_OK)
127  ResetCombatAction(action, 10000);
128  break;
129  case KERISTRASZA_ACTION_CRYSTALFIRE_BREATH:
130  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_CRYSTALFIRE_BREATH : SPELL_CRYSTALFIRE_BREATH_H) == CAST_OK)
131  ResetCombatAction(action, 15000);
132  break;
133  case KERISTRASZA_ACTION_CRYSTALLIZE:
134  // different spells for heroic and non heroic
135  if (m_isRegularMode)
136  {
137  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_CRYSTAL_CHAINS, SELECT_FLAG_PLAYER))
138  {
139  if (DoCastSpellIfCan(pTarget, SPELL_CRYSTAL_CHAINS) == CAST_OK)
140  ResetCombatAction(action, 25000);
141  }
142  }
143  else
144  {
145  if (DoCastSpellIfCan(m_creature, SPELL_CRYSTALLIZE) == CAST_OK)
146  {
147  ResetCombatAction(action, 25000);
148  DoScriptText(SAY_CRYSTAL_NOVA, m_creature);
149  }
150 
151  }
152  break;
153  case KERISTRASZA_ACTION_ENRAGE:
154  if (m_creature->GetHealthPercent() < 25.0f)
155  {
156  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
157  {
158  DoScriptText(SAY_ENRAGE, m_creature);
159  SetActionReadyStatus(action, false);
160  }
161  }
162  break;
163  }
164  }
