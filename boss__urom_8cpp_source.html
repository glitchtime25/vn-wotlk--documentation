136  void AttackStart(Unit* pWho) override
137  {
138  if (m_uiPlatformPhase < MAX_PLATFORMS)
139  {
140  if (m_bIsTeleporting)
141  return;
142 
143  // Summon the trash mobs pack
144  m_bIsTeleporting = true;
145  m_attackTarget = pWho->GetObjectGuid();
146  m_creature->InterruptNonMeleeSpells(false);
147  DoSpawnTrashPack();
148 
149  // teleport to next platform and spawn adds
150  switch (m_uiPlatformPhase)
151  {
152  case 0:
153  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_MENAGERIE_1) == CAST_OK)
154  DoScriptText(SAY_SUMMON_1, m_creature);
155  break;
156  case 1:
157  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_MENAGERIE_2) == CAST_OK)
158  DoScriptText(SAY_SUMMON_2, m_creature);
159  break;
160  case 2:
161  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_MENAGERIE_3) == CAST_OK)
162  DoScriptText(SAY_SUMMON_3, m_creature);
163  break;
164  }
165  }
166  // Boss has teleported in the central ring - start normal combat
167  else if (m_bIsPlatformPhase)
168  {
169  DoScriptText(SAY_AGGRO, m_creature);
170  m_creature->InterruptNonMeleeSpells(false);
171  m_bIsPlatformPhase = false;
172 
173  ScriptedAI::AttackStart(pWho);
174  }
175  }

232  void DoSpawnTrashPack()
233  {
234  float fX, fY, fZ;
235 
236  // Summon the 3 mobs contained in the pack
237  for (uint8 i = 0; i < MAX_PLATFORMS; ++i)
238  {
239  m_creature->GetNearPoint(m_creature, fX, fY, fZ, 0, 10.0f, M_PI_F / 2 * i);
240  m_creature->SummonCreature(uiTrashPacks[m_vuiTrashPacksIds[m_uiPlatformPhase]][i], fX, fY, fZ, 0.0f, TEMPSPAWN_TIMED_OOC_DESPAWN, 30000);
241  }
242 
243  // Summon a fourth mob, which can be random
244  uint32 uiEntry = uiTrashPacks[m_vuiTrashPacksIds[m_uiPlatformPhase]][urand(0, 2)];
245  m_creature->GetNearPoint(m_creature, fX, fY, fZ, 0, 10.0f, M_PI_F / 2 * 3);
246  m_creature->SummonCreature(uiEntry, fX, fY, fZ, 0.0f, TEMPSPAWN_TIMED_OOC_DESPAWN, 30000);

262  void UpdateAI(const uint32 uiDiff) override
263  {
264  // Set the Arcane Shield on out of combat timer
265  if (m_uiArcaneShieldTimer)
266  {
267  if (m_uiArcaneShieldTimer <= uiDiff)
268  {
269  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_SHIELD) == CAST_OK)
270  m_uiArcaneShieldTimer = 0;
271  }
272  else
273  m_uiArcaneShieldTimer -= uiDiff;
274  }
275 
276  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
277  return;
278 
279  // Don't use any combat abilities during the platform transition
280  if (m_bIsPlatformPhase)
281  return;
282 
283  if (m_uiExplosionTimer)
284  {
285  if (m_uiExplosionTimer <= uiDiff)
286  {
287  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ARCANE_EXPLOSION : SPELL_ARCANE_EXPLOSION_H) == CAST_OK)
288  {
289  DoScriptText(EMOTE_EXPLOSION, m_creature);
290  m_uiExplosionTimer = 0;
291  }
292  }
293  else
294  m_uiExplosionTimer -= uiDiff;
295  }
296 
297  if (m_uiExplosionExpireTimer)
298  {
299  if (m_uiExplosionExpireTimer <= uiDiff)
300  {
301  // Teleport to the original location
302  m_creature->NearTeleportTo(m_fX, m_fY, m_fZ, 0);
303 
304  // Resume combat movement
305  SetCombatMovement(true);
306  m_creature->GetMotionMaster()->Clear();
307  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
308  m_uiExplosionExpireTimer = 0;
309  }
310  else
311  m_uiExplosionExpireTimer -= uiDiff;
312 
313  // Don't decrease timers during the explosion event
314  return;
315  }
316 
317  if (m_uiTeleportTimer < uiDiff)
318  {
319  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT) == CAST_OK)
320  {
321  DoScriptText(urand(0, 1) ? SAY_EXPLOSION_1 : SAY_EXPLOSION_2, m_creature);
322 
323  // Store the original position - boss needs to be teleported back
324  m_creature->GetPosition(m_fX, m_fY, m_fZ);
325 
326  // Stop movement until he casts the arcane explosion
327  SetCombatMovement(false);
328  m_creature->GetMotionMaster()->MoveIdle();
329  m_uiTeleportTimer = 20000;
330  m_uiExplosionExpireTimer = m_bIsRegularMode ? 9500 : 7500;
331  m_uiExplosionTimer = 1000;
332  }
333  }
334  else
335  m_uiTeleportTimer -= uiDiff;
336 
337  if (m_uiFrostBombTimer < uiDiff)
338  {
339  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FROSTBOMB) == CAST_OK)
340  m_uiFrostBombTimer = urand(4000, 6000);
341  }
342  else
343  m_uiFrostBombTimer -= uiDiff;
344 
345  if (m_uiTimeBombTimer < uiDiff)
346  {
347  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
348  {
349  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_TIME_BOMB : SPELL_TIME_BOMB_H) == CAST_OK)
350  m_uiTimeBombTimer = urand(10000, 15000);
351  }
352  }
353  else
354  m_uiTimeBombTimer -= uiDiff;
355 
356  DoMeleeAttackIfReady();
357  }
