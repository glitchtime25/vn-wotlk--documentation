35 void WaypointMovementGenerator<Creature>::LoadPath(Creature& creature, int32 pathId, WaypointPathOrigin wpOrigin, uint32 overwriteEntry)
36 {
37  DETAIL_FILTER_LOG(LOG_FILTER_AI_AND_MOVEGENSS, "LoadPath: loading waypoint path for %s", creature.GetGuidStr().c_str());
38 
39  if (!overwriteEntry)
40  overwriteEntry = creature.GetEntry();
41 
42  //sLog.outErrorScriptLib("LoadPath: Assign creature guid(%u) and entry(%u) path %i", creature.GetDbGuid(), overwriteEntry, pathId);
43 
44  if (wpOrigin == PATH_NO_PATH && pathId == 0)
45  i_path = sWaypointMgr.GetDefaultPath(overwriteEntry, creature.GetDbGuid(), &m_PathOrigin);
46  else
47  {
48  m_PathOrigin = wpOrigin == PATH_NO_PATH ? PATH_FROM_ENTRY : wpOrigin;
49  i_path = sWaypointMgr.GetPathFromOrigin(overwriteEntry, creature.GetDbGuid(), pathId, m_PathOrigin);
50  }
51  m_pathId = pathId;
52 
53  // No movement found for entry nor guid
54  if (!i_path)
55  {
56  if (m_PathOrigin == PATH_FROM_EXTERNAL)
57  sLog.outErrorScriptLib("WaypointMovementGenerator::LoadPath: %s doesn't have waypoint path %i", creature.GetGuidStr().c_str(), pathId);
58  else
59  sLog.outErrorDb("WaypointMovementGenerator::LoadPath: %s doesn't have waypoint path %i", creature.GetGuidStr().c_str(), pathId);
60  return;
61  }
62 
63  if (i_path->empty())
64  return;
65  // Initialize the i_currentNode to point to the first node
66  i_currentNode = i_path->begin()->first;
67  m_lastReachedWaypoint = 0;
68  m_currentWaypointNode = i_path->begin();
69 
70  if (i_path->size() < 2)
71  {
72  auto& node = i_path->begin()->second;
73  if (node.delay == 0)
74  {
75  // db dev seem to use one wp for some specific script without delay
76  sLog.outErrorDb("WaypointMovementGenerator::LoadPath()> creature %s have only one wp and without any delay. Delay forced to 5 sec!",
77  creature.GetGuidStr().c_str());
78 
79  // little hack to change db data
80  auto modNode = const_cast<WaypointNode&>(node);
81  modNode.delay = 5000;
82  }
83  }
84 }

95  i_nextMoveTime.Reset(initialDelay);
96  if (!i_path)
97  {
98  sLog.outErrorDb("void WaypointMovementGenerator<Creature>::InitializeWaypointPath> unable to intialize path for %s", u.GetGuidStr().c_str());
99  return;
100  }
101 
102  i_nextMoveTime.Reset(initialDelay);
103 
104  uint32 startPoint = -1; // TODO add an argument to set the start point
105  if (startPoint >= 0)
106  {
107  auto selPoint = i_path->find(startPoint);
108  if (selPoint != i_path->end())
109  {
110  i_currentNode = startPoint;
111  m_currentWaypointNode = selPoint;
112  }
113  else
114  {
115  m_currentWaypointNode = i_path->begin();
116  i_currentNode = m_currentWaypointNode->first;

118  }
119

121  if (m_currentWaypointNode != i_path->begin())
122  {
123  auto lastNode = m_currentWaypointNode;
124  --lastNode;
125  m_lastReachedWaypoint = lastNode->first;
126  }
127  else
128  m_lastReachedWaypoint = 0;
129 
130  // Start moving if possible
131  SendNextWayPointPath(u);
132  m_resetPoint = u.GetPosition(u.GetTransport());

173 void WaypointMovementGenerator<Creature>::OnArrived(Creature& creature)
174 {
175  // already arrived?
176  if (m_lastReachedWaypoint == i_currentNode)
177  return;
178 
179  m_lastReachedWaypoint = m_currentWaypointNode->first;
180 
181  WaypointNode const& node = m_currentWaypointNode->second;
182 
183  if (node.script_id)
184  {
185  DEBUG_FILTER_LOG(LOG_FILTER_AI_AND_MOVEGENSS, "Creature movement start script %u at point %u for %s.", node.script_id, i_currentNode, creature.GetGuidStr().c_str());
186  creature.GetMap()->ScriptsStart(SCRIPT_TYPE_CREATURE_MOVEMENT, node.script_id, &creature, m_guid ? creature.GetMap()->GetWorldObject(m_guid) : &creature);
187  }
188 
189  // Inform script
190  if (creature.AI())
191  {
192  bool inform = false;
193  if (m_PathOrigin == PATH_FROM_EXTERNAL)
194  inform = true;
195  else if (m_PathOrigin == PATH_FROM_WAYPOINT_PATH) // temporary transitional code for transitioning from script_texts to waypoint_path
196  {
197  if (m_pathId == creature.AI()->GetCurrentWaypointPath())
198  inform = true;
199  }
200 
201  uint32 type = WAYPOINT_MOTION_TYPE;
202  if (inform)
203  type = EXTERNAL_WAYPOINT_MOVE;
204  InformAI(creature, type, i_currentNode);
205  }
206 
207  // save position and orientation in case of GetResetPosition() call
208  m_resetPoint = creature.GetPosition(creature.GetTransport());
209 
210  // AI can modify node delay so we have to compute it
211  int32 newWaitTime = node.delay + m_scriptTime;
212 
213  // switch to next node
214  SwitchToNextNode(creature, m_currentWaypointNode);
215 
216  // Wait delay ms
217  if (newWaitTime > 0)
218  {
219  creature.clearUnitState(UNIT_STAT_ROAMING_MOVE);
220  Stop(newWaitTime);
221 
222  // if script modified the node delay we already sent path to client to move the creature so we have to stop it manually
223  if (!creature.movespline->Finalized() && node.delay == 0 && m_scriptTime != 0)
224  {
225  Movement::MoveSplineInit init(creature);
226  init.Stop();
227  }
228  }
229  else if (newWaitTime == 0 && node.delay != 0)
230  {
231  // we have to manually restart movement as script changed the node delay to 0
232  SendNextWayPointPath(creature);
233  }
234 
235  // Fire event only if it is already moving to the next node (for normal start it is already done in SendNewPath)
236  if (!m_nodeIndexes.empty() && !Stopped(creature))
237  {
238  bool inform = false;
239  if (m_PathOrigin == PATH_FROM_EXTERNAL)
240  inform = true;
241  else if (m_PathOrigin == PATH_FROM_WAYPOINT_PATH) // temporary transitional code for transitioning from script_texts to waypoint_path
242  {
243  if (m_pathId == creature.AI()->GetCurrentWaypointPath())
244  inform = true;
245  }
246 
247  // Inform AI that we start to move
248  if (creature.AI() && inform)
249  InformAI(creature, uint32(EXTERNAL_WAYPOINT_MOVE_START), m_currentWaypointNode->first);
250  }
251 
252  i_currentNode = m_currentWaypointNode->first;
253  m_scriptTime = 0;
254 }

257 uint32 WaypointMovementGenerator<Creature>::BuildIntPath(PointsArray& path, Creature& creature, Vector3 const& endPos)
258 {
259  Vector3 startPos = path.back();
260  uint32 travelTime = 0;
261  auto speedType = MovementInfo::GetSpeedType(creature.m_movementInfo.GetMovementFlags());
262  float creatureSpeed = creature.GetSpeed(speedType);
263 
264  PathFinder pathfinder(&creature, true);
265  pathfinder.calculate(startPos, endPos, true);
266  auto genPath = pathfinder.getPath();
267 
268  Vector3 firstPoint = startPos;
269  bool first = true;
270  for (auto& point : genPath)
271  {
272  if (first) // first point is already in path
273  {
274  first = false;
275  continue;
276  }
277  const Vector3 offset = point - firstPoint;
278  const float distance = offset.magnitude();
279  travelTime += distance / creatureSpeed * 1000;
280  path.push_back(point);
281  }
282  const Vector3 offset = endPos - firstPoint;
283  const float distance = offset.magnitude();
284  travelTime = distance / creatureSpeed * 1000;
285 
286  return travelTime;
287 }

301 // client need to receive new path before the end of previous path (much higher value have been seen in sniff)

304 // build and send path to next node
305 void WaypointMovementGenerator<Creature>::SendNextWayPointPath(Creature& creature)
306 {
307  // make sure to reset spline index as its new path
308  m_nodeIndexes.clear();
309 
310  // prevent movement while casting spells with cast time or channel time
311  if (!creature.IsAlive() || creature.IsImmobilizedState() || creature.IsNonMeleeSpellCasted(false, false, true, true))
312  {
313  if (!creature.movespline->Finalized())
314  {
315  if (creature.IsClientControlled())
316  creature.StopMoving(true);
317  else
318  creature.InterruptMoving();
319  }
320  return;
321  }
322 
323  creature.addUnitState(UNIT_STAT_ROAMING_MOVE);
324  m_speedChanged = false;
325 
326  WaypointNode const* nextNode = &m_currentWaypointNode->second;
327 
328  bool inform = false;
329  if (m_PathOrigin == PATH_FROM_EXTERNAL)
330  inform = true;
331  else if (m_PathOrigin == PATH_FROM_WAYPOINT_PATH && creature.AI()) // temporary transitional code for transitioning from script_texts to waypoint_path
332  {
333  if (m_pathId == creature.AI()->GetCurrentWaypointPath())
334  inform = true;
335  }
336 
337  // Inform AI that we start to move or reached last node
338  if (creature.AI() && inform)
339  {
340  uint32 type = uint32(EXTERNAL_WAYPOINT_MOVE_START);
341  if (i_path->begin()->first == i_currentNode && i_path->rbegin()->first == m_lastReachedWaypoint)
342  type = uint32(EXTERNAL_WAYPOINT_FINISHED_LAST);
343 
344  InformAI(creature, type, i_currentNode);
345 
346  if (creature.IsDead() || !creature.IsInWorld()) // Might have happened with above calls
347  return;
348  }
349 
350  // will contain generated path
351  PointsArray genPath;
352  genPath.reserve(20); // little optimization
353 
354  Vector3 startPos(creature.GetPositionX(), creature.GetPositionY(), creature.GetPositionZ());
355  if (!creature.movespline->Finalized())
356  {
357  if (GenericTransport* transport = creature.GetTransport())
358  transport->CalculatePassengerPosition(startPos.x, startPos.y, startPos.z, nullptr);
359  startPos = creature.movespline->ComputePosition();
360  }
361 
362  genPath.push_back(startPos);
363 
364  // compute path to next node and put it in the path
365  uint32 travelTime = BuildIntPath(genPath, creature, Vector3(nextNode->x, nextNode->y, nextNode->z));
366 
367  bool looped = false;
368  auto currPointItr = m_currentWaypointNode;
369  // add more node until travel time is big enough
370  while (!nextNode->delay && travelTime < MinimumPathTime)
371  {
372  // we'll add path to node after this one too to make animation more smoother
373  auto nodeAfterItr = currPointItr;
374 
375  if (!GetNodeAfter(nodeAfterItr, looped))
376  {
377  if (looped) //that should never happen. (Well can but with abnormally short path)
378  break;
379 
380  // we reach the end of the path better is to retry in the opposite direction
381  looped = true;
382  if (!GetNodeAfter(nodeAfterItr, true))
383  break; // did all the path and not reached MinimumPathTime?
384  }
385 
386  auto const nodeAfter = nodeAfterItr->second;
387 
388  // extend path only if next node is different than current node
389  m_nodeIndexes.push_back(genPath.size() - 1);
390  Vector3 nodeAfterCoord(nodeAfter.x, nodeAfter.y, nodeAfter.z);
391  travelTime += BuildIntPath(genPath, creature, nodeAfterCoord);
392  currPointItr = nodeAfterItr;
393  nextNode = &nodeAfterItr->second;
394  }
395 
396  Movement::MoveSplineInit init(creature);
397  init.MovebyPath(genPath);
398  if (nextNode->orientation != 100 && nextNode->delay != 0)
399  init.SetFacing(nextNode->orientation);
400  if (m_forcedMovement == FORCED_MOVEMENT_WALK)
401  init.SetWalk(true);
402  else if (m_forcedMovement == FORCED_MOVEMENT_RUN)
403  init.SetWalk(false);
404  else
405  init.SetWalk(!creature.hasUnitState(UNIT_STAT_RUNNING));
406 
407  // send path to client
408  m_pathDuration = init.Launch();
409 
410  // remove presend time if next node have no delay
411  if (nextNode->delay == 0)
412  m_pathDuration -= PreSendTime;
413  if (i_path->size() == 1)
414  m_lastReachedWaypoint = 0;
415 }

429 bool WaypointMovementGenerator<Creature>::Update(Creature& creature, const uint32& diff)
430 {
431  // Waypoint movement can be switched on/off
432  // This is quite handy for escort quests and other stuff
433  if (creature.hasUnitState(UNIT_STAT_NOT_MOVE))
434  {
435  creature.clearUnitState(UNIT_STAT_ROAMING_MOVE);
436  return true;
437  }
438 
439  // prevent a crash at empty waypoint path.
440  if (!i_path || i_path->empty())
441  {
442  creature.clearUnitState(UNIT_STAT_ROAMING_MOVE);
443  return true;
444  }
445 
446  if (Stopped(creature) || m_speedChanged)
447  {
448  // If a script just have set the waypoint to be paused or stopped we have to check
449  // if the client did get a path for this creature. If it is the case, we have to
450  // explicitly send stop so the client knows that we want that creature to be stopped
451  if (!creature.movespline->Finalized())
452  {
453  Movement::MoveSplineInit init(creature);
454  init.Stop();
455  }
456 
457  if (CanMove(diff, creature))
458  SendNextWayPointPath(creature);
459  }
460  else
461  {
462  if (creature.IsStopped())
463  {
464  if (creature.IsNonMeleeSpellCasted(false, false, true, true))
465  return true;
466  Stop(1000);
467  }
468  else
469  {
470  if (creature.movespline->Finalized())
471  {
472  // we arrived to a node either by movespline finalized or node reached while creature continue to move
473  OnArrived(creature); // fire script events
474  if (!Stopped(creature)) // check if not stopped in OnArrived
475  SendNextWayPointPath(creature); // restart movement
476  }
477  else
478  {
479  if (m_pathDuration <= 0)
480  {
481  // time to send new packet
482  if (m_currentWaypointNode->second.delay == 0)
483  SendNextWayPointPath(creature);
484  }
485  else
486  m_pathDuration -= diff;
487 
488  if (!m_nodeIndexes.empty() && creature.movespline->currentPathIdx() >= m_nodeIndexes.front())
489  {
490  // node reached while moving
491  m_nodeIndexes.pop_front();
492  OnArrived(creature); // fire script events
493  }
494  }
495  }
496  }
497 
498  return true;
499 }

515 bool WaypointMovementGenerator<Creature>::GetResetPosition(Creature&, float& x, float& y, float& z, float& o) const
516 {
517  x = m_resetPoint.x;
518  y = m_resetPoint.y;
519  z = m_resetPoint.z;
520  o = m_resetPoint.o;
521 
522  return true;
523  // prevent a crash at empty waypoint path.
524  if (!i_path || i_path->empty())
525  return false;
526 
527  WaypointPath::const_iterator lastPoint = i_path->find(m_lastReachedWaypoint);
528  // Special case: Before the first waypoint is reached, m_lastReachedWaypoint is set to 0 (which may not be contained in i_path)
529  if (!m_lastReachedWaypoint && lastPoint == i_path->end())
530  return false;
531 
532  MANGOS_ASSERT(lastPoint != i_path->end());
533 
534  WaypointNode const* curWP = &(lastPoint->second);
535 
536  x = curWP->x;
537  y = curWP->y;
538  z = curWP->z;
539 
540  if (curWP->orientation != 100)
541  o = curWP->orientation;
542  else // Calculate the resulting angle based on positions between previous and current waypoint
543  {
544  WaypointNode const* prevWP;
545  if (lastPoint != i_path->begin()) // Not the first waypoint
546  {
547  --lastPoint;
548  prevWP = &(lastPoint->second);
549  }
550  else // Take the last waypoint (crbegin()) as previous
551  prevWP = &(i_path->rbegin()->second);
552 
553  float dx = x - prevWP->x;
554  float dy = y - prevWP->y;
555  o = atan2(dy, dx); // returns value between -Pi..Pi
556 
557  o = (o >= 0) ? o : 2 * M_PI_F + o;
558  }
559 
560  return true;
561 }

587 bool WaypointMovementGenerator<Creature>::SetNextWaypoint(uint32 pointId)
588 {
589  if (!i_path || i_path->empty())
590  return false;
591 
592  // point id should not be bigger than the size of the path -1
593  pointId = uint32(pointId % i_path->size());
594 
595  WaypointPath::const_iterator currPoint = i_path->find(pointId);
596  if (currPoint == i_path->end())
597  return false;
598 
599  // Allow Moving with next tick
600  // Handle allow movement this way to not interact with PAUSED state.
601  // If this function is called while PAUSED, it will move properly when unpaused.
602  i_nextMoveTime.Reset(1);
603 
604  // Set the point
605  i_currentNode = pointId;
606  m_currentWaypointNode = currPoint;
607 
608  // set last reached point accordingly to avoid going back to point 0 if the
609  // movegen is interrupted before reaching next point
610  if (pointId > 0)
611  m_lastReachedWaypoint = pointId - 1;
612  else
613  m_lastReachedWaypoint = i_path->rbegin()->first;
614  return true;
615 }
