217  void JustDidDialogueStep(int32 iEntry) override
218  {
219  switch (iEntry)
220  {
221  case SPELL_ETHEREAL_TELEPORT:
222  DoCastSpellIfCan(m_creature, SPELL_ETHEREAL_TELEPORT);
223  m_creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
224  break;
225  case NPC_ETHEREAL_SPELLBINDER:
226  if (m_creature->IsTemporarySummon())
227  {
228  if (Player* pSummoner = m_creature->GetMap()->GetPlayer(m_creature->GetSpawnerGuid()))
229  DoScriptText(SAY_SPAWN, m_creature, pSummoner);
230  }
231  break;
232  case SAY_ESCORT_READY:
233  m_creature->HandleEmote(EMOTE_STATE_NONE);
234  m_creature->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
235  break;
236  case SAY_REST:
237  m_creature->HandleEmote(EMOTE_STATE_NONE);
238  break;
239  case NPC_SHADOW_LORD_XIRAXIS:
240  if (Creature* pXiraxis = m_creature->GetMap()->GetCreature(m_xiraxisGuid))
241  {
242  pXiraxis->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
243  pXiraxis->AI()->AttackStart(m_creature);
244  }
245  break;
246  case SAY_EVENT_COMPLETE:
247  SetRun();
248  SetEscortPaused(false);
249  break;
250  }
251  }
