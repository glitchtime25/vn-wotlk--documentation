30 void WorldStateVariableManager::Initialize(uint32 completedEncounterMask)
31 {
32  auto bounds = sObjectMgr.GetDungeonEncounterBoundsByMap(m_owner->GetId());
33  for (auto itr = bounds.first; itr != bounds.second; ++itr)
34  {
35  DungeonEncounterEntry const* encounter = (*itr).second.dbcEntry;
36  if (m_owner->GetDifficulty() != encounter->Difficulty)
37  continue;
38 
39  if (encounter->CompleteWorldStateID) // use official data whenever available
40  SetVariable(encounter->CompleteWorldStateID, ((1 << encounter->encounterIndex) & completedEncounterMask) != 0);
41  else // phase this out eventually
42  SetEncounterVariable(encounter->Id, ((1 << encounter->encounterIndex) & completedEncounterMask) != 0);
43  }
44 }

101 void WorldStateVariableManager::BroadcastVariable(int32 Id) const
102 {
103  auto const& lPlayers = m_owner->GetPlayers();
104  if (!lPlayers.isEmpty())
105  {
106  WorldStateVariable const* variable = GetVariableData(Id);
107  MANGOS_ASSERT(variable); // if we are broadcasting a variable it must be initialized
108  int32 value = variable->value;
109  bool queryIds = variable->zoneId || variable->areaId;
110  for (const auto& lPlayer : lPlayers)
111  {
112  if (Player* player = lPlayer.getSource())
113  {
114  uint32 zoneId, areaId;
115  if (queryIds) // optimization
116  player->GetZoneAndAreaId(zoneId, areaId);
117  if ((!variable->zoneId || variable->zoneId == zoneId) &&
118  (!variable->areaId || variable->areaId == areaId))
119  player->SendUpdateWorldState(Id, value);
120  }
121  }
122  }
