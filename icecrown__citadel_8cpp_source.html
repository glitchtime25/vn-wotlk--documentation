151 void instance_icecrown_citadel::DoHandleCitadelAreaTrigger(uint32 uiTriggerId, Player* pPlayer)
152 {
153  if (uiTriggerId == AT_MARROWGAR_INTRO && !m_bHasMarrowgarIntroYelled)
154  {
155  if (Creature* pMarrowgar = GetSingleCreatureFromStorage(NPC_LORD_MARROWGAR))
156  {
157  DoScriptText(SAY_MARROWGAR_INTRO, pMarrowgar);
158  m_bHasMarrowgarIntroYelled = true;
159  }
160  }
161  else if (uiTriggerId == AT_DEATHWHISPER_INTRO && !m_bHasDeathwhisperIntroYelled)
162  {
163  StartNextDialogueText(SAY_DEATHWHISPER_SPEECH_1);
164  m_bHasDeathwhisperIntroYelled = true;
165  }
166  else if (uiTriggerId == AT_SINDRAGOSA_PLATFORM)
167  {
168  if (Creature* pSindragosa = GetSingleCreatureFromStorage(NPC_SINDRAGOSA))
169  {
170  if (pSindragosa->IsAlive() && !pSindragosa->IsInCombat())
171  pSindragosa->SetInCombatWithZone();
172  }
173  else
174  {
175  if (!m_bHasRimefangLanded)
176  {
177  if (Creature* pRimefang = GetSingleCreatureFromStorage(NPC_RIMEFANG))
178  {
179  pRimefang->AI()->AttackStart(pPlayer);
180  m_bHasRimefangLanded = true;
181  }
182  }
183 
184  if (!m_bHasSpinestalkerLanded)
185  {
186  if (Creature* pSpinestalker = GetSingleCreatureFromStorage(NPC_SPINESTALKER))
187  {
188  pSpinestalker->AI()->AttackStart(pPlayer);
189  m_bHasSpinestalkerLanded = true;
190  }
191  }
192  }
193  }
