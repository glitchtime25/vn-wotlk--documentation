32 void WorldSession::SendTradeStatus(TradeStatusInfo const& info) const
33 {
34  WorldPacket data(SMSG_TRADE_STATUS, 13);
35  data << uint32(info.Status);
36 
37  switch (info.Status)
38  {
39  case TRADE_STATUS_BEGIN_TRADE:
40  data << info.TraderGuid; // CGTradeInfo::m_tradingPlayer
41  break;
42  case TRADE_STATUS_OPEN_WINDOW:
43  data << uint32(0); // CGTradeInfo::m_tradeID
44  break;
45  case TRADE_STATUS_CLOSE_WINDOW:
46  data << uint32(info.Result); // InventoryResult
47  data << uint8(info.IsTargetResult); // bool isTargetError; used for: EQUIP_ERR_BAG_FULL, EQUIP_ERR_CANT_CARRY_MORE_OF_THIS, EQUIP_ERR_MISSING_REAGENT, EQUIP_ERR_ITEM_MAX_LIMIT_CATEGORY_COUNT_EXCEEDED
48  data << uint32(info.ItemLimitCategoryId); // ItemLimitCategory.dbc entry
49  break;
50  case TRADE_STATUS_WRONG_REALM:
51  case TRADE_STATUS_NOT_ON_TAPLIST:
52  data << uint8(info.Slot); // Trade slot; -1 here clears CGTradeInfo::m_tradeMoney
53  break;
54  default:
55  break;
56  }
57 
58  SendPacket(data);
59 }

73 void WorldSession::SendUpdateTrade(bool trader_state /*= true*/) const
74 {
75  TradeData* view_trade = trader_state ? _player->GetTradeData()->GetTraderData() : _player->GetTradeData();
76 
77  WorldPacket data(SMSG_TRADE_STATUS_EXTENDED, (100)); // guess size
78  data << uint8(trader_state ? 1 : 0); // send trader or own trade windows state (last need for proper show spell apply to non-trade slot)
79  data << uint32(0); // CGTradeInfo::m_tradeID
80  data << uint32(TRADE_SLOT_COUNT); // trade slots count/number?, = next field in most cases
81  data << uint32(TRADE_SLOT_COUNT); // trade slots count/number?, = prev field in most cases
82  data << uint32(view_trade->GetMoney()); // trader gold
83  data << uint32(view_trade->GetSpell()); // spell casted on lowest slot item
84 
85  for (uint8 i = 0; i < TRADE_SLOT_COUNT; ++i)
86  {
87  data << uint8(i); // trade slot number, if not specified, then end of packet
88 
89  if (Item* item = view_trade->GetItem(TradeSlots(i)))
90  {
91  data << uint32(item->GetProto()->ItemId); // entry
92  data << uint32(item->GetProto()->DisplayInfoID);// display id
93  data << uint32(item->GetCount()); // stack count
94  // wrapped: hide stats but show giftcreator name
95  data << uint32(item->HasFlag(ITEM_FIELD_FLAGS, ITEM_DYNFLAG_WRAPPED) ? 1 : 0);
96  data << item->GetGuidValue(ITEM_FIELD_GIFTCREATOR);
97 
98  data << uint32(item->GetEnchantmentId(PERM_ENCHANTMENT_SLOT));
99  for (uint32 enchant_slot = SOCK_ENCHANTMENT_SLOT; enchant_slot < SOCK_ENCHANTMENT_SLOT + MAX_GEM_SOCKETS; ++enchant_slot)
100  data << uint32(item->GetEnchantmentId(EnchantmentSlot(enchant_slot)));
101  // creator
102  data << item->GetGuidValue(ITEM_FIELD_CREATOR);
103  data << uint32(item->GetSpellCharges()); // charges
104  data << uint32(item->GetItemSuffixFactor()); // SuffixFactor
105  data << uint32(item->GetItemRandomPropertyId());// random properties id
106  data << uint32(item->GetProto()->LockID); // lock id
107  // max durability
108  data << uint32(item->GetUInt32Value(ITEM_FIELD_MAXDURABILITY));
109  // durability
110  data << uint32(item->GetUInt32Value(ITEM_FIELD_DURABILITY));
111  }
112  else
113  {
114  for (uint8 j = 0; j < 18; ++j)
115  data << uint32(0);
116  }
117  }
118  SendPacket(data);
119 }

124 void WorldSession::moveItems(Item* myItems[], Item* hisItems[])
125 {
126  Player* trader = _player->GetTrader();
127  if (!trader)
128  return;
129 
130  for (int i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
131  {
132  ItemPosCountVec traderDst;
133  ItemPosCountVec playerDst;
134  bool traderCanTrade = (myItems[i] == nullptr || trader->CanStoreItem(NULL_BAG, NULL_SLOT, traderDst, myItems[i], false) == EQUIP_ERR_OK);
135  bool playerCanTrade = (hisItems[i] == nullptr || _player->CanStoreItem(NULL_BAG, NULL_SLOT, playerDst, hisItems[i], false) == EQUIP_ERR_OK);
136  if (traderCanTrade && playerCanTrade)
137  {
138  // Ok, if trade item exists and can be stored
139  // If we trade in both directions we had to check, if the trade will work before we actually do it
140  // A roll back is not possible after we stored it
141  if (myItems[i])
142  {
143  // logging
144  DEBUG_LOG("partner storing: %s", myItems[i]->GetGuidStr().c_str());
145  if (_player->GetSession()->GetSecurity() > SEC_PLAYER && sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
146  {
147  sLog.outCommand(_player->GetSession()->GetAccountId(), "GM %s (Account: %u) trade: %s (Entry: %d Count: %u) to player: %s (Account: %u)",
148  _player->GetName(), _player->GetSession()->GetAccountId(),
149  myItems[i]->GetProto()->Name1, myItems[i]->GetEntry(), myItems[i]->GetCount(),
150  trader->GetName(), trader->GetSession()->GetAccountId());
151  }
152 
153  // store
154  trader->MoveItemToInventory(traderDst, myItems[i], true, true);
155  }
156 
157  if (hisItems[i])
158  {
159  // logging
160  DEBUG_LOG("player storing: %s", hisItems[i]->GetGuidStr().c_str());
161  if (trader->GetSession()->GetSecurity() > SEC_PLAYER && sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
162  {
163  sLog.outCommand(trader->GetSession()->GetAccountId(), "GM %s (Account: %u) trade: %s (Entry: %d Count: %u) to player: %s (Account: %u)",
164  trader->GetName(), trader->GetSession()->GetAccountId(),
165  hisItems[i]->GetProto()->Name1, hisItems[i]->GetEntry(), hisItems[i]->GetCount(),
166  _player->GetName(), _player->GetSession()->GetAccountId());
167  }
168 
169  // store
170  _player->MoveItemToInventory(playerDst, hisItems[i], true, true);
171  }
172  }
173  else
174  {
175  // in case of fatal error log error message
176  // return the already removed items to the original owner
177  if (myItems[i])
178  {
179  if (!traderCanTrade)
180  sLog.outError("trader can't store item: %s", myItems[i]->GetGuidStr().c_str());
181  if (_player->CanStoreItem(NULL_BAG, NULL_SLOT, playerDst, myItems[i], false) == EQUIP_ERR_OK)
182  _player->MoveItemToInventory(playerDst, myItems[i], true, true);
183  else
184  sLog.outError("player can't take item back: %s", myItems[i]->GetGuidStr().c_str());
185  }
186  // return the already removed items to the original owner
187  if (hisItems[i])
188  {
189  if (!playerCanTrade)
190  sLog.outError("player can't store item: %s", hisItems[i]->GetGuidStr().c_str());
191  if (trader->CanStoreItem(NULL_BAG, NULL_SLOT, traderDst, hisItems[i], false) == EQUIP_ERR_OK)
192  trader->MoveItemToInventory(traderDst, hisItems[i], true, true);
193  else
194  sLog.outError("trader can't take item back: %s", hisItems[i]->GetGuidStr().c_str());
195  }
196  }
197  }

201 static void setAcceptTradeMode(TradeData* myTrade, TradeData* hisTrade, Item** myItems, Item** hisItems)
202 {
203  myTrade->SetInAcceptProcess(true);
204  hisTrade->SetInAcceptProcess(true);
205 
206  // store items in local list and set 'in-trade' flag
207  for (int i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
208  {
209  if (Item* item = myTrade->GetItem(TradeSlots(i)))
210  {
211  DEBUG_LOG("player trade %s bag: %u slot: %u", item->GetGuidStr().c_str(), item->GetBagSlot(), item->GetSlot());
212  // Can return nullptr
213  myItems[i] = item;
214  myItems[i]->SetInTrade();
215  }
216 
217  if (Item* item = hisTrade->GetItem(TradeSlots(i)))
218  {
219  DEBUG_LOG("partner trade %s bag: %u slot: %u", item->GetGuidStr().c_str(), item->GetBagSlot(), item->GetSlot());
220  hisItems[i] = item;
221  hisItems[i]->SetInTrade();
222  }
223  }

244 void WorldSession::HandleAcceptTradeOpcode(WorldPacket& recvPacket)
245 {
246  recvPacket.read_skip<uint32>(); // 7, amount traded slots ?
247 
248  TradeData* my_trade = _player->m_trade;
249  if (!my_trade)
250  return;
251 
252  Player* trader = my_trade->GetTrader();
253 
254  TradeData* his_trade = trader->m_trade;
255  if (!his_trade)
256  return;
257 
258  Item* myItems[TRADE_SLOT_TRADED_COUNT] = { nullptr, nullptr, nullptr, nullptr, nullptr, nullptr };
259  Item* hisItems[TRADE_SLOT_TRADED_COUNT] = { nullptr, nullptr, nullptr, nullptr, nullptr, nullptr };
260 
261  // set before checks to properly undo at problems (it already set in to client)
262  my_trade->SetAccepted(true);
263 
264  TradeStatusInfo info;
265  if (!_player->IsWithinDistInMap(trader, TRADE_DISTANCE, false))
266  {
267  info.Status = TRADE_STATUS_TARGET_TO_FAR;
268  SendTradeStatus(info);
269  my_trade->SetAccepted(false);
270  return;
271  }
272 
273  // not accept case incorrect money amount
274  if (my_trade->GetMoney() > _player->GetMoney())
275  {
276  info.Status = TRADE_STATUS_CLOSE_WINDOW;
277  info.Result = EQUIP_ERR_NOT_ENOUGH_MONEY;
278  SendTradeStatus(info);
279  my_trade->SetAccepted(false, true);
280  return;
281  }
282 
283  // not accept case incorrect money amount
284  if (his_trade->GetMoney() > trader->GetMoney())
285  {
286  info.Status = TRADE_STATUS_CLOSE_WINDOW;
287  info.Result = EQUIP_ERR_NOT_ENOUGH_MONEY;
288  trader->GetSession()->SendTradeStatus(info);
289  his_trade->SetAccepted(false, true);
290  return;
291  }
292 
293  // not accept if some items now can't be trade (cheating)
294  for (int i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
295  {
296  if (Item* item = my_trade->GetItem(TradeSlots(i)))
297  {
298  if (!item->CanBeTraded())
299  {
300  info.Status = TRADE_STATUS_TRADE_CANCELED;
301  SendTradeStatus(info);
302  return;
303  }
304  }
305 
306  if (Item* item = his_trade->GetItem(TradeSlots(i)))
307  {
308  if (!item->CanBeTraded())
309  {
310  info.Status = TRADE_STATUS_TRADE_CANCELED;
311  SendTradeStatus(info);
312  return;
313  }
314  }
315  }
316 
317  if (his_trade->IsAccepted())
318  {
319  setAcceptTradeMode(my_trade, his_trade, myItems, hisItems);
320 
321  Spell* my_spell = nullptr;
322  SpellCastTargets my_targets;
323 
324  Spell* his_spell = nullptr;
325  SpellCastTargets his_targets;
326 
327  // not accept if spell can't be casted now (cheating)
328  if (uint32 my_spell_id = my_trade->GetSpell())
329  {
330  SpellEntry const* spellEntry = sSpellTemplate.LookupEntry<SpellEntry>(my_spell_id);
331  Item* castItem = my_trade->GetSpellCastItem();
332 
333  if (!spellEntry || !his_trade->GetItem(TRADE_SLOT_NONTRADED) ||
334  (my_trade->HasSpellCastItem() && !castItem))
335  {
336  clearAcceptTradeMode(my_trade, his_trade);
337  clearAcceptTradeMode(myItems, hisItems);
338 
339  my_trade->SetSpell(0);
340  return;
341  }
342 
343  my_spell = new Spell(_player, spellEntry, TRIGGERED_OLD_TRIGGERED);
344  my_spell->SetCastItem(castItem);
345  my_targets.setTradeItemTarget(_player);
346  my_spell->m_targets = my_targets;
347 
348  SpellCastResult res = my_spell->CheckCast(true);
349  if (res != SPELL_CAST_OK)
350  {
351  my_spell->SendCastResult(res);
352 
353  clearAcceptTradeMode(my_trade, his_trade);
354  clearAcceptTradeMode(myItems, hisItems);
355 
356  delete my_spell;
357  my_trade->SetSpell(0);
358  return;
359  }
360  }
361 
362  // not accept if spell can't be casted now (cheating)
363  if (uint32 his_spell_id = his_trade->GetSpell())
364  {
365  SpellEntry const* spellEntry = sSpellTemplate.LookupEntry<SpellEntry>(his_spell_id);
366  Item* castItem = his_trade->GetSpellCastItem();
367 
368  if (!spellEntry || !my_trade->GetItem(TRADE_SLOT_NONTRADED) ||
369  (his_trade->HasSpellCastItem() && !castItem))
370  {
371  delete my_spell;
372  his_trade->SetSpell(0);
373 
374  clearAcceptTradeMode(my_trade, his_trade);
375  clearAcceptTradeMode(myItems, hisItems);
376  return;
377  }
378 
379  his_spell = new Spell(trader, spellEntry, TRIGGERED_OLD_TRIGGERED);
380  his_spell->SetCastItem(castItem);
381  his_targets.setTradeItemTarget(trader);
382  his_spell->m_targets = his_targets;
383 
384  SpellCastResult res = his_spell->CheckCast(true);
385  if (res != SPELL_CAST_OK)
386  {
387  his_spell->SendCastResult(res);
388 
389  clearAcceptTradeMode(my_trade, his_trade);
390  clearAcceptTradeMode(myItems, hisItems);
391 
392  delete my_spell;
393  delete his_spell;
394 
395  his_trade->SetSpell(0);
396  return;
397  }
398  }
399 
400  // inform partner client
401  info.Status = TRADE_STATUS_TRADE_ACCEPT;
402  trader->GetSession()->SendTradeStatus(info);
403 
404  // test if item will fit in each inventory
405  TradeStatusInfo myCanCompleteInfo, hisCanCompleteInfo;
406  hisCanCompleteInfo.Result = trader->CanStoreItems(myItems, TRADE_SLOT_TRADED_COUNT, &hisCanCompleteInfo.ItemLimitCategoryId);
407  myCanCompleteInfo.Result = _player->CanStoreItems(hisItems, TRADE_SLOT_TRADED_COUNT, &myCanCompleteInfo.ItemLimitCategoryId);
408 
409  clearAcceptTradeMode(myItems, hisItems);
410 
411  // in case of missing space report error
412  if (myCanCompleteInfo.Result != EQUIP_ERR_OK)
413  {
414  clearAcceptTradeMode(my_trade, his_trade);
415 
416  myCanCompleteInfo.Status = TRADE_STATUS_CLOSE_WINDOW;
417  trader->GetSession()->SendTradeStatus(myCanCompleteInfo);
418  myCanCompleteInfo.IsTargetResult = true;
419  SendTradeStatus(myCanCompleteInfo);
420  my_trade->SetAccepted(false);
421  his_trade->SetAccepted(false);
422  return;
423  }
424  if (hisCanCompleteInfo.Result != EQUIP_ERR_OK)
425  {
426  clearAcceptTradeMode(my_trade, his_trade);
427 
428  hisCanCompleteInfo.Status = TRADE_STATUS_CLOSE_WINDOW;
429  SendTradeStatus(hisCanCompleteInfo);
430  hisCanCompleteInfo.IsTargetResult = true;
431  trader->GetSession()->SendTradeStatus(hisCanCompleteInfo);
432  my_trade->SetAccepted(false);
433  his_trade->SetAccepted(false);
434  return;
435  }
436 
437  // execute trade: 1. remove
438  for (int i = 0; i < TRADE_SLOT_TRADED_COUNT; ++i)
439  {
440  if (Item* item = myItems[i])
441  {
442  item->SetGuidValue(ITEM_FIELD_GIFTCREATOR, _player->GetObjectGuid());
443  _player->MoveItemFromInventory(item->GetBagSlot(), item->GetSlot(), true);
444  }
445  if (Item* item = hisItems[i])
446  {
447  item->SetGuidValue(ITEM_FIELD_GIFTCREATOR, trader->GetObjectGuid());
448  trader->MoveItemFromInventory(item->GetBagSlot(), item->GetSlot(), true);
449  }
450  }
451 
452  // execute trade: 2. store
453  moveItems(myItems, hisItems);
454 
455  // logging money
456  if (sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
457  {
458  if (_player->GetSession()->GetSecurity() > SEC_PLAYER && my_trade->GetMoney() > 0)
459  {
460  sLog.outCommand(_player->GetSession()->GetAccountId(), "GM %s (Account: %u) give money (Amount: %u) to player: %s (Account: %u)",
461  _player->GetName(), _player->GetSession()->GetAccountId(),
462  my_trade->GetMoney(),
463  trader->GetName(), trader->GetSession()->GetAccountId());
464  }
465  if (trader->GetSession()->GetSecurity() > SEC_PLAYER && his_trade->GetMoney() > 0)
466  {
467  sLog.outCommand(trader->GetSession()->GetAccountId(), "GM %s (Account: %u) give money (Amount: %u) to player: %s (Account: %u)",
468  trader->GetName(), trader->GetSession()->GetAccountId(),
469  his_trade->GetMoney(),
470  _player->GetName(), _player->GetSession()->GetAccountId());
471  }
472  }
473 
474  // update money
475  _player->ModifyMoney(-int32(my_trade->GetMoney()));
476  _player->ModifyMoney(his_trade->GetMoney());
477  trader->ModifyMoney(-int32(his_trade->GetMoney()));
478  trader->ModifyMoney(my_trade->GetMoney());
479 
480  if (my_spell)
481  my_spell->SpellStart(&my_targets);
482 
483  if (his_spell)
484  his_spell->SpellStart(&his_targets);
485 
486  // cleanup
487  clearAcceptTradeMode(my_trade, his_trade);
488  delete _player->m_trade;
489  _player->m_trade = nullptr;
490  delete trader->m_trade;
491  trader->m_trade = nullptr;
492 
493  // desynchronized with the other saves here (SaveInventoryAndGoldToDB() not have own transaction guards)
494  CharacterDatabase.BeginTransaction();
495  _player->SaveInventoryAndGoldToDB();
496  trader->SaveInventoryAndGoldToDB();
497  CharacterDatabase.CommitTransaction();
498 
499  info.Status = TRADE_STATUS_TRADE_COMPLETE;
500  trader->GetSession()->SendTradeStatus(info);
501  SendTradeStatus(info);
502  }
503  else
504  {
505  info.Status = TRADE_STATUS_TRADE_ACCEPT;
506  trader->GetSession()->SendTradeStatus(info);
507  }
508 }

548 void WorldSession::HandleInitiateTradeOpcode(WorldPacket& recvPacket)
549 {
550  ObjectGuid otherGuid;
551  recvPacket >> otherGuid;
552 
553  if (GetPlayer()->m_trade)
554  return;
555 
556  TradeStatusInfo info;
557  if (!GetPlayer()->IsAlive())
558  {
559  info.Status = TRADE_STATUS_YOU_DEAD;
560  SendTradeStatus(info);
561  return;
562  }
563 
564  if (GetPlayer()->IsStunned())
565  {
566  info.Status = TRADE_STATUS_YOU_STUNNED;
567  SendTradeStatus(info);
568  return;
569  }
570 
571  if (isLogingOut())
572  {
573  info.Status = TRADE_STATUS_YOU_LOGOUT;
574  SendTradeStatus(info);
575  return;
576  }
577 
578  if (GetPlayer()->IsTaxiFlying())
579  {
580  info.Status = TRADE_STATUS_TARGET_TO_FAR;
581  SendTradeStatus(info);
582  return;
583  }
584 
585  Player* pOther = ObjectAccessor::FindPlayer(otherGuid);
586 
587  if (!pOther)
588  {
589  info.Status = TRADE_STATUS_NO_TARGET;
590  SendTradeStatus(info);
591  return;
592  }
593 
594  if (pOther == GetPlayer() || pOther->m_trade)
595  {
596  info.Status = TRADE_STATUS_BUSY;
597  SendTradeStatus(info);
598  return;
599  }
600 
601  if (!pOther->IsAlive())
602  {
603  info.Status = TRADE_STATUS_TARGET_DEAD;
604  SendTradeStatus(info);
605  return;
606  }
607 
608  if (pOther->IsTaxiFlying())
609  {
610  info.Status = TRADE_STATUS_TARGET_TO_FAR;
611  SendTradeStatus(info);
612  return;
613  }
614 
615  if (pOther->IsStunned())
616  {
617  info.Status = TRADE_STATUS_TARGET_STUNNED;
618  SendTradeStatus(info);
619  return;
620  }
621 
622  if (pOther->GetSession()->isLogingOut())
623  {
624  info.Status = TRADE_STATUS_TARGET_LOGOUT;
625  SendTradeStatus(info);
626  return;
