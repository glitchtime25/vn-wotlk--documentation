195  {
196  switch (pSummoned->GetEntry())

199  {
200  int32 uiSeat = (int32)SEAT_ID_RIGHT;

202  pSummoned->CastSpell(pSummoned, SPELL_ARM_VISUAL, TRIGGERED_OLD_TRIGGERED);
203 
204  if (m_creature->GetVictim())
205  pSummoned->AI()->AttackStart(m_creature->GetVictim());
206  break;
207  }
208  case NPC_LEFT_ARM:
209  {
210  int32 uiSeat = (int32)SEAT_ID_LEFT;
211  pSummoned->CastCustomSpell(m_creature, SPELL_RIDE_KOLOGARN_ARMS, &uiSeat, nullptr, nullptr, TRIGGERED_OLD_TRIGGERED);
212  pSummoned->CastSpell(pSummoned, SPELL_ARM_VISUAL, TRIGGERED_OLD_TRIGGERED);
213 
214  if (m_creature->GetVictim())
215  pSummoned->AI()->AttackStart(m_creature->GetVictim());
216  break;
217  }
218  case NPC_FOCUSED_EYEBEAM_RIGHT:
219  case NPC_FOCUSED_EYEBEAM_LEFT:
220  // force despawn - if the npc gets in combat it won't despawn automatically
221  pSummoned->ForcedDespawn(10000);
222 
223  // cast visuals and damage spell
224  pSummoned->CastSpell(m_creature, pSummoned->GetEntry() == NPC_FOCUSED_EYEBEAM_LEFT ? SPELL_EYEBEAM_VISUAL_LEFT : SPELL_EYEBEAM_VISUAL_RIGHT, TRIGGERED_OLD_TRIGGERED);
225  pSummoned->CastSpell(pSummoned, m_bIsRegularMode ? SPELL_EYEBEAM_PERIODIC : SPELL_EYEBEAM_PERIODIC_H, TRIGGERED_OLD_TRIGGERED);
226 
227  // follow the summoner
228  if (pSummoned->IsTemporarySummon())
229  {
230  if (Unit* pPlayer = m_creature->GetMap()->GetUnit(pSummoned->GetSpawnerGuid()))
231  pSummoned->GetMotionMaster()->MoveChase(pPlayer);
232  }
233  break;
234  }

313  void UpdateAI(const uint32 uiDiff) override
314  {
315  if (m_uiMountArmsTimer)
316  {
317  if (m_uiMountArmsTimer <= uiDiff)
318  {
319  m_creature->SummonCreature(NPC_RIGHT_ARM, afKoloArmsLoc[0], afKoloArmsLoc[1], afKoloArmsLoc[2], afKoloArmsLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
320  m_creature->SummonCreature(NPC_LEFT_ARM, afKoloArmsLoc[0], afKoloArmsLoc[1], afKoloArmsLoc[2], afKoloArmsLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
321  m_uiMountArmsTimer = 0;
322  }
323  else
324  m_uiMountArmsTimer -= uiDiff;
325  }
326 
327  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
328  return;
329 
330  if (m_uiRespawnLeftTimer && m_uiRespawnRightTimer)
331  {
332  if (m_uiStoneShoutTimer < uiDiff)
333  {
334  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_STONE_SHOUT : SPELL_STONE_SHOUT_H) == CAST_OK)
335  m_uiStoneShoutTimer = urand(3000, 4000);
336  }
337  else
338  m_uiStoneShoutTimer -= uiDiff;
339  }
340  else
341  {
342  if (m_uiOverheadSmashTimer < uiDiff)
343  {
344  CanCastResult castResult;
345  if (!m_uiRespawnLeftTimer && !m_uiRespawnRightTimer)
346  castResult = DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_OVERHEAD_SMASH : SPELL_OVERHEAD_SMASH_H);
347  else
348  castResult = DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_ONE_ARMED_SMASH : SPELL_ONE_ARMED_SMASH_H);
349 
350  if (castResult == CAST_OK)
351  m_uiOverheadSmashTimer = 15000;
352  }
353  else
354  m_uiOverheadSmashTimer -= uiDiff;
355  }
356 
357  if (m_uiEyebeamTimer < uiDiff)
358  {
359  if (DoCastSpellIfCan(m_creature, SPELL_FOCUSED_EYEBEAM_SUMMON) == CAST_OK)
360  m_uiEyebeamTimer = 20000;
361  }
362  else
363  m_uiEyebeamTimer -= uiDiff;
364 
365  // respawn left arm if killed
366  if (m_uiRespawnLeftTimer)
367  {
368  if (m_uiRespawnLeftTimer <= uiDiff)
369  {
370  DoScriptText(EMOTE_ARM_LEFT, m_creature);
371  m_creature->SummonCreature(NPC_LEFT_ARM, afKoloArmsLoc[0], afKoloArmsLoc[1], afKoloArmsLoc[2], afKoloArmsLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
372  m_uiRespawnLeftTimer = 0;
373  }
374  else
375  m_uiRespawnLeftTimer -= uiDiff;
376  }
377  // use left arm ability if available - spell always cast by Kologarn
378  else
379  {
380  if (m_uiShockwaveTimer < uiDiff)
381  {
382  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ARM_SWEEP : SPELL_ARM_SWEEP_H) == CAST_OK)
383  {
384  DoScriptText(SAY_SHOCKWAVE, m_creature);
385  m_uiShockwaveTimer = 17000;
386  }
387  }
388  else
389  m_uiShockwaveTimer -= uiDiff;
390  }
391 
392  // respawn right arm if killed
393  if (m_uiRespawnRightTimer)
394  {
395  if (m_uiRespawnRightTimer <= uiDiff)
396  {
397  DoScriptText(EMOTE_ARM_RIGHT, m_creature);
398  m_creature->SummonCreature(NPC_RIGHT_ARM, afKoloArmsLoc[0], afKoloArmsLoc[1], afKoloArmsLoc[2], afKoloArmsLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
399  m_uiRespawnRightTimer = 0;
400  }
401  else
402  m_uiRespawnRightTimer -= uiDiff;
403  }
404  // use right arm ability if available - spell always cast by Kologarn
405  else
406  {
407  if (m_uiStoneGripTimer < uiDiff)
408  {
409  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_STONE_GRIP : SPELL_STONE_GRIP_H) == CAST_OK)
410  {
411  DoScriptText(SAY_GRAB, m_creature);
412  DoScriptText(EMOTE_STONE_GRIP, m_creature);
413  m_uiStoneGripTimer = urand(20000, 30000);
414  }
415  }
416  else
417  m_uiStoneGripTimer -= uiDiff;
418  }
419 
420  if (m_uiBerserkTimer)
421  {
422  if (m_uiBerserkTimer <= uiDiff)
423  {
424  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
425  {
426  if (m_pInstance)

428  if (Creature* pRightArm = m_pInstance->GetSingleCreatureFromStorage(NPC_RIGHT_ARM))
429  pRightArm->CastSpell(pRightArm, SPELL_BERSERK, TRIGGERED_OLD_TRIGGERED);
430  if (Creature* pLeftArm = m_pInstance->GetSingleCreatureFromStorage(NPC_LEFT_ARM))
431  pLeftArm->CastSpell(pLeftArm, SPELL_BERSERK, TRIGGERED_OLD_TRIGGERED);
432  }
433 
434  DoScriptText(SAY_BERSERK, m_creature);
435  m_uiBerserkTimer = 0;
436  }
437  }
438  else
439  m_uiBerserkTimer -= uiDiff;
440  }
441 
442  if (m_uiDisarmedTimer)
443  {
444  if (m_uiDisarmedTimer <= uiDiff)
445  {
446  if (m_pInstance)
447  m_pInstance->SetSpecialAchievementCriteria(TYPE_ACHIEV_DISARMED, false);
448  m_uiDisarmedTimer = 0;
449  }
450  else
451  m_uiDisarmedTimer -= uiDiff;
452  }
453 
454  // melee range check
455  if (!m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
456  {
457  if (m_uiPetrifyingBreathTimer < uiDiff)
458  {
459  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_PETRIFYING_BREATH : SPELL_PETRIFYING_BREATH_H) == CAST_OK)
460  m_uiPetrifyingBreathTimer = 4000;
461  }
462  else
463  m_uiPetrifyingBreathTimer -= uiDiff;
464  }
465  else
466  DoMeleeAttackIfReady();
467  }
