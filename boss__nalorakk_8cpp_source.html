161  void ExecuteAction(uint32 action) override
162  {
163  switch (action)
164  {
165  case NALORAKK_ACTION_BERSERK:
166  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
167  {
168  DoScriptText(SAY_BERSERK, m_creature);
169  DisableCombatAction(action);
170  }
171  return;
172  case NALORAKK_ACTION_BEAR_SHAPE:
173  if (m_bIsInBearForm)
174  {
175  DoScriptText(SAY_TOTROLL, m_creature);
176  ResetCombatAction(action, 45000);
177  m_bIsInBearForm = false;
178  // Reset troll form timers
179  DisableCombatAction(NALORAKK_ACTION_SLASH);
180  DisableCombatAction(NALORAKK_ACTION_REND);
181  DisableCombatAction(NALORAKK_ACTION_ROAR);
182  ResetCombatAction(NALORAKK_ACTION_SURGE, urand(15000, 32000));
183  ResetCombatAction(NALORAKK_ACTION_SWIPE, urand(7000, 20000));
184  ResetCombatAction(NALORAKK_ACTION_MANGLE, urand(3000, 20000));
185  SetEquipmentSlots(true);
186  }
187  else
188  {
189  if (DoCastSpellIfCan(nullptr, SPELL_BEAR_SHAPE, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
190  {
191  DoScriptText(SAY_TOBEAR_EMOTE, m_creature);
192  DoScriptText(SAY_TOBEAR, m_creature);
193  ResetCombatAction(action, 30000);
194  m_bIsInBearForm = true;
195  // Reset bear form timers
196  DisableCombatAction(NALORAKK_ACTION_SWIPE);
197  DisableCombatAction(NALORAKK_ACTION_MANGLE);
198  DisableCombatAction(NALORAKK_ACTION_SURGE);
199  ResetCombatAction(NALORAKK_ACTION_SLASH, urand(5000, 8000));
200  ResetCombatAction(NALORAKK_ACTION_REND, urand(6000, 10000));
201  ResetCombatAction(NALORAKK_ACTION_ROAR, urand(15000, 25000));
202  SetEquipmentSlots(false, 0, 0, 0);
203  m_creature->resetAttackTimer(BASE_ATTACK);
204  }
205  }
206  return;
207  case NALORAKK_ACTION_SWIPE:
208  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BRUTAL_SWIPE) == CAST_OK)
209  ResetCombatAction(action, urand(7000, 15000));
210  return;
211  case NALORAKK_ACTION_MANGLE:
212  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MANGLE) == CAST_OK)
213  ResetCombatAction(action, urand(3000, 15000));
214  return;
215  case NALORAKK_ACTION_SURGE:
216  {
217  // select a random unit other than the main tank
218  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IN_MELEE_RANGE);
219 
220  // if there aren't other units, cast on random
221  if (!target)
222  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER);
223 
224  if (DoCastSpellIfCan(target, SPELL_SURGE) == CAST_OK)
225  {
226  DoScriptText(SAY_SURGE, m_creature);
227  ResetCombatAction(action, urand(15000, 32500));
228  }
229  return;
230  }
231  case NALORAKK_ACTION_SLASH:
232  {
233  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_LACERATING_SLASH) == CAST_OK)
234  ResetCombatAction(action, 20000);
235  return;
236  }
237  case NALORAKK_ACTION_REND:
238  {
239  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_REND_FLESH) == CAST_OK)
240  ResetCombatAction(action, urand(6000, 10000));
241  return;
242  }
243  case NALORAKK_ACTION_ROAR:
244  {
245  if (DoCastSpellIfCan(m_creature, SPELL_DEAFENING_ROAR) == CAST_OK)
246  ResetCombatAction(action, urand(15000, 20000));
247  return;
248  }
249  }
250  }
