102 void instance_draktharon_keep::DoSortNovosDummies()
103 {
104  // Sorting once is good enough
105  if (m_lNovosDummyGuids.empty())
106  return;
107 
108  Creature* pNovos = GetSingleCreatureFromStorage(NPC_NOVOS);
109  if (!pNovos)
110  return;
111 
112  // First sort the Dummies to the Crystals
113  for (auto& i : m_aNovosCrystalInfo)
114  {
115  GameObject* pCrystal = instance->GetGameObject(i.m_crystalGuid);
116  if (!pCrystal)
117  continue;
118 
119  for (GuidList::iterator itr = m_lNovosDummyGuids.begin(); itr != m_lNovosDummyGuids.end();)
120  {
121  Creature* pDummy = instance->GetCreature(*itr);
122  if (!pDummy)
123  {
124  m_lNovosDummyGuids.erase(itr++);
125  continue;
126  }
127 
128  // Check if dummy fits to crystal
129  if (pCrystal->IsWithinDistInMap(pDummy, INTERACTION_DISTANCE, false))
130  {
131  i.m_channelGuid = pDummy->GetObjectGuid();
132  m_lNovosDummyGuids.erase(itr);
133  break;
134  }
135 
136  ++itr;
137  }
138  }
139 
140  // Find the crystal channel target (above Novos)
141  float fNovosX, fNovosY, fNovosZ;
142  pNovos->GetRespawnCoord(fNovosX, fNovosY, fNovosZ);
143  for (GuidList::iterator itr = m_lNovosDummyGuids.begin(); itr != m_lNovosDummyGuids.end();)
144  {
145  Creature* pDummy = instance->GetCreature(*itr);
146  if (!pDummy)
147  {
148  m_lNovosDummyGuids.erase(itr++);
149  continue;
150  }
151 
152  // As the wanted dummy is exactly above Novos, check small range, and only 2d
153  if (pDummy->IsWithinDist2d(fNovosX, fNovosY, 5.0f))
154  {
155  m_novosChannelGuid = pDummy->GetObjectGuid();
156  m_lNovosDummyGuids.erase(itr);
157  break;
158  }
159 
160  ++itr;
161  }
162 
163  // Summon positions (at end of stairs)
164  for (GuidList::iterator itr = m_lNovosDummyGuids.begin(); itr != m_lNovosDummyGuids.end();)
165  {
166  Creature* pDummy = instance->GetCreature(*itr);
167  if (!pDummy)
168  {
169  m_lNovosDummyGuids.erase(itr++);
170  continue;
171  }
172 
173  // The wanted dummies are quite above Novos
174  if (pDummy->GetPositionZ() > fNovosZ + 20.0f)
175  {
176  m_vSummonDummyGuids.push_back(pDummy->GetObjectGuid());
177  m_lNovosDummyGuids.erase(itr++);
178  }
179  else
180  ++itr;
181  }
182 
183  // Clear remaining (unused) dummies
184  m_lNovosDummyGuids.clear();
185 }

187 Creature* instance_draktharon_keep::GetNextCrystalTarget(Creature* pCrystalHandler, uint8& uiIndex) const
188 {
189  Creature* pTarget = nullptr;
190  uiIndex = 0;
191 
192  for (uint8 i = 0; i < MAX_CRYSTALS; ++i)
193  {
194  Creature* pDummy = instance->GetCreature(m_aNovosCrystalInfo[i].m_channelGuid);
195  // Return the nearest 'unused' crystal dummy
196  // unused means, that the crystal was not already used, and the dummy-npc doesn't have the aura that will trigger the use on remove
197  if (pDummy && !m_aNovosCrystalInfo[i].m_bWasUsed && (!pTarget || pCrystalHandler->GetDistanceOrder(pDummy, pTarget)) && !pDummy->HasAura(aCrystalHandlerDeathSpells[i]))
198  {
199  pTarget = pDummy;
200  uiIndex = i;
201  }
202  }
203 
204  return pTarget;
205 }

237 void instance_draktharon_keep::SetData(uint32 uiType, uint32 uiData)
238 {
239  switch (uiType)
240  {
241  case TYPE_TROLLGORE:
242  if (uiData == IN_PROGRESS)
243  m_bTrollgoreConsume = true;
244  if (uiData == SPECIAL)
245  m_bTrollgoreConsume = false;
246  m_auiEncounter[uiType] = uiData;
247  break;
248  case TYPE_NOVOS:
249  if (uiData == IN_PROGRESS)
250  {
251  // Sort the dummies
252  DoSortNovosDummies();
253 
254  // Cast some visual spells
255  Creature* pTarget = instance->GetCreature(m_novosChannelGuid);
256  for (auto& i : m_aNovosCrystalInfo)
257  {
258  Creature* pCaster = instance->GetCreature(i.m_channelGuid);
259  if (pCaster && pTarget)
260  pCaster->CastSpell(pTarget, SPELL_BEAM_CHANNEL, TRIGGERED_NONE);
261 
262  i.m_bWasUsed = false;
263  }
264 
265  // Achievement related
266  m_bNovosAddGrounded = false;
267  }
268  else if (uiData == SPECIAL)
269  {
270  // Achievement related
271  m_bNovosAddGrounded = true;
272  }
273  else if (uiData == FAIL)
274  {
275  // Interrupt casted spells
276  for (auto& i : m_aNovosCrystalInfo)
277  {
278  Creature* pDummy = instance->GetCreature(i.m_channelGuid);
279  if (pDummy)
280  pDummy->InterruptNonMeleeSpells(false);
281  // And reset used crystals
282  if (i.m_bWasUsed)
283  DoUseDoorOrButton(i.m_crystalGuid);
284  }
285  }
286  m_auiEncounter[uiType] = uiData;
287  break;
288  case TYPE_KING_DRED:
289  if (uiData == IN_PROGRESS)
290  m_uiDreadAddsKilled = 0;
291  m_auiEncounter[uiType] = uiData;
292  break;
293  case TYPE_THARONJA:
294  m_auiEncounter[uiType] = uiData;
295  break;
296  }
297 
298  if (uiData == DONE)
299  {
300  OUT_SAVE_INST_DATA;
301 
302  std::ostringstream saveStream;
303  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
304 
305  m_strInstData = saveStream.str();
306 
307  SaveToDB();
308  OUT_SAVE_INST_DATA_COMPLETE;
309  }
310 }
