162  void DoSpawnGuests()
163  {
164  // not if m_creature are dead, so avoid
165  if (!m_creature->IsAlive())
166  return;
167 
168  // it's empty, so first time
169  if (m_vGuestsEntryList.empty())
170  {
171  // pre-allocate size for speed
172  m_vGuestsEntryList.resize(MAX_GUESTS);
173 
174  // fill vector array with entries from creature array
175  for (uint8 i = 0; i < MAX_GUESTS; ++i)
176  m_vGuestsEntryList[i] = guests[i];
177 
178  std::shuffle(m_vGuestsEntryList.begin(), m_vGuestsEntryList.end(), *GetRandomGenerator());
179 
180  // Summon the 4 entries
181  for (uint8 i = 0; i < MAX_ACTIVE_GUESTS; ++i)
182  m_creature->SummonCreature(m_vGuestsEntryList[i], guestLocations[i][0], guestLocations[i][1], guestLocations[i][2], guestLocations[i][3], TEMPSPAWN_CORPSE_DESPAWN, 0);
183  }
184  // Resummon the killed adds
185  else
186  {
187  if (!m_instance)
188  return;
189 
190  for (uint8 i = 0; i < MAX_ACTIVE_GUESTS; ++i)
191  {
192  // If we already have the creature on the map, then don't summon it
193  if (Creature* add = m_instance->GetSingleCreatureFromStorage(m_vGuestsEntryList[i], true))
194  {
195  if (add->IsInCombat())
196  add->AI()->EnterEvadeMode();
197  continue;
198  }
199 
200  m_creature->SummonCreature(m_vGuestsEntryList[i], guestLocations[i][0], guestLocations[i][1], guestLocations[i][2], guestLocations[i][3], TEMPSPAWN_CORPSE_DESPAWN, 0);
201  }
202  }
203  }

205  // Wrapper to remove the Garrote aura on death and on evade - ToDo: maybe find a better way for this!
206  void DoRemoveGarroteAura()
207  {
208  // remove aura from spell Garrote when Moroes dies
209  Map* pMap = m_creature->GetMap();
210  if (pMap->IsDungeon())
211  {
212  Map::PlayerList const& PlayerList = pMap->GetPlayers();
213 
214  if (PlayerList.isEmpty())
215  return;
216 
217  for (const auto& i : PlayerList)
218  {
219  if (i.getSource()->IsAlive() && i.getSource()->HasAura(SPELL_GARROTE))
220  i.getSource()->RemoveAurasDueToSpell(SPELL_GARROTE);
221  }
222  }

225  void ExecuteAction(uint32 action) override
226  {
227  switch (action)
228  {
229  case MOROES_ACTION_VANISH:
230  {
231  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_GARROTE, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA);
232  if (!target) // if no target without garrote found - select any random
233  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER);
234  if (!target)
235  break;
236  target->CastSpell(nullptr, SPELL_TAUNT, TRIGGERED_OLD_TRIGGERED); // TODO: Needs to send both packets
237  m_creature->SelectHostileTarget(); // apply taunt before vanish
238  m_creature->RemoveSpellsCausingAura(SPELL_AURA_MOD_STALKED); // hunters mark
239  DoCastSpellIfCan(nullptr, SPELL_VANISH);
240  SetCombatScriptStatus(true);
241  SetMeleeEnabled(false);
242  m_attackAngle = M_PI_F;
243  DoStartMovement(m_creature->GetVictim());
244  ResetCombatAction(action, GetSubsequentActionTimer(action));
245  ResetTimer(MOROES_ACTION_GAROTTE, 9500);
246  break;
247  }
248  case MOROES_ACTION_BLIND:
249  {
250  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, SPELL_GARROTE, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA | SELECT_FLAG_SKIP_TANK);
251  if (!target) // if no target without garrote found - select any random
252  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER);
253  if (!target)
254  break;
255 
256  if (DoCastSpellIfCan(target, SPELL_BLIND) == CAST_OK)
257  ResetCombatAction(action, GetSubsequentActionTimer(action));
258  break;
259  }
260  case MOROES_ACTION_GOUGE:
261  {
262  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_GOUGE) == CAST_OK)
263  ResetCombatAction(action, GetSubsequentActionTimer(action));
264  break;
265  }
266  case MOROES_ACTION_ENRAGE:
267  {
268  if (m_creature->GetHealthPercent() < 30.0f)
269  {
270  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
271  {
272  DoScriptText(SAY_FRENZY, m_creature);
273  SetActionReadyStatus(action, false);
274  }
275  }
276  break;
277  }
278  }
279  }
