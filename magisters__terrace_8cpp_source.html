101 void instance_magisters_terrace::SetData(uint32 uiType, uint32 uiData)
102 {
103  switch (uiType)
104  {
105  case TYPE_SELIN:
106  if (uiData == DONE)
107  DoUseDoorOrButton(GO_SELIN_DOOR);
108  if (uiData == FAIL)
109  {
110  // Reset crystals - respawn and kill is handled by creature linking
111  for (GuidList::const_iterator itr = m_lFelCrystalGuid.begin(); itr != m_lFelCrystalGuid.end(); ++itr)
112  {
113  if (Creature* pTemp = instance->GetCreature(*itr))
114  {
115  if (!pTemp->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE))
116  pTemp->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
117 
118  if (pTemp->IsAlive())
119  {
120  if (pTemp->IsInCombat())
121  {
122  pTemp->ForcedDespawn();
123  pTemp->Respawn();
124  }
125  else
126  pTemp->CastSpell(nullptr, SPELL_FEL_CRYSTAL_VISUAL, TRIGGERED_NONE);
127  }
128  else
129  pTemp->Respawn();
130  }
131  }
132  }
133  if (uiData == IN_PROGRESS)
134  {
135  // Stop channeling when the fight starts
136  for (GuidList::const_iterator itr = m_lFelCrystalGuid.begin(); itr != m_lFelCrystalGuid.end(); ++itr)
137  {
138  if (Creature* pTemp = instance->GetCreature(*itr))
139  pTemp->InterruptNonMeleeSpells(false);
140  }
141  }
142  DoUseDoorOrButton(GO_SELIN_ENCOUNTER_DOOR);
143  m_auiEncounter[uiType] = uiData;
144  break;
145  case TYPE_VEXALLUS:
146  if (uiData == DONE)
147  DoUseDoorOrButton(GO_VEXALLUS_DOOR);
148  m_auiEncounter[uiType] = uiData;
149  break;
150  case TYPE_DELRISSA:
151  if (uiData == DONE)
152  DoUseDoorOrButton(GO_DELRISSA_DOOR);
153  m_auiEncounter[uiType] = uiData;
154  break;
155  case TYPE_KAELTHAS:
156  DoUseDoorOrButton(GO_KAEL_DOOR);
157  if (uiData == DONE)
158  DoToggleGameObjectFlags(GO_ESCAPE_QUEL_DANAS, GO_FLAG_NO_INTERACT, false);
159  m_auiEncounter[uiType] = uiData;
160  break;
161  }
162 
163  if (uiData == DONE)
164  {
165  OUT_SAVE_INST_DATA;
166 
167  std::ostringstream saveStream;
168  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
169 
170  m_strInstData = saveStream.str();
171 
172  SaveToDB();
173  OUT_SAVE_INST_DATA_COMPLETE;
174  }
175 }
