74 void MassMailMgr::Update(bool sendall /*= false*/)
75 {
76  if (m_massMails.empty())
77  return;
78 
79  uint32 maxcount = sWorld.getConfig(CONFIG_UINT32_MASS_MAILER_SEND_PER_TICK);
80 
81  do
82  {
83  MassMail& task = m_massMails.front();
84 
85  while (!task.m_receivers.empty() && (sendall || maxcount > 0))
86  {
87  uint32 receiver_lowguid = *task.m_receivers.begin();
88  task.m_receivers.erase(task.m_receivers.begin());
89 
90  ObjectGuid receiver_guid = ObjectGuid(HIGHGUID_PLAYER, receiver_lowguid);
91  Player* receiver = sObjectMgr.GetPlayer(receiver_guid);
92 
93  // last case. can be just send
94  if (task.m_receivers.empty())
95  {
96  // prevent mail return
97  task.m_protoMail->SendMailTo(MailReceiver(receiver, receiver_guid), task.m_sender, MAIL_CHECK_MASK_RETURNED);
98 
99  if (!sendall)
100  --maxcount;
101  break;
102  }
103 
104  // need clone draft
105  MailDraft draft;
106  draft.CloneFrom(*task.m_protoMail);
107 
108  // prevent mail return
109  draft.SendMailTo(MailReceiver(receiver, receiver_guid), task.m_sender, MAIL_CHECK_MASK_RETURNED);
110 
111  if (!sendall)
112  --maxcount;
113  }
114 
115  if (task.m_receivers.empty())
116  m_massMails.pop_front();
117  }
118  while (!m_massMails.empty() && (sendall || maxcount > 0));
119 }
