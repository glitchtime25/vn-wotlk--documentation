97  void UpdateAI(const uint32 diff) override
98  {
99  if (m_bEvent)
100  {
101  if (m_bIsMovingToLunch)
102  return;
103 
104  if (m_uiEventTimer < diff)
105  {
106  m_uiEventTimer = 5000;
107  ++m_uiEventPhase;
108 
109  switch (m_uiEventPhase)
110  {
111  case 1:
112  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid))
113  {
114  GameObject* pGo = pPlayer->GetGameObject(SPELL_LUNCH);
115 
116  // Workaround for broken function GetGameObject
117  if (!pGo)
118  {
119  const SpellEntry* pSpell = GetSpellStore()->LookupEntry<SpellEntry>(SPELL_LUNCH);
120 
121  uint32 uiGameobjectEntry = pSpell->EffectMiscValue[EFFECT_INDEX_1];
122 
123  pGo = GetClosestGameObjectWithEntry(pPlayer, uiGameobjectEntry, 2 * INTERACTION_DISTANCE);
124  }
125 
126  if (pGo)
127  {
128  m_bIsMovingToLunch = true;
129 
130  float fX, fY, fZ;
131  pGo->GetContactPoint(m_creature, fX, fY, fZ, CONTACT_DISTANCE);
132 
133  m_creature->GetMotionMaster()->MovePoint(POINT_ID, fX, fY, fZ);
134  }
135  }
136  break;
137  case 2:
138  DoScriptText(EMOTE_EAT_LUNCH, m_creature);
139  m_creature->HandleEmote(EMOTE_STATE_USESTANDING);
140  break;
141  case 3:
142  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid))
143  pPlayer->TalkedToCreature(m_creature->GetEntry(), m_creature->GetObjectGuid());
144 
145  m_creature->UpdateEntry(NPC_KYLE_FRIENDLY);
146  break;
147  case 4:
148  m_uiEventTimer = 30000;
149  DoScriptText(EMOTE_DANCE, m_creature);
150  m_creature->HandleEmote(EMOTE_STATE_DANCESPECIAL);
151  break;
152  case 5:
153  m_creature->HandleEmote(EMOTE_STATE_NONE);
154  Reset();
155  m_creature->GetMotionMaster()->Clear();
156  break;
157  }
158  }
159  else
160  m_uiEventTimer -= diff;
161  }
162  }
