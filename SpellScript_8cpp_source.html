66 void SpellScriptMgr::LoadScripts()
67 {
68  // load all scripts with names
69  LoadDeathKnightScripts();
70  LoadDruidScripts();
71  LoadHunterScripts();
72  LoadMageScripts();
73  LoadPaladinScripts();
74  LoadPriestScripts();
75  LoadRogueScripts();
76  LoadShamanScripts();
77  LoadWarlockScripts();
78  LoadWarriorScripts();
79  LoadScalingScripts();
80 
81  // load names from DB and pair names to Ids
82  std::unique_ptr<QueryResult> result(WorldDatabase.Query("SELECT Id, ScriptName FROM spell_scripts"));
83  if (result)
84  {
85  do
86  {
87  Field* fields = result->Fetch();
88  uint32 spellId = fields[0].GetUInt32();
89  std::string scriptName = fields[1].GetCppString();
90 
91  if (!sSpellTemplate.LookupEntry<SpellEntry>(spellId))
92  {
93  sLog.outErrorDb("Spell %u has script %s in spell_scripts but spell does not exist. Skipping.", spellId, scriptName.data());
94  continue;
95  }
96 
97  bool found = false;
98  if (SpellScript* script = GetSpellScript(scriptName))
99  SetSpellScript(spellId, script), found = true;
100  if (AuraScript* script = GetAuraScript(scriptName))
101  SetAuraScript(spellId, script), found = true;
102 
103  if (!found)
104  {
105  sLog.outErrorDb("Spell %u has script %s in spell_scripts but script does not exist. Skipping.", spellId, scriptName.data());
106  continue;
107  }
108  } while (result->NextRow());
109  }
110 }
