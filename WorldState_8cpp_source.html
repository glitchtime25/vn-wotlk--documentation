129 void WorldState::Load()
130 {
131  std::unique_ptr<QueryResult> result(CharacterDatabase.Query("SELECT Id, Data FROM world_state"));
132 
133  if (result)
134  {
135  // always one row
136  do
137  {
138  Field* fields = result->Fetch();
139  uint32 id = fields[0].GetUInt32();
140  std::string data = fields[1].GetCppString();
141  std::istringstream loadStream(data);
142  switch (id)
143  {
144  case SAVE_ID_EMERALD_DRAGONS:
145  {
146  auto curTime = World::GetCurrentClockTime();
147  uint64 respawnTime;
148  if (data.size())
149  {
150  loadStream >> m_emeraldDragonsState >> respawnTime;
151  for (uint32 i = 0; i < 4; ++i)
152  loadStream >> m_emeraldDragonsChosenPositions[i];
153  if (respawnTime)
154  {
155  TimePoint respawnTimePoint = std::chrono::time_point_cast<std::chrono::milliseconds>(Clock::from_time_t(respawnTime));
156  m_emeraldDragonsTimer = std::chrono::duration_cast<std::chrono::milliseconds>(respawnTimePoint - curTime).count();
157  }
158  else m_emeraldDragonsTimer = 0;
159  }
160  break;
161  }
162  case SAVE_ID_AHN_QIRAJ:
163  {
164  if (data.size())
165  {
166  auto curTime = World::GetCurrentClockTime();
167  uint64 time;
168  try
169  {
170  loadStream >> m_aqData.m_phase >> time;
171  if (time)
172  {
173  TimePoint timePoint = std::chrono::time_point_cast<std::chrono::milliseconds>(Clock::from_time_t(time));
174  m_aqData.m_timer = std::chrono::duration_cast<std::chrono::milliseconds>(timePoint - curTime).count();
175  }
176  for (uint32 i = 0; i < RESOURCE_MAX; ++i)
177  loadStream >> m_aqData.m_WarEffortCounters[i];
178  loadStream >> m_aqData.m_phase2Tier >> m_aqData.m_killedBosses;
179  }
180  catch (std::exception& e)
181  {
182  sLog.outError("%s", e.what());
183  memset(m_loveIsInTheAirData.counters, 0, sizeof(LoveIsInTheAir));
184  }
185  }
186  break;
187  }
188  case SAVE_ID_QUEL_DANAS:
189  {
190  if (data.size())
191  {
192  try
193  {
194  loadStream >> m_sunsReachData.m_phase >> m_sunsReachData.m_subphaseMask;
195  for (uint32 i = 0; i < COUNTERS_MAX; ++i)
196  loadStream >> m_sunsReachData.m_sunsReachReclamationCounters[i];
197  loadStream >> m_sunsReachData.m_gate;
198  for (uint32 i = 0; i < COUNTERS_MAX_GATES; ++i)
199  loadStream >> m_sunsReachData.m_gateCounters[i];
200  }
201  catch (std::exception& e)
202  {
203  sLog.outError("%s", e.what());
204  m_sunsReachData.m_phase = 0;
205  m_sunsReachData.m_subphaseMask = 0;
206  m_sunsReachData.m_gate = SUNWELL_ARCHONISUS_GATE3_OPEN;
207  memset(m_sunsReachData.m_sunsReachReclamationCounters, 0, sizeof(m_sunsReachData.m_sunsReachReclamationCounters));
208  memset(m_sunsReachData.m_gateCounters, 0, sizeof(m_sunsReachData.m_gateCounters));
209  }
210  }
211  else
212  {
213  m_sunsReachData.m_phase = 0;
214  m_sunsReachData.m_subphaseMask = 0;
215  m_sunsReachData.m_gate = SUNWELL_ARCHONISUS_GATE3_OPEN;
216  memset(m_sunsReachData.m_sunsReachReclamationCounters, 0, sizeof(m_sunsReachData.m_sunsReachReclamationCounters));
217  memset(m_sunsReachData.m_gateCounters, 0, sizeof(m_sunsReachData.m_gateCounters));
218  }
219  break;
220  }
221  case SAVE_ID_HIGHLORD_KRUUL:
222  {
223  auto curTime = World::GetCurrentClockTime();
224  uint64 respawnTime;
225  if (data.size())
226  {
227  loadStream >> m_highlordKruulTimer >> respawnTime;
228  if (respawnTime)
229  {
230  TimePoint respawnTimePoint = std::chrono::time_point_cast<std::chrono::milliseconds>(Clock::from_time_t(respawnTime));
231  m_highlordKruulTimer = std::chrono::duration_cast<std::chrono::milliseconds>(respawnTimePoint - curTime).count();
232  m_highlordKruulSpawned = true;
233  }
234  else m_highlordKruulTimer = 0;
235  }
236  else
237  {
238  m_highlordKruulSpawned = false;
239  m_highlordKruulTimer = 0;
240  }
241  break;
242  }
243  case SAVE_ID_EXPANSION_RELEASE:
244  if (data.size())
245  {
246  uint32 expansion; // need to read as number not as character
247  loadStream >> expansion;
248  m_expansion = expansion;
249  }
250  else
251  m_expansion = sWorld.getConfig(CONFIG_UINT32_EXPANSION);
252  break;
253  case SAVE_ID_LOVE_IS_IN_THE_AIR:
254  if (data.size())
255  {
256  try
257  {
258  for (uint32 i = 0; i < LOVE_LEADER_MAX; ++i)
259  loadStream >> m_loveIsInTheAirData.counters[i];
260  }
261  catch (std::exception & e)
262  {
263  sLog.outError("%s", e.what());
264  memset(m_loveIsInTheAirData.counters, 0, sizeof(LoveIsInTheAir));
265  }
266  }
267  else
268  memset(m_loveIsInTheAirData.counters, 0, sizeof(LoveIsInTheAir));
269  break;
270  case SAVE_ID_SCOURGE_INVASION:
271  if (data.size())
272  {
273  try
274  {
275  uint32 state;
276  loadStream >> state;
277  m_siData.m_state = SIState(state);
278  for (uint32 i = 0; i < SI_TIMER_MAX; ++i)
279  {
280  uint64 time;
281  loadStream >> time;
282  m_siData.m_timers[i] = std::chrono::time_point_cast<std::chrono::milliseconds>(Clock::from_time_t(time));
283  }
284  loadStream >> m_siData.m_battlesWon >> m_siData.m_lastAttackZone;
285  for (uint32 i = 0; i < SI_REMAINING_MAX; ++i)
286  loadStream >> m_siData.m_remaining[i];
287  }
288  catch (std::exception& e)
289  {
290  sLog.outError("%s", e.what());
291  m_siData.Reset();
292  }
293  }
294  break;
295  }
296  }
297  while (result->NextRow());
298  }
299  StartWarEffortEvent();
300  SpawnWarEffortGos();
301  if (m_siData.m_state == STATE_1_ENABLED)
302  {
303  StartScourgeInvasion(false);
304  HandleDefendedZones();
305  }
306  RespawnEmeraldDragons();
307  StartSunsReachPhase(true);
308  StartSunwellGatePhase();
309  HandleSunsReachSubPhaseTransition(m_sunsReachData.m_subphaseMask, true);
310  StartExpansionEvent();
311 }

313 void WorldState::Save(SaveIds saveId)
314 {
315  switch (saveId)
316  {
317  case SAVE_ID_EMERALD_DRAGONS:
318  {
319  uint64 time;
320  if (m_emeraldDragonsTimer)
321  {
322  auto curTime = World::GetCurrentClockTime();
323  auto respawnTime = std::chrono::milliseconds(m_emeraldDragonsTimer) + curTime;
324  time = uint64(Clock::to_time_t(respawnTime));
325  }
326  else time = 0;
327  std::string dragonsData = std::to_string(m_emeraldDragonsState) + " " + std::to_string(time);
328  for (uint32 i = 0; i < 4; ++i)
329  dragonsData += " " + std::to_string(m_emeraldDragonsChosenPositions[i]);
330  CharacterDatabase.PExecute("DELETE FROM world_state WHERE Id='%u'", SAVE_ID_EMERALD_DRAGONS);
331  CharacterDatabase.PExecute("INSERT INTO world_state(Id,Data) VALUES('%u','%s')", SAVE_ID_EMERALD_DRAGONS, dragonsData.data());
332  break;
333  }
334  case SAVE_ID_AHN_QIRAJ:
335  {
336  std::string expansionData = m_aqData.GetData();
337  SaveHelper(expansionData, SAVE_ID_AHN_QIRAJ);
338  break;
339  }
340  case SAVE_ID_QUEL_DANAS:
341  {
342  std::string expansionData = m_sunsReachData.GetData();
343  SaveHelper(expansionData, SAVE_ID_QUEL_DANAS);
344  break;
345  }
346  case SAVE_ID_EXPANSION_RELEASE:
347  {
348  std::string expansionData = std::to_string(m_expansion);
349  SaveHelper(expansionData, SAVE_ID_EXPANSION_RELEASE);
350  break;
351  }
352  case SAVE_ID_LOVE_IS_IN_THE_AIR:
353  {
354  std::string loveData;
355  for (uint32 i = 0; i < LOVE_LEADER_MAX; ++i)
356  {
357  if (i != 0)
358  loveData += " ";
359  loveData += std::to_string(m_loveIsInTheAirData.counters[i]);
360  }
361  SaveHelper(loveData, SAVE_ID_LOVE_IS_IN_THE_AIR);
362  break;
363  }
364  case SAVE_ID_SCOURGE_INVASION:
365  {
366  std::string siData = m_siData.GetData();
367  SaveHelper(siData, SAVE_ID_SCOURGE_INVASION);
368  break;
369  }
370  default: break;
371  }
