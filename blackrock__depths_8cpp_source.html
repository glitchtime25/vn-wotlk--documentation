375  void UpdateEscortAI(const uint32 uiDiff) override
376  {
377  if (!m_pInstance)
378  return;
379 
380  if (m_pInstance->GetData(TYPE_RING_OF_LAW) == FAIL)
381  {
382  // Reset Doors
383  if (m_uiEventPhase >= 10) // North Gate is opened
384  {
385  m_pInstance->DoUseDoorOrButton(GO_ARENA_2);
386  m_pInstance->DoUseDoorOrButton(GO_ARENA_4);
387  }
388  else if (m_uiEventPhase >= 4) // East Gate is opened
389  {
390  m_pInstance->DoUseDoorOrButton(GO_ARENA_1);
391  m_pInstance->DoUseDoorOrButton(GO_ARENA_4);
392  }
393 
394  // Despawn Summoned Mobs
395  for (GuidList::const_iterator itr = m_lSummonedGUIDList.begin(); itr != m_lSummonedGUIDList.end(); ++itr)
396  {
397  if (Creature* pSummoned = m_creature->GetMap()->GetCreature(*itr))
398  pSummoned->ForcedDespawn();
399  }
400  m_lSummonedGUIDList.clear();
401 
402  // Despawn NPC
403  m_creature->ForcedDespawn();
404  return;
405  }
406 
407  if (m_uiEventTimer)
408  {
409  if (m_uiEventTimer <= uiDiff)
410  {
411  switch (m_uiEventPhase)
412  {
413  case 0:
414  // Shortly after spawn, start walking
415  DoCastSpellIfCan(m_creature, SPELL_ASHCROMBES_TELEPORT_A, CAST_TRIGGERED);
416  DoScriptText(SAY_START_2, m_creature);
417  m_pInstance->DoUseDoorOrButton(GO_ARENA_4);
418  // Some of the NPCs in the crowd do cheer emote at event start
419  // we randomly select 25% of the NPCs to do this
420  m_pInstance->GetArenaCrowdGuid(m_lArenaCrowd);
421  for (auto itr : m_lArenaCrowd)
422  {
423  if (Creature* pSpectator = m_creature->GetMap()->GetCreature(itr))
424  {
425  if (urand(0, 3) < 1)
426  pSpectator->HandleEmote(EMOTE_ONESHOT_CHEER);
427  }
428  }
429  Start(false);
430  SetEscortPaused(false);
431  m_uiEventTimer = 0;
432  break;
433  case 1:
434  // Start walking towards wall
435  SetEscortPaused(false);
436  m_uiEventTimer = 0;
437  break;
438  case 2:
439  m_uiEventTimer = 2000;
440  break;
441  case 3:
442  // Open East Gate
443  DoCastSpellIfCan(m_creature, SPELL_ARENA_FLASH_A, CAST_TRIGGERED);
444  DoCastSpellIfCan(m_creature, SPELL_ARENA_FLASH_B, CAST_TRIGGERED);
445  m_pInstance->DoUseDoorOrButton(GO_ARENA_1);
446  m_uiEventTimer = 3000;
447  break;

449  // timer for teleport out spell which has 2000 ms cast time
450  DoCastSpellIfCan(m_creature, SPELL_ASHCROMBES_TELEPORT_B, CAST_TRIGGERED);

490  if (m_pInstance->GetData(TYPE_RING_OF_LAW) == SPECIAL && m_uiPhase == PHASE_MOBS)

492  m_uiPhase = PHASE_GLADIATORS;
493  SummonRingMob(NPC_THELDREN, 1, POS_NORTH);
