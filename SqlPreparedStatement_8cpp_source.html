94 void SqlPlainPreparedStatement::bind(const SqlStmtParameters& holder)
95 {
96  // verify if we bound all needed input parameters
97  if (m_nParams != holder.boundParams())
98  {
99  MANGOS_ASSERT(false);
100  return;
101  }
102 
103  // reset resulting plain SQL request
104  m_szPlainRequest = m_szFmt;
105  size_t nLastPos = 0;
106 
107  SqlStmtParameters::ParameterContainer const& _args = holder.params();
108 
109  SqlStmtParameters::ParameterContainer::const_iterator iter_last = _args.end();
110  for (SqlStmtParameters::ParameterContainer::const_iterator iter = _args.begin(); iter != iter_last; ++iter)
111  {
112  // bind parameter
113  const SqlStmtFieldData& data = (*iter);
114 
115  std::ostringstream fmt;
116  DataToString(data, fmt);
117 
118  nLastPos = m_szPlainRequest.find('?', nLastPos);
119  if (nLastPos != std::string::npos)
120  {
121  std::string tmp = fmt.str();
122  m_szPlainRequest.replace(nLastPos, 1, tmp);
123  nLastPos += tmp.length();
124  }
125  }
126 }
