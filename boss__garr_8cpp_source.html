95  void ExecuteAction(uint32 action) override
96  {
97  switch (action)
98  {
99  case GARR_ANTI_MAGIC_PULSE:
100  {
101  if (DoCastSpellIfCan(nullptr, SPELL_ANTIMAGICPULSE) == CAST_OK)
102  ResetCombatAction(action, urand(15 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
103  break;
104  }
105  case GARR_MAGMA_SHACKLES:
106  {
107  if (DoCastSpellIfCan(nullptr, SPELL_MAGMASHACKLES) == CAST_OK)
108  ResetCombatAction(action, urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS));
109  break;
110  }
111  case GARR_MASSIVE_ERUPTION:
112  {
113  if (DoCastSpellIfCan(m_creature, SPELL_ERUPTION_TRIGGER) == CAST_OK)
114  {
115  DoScriptText(EMOTE_MASSIVE_ERUPTION, m_creature);
116  ResetCombatAction(action, 20 * IN_MILLISECONDS);
117  }
118  break;
119  }
120  }
121  }
