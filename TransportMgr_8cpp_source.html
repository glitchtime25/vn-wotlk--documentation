81 bool TransportMgr::GenerateWaypoints(GameObjectInfo const* goinfo, TransportTemplate& transportTemplate)
82 {
83  uint32 pathid = goinfo->moTransport.taxiPathId;
84  if (pathid >= sTaxiPathNodesByPath.size())
85  return false;
86 
87  uint32 pathId = goinfo->moTransport.taxiPathId;
88  TaxiPathNodeList const& path = sTaxiPathNodesByPath[pathId];
89  std::vector<KeyFrame>& keyFrames = transportTemplate.keyFrames;
90  Movement::PointsArray splinePath, allPoints;
91  bool mapChange = false;
92  for (size_t i = 0; i < path.size(); ++i)
93  allPoints.push_back(G3D::Vector3(path[i]->x, path[i]->y, path[i]->z));
94 
95  // Add extra points to allow derivative calculations for all path nodes
96  allPoints.insert(allPoints.begin(), allPoints.front().lerp(allPoints[1], -0.2f));
97  allPoints.push_back(allPoints.back().lerp(allPoints[allPoints.size() - 2], -0.2f));
98  allPoints.push_back(allPoints.back().lerp(allPoints[allPoints.size() - 2], -1.0f));
99 
100  SplineRawInitializer initer(allPoints);
101  TransportSpline orientationSpline;
102  orientationSpline.init_spline_custom(initer);
103  orientationSpline.initLengths();
104 
105  for (size_t i = 1; i < path.size() - 1; ++i)
106  {
107  if (!mapChange)
108  {
109  TaxiPathNodeEntry const& node_i = *path[i];
110  if (i != path.size() - 1 && (node_i.actionFlag & 1 || node_i.mapid != path[i + 1]->mapid))
111  {
112  keyFrames.back().Teleport = true;
113  mapChange = true;
114  }
115  else
116  {
117  KeyFrame k(node_i);
118  G3D::Vector3 h;
119  orientationSpline.evaluate_derivative(i + 1, 0.0f, h);
120  k.InitialOrientation = MapManager::NormalizeOrientation(atan2(h.y, h.x) + M_PI);
121 
122  keyFrames.push_back(k);
123  splinePath.push_back(G3D::Vector3(node_i.x, node_i.y, node_i.z));
124  transportTemplate.mapsUsed.insert(k.Node->mapid);
125  }
126  }
127  else
128  mapChange = false;
129  }
130 
131  MANGOS_ASSERT(!keyFrames.empty());
132 
133  if (transportTemplate.mapsUsed.size() > 1)
134  {
135  for (const auto itr : transportTemplate.mapsUsed)
136  MANGOS_ASSERT(!sMapStore.LookupEntry(itr)->Instanceable());
137 
138  transportTemplate.inInstance = false;
139  }
140  else
141  transportTemplate.inInstance = sMapStore.LookupEntry(*transportTemplate.mapsUsed.begin())->Instanceable();
142 
143  // last to first is always "teleport", even for closed paths
144  keyFrames.back().Teleport = true;
145 
146  float const speed = float(goinfo->moTransport.moveSpeed);
147  float const accel = float(goinfo->moTransport.accelRate);
148  float const accel_dist = 0.5f * speed * speed / accel;
149 
150  transportTemplate.accelTime = speed / accel;
151  transportTemplate.accelDist = accel_dist;
152 
153  int32 firstStop = -1;
154  int32 lastStop = -1;
155 
156  // first cell is arrived at by teleportation :S
157  keyFrames[0].DistFromPrev = 0;
158  keyFrames[0].Index = 1;
159  if (keyFrames[0].IsStopFrame())
160  {
161  firstStop = 0;
162  lastStop = 0;
163  }
164 
165  // find the rest of the distances between key points
166  // Every path segment has its own spline
167  size_t start = 0;
168  for (size_t i = 1; i < keyFrames.size(); ++i)
169  {
170  if (keyFrames[i - 1].Teleport || i + 1 == keyFrames.size())
171  {
172  size_t extra = !keyFrames[i - 1].Teleport ? 1 : 0;
173  TransportSpline* spline = new TransportSpline();
174  spline->init_spline(&splinePath[start], i - start + extra, Movement::SplineBase::ModeCatmullrom);
175  spline->initLengths();
176  for (size_t j = start; j < i + extra; ++j)
177  {
178  keyFrames[j].Index = j - start + 1;
179  keyFrames[j].DistFromPrev = spline->length(j - start, j + 1 - start);
180  if (j > 0)
181  keyFrames[j - 1].NextDistFromPrev = keyFrames[j].DistFromPrev;
182  keyFrames[j].Spline = spline;
183  }
184 
185  if (keyFrames[i - 1].Teleport)
186  {
187  keyFrames[i].Index = i - start + 1;
188  keyFrames[i].DistFromPrev = 0.0f;
189  keyFrames[i - 1].NextDistFromPrev = 0.0f;
190  keyFrames[i].Spline = spline;
191  }
192 
193  start = i;
194  }
195 
196  if (keyFrames[i].IsStopFrame())
197  {
198  // remember first stop frame
199  if (firstStop == -1)
200  firstStop = i;
201  lastStop = i;
202  }
203  }
204 
205  keyFrames.back().NextDistFromPrev = keyFrames.front().DistFromPrev;
206 
207  if (firstStop == -1 || lastStop == -1)
208  firstStop = lastStop = 0;
209 
210  // at stopping keyframes, we define distSinceStop == 0,
211  // and distUntilStop is to the next stopping keyframe.
212  // this is required to properly handle cases of two stopping frames in a row (yes they do exist)
213  float tmpDist = 0.0f;
214  for (size_t i = 0; i < keyFrames.size(); ++i)
215  {
216  int32 j = (i + lastStop) % keyFrames.size();
217  if (keyFrames[j].IsStopFrame() || j == lastStop)
218  tmpDist = 0.0f;
219  else
220  tmpDist += keyFrames[j].DistFromPrev;
221  keyFrames[j].DistSinceStop = tmpDist;
222  }
223 
224  tmpDist = 0.0f;
225  for (int32 i = int32(keyFrames.size()) - 1; i >= 0; i--)
226  {
227  int32 j = (i + firstStop) % keyFrames.size();
228  tmpDist += keyFrames[(j + 1) % keyFrames.size()].DistFromPrev;
229  keyFrames[j].DistUntilStop = tmpDist;
230  if (keyFrames[j].IsStopFrame() || j == firstStop)
231  tmpDist = 0.0f;
232  }
233 
234  for (auto& keyFrame : keyFrames)
235  {
236  float total_dist = keyFrame.DistSinceStop + keyFrame.DistUntilStop;
237  if (total_dist < 2 * accel_dist) // won't reach full speed
238  {
239  if (keyFrame.DistSinceStop < keyFrame.DistUntilStop) // is still accelerating
240  {
241  // calculate accel+brake time for this short segment
242  float segment_time = 2.0f * sqrt((keyFrame.DistUntilStop + keyFrame.DistSinceStop) / accel);
243  // substract acceleration time
244  keyFrame.TimeTo = segment_time - sqrt(2 * keyFrame.DistSinceStop / accel);
245  }
246  else // slowing down
247  keyFrame.TimeTo = sqrt(2 * keyFrame.DistUntilStop / accel);
248  }
249  else if (keyFrame.DistSinceStop < accel_dist) // still accelerating (but will reach full speed)
250  {
251  // calculate accel + cruise + brake time for this long segment
252  float segment_time = (keyFrame.DistUntilStop + keyFrame.DistSinceStop) / speed + (speed / accel);
253  // substract acceleration time
254  keyFrame.TimeTo = segment_time - sqrt(2 * keyFrame.DistSinceStop / accel);
255  }
256  else if (keyFrame.DistUntilStop < accel_dist) // already slowing down (but reached full speed)
257  keyFrame.TimeTo = sqrt(2 * keyFrame.DistUntilStop / accel);
258  else // at full speed
259  keyFrame.TimeTo = (keyFrame.DistUntilStop / speed) + (0.5f * speed / accel);
260  }
261 
262  // calculate tFrom times from tTo times
263  float segmentTime = 0.0f;
264  for (size_t i = 0; i < keyFrames.size(); ++i)
265  {
266  int32 j = (i + lastStop) % keyFrames.size();
267  if (keyFrames[j].IsStopFrame() || j == lastStop)
268  segmentTime = keyFrames[j].TimeTo;
269  keyFrames[j].TimeFrom = segmentTime - keyFrames[j].TimeTo;
270  }
271 
272  // calculate path times
273  keyFrames[0].ArriveTime = 0;
274  float curPathTime = 0.0f;
275  if (keyFrames[0].IsStopFrame())
276  {
277  curPathTime = float(keyFrames[0].Node->delay);
278  keyFrames[0].DepartureTime = uint32(curPathTime * IN_MILLISECONDS);
279  }
280 
281  for (size_t i = 1; i < keyFrames.size(); ++i)
282  {
283  curPathTime += keyFrames[i - 1].TimeTo;
284  if (keyFrames[i].IsStopFrame())
285  {
286  keyFrames[i].ArriveTime = uint32(curPathTime * IN_MILLISECONDS);
287  keyFrames[i - 1].NextArriveTime = keyFrames[i].ArriveTime;
288  curPathTime += float(keyFrames[i].Node->delay);
289  keyFrames[i].DepartureTime = uint32(curPathTime * IN_MILLISECONDS);
290  }
291  else
292  {
293  curPathTime -= keyFrames[i].TimeTo;
294  keyFrames[i].ArriveTime = uint32(curPathTime * IN_MILLISECONDS);
295  keyFrames[i - 1].NextArriveTime = keyFrames[i].ArriveTime;
296  keyFrames[i].DepartureTime = keyFrames[i].ArriveTime;
297  }
298  }
299 
300  keyFrames.back().NextArriveTime = keyFrames.back().DepartureTime;
301  // the client destroys a transport by itself after a while, refresh is needed mid course
302  // Feathermoon 303 & Teldrassil 293 ferries
303  if (pathId == 303 || pathId == 293)
304  keyFrames[12].Update = true;
305  transportTemplate.pathTime = keyFrames.back().DepartureTime;
306 
307  return true;
308 }
