39 Unit* SelectRandomFlameshockerSpawnTarget(Creature* unit, Creature* except, float radius)
40 {
41  CreatureList targets;
42 
43  MaNGOS::AnyUnitInObjectRangeCheck u_check(unit, radius);
44  MaNGOS::CreatureListSearcher<MaNGOS::AnyUnitInObjectRangeCheck> searcher(targets, u_check);
45  Cell::VisitAllObjects(unit, searcher, radius);
46 
47  // remove current target
48  if (except)
49  targets.remove(except);
50 
51  for (auto tIter = targets.begin(); tIter != targets.end();)
52  {
53  // || !(*tIter)->ToCreature()->CanSummonGuards() - CREATURE_FLAG_EXTRA_SUMMON_GUARD
54  if ((*tIter)->IsCivilian() || (*tIter)->GetZoneId() != unit->GetZoneId() || !unit->CanAttack((*tIter)) || GetClosestCreatureWithEntry((*tIter), NPC_FLAMESHOCKER, VISIBILITY_DISTANCE_TINY))
55  {
56  auto tIter2 = tIter;
57  ++tIter;
58  targets.erase(tIter2);
59  }
60  else
61  ++tIter;
62  }
63 
64  // no appropriate targets
65  if (targets.empty())
66  return nullptr;
67 
68  // select random
69  std::vector<Creature*> random(targets.begin(), targets.end());
70  std::shuffle(random.begin(), random.end(), *GetRandomGenerator());
71 
72  return random[0];
73 }
