34 void WorldSession::HandleUseItemOpcode(WorldPacket& recvPacket)
35 {
36  uint8 bagIndex, slot;
37  uint8 cast_flags; // flags (if 0x02 - some additional data are received)
38  uint8 cast_count; // next cast if exists (single or not)
39  ObjectGuid itemGuid;
40  uint32 glyphIndex; // something to do with glyphs?
41  uint32 spellid; // casted spell id
42 
43  recvPacket >> bagIndex >> slot >> cast_count >> spellid >> itemGuid >> glyphIndex >> cast_flags;
44 
45  // TODO: add targets.read() check
46  Player* pUser = _player;
47 
48  // ignore for remote control state
49  if (!pUser->IsSelfMover())
50  {
51  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
52  return;
53  }
54 
55  // reject fake data
56  if (glyphIndex >= MAX_GLYPH_SLOT_INDEX)
57  {
58  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
59  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
60  return;
61  }
62 
63  Item* pItem = pUser->GetItemByPos(bagIndex, slot);
64  if (!pItem)
65  {
66  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
67  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
68  return;
69  }
70 
71  if (pItem->GetObjectGuid() != itemGuid)
72  {
73  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
74  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
75  return;
76  }
77 
78  DETAIL_LOG("WORLD: CMSG_USE_ITEM packet, bagIndex: %u, slot: %u, cast_count: %u, spellid: %u, Item: %u, glyphIndex: %u, unk_flags: %u, data length = %u", bagIndex, slot, cast_count, spellid, pItem->GetEntry(), glyphIndex, cast_flags, (uint32)recvPacket.size());
79 
80  ItemPrototype const* proto = pItem->GetProto();
81  if (!proto)
82  {
83  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
84  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
85  return;
86  }
87 
88  // some item classes can be used only in equipped state
89  if (proto->InventoryType != INVTYPE_NON_EQUIP && !pItem->IsEquipped())
90  {
91  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
92  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
93  return;
94  }
95 
96  InventoryResult msg = pUser->CanUseItem(pItem);
97  if (msg != EQUIP_ERR_OK)
98  {
99  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
100  pUser->SendEquipError(msg, pItem, nullptr);
101  return;
102  }
103 
104  // not allow use item from trade (cheat way only)
105  if (pItem->IsInTrade())
106  {
107  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
108  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
109  return;
110  }
111 
112  // only allow conjured consumable, bandage, poisons (all should have the 2^21 item flag set in DB)
113  if (proto->Class == ITEM_CLASS_CONSUMABLE &&
114  !(proto->Flags & ITEM_FLAG_IGNORE_DEFAULT_ARENA_RESTRICTIONS) &&
115  pUser->InArena())
116  {
117  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
118  pUser->SendEquipError(EQUIP_ERR_NOT_DURING_ARENA_MATCH, pItem, nullptr);
119  return;
120  }
121 
122  if (pUser->IsInCombat())
123  {
124  for (const auto& Spell : proto->Spells)
125  {
126  if (SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(Spell.SpellId))
127  {
128  if (IsNonCombatSpell(spellInfo))
129  {
130  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
131  pUser->SendEquipError(EQUIP_ERR_NOT_IN_COMBAT, pItem, nullptr);
132  return;
133  }
134  }
135  }
136 
137  // Prevent potion drink if another potion in processing (client have potions disabled in like case)
138  if (pItem->IsPotion() && pUser->GetLastPotionId())
139  {
140  recvPacket.rpos(recvPacket.wpos()); // prevent spam at not read packet tail
141  pUser->SendEquipError(EQUIP_ERR_OBJECT_IS_BUSY, pItem, nullptr);
142  return;
143  }
144  }
145 
146  // check also BIND_WHEN_PICKED_UP and BIND_QUEST_ITEM for .additem or .additemset case by GM (not binded at adding to inventory)
147  if (pItem->GetProto()->Bonding == BIND_WHEN_USE || pItem->GetProto()->Bonding == BIND_WHEN_PICKED_UP || pItem->GetProto()->Bonding == BIND_QUEST_ITEM)
148  {
149  if (!pItem->IsSoulBound())
150  {
151  pItem->SetState(ITEM_CHANGED, pUser);
152  pItem->SetBinding(true);
153  }
154  }
155 
156  SpellCastTargets targets;
157 
158  recvPacket >> targets.ReadForCaster(pUser);
159 
160  // some spell cast packet including more data (for projectiles)
161  HandleClientCastFlags(recvPacket, cast_flags, targets);
162 
163  targets.Update(pUser);
164 
165  if (!pItem->IsTargetValidForItemUse(targets.getUnitTarget()))
166  {
167  // free gray item after use fail
168  pUser->SendEquipError(EQUIP_ERR_NONE, pItem, nullptr);
169 
170  // send spell error
171  if (SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellid))
172  {
173  // for implicit area/coord target spells
174  if (IsPointEffectTarget(SpellTarget(spellInfo->EffectImplicitTargetA[EFFECT_INDEX_0])) ||
175  IsAreaEffectTarget(SpellTarget(spellInfo->EffectImplicitTargetA[EFFECT_INDEX_0])))
176  Spell::SendCastResult(_player, spellInfo, cast_count, SPELL_FAILED_NO_VALID_TARGETS);
177  // for explicit target spells
178  else
179  Spell::SendCastResult(_player, spellInfo, cast_count, SPELL_FAILED_BAD_TARGETS);
180  }
181  return;
182  }
183 
184  _player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_ITEM_USE);
185 
186  // Note: If script stop casting it must send appropriate data to client to prevent stuck item in gray state.
187  if (!sScriptDevAIMgr.OnItemUse(pUser, pItem, targets))
188  {
189  // no script or script not process request by self
190  pUser->CastItemUseSpell(pItem, targets, cast_count, glyphIndex, spellid);
191  }
192 }

201 {
202  DETAIL_LOG("WORLD: CMSG_OPEN_ITEM packet, data length = " SIZEFMTD, recvPacket.size());
203 
204  uint8 bagIndex, slot;
205 
206  recvPacket >> bagIndex >> slot;
207 
208  DETAIL_LOG("bagIndex: %u, slot: %u", bagIndex, slot);
209 
210  Player* pUser = _player;
211 
212  // ignore for remote control state
213  if (!pUser->IsSelfMover())
214  return;
215 
216  Item* pItem = pUser->GetItemByPos(bagIndex, slot);
217  if (!pItem)
218  {
219  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
220  return;
221  }
222 
223  ItemPrototype const* proto = pItem->GetProto();
224  if (!proto)
225  {
226  pUser->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, pItem, nullptr);
227  return;
228  }
229 
230  // locked item
231  uint32 lockId = proto->LockID;
232  if (lockId && !pItem->HasFlag(ITEM_FIELD_FLAGS, ITEM_DYNFLAG_UNLOCKED))
233  {
234  LockEntry const* lockInfo = sLockStore.LookupEntry(lockId);
235 
236  if (!lockInfo)
237  {
238  pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, pItem, nullptr);
239  sLog.outError("WORLD::OpenItem: item [guid = %u] has an unknown lockId: %u!", pItem->GetGUIDLow(), lockId);
240  return;
241  }
242 
243  // required picklocking
244  if (lockInfo->Skill[1] || lockInfo->Skill[0])
245  {
246  pUser->SendEquipError(EQUIP_ERR_ITEM_LOCKED, pItem, nullptr);
247  return;
248  }
249  }
250 
251  if (pItem->HasFlag(ITEM_FIELD_FLAGS, ITEM_DYNFLAG_WRAPPED))// wrapped?
252  {
253  QueryResult* result = CharacterDatabase.PQuery("SELECT entry, flags FROM character_gifts WHERE item_guid = '%u'", pItem->GetGUIDLow());
254  if (result)
255  {
256  Field* fields = result->Fetch();
257  uint32 entry = fields[0].GetUInt32();
258  uint32 flags = fields[1].GetUInt32();
259 
260  pItem->SetGuidValue(ITEM_FIELD_GIFTCREATOR, ObjectGuid());
261  pItem->SetEntry(entry);
262  pItem->SetUInt32Value(ITEM_FIELD_FLAGS, flags);
263  pItem->SetState(ITEM_CHANGED, pUser);
264  delete result;
265  }
266  else
267  {
268  sLog.outError("Wrapped item %u don't have record in character_gifts table and will deleted", pItem->GetGUIDLow());
269  pUser->DestroyItem(pItem->GetBagSlot(), pItem->GetSlot(), true);
270  return;
271  }
272 
273  static SqlStatementID delGifts ;
274 
275  SqlStatement stmt = CharacterDatabase.CreateStatement(delGifts, "DELETE FROM character_gifts WHERE item_guid = ?");
276  stmt.PExecute(pItem->GetGUIDLow());
277  }
278  else
279  {
280  Loot*& loot = pItem->m_loot;
281  if (!loot)
282  loot = new Loot(pUser, pItem, LOOT_PICKPOCKETING);
283 
284  loot->ShowContentTo(pUser);
285  }
286 }

288 void WorldSession::HandleGameObjectUseOpcode(WorldPacket& recv_data)
289 {
290  ObjectGuid guid;
291 
292  recv_data >> guid;
293 
294  DEBUG_LOG("WORLD: Received opcode CMSG_GAMEOBJ_USE guid: %s", guid.GetString().c_str());
295 
296  if (_player->IsBeingTeleported())
297  return;
298 
299  GameObject* obj = _player->GetMap()->GetGameObject(guid);
300  if (!obj)
301  return;
302 
303  if (!obj->IsAtInteractDistance(_player))
304  return;
305 
306  // Additional check preventing exploits (ie loot despawned chests)
307  if (!obj->IsSpawned())
308  {
309  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for despawned GameObject (Entry %u), didn't expect this to happen.", obj->GetEntry());
310  return;
311  }
312 
313  // ignore for remote control state
314  if (!_player->IsSelfMover())
315  {
316  // check player on vehicle
317  if (!_player->GetTransportInfo() || !_player->GetTransportInfo()->IsOnVehicle() || !obj->GetGOInfo()->IsUsableMounted())
318  return;
319  }
320 
321  // Never expect this opcode for some type GO's
322  if (obj->GetGoType() == GAMEOBJECT_TYPE_GENERIC)
323  {
324  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for not allowed GameObject type %u (Entry %u), didn't expect this to happen.", obj->GetGoType(), obj->GetEntry());
325  return;
326  }
327 
328  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED)) // we should not allow use of a locked GO
329  return;
330 
331  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_IN_USE))
332  return;
333 
334  // Never expect this opcode for non intractable GO's
335  if (obj->HasFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT))
336  {
337  sLog.outError("HandleGameObjectUseOpcode: CMSG_GAMEOBJ_USE for GameObject (Entry %u) with non intractable flag (Flags %u), didn't expect this to happen.", obj->GetEntry(), obj->GetUInt32Value(GAMEOBJECT_FLAGS));
338  return;
339  }
340 
341  // client checks this but needs recheck
342  if (obj->GetGOInfo()->CannotBeUsedUnderImmunity() && _player->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE))
343  return;
344 
345  // code meant to be in CanUseNow
346  if (obj->GetGoType() == GAMEOBJECT_TYPE_CHAIR)
347  {
348  float x, y;
349  std::tie(x, y) = obj->GetClosestChairSlotPosition(_player);
350  if (_player->GetDistance(x, y, obj->GetPositionZ(), DIST_CALC_NONE) > 3.f * 3.f)
351  return;
352  }
353 
354  obj->Use(_player);
355 }

379 {
380  uint32 spellId;
381  uint8 cast_count, cast_flags;
382  recvPacket >> cast_count;
383  recvPacket >> spellId;
384  recvPacket >> cast_flags; // flags (if 0x02 - some additional data are received)
385 
386  // ignore for remote control state (for player case)
387  Unit* mover = _player->GetMover();
388  if (mover != _player && mover->GetTypeId() == TYPEID_PLAYER)
389  {
390  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
391  return;
392  }
393 
394  DEBUG_LOG("WORLD: CMSG_CAST_SPELL, spellId - %u, cast_count: %u, unk_flags %u, data length = " SIZEFMTD,
395  spellId, cast_count, cast_flags, recvPacket.size());
396 
397  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
398  if (!spellInfo)
399  {
400  sLog.outError("WORLD: unknown spell id %u", spellId);
401  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
402  return;
403  }
404 
405  Aura* triggeredByAura = mover->GetTriggeredByClientAura(spellId);
406 
407  Unit* caster = mover;
408  if (mover->GetTypeId() == TYPEID_PLAYER)
409  {
410  // not have spell in spellbook or spell passive and not casted by client
411 
412  if ((!((Player*)mover)->HasActiveSpell(spellId) && !triggeredByAura) || IsPassiveSpell(spellInfo))
413  {
414  sLog.outError("World: %s casts spell %u which he shouldn't have", mover->GetGuidStr().c_str(), spellId);
415  // cheater? kick? ban?
416  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
417  return;
418  }
419  }
420  else
421  {
422  // If the vehicle creature does not have the spell but it allows the passenger to cast own spells
423  // change caster to player and let him cast
424  if (!_player->GetTransportInfo() || _player->GetTransportInfo()->GetTransport() != caster || Spell::CheckVehicle(_player, *spellInfo) != SPELL_CAST_OK)
425  {
426  bool isPassive = IsPassiveSpell(spellInfo);
427  // not have spell in spellbook or spell passive and not casted by client
428  if (!mover->HasSpell(spellId) || isPassive)
429  {
430  if (!_player->HasSpell(spellId) || isPassive)
431  {
432  // cheater? kick? ban?
433  recvPacket.rpos(recvPacket.wpos()); // prevent spam at ignore packet
434  return;
435  }
436  }
437  else
438  caster = _player;
439  }
440  }
441 
442  // client provided targets
443  SpellCastTargets targets;
444 
445 #ifdef BUILD_PLAYERBOT
446  recvPacket >> targets.ReadForCaster(caster);
447 #else
448  recvPacket >> targets.ReadForCaster(_player);
449 #endif
450 
451  // some spell cast packet including more data (for projectiles)
452  HandleClientCastFlags(recvPacket, cast_flags, targets);
453 
454  // auto-selection buff level base at target level (in spellInfo)
455  if (Unit* target = targets.getUnitTarget())
456  {
457  // if rank not found then function return nullptr but in explicit cast case original spell can be casted and later failed with appropriate error message
458  if (SpellEntry const* actualSpellInfo = sSpellMgr.SelectAuraRankForLevel(spellInfo, target->GetLevel()))
459  spellInfo = actualSpellInfo;
460  }
461 
462  if (HasMissingTargetFromClient(spellInfo))
463  targets.setUnitTarget(caster->GetTarget());
464 
465  if (_player->HasQueuedSpell())
466  return;
467 
468  bool handled = false;
469  Spell* spell = new Spell(caster, spellInfo, triggeredByAura ? TRIGGERED_OLD_TRIGGERED : TRIGGERED_NONE, caster->GetObjectGuid(), triggeredByAura ? triggeredByAura->GetSpellProto() : nullptr);
470  spell->m_cast_count = cast_count; // set count of casts
471  spell->m_clientCast = true;
472  if (!triggeredByAura && (caster->HasGCD(spellInfo) || !caster->IsSpellReady(*spellInfo)))
473  {
474  if (caster->HasGCDOrCooldownWithinMargin(*spellInfo))
475  {
476  handled = true;
477  _player->SetQueuedSpell(spell);
478  GetMessager().AddMessage([guid = caster->GetObjectGuid(), isPlayer = caster != mover, targets = targets](WorldSession* session) mutable
479  {
480  if (session->GetPlayer()) // in case of logout
481  {
482  // in case of mind control end
483  if ((isPlayer && session->GetPlayer()->GetObjectGuid() == guid) || (!isPlayer && session->GetPlayer()->GetMover()->GetObjectGuid() == guid))
484  session->GetPlayer()->CastQueuedSpell(targets);
485  else
486  session->GetPlayer()->ClearQueuedSpell();
487  }
488  });
489  }
490  }
491 
492  if (!handled)
493  spell->SpellStart(&targets, triggeredByAura);
494 }
