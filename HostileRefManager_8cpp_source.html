41 void HostileRefManager::threatAssist(Unit* victim, float threat, SpellEntry const* threatSpell, bool singleTarget, bool ignoreTimer)
42 {
43  if (threatSpell->HasAttribute(SPELL_ATTR_EX2_NO_INITIAL_THREAT) || threatSpell->HasAttribute(SPELL_ATTR_EX_NO_THREAT) || !getOwner()->CanEnterCombat() || !victim->CanEnterCombat())
44  return;
45 
46  if (threatSpell->HasAttribute(SPELL_ATTR_EX4_NO_HELPFUL_THREAT))
47  return;
48 
49  HostileReference* ref = getFirst();
50  if (!ref)
51  return;
52 
53  std::vector<HostileReference*> validRefs;
54  while (ref)
55  {
56  Unit* owner = ref->getSource()->getOwner();
57  if (!owner->IsCrowdControlled() && owner != victim && owner->CanAttack(victim))
58  validRefs.push_back(ref);
59  ref = ref->next();
60  }
61 
62  uint32 size = singleTarget ? 1 : validRefs.size(); // if singleTarget do not devide threat
63  float threatPerTarget = threat / size;
64  for (HostileReference* validReference : validRefs)
65  {
66  validReference->getSource()->addThreat(victim, threatPerTarget, false, (threatSpell ? GetSpellSchoolMask(threatSpell) : SPELL_SCHOOL_MASK_NORMAL), threatSpell, true);
67  if (!ignoreTimer)
68  victim->GetCombatManager().TriggerCombatTimer(validReference->getSource()->getOwner());
69  }
70 }

235 void HostileRefManager::HandleSuppressed(bool apply, bool immunity)
236 {
237  if (apply)
238  {
239  for (auto& data : *this)
240  {
241  HostileReference& ref = static_cast<HostileReference&>(data);
242  if (immunity)
243  {
244  Unit* source = ref.getSource()->getOwner();
245  Unit* target = ref.getTarget();
246  if (!target->IsImmuneToDamage(source->GetMainAttackSchoolMask()))
247  continue;
248  }
249  ref.SetHostileState(STATE_SUPPRESSED);
250  }
251  }
252  else
253  {
254  for (auto& data : *this)
255  {
256  HostileReference& ref = static_cast<HostileReference&>(data);
257  Unit* source = ref.getSource()->getOwner();
258  Unit* target = ref.getTarget();
259  if (!source->IsSuppressedTarget(target))
260  {
261  ref.SetSuppressabilityToggle();
262  continue;
263  }
264  ref.SetHostileState(STATE_SUPPRESSED);
265  }
266  }
267 }
