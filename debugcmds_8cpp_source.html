124 bool ChatHandler::HandleDebugSendOpcodeCommand(char* /*args*/)
125 {
126  Unit* unit = getSelectedUnit();
127  if (!unit || (unit->GetTypeId() != TYPEID_PLAYER))
128  unit = m_session->GetPlayer();
129 
130  std::ifstream stream("opcode.txt");
131  if (!stream.is_open())
132  return false;
133 
134  uint32 opcode = 0;
135  if (!(stream >> opcode))
136  {
137  stream.close();
138  return false;
139  }
140 
141  WorldPacket data(Opcodes(opcode), 0);
142 
143  std::string type;
144  while (stream >> type)
145  {
146  if (type.empty())
147  break;
148 
149  if (type == "uint8")
150  {
151  uint16 value;
152  stream >> value;
153  data << uint8(value);
154  }
155  else if (type == "uint16")
156  {
157  uint16 value;
158  stream >> value;
159  data << value;
160  }
161  else if (type == "uint32")
162  {
163  uint32 value;
164  stream >> value;
165  data << value;
166  }
167  else if (type == "uint64")
168  {
169  uint64 value;
170  stream >> value;
171  data << value;
172  }
173  else if (type == "float")
174  {
175  float value;
176  stream >> value;
177  data << value;
178  }
179  else if (type == "string")
180  {
181  std::string value;
182  stream >> value;
183  data << value;
184  }
185  else if (type == "pguid")
186  {
187  data << unit->GetPackGUID();
188  }
189  else
190  {
191  DEBUG_LOG("Sending opcode: unknown type '%s'", type.c_str());
192  break;
193  }
194  }
195  stream.close();
196 
197  DEBUG_LOG("Sending opcode %u, %s", data.GetOpcode(), data.GetOpcodeName());
198 
199  data.hexlike();
200  ((Player*)unit)->GetSession()->SendPacket(data);
201 
202  PSendSysMessage(LANG_COMMAND_OPCODESENT, data.GetOpcode(), unit->GetName());
203 
204  return true;
205 }
