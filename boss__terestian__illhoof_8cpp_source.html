106  uint32 GetSubsequentActionTimer(uint32 id)

121  void ExecuteAction(uint32 action) override
122  {
123  switch (action)
124  {
125  case ILLHOOF_ACTION_SUMMON_KILREK:
126  {
127  DoCastSpellIfCan(m_creature, SPELL_SUMMON_IMP);
128  DisableCombatAction(action);
129  return;
130  }
131  case ILLHOOF_ACTION_SACRIFICE:
132  {
133 #ifdef PRENERF_2_0_3
134  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SACRIFICE, SELECT_FLAG_PLAYER))
135 #else
136  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_SACRIFICE, SELECT_FLAG_PLAYER))
137 #endif
138  {
139  if (DoCastSpellIfCan(target, SPELL_SACRIFICE) == CAST_OK)
140  {
141  DoCastSpellIfCan(m_creature, SPELL_SUMMON_DEMONCHAINS, CAST_TRIGGERED);
142  DoScriptText(urand(0, 1) ? SAY_SACRIFICE1 : SAY_SACRIFICE2, m_creature);
143  m_sacrificeGuid = target->GetObjectGuid();
144  }
145  }
146  ResetCombatAction(action, GetSubsequentActionTimer(action));
147  return;
148  }
149  case ILLHOOF_ACTION_SHADOWBOLT:
150  {
151  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SHADOW_BOLT);
152  ResetCombatAction(action, GetSubsequentActionTimer(action));
153  return;
154  }
155  case ILLHOOF_ACTION_SUMMON:
156  {
157  if (DoCastSpellIfCan(nullptr, SPELL_FIENDISH_PORTAL) == CAST_OK)
158  DoScriptText(urand(0, 1) ? SAY_SUMMON1 : SAY_SUMMON2, m_creature);
159  DisableCombatAction(action);
160  return;
161  }
162  case ILLHOOF_ACTION_BERSERK:
163  {
164  DoCastSpellIfCan(nullptr, SPELL_BERSERK);
165  DisableCombatAction(action);
166  return;
167  }
168  }
169  }

210  summoned->SetInCombatWithZone();

213  m_creature->RemoveAurasDueToSpell(SPELL_BROKEN_PACT);
214  break;
215  case NPC_DEMONCHAINS:
216  m_chainsGuid = summoned->GetObjectGuid();
