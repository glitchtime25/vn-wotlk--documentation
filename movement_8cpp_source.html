245 void Movement::OrderAck(uint16 opcode, uint32 counter)
246 {
247  _ackHistory.push_back({ opcode, WorldTimer::getMSTime() });
248 
249  sLog.outDebug("ORDER: %s (%u) ACK %s (counter = %u)", LookupOpcodeName(opcode), opcode, _me->GetName(), counter);
250 
251  if (_orders.empty())
252  {
253  if (sAnticheatConfig.IsEnabled(CHEAT_TYPE_BAD_ORDER_ACK))
254  _anticheat->RecordCheatInternal(CHEAT_TYPE_BAD_ORDER_ACK, "Received ACK %s (%u) but no orders are pending",
255  LookupOpcodeName(opcode), opcode);
256 
257  return;
258  }
259 
260  // if this detection method is disabled, just empty the results and stop
261  if (!sAnticheatConfig.IsEnabled(CHEAT_TYPE_BAD_ORDER_ACK))
262  {
263  _orders.clear();
264  return;
265  }
266 
267  // FIXME: there is currently a bug in the core whereby SMSG_FORCE_* opcodes are sent to a world session for
268  // units other than the one they are controlling (i.e. a pet), prompting multiple consecutive ACKs, which
269  // are not guaranteed to arrive in the same order they were sent. once that bug is fixed, the below code
270  // can be replaced with one which is both faster and more strict, where only the front of the queue is
271  // checked against the data received from the client.
272  for (auto i = _orders.begin(); i != _orders.end(); ++i)
273  {
274  auto const response = GetOrderResponse(i->opcode);
275 
276  // found? remove from the queue and return
277  if (response == opcode && counter == i->counter)
278  {
279  _orders.erase(i);
280  return;
281  }
282  }
283 
284  // if execution reaches here, it means that the acknowledgement we received was not in the queue to be received.
285  // this should never happen for legitimate players.
286 
287  _anticheat->RecordCheatInternal(CHEAT_TYPE_BAD_ORDER_ACK, "Received ACK %s (%u) counter = %u which was not pending",
288  LookupOpcodeName(opcode), opcode, counter);
289 }

313 bool Movement::HandleAnticheatTests(MovementInfo& movementInfo, WorldSession* session, const WorldPacket& packet)
314 {
315  MANGOS_ASSERT(!!_me);
316  MANGOS_ASSERT(session);
317 
318  if (!sAnticheatConfig.EnableAnticheat())
319  return true;
320 
321  // TODO: Currently anticheat is disabled with Mind Controlled players!
322  if (!_me->IsSelfMover())
323  {
324  _wasMovingOther = true;
325  return true;
326  }
327  // if we were moving someone else, but are no longer, skip one more tick
328  else if (_wasMovingOther)
329  {
330  _wasMovingOther = false;
331  return true;
332  }
333 
334  auto const opcode = packet.GetOpcode();
335 
336  // Dont accept movement packets while movement is controlled by server (fear, charge, etc..)
337  if (!_me->movespline->Finalized())
338  return false;

340  // two dimensional distance traveled
341  auto const dist2d =
342  (movementInfo.pos.x - GetLastMovementInfo().pos.x) * (movementInfo.pos.x - GetLastMovementInfo().pos.x) +
343  (movementInfo.pos.y - GetLastMovementInfo().pos.y) * (movementInfo.pos.y - GetLastMovementInfo().pos.y);
344 
345  _totalDistanceTraveled += sqrtf(dist2d);
346 
347  if (_inKnockBack && opcode != MSG_MOVE_FALL_LAND)
348  movementInfo.jump = GetLastMovementInfo().jump;
349 
350  if (opcode == CMSG_MOVE_FEATHER_FALL_ACK)
351  {
352  GetLastMovementInfo().jump.startClientTime = movementInfo.jump.startClientTime = movementInfo.ctime;
353  //GetLastMovementInfo().jump.start = movementInfo.jump.start = movementInfo.pos;

356 
357  /*if ((movementInfo.moveFlags & MOVE_ROOT) && (movementInfo.moveFlags & MOVE_MASK_MOVING_OR_TURN))
358  APPEND_CHEAT(CHEAT_TYPE_ROOT_MOVE);*/
359 
360  if (sAnticheatConfig.IsEnabled(CHEAT_TYPE_MULTIJUMP) &&
361  opcode == MSG_MOVE_JUMP &&
362  GetLastMovementInfo().moveFlags & MOVEFLAG_FALLING)
363  {
364  _anticheat->RecordCheatInternal(CHEAT_TYPE_MULTIJUMP, "Jump start %u", movementInfo.jump.startClientTime);
365 
366  std::string str;
367  DumpMovement(str);
368 
369  sLog.outBasic("Multijump debug output:\n%s", str.c_str());
370  }
371 
372  if (opcode == MSG_MOVE_STOP_SWIM && (movementInfo.moveFlags & MOVEFLAG_SWIMMING))
373  _anticheat->RecordCheatInternal(CHEAT_TYPE_FLY_HACK);
374 
375  if (!movementInfo.ctime && sAnticheatConfig.IsEnabled(CHEAT_TYPE_NULL_CLIENT_TIME))
376  _anticheat->RecordCheatInternal(CHEAT_TYPE_NULL_CLIENT_TIME);
377 
378  // This can happen if the player is knocked back while jumping

386  GetXYFlagBasedClientSpeed(GetLastMovementInfo().moveFlags) :
387  GetXYFlagBasedClientSpeed(movementInfo.moveFlags);

391  auto const oldClientSpeed = GetXYFlagBasedClientSpeed(GetLastMovementInfo().moveFlags);

489  _anticheat->RecordCheatInternal(CHEAT_TYPE_FORBIDDEN, "Too high in Warsong Gulch (z = %f)", movementInfo.pos.z);

494  {
495  // Battleground not started. Players should be in their starting areas.

497  _anticheat->RecordCheatInternal(CHEAT_TYPE_FORBIDDEN, "Outside alliance starting area (%f, %f, %f)",

501  movementInfo.pos.x, movementInfo.pos.y, movementInfo.pos.z);

503  }

507  if (movementInfo.moveFlags & MOVEFLAG_ONTRANSPORT &&
508  sAnticheatConfig.IsEnabled(CHEAT_TYPE_TELE_TO_TRANSPORT) &&
509  GetLastMovementInfo().ctime &&
510  !(GetLastMovementInfo().moveFlags & MOVEFLAG_ONTRANSPORT))

520 
521  // Distance computation related
522  if (!_me->IsTaxiFlying() &&
523  !(movementInfo.moveFlags & MOVEFLAG_ONTRANSPORT) &&
524  sAnticheatConfig.EnableAntiSpeedHack() &&
525  !_justTeleported)
526  {
527  float allowedDXY = 0.0f;
528  float allowedDZ = 0.0f;
529  float realDistance2D_sq = 0.0f;
530 
531  auto const dt = movementInfo.ctime - GetLastMovementInfo().ctime;
532 
533  // Check vs extrapolation
534  if (sAnticheatConfig.EnableExtrapolation())
535  {
536  Position extrap;
537 
538  // predict destination given the last movement position, direction, and flags, and compare to value reported by the client
539  if (ExtrapolateMovement(GetLastMovementInfo(), dt, extrap))
540  {
541  auto const includeZ = !!((movementInfo.moveFlags | GetLastMovementInfo().moveFlags) & MOVEFLAG_FALLING);
542 
543  // how far our extrapolation predicted they will go (squared)
544  auto const extrapDist2 = DistanceSquared(GetLastMovementInfo().pos, extrap, includeZ);
545 
546  // how far they actually went (squared)
547  auto const theirDist2 = DistanceSquared(GetLastMovementInfo().pos, movementInfo.pos, includeZ);
548 
549  // if they went further than we expected, record it
550  if (theirDist2 > extrapDist2)
551  {
552  constexpr float minErr = 1.f;
553 
554  auto const extrapDist = sqrt(extrapDist2);
555  auto const theirDist = sqrt(theirDist2);
556 
557  auto const delta = theirDist - extrapDist;
558 
559  if (delta > extrapolationEpsilon)
560  {
561  overSpeedDistanceTick += delta;
562  overSpeedDistanceTotal += delta;
563  }
564 
565  if (delta >= minErr)
566  if (auto const anticheat = dynamic_cast<AnticheatLib *>(GetAnticheatLib()))
567  anticheat->OfferExtrapolationData(
568  GetLastMovementInfo(),
569  clientSpeeds[GetMoveType(GetLastMovementInfo().moveFlags)],
570  clientSpeeds[GetMoveType(movementInfo.moveFlags)],
571  movementInfo, extrap, delta);
572  }
573  }
574  }
575  else if (GetMaxAllowedDist(GetLastMovementInfo(), dt, allowedDXY, allowedDZ))
576  {
577  // Allow some margin
578  allowedDXY += 0.5f;
579  allowedDZ += 0.5f;
580 
581  realDistance2D_sq = DistanceSquared(movementInfo.pos, GetLastMovementInfo().pos, false);
582 
583  if (realDistance2D_sq > allowedDXY * allowedDXY * 1.1f)
584  {
585  auto const dist = sqrt(realDistance2D_sq) - allowedDXY;
586 
587  if (dist > extrapolationEpsilon)
588  {
589  overSpeedDistanceTick += dist;
590  overSpeedDistanceTotal += dist;
591  }
592  }
593 
594  if (sAnticheatConfig.IsEnabled(CHEAT_TYPE_OVERSPEED_Z) && fabs(movementInfo.pos.z - GetLastMovementInfo().pos.z) > allowedDZ)
595  _anticheat->RecordCheatInternal(CHEAT_TYPE_OVERSPEED_Z, "z changed from %f to %f, difference of %f, max allowed: %f",
596  GetLastMovementInfo().pos.z, movementInfo.pos.z, fabs(movementInfo.pos.z - GetLastMovementInfo().pos.z), allowedDZ);
597  }
598 
599  // Client should send heartbeats every 500ms at the most (it can send them sooner
600  // if movement flags change, i.e. entering or exiting a transport)
601  if (sAnticheatConfig.IsEnabled(CHEAT_TYPE_SKIPPED_HEARTBEATS) &&
602  dt > 1000 &&
603  GetLastMovementInfo().ctime &&
604  GetLastMovementInfo().moveFlags & MOVEFLAG_MASK_MOVING)
605  _anticheat->RecordCheatInternal(CHEAT_TYPE_SKIPPED_HEARTBEATS, "Time since last movement: %d", dt);
606  }
607 
608  //GetLastMovementInfo() = movementInfo;
609  //GetLastMovementInfo().UpdateTime(WorldTimer::getMSTime());
610  // This is required for proper movement interpolation
611  if (opcode == MSG_MOVE_JUMP)
612  _jumpInitialSpeed = 7.95797334f;
613  else if (opcode == MSG_MOVE_FALL_LAND)
614  {
615  _jumpInitialSpeed = -9.645f;
616  _inKnockBack = false;
617  }
618 
619  if (_justTeleported)
620  {
621  _moveHistory.clear();
622  _justTeleported = false;
623  }
624 
625  _moveHistory.push_back({ static_cast<uint16>(packet.GetOpcode()), movementInfo });
626 
627  return true;

737 bool Movement::ExtrapolateMovement(MovementInfo const& mi, uint32 diffMs, Position &pos) const
738 {
739  // TODO: These cases are not handled in movement extrapolation
740  // - Transports
741  if (mi.moveFlags & (MOVEFLAG_PITCH_UP | MOVEFLAG_PITCH_DOWN | MOVEFLAG_ONTRANSPORT))
742  return false;
743  // - Not correctly handled yet (issues regarding feather fall auras)
744  if (mi.moveFlags & MOVEFLAG_FALLINGFAR)
745  return false;
746  // - Server side movement (should be easy to extrapolate actually)
747  if (!_me->movespline->Finalized())
748  return false;
749  // Dernier paquet pas a jour (connexion, TP autre map ...)
750  if (mi.ctime == 0)
751  return false;
752 
753  // if we are not moving, position shouldn't change
754  if (!(mi.moveFlags & MOVEFLAG_MASK_MOVING))
755  {
756  pos = mi.pos;
757  return true;
758  }
759 
760  pos = mi.pos;
761 
762  float o = pos.o;
763 
764  // Not allowed to move
765  /*if (mi.moveFlags & MOVE_ROOT)
766  return true;*/
767 
768  auto const speed = clientSpeeds[GetMoveType(mi.moveFlags)];
769 
770  if (mi.moveFlags & MOVEFLAG_BACKWARD)
771  o += M_PI_F;
772  else if (mi.moveFlags & MOVEFLAG_STRAFE_LEFT)
773  {
774  if (mi.moveFlags & MOVEFLAG_FORWARD)
775  o += M_PI_F / 4;
776  else
777  o += M_PI_F / 2;
778  }
779  else if (mi.moveFlags & MOVEFLAG_STRAFE_RIGHT)
780  {
781  if (mi.moveFlags & MOVEFLAG_FORWARD)
782  o -= M_PI_F / 4;
783  else
784  o -= M_PI_F / 2;
785  }
786 
787  if (mi.moveFlags & MOVEFLAG_FALLING)
788  {
789  float diffT = WorldTimer::getMSTimeDiff(mi.jump.startClientTime, diffMs + mi.ctime) / 1000.f;
790  pos.x = mi.jump.start.x;
791  pos.y = mi.jump.start.y;
792  pos.z = mi.jump.start.z;
793  // Fatal error. Avoid crashing here ...
794  if (!pos.x || !pos.y || !pos.z || diffT > 10000.f)
795  return false;
796  pos.x += mi.jump.cosAngle * mi.jump.xyspeed * diffT;
797  pos.y += mi.jump.sinAngle * mi.jump.xyspeed * diffT;
798  pos.z -= ::Movement::computeFallElevation(diffT, mi.moveFlags & MOVEFLAG_SAFE_FALL, -_jumpInitialSpeed);
799  }
800  else if (mi.moveFlags & (MOVEFLAG_TURN_LEFT | MOVEFLAG_TURN_RIGHT))
801  {
802  if (mi.moveFlags & MOVEFLAG_MASK_MOVING)
803  {
804  // Every 2 sec
805  float T = 0.75f * (clientSpeeds[MOVE_TURN_RATE]) * (diffMs / 1000.f);
806  float R = 1.295f * speed / M_PI * cos(mi.s_pitch);
807  pos.z += diffMs * speed / 1000.0f * sin(mi.s_pitch);
808  // Find the center of the circle we are moving on
809  if (mi.moveFlags & MOVEFLAG_TURN_LEFT)
810  {
811  pos.x += R * cos(o + M_PI / 2);
812  pos.y += R * sin(o + M_PI / 2);
813  pos.o += T;
814  T = T - M_PI / 2.0f;
815  }
816  else
817  {
818  pos.x += R * cos(o - M_PI / 2);
819  pos.y += R * sin(o - M_PI / 2);
820  pos.o -= T;
821  T = -T + M_PI / 2.0f;
822  }
823  pos.x += R * cos(o + T);
824  pos.y += R * sin(o + T);
825  }
826  else
827  {
828  float diffO = clientSpeeds[MOVE_TURN_RATE] * diffMs / 1000.f;
829  if (mi.moveFlags & MOVEFLAG_TURN_LEFT)
830  pos.o += diffO;
831  else
832  pos.o -= diffO;
833  return true;
834  }
835  }
836  else if (mi.moveFlags & MOVEFLAG_MASK_MOVING)
837  {
838  auto const dist = speed * diffMs / 1000.f;
839 
840  pos.x += dist * cos(o) * cos(mi.s_pitch);
841  pos.y += dist * sin(o) * cos(mi.s_pitch);
842  pos.z += dist * sin(mi.s_pitch);
843  }
844  else // If we reach here, we did not move
845  return true;
846 
847  if (!MaNGOS::IsValidMapCoord(pos.x, pos.y, pos.z, pos.o))
848  return false;
849 
850  if (!(mi.moveFlags & (MOVEFLAG_FALLING | MOVEFLAG_FALLINGFAR | MOVEFLAG_SWIMMING | MOVEFLAG_WATERWALKING)))
851  pos.z = _me->GetMap()->GetHeight(_me->GetPhaseMask(), pos.x, pos.y, pos.z);
852 
853  return _me->GetMap()->IsInLineOfSight(mi.pos.x, mi.pos.y, mi.pos.z + 0.5f, pos.x, pos.y, pos.z + 0.5f, _me->GetPhaseMask(), false);
854 }
