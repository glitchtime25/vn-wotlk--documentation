58  void UpdateAI(const uint32 uiDiff) override
59  {
60  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
61  return;
62 
63  // Arcane Missiles Timer
64  if (m_uiArcaneMissilesTimer < uiDiff)
65  {
66  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ARCANE_MISSILES) == CAST_OK)
67  m_uiArcaneMissilesTimer = 8000;
68  }
69  else
70  m_uiArcaneMissilesTimer -= uiDiff;
71 
72  // Shadow Shield Timer
73  if (m_uiShadowShieldTimer < uiDiff)
74  {
75  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_SHIELD) == CAST_OK)
76  m_uiShadowShieldTimer = urand(14000, 28000);
77  }
78  else
79  m_uiShadowShieldTimer -= uiDiff;
80 
81  // Curse Timer
82  if (m_uiCurseTimer < uiDiff)
83  {
84  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CURSE_SD) == CAST_OK)
85  m_uiCurseTimer = urand(15000, 27000);
86  }
87  else
88  m_uiCurseTimer -= uiDiff;
89 
90  // Teleporting Random Target to one of the six pre boss rooms and spawn 3-4 skeletons near the gamer.
91  // We will only telport if gandling has more than 3% of hp so teleported gamers can always loot.
92  if (m_creature->GetHealthPercent() > 3.0f)
93  {
94  if (m_uiTeleportTimer < uiDiff)
95  {
96  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SHADOW_PORTAL, SELECT_FLAG_PLAYER))
97  {
98  if (DoCastSpellIfCan(pTarget, SPELL_SHADOW_PORTAL) == CAST_OK)
99  m_uiTeleportTimer = urand(20000, 35000);
100  }
101  }
102  else
103  m_uiTeleportTimer -= uiDiff;
104  }
105 
106  DoMeleeAttackIfReady();
107  }
