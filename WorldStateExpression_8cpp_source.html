27 void HexStrToByteArray(std::string_view str, uint8* out, size_t outlen, bool reverse)
28 {
29  MANGOS_ASSERT(str.size() == (2 * outlen));
30 
31  int32 init = 0;
32  int32 end = int32(str.length());
33  int8 op = 1;
34 
35  if (reverse)
36  {
37  init = int32(str.length() - 2);
38  end = -2;
39  op = -1;
40  }
41 
42  uint32 j = 0;
43  for (int32 i = init; i != end; i += 2 * op)
44  {
45  char buffer[3] = { str[i], str[i + 1], '\0' };
46  out[j++] = uint8(strtoul(buffer, nullptr, 16));
47  }
48 }

311 int32 EvalSingleValue(ByteBuffer& buffer, Map const* map)
312 {
313  WorldStateExpressionValueType valueType = buffer.read<WorldStateExpressionValueType>();
314  int32 value = 0;
315 
316  switch (valueType)
317  {
318  case WorldStateExpressionValueType::CONSTANT:
319  {
320  value = buffer.read<int32>();
321  break;
322  }
323  case WorldStateExpressionValueType::WORLDSTATE:
324  {
325  uint32 worldStateId = buffer.read<uint32>();
326  value = map->GetVariableManager().GetVariable(worldStateId);
327  break;
328  }
329  case WorldStateExpressionValueType::FUNCTION:
330  {
331  uint32 functionType = buffer.read<uint32>();
332  int32 arg1 = EvalSingleValue(buffer, map);
333  int32 arg2 = EvalSingleValue(buffer, map);
334 
335  if (functionType >= WSE_FUNCTION_MAX)
336  return 0;
337 
338  value = WorldStateExpressionFunctions[functionType](map, arg1, arg2);
339  break;
340  }
341  default:
342  break;
343  }
344 
345  return value;
346 }

348 int32 EvalValue(ByteBuffer& buffer, Map const* map)
349 {
350  int32 leftValue = EvalSingleValue(buffer, map);
351 
352  WorldStateExpressionOperatorType operatorType = buffer.read<WorldStateExpressionOperatorType>();
353  if (operatorType == WorldStateExpressionOperatorType::NONE)
354  return leftValue;
355 
356  int32 rightValue = EvalSingleValue(buffer, map);
357 
358  switch (operatorType)
359  {
360  case WorldStateExpressionOperatorType::SUM: return leftValue + rightValue;
361  case WorldStateExpressionOperatorType::SUBTRACTION: return leftValue - rightValue;
362  case WorldStateExpressionOperatorType::MULTIPLICATION: return leftValue * rightValue;
363  case WorldStateExpressionOperatorType::DIVISION: return !rightValue ? 0 : leftValue / rightValue;
364  case WorldStateExpressionOperatorType::REMAINDER: return !rightValue ? 0 : leftValue % rightValue;
365  default:
366  break;
367  }
368 
369  return leftValue;
370 }

372 bool EvalRelOp(ByteBuffer& buffer, Map const* map)
373 {
374  int32 leftValue = EvalValue(buffer, map);
375 
376  ConditionOperation compareLogic = buffer.read<ConditionOperation>();
377  if (compareLogic == ConditionOperation::NONE)
378  return leftValue != 0;
379 
380  int32 rightValue = EvalValue(buffer, map);
381 
382  return ConditionEntry::CheckOp(compareLogic, leftValue, rightValue);

423 {

449  case ConditionLogic::OR: finalResult = finalResult || secondResult; break;

451  default:

453  }
454 
455  if (buffer.rpos() < buffer.size())
456  break;
457 
458  resultLogic = buffer.read<ConditionLogic>();
459  }
460 
461  return finalResult;
