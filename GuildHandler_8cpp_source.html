74  recvPacket >> Invitedname;

78

88  SendGuildCommandResult(GUILD_CREATE_S, "", ERR_GUILD_PLAYER_NOT_IN_GUILD);

90  }
91 
92  // OK result but not send invite
93  if (player->GetSocial()->HasIgnore(GetPlayer()->GetObjectGuid()))
94  return;
95 
96  // not let enemies sign guild charter

98  {
99  SendGuildCommandResult(GUILD_INVITE_S, Invitedname, ERR_GUILD_NOT_ALLIED);
100  return;
101  }
102 
103  if (player->GetGuildId())
104  {
105  plname = player->GetName();
106  SendGuildCommandResult(GUILD_INVITE_S, plname, ERR_ALREADY_IN_GUILD_S);
107  return;
108  }
109 
110  if (player->GetGuildIdInvited())
111  {
112  plname = player->GetName();
113  SendGuildCommandResult(GUILD_INVITE_S, plname, ERR_ALREADY_INVITED_TO_GUILD_S);
114  return;
115  }
116 
117  if (!guild->HasRankRight(GetPlayer()->GetRank(), GR_RIGHT_INVITE))
118  {
119  SendGuildCommandResult(GUILD_INVITE_S, "", ERR_GUILD_PERMISSIONS);
120  return;
121  }
122 
123  DEBUG_LOG("Player %s Invited %s to Join his Guild", GetPlayer()->GetName(), Invitedname.c_str());
124 
125  player->SetGuildIdInvited(GetPlayer()->GetGuildId());
126  // Put record into guildlog
127  guild->LogGuildEvent(GUILD_EVENT_LOG_INVITE_PLAYER, GetPlayer()->GetObjectGuid(), player->GetObjectGuid());
128 
129  WorldPacket data(SMSG_GUILD_INVITE, (8 + 10)); // guess size
130  data << GetPlayer()->GetName();
131  data << guild->GetName();
132  player->GetSession()->SendPacket(data);
133 
134  DEBUG_LOG("WORLD: Sent (SMSG_GUILD_INVITE)");
135 }

596 void WorldSession::HandleGuildRankOpcode(WorldPacket& recvPacket)
597 {
598  std::string rankname;
599  uint32 rankId;
600  uint32 rights, MoneyPerDay;
601 
602  DEBUG_LOG("WORLD: Received opcode CMSG_GUILD_RANK");
603 
604  Guild* guild = sGuildMgr.GetGuildById(GetPlayer()->GetGuildId());
605  if (!guild)
606  {
607  recvPacket.rpos(recvPacket.wpos()); // set to end to avoid warnings spam
608  SendGuildCommandResult(GUILD_CREATE_S, "", ERR_GUILD_PLAYER_NOT_IN_GUILD);
609  return;
610  }
611 
612  if (GetPlayer()->GetObjectGuid() != guild->GetLeaderGuid())
613  {
614  recvPacket.rpos(recvPacket.wpos()); // set to end to avoid warnings spam
615  SendGuildCommandResult(GUILD_INVITE_S, "", ERR_GUILD_PERMISSIONS);
616  return;
617  }
618 
619  recvPacket >> rankId;
620  recvPacket >> rights;
621  recvPacket >> rankname;
622  recvPacket >> MoneyPerDay;
623 
624  for (int i = 0; i < GUILD_BANK_MAX_TABS; ++i)
625  {
626  uint32 BankRights;
627  uint32 BankSlotPerDay;
628 
629  recvPacket >> BankRights;
630  recvPacket >> BankSlotPerDay;
631  guild->SetBankRightsAndSlots(rankId, uint8(i), uint16(BankRights & 0xFF), uint16(BankSlotPerDay), true);
632  }
633 
634  DEBUG_LOG("WORLD: Changed RankName to %s , Rights to 0x%.4X", rankname.c_str(), rights);
635 
636  guild->SetBankMoneyPerDay(rankId, MoneyPerDay);
637  guild->SetRankName(rankId, rankname);
638 
639  if (rankId == GR_GUILDMASTER) // prevent loss leader rights
640  rights = GR_RIGHT_ALL;
641 
642  guild->SetRankRights(rankId, rights);
643 
644  guild->Query(this);
645  guild->Roster(); // broadcast for tab rights update
646 }
