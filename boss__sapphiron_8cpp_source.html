116  void Reset() override
117  {

142  void Aggro(Unit* /*who*/) override
143  {

149  void MovementInform(uint32 type, uint32 /*pointId*/) override
150  {
151  if (type == POINT_MOTION_TYPE && m_phase == PHASE_LIFT_OFF)
152  {
153  DoCastSpellIfCan(m_creature, SPELL_SUMMON_WING_BUFFET);
154 
155  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
156  DoBroadcastText(EMOTE_SAPPHIRON_FLY, m_creature);
157  m_creature->SetHover(true);
158  m_creature->CastSpell(nullptr, SPELL_DRAGON_HOVER, TRIGGERED_OLD_TRIGGERED);
159  SetCombatScriptStatus(false);
160  m_creature->SetSpellList(m_isRegularMode ? SPELLSET_AIR_10N : SPELLSET_AIR_25N);
161  }
162  }
163 
164  void ExecuteAction(uint32 action) override
165  {
166  switch(action)
167  {
168  case SAPPHIRON_BERSERK:
169  {
170  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
171  {
172  DoBroadcastText(EMOTE_GENERIC_ENRAGED, m_creature);
173  DisableCombatAction(action);
174  }
175  return;
176  }
177  case SAPPHIRON_AIR_PHASE:
178  {
179  if (m_creature->GetHealthPercent() > 10.f)
180  {
181  m_phase = PHASE_LIFT_OFF;
182  SetCombatScriptStatus(true);
183  SetDeathPrevention(true);
184  m_creature->InterruptNonMeleeSpells(false);
185  SetCombatMovement(false);
186  SetMeleeEnabled(false);
187  m_creature->AttackStop(false, true);
188  m_creature->SetTarget(nullptr);
189  m_creature->GetMotionMaster()->MovePoint(1, aLiftOffPosition[0], aLiftOffPosition[1], aLiftOffPosition[2], FORCED_MOVEMENT_RUN);
190  }
191  DisableCombatAction(SAPPHIRON_AIR_PHASE);
192  return;
193  }
194  case SAPPHIRON_LANDING_PHASE:
195  {
196  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
197  DoBroadcastText(EMOTE_SAPPHIRON_LAND, m_creature);
198  if (Creature* buffetDummy = GetClosestCreatureWithEntry(m_creature, NPC_WING_BUFFET, 10.f))
199  buffetDummy->ForcedDespawn(); // Despawn Dummy manually due to different Air-Phase durations between 10 and 25 versions
200  ResetCombatAction(SAPPHIRON_GROUND_PHASE, 2s);
201  DisableCombatAction(SAPPHIRON_LANDING_PHASE);
202  return;
203  }
204  case SAPPHIRON_GROUND_PHASE:
205  {
206  m_phase = PHASE_GROUND;
207  m_creature->SetSpellList(m_isRegularMode ? SPELLSET_10N : SPELLSET_25N);
208  SetDeathPrevention(false);
209  SetCombatMovement(true);
210  SetMeleeEnabled(true);
211  m_creature->RemoveAurasDueToSpell(SPELL_DRAGON_HOVER);
212  m_creature->SetHover(false);
213  m_creature->GetMotionMaster()->Clear(false);
214  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
215 
216  ResetCombatAction(SAPPHIRON_AIR_PHASE, GetSubsequentActionTimer(SAPPHIRON_AIR_PHASE));
217  DisableCombatAction(SAPPHIRON_GROUND_PHASE);
218  return;
219  }
220  }
221  ResetCombatAction(action, GetSubsequentActionTimer(action));
222  }

226 {

228 
229  if (!instance)

244 struct IceBolt : public SpellScript
245 {
246  void OnEffectExecute(Spell* spell, SpellEffectIndex /* effIdx */) const override
247  {
248  Creature* caster = static_cast<Creature*>(spell->GetCaster());
249  if (!caster)
250  return;
251  boss_sapphironAI* boss_ai = dynamic_cast<boss_sapphironAI*>(caster->AI());
252  if (!boss_ai)
253  return;
254 
255  if (Unit* target = caster->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_ICEBOLT, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA))
256  caster->CastSpell(target, SPELL_ICEBOLT, TRIGGERED_NONE);
257 
258  if (boss_ai->m_iceboltCount >= (boss_ai->m_isRegularMode ? 2 : 3))
259  {
260  caster->CastSpell(nullptr, SPELL_FROST_BREATH, TRIGGERED_IGNORE_COOLDOWNS | TRIGGERED_IGNORE_GCD);
261  caster->CastSpell(nullptr, SPELL_FROST_BREATH_DUMMY, TRIGGERED_IGNORE_COOLDOWNS | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_IGNORE_GCD);
262  DoBroadcastText(EMOTE_BREATH, caster);
263  caster->SetSpellList(boss_ai->m_isRegularMode ? SPELLSET_NULL_10N : SPELLSET_NULL_25N);
264  boss_ai->ResetCombatAction(SAPPHIRON_LANDING_PHASE, 10s);
265  }
266 
267  boss_ai->m_iceboltCount += 1;
268  }
