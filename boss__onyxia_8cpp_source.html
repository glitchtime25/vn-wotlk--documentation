312  void UpdateAI(const uint32 uiDiff) override
313  {
314  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
315  return;
316 
317  switch (m_uiPhase)
318  {
319  case PHASE_END: // Here is room for additional summoned whelps and Erruption
320  if (m_uiBellowingRoarTimer < uiDiff)
321  {
322  if (DoCastSpellIfCan(m_creature, SPELL_BELLOWINGROAR) == CAST_OK)
323  m_uiBellowingRoarTimer = urand(15000, 45000);
324  }
325  else
326  m_uiBellowingRoarTimer -= uiDiff;
327  // no break, phase 3 will use same abilities as in 1
328  case PHASE_START:
329  {
330  if (m_uiFlameBreathTimer < uiDiff)
331  {
332  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_FLAMEBREATH : SPELL_FLAMEBREATH_H) == CAST_OK)
333  m_uiFlameBreathTimer = urand(10000, 20000);
334  }
335  else
336  m_uiFlameBreathTimer -= uiDiff;
337 
338  if (m_uiTailSweepTimer < uiDiff)
339  {
340  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_TAILSWEEP : SPELL_TAILSWEEP_H) == CAST_OK)
341  m_uiTailSweepTimer = urand(15000, 20000);
342  }
343  else
344  m_uiTailSweepTimer -= uiDiff;
345 
346  if (m_uiCleaveTimer < uiDiff)
347  {
348  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
349  m_uiCleaveTimer = urand(2000, 5000);
350  }
351  else
352  m_uiCleaveTimer -= uiDiff;
353 
354  if (m_uiWingBuffetTimer < uiDiff)
355  {
356  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_WINGBUFFET : SPELL_WINGBUFFET_H) == CAST_OK)
357  m_uiWingBuffetTimer = urand(15000, 30000);
358  }
359  else
360  m_uiWingBuffetTimer -= uiDiff;
361 
362  if (m_uiCheckInLairTimer < uiDiff)
363  {
364  if (m_pInstance)
365  {
366  Creature* pOnyTrigger = m_pInstance->GetSingleCreatureFromStorage(NPC_ONYXIA_TRIGGER);
367  if (pOnyTrigger && !m_creature->IsWithinDistInMap(pOnyTrigger, 90.0f, false))
368  {
369  if (!m_bHasYelledLured)
370  {
371  m_bHasYelledLured = true;
372  DoScriptText(SAY_KITE, m_creature);
373  }
374  DoCastSpellIfCan(m_creature, SPELL_BREATH_ENTRANCE);
375  }
376  else
377  m_bHasYelledLured = false;
378  }
379  m_uiCheckInLairTimer = 3000;
380  }
381  else
382  m_uiCheckInLairTimer -= uiDiff;
383 
384  if (m_uiPhase == PHASE_START && m_creature->GetHealthPercent() < 65.0f)
385  {
386  m_uiPhase = PHASE_TO_LIFTOFF;
387  DoScriptText(SAY_PHASE_2_TRANS, m_creature);
388  SetCombatMovement(false);
389  m_creature->GetMotionMaster()->MoveIdle();
390  m_creature->SetTarget(nullptr);
391 
392  float fGroundZ = m_creature->GetMap()->GetHeight(m_creature->GetPhaseMask(), aMoveData[POINT_ID_SOUTH].fX, aMoveData[POINT_ID_SOUTH].fY, aMoveData[POINT_ID_SOUTH].fZ);
393  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LIFTOFF, aMoveData[POINT_ID_SOUTH].fX, aMoveData[POINT_ID_SOUTH].fY, fGroundZ);
394  return;
395  }
396 
397  DoMeleeAttackIfReady();
398  break;
399  }
400  case PHASE_BREATH:
401  {
402  if (m_creature->GetHealthPercent() < 40.0f)
403  {
404  m_uiPhase = PHASE_BREATH_POST;
405  DoScriptText(SAY_PHASE_3_TRANS, m_creature);
406 
407  float fGroundZ = m_creature->GetMap()->GetHeight(m_creature->GetPhaseMask(), m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ());
408  m_creature->GetMotionMaster()->MovePointTOL(POINT_ID_LAND, m_creature->GetPositionX(), m_creature->GetPositionY(), fGroundZ, false);
409  return;
410  }
411 
412  if (m_uiMovementTimer < uiDiff)
413  {
414  // 3 possible actions
415  switch (urand(0, 2))
416  {
417  case 0: // breath
418  DoScriptText(EMOTE_BREATH, m_creature);
419  DoCastSpellIfCan(m_creature, aMoveData[m_uiMovePoint].uiSpellId, CAST_INTERRUPT_PREVIOUS);
420  m_uiMovePoint += NUM_MOVE_POINT / 2;
421  m_uiMovePoint %= NUM_MOVE_POINT;
422  m_uiMovementTimer = 25000;
423  return;
424  case 1: // a point on the left side
425  {
426  // C++ is stupid, so add -1 with +7
427  m_uiMovePoint += NUM_MOVE_POINT - 1;
428  m_uiMovePoint %= NUM_MOVE_POINT;
429  break;
430  }
431  case 2: // a point on the right side
432  m_uiMovePoint = (m_uiMovePoint + 1) % NUM_MOVE_POINT;
433  break;
434  }
435 
436  m_uiMovementTimer = urand(15000, 25000);
437  m_creature->GetMotionMaster()->MovePoint(m_uiMovePoint, aMoveData[m_uiMovePoint].fX, aMoveData[m_uiMovePoint].fY, aMoveData[m_uiMovePoint].fZ);
438  }
439  else
440  m_uiMovementTimer -= uiDiff;
441 
442  if (m_uiFireballTimer < uiDiff)
443  {
444  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
445  {
446  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_FIREBALL : SPELL_FIREBALL_H) == CAST_OK)
447  m_uiFireballTimer = urand(3000, 5000);
448  }
449  }
450  else
451  m_uiFireballTimer -= uiDiff; // engulfingflames is supposed to be activated by a fireball but haven't come by
452 
453  if (m_bIsSummoningWhelps)
454  {
455  if (DidSummonWhelps(uiDiff))
456  {
457  m_bIsSummoningWhelps = false;
458  m_uiSummonCount = 0;
459  m_uiSummonWhelpsTimer = 80000; // 90s - 10s for summoning
460  }
461  }
462  else
463  {
464  if (m_uiSummonWhelpsTimer < uiDiff)
465  m_bIsSummoningWhelps = true;
466  else
467  m_uiSummonWhelpsTimer -= uiDiff;
468  }
469 
470  if (m_uiSummonGuardTimer < uiDiff)
471  {
472  if (!m_creature->IsNonMeleeSpellCasted(false))
473  {
474  m_creature->CastSpell(afSpawnLocations[2][0], afSpawnLocations[2][1], afSpawnLocations[2][2], SPELL_SUMMON_LAIR_GUARD, TRIGGERED_OLD_TRIGGERED);
475  m_uiSummonGuardTimer = 30000;
476  }
477  }
478  else
479  m_uiSummonGuardTimer -= uiDiff;
480 
481  break;
482  }
483  case PHASE_BREATH_PRE: // Summon first rounds of whelps
484  DidSummonWhelps(uiDiff);
485  // no break here
486  default: // Phase-switching phases
487  if (!m_uiPhaseTimer)
488  break;
489  if (m_uiPhaseTimer <= uiDiff)
490  {
491  switch (m_uiPhase)
492  {
493  case PHASE_TO_LIFTOFF:
494  m_uiPhase = PHASE_BREATH_PRE;
495  if (m_pInstance)
496  m_pInstance->SetData(TYPE_ONYXIA, DATA_LIFTOFF);
497  m_creature->GetMotionMaster()->MovePointTOL(POINT_ID_IN_AIR, aMoveData[POINT_ID_SOUTH].fX, aMoveData[POINT_ID_SOUTH].fY, aMoveData[POINT_ID_SOUTH].fZ, true);
498  break;
499  case PHASE_BREATH_PRE:
500  m_creature->GetMotionMaster()->MovePoint(POINT_ID_INIT_NORTH, aMoveData[POINT_ID_NORTH].fX, aMoveData[POINT_ID_NORTH].fY, aMoveData[POINT_ID_NORTH].fZ);
501  break;
502  case PHASE_BREATH_POST:
503  m_uiPhase = PHASE_END;
504  m_creature->SetTarget(m_creature->GetVictim());
505  SetCombatMovement(true, true);
506  DoCastSpellIfCan(m_creature, SPELL_BELLOWINGROAR);
507  break;
508  }
509  m_uiPhaseTimer = 0;
510  }
511  else
512  m_uiPhaseTimer -= uiDiff;
513  break;
514  }
515  }
