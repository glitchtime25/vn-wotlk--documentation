214  void SwitchPhases()
215  {
216  if (m_uiActivePhase == BEAM_PHASE)
217  {
218  if (DoCastSpellIfCan(m_creature, SPELL_SHADOWFORM, CAST_TRIGGERED) == CAST_OK)
219  {
220  m_creature->RemoveAurasDueToSpell(SPELL_EMPOWERMENT);
221 #ifdef PRENERF_2_0_3

224 #endif
225  DoCastSpellIfCan(m_creature, SPELL_NETHERSPITE_ROAR);
226 
227  SetCombatMovement(false);
228  SetMeleeEnabled(false);
229 
230  m_uiActivePhase = BANISH_PHASE;
231  DoScriptText(EMOTE_PHASE_BANISH, m_creature);
232 
233  DisableCombatAction(NETHERSPITE_EMPOWERMENT);
234  DisableCombatAction(NETHERSPITE_VOID_ZONE);
235  ResetCombatAction(NETHERSPITE_FACE_RANDOM_TARGET, 5000);
236  ResetCombatAction(NETHERSPITE_NETHERBREATH, 5000);
237  ResetCombatAction(NETHERSPITE_DUMMY_NUKE, 6000);
238  ResetCombatAction(NETHERSPITE_PHASE_CHANGE, 30000);
239 
240  DespawnPortals();
241  }
242  }
243  else
244  {
245 #ifdef PRENERF_2_0_3
246 #else
247  DoCastSpellIfCan(m_creature, SPELL_NETHERBURN, CAST_TRIGGERED);
248 #endif
249  m_creature->RemoveAurasDueToSpell(SPELL_SHADOWFORM);
250  DoCastSpellIfCan(m_creature, SPELL_NETHERSPITE_ROAR);
251  SetCombatMovement(true);
252  SetMeleeEnabled(true);
253  m_creature->FixateTarget(nullptr);
254 
255  if (m_creature->GetVictim())
256  {
257  m_creature->SetTarget(m_creature->GetVictim());
258  DoStartMovement(m_creature->GetVictim());
259  }
260 
261  m_uiActivePhase = BEAM_PHASE;
262  DoScriptText(EMOTE_PHASE_BEAM, m_creature);
263 
264  DoSummonPortals();
265  DisableCombatAction(NETHERSPITE_FACE_RANDOM_TARGET);
266  DisableCombatAction(NETHERSPITE_NETHERBREATH);
267  DisableCombatAction(NETHERSPITE_DUMMY_NUKE);

271  }

304  void ExecuteAction(uint32 action) override
305  {
306  switch (action)
307  {
308  case NETHERSPITE_PHASE_CHANGE:
309  {
310  SwitchPhases();
311  break;
312  }
313  case NETHERSPITE_ENRAGE:
314  {
315  if (DoCastSpellIfCan(m_creature, SPELL_NETHER_INFUSION) == CAST_OK)
316  DisableCombatAction(action);
317  break;
318  }
319  case NETHERSPITE_EMPOWERMENT:
320  {
321  if (DoCastSpellIfCan(m_creature, SPELL_EMPOWERMENT) == CAST_OK)
322  {
323  DoCastSpellIfCan(m_creature, SPELL_PORTAL_ATTUNEMENT, CAST_TRIGGERED);
324  DisableCombatAction(action);
325  }
326  break;
327  }
328  case NETHERSPITE_VOID_ZONE:
329  {
330  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
331  if (DoCastSpellIfCan(target, SPELL_VOID_ZONE) == CAST_OK)
332  ResetCombatAction(action, 15000);
333  break;
334  }
335  case NETHERSPITE_FACE_RANDOM_TARGET:
336  {
337  if (DoCastSpellIfCan(nullptr, SPELL_FACE_RANDOM_TARGET) == CAST_OK)
338  ResetCombatAction(action, 15000);
339  break;
340  }
341  case NETHERSPITE_NETHERBREATH:
342  {
343  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_NETHERBREATH, SELECT_FLAG_PLAYER))
344  if (DoCastSpellIfCan(target, SPELL_NETHERBREATH) == CAST_OK)
345  ResetCombatAction(action, 6000);
346  break;
347  }
348  case NETHERSPITE_DUMMY_NUKE:
349  {
350  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_DUMMY_NUKE, SELECT_FLAG_PLAYER))
351  if (DoCastSpellIfCan(target, SPELL_DUMMY_NUKE) == CAST_OK)
352  ResetCombatAction(action, 2000);
353  break;
354  }
355  }
