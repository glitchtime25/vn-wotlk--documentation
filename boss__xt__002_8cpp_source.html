216  void UpdateAI(const uint32 uiDiff) override
217  {
218  if (!m_pInstance)
219  {
220  script_error_log("Instance Ulduar: ERROR Failed to load instance data for this instace.");
221  return;
222  }
223 
224  // The heart needs to be mounted manually, not by vehicle_accessories
225  if (m_uiMountTimer)
226  {
227  if (m_uiMountTimer <= uiDiff)
228  {
229  if (Creature* pHeart = m_pInstance->GetSingleCreatureFromStorage(NPC_HEART_DECONSTRUCTOR))
230  {
231  // safeguard in case the Heart isn't respawned
232  if (!pHeart->IsAlive())
233  pHeart->Respawn();
234 
235  pHeart->AI()->EnterEvadeMode();
236  m_creature->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
237  pHeart->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
238  pHeart->CastSpell(m_creature, SPELL_HEART_RIDE_VEHICLE, TRIGGERED_OLD_TRIGGERED);
239  }
240 
241  m_uiMountTimer = 0;
242  }
243  else
244  m_uiMountTimer -= uiDiff;
245  }
246 
247  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
248  return;
249 
250  if (m_uiBerserkTimer)
251  {
252  if (m_uiBerserkTimer <= uiDiff)
253  {
254  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
255  {
256  DoScriptText(SAY_BERSERK, m_creature);
257  m_uiBerserkTimer = 0;
258  }
259  }
260  else
261  m_uiBerserkTimer -= uiDiff;
262  }
263 
264  switch (m_uiPhase)
265  {
266  case PHASE_NORMAL:
267 
268  if (m_uiLightBombTimer < uiDiff)
269  {
270  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SEARING_LIGHT : SPELL_SEARING_LIGHT_H) == CAST_OK)
271  m_uiLightBombTimer = 20000;
272  }
273  else
274  m_uiLightBombTimer -= uiDiff;
275 
276  if (m_uiGravityBombTimer < uiDiff)
277  {
278  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_GRAVITY_BOMB : SPELL_GRAVITY_BOMB_H) == CAST_OK)
279  m_uiGravityBombTimer = 20000;
280  }
281  else
282  m_uiGravityBombTimer -= uiDiff;
283 
284  if (m_uiTanctrumTimer < uiDiff)
285  {
286  if (DoCastSpellIfCan(m_creature, SPELL_TYMPANIC_TANTRUM) == CAST_OK)
287  {
288  DoScriptText(SAY_TANCTRUM, m_creature);
289  DoScriptText(EMOTE_EARTH_QUAKE, m_creature);
290  m_uiTanctrumTimer = 60000;
291  }
292  }
293  else
294  m_uiTanctrumTimer -= uiDiff;
295 
296  // start heart stage transition
297  if (m_creature->GetHealthPercent() < float(100 - 25 * m_uiHeartStage))
298  {
299  DoScriptText(SAY_HEART_OPEN, m_creature);
300 
301  ++m_uiHeartStage;
302  m_uiHeartTimer = 5000;
303  m_uiPhase = PHASE_TRANSITION;
304 
305  // stop all movement
306  m_creature->GetMotionMaster()->MoveIdle();
307  }
308 
309  DoMeleeAttackIfReady();
310 
311  break;
312  case PHASE_TRANSITION:
313 
314  if (m_uiHeartTimer < uiDiff)
315  {
316  // inform the heart about the phase switch
317  if (Creature* pHeart = m_pInstance->GetSingleCreatureFromStorage(NPC_HEART_DECONSTRUCTOR))
318  SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, pHeart);
319 
320  DoScriptText(EMOTE_HEART, m_creature);
321 
322  m_creature->SetStandState(UNIT_STAND_STATE_CUSTOM);
323  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
324 
325  m_uiHeartTimer = 30000;
326  m_uiPhase = PHASE_HEART;
327  }
328  else
329  m_uiHeartTimer -= uiDiff;
330 
331  break;
332  case PHASE_HEART:
333 
334  // reset to normal phase when timer expires
335  if (m_uiHeartTimer < uiDiff)
336  {
337  DoResetToNormalPhase();
338  m_uiHeartTimer = 0;
339 
340  // mount the heart back inside if not already killed
341  if (m_pInstance && m_pInstance->GetData(TYPE_XT002_HARD) != DONE)
342  {
343  if (Creature* pHeart = m_pInstance->GetSingleCreatureFromStorage(NPC_HEART_DECONSTRUCTOR))
344  {
345  pHeart->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
346  m_creature->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
347  pHeart->CastSpell(m_creature, SPELL_HEART_RIDE_VEHICLE, TRIGGERED_OLD_TRIGGERED);
348 
349  // no spell found for this
350  pHeart->SetHealth(pHeart->GetMaxHealth());
351  }
352  }
353  }
354  else
355  m_uiHeartTimer -= uiDiff;
356 
357  break;
358  }
359  }
