45 template<class T, typename D>
46 bool TargetedMovementGeneratorMedium<T, D>::Update(T& owner, const uint32& time_diff)
47 {
48  if (!i_target.isValid() || !i_target->IsInWorld())
49  return !static_cast<D*>(this)->RemoveOnInvalid();
50 
51  // Trying to detect error
52  if (i_target->GetMap() != owner.GetMap())
53  {
54  sLog.outCustomLog("TargetedMovementGeneratorMedium::Update(): Target %s left map id %u for map id %u out of order!",
55  i_target->GetGuidStr().c_str(), i_target->GetMapId(), owner.GetMapId());
56  return !static_cast<D*>(this)->RemoveOnInvalid();
57  }
58 
59  if (i_target->GetTransport() != owner.GetTransport())
60  {
61  HandleMovementFailure(owner);
62  return true;
63  }
64 
65  if (!owner.IsAlive())
66  return true;
67 
68  // prevent movement while casting spells with cast time or channel time
69  if (owner.IsNonMeleeSpellCasted(false, false, true, true))
70  {
71  if (!owner.movespline->Finalized())
72  {
73  if (owner.IsClientControlled())
74  owner.StopMoving(true);
75  else
76  owner.InterruptMoving();
77  }
78  return true;
79  }
80 
81  if (_hasUnitStateNotMove(owner) || owner.IsImmobilizedState())
82  {
83  HandleMovementFailure(owner);
84  return true;
85  }
86 
87  // prevent crash after creature killed pet
88  if (static_cast<D*>(this)->_lostTarget(owner))
89  {
90  HandleMovementFailure(owner);
91  return true;
92  }
93 
94  HandleTargetedMovement(owner, time_diff);
95 
96  if (owner.movespline->Finalized() && !i_targetReached)
97  HandleFinalizedMovement(owner);
98 
99  return true;
100 }

222 void ChaseMovementGenerator::HandleTargetedMovement(Unit& owner, const uint32& time_diff)
223 {
224  bool targetMoved = false;
225  G3D::Vector3 currentTargetPos;
226  this->i_target->GetPosition(currentTargetPos.x, currentTargetPos.y, currentTargetPos.z, owner.GetTransport());
227  this->i_recheckDistance.Update(time_diff);
228  if (m_closenessAndFanningTimer) // here because always need to update timer, cant reuse the timer class because we need a disablable timer
229  {
230  if (m_closenessAndFanningTimer <= time_diff)
231  {
232  m_closenessAndFanningTimer = 0;
233  m_closenessExpired = true;
234  }
235  else m_closenessAndFanningTimer -= time_diff;
236  }
237  if (!this->i_recheckDistance.Passed())
238  return;
239 
240  this->i_recheckDistance.Reset(250);
241  auto vectorDest = owner.movespline->FinalDestination();
242  Position dest(vectorDest.x, vectorDest.y, vectorDest.z, 0.f);
243  if (dest.x == 0 && dest.y == 0 && dest.z == 0)
244  owner.GetPosition(dest.x, dest.y, dest.z, owner.GetTransport());
245  if (m_currentMode != CHASE_MODE_DISTANCING)
246  {
247  targetMoved = this->RequiresNewPosition(owner, dest); // uses transport coordinates
248 
249  if ((this->i_speedChanged && !owner.movespline->Finalized()) || targetMoved)
250  {
251  float x, y, z;
252 
253  // i_path can be nullptr in case this is the first call for this MMGen (via Update)
254  // Can happen for example if no path was created on MMGen-Initialize because of the owner being stunned
255  if (targetMoved || !this->i_path)
256  _getLocation(owner, x, y, z);
257  else
258  {
259  // the destination has not changed, we just need to refresh the path (usually speed change)
260  G3D::Vector3 end = this->i_path->getEndPosition(); // supplies transport coordinates when on a transport
261  x = end.x;
262  y = end.y;
263  z = end.z;
264  if (GenericTransport* transport = owner.GetTransport()) // dispatch spline position needs global coordinates
265  transport->CalculatePassengerPosition(dest.x, dest.y, dest.z);
266  }
267 
268  if (owner.GetDistance(x, y, z, DIST_CALC_NONE) > 0.3f)
269  {
270  if (DispatchSplineToPosition(owner, x, y, z, EnableWalking(), true, true, true))
271  {
272  this->i_targetReached = false;
273  this->i_speedChanged = false;
274  /* m_prevTargetPos is updated on making new spline (normal and distancing) and also on reaching target
275  is used for determining if player moved towards target whilst the spline was going on to stop the spline prematurely
276  and prevent it going behind targets back - it will still occur in rare cases due to PF and lag */
277  this->i_target->GetPosition(this->i_lastTargetPos.x, this->i_lastTargetPos.y, this->i_lastTargetPos.z, owner.GetTransport());
278  m_closenessAndFanningTimer = 0;
279  return;
280  }
281  if (m_reachable == false)
282  return;
283  }
284  if (!IsReachablePositionToTarget(owner, owner.GetPositionX(), owner.GetPositionY(), owner.GetPositionZ(), *this->i_target.getTarget()))
285  m_reachable = false;
286  return;
287  }
288  else if (!targetMoved) // we do not need new position and we are reachable
289  m_reachable = true;
290  }
291  else if (this->i_speedChanged)
292  {
293  if (GenericTransport* transport = owner.GetTransport()) // dispatch spline position needs global coordinates
294  transport->CalculatePassengerPosition(dest.x, dest.y, dest.z);
295  if (DispatchSplineToPosition(owner, dest.x, dest.y, dest.z, false, false, true))
296  {
297  this->i_speedChanged = false;
298  return;
299  }
300  }
301 
302  // while spline is engaged we have two cases: Running to target or distancing when target was standing in model
303  if (!this->i_targetReached)
304  {
305  if (owner.movespline->Finalized())
306  return;
307  else if (m_currentMode == CHASE_MODE_NORMAL)
308  {
309  // TODO: This code would benefit greatly if "Get Target Position in 250 ms" function existed
310  if (currentTargetPos != this->i_lastTargetPos)
311  {
312  Position ownerPos;
313  owner.GetPosition(ownerPos.x, ownerPos.y, ownerPos.z, owner.GetTransport());
314  float distFromDestination = owner.GetDistance(dest.x, dest.y, dest.z, DIST_CALC_NONE, owner.GetTransport());
315  float distOwnerFromTarget = this->i_target->GetDistance(ownerPos.x, ownerPos.y, ownerPos.z, DIST_CALC_NONE, owner.GetTransport());
316  // Explanation of magic: comparing distances between target and destination makes it so we know when mob is between destination and owner
317  // - thats when forcible spline stop is needed
318  float targetDist = this->i_target->GetCombinedCombatReach(&owner, this->i_offset == 0.f ? true : false);
319  if (distFromDestination > distOwnerFromTarget)
320  {
321  if (this->i_target->GetDistance(ownerPos.x, ownerPos.y, ownerPos.z, DIST_CALC_NONE, owner.GetTransport()) < targetDist * targetDist)
322  {
323  if (owner.IsClientControlled())
324  owner.StopMoving(true);
325  else
326  owner.InterruptMoving();
327  }
328  }
329  }
330  }
331  }
332  else
333  {
334  // When creatures use backpedaling, they are caught in an endless cycle of it, its not critical, since they arrive at each other with precision anyhow
335  if (this->i_target->GetTypeId() != TYPEID_PLAYER)
336  return;
337 
338  if (m_closenessExpired)
339  Backpedal(owner);
340  }
341 }

391 void ChaseMovementGenerator::Backpedal(Unit& owner)
392 {
393  // do not allow backpedalling during scripts
394  if (!owner.AI() || owner.AI()->GetCombatScriptStatus())
395  return;
396 
397  m_closenessExpired = false;
398  m_closenessAndFanningTimer = CHASE_CLOSENESS_TIMER; // Just in case path doesnt generate
399  float targetDist = std::min(this->i_target->GetCombinedCombatReach(&owner, false), 3.75f);
400  float x, y, z;
401  owner.GetPosition(x, y, z);
402  // TODO: add calculation of circle on ground to core and use that
403  if (i_target->GetDistance(x, y, z, DIST_CALC_NONE) < (targetDist * targetDist) * 0.33f) // is too close
404  {
405  float ori = MapManager::NormalizeOrientation(owner.GetOrientation() + M_PI_F);
406  i_target->GetNearPoint(&owner, x, y, z, owner.GetObjectBoundingRadius(), targetDist * 0.75f, ori);
407  }
408  else
409  {
410  FanOut(owner); // if already on the radius - check fanning
411  return;
412  }
413 
414  if (DispatchSplineToPosition(owner, x, y, z, true, false))
415  {
416  this->i_targetReached = false;
417  m_currentMode = CHASE_MODE_BACKPEDAL;
418  this->i_speedChanged = false;
419  m_closenessAndFanningTimer = 0; // restart on reaching target
420  }

427 void ChaseMovementGenerator::FanOut(Unit& owner)
428 {
429  if (!m_fanningEnabled)
430  return;
431 
432  Unit* collider = nullptr;
433  MaNGOS::AnyUnitFulfillingConditionInRangeCheck collisionCheck(&owner, [&](Unit* unit)->bool
434  {
435  return &owner != unit && unit->GetVictim() && unit->GetVictim() == this->i_target.getTarget() && !unit->IsMoving() && !unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE) && !unit->IsBoarded();
436  }, fanningRadius * fanningRadius, DIST_CALC_NONE);
437  MaNGOS::UnitSearcher<MaNGOS::AnyUnitFulfillingConditionInRangeCheck> checker(collider, collisionCheck);
438  Cell::VisitAllObjects(&owner, checker, fanningRadius);
439 
440  if (collider) // position collision found - need to find a new spot
441  {
442  int32 direction = irand(0, 1); // blizzlike behaviour
443  if (direction == 0) direction = -1;
444  float ori = MapManager::NormalizeOrientation(owner.GetOrientation() + M_PI_F + frand(fanAngleMin, fanAngleMax) * direction);
445  float x, y, z;
446  float targetDist = this->i_target->GetCombinedCombatReach(&owner, false);
447  i_target->GetNearPoint(&owner, x, y, z, owner.GetObjectBoundingRadius(), targetDist, ori);
448  if (DispatchSplineToPosition(owner, x, y, z, true, false))
449  {
450  this->i_targetReached = false;
451  m_currentMode = CHASE_MODE_FANNING;
452  this->i_speedChanged = false;
453  m_closenessAndFanningTimer = 0; // restart on reaching target
454  return;
455  }
456  // if first direction failed try second
457  ori = MapManager::NormalizeOrientation(owner.GetOrientation() + M_PI_F + frand(fanAngleMin, fanAngleMax) * -direction);
458  targetDist = this->i_target->GetCombinedCombatReach(&owner, false);
459  i_target->GetNearPoint(&owner, x, y, z, owner.GetObjectBoundingRadius(), targetDist, ori);
460  if (DispatchSplineToPosition(owner, x, y, z, true, false))
461  {
462  this->i_targetReached = false;
463  m_currentMode = CHASE_MODE_FANNING;
464  this->i_speedChanged = false;
465  m_closenessAndFanningTimer = 0; // restart on reaching target
466  return;
467  }
468  // both failed - do nothing
469  }

473 {

476  for (ObjectGuid guid : m_spawns)

478  whisp->ForcedDespawn();

498  {

505  {

507  {
508  gen = false;
509  break;
510  }
511  }
512  }
513  }
514 
515  if (owner.IsDebuggingMovement())
516  {

522  message += (owner.IsWithinDist3d(x, y, z, 200.f) ? "Within 200f " : "") + std::string(owner.IsWithinLOS(x, y, z + i_target->GetCollisionHeight()) ? "Within LOS " : "") +
523  ((this->i_path->getPathType() & PATHFIND_NOPATH) ? " No straight path" : "") + "\n";
524  static_cast<Player*>(i_target.getTarget())->SendMessageToPlayer(message);
525  std::ostringstream out;
526  out.precision(10);
527  out << ".go xyz " << std::fixed << pos.x << " " << pos.y << " " << pos.z << std::endl;
528  out << ".go xyz " << std::fixed << x << " " << y << " " << z << std::endl;
529  sLog.outCustomLog("%s", out.str().data());
530  }
531  }
532 
533  if (!gen || (this->i_path->getPathType() & (PATHFIND_NOPATH | PATHFIND_INCOMPLETE)))
534  {
535  this->i_path->calculate(x, y, z);
536  if (this->i_path->getPathType() & PATHFIND_NOPATH)
537  return false;
538  }
539

541

544

845 bool FollowMovementGenerator::Move(Unit& owner, float x, float y, float z)
846 {
847  if (!owner.movespline->Finalized())
848  owner.UpdateSplinePosition(true);
849 
850  if (!i_path)
851  i_path = new PathFinder(&owner);
852 
853  bool unstuck = false;
854 
855  i_path->calculate(x, y, z);
856 
857  auto& path = i_path->getPath();
858 
859  if (i_path->getPathType() & (PATHFIND_NOPATH | PATHFIND_SHORTCUT))
860  {
861  if (!IsUnstuckAllowed(owner))
862  return false;
863 
864  unstuck = true;
865  }
866  else if (i_target->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
867  {
868  // Trim path according to LoS when target is a PC unit
869  // Follower needs to be able to see predicted location to prevent issues and exploits
870  const Map* map = owner.GetMap();
871  const float height = owner.GetCollisionHeight();
872  const GenericTransport* transport = owner.GetTransport();
873 
874  for (size_t i = 1; i < path.size(); ++i)
875  {
876  float lx = path[i - 1].x, ly = path[i - 1].y, lz = path[i - 1].z;
877  float nx = path[i].x, ny = path[i].y, nz = path[i].z;
878 
879  // Convert local transport spline to global spline for visibility check
880  if (transport)
881  {
882  transport->CalculatePassengerPosition(lx, ly, lz);
883  transport->CalculatePassengerPosition(nx, ny, nz);
884  }
885 
886  if (map->IsInLineOfSight(lx, ly, (lz + height), nx, ny, (nz + height), owner.GetPhaseMask(), true))
887  continue;
888 
889  if (i != 1)
890  path.resize(i + 1);
891  else if (!IsUnstuckAllowed(owner))
892  return false;
893  else
894  unstuck = true;
895  break;
896  }
897  }
898 
899  if (unstuck)
900  {
901  float o;
902  _getOrientation(owner, o);
903  _getLocation(owner, x, y, z, false);
904 
905  // Do a final sanity LoS check when unstucking self to prevent landing on a wrong walkable surface
906  // Target needs to be able to see this new location to prevent issues and exploits
907  if (i_target->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
908  if (!i_target->IsWithinLOS(x, y, (z + owner.GetCollisionHeight()), true))
909  i_target->GetPosition(x, y, z);
910 
911  if (owner.GetTypeId() == TYPEID_PLAYER)
912  owner.NearTeleportTo(x, y, z, o);
913  else
914  {
915  owner.GetMap()->CreatureRelocation(static_cast<Creature*>(&owner), x, y, z, o);
916  owner.SendHeartBeat();
917  }
918 
919  return false;
920  }
921 
922  _addUnitStateMove(owner);
923 
924  Movement::MoveSplineInit init(owner);
925  init.MovebyPath(path);
926  init.SetWalk(EnableWalking());
927  init.SetVelocity(GetSpeed(owner));
928  init.Launch();
929 
930  return true;
931 }

1067 void FollowMovementGenerator::HandleTargetedMovement(Unit& owner, const uint32& time_diff)
1068 {
1069  static const MovementFlags detected = MovementFlags(MOVEFLAG_MASK_MOVING_FORWARD | MOVEFLAG_BACKWARD | MOVEFLAG_PITCH_UP | MOVEFLAG_PITCH_DOWN);
1070  static const MovementFlags ignored = MovementFlags(MOVEFLAG_FALLING | MOVEFLAG_FALLINGFAR);
1071 
1072  const bool followerMoving = owner.m_movementInfo.HasMovementFlag(detected);
1073 
1074  if (owner.IsNoMountedFollow() && i_target->IsMounted())
1075  return;
1076 
1077  // Detect target movement and relocation (ignore jumping in place and long falls)
1078  const bool targetMovementIgnored = i_target->m_movementInfo.HasMovementFlag(ignored);
1079  const bool targetMovingLast = m_targetMoving;
1080  m_targetMoving = (!targetMovementIgnored && i_target->m_movementInfo.HasMovementFlag(detected));
1081 
1082  bool targetRelocation = false;
1083  bool targetOrientation = false;
1084  bool targetSpeedChanged = (i_speedChanged && m_targetMoving && targetMovingLast && followerMoving);
1085  i_speedChanged = false;
1086 
1087  if (m_targetMoving && !targetMovingLast) // Movement just started: force update
1088  targetRelocation = !followerMoving;
1089  else if (!m_targetMoving && targetMovingLast && followerMoving) // Movement just ended: delay update further
1090  i_recheckDistance.Reset(_getPollRate(owner, m_targetMoving, targetMovingLast));
1091  else // Periodic dist poll: fast when moving, slow when stationary
1092  {
1093  i_recheckDistance.Update(time_diff);
1094 
1095  if (!targetMovementIgnored && (i_recheckDistance.Passed() || targetSpeedChanged))
1096  {
1097  i_recheckDistance.Reset(_getPollRate(owner, m_targetMoving, targetMovingLast));
1098 
1099  G3D::Vector3 currentTargetPos;
1100  i_target->GetPosition(currentTargetPos.x, currentTargetPos.y, currentTargetPos.z, owner.GetTransport());
1101  Position actualPos = owner.GetPosition(owner.GetTransport());
1102 
1103  // Tracking PC units requires additional orientation tick
1104  if (i_target->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
1105  {
1106  targetRelocation = (currentTargetPos != i_lastTargetPos || RequiresNewPosition(owner, actualPos));
1107  targetOrientation = (!targetRelocation && !m_targetMoving && !m_targetFaced);
1108  }
1109  // Tracking NPC units has no orientation tick and relaxed tracking for wandering targets
1110  else
1111  {
1112  if (i_target->GetMotionMaster()->GetCurrentMovementGeneratorType() == RANDOM_MOTION_TYPE)
1113  targetRelocation = RequiresNewPosition(owner, actualPos);
1114  else
1115  targetRelocation = (currentTargetPos != i_lastTargetPos || RequiresNewPosition(owner, actualPos));
1116  }
1117 
1118  targetSpeedChanged = (targetSpeedChanged && targetRelocation);
1119  i_lastTargetPos = currentTargetPos;
1120  }
1121  }
1122 
1123  // Decide whether it's suitable time to update position or orientation
1124  if (targetRelocation || targetSpeedChanged)
1125  {
1126  i_recheckDistance.Reset(_getPollRate(owner, m_targetMoving, targetMovingLast));
1127  _setLocation(owner, m_targetMoving);
1128  }
1129  else if (targetOrientation && !i_faceTarget && i_targetReached)
1130  _setOrientation(owner);
1131 }
