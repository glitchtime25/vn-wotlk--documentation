49 void LoadSkillDiscoveryTable()
50 {
51  SkillDiscoveryStore.clear(); // need for reload
52 
53  uint32 count = 0;
54 
55  // 0 1 2 3
56  QueryResult* result = WorldDatabase.Query("SELECT spellId, reqSpell, reqSkillValue, chance FROM skill_discovery_template");
57 
58  if (!result)
59  {
60  sLog.outString(">> Loaded 0 skill discovery definitions. DB table `skill_discovery_template` is empty.");
61  sLog.outString();
62  return;
63  }
64 
65  BarGoLink bar(result->GetRowCount());
66 
67  std::ostringstream ssNonDiscoverableEntries;
68  std::set<uint32> reportedReqSpells;
69 
70  do
71  {
72  Field* fields = result->Fetch();
73  bar.step();
74 
75  uint32 spellId = fields[0].GetUInt32();
76  int32 reqSkillOrSpell = fields[1].GetInt32();
77  uint32 reqSkillValue = fields[2].GetInt32();
78  float chance = fields[3].GetFloat();
79 
80  if (chance <= 0) // chance
81  {
82  ssNonDiscoverableEntries << "spellId = " << spellId << " reqSkillOrSpell = " << reqSkillOrSpell
83  << " reqSkillValue = " << reqSkillValue << " chance = " << chance << "(chance problem)\n";
84  continue;
85  }
86 
87  if (reqSkillOrSpell > 0) // spell case
88  {
89  SpellEntry const* reqSpellEntry = sSpellTemplate.LookupEntry<SpellEntry>(reqSkillOrSpell);
90  if (!reqSpellEntry)
91  {
92  if (reportedReqSpells.find(reqSkillOrSpell) == reportedReqSpells.end())
93  {
94  sLog.outErrorDb("Spell (ID: %u) have nonexistent spell (ID: %i) in `reqSpell` field in `skill_discovery_template` table", spellId, reqSkillOrSpell);
95  reportedReqSpells.insert(reqSkillOrSpell);
96  }
97  continue;
98  }
99 
100  // mechanic discovery
101  if (reqSpellEntry->Mechanic != MECHANIC_DISCOVERY &&
102  // explicit discovery ability
103  !IsExplicitDiscoverySpell(reqSpellEntry))
104  {
105  if (reportedReqSpells.find(reqSkillOrSpell) == reportedReqSpells.end())
106  {
107  sLog.outErrorDb("Spell (ID: %u) not have MECHANIC_DISCOVERY (28) value in Mechanic field in spell.dbc"
108  " and not 100%% chance random discovery ability but listed for spellId %u (and maybe more) in `skill_discovery_template` table",
109  reqSkillOrSpell, spellId);
110  reportedReqSpells.insert(reqSkillOrSpell);
111  }
112  continue;
113  }
114 
115  SkillDiscoveryStore[reqSkillOrSpell].push_back(SkillDiscoveryEntry(spellId, reqSkillValue, chance));
116  }
117  else if (reqSkillOrSpell == 0) // skill case
118  {
119  SkillLineAbilityMapBounds bounds = sSpellMgr.GetSkillLineAbilityMapBoundsBySpellId(spellId);
120 
121  if (bounds.first == bounds.second)
122  {
123  sLog.outErrorDb("Spell (ID: %u) not listed in `SkillLineAbility.dbc` but listed with `reqSpell`=0 in `skill_discovery_template` table", spellId);
124  continue;
125  }
126 
127  for (SkillLineAbilityMap::const_iterator _spell_idx = bounds.first; _spell_idx != bounds.second; ++_spell_idx)
128  SkillDiscoveryStore[-int32(_spell_idx->second->skillId)].push_back(SkillDiscoveryEntry(spellId, reqSkillValue, chance));
129  }
130  else
131  {
132  sLog.outErrorDb("Spell (ID: %u) have negative value in `reqSpell` field in `skill_discovery_template` table", spellId);
133  continue;
134  }
135 
136  ++count;
137  }
138  while (result->NextRow());
139 
140  delete result;
141 
142  if (!ssNonDiscoverableEntries.str().empty())
143  sLog.outErrorDb("Some items can't be successfully discovered: have in chance field value < 0.000001 in `skill_discovery_template` DB table . List:\n%s", ssNonDiscoverableEntries.str().c_str());
144 
145  // report about empty data for explicit discovery spells
146  for (uint32 spell_id = 1; spell_id < sSpellTemplate.GetMaxEntry(); ++spell_id)
147  {
148  SpellEntry const* spellEntry = sSpellTemplate.LookupEntry<SpellEntry>(spell_id);
149  if (!spellEntry)
150  continue;
151 
152  // skip not explicit discovery spells
153  if (!IsExplicitDiscoverySpell(spellEntry))
154  continue;
155 
156  if (SkillDiscoveryStore.find(spell_id) == SkillDiscoveryStore.end())
157  sLog.outErrorDb("Spell (ID: %u) is 100%% chance random discovery ability but not have data in `skill_discovery_template` table", spell_id);
158  }
159 
160  sLog.outString(">> Loaded %u skill discovery definitions", count);
161  sLog.outString();
162 }

164 uint32 GetExplicitDiscoverySpell(uint32 spellId, Player* player)
165 {
166  // explicit discovery spell chances (always success if case exist)
167  // in this case we have both skill and spell
168  SkillDiscoveryMap::const_iterator tab = SkillDiscoveryStore.find(spellId);
169  if (tab == SkillDiscoveryStore.end())
170  return 0;
171 
172  SkillLineAbilityMapBounds bounds = sSpellMgr.GetSkillLineAbilityMapBoundsBySpellId(spellId);
173  uint32 skillvalue = bounds.first != bounds.second ? player->GetSkillValue(bounds.first->second->skillId) : 0;
174 
175  float full_chance = 0;
176  for (auto item_iter : tab->second)
177  if (item_iter.reqSkillValue <= skillvalue)
178  if (!player->HasSpell(item_iter.spellId))
179  full_chance += item_iter.chance;
180 
181  float rate = full_chance / 100.0f;
182  float roll = rand_chance_f() * rate; // roll now in range 0..full_chance
183 
184  for (auto item_iter : tab->second)
185  {
186  if (item_iter.reqSkillValue > skillvalue)
187  continue;
188 
189  if (player->HasSpell(item_iter.spellId))
190  continue;
191 
192  if (item_iter.chance > roll)
193  return item_iter.spellId;
194 
195  roll -= item_iter.chance;
196  }
197 
198  return 0;
199 }

201 uint32 GetSkillDiscoverySpell(uint32 skillId, uint32 spellId, Player* player)
202 {
203  uint32 skillvalue = skillId ? player->GetSkillValue(skillId) : 0;
204 
205  // check spell case
206  SkillDiscoveryMap::const_iterator tab = SkillDiscoveryStore.find(spellId);
207 
208  if (tab != SkillDiscoveryStore.end())
209  {
210  for (auto item_iter : tab->second)
211  {
212  if (roll_chance_f(item_iter.chance * sWorld.getConfig(CONFIG_FLOAT_RATE_SKILL_DISCOVERY)) &&
213  item_iter.reqSkillValue <= skillvalue &&
214  !player->HasSpell(item_iter.spellId))
215  return item_iter.spellId;
216  }
217 
218  return 0;
219  }
220 
221  if (!skillId)
222  return 0;
223 
224  // check skill line case
225  tab = SkillDiscoveryStore.find(-(int32)skillId);
226  if (tab != SkillDiscoveryStore.end())
227  {
228  for (auto item_iter : tab->second)
229  {
230  if (roll_chance_f(item_iter.chance * sWorld.getConfig(CONFIG_FLOAT_RATE_SKILL_DISCOVERY)) &&
