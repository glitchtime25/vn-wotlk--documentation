120  void StartEruptions()
121  {
122  uint32 spellId = m_phase == PHASE_GROUND ? SPELL_PLAGUE_WAVE_SLOW : SPELL_PLAGUE_WAVE_FAST;
123  if (Creature* trigger = GetClosestCreatureWithEntry(m_creature, NPC_PLAGUE_WAVE, 100.0f))
124  {
125  trigger->RemoveAllAuras();

127  }
128  }

130  void StopEruptions()

133  trigger->RemoveAllAuras();
134  }

136  void HandleGroundPhase()
137  {
138  m_phase = PHASE_GROUND;
139  StopEruptions();
140  m_creature->InterruptNonMeleeSpells(true);
141  SetCombatScriptStatus(false);
142  SetMeleeEnabled(true);
143  SetCombatMovement(true, true);
144  DoBroadcastText(EMOTE_RETURN, m_creature);
145  ResetCombatAction(HEIGAN_PHASE_PLATFORM, GetSubsequentActionTimer(HEIGAN_PHASE_PLATFORM));
146  ResetTimer(HEIGAN_ERUPTIONS, 5s);
147  AddInitialCooldowns();
148  }

150  void HandleTaunt()
151  {
152  if (!m_creature->IsAlive() || !m_creature->IsInCombat())
153  return;
154  switch (urand(0, 3))
155  {
156  case 0: DoBroadcastText(SAY_TAUNT1, m_creature); break;
157  case 1: DoBroadcastText(SAY_TAUNT2, m_creature); break;
158  case 2: DoBroadcastText(SAY_TAUNT3, m_creature); break;
159  case 3: DoBroadcastText(SAY_TAUNT4, m_creature); break;
160  }
161  ResetTimer(HEIGAN_TAUNT, GetSubsequentActionTimer(HEIGAN_TAUNT));
162  }
163 
164  void HandleStartChanneling()
165  {

170  std::chrono::milliseconds GetSubsequentActionTimer(uint32 action)

174  case HEIGAN_TAUNT: return RandomTimer(20s, 70s);

178  }
179  }

181  void ExecuteAction(uint32 action) override
182  {
183  switch (action)
184  {
185  case HEIGAN_PHASE_PLATFORM:
186  {
187  if (DoCastSpellIfCan(nullptr, SPELL_TELEPORT) == CAST_OK)
188  {
189  StopEruptions();
190  SetAIImmobilizedState(true);
191  DoBroadcastText(EMOTE_TELEPORT, m_creature);
192  SetReactState(REACT_PASSIVE);
193  SetCombatScriptStatus(true);
194  m_creature->SetTarget(nullptr);
195  SetMeleeEnabled(false);
196  SetCombatMovement(false);
197  m_phase = PHASE_PLATFORM;
198  DisableCombatAction(action);
199 
200  ResetCombatAction(HEIGAN_START_CHANNELING, 1s);
201  ResetTimer(HEIGAN_PHASE_GROUND, GetSubsequentActionTimer(HEIGAN_PHASE_GROUND));
202  ResetTimer(HEIGAN_ERUPTIONS, 5s);
203  }
204  return;
205  }
206  }
207  ResetCombatAction(action, GetSubsequentActionTimer(action));
208  }
