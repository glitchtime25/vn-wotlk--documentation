195 void metric::metric::send()
196 {
197  std::vector<std::unique_ptr<Measurement>> measurements;
198 
199  // Scope swapping
200  {
201  std::lock_guard<std::mutex> guard(m_queueWriteLock);
202  std::swap(measurements, m_measurementQueue);
203  }
204 
205  sLog.outDetail("Sending %zu measurements!", measurements.size());
206 
207  using boost::asio::ip::tcp;
208 
209  boost::system::error_code error;
210 
211  // Hostname resolution
212  tcp::resolver resolver(m_writeService);
213  tcp::resolver::query query(m_connectionInfo.hostname, std::to_string(m_connectionInfo.port));
214  tcp::resolver::iterator endpoint_iterator = resolver.resolve(query, error);
215 
216  if (error)
217  {
218  sLog.outError("metric::metric::send resolve aborted, %s", error.message().c_str());
219  return;
220  }
221 
222  error = boost::asio::error::host_not_found;
223 
224  tcp::resolver::iterator end;
225 
226  tcp::socket socket(m_writeService);
227  while (error && endpoint_iterator != end)
228  {
229  socket.close();
230  socket.connect(*endpoint_iterator++, error);
231  }
232 
233  if (error)
234  {
235  sLog.outError("metric::metric::send connect aborted, %s", error.message().c_str());
236  return;
237  }
238 
239  std::stringstream payload;
240  for (auto const& measurement : measurements)
241  {
242  if (&measurement != &measurements.front())
243  payload << "\n";
244 
245  payload << *measurement;
246  }
247 
248  boost::asio::streambuf request;
249  std::ostream request_stream(&request);
250 
251  // Write request
252  request_stream << "POST " << "/write?db=" << m_connectionInfo.database << "&u=" << m_connectionInfo.username << "&p=" << m_connectionInfo.password << " HTTP/1.1\r\n";
253  request_stream << "Host: " << m_connectionInfo.hostname << "\r\n";
254  request_stream << "Content-Length:" << std::to_string(payload.tellp()) << "\r\n";
255  request_stream << "Connection: close\r\n\r\n";
256  request_stream << payload.rdbuf();
257 
258  // Send the request.
259  boost::asio::write(socket, request, error);
260  if (error)
261  {
262  sLog.outError("metric::metric::send write aborted, %s", error.message().c_str());
263  return;
264  }
265 
266  // Read the response status line. The response streambuf will automatically
267  // grow to accommodate the entire line. The growth may be limited by passing
268  // a maximum size to the streambuf constructor.
269  boost::asio::streambuf response;
270  boost::asio::read_until(socket, response, "\r\n", error);
271  if (error)
272  {
273  sLog.outError("metric::metric::send read_until aborted, %s", error.message().c_str());
274  return;
275  }
276 
277  // Check that response is OK.
278  std::string http_version;
279  unsigned int status_code;
280  std::string status_message;
281 
282  std::istream response_stream(&response);
283  response_stream >> http_version;
284  response_stream >> status_code;
285  std::getline(response_stream, status_message);
286 
287  if (!response_stream || http_version.substr(0, 5) != "HTTP/")
288  {
289  sLog.outError("metric::metric::send received invalid response");
290  return;
291  }
292 
293  if (status_code < 200 || status_code >= 300)
294  {
295  // Should restore measurements back into queue
296 
297  sLog.outError("metric::metric::send response returned with status code %u", status_code);
298 
299  // Read the response headers, which are terminated by a blank line.
300  // if this errors out its ok
301  boost::asio::read_until(socket, response, "\r\n\r\n", error);
302 
303  // Process the response headers.
304  std::string header;
305  while (std::getline(response_stream, header) && header != "\r")
306  std::cout << header << "\n";
307 
308  std::cout << "\n";
309 
310  // Write whatever content we already have to output.
311  if (response.size() > 0)
312  std::cout << &response;
313 
314  // Read until EOF, writing data to output as we go.
315  while (boost::asio::read(socket, response, boost::asio::transfer_at_least(1), error))
316  std::cout << &response;
317 
318  if (error != boost::asio::error::eof)
319  throw boost::system::system_error(error);
320 
321  return;
322  }
323 }
