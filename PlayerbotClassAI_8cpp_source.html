133 CombatManeuverReturns PlayerbotClassAI::Buff(bool (*BuffHelper)(PlayerbotAI*, uint32, Unit*), uint32 spellId, uint32 type, bool mustBeOOC)
134 {
135  if (!m_bot.IsAlive() || m_bot.IsInDuel()) return RETURN_NO_ACTION_ERROR;
136  if (mustBeOOC && m_bot.IsInCombat()) return RETURN_NO_ACTION_ERROR;
137 
138  if (spellId == 0) return RETURN_NO_ACTION_OK;
139 
140  // First, fill the list of targets
141  if (m_bot.GetGroup())
142  {
143  Group::MemberSlotList const& groupSlot = m_bot.GetGroup()->GetMemberSlots();
144  for (const auto & memberItr : groupSlot)
145  {
146  Player* member = sObjectMgr.GetPlayer(memberItr.guid);
147  if (!member || !member->IsAlive() || member->IsInDuel())
148  continue;
149  // Guess the job of current member
150  JOB_TYPE job = GetTargetJob(member);
151  // If job matches requested job type or (mana target is requested and member is mana user (or shapeshift druid)): buff it
152  if (job & type ||
153  (type & JOB_MANAONLY && (member->getClass() == CLASS_DRUID || member->GetPowerType() == POWER_MANA)))
154  {
155  if (BuffHelper(&m_ai, spellId, member))
156  return RETURN_CONTINUE;
157  }
158  }
159  }
160  else
161  {
162  // Buff master if he/she is eligible
163  if (!m_master.IsInDuel()
164  && (GetTargetJob(&m_master) & type
165  || (type & JOB_MANAONLY && (m_master.getClass() == CLASS_DRUID || m_master.GetPowerType() == POWER_MANA))))
166  if (BuffHelper(&m_ai, spellId, &m_master))
167  return RETURN_CONTINUE;
168  // Do not check job or power type - any buff you have is always useful to self
169  if (BuffHelper(&m_ai, spellId, &m_bot))
170  return RETURN_CONTINUE;
171  }
172 
173  return RETURN_NO_ACTION_OK;
174 }

185 bool PlayerbotClassAI::NeedGroupBuff(uint32 groupBuffSpellId, uint32 singleBuffSpellId)
186 {
187  uint8 unbuffedTargets = 0;
188  // Check group players to avoid using reagent and mana with an expensive group buff
189  // when only two players or less need it
190  if (m_bot.GetGroup())
191  {
192  Group::MemberSlotList const& groupSlot = m_bot.GetGroup()->GetMemberSlots();
193  for (const auto& memberItr : groupSlot)
194  {
195  Player* member = sObjectMgr.GetPlayer(memberItr.guid);
196  if (!member || !member->IsAlive())
197  continue;
198  // Check if group member needs buff
199  if (!member->HasAura(groupBuffSpellId, EFFECT_INDEX_0) && !member->HasAura(singleBuffSpellId, EFFECT_INDEX_0))
200  unbuffedTargets++;
201  // Don't forget about pet
202  Pet* pet = member->GetPet();
203  if (pet && !pet->HasAuraType(SPELL_AURA_MOD_UNATTACKABLE) && !(pet->HasAura(groupBuffSpellId, EFFECT_INDEX_0) || pet->HasAura(singleBuffSpellId, EFFECT_INDEX_0)))
204  unbuffedTargets++;
205  }
206  // threshold is set to 2 targets because beyond that value, the group buff cost is cheaper in mana
207  return unbuffedTargets >= 3;
208  }
209  return false; // no group, no group buff
210 }

221

254 Player* PlayerbotClassAI::GetHealTarget(JOB_TYPE type, bool onlyPickFromSameGroup)
255 {
256  if (!m_bot.IsAlive() || m_bot.IsInDuel()) return nullptr;
257 
258  // define seperately for sorting purposes - DO NOT CHANGE ORDER!
259  std::vector<heal_priority> targets;
260  uint8 uiHealthPercentage;
261 
262  // First, fill the list of targets
263  if (m_bot.GetGroup())
264  {
265  Group::MemberSlotList const& groupSlot = m_bot.GetGroup()->GetMemberSlots();
266  for (Group::member_citerator itr = groupSlot.begin(); itr != groupSlot.end(); itr++)
267  {
268  Player* groupMember = sObjectMgr.GetPlayer(itr->guid);
269  if (!groupMember || !groupMember->IsAlive() || groupMember->IsInDuel()
270  || (!m_bot.GetGroup()->SameSubGroup(&m_bot, groupMember) && onlyPickFromSameGroup))
271  continue;
272  JOB_TYPE job = GetTargetJob(groupMember);
273  if (job & type)
274  {
275  uiHealthPercentage = int(groupMember->GetMaxHealth() != 0 ? groupMember->GetHealth() * 100 / groupMember->GetMaxHealth() : 0);
276  targets.push_back(heal_priority(groupMember, uiHealthPercentage, job));
277  }
278  }
279  }
280  else
281  {
282  targets.push_back(heal_priority(&m_bot, m_bot.GetHealthPercent(), GetTargetJob(&m_bot)));
283  if (!m_master.IsInDuel())
284  {
285  uiHealthPercentage = int(m_master.GetMaxHealth() != 0 ? m_master.GetHealth() * 100 / m_master.GetMaxHealth() : 0);
286  targets.push_back(heal_priority(&m_master, uiHealthPercentage, GetTargetJob(&m_master)));
287  }
288  }
289 
290  // Sorts according to type: Main tank first, healers second, then regular tanks, then master followed by DPS, thanks to the order of the TYPE enum
291  std::sort(targets.begin(), targets.end());
292 
293  uint8 uCount = 0, i = 0;
294  // x is used as 'target found' variable; i is used as the targets iterator throughout all 6 types.
295  int16 x = -1;
296 
297  // Try to find a main tank in need of healing (if multiple, the lowest health one)
298  while (true)
299  {
300  // This works because we sorted it above
301  if (uint32(uCount + i) >= uint32(targets.size()) || !(targets.at(uCount).type & JOB_MAIN_TANK)) break;
302  uCount++;
303  }
304 
305  // We have uCount main tanks in the targets, check if any qualify for priority healing
306  for (; uCount > 0; uCount--, i++)
307  {
308  if (targets.at(i).hp <= m_MinHealthPercentTank)
309  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
310  x = i;
311  }
312  if (x > -1) return targets.at(x).p;
313 
314  // Try to find a healer in need of healing (if multiple, the lowest health one)
315  while (true)
316  {
317  if (uint32(uCount + i) >= uint32(targets.size()) || !(targets.at(uCount).type & (JOB_HEAL | JOB_MAIN_HEAL))) break;
318  uCount++;
319  }
320 
321  for (; uCount > 0; uCount--, i++)
322  {
323  if (targets.at(i).hp <= m_MinHealthPercentHealer)
324  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
325  x = i;
326  }
327  if (x > -1) return targets.at(x).p;
328 
329  // Try to find a tank in need of healing (if multiple, the lowest health one)
330  while (true)
331  {
332  if (uint32(uCount + i) >= uint32(targets.size()) || !(targets.at(uCount).type & JOB_TANK)) break;
333  uCount++;
334  }
335 
336  for (; uCount > 0; uCount--, i++)
337  {
338  if (targets.at(i).hp <= m_MinHealthPercentTank)
339  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
340  x = i;
341  }
342  if (x > -1) return targets.at(x).p;
343 
344  // Try to find master in need of healing (lowest health one first)
345  if (m_MinHealthPercentMaster != m_MinHealthPercentDPS)
346  {
347  while (true)
348  {
349  if (uint32(uCount + i) >= uint32(targets.size()) || !(targets.at(uCount).type & JOB_MASTER)) break;
350  uCount++;
351  }
352 
353  for (; uCount > 0; uCount--, i++)
354  {
355  if (targets.at(i).hp <= m_MinHealthPercentMaster)
356  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
357  x = i;
358  }
359  if (x > -1) return targets.at(x).p;
360  }
361 
362  // Try to find anyone else in need of healing (lowest health one first)
363  while (true)
364  {
365  if (uint32(uCount + i) >= uint32(targets.size())) break;
366  uCount++;
367  }
368 
369  for (; uCount > 0; uCount--, i++)
370  {
371  if (targets.at(i).hp <= m_MinHealthPercentDPS)
372  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
373  x = i;
374  }
375  if (x > -1) return targets.at(x).p;
376 
377  // Nobody is critical, find anyone hurt at all, return lowest (let the healer sort out if it's worth healing or not)
378  for (i = 0, uCount = targets.size(); uCount > 0; uCount--, i++)
379  {
380  if (targets.at(i).hp < 100)
381  if (x == -1 || targets.at(x).hp > targets.at(i).hp)
382  x = i;
383  }
384  if (x > -1) return targets.at(x).p;
385 
386  return nullptr;
387 }

499 bool PlayerbotClassAI::FleeFromPointIfCan(uint32 radius, Unit* pTarget, float x0, float y0, float z0, float forcedAngle /* = 0.0f */)
500 {
501  // Get relative position to current target
502  // the bot will try to move on a tangential axis from it
503  float dist_from_target, angle_to_target;
504  if (pTarget)
505  {
506  dist_from_target = pTarget->GetDistance(&m_bot);
507  if (dist_from_target > 0.2f)
508  angle_to_target = pTarget->GetAngle(&m_bot);
509  else
510  angle_to_target = frand(0, 2 * M_PI_F);
511  }
512  else
513  {
514  dist_from_target = 0.0f;
515  angle_to_target = frand(0, 2 * M_PI_F);
516  }
517 
518  // Find coords to move to
519  // The bot will move for a distance equal to the spell radius + 1 yard for more safety
520  float dist = radius + 1.0f;
521 
522  float moveAngles[3] = {- M_PI_F / 2, M_PI_F / 2, 0.0f};
523  float angle, x, y, z;
524  bool foundCoords;
525  for (uint8 i = 0; i < 3; i++)
526  {
527  // define an angle tangential to target's direction
528  angle = angle_to_target + moveAngles[i];
529  // if an angle was provided, use it instead but only for the first iteration in case this does not lead to a valid point
530  if (forcedAngle != 0.0f)
531  {
532  angle = forcedAngle;
533  forcedAngle = 0.0f;
534  }
535  foundCoords = true;
536 
537  x = x0 + dist * cos(angle);
538  y = y0 + dist * sin(angle);
539  z = z0 + 0.5f;
540 
541  // try to fix z
542  if (!m_bot.GetMap()->GetHeightInRange(m_bot.GetPhaseMask(), x, y, z))
543  foundCoords = false;
544 
545  // check any collision
546  float testZ = z + 0.5f; // needed to avoid some false positive hit detection of terrain or passable little object
547  if (m_bot.GetMap()->GetHitPosition(x0, y0, z0 + 0.5f, x, y, testZ, m_bot.GetPhaseMask(), -0.1f))
548  {
549  z = testZ;
550  if (!m_bot.GetMap()->GetHeightInRange(m_bot.GetPhaseMask(), x, y, z))
551  foundCoords = false;
552  }
553 
554  if (foundCoords)
555  {
556  m_ai.InterruptCurrentCastingSpell();
557  m_bot.GetMotionMaster()->MovePoint(0, x, y, z);
558  m_ai.SetIgnoreUpdateTime(2);
559  return true;
560  }
561  }
562 
563  return false;
564 }
