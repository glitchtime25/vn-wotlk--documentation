176  void HandleSubmergeActions()
177  {
178  uint32 timer = 0;
179  switch (m_submergePhases)
180  {
181  case 0:
182  {
183  m_creature->RemoveAurasDueToSpell(SPELL_SUBMERGE);
184  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
185  timer = 2000;
186  break;
187  }
188  case 1:
189  {
190  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
191  timer = 1000;
192  break;
193  }
194  case 2:
195  {
196  ResetCombatAction(LURKER_WHIRL, 0);
197  ResetCombatAction(LURKER_SUBMERGE_PHASE, 2 * MINUTE * IN_MILLISECONDS);
198  ResetCombatAction(LURKER_SPOUT, 2000);
199  ResetCombatAction(LURKER_GEYSER, 50000);
200  ResetCombatAction(LURKER_WATERBOLT, 2000);
201  SetCombatScriptStatus(false);
202  break;
203  }
204  }
205  ++m_submergePhases;
206  if (timer)
207  ResetTimer(LURKER_SUBMERGE_ACTIONS, timer);
208  }

226  void ExecuteAction(uint32 action) override
227  {
228  switch (action)
229  {
230  case LURKER_SUBMERGE_PHASE:
231  {
232  m_creature->CastSpell(nullptr, SPELL_CLEAR_ALL_DEBUFFS, TRIGGERED_NONE);
233  if (DoCastSpellIfCan(nullptr, SPELL_SUBMERGE) == CAST_OK)
234  {
235  DoSummonCoilfangNaga();
236  SetCombatScriptStatus(true);
237  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE); // UNIT_STAND_STATE_CUSTOM
238  m_creature->SetStandState(UNIT_STAND_STATE_CUSTOM); // TODO investigate Submerge visual spell to see if it shouldnt do this instead in aura
239  m_submergePhases = 0;
240  ResetTimer(LURKER_SUBMERGE_ACTIONS, MINUTE * IN_MILLISECONDS);
241  DisableCombatAction(LURKER_WHIRL);
242  DisableCombatAction(LURKER_SUBMERGE_PHASE);
243  DisableCombatAction(LURKER_SPOUT);
244  DisableCombatAction(LURKER_GEYSER);
245  DisableCombatAction(LURKER_WATERBOLT);
246  break;
247  }
248  break;
249  }
250  case LURKER_SPOUT:
251  {
252  if (DoCastSpellIfCan(nullptr, SPELL_SPOUT) == CAST_OK)
253  {
254  DoScriptText(EMOTE_DEEP_BREATH, m_creature);
255 
256  // Remove the target focus but allow the boss to face the current victim
257  SetCombatScriptStatus(true);
258  m_meleeEnabled = false;
259  m_creature->MeleeAttackStop(m_creature->GetVictim());
260  m_creature->SetTarget(nullptr);
261 
262  ResetTimer(LURKER_SPOUT_END, 16000);
263  ResetCombatAction(action, 45000);
264  }
265  break;
266  }
267  case LURKER_WHIRL:
268  {
269  if (DoCastSpellIfCan(nullptr, SPELL_WHIRL) == CAST_OK)
270  ResetCombatAction(action, 18000);
271  break;
272  }
273  case LURKER_GEYSER:
274  {
275  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_GEYSER, SELECT_FLAG_PLAYER))
276  if (DoCastSpellIfCan(target, SPELL_GEYSER) == CAST_OK)
277  ResetCombatAction(action, urand(50000, 60000));
278  break;
279  }
280  case LURKER_WATERBOLT:
281  {
282  uint32 timer = 500;
283  // If victim exists we have a target in melee range
284  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
285  m_rangeCheckState = -1;
286  // Spam Waterbolt spell when not tanked
287  else
288  {
289  ++m_rangeCheckState;
290  if (m_rangeCheckState > 1)
291  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_WATERBOLT, SELECT_FLAG_PLAYER))
292  if (DoCastSpellIfCan(target, SPELL_WATERBOLT) == CAST_OK)
293  timer = 2500;
294  }
295  ResetCombatAction(action, timer);
296  break;
297  }
298  }
299  }

302 // Cast the spell that should summon the Lurker-Below
303 bool GOUse_go_strange_pool(Player* /*player*/, GameObject* go)
304 {
305  // There is some chance to fish The Lurker Below, sources are from 20s to 10minutes, average 5min => 20 tries, hence 5%
306  if (urand(0, 99) < 10)
307  {
308  if (ScriptedInstance* pInstance = (ScriptedInstance*)go->GetInstanceData())
309  {
310  if (pInstance->GetData(TYPE_THELURKER_EVENT) == NOT_STARTED || pInstance->GetData(TYPE_THELURKER_EVENT) == FAIL)
311  {
312  pInstance->SetData(TYPE_THELURKER_EVENT, IN_PROGRESS);
313  go->SetRespawnTime(7 * DAY); // dont respawn until reset
314  return true;
315  }
316  }
317  }
318  return false;
319 }
