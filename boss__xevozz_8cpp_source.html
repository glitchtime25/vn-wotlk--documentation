141  void ExecuteAction(uint32 action) override
142  {
143  switch (action)
144  {
145  case XEVOZZ_ARCANE_VOLLEY:
146  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_ARCANE_VOLLEY : SPELL_ARCANE_VOLLEY_H) == CAST_OK)
147  ResetCombatAction(action, 15000);
148  break;
149  case XEVOZZ_ARCANE_BUFFET:
150  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
151  {
152  if (DoCastSpellIfCan(pTarget, m_isRegularMode ? SPELL_ARCANE_BUFFET : SPELL_ARCANE_BUFFET_H) == CAST_OK)
153  ResetCombatAction(action, 15000);
154  }
155  break;
156  case XEVOZZ_SUMMON_SPHERE:
157  if (DoCastSpellIfCan(nullptr, aSummonSphereSpells[urand(0, 2)]) == CAST_OK)
158  {
159  switch (urand(0, 2))
160  {
161  case 0: DoScriptText(SAY_SUMMON_1, m_creature); break;
162  case 1: DoScriptText(SAY_SUMMON_2, m_creature); break;
163  case 2: DoScriptText(SAY_SUMMON_3, m_creature); break;
164  }
165  DisableCombatAction(action);
166 
167  // on heroic a second summon happens immediately
168  if (!m_isRegularMode)
169  ResetCombatAction(XEVOZZ_SUMMON_SPHERE_H, 3000);
170  }
171  break;
172  case XEVOZZ_SUMMON_SPHERE_H:
173  if (DoCastSpellIfCan(nullptr, aSummonSphereHeroicSpells[urand(0, 2)]) == CAST_OK)
174  DisableCombatAction(action);
175  break;
176  }
