146 {
147  if (!target)
148  return;
149 
150  uint8 updatetype = UPDATETYPE_CREATE_OBJECT;
151  uint16 updateFlags = m_updateFlag;

154  if (target == this) // building packet for yourself
155  updateFlags |= UPDATEFLAG_SELF;
156 
157  if (m_itsNewObject)
158  {
159  switch (GetObjectGuid().GetHigh())
160  {
161  case HighGuid::HIGHGUID_DYNAMICOBJECT:
162  case HighGuid::HIGHGUID_CORPSE:
163  case HighGuid::HIGHGUID_PLAYER:
164  case HighGuid::HIGHGUID_UNIT:

173  }
174 
175  if (isType(TYPEMASK_UNIT))
176  {
177  if (static_cast<const Unit*>(this)->GetVictim())
178  updateFlags |= UPDATEFLAG_HAS_ATTACKING_TARGET;
179  }

181  // DEBUG_LOG("BuildCreateUpdate: update-type: %u, object-type: %u got updateFlags: %X", updatetype, m_objectTypeId, updateFlags);
182 
183  ByteBuffer buf(500);
184  buf << uint8(updatetype);
185  buf << GetPackGUID();
186  buf << uint8(m_objectTypeId);
187 
188  BuildMovementUpdate(&buf, updateFlags);

190  UpdateMask updateMask;

285 {

287 
288  *data << uint16(updateFlags); // update flags
289 
290  // 0x20
291  if (updateFlags & UPDATEFLAG_LIVING)
292  {
293  Unit const* unit = static_cast<Unit const*>(this);
294 
295  // TODO: Remove this when moveflag is properly used
296  if (unit->GetTransport() || unit->IsBoarded())
297  const_cast<Unit*>(unit)->m_movementInfo.AddMovementFlag(MOVEFLAG_ONTRANSPORT);
298  else
299  const_cast<Unit*>(unit)->m_movementInfo.RemoveMovementFlag(MOVEFLAG_ONTRANSPORT);
300 
301  // Write movement info
302  // TODO: Move boarding and unboarding and root enable/disable to movement generator
303  if ((unit->m_movementInfo.GetMovementFlags() & MOVEFLAG_SPLINE_ENABLED) && unit->movespline->IsBoarding())
304  {
305  MovementInfo info = unit->m_movementInfo;
306  info.RemoveMovementFlag(MOVEFLAG_ROOT);
307  *data << info;
308  }
309  else
310  *data << unit->m_movementInfo;
311 
312  // Unit speeds
313  *data << float(unit->GetSpeed(MOVE_WALK));
314  *data << float(unit->GetSpeed(MOVE_RUN));
315  *data << float(unit->GetSpeed(MOVE_RUN_BACK));
316  *data << float(unit->GetSpeed(MOVE_SWIM));
317  *data << float(unit->GetSpeed(MOVE_SWIM_BACK));
318  *data << float(unit->GetSpeed(MOVE_FLIGHT));
319  *data << float(unit->GetSpeed(MOVE_FLIGHT_BACK));
320  *data << float(unit->GetSpeed(MOVE_TURN_RATE));
321  *data << float(unit->GetSpeed(MOVE_PITCH_RATE));
322 
323  // 0x08000000
324  if (unit->m_movementInfo.GetMovementFlags() & MOVEFLAG_SPLINE_ENABLED)
325  Movement::PacketBuilder::WriteCreate(*unit->movespline, *data);
326  }
327  else
328  {
329  if (updateFlags & UPDATEFLAG_POSITION)
330  {
331  WorldObject const* wo = static_cast<WorldObject const*>(this);
332 
333  GenericTransport* transport = wo->GetTransport();
334 
335  if (transport)
336  *data << transport->GetPackGUID();
337  else
338  *data << uint8(0);
339 
340  *data << wo->GetPositionX();
341  *data << wo->GetPositionY();
342  *data << wo->GetPositionZ();
343 
344  if (transport)
345  {
346  *data << wo->GetTransOffsetX();
347  *data << wo->GetTransOffsetY();
348  *data << wo->GetTransOffsetZ();
349  }
350  else
351  {
352  *data << wo->GetPositionX();
353  *data << wo->GetPositionY();
354  *data << wo->GetPositionZ();
355  }
356 
357  *data << wo->GetOrientation();
358 
359  if (GetTypeId() == TYPEID_CORPSE)
360  *data << float(wo->GetOrientation());
361  else
362  *data << float(0);
363  }
364  else
365  {
366  // 0x40
367  if (updateFlags & UPDATEFLAG_HAS_POSITION)
368  {
369  // 0x02
370  if (updateFlags & UPDATEFLAG_TRANSPORT)
371  {
372  GameObject const* go = static_cast<GameObject const*>(this);
373  *data << float(go->GetStationaryX());
374  *data << float(go->GetStationaryY());
375  *data << float(go->GetStationaryZ());
376  *data << float(go->GetStationaryO());
377  }
378  else
379  {
380  WorldObject const* wo = static_cast<WorldObject const*>(this);
381  *data << float(wo->GetPositionX());
382  *data << float(wo->GetPositionY());
383  *data << float(wo->GetPositionZ());
384  *data << float(wo->GetOrientation());
385  }
386  }
387  }
388  }
389 
390  // 0x8
391  if (updateFlags & UPDATEFLAG_LOWGUID)
392  {
393  switch (GetTypeId())
394  {
395  case TYPEID_OBJECT:
396  case TYPEID_ITEM:
397  case TYPEID_CONTAINER:
398  case TYPEID_GAMEOBJECT:
399  case TYPEID_DYNAMICOBJECT:
400  case TYPEID_CORPSE:
401  *data << uint32(GetGUIDLow()); // GetGUIDLow()
402  break;
403  case TYPEID_UNIT:
404  *data << uint32(0x0000000B); // unk, can be 0xB or 0xC
405  break;
406  case TYPEID_PLAYER:
407  if (updateFlags & UPDATEFLAG_SELF)
408  *data << uint32(0x0000002F); // unk, can be 0x15 or 0x22
409  else
410  *data << uint32(0x00000008); // unk, can be 0x7 or 0x8
411  break;
412  default:
413  *data << uint32(0x00000000); // unk
414  break;
415  }
416  }
417 
418  // 0x10
419  if (updateFlags & UPDATEFLAG_HIGHGUID)
420  {
421  switch (GetTypeId())
422  {
423  case TYPEID_OBJECT:
424  case TYPEID_ITEM:
425  case TYPEID_CONTAINER:
426  case TYPEID_GAMEOBJECT:
427  case TYPEID_DYNAMICOBJECT:
428  case TYPEID_CORPSE:
429  *data << uint32(GetObjectGuid().GetHigh()); // GetGUIDHigh()
430  break;
431  case TYPEID_UNIT:
432  *data << uint32(0x0000000B); // unk, can be 0xB or 0xC
433  break;
434  case TYPEID_PLAYER:
435  if (updateFlags & UPDATEFLAG_SELF)
436  *data << uint32(0x0000002F); // unk, can be 0x15 or 0x22
437  else
438  *data << uint32(0x00000008); // unk, can be 0x7 or 0x8
439  break;
440  default:
441  *data << uint32(0x00000000); // unk
442  break;
443  }
444  }
445 
446  // 0x4
447  if (updateFlags & UPDATEFLAG_HAS_ATTACKING_TARGET) // packed guid (current target guid)

449  if (((Unit*) this)->GetVictim())
450  *data << ((Unit*) this)->GetVictim()->GetPackGUID();

465  *data << uint32(((Unit*)this)->GetVehicleInfo()->GetVehicleEntry()->m_ID); // vehicle id
466  *data << float(((WorldObject*)this)->GetOrientation());
467  }
468 
469  // 0x200

472  *data << int64(((GameObject*)this)->GetPackedLocalRotation());
473  }

476 void Object::BuildValuesUpdate(uint8 updatetype, ByteBuffer* data, UpdateMask* updateMask, Player* target) const
477 {
478  if (!target)
479  return;
480 
481  bool IsActivateToQuest = false;
482  bool IsPerCasterAuraState = false;
483 
484  if (updatetype == UPDATETYPE_CREATE_OBJECT || updatetype == UPDATETYPE_CREATE_OBJECT2)
485  {
486  if (isType(TYPEMASK_GAMEOBJECT) && !((GameObject*)this)->IsDynTransport())
487  {
488  if (((GameObject*)this)->ActivateToQuest(target) || target->IsGameMaster())
489  IsActivateToQuest = true;
490 
491  updateMask->SetBit(GAMEOBJECT_DYNAMIC);
492  }
493  else if (isType(TYPEMASK_UNIT))
494  {
495  if (((Unit*)this)->HasAuraState(AURA_STATE_CONFLAGRATE))
496  {
497  IsPerCasterAuraState = true;
498  updateMask->SetBit(UNIT_FIELD_AURASTATE);
499  }
500  }
501  }
502  else // case UPDATETYPE_VALUES
503  {
504  if (isType(TYPEMASK_GAMEOBJECT) && !((GameObject*)this)->IsDynTransport())
505  {
506  if (((GameObject*)this)->ActivateToQuest(target) || target->IsGameMaster())
507  IsActivateToQuest = true;
508 
509  updateMask->SetBit(GAMEOBJECT_DYNAMIC);
510  updateMask->SetBit(GAMEOBJECT_BYTES_1); // why do we need this here?
511  }
512  else if (isType(TYPEMASK_UNIT))
513  {
514  if (((Unit*)this)->HasAuraState(AURA_STATE_CONFLAGRATE))
515  {
516  IsPerCasterAuraState = true;
517  updateMask->SetBit(UNIT_FIELD_AURASTATE);
518  }
519  }
520  }
521 
522  MANGOS_ASSERT(updateMask && updateMask->GetCount() == m_valuesCount);
523 
524  *data << (uint8)updateMask->GetBlockCount();
525  data->append(updateMask->GetMask(), updateMask->GetLength());
526 
527  // 2 specialized loops for speed optimization in non-unit case
528  if (isType(TYPEMASK_UNIT)) // unit (creature/player) case
529  {
530  for (uint16 index = 0; index < m_valuesCount; ++index)
531  {
532  if (updateMask->GetBit(index))
533  {
534  if (index == UNIT_NPC_FLAGS)
535  {
536  uint32 appendValue = m_uint32Values[index];
537 
538  if (GetTypeId() == TYPEID_UNIT)
539  {
540  if (!target->canSeeSpellClickOn((Creature*)this))
541  appendValue &= ~UNIT_NPC_FLAG_SPELLCLICK;
542 
543  if (appendValue & UNIT_NPC_FLAG_TRAINER)
544  {
545  if (!((Creature*)this)->IsTrainerOf(target, false))
546  appendValue &= ~(UNIT_NPC_FLAG_TRAINER | UNIT_NPC_FLAG_TRAINER_CLASS | UNIT_NPC_FLAG_TRAINER_PROFESSION);
547  }
548 
549  if (appendValue & UNIT_NPC_FLAG_STABLEMASTER)
550  {
551  if (target->getClass() != CLASS_HUNTER)
552  appendValue &= ~UNIT_NPC_FLAG_STABLEMASTER;
553  }
554 
555  if (appendValue & UNIT_NPC_FLAG_FLIGHTMASTER)
556  {
557  QuestRelationsMapBounds bounds = sObjectMgr.GetCreatureQuestRelationsMapBounds(((Creature*)this)->GetEntry());
558  for (QuestRelationsMap::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
559  {
560  Quest const* pQuest = sObjectMgr.GetQuestTemplate(itr->second);
561  if (target->CanSeeStartQuest(pQuest))
562  {
563  appendValue &= ~UNIT_NPC_FLAG_FLIGHTMASTER;
564  break;
565  }
566  }
567 
568  bounds = sObjectMgr.GetCreatureQuestInvolvedRelationsMapBounds(((Creature*)this)->GetEntry());
569  for (QuestRelationsMap::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
570  {
571  Quest const* pQuest = sObjectMgr.GetQuestTemplate(itr->second);
572  if (target->CanRewardQuest(pQuest, false))
573  {
574  appendValue &= ~UNIT_NPC_FLAG_FLIGHTMASTER;
575  break;
576  }
577  }
578  }
579  }
580 
581  *data << uint32(appendValue);
582  }
583  else if (index == UNIT_FIELD_AURASTATE)
584  {
585  if (IsPerCasterAuraState)
586  {
587  // IsPerCasterAuraState set if related pet caster aura state set already
588  if (((Unit*)this)->HasAuraStateForCaster(AURA_STATE_CONFLAGRATE, target->GetObjectGuid()))
589  *data << m_uint32Values[index];
590  else
591  *data << (m_uint32Values[index] & ~(1 << (AURA_STATE_CONFLAGRATE - 1)));
592  }
593  else
594  *data << m_uint32Values[index];
595  }
596  // FIXME: Some values at server stored in float format but must be sent to client in uint32 format
597  else if (index >= UNIT_FIELD_BASEATTACKTIME && index <= UNIT_FIELD_RANGEDATTACKTIME)
598  {
599  // convert from float to uint32 and send
600  *data << uint32(m_floatValues[index] < 0 ? 0 : m_floatValues[index]);
601  }
602 
603  // there are some float values which may be negative or can't get negative due to other checks
604  else if ((index >= UNIT_FIELD_NEGSTAT0 && index <= UNIT_FIELD_NEGSTAT4) ||
605  (index >= UNIT_FIELD_RESISTANCEBUFFMODSPOSITIVE && index <= (UNIT_FIELD_RESISTANCEBUFFMODSPOSITIVE + 6)) ||
606  (index >= UNIT_FIELD_RESISTANCEBUFFMODSNEGATIVE && index <= (UNIT_FIELD_RESISTANCEBUFFMODSNEGATIVE + 6)) ||
607  (index >= UNIT_FIELD_POSSTAT0 && index <= UNIT_FIELD_POSSTAT4))
608  {
609  *data << uint32(m_floatValues[index]);
610  }
611  else if (index == UNIT_FIELD_HEALTH || index == UNIT_FIELD_MAXHEALTH)
612  {
613  uint32 value = m_uint32Values[index];
614 
615  // Fog of War: replace absolute health values with percentages for non-allied units according to settings
616  if (!static_cast<const Unit*>(this)->IsFogOfWarVisibleHealth(target) &&
617  !target->CanSeeSpecialInfoOf(static_cast<const Unit*>(this)))
618  {
619  switch (index)
620  {
621  case UNIT_FIELD_HEALTH: value = uint32(ceil((100.0 * value) / m_uint32Values[UNIT_FIELD_MAXHEALTH])); break;
622  case UNIT_FIELD_MAXHEALTH: value = 100; break;
623  }
624  }
625 
626  *data << value;
627  }
628  else if (index == UNIT_FIELD_FLAGS)
629  {
630  uint32 value = m_uint32Values[index];
631 
632  // For gamemasters in GM mode:
633  if (target->IsGameMaster())
634  {
635  // Gamemasters should be always able to select units - remove not selectable flag:
636  value &= ~UNIT_FLAG_UNINTERACTIBLE;
637  }
638 
639  // Client bug workaround: Fix for missing chat channels when resuming taxi flight on login
640  // Client does not send any chat joining attempts by itself when taxi flag is on
641  if (target == this && (value & UNIT_FLAG_TAXI_FLIGHT))
642  {
643  if (sWorld.getConfig(CONFIG_BOOL_TAXI_FLIGHT_CHAT_FIX))
644  if (WorldSession* session = static_cast<Player const*>(this)->GetSession())
645  if (!session->IsInitialZoneUpdated())
646  value &= ~UNIT_FLAG_TAXI_FLIGHT;
647  }
648 
649  *data << value;
650  }
651  // Hide special-info for non empathy-casters,
652  // Hide lootable animation for unallowed players
653  // Handle tapped flag
654  else if (index == UNIT_DYNAMIC_FLAGS)
655  {
656  Creature const* creature = static_cast<Creature const*>(this);
657  uint32 dynflagsValue = m_uint32Values[index];
658  bool setTapFlags = false;
659 
660  if (creature->IsAlive())
661  {
662  // Checking SPELL_AURA_EMPATHY and caster
663  if (dynflagsValue & UNIT_DYNFLAG_SPECIALINFO)
664  {
665  bool bIsEmpathy = false;
666  bool bIsCaster = false;
667  Unit::AuraList const& mAuraEmpathy = creature->GetAurasByType(SPELL_AURA_EMPATHY);
668  for (Unit::AuraList::const_iterator itr = mAuraEmpathy.begin(); !bIsCaster && itr != mAuraEmpathy.end(); ++itr)
669  {
670  bIsEmpathy = true; // Empathy by aura set
671  if ((*itr)->GetCasterGuid() == target->GetObjectGuid())
672  bIsCaster = true; // target is the caster of an empathy aura
673  }
674  if (bIsEmpathy && !bIsCaster) // Empathy by aura, but target is not the caster
675  dynflagsValue &= ~UNIT_DYNFLAG_SPECIALINFO;
676  }
677 
678  // creature is alive so, not lootable
679  dynflagsValue = dynflagsValue & ~UNIT_DYNFLAG_LOOTABLE;
680  if (creature->IsInCombat())
681  {
682  // as creature is in combat we have to manage tap flags
683  setTapFlags = true;
684  }
685  else
686  {
687  // creature is not in combat so its not tapped
688  dynflagsValue = dynflagsValue & ~(UNIT_DYNFLAG_TAPPED | UNIT_DYNFLAG_TAPPED_BY_PLAYER);
689  //sLog.outString(">> %s is not in combat so not tapped by %s", this->GetGuidStr().c_str(), target->GetGuidStr().c_str());
690  }
691  }
692  else
693  {
694  // check loot flag
695  if (creature->m_loot && creature->m_loot->CanLoot(target))
696  {
697  // creature is dead and this player can loot it
698  dynflagsValue = dynflagsValue | UNIT_DYNFLAG_LOOTABLE;
699  //sLog.outString(">> %s is lootable for %s", this->GetGuidStr().c_str(), target->GetGuidStr().c_str());
700  }
701  else
702  {
703  // creature is dead but this player cannot loot it
704  dynflagsValue = dynflagsValue & ~UNIT_DYNFLAG_LOOTABLE;
705  //sLog.outString(">> %s is not lootable for %s", this->GetGuidStr().c_str(), target->GetGuidStr().c_str());
706  }
707 
708  // as creature is died we have to manage tap flags
709  setTapFlags = true;
710  }
711 
712  // check tap flags
713  if (setTapFlags)
714  {
715  dynflagsValue = dynflagsValue | UNIT_DYNFLAG_TAPPED;
716  if (creature->IsTappedBy(target))
717  {
718  // creature is in combat or died and tapped by this player
719  dynflagsValue = dynflagsValue | UNIT_DYNFLAG_TAPPED_BY_PLAYER;
720  //sLog.outString(">> %s is tapped by %s", this->GetGuidStr().c_str(), target->GetGuidStr().c_str());
721  }
722  else
723  {
724  // creature is in combat or died but not tapped by this player
725  dynflagsValue = dynflagsValue & ~UNIT_DYNFLAG_TAPPED_BY_PLAYER;
726  //sLog.outString(">> %s is not tapped by %s", this->GetGuidStr().c_str(), target->GetGuidStr().c_str());
727  }
728  }
729

735  }
736 
737  *data << dynflagsValue;
738  }

740  {
741  uint32 value = m_uint32Values[index];
742

744  if (this != target && GetTypeId() == TYPEID_PLAYER)
745  {
746  Player const* thisPlayer = static_cast<Player const*>(this);
747 
748  if (sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_GROUP) && target->IsInGroup(thisPlayer))

753  {
754  switch (targetTeam)

756  case ALLIANCE: value = 1054; break; // "Alliance Generic"
757  case HORDE: value = 1495; break; // "Horde Generic"
758  }
759  }
760  }
761  }

764  }

766  {
767  // send in current format (float as float, uint32 as uint32)
768  *data << m_uint32Values[index];
769  }
770  }
771  }

816  // most likely related to "can enter map" and then should be 0 if can not enter
817

834  break;

841  default:
842  // unknown, not happen.

844  *data << uint16(-1);
845  break;
846  }
847  }

867  }
868  }
869  }

873  {
874  if (updateMask->GetBit(index))
875  {
876  // send in current format (float as float, uint32 as uint32)
