198  void ExecuteAction(uint32 action) override
199  {
200  switch (action)
201  {
202  case AKILZON_ACTION_BERSERK:
203  {
204  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
205  {
206  DoScriptText(SAY_ENRAGE, m_creature);
207  DisableCombatAction(action);
208  }
209  break;
210  }
211  case AKILZON_ACTION_STATIC_DISRUPT:
212  {
213  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
214  {
215  if (DoCastSpellIfCan(target, SPELL_STATIC_DISRUPTION) == CAST_OK)
216  ResetCombatAction(action, urand(7000, 14000));
217  }
218  break;
219  }
220  case AKILZON_ACTION_CALL_LIGHTNING:
221  {
222  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 0, nullptr, SELECT_FLAG_PLAYER))
223  {
224  if (DoCastSpellIfCan(target, SPELL_CALL_LIGHTNING) == CAST_OK)
225  ResetCombatAction(action, urand(15000, 25000));
226  }
227  break;
228  }
229  case AKILZON_ACTION_GUST_OF_WIND:
230  {
231  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
232  {
233  if (DoCastSpellIfCan(target, SPELL_GUST_OF_WIND) == CAST_OK)
234  ResetCombatAction(action, urand(20000, 30000));
235  }
236  break;
237  }
238  case AKILZON_ACTION_STORM_WEATHER:
239  {
240  // change weather 8.5 seconds prior to storm
241  if (m_instance)
242  m_instance->ChangeWeather(true);
243  ResetCombatAction(action, urand(54000, 60000));
244  ResetCombatAction(AKILZON_ACTION_STORM, 8500);
245  break;
246  }
247  case AKILZON_ACTION_STORM:
248  {
249  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
250  {
251  if (DoCastSpellIfCan(target, SPELL_ELECTRICAL_STORM) == CAST_OK)
252  {
253  DoScriptText(EMOTE_STORM, m_creature);
254  DisableCombatAction(action);
255  }
256  }
257  break;
258  }
259  case AKILZON_ACTION_SUMMON_EAGLE:
260  {
261  DoScriptText(urand(0, 1) ? SAY_SUMMON : SAY_SUMMON_ALT, m_creature);
262  DoSummonEagles();
263  ResetCombatAction(action, 60000);
264  break;
265  }
266  }
267  }
