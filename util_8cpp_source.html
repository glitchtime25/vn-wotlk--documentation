34  float computeFallTime(float path_length, bool isSafeFall)
35  {
36  if (path_length < 0.f)
37  return 0.f;
38 
39  float time;
40  if (isSafeFall)
41  {
42  if (path_length >= terminal_savefall_length)
43  time = (path_length - terminal_savefall_length) / terminalSavefallVelocity + terminalSavefallVelocity / gravity;
44  else
45  time = sqrtf(2.f * path_length / gravity);
46  }
47  else
48  {
49  if (path_length >= terminal_length)
50  time = (path_length - terminal_length) / terminalVelocity + terminalFallTime;
51  else
52  time = sqrtf(2.f * path_length / gravity);
53  }
54 
55  return time;
56  }

58  float computeFallElevation(float t_passed, bool isSafeFall, float start_velocity)
59  {
60  float termVel;
61  float result;
62 
63  if (isSafeFall)
64  termVel = terminalSavefallVelocity;
65  else
66  termVel = terminalVelocity;
67 
68  if (start_velocity > termVel)
69  start_velocity = termVel;
70 
71  float terminal_time = terminalFallTime - start_velocity / gravity; // the time that needed to reach terminalVelocity
72 
73  if (t_passed > terminal_time)
74  {
75  result = terminalVelocity * (t_passed - terminal_time) +
76  start_velocity * terminal_time + gravity * terminal_time * terminal_time * 0.5f;
77  }
78  else
79  result = t_passed * (start_velocity + t_passed * gravity * 0.5f);
80 
81  return result;
