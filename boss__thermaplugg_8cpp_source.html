151  void UpdateAI(const uint32 uiDiff) override
152  {
153  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
154  return;
155 
156  // Movement of Summoned mobs
157  if (!m_lLandedBombGUIDs.empty())
158  {
159  for (GuidList::const_iterator itr = m_lLandedBombGUIDs.begin(); itr != m_lLandedBombGUIDs.end(); ++itr)
160  {
161  if (Creature* pBomb = m_creature->GetMap()->GetCreature(*itr))
162  pBomb->GetMotionMaster()->MoveFollow(m_creature, 0.0f, 0.0f);
163  }
164  m_lLandedBombGUIDs.clear();
165  }
166 
167  if (!m_bIsPhaseTwo && m_creature->GetHealthPercent() < 50.0f)
168  {
169  DoScriptText(SAY_PHASE, m_creature);
170  m_bIsPhaseTwo = true;
171  }
172 
173  if (m_uiKnockAwayTimer < uiDiff)
174  {
175  if (m_bIsPhaseTwo)
176  {
177  if (DoCastSpellIfCan(m_creature, SPELL_KNOCK_AWAY_AOE) == CAST_OK)
178  m_uiKnockAwayTimer = 12000;
179  }
180  else
181  {
182  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_KNOCK_AWAY) == CAST_OK)
183  m_uiKnockAwayTimer = urand(17000, 20000);
184  }
185  }
186  else
187  m_uiKnockAwayTimer -= uiDiff;
188 
189  if (m_uiActivateBombTimer < uiDiff)
190  {
191  if (DoCastSpellIfCan(m_creature, m_bIsPhaseTwo ? SPELL_ACTIVATE_BOMB_B : SPELL_ACTIVATE_BOMB_A) == CAST_OK)
192  {
193  m_uiActivateBombTimer = (m_bIsPhaseTwo ? urand(6, 12) : urand(12, 17)) * IN_MILLISECONDS;
194  if (!urand(0, 5)) // TODO, chance/ place for this correct?
195  DoScriptText(SAY_BOMB, m_creature);
196  }
197  }
198  else
199  m_uiActivateBombTimer -= uiDiff;
200 
201  // Spawn bombs
202  if (m_asBombFaces)
203  {
204  for (uint8 i = 0; i < MAX_GNOME_FACES; ++i)
205  {
206  if (m_asBombFaces[i].m_bActivated)
207  {
208  if (m_asBombFaces[i].m_uiBombTimer < uiDiff)
209  {
210  // Calculate the spawning position as 90% between face and thermaplugg spawn-pos, and hight hardcoded
211  float fX = 0.0f, fY = 0.0f;
212  if (GameObject* pFace = m_creature->GetMap()->GetGameObject(m_asBombFaces[i].m_gnomeFaceGuid))
213  {
214  fX = 0.35 * m_afSpawnPos[0] + 0.65 * pFace->GetPositionX();
215  fY = 0.35 * m_afSpawnPos[1] + 0.65 * pFace->GetPositionY();
216  }
217  m_creature->SummonCreature(NPC_WALKING_BOMB, fX, fY, fBombSpawnZ, 0.0f, TEMPSPAWN_CORPSE_DESPAWN, 0);
218  m_asBombFaces[i].m_uiBombTimer = urand(10000, 25000); // TODO
219  }
220  else
221  m_asBombFaces[i].m_uiBombTimer -= uiDiff;
222  }
223  }
224  }
225 
226  DoMeleeAttackIfReady();
227  }
