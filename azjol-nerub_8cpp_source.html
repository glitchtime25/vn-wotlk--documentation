62 void instance_azjol_nerub::OnCreatureCreate(Creature* pCreature)
63 {
64  switch (pCreature->GetEntry())
65  {
66  case NPC_KRIKTHIR:
67  case NPC_GASHRA:
68  case NPC_NARJIL:
69  case NPC_SILTHIK:
70  case NPC_HADRONOX:
71  case NPC_ANUBARAK:
72  m_npcEntryGuidStore[pCreature->GetEntry()] = pCreature->GetObjectGuid();
73  break;
74  case NPC_WORLD_TRIGGER:
75  // sort triggers based on position
76  if (pCreature->GetPositionZ() < 150.0f)
77  m_darterSummonTarget = pCreature->GetObjectGuid();
78  else if (pCreature->GetPositionZ() < 230.0f)
79  m_anubSummonTarget = pCreature->GetObjectGuid();
80  else if (pCreature->GetPositionZ() < 245.0f && pCreature->GetPositionY() < 360.0f)
81  m_guardianSummonTarget = pCreature->GetObjectGuid();
82  else if (pCreature->GetPositionZ() > 270.0f && pCreature->GetPositionZ() < 285.0f && pCreature->GetPositionY() < 410.0f)
83  m_venomancerSummonTarget = pCreature->GetObjectGuid();
84  break;
85  case NPC_ANUBAR_CRUSHER:
86  m_lCrusherGuids.push_back(pCreature->GetObjectGuid());
87  break;
88  case NPC_WORLD_TRIGGER_LARGE:
89  if (pCreature->IsTemporarySummon())
90  return;
91 
92  m_lSpiderTriggersGuids.push_back(pCreature->GetObjectGuid());
93 
94  // sort the triggers
95  if (pCreature->GetPositionZ() > 750.0f)
96  {
97  if (pCreature->GetPositionX() < 500.0f)
98  m_rightTargetTriggerGuid = pCreature->GetObjectGuid();
99  else
100  m_leftTargetTriggerGuid = pCreature->GetObjectGuid();
101  }
102  break;
103  }
104 }

106 void instance_azjol_nerub::OnCreatureRespawn(Creature* pCreature)
107 {
108  switch (pCreature->GetEntry())
109  {
110  // following have passive behavior movement
111  case NPC_WORLD_TRIGGER_LARGE:
112  case NPC_WORLD_TRIGGER:
113  case NPC_IMPALE_TARGET:
114  pCreature->AI()->SetReactState(REACT_PASSIVE);
115  pCreature->SetCanEnterCombat(false);
116  break;
117  // following creatures have WP movement
118  case NPC_ANUBAR_CHAMPION_1:
119  case NPC_ANUBAR_CRYPT_FIEND_1:
120  case NPC_ANUBAR_NECROMANCER_1:
121  case NPC_ANUBAR_GUARDIAN:
122  case NPC_ANUBAR_VENOMANCER:
123  pCreature->SetWalk(false);
124  pCreature->GetMotionMaster()->MoveWaypoint();
125  break;
126  // following creature has levitate and WP movement
127  case NPC_ANUBAR_DARTER:
128  pCreature->SetLevitate(true);
129  pCreature->SetWalk(false);
130  pCreature->GetMotionMaster()->MoveWaypoint();
131  break;
132  // following creatures have WP movement with variable path
133  case NPC_ANUBAR_CHAMPION_2:
134  case NPC_ANUBAR_CRYPT_FIEND_2:
135  case NPC_ANUBAR_NECROMANCER_2:
136  pCreature->SetWalk(false);
137  pCreature->GetMotionMaster()->MoveWaypoint(pCreature->GetPositionX() < 500.0f ? 0 : 1);
138  break;
139  // following creatures attack the player directly
140  case NPC_ANUBAR_CHAMPION_3:
141  case NPC_ANUBAR_CRYPT_FIEND_3:
142  case NPC_ANUBAR_NECROMANCER_3:
143  pCreature->CastSpell(pCreature, SPELL_CHECK_RESET, TRIGGERED_OLD_TRIGGERED);
144 
145  if (Player* pTarget = instance->GetPlayer(m_playerGuid))
146  {
147  if (pTarget->IsAlive())
148  pCreature->AI()->AttackStart(pTarget);
149  }
150  else if (Player* pPlayer = GetPlayerInMap(true))
151  pCreature->AI()->AttackStart(pPlayer);
152  else if (Creature* pHadronox = GetSingleCreatureFromStorage(NPC_HADRONOX))
153  pCreature->AI()->AttackStart(pHadronox);
154  break;
155  // following creatures have jump movement
156  case NPC_ANUBAR_ASSASSIN:
157  if (Creature* pTrigger = instance->GetCreature(m_anubSummonTarget))
158  {
159  float fX, fY, fZ;
160  pCreature->GetRandomPoint(pTrigger->GetPositionX(), pTrigger->GetPositionY(), pTrigger->GetPositionZ(), 15.0f, fX, fY, fZ);
161  pCreature->GetMotionMaster()->MoveJump(fX, fY, fZ, 3 * pCreature->GetSpeed(MOVE_RUN), 10.0f);
162  }
163  break;
164  }

167 void instance_azjol_nerub::OnCreatureDeath(Creature* pCreature)
168 {
169  uint32 uiEntry = pCreature->GetEntry();
170  if (uiEntry == NPC_GASHRA || uiEntry == NPC_NARJIL || uiEntry == NPC_SILTHIK)
171  {
172  if (m_auiEncounter[TYPE_KRIKTHIR] == NOT_STARTED)
173  m_uiWatcherTimer = 5000;
174 
175  // Set achiev criteriat to false if one of the watchers dies
176  m_bWatchHimDie = false;
177  }
178  else if (uiEntry == NPC_ANUBAR_CRUSHER)

180  m_lCrusherGuids.remove(pCreature->GetObjectGuid());
181 
182  // make boss passive to ignore mobs and speed up the tunnel movement
183  if (m_lCrusherGuids.empty())

186  {
187  pHadronox->AI()->EnterEvadeMode();
188  pHadronox->AI()->SetReactState(REACT_PASSIVE);
189 
190  pHadronox->SetWalk(false);
191  pHadronox->GetMotionMaster()->Clear(false, true);
192  pHadronox->GetMotionMaster()->MoveWaypoint();
193  }
194  }

198 void instance_azjol_nerub::OnCreatureEnterCombat(Creature* pCreature)
199 {
200  uint32 uiEntry = pCreature->GetEntry();
201 
202  if (uiEntry == NPC_GASHRA || uiEntry == NPC_NARJIL || uiEntry == NPC_SILTHIK)
203  {
204  // Creature enter combat is not equal to having a victim yet.
205  if (!m_playerGuid && pCreature->GetVictim())
206  m_playerGuid = pCreature->GetVictim()->GetBeneficiaryPlayer()->GetObjectGuid();
207  }
208  else if (uiEntry == NPC_ANUBAR_CRUSHER)
209  {
210  // there is only one Crusher which is spawned in DB; the script reacts only to that one
211  if (pCreature->IsTemporarySummon())
212  return;
213 
214  // Only for the first try
215  if (GetData(TYPE_HADRONOX) == IN_PROGRESS)
216  return;
217 
218  DoScriptText(SAY_CRUSHER_AGGRO, pCreature);
219 
220  if (pCreature->GetVictim())
221  m_playerGuid = pCreature->GetVictim()->GetBeneficiaryPlayer()->GetObjectGuid();
222 
223  // spawn and store additional triggers for player attack
224  for (uint8 i = 0; i < MAX_GAUNTLET_SPELLS; ++i)
225  if (Creature* pTrigger = pCreature->SummonCreature(triggerSummonData[i].entry, triggerSummonData[i].x, triggerSummonData[i].y, triggerSummonData[i].z, triggerSummonData[i].o, TEMPSPAWN_DEAD_DESPAWN, 0))
226  m_playerTriggerGuid[i] = pTrigger->GetObjectGuid();
227 
228  // Spawn 6 extra spawns; movement in DB
229  for (uint8 i = 0; i < MAX_GAUNTLET_SPAWNS; ++i)
230  {
231  if (Creature* pMob = pCreature->SummonCreature(gauntletSummonData[i].entry, gauntletSummonData[i].x, gauntletSummonData[i].y, gauntletSummonData[i].z, gauntletSummonData[i].o, TEMPSPAWN_DEAD_DESPAWN, 0))
232  {
233  pMob->SetWalk(false);
234  pMob->GetMotionMaster()->MoveWaypoint(gauntletSummonData[i].pathId);
235  }
236  }
237 
238  // start gauntlet; spider moves up the tunnel
239  if (Creature* pHadronox = GetSingleCreatureFromStorage(NPC_HADRONOX))
240  DoScriptText(EMOTE_MOVE_TUNNEL, pHadronox);
241 
242  SetData(TYPE_HADRONOX, IN_PROGRESS);
243 
244  m_uiSpiderSummonTimer = 20000;
245  }
246 }

267 void instance_azjol_nerub::SetData(uint32 uiType, uint32 uiData)
268 {

272  m_auiEncounter[uiType] = uiData;

288  m_auiEncounter[uiType] = uiData;
289  DoUseDoorOrButton(GO_DOOR_ANUBARAK_1);

293  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_ANUB_ID);
294  break;
295  }
296 
297  if (uiData == DONE)
298  {
299  OUT_SAVE_INST_DATA;
300 
301  std::ostringstream saveStream;

303 
304  m_strInstData = saveStream.str();

355 // Method that will send watchers to engange in combat for Krikthir
356 void instance_azjol_nerub::DoSendWatcherOrKrikthir()
357 {
358  Creature* pAttacker = nullptr;
359  Creature* pKrikthir = GetSingleCreatureFromStorage(NPC_KRIKTHIR);
360 
361  if (!pKrikthir)
362  return;
363 
364  for (unsigned int aWatcher : aWatchers)
365  {
366  if (Creature* pTemp = GetSingleCreatureFromStorage(aWatcher))
367  {
368  if (pTemp->IsAlive())
369  {
370  if (pAttacker && urand(0, 1))
371  continue;
372  pAttacker = pTemp;
373  }
374  }
375  }
376 
377  if (pAttacker)
378  {
379  switch (urand(0, 2))
380  {
381  case 0: DoScriptText(SAY_SEND_GROUP_1, pKrikthir); break;
382  case 1: DoScriptText(SAY_SEND_GROUP_2, pKrikthir); break;
383  case 2: DoScriptText(SAY_SEND_GROUP_3, pKrikthir); break;
384  }
385  }
386  else
387  pAttacker = pKrikthir;
388 
389  if (Player* pTarget = instance->GetPlayer(m_playerGuid))
390  {
391  if (pTarget->IsAlive())
392  pAttacker->AI()->AttackStart(pTarget);
393  }
394 }

396 // Method to reset the web triggers that spawn the gauntlet mobs
397 void instance_azjol_nerub::ResetHadronoxTriggers()
398 {
399  // Drop the summon auras from the triggers
400  for (const auto& guid : m_lSpiderTriggersGuids)
401  {
402  if (Creature* pTrigger = instance->GetCreature(guid))
403  {
404  pTrigger->RemoveAurasDueToSpell(SPELL_SUMMON_CHAMPION_S);
405  pTrigger->RemoveAurasDueToSpell(SPELL_SUMMON_NECROMANCER_S);
406  pTrigger->RemoveAurasDueToSpell(SPELL_SUMMON_CRYPT_FIEND_S);
407  }
408  }
409 
410  // despawn temp triggers
411  for (uint8 i = 0; i < MAX_GAUNTLET_SPELLS; ++i)
412  if (Creature* pTrigger = instance->GetCreature(m_playerTriggerGuid[i]))
413  pTrigger->ForcedDespawn();
414 
415  // stop the timer
416  m_uiSpiderSummonTimer = 0;
417 }

419 void instance_azjol_nerub::Update(uint32 uiDiff)
420 {
421  // timer to engange watchers for Krikthir
422  if (m_uiWatcherTimer)
423  {
424  if (m_uiWatcherTimer <= uiDiff)
425  {
426  DoSendWatcherOrKrikthir();
427  m_uiWatcherTimer = 0;
428  }
429  else
430  m_uiWatcherTimer -= uiDiff;
431  }
432 
433  // apply period auras on the triggers
434  if (m_uiPeriodicAuraTimer)
435  {
436  if (m_uiPeriodicAuraTimer <= uiDiff)
437  {
438  for (const auto& guid : m_lSpiderTriggersGuids)
439  if (Creature* pTrigger = instance->GetCreature(guid))
440  pTrigger->CastSpell(pTrigger, aSpiderSummonSpells[m_uiPeriodicAuraStage], TRIGGERED_OLD_TRIGGERED);
441 
442  ++m_uiPeriodicAuraStage;
443  m_uiPeriodicAuraTimer = m_uiPeriodicAuraStage == 2 ? 0 : 5000;
444  }
445  else
446  m_uiPeriodicAuraTimer -= uiDiff;
447  }
448 
449  // spider summon timer
450  if (m_uiSpiderSummonTimer)
451  {
452  if (m_uiSpiderSummonTimer <= uiDiff)
453  {
454  for (uint8 i = 0; i < MAX_GAUNTLET_SPELLS; ++i)
455  if (Creature* pTrigger = instance->GetCreature(m_playerTriggerGuid[i]))
456  pTrigger->CastSpell(pTrigger, aPlayerSummonSpells[i], TRIGGERED_OLD_TRIGGERED);
457 
458  m_uiSpiderSummonTimer = 60000;
459  }
460  else
461  m_uiSpiderSummonTimer -= uiDiff;
462  }
463 }
