28 void TemporarySpawn::Update(const uint32 diff)
29 {
30  switch (m_type)
31  {
32  case TEMPSPAWN_MANUAL_DESPAWN:
33  break;
34 
35  case TEMPSPAWN_TIMED_DESPAWN:
36  {
37  if (IsExpired())
38  {
39  UnSummon();
40  return;
41  }
42  break;
43  }
44 
45  case TEMPSPAWN_TIMED_OOC_DESPAWN:
46  {
47  if (IsAlive())
48  {
49  if (!IsInCombat())
50  {
51  if (IsExpired())
52  {
53  UnSummon();
54  return;
55  }
56  }
57  else m_expirationTimestamp = GetMap()->GetCurrentClockTime() + std::chrono::milliseconds(m_lifetime);
58  }
59  else if (IsDespawned())
60  {
61  UnSummon();
62  return;
63  }
64  break;
65  }
66 
67  case TEMPSPAWN_CORPSE_TIMED_DESPAWN:
68  {
69  if (IsCorpse())
70  {
71  if (IsExpired())
72  {
73  UnSummon();
74  return;
75  }
76  }
77  if (IsDespawned())
78  {
79  UnSummon();
80  return;
81  }
82  break;
83  }
84 
85  case TEMPSPAWN_CORPSE_DESPAWN:
86  {
87  // if m_deathState is DEAD, CORPSE was skipped
88  if (IsDead())
89  {
90  UnSummon();
91  return;
92  }
93 
94  break;
95  }
96 
97  case TEMPSPAWN_DEAD_DESPAWN:
98  {
99  if (IsDespawned())
100  {
101  UnSummon();
102  return;
103  }
104  break;
105  }
106 
107  case TEMPSPAWN_TIMED_OOC_OR_CORPSE_DESPAWN:
108  {
109  // if m_deathState is DEAD, CORPSE was skipped
110  if (IsDead())
111  {
112  UnSummon();
113  return;
114  }
115 
116  if (!IsInCombat())
117  {
118  if (IsExpired())
119  {
120  UnSummon();
121  return;
122  }
123  }
124  else m_expirationTimestamp = GetMap()->GetCurrentClockTime() + std::chrono::milliseconds(m_lifetime);
125 
126  break;
127  }
128 
129  case TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN:
130  {
131  // if m_deathState is DEAD, CORPSE was skipped
132  if (IsDespawned())
133  {
134  UnSummon();
135  return;
136  }
137 
138  if (!IsInCombat() && IsAlive() && !GetCharmerGuid())
139  {
140  if (IsExpired())
141  {
142  UnSummon();
143  return;
144  }
145  }
146  else m_expirationTimestamp = GetMap()->GetCurrentClockTime() + std::chrono::milliseconds(m_lifetime);
147 
148  break;
149  }
150 
151  case TEMPSPAWN_TIMED_OR_CORPSE_DESPAWN:
152  {
153  // if m_deathState is DEAD, CORPSE was skipped
154  if (IsDead())
155  {
156  UnSummon();
157  return;
158  }
159  if (IsExpired())
160  {
161  UnSummon();
162  return;
163  }
164  break;
165  }
166 
167  case TEMPSPAWN_TIMED_OR_DEAD_DESPAWN:
168  {
169  // if m_deathState is DEAD, CORPSE was skipped
170  if (IsDespawned())
171  {
172  UnSummon();
173  return;
174  }
175  if (!GetCharmerGuid())
176  {
177  if (IsExpired())
178  {
179  UnSummon();
180  return;
181  }
182  }
183  break;
184  }
185 
186  default:
187  UnSummon();
188  sLog.outError("Temporary summoned creature (entry: %u) have unknown type %u of ", GetEntry(), m_type);
189  break;
190  }
191 
192  switch (m_deathState)
193  {
194  case ALIVE:
195  if (m_linkedToOwnerAura & TEMPSPAWN_LINKED_AURA_OWNER_CHECK)
196  {
197  if (!CheckAuraOnOwner())
198  UnSummon();
199  }
200  break;
201 
202  case DEAD:
203  case CORPSE:
204  if (m_linkedToOwnerAura & TEMPSPAWN_LINKED_AURA_REMOVE_OWNER)
205  {
206  RemoveAuraFromOwner();
207  m_linkedToOwnerAura = 0; // we dont need to recheck
208  }
209 
210  default:
211  break;
212  }
213 
214  Creature::Update(diff);
215 }

236 void TemporarySpawn::UnSummon()
237 {

239 
240  if (m_linkedToOwnerAura & TEMPSPAWN_LINKED_AURA_REMOVE_OWNER)
241  RemoveAuraFromOwner();
242 
243  if (GetSpawnerGuid().IsCreatureOrVehicle())
244  {

247  sum->AI()->SummonedCreatureDespawn(this);
