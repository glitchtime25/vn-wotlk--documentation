259  // Function to respawn GOs
260  void DoRespawnObjects(uint32 uiEntry, float fRange)
261  {
262  GameObjectList lBarrelsInRange;
263  m_lExplosivesGuidsList.clear();
264  GetGameObjectListWithEntryInGrid(lBarrelsInRange, m_creature, uiEntry, fRange);
265 
266  if (lBarrelsInRange.empty())
267  {
268  script_error_log("Bloodmyst Isle: ERROR Failed to find any gameobjects of entry %u in range.", uiEntry);
269  return;
270  }
271 
272  // respawn explosives and store for future use
273  for (GameObjectList::const_iterator itr = lBarrelsInRange.begin(); itr != lBarrelsInRange.end(); ++itr)
274  {
275  (*itr)->SetRespawnTime(5 * MINUTE);
276  (*itr)->Refresh();
277  m_lExplosivesGuidsList.push_back((*itr)->GetObjectGuid());
278  }
279  }
280 
281  // Function to use the recently spawned GOs
282  void DoUseExplosiveTraps()
283  {
284  for (GuidList::const_iterator itr = m_lExplosivesGuidsList.begin(); itr != m_lExplosivesGuidsList.end(); ++itr)
285  {
286  if (GameObject* pTrap = m_creature->GetMap()->GetGameObject(*itr))
287  {
288  pTrap->SendGameObjectCustomAnim(pTrap->GetObjectGuid());

290  }
291  }
292  }

294  // Function to respawn fire GOs
295  void DoRespawnFires(bool bFirstEvent)
296  {
297  GameObjectList lFiresInRange;
298  GetGameObjectListWithEntryInGrid(lFiresInRange, m_creature, GO_COIL_FIRE_L, 110.0f);
299  GetGameObjectListWithEntryInGrid(lFiresInRange, m_creature, GO_COIL_FIRE_S, 110.0f);
300 
301  if (lFiresInRange.empty())
302  {
303  script_error_log("Bloodmyst Isle: ERROR Failed to find any gameobjects of entry %u and %u in range.", GO_COIL_FIRE_L, GO_COIL_FIRE_S);
304  return;
305  }
306 
307  for (GameObjectList::const_iterator itr = lFiresInRange.begin(); itr != lFiresInRange.end(); ++itr)
308  {
309  if ((bFirstEvent && (*itr)->GetPositionZ() < 150.0f) || (!bFirstEvent && (*itr)->GetPositionZ() > 150.0f))
310  {
311  (*itr)->SetRespawnTime(5 * MINUTE);
312  (*itr)->Refresh();
313  }
314  }

317  void JustDidDialogueStep(int32 iEntry) override
318  {
319  switch (iEntry)
320  {
321  case POINT_ID_TEXT_1:
322  if (Player* pPlayer = GetPlayerForEscort())
323  DoScriptText(SAY_ESCORT_LEGOSO_3, m_creature, pPlayer);
324  break;
325  case POINT_ID_TEXT_2:
326  if (Player* pPlayer = GetPlayerForEscort())
327  DoScriptText(SAY_ESCORT_LEGOSO_6, m_creature, pPlayer);
328  break;
329  case POINT_ID_TEXT_3:
330  if (Player* pPlayer = GetPlayerForEscort())
331  DoScriptText(SAY_ESCORT_LEGOSO_10, m_creature, pPlayer);
332  break;
333  case SAY_ESCORT_LEGOSO_4:
334  // first event
335  SetEscortPaused(false);
336  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
337  m_creature->HandleEmote(EMOTE_STATE_NONE);
338 
339  DoRespawnObjects(GO_EXPLOSIVES_1, 10.0f);
340  break;
341  case POINT_ID_COUNT_FINISHED:
342  // different actions depending on the event
343  if (m_bFirstExplosives)
344  {
345  StartNextDialogueText(POINT_ID_EXPLOSIVES_2);
346  m_bFirstExplosives = false;
347  }
348  break;
349  case POINT_ID_EXPLOSIVES_2:
350  DoUseExplosiveTraps();
351  break;
352  case POINT_ID_EXPLOSIVES_3:
353  m_creature->HandleEmote(EMOTE_ONESHOT_CHEER_NOSHEATHE);
354  break;
355  case POINT_ID_EXPLOSIVES_4:
356  DoRespawnFires(true);
357  SetEscortPaused(false);
358  break;
359  case POINT_ID_EXPLOSIVES_6:
360  // second event
361  SetEscortPaused(false);
362  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
363 
364  DoRespawnObjects(GO_EXPLOSIVES_2, 10.0f);
365  break;
366  case SAY_LEGOSO_AGGRO:
367  // explode traps and start combat event
368  DoUseExplosiveTraps();
369  if (Creature* pSironas = GetClosestCreatureWithEntry(m_creature, NPC_SIRONAS, 40.0f))
370  {
371  m_creature->SetFacingToObject(pSironas);
372  pSironas->InterruptNonMeleeSpells(false);
373  pSironas->RemoveAllAurasOnEvade();
374  // ToDo: find the proper spell to scale up Sironas
375  }
376  break;
377  case POINT_ID_EXPLOSIVES_7:
378  DoRespawnFires(false);
379  if (Creature* pSironas = GetClosestCreatureWithEntry(m_creature, NPC_SIRONAS, 40.0f))
380  {
381  pSironas->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
382  pSironas->AI()->AttackStart(m_creature);
383  }
384  break;
385  case QUEST_ENDING_THEIR_WORLD:
386  m_creature->HandleEmote(EMOTE_ONESHOT_CHEER_NOSHEATHE);
387  // complete quest
388  if (Player* pPlayer = GetPlayerForEscort())
389  pPlayer->RewardPlayerAndGroupAtEventExplored(QUEST_ENDING_THEIR_WORLD, m_creature);
390  break;
391  case POINT_ID_EXPLOSIVES_9:
392  // event complete
393  SetEscortPaused(false);
394  break;
395  }
396  }
