358 {

361  return nullptr;

364  sObjectMgr.GetPlayerNameByGUID(inviteeGuid, name);
365

368  {
369  SendCalendarCommandResult(sender, CALENDAR_ERROR_ALREADY_INVITED_TO_EVENT_S, name.c_str());
370  return nullptr;
371  }
372 
373  // check if player can still have new invite (except for event creator)
374  if (!event->IsGuildAnnouncement() && event->CreatorGuid != inviteeGuid)
375  {
376  if (!CanAddInviteTo(inviteeGuid))
377  {
378  SendCalendarCommandResult(sender, CALENDAR_ERROR_OTHER_INVITES_EXCEEDED_S, name.c_str());

384 
385  if (!event->IsGuildAnnouncement())
386  SendCalendarEventInvite(invite);
387 
388  if (!event->IsGuildEvent() || invite->InviteeGuid == event->CreatorGuid)
389  SendCalendarEventInviteAlert(invite);
390 
391  if (event->IsGuildAnnouncement())
392  {
393  // no need to realy add invite for announcements
394  delete invite;

397 
398  DEBUG_FILTER_LOG(LOG_FILTER_CALENDAR, "Add Invite> eventId[" UI64FMTD "], senderGuid[%s], inviteGuid[%s], Status[%u], rank[%u], text[%s], time[%u]",

409  invite->InviteId,
410  event->EventId,
411  inviteeGuid.GetCounter(),
412  senderGuid.GetCounter(),
413  uint32(status),
414  uint32(statusTime),
415  uint32(rank));
416 
417  return invite;
