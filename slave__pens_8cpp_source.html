70 void instance_slave_pens::Update(const uint32 diff)
71 {
72  if (m_quagmirranTimer)
73  {
74  if (m_quagmirranTimer <= diff)
75  {
76  m_quagmirranTimer = 3000;
77  if (Creature* quagmirran = GetSingleCreatureFromStorage(NPC_QUAGMIRRAN))
78  {
79  if (quagmirran->IsAlive() && !quagmirran->IsInCombat())
80  {
81  for (const auto& data : instance->GetPlayers())
82  {
83  if (data.getSource()->GetDistance(quagmirran, true, DIST_CALC_NONE) < 140.f * 140.f)
84  {
85  m_quagmirranTimer = 0;
86  quagmirran->GetMotionMaster()->MoveWaypoint();
87  }
88  }
89  }
90  else
91  m_quagmirranTimer = 0;
92  }
93  else
94  m_quagmirranTimer = 0;
95  }
96  else m_quagmirranTimer -= diff;
97  }
