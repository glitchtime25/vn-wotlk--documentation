197  DoUseDoorOrButton(GO_PORTCULLIS_EXIT_2);
198  }
199  m_auiEncounter[uiType] = uiData;
200  break;
201  case TYPE_BELLOW_1:
202  m_auiEncounter[uiType] = uiData;
203  break;
204  case TYPE_BELLOW_2:
205  m_auiEncounter[uiType] = uiData;
206  break;
207  case TYPE_BELLOW_3:
208  m_auiEncounter[uiType] = uiData;
209  break;
210  }
211 
212  if (uiData == DONE)
213  {
214  OUT_SAVE_INST_DATA;
215 
216  std::ostringstream saveStream;

220  m_strInstData = saveStream.str();
221 
222  SaveToDB();
