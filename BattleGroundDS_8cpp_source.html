42 void BattleGroundDS::Update(uint32 diff)
43 {
44  BattleGround::Update(diff);
45 
46  // push people from the tubes
47  if (m_uiFlushTimer)
48  {
49  if (m_uiFlushTimer <= diff)
50  {
51  // cast flush spell
52  for (const auto& guid : m_lGateTriggersGuids)
53  if (Creature* trigger = GetBgMap()->GetCreature(guid))
54  trigger->CastSpell(trigger, BG_DS_SPELL_FLUSH, TRIGGERED_OLD_TRIGGERED);
55 
56  if (Creature* trigger = GetBgMap()->GetCreature(m_worldTriggerGuid))
57  {
58  // knockback players manually due to missing triggered spell 61698
59  for (const auto& itr : GetPlayers())
60  {
61  Player* plr = sObjectMgr.GetPlayer(itr.first);
62  if (!plr)
63  continue;
64 
65  if (plr->GetPositionZ() < 11.0f)
66  continue;
67 
68  // knockback player depending on location
69  if (plr->IsWithinDist2d(1249.43f, 764.601f, 40))
70  trigger->CastSpell(plr, BG_DS_SPELL_TRACTOR_BEAM, TRIGGERED_NONE);
71 
72  if (plr->IsWithinDist2d(1333.51f, 818.031f, 40))
73  trigger->CastSpell(plr, BG_DS_SPELL_TRACTOR_BEAM, TRIGGERED_NONE);
74  }
75  }
76 
77  m_uiFlushTimer = 0;
78  }
79  else
80  m_uiFlushTimer -= diff;
81  }
82 
83  // Waterfall
84  if (m_uiWaterfallTimer < diff)
85  {
86  switch (m_uiWaterfallStage)
87  {
88  case 0:
89  // prepare waterfall
90  if (GameObject* pAnim = GetBgMap()->GetGameObject(m_waterfallAnimGuid))
91  pAnim->UseDoorOrButton();
92  ++m_uiWaterfallStage;
93  m_uiWaterfallTimer = 5000;
94  break;
95  case 1:
96  // set collision
97  if (GameObject* pCollision = GetBgMap()->GetGameObject(m_waterfallCollisionGuid))
98  pCollision->UseDoorOrButton();
99  ++m_uiWaterfallStage;
100  m_uiWaterfallSpellTimer = 1500;
101  m_uiWaterfallTimer = 25000;
102  break;
103  case 2:
104  // reset waterfall
105  if (GameObject* pCollision = GetBgMap()->GetGameObject(m_waterfallCollisionGuid))
106  pCollision->ResetDoorOrButton();
107  if (GameObject* pAnim = GetBgMap()->GetGameObject(m_waterfallAnimGuid))
108  pAnim->ResetDoorOrButton();
109  m_uiWaterfallStage = 0;
110  m_uiWaterfallSpellTimer = 0;
111  m_uiWaterfallTimer = 30000;
112  break;
113  }
114  }
115  else
116  m_uiWaterfallTimer -= diff;
117 
118  // cast water spout spell on timer
119  if (m_uiWaterfallSpellTimer)
120  {
121  if (m_uiWaterfallSpellTimer <= diff)
122  {
123  if (Creature* trigger = GetBgMap()->GetCreature(m_waterfallTriggerGuid))
124  trigger->CastSpell(trigger, BG_DS_SPELL_WATER_SPOUT, TRIGGERED_OLD_TRIGGERED);
125 
126  m_uiWaterfallSpellTimer = 1500;
127  }
128  else
129  m_uiWaterfallSpellTimer -= diff;
130  }
131 }
