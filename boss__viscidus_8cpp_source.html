225  void SetPhase(uint8 phase)
226  {
227  // Clean-up any potential phase aura before setting new phase
228  m_creature->RemoveAurasDueToSpell(SPELL_VISCIDUS_SLOWED);
229  m_creature->RemoveAurasDueToSpell(SPELL_VISCIDUS_SLOWED_MORE);
230  m_creature->RemoveAurasDueToSpell(SPELL_VISCIDUS_FREEZE);
231 
232  switch (phase)
233  {
234  case PHASE_NORMAL:
235  m_hitCount = 0;
236  DoCastSpellIfCan(nullptr, SPELL_MEMBRANE_VISCIDUS, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
237  DoCastSpellIfCan(nullptr, SPELL_VISCIDUS_WEAKNESS, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
238  DoCastSpellIfCan(m_creature, SPELL_HATE_TO_ZERO);
239  break;
240  case PHASE_SLOWED:
241  {
242  DoScriptText(EMOTE_SLOW, m_creature);
243  DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_SLOWED, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
244  break;
245  }
246  case PHASE_SLOWED_MORE:
247  {
248  DoScriptText(EMOTE_FREEZE, m_creature);
249  DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_SLOWED_MORE, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
250  break;
251  }
252  case PHASE_FROZEN:
253  {
254  // Reset hit count for frozen phase and shattering
255  m_hitCount = 0;
256  DoScriptText(EMOTE_FROZEN, m_creature);
257  DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_FREEZE, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
258  break;
259  }
260  case PHASE_EXPLODED:
261  {
262  m_creature->RemoveAurasDueToSpell(SPELL_VISCIDUS_FREEZE);
263  // In case there is less than 10% HP left, Viscidus triggers suicide
264  if (m_creature->GetHealthPercent() <= 10.0f)
265  DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_SUICIDE_TRIGGER, CAST_TRIGGERED);
266  // Else, he explodes in small globs that must be destroyed
267  else if (DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_EXPLODE, CAST_TRIGGERED|CAST_INTERRUPT_PREVIOUS) == CAST_OK)
268  ResetTimer(VISCIDUS_EXPLODE, 2 * IN_MILLISECONDS);
269  break;
270  }
271  case PHASE_REJOIN:
272  {
273  m_creature->RemoveAurasDueToSpell(SPELL_INVIS_STALKER);
274  m_creature->RemoveAurasDueToSpell(SPELL_STUN_SELF);
275  m_creature->RemoveAurasDueToSpell(SPELL_INVIS_SELF);
276 
277  auto heal = (int32)((float)m_creature->GetMaxHealth() * 0.05f * (float)(m_aliveGlobs));
278  m_creature->CastCustomSpell(m_creature, SPELL_VISCIDUS_GROWS, &heal, nullptr, nullptr, TRIGGERED_IGNORE_UNSELECTABLE_FLAG|TRIGGERED_IGNORE_UNATTACKABLE_FLAG);
279  ResetTimer(VISCIDUS_REJOIN, 1 * IN_MILLISECONDS);
280  break;
281  }
282  case PHASE_SUICIDE:
283  {
284  DoCastSpellIfCan(m_creature, SPELL_VISCIDUS_SUICIDE_TRIGGER, CAST_TRIGGERED);
285  break;
286  }
287  // Unexpected phase requested : return before setting
288  default:
289  return;
290  }
291  m_phase = phase;
292  }

294  void IncreaseHitCount()
295  {
296  ++m_hitCount;
297 
298  switch (m_phase)
299  {
300  case PHASE_NORMAL:
301  {
302  if (m_hitCount == HITCOUNT_SLOW)
303  SetPhase(PHASE_SLOWED);
304  break;
305  }
306  case PHASE_SLOWED:
307  {
308  if (m_hitCount == HITCOUNT_SLOW_MORE)
309  SetPhase(PHASE_SLOWED_MORE);
310  break;
311  }
312  case PHASE_SLOWED_MORE:
313  {
314  if (m_hitCount == HITCOUNT_FREEZE)
315  SetPhase(PHASE_FROZEN);
316  break;
317  }
318  case PHASE_FROZEN:
319  {
320  if (m_hitCount == HITCOUNT_CRACK)
321  DoScriptText(EMOTE_CRACK, m_creature);
322  else if (m_hitCount == HITCOUNT_SHATTER)
323  DoScriptText(EMOTE_SHATTER, m_creature);
324  else if (m_hitCount == HITCOUNT_EXPLODE)
325  SetPhase(PHASE_EXPLODED);
326  break;
327  }
328  default:
329  return;
330  }
331  }

333  void ReceiveAIEvent(AIEventType eventType, Unit* /*sender*/, Unit* /*invoker*/, uint32 /*miscValue*/) override
334  {
335  if (eventType == AI_EVENT_CUSTOM_A)
336  {
337  if (m_phase == PHASE_EXPLODED)
338  return;
339 
340  // reset phase if not already exploded
341  m_phase = PHASE_NORMAL;
342  m_hitCount = 0;
343  }
344  else if (eventType == AI_EVENT_CUSTOM_B)
345  {
346  if (m_phase >= PHASE_NORMAL && m_phase < PHASE_FROZEN)
347  IncreaseHitCount();
348  }
349  else if (eventType == AI_EVENT_CUSTOM_C && m_phase == PHASE_FROZEN)
350  IncreaseHitCount();
351  }

374  break;
375  }
376  default:
377  break;

381  void HandleExplode()
382  {
383  // At this point, Viscidus should play an emote like dying or exploding but this does not appear clearly in sniff (maybe a spell visual or stand state change?)
384 
385  // Summon globs
386  DoCastSpellIfCan(m_creature, SPELL_SUMMON_GLOBS, CAST_TRIGGERED);
387  m_aliveGlobs = 0;
388 
389  // Make invisible and stun self
390  DoCastSpellIfCan(nullptr, SPELL_INVIS_SELF, CAST_TRIGGERED);
391  DoCastSpellIfCan(nullptr, SPELL_INVIS_STALKER, CAST_TRIGGERED);
392  DoCastSpellIfCan(nullptr, SPELL_STUN_SELF, CAST_TRIGGERED);
393 
394  // Reset hit count for normal and freezing phase
395  m_hitCount = 0;
396  m_creature->RemoveAurasDueToSpell(SPELL_MEMBRANE_VISCIDUS);
397  m_creature->RemoveAurasDueToSpell(SPELL_VISCIDUS_WEAKNESS);
398  m_lGlobesGuidList.clear();
399 
400  DoCastSpellIfCan(m_creature, SPELL_SUMMON_GLOBS, TRIGGERED_IGNORE_GCD);
401  }
