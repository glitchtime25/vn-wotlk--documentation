35  bool IntersectTriangle(MeshTriangle const& tri, std::vector<Vector3>::const_iterator points, G3D::Ray const& ray, float& distance)
36  {
37 #define EPS 1e-5f
38 
39  // See RTR2 ch. 13.7 for the algorithm.
40 
41  Vector3 const idx0 = points[tri.idx0];
42  Vector3 const e1 = points[tri.idx1] - idx0;
43  Vector3 const e2 = points[tri.idx2] - idx0;
44  Vector3 const p(ray.direction().cross(e2));
45  float a = e1.dot(p);
46 
47  if (fabs(a) < EPS)
48  {
49  // Determinant is ill-conditioned; abort early
50  return false;
51  }
52 
53  float const f = 1.0f / a;
54  Vector3 const s(ray.origin() - idx0);
55  float const u = f * s.dot(p);
56 
57  if ((u < 0.0f) || (u > 1.0f))
58  {
59  // We hit the plane of the m_geometry, but outside the m_geometry
60  return false;
61  }
62 
63  Vector3 const q(s.cross(e1));
64  float const v = f * ray.direction().dot(q);
65 
66  if ((v < 0.0f) || ((u + v) > 1.0f))
67  {
68  // We hit the plane of the triangle, but outside the triangle
69  return false;
70  }
71 
72  float const t = f * e2.dot(q);
73 
74  if ((t > 0.0f) && (t < distance))
75  {
76  // This is a new hit, closer than the previous one
77  distance = t;
78 
79  /* baryCoord[0] = 1.0 - u - v;
80  baryCoord[1] = u;
81  baryCoord[2] = v; */
82 
83  return true;
84  }
85  // This hit is after the previous hit, so ignore it
86  return false;
87  }

159  bool WmoLiquid::GetLiquidHeight(Vector3 const& pos, float& liqHeight) const
160  {
161  float tx_f = (pos.x - iCorner.x) / LIQUID_TILE_SIZE;
162  if (tx_f < 0.0f)
163  return false;
164 
165  float ty_f = (pos.y - iCorner.y) / LIQUID_TILE_SIZE;
166  if (ty_f < 0.0f)
167  return false;
168 
169  uint32 tx = uint32(tx_f);
170  if (tx >= iTilesX)
171  return false;
172 
173  uint32 ty = uint32(ty_f);
174  if (ty >= iTilesY)
175  return false;
176 
177  // check if tile shall be used for liquid level
178  // checking for 0x08 *might* be enough, but disabled tiles always are 0x?F:
179  if ((iFlags[tx + ty * iTilesX] & 0x0F) == 0x0F)
180  return false;
181 
182  // (dx, dy) coordinates inside tile, in [0,1]^2
183  float dx = tx_f - float(tx);
184  float dy = ty_f - float(ty);
185 
186  /* Tesselate tile to two triangles (not sure if client does it exactly like this)
187 
188  ^ dy
189  |
190  1 x---------x (1,1)
191  | (b) / |
192  | / |
193  | / |
194  | / (a) |
195  x---------x---> dx
196  0 1
197  */
198  uint32 const rowOffset = iTilesX + 1;
199  float sx;
200  float sy;
201 
202  if (dx > dy) // case (a)
203  {
204  uint32 const tyPcmp = ty * rowOffset;
205  sx = iHeight[tx + 1 + tyPcmp] - iHeight[tx + tyPcmp];
206  sy = iHeight[tx + 1 + (ty + 1) * rowOffset] - iHeight[tx + 1 + tyPcmp];
207  }
208  else // case (b)
209  {
210  uint32 const tyPcmp = (ty + 1) * rowOffset;
211  sx = iHeight[tx + 1 + tyPcmp] - iHeight[tx + tyPcmp];
212  sy = iHeight[tx + tyPcmp] - iHeight[tx + ty * rowOffset];
213  }
214 
215  liqHeight = iHeight[tx + ty * rowOffset] + dx * sx + dy * sy;
216 
217  return true;
218  }
