31 void WorldSession::HandleAutostoreLootItemOpcode(WorldPacket& recv_data)
32 {
33  uint8 itemSlot;
34  recv_data >> itemSlot;
35 
36  DEBUG_LOG("WORLD: CMSG_AUTOSTORE_LOOT_ITEM > requesting item in slot %u", uint32(itemSlot));
37 
38  Loot* loot = sLootMgr.GetLoot(_player);
39 
40  if (!loot)
41  {
42  sLog.outError("HandleAutostoreLootItemOpcode> Cannot retrieve loot for player %s", _player->GetGuidStr().c_str());
43  return;
44  }
45 
46  ObjectGuid const& lguid = loot->GetLootGuid();
47 
48  LootItem* lootItem = loot->GetLootItemInSlot(itemSlot);
49 
50  if (!lootItem)
51  {
52  _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
53  loot->SendReleaseFor(_player);
54  return;
55  }
56 
57  // item may be blocked by roll system or already looted or another cheating possibility
58  if (lootItem->isBlocked || lootItem->GetSlotTypeForSharedLoot(_player, loot) == MAX_LOOT_SLOT_TYPE)
59  {
60  sLog.outDebug("HandleAutostoreLootItemOpcode> %s have no right to loot itemId(%u)", _player->GetGuidStr().c_str(), lootItem->itemId);
61  loot->SendReleaseFor(_player);
62  return;
63  }
64 
65  InventoryResult result = loot->SendItem(_player, lootItem);
66 
67  if (result == EQUIP_ERR_OK && lguid.IsItem())
68  {
69  if (Item* item = _player->GetItemByGuid(lguid))
70  item->SetLootState(ITEM_LOOT_CHANGED);
71  }
72 }

89 void WorldSession::HandleLootOpcode(WorldPacket& recv_data)
90 {
91  DEBUG_LOG("WORLD: CMSG_LOOT");
92 
93  ObjectGuid lguid;
94  recv_data >> lguid;
95 
96  // Check possible cheat
97  if (!_player->IsAlive())
98  return;
99 
100  if (!_player->IsInWorld())
101  return;
102 
103  if (!_player->IsStandState())
104  {
105  _player->SendLootError(lguid, LOOT_ERROR_NOTSTANDING);
106  return;
107  }
108 
109  if (_player->IsStunned())
110  {
111  _player->SendLootError(lguid, LOOT_ERROR_STUNNED);
112  return;
113  }
114 
115  if (!lguid.IsItem())
116  {
117  if (lguid.IsUnit())
118  {
119  if (Unit* unit = _player->GetMap()->GetUnit(lguid))
120  {
121  float range = std::max(5.f, unit->GetCombatReach() + (4.f / 3.f) + _player->GetCombatReach());
122  if (range * range < _player->GetDistance(unit, true, DIST_CALC_NONE))
123  {
124  _player->SendLootError(lguid, LOOT_ERROR_TOO_FAR);
125  return;
126  }
127  }
128  }
129  else if (lguid.IsGameObject())
130  {
131  if (GameObject* go = _player->GetMap()->GetGameObject(lguid))
132  {
133  if (!go->IsAtInteractDistance(_player))
134  {
135  _player->SendLootError(lguid, LOOT_ERROR_TOO_FAR);
136  return;
137  }
138  }
139  }
140  }
141 
142  if (Loot* loot = sLootMgr.GetLoot(_player, lguid))
143  {
144  // remove stealth aura
145  _player->DoLoot();
146 
147  loot->ShowContentTo(_player);
148  }
149 
150  // interrupt cast
151  if (GetPlayer()->IsNonMeleeSpellCasted(false))
152  GetPlayer()->InterruptNonMeleeSpells(false);
153 }

166 void WorldSession::HandleLootMasterGiveOpcode(WorldPacket& recv_data)
167 {
168  uint8 itemSlot; // slot sent in LOOT_RESPONSE
169  ObjectGuid lootguid; // the guid of the loot object owner
170  ObjectGuid targetGuid; // the item receiver guid
171 
172  recv_data >> lootguid >> itemSlot >> targetGuid;
173 
174  Player* target = ObjectAccessor::FindPlayer(targetGuid);
175  if (!target)
176  {
177  _player->SendLootError(lootguid, LOOT_ERROR_PLAYER_NOT_FOUND);
178  sLog.outError("WorldSession::HandleLootMasterGiveOpcode> Cannot retrieve target %s", targetGuid.GetString().c_str());
179  return;
180  }
181 
182  DEBUG_LOG("WorldSession::HandleLootMasterGiveOpcode> Giver = %s, Target = %s.", _player->GetGuidStr().c_str(), targetGuid.GetString().c_str());
183 
184  Loot* pLoot = sLootMgr.GetLoot(_player, lootguid);
185 
186  if (!pLoot)
187  {
188  sLog.outError("WorldSession::HandleLootMasterGiveOpcode> Cannot retrieve loot for player %s", _player->GetGuidStr().c_str());
189  return;
190  }
191 
192  if (_player->GetObjectGuid() != pLoot->GetMasterLootGuid())
193  {
194  _player->SendLootError(lootguid, LOOT_ERROR_DIDNT_KILL);
195  sLog.outError("WorldSession::HandleLootMasterGiveOpcode> player %s is not the loot master!", _player->GetGuidStr().c_str());
196  return;
197  }
198 
199  if (!_player->IsInGroup(target) || !_player->IsInMap(target))
200  {
201  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_OTHER);
202  sLog.outError("WorldSession::HandleLootMasterGiveOpcode> Player %s tried to give an item to ineligible player %s !", _player->GetGuidStr().c_str(), target->GetGuidStr().c_str());
203  return;
204  }
205 
206  LootItem* lootItem = pLoot->GetLootItemInSlot(itemSlot);
207 
208  if (!lootItem)
209  {
210  _player->SendEquipError(EQUIP_ERR_ITEM_NOT_FOUND, nullptr, nullptr);
211  return;
212  }
213 
214  // item may be already looted or another cheating possibility
215  if (lootItem->GetSlotTypeForSharedLoot(_player, pLoot) == MAX_LOOT_SLOT_TYPE)
216  {
217  sLog.outError("HandleAutostoreLootItemOpcode> %s have no right to loot itemId(%u)", _player->GetGuidStr().c_str(), lootItem->itemId);
218  return;
219  }
220 
221  if (!lootItem->IsAllowed(target, pLoot))
222  {
223  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_OTHER);
224  return;
225  }
226 
227  InventoryResult result = pLoot->SendItem(target, lootItem);
228  if (result != EQUIP_ERR_OK)
229  {
230  if (result == EQUIP_ERR_CANT_CARRY_MORE_OF_THIS)
231  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_UNIQUE_ITEM);
232  else if (result == EQUIP_ERR_INVENTORY_FULL)
233  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_INV_FULL);
234  else
235  _player->SendLootError(lootguid, LOOT_ERROR_MASTER_OTHER);
236  }
237 }
