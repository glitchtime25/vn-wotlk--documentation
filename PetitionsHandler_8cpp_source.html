49 void WorldSession::HandlePetitionBuyOpcode(WorldPacket& recv_data)
50 {
51  DEBUG_LOG("Received opcode CMSG_PETITION_BUY");
52  recv_data.hexlike();
53 
54  ObjectGuid guidNPC;
55  uint32 clientIndex; // 1 for guild and arenaslot+1 for arenas in client
56  std::string name;
57 
58  recv_data >> guidNPC; // NPC GUID
59  recv_data.read_skip<uint32>(); // 0
60  recv_data.read_skip<uint64>(); // 0
61  recv_data >> name; // name
62  recv_data.read_skip<std::string>(); // some string
63  recv_data.read_skip<uint32>(); // 0
64  recv_data.read_skip<uint32>(); // 0
65  recv_data.read_skip<uint32>(); // 0
66  recv_data.read_skip<uint32>(); // 0
67  recv_data.read_skip<uint32>(); // 0
68  recv_data.read_skip<uint32>(); // 0
69  recv_data.read_skip<uint32>(); // 0
70  recv_data.read_skip<uint16>(); // 0
71  recv_data.read_skip<uint32>(); // 0
72  recv_data.read_skip<uint32>(); // 0
73  recv_data.read_skip<uint32>(); // 0
74 
75  for (int i = 0; i < 10; ++i)
76  recv_data.read_skip<std::string>();
77 
78  recv_data >> clientIndex; // index
79  recv_data.read_skip<uint32>(); // 0
80 
81  DEBUG_LOG("Petitioner %s tried sell petition: name %s", guidNPC.GetString().c_str(), name.c_str());
82 
83  // prevent cheating
84  Creature* pCreature = GetPlayer()->GetNPCIfCanInteractWith(guidNPC, UNIT_NPC_FLAG_PETITIONER);
85  if (!pCreature)
86  {
87  DEBUG_LOG("WORLD: HandlePetitionBuyOpcode - %s not found or you can't interact with him.", guidNPC.GetString().c_str());
88  return;
89  }
90 
91  uint32 charterid;
92  uint32 cost;
93  uint32 type;
94  if (pCreature->isTabardDesigner())
95  {
96  // if tabard designer, then trying to buy a guild charter.
97  // do not let if already in guild.
98  if (_player->GetGuildId())
99  return;
100 
101  charterid = GUILD_CHARTER;
102  cost = GUILD_CHARTER_COST;
103  type = 9;
104  }
105  else
106  {
107  // TODO: find correct opcode
108  if (_player->GetLevel() < sWorld.getConfig(CONFIG_UINT32_MAX_PLAYER_LEVEL))
109  {
110  SendNotification(LANG_ARENA_ONE_TOOLOW, sWorld.getConfig(CONFIG_UINT32_MAX_PLAYER_LEVEL));
111  return;
112  }
113 
114  switch (clientIndex) // arenaSlot+1 as received from client (1 from 3 case)
115  {
116  case 1:
117  charterid = ARENA_TEAM_CHARTER_2v2;
118  cost = ARENA_TEAM_CHARTER_2v2_COST;
119  type = 2; // 2v2
120  break;
121  case 2:
122  charterid = ARENA_TEAM_CHARTER_3v3;
123  cost = ARENA_TEAM_CHARTER_3v3_COST;
124  type = 3; // 3v3
125  break;
126  case 3:
127  charterid = ARENA_TEAM_CHARTER_5v5;
128  cost = ARENA_TEAM_CHARTER_5v5_COST;
129  type = 5; // 5v5
130  break;
131  default:
132  DEBUG_LOG("unknown selection at buy arena petition: %u", clientIndex);
133  return;
134  }
135 
136  if (_player->GetArenaTeamId(clientIndex - 1)) // arenaSlot+1 as received from client
137  {
138  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ALREADY_IN_ARENA_TEAM);
139  return;
140  }
141  }
142 
143  // Check guild petition name (use whisper type - 6)
144  if (sAnticheatLib->ValidateGuildName(name))
145  {
146  sLog.outBasic("Attempt to create guild petition with spam name \"%s\"", name.c_str());
147  SendGuildCommandResult(GUILD_CREATE_S, name, ERR_GUILD_NAME_INVALID);
148  return;
149  }
150 
151  if (type == 9)
152  {
153  if (sGuildMgr.GetGuildByName(name))
154  {
155  SendGuildCommandResult(GUILD_CREATE_S, name, ERR_GUILD_NAME_EXISTS_S);
156  return;
157  }
158  if (sObjectMgr.IsReservedName(name) || !ObjectMgr::IsValidCharterName(name))
159  {
160  SendGuildCommandResult(GUILD_CREATE_S, name, ERR_GUILD_NAME_INVALID);
161  return;
162  }
163  }
164  else
165  {
166  if (sObjectMgr.GetArenaTeamByName(name))
167  {
168  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ARENA_TEAM_NAME_EXISTS_S);
169  return;
170  }
171  if (sObjectMgr.IsReservedName(name) || !ObjectMgr::IsValidCharterName(name))
172  {
173  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, name, "", ERR_ARENA_TEAM_NAME_INVALID);
174  return;
175  }
176  }
177 
178  ItemPrototype const* pProto = ObjectMgr::GetItemPrototype(charterid);
179  if (!pProto)
180  {
181  _player->SendBuyError(BUY_ERR_CANT_FIND_ITEM, nullptr, charterid, 0);
182  return;
183  }
184 
185  if (_player->GetMoney() < cost)
186  {
187  // player hasn't got enough money
188  _player->SendBuyError(BUY_ERR_NOT_ENOUGHT_MONEY, pCreature, charterid, 0);
189  return;
190  }
191 
192  ItemPosCountVec dest;
193  InventoryResult msg = _player->CanStoreNewItem(NULL_BAG, NULL_SLOT, dest, charterid, pProto->BuyCount);
194  if (msg != EQUIP_ERR_OK)
195  {
196  _player->SendEquipError(msg, nullptr, nullptr, charterid);
197  return;
198  }
199 
200  _player->ModifyMoney(-(int32)cost);
201  Item* charter = _player->StoreNewItem(dest, charterid, true);
202  if (!charter)
203  return;
204 
205  charter->SetUInt32Value(ITEM_FIELD_ENCHANTMENT_1_1, charter->GetGUIDLow());
206  // ITEM_FIELD_ENCHANTMENT_1_1 is guild/arenateam id
207  // ITEM_FIELD_ENCHANTMENT_1_1+1 is current signatures count (showed on item)
208  charter->SetState(ITEM_CHANGED, _player);
209  _player->SendNewItem(charter, 1, true, false);
210 
211  // a petition is invalid, if both the owner and the type matches
212  // we checked above, if this player is in an arenateam, so this must be data corruption
213  QueryResult* result = CharacterDatabase.PQuery("SELECT petitionguid FROM petition WHERE ownerguid = '%u' AND type = '%u'", _player->GetGUIDLow(), type);
214 
215  std::ostringstream ssInvalidPetitionGUIDs;
216 
217  if (result)
218  {
219  do
220  {
221  Field* fields = result->Fetch();
222  ssInvalidPetitionGUIDs << "'" << fields[0].GetUInt32() << "' , ";
223  }
224  while (result->NextRow());
225 
226  delete result;
227  }
228 
229  // delete petitions with the same guid as this one
230  ssInvalidPetitionGUIDs << "'" << charter->GetGUIDLow() << "'";
231 
232  DEBUG_LOG("Invalid petition GUIDs: %s", ssInvalidPetitionGUIDs.str().c_str());
233  CharacterDatabase.escape_string(name);
234  CharacterDatabase.BeginTransaction();
235  CharacterDatabase.PExecute("DELETE FROM petition WHERE petitionguid IN ( %s )", ssInvalidPetitionGUIDs.str().c_str());
236  CharacterDatabase.PExecute("DELETE FROM petition_sign WHERE petitionguid IN ( %s )", ssInvalidPetitionGUIDs.str().c_str());
237  CharacterDatabase.PExecute("INSERT INTO petition (ownerguid, petitionguid, name, type) VALUES ('%u', '%u', '%s', '%u')",
238  _player->GetGUIDLow(), charter->GetGUIDLow(), name.c_str(), type);
239  CharacterDatabase.CommitTransaction();
240 }

242 void WorldSession::HandlePetitionShowSignOpcode(WorldPacket& recv_data)
243 {
244  // ok
245  DEBUG_LOG("Received opcode CMSG_PETITION_SHOW_SIGNATURES");
246  // recv_data.hexlike();
247 
248  uint8 signs = 0;
249  ObjectGuid petitionguid;
250  recv_data >> petitionguid; // petition guid
251 
252  // solve (possible) some strange compile problems with explicit use GUID_LOPART(petitionguid) at some GCC versions (wrong code optimization in compiler?)
253  uint32 petitionguid_low = petitionguid.GetCounter();
254 
255  QueryResult* result = CharacterDatabase.PQuery("SELECT type FROM petition WHERE petitionguid = '%u'", petitionguid_low);
256  if (!result)
257  {
258  sLog.outError("any petition on server...");
259  return;
260  }
261  Field* fields = result->Fetch();
262  uint32 type = fields[0].GetUInt32();
263  delete result;
264 
265  // if guild petition and has guild => error, return;
266  if (type == 9 && _player->GetGuildId())
267  return;
268 
269  result = CharacterDatabase.PQuery("SELECT playerguid FROM petition_sign WHERE petitionguid = '%u'", petitionguid_low);
270 
271  // result==nullptr also correct in case no sign yet
272  if (result)
273  signs = (uint8)result->GetRowCount();
274 
275  DEBUG_LOG("CMSG_PETITION_SHOW_SIGNATURES petition: %s", petitionguid.GetString().c_str());
276 
277  WorldPacket data(SMSG_PETITION_SHOW_SIGNATURES, (8 + 8 + 4 + 1 + signs * 12));
278  data << ObjectGuid(petitionguid); // petition guid
279  data << _player->GetObjectGuid(); // owner guid
280  data << uint32(petitionguid_low); // guild guid (in mangos always same as GUID_LOPART(petitionguid)
281  data << uint8(signs); // sign's count
282 
283  for (uint8 i = 1; i <= signs; ++i)
284  {
285  Field* fields2 = result->Fetch();
286  ObjectGuid signerGuid = ObjectGuid(HIGHGUID_PLAYER, fields2[0].GetUInt32());
287 
288  data << ObjectGuid(signerGuid); // Player GUID
289  data << uint32(0); // there 0 ...
290 
291  result->NextRow();
292  }
293  delete result;
294  SendPacket(data);
295 }

311 void WorldSession::SendPetitionQueryOpcode(ObjectGuid petitionguid) const
312 {
313  uint32 petitionLowGuid = petitionguid.GetCounter();
314 
315  QueryResult* result = CharacterDatabase.PQuery(
316  "SELECT ownerguid, name, "
317  " (SELECT COUNT(playerguid) FROM petition_sign WHERE petition_sign.petitionguid = '%u') AS signs, "
318  " type "
319  "FROM petition WHERE petitionguid = '%u'", petitionLowGuid, petitionLowGuid);
320 
321  if (!result)
322  {
323  DEBUG_LOG("CMSG_PETITION_QUERY failed for petition (GUID: %u)", petitionLowGuid);
324  return;
325  }
326 
327  Field* fields = result->Fetch();
328  ObjectGuid ownerGuid = ObjectGuid(HIGHGUID_PLAYER, fields[0].GetUInt32());
329  std::string name = fields[1].GetCppString();
330  //uint8 signs = fields[2].GetUInt8();
331  uint32 type = fields[3].GetUInt32();
332  delete result;
333 
334  WorldPacket data(SMSG_PETITION_QUERY_RESPONSE, (4 + 8 + name.size() + 1 + 1 + 4 * 12 + 2 + 10));
335  data << uint32(petitionLowGuid); // guild/team guid (in mangos always same as GUID_LOPART(petition guid)
336  data << ObjectGuid(ownerGuid); // charter owner guid
337  data << name; // name (guild/arena team)
338  data << uint8(0); // some string
339  if (type == 9)
340  {
341  data << uint32(9);
342  data << uint32(9);
343  data << uint32(0); // bypass client - side limitation, a different value is needed here for each petition
344  }
345  else
346  {
347  data << uint32(type - 1);
348  data << uint32(type - 1);
349  data << uint32(type); // bypass client - side limitation, a different value is needed here for each petition
350  }
351  data << uint32(0); // 5
352  data << uint32(0); // 6
353  data << uint32(0); // 7
354  data << uint32(0); // 8
355  data << uint16(0); // 9 2 bytes field
356  data << uint32(0); // 10
357  data << uint32(0); // 11
358  data << uint32(0); // 13 count of next strings?
359 
360  for (int i = 0; i < 10; ++i)
361  data << uint8(0); // some string
362 
363  data << uint32(0); // 14
364 
365  if (type == 9)
366  data << uint32(0); // 15 0 - guild, 1 - arena team
367  else
368  data << uint32(1);
369 
370  SendPacket(data);
371 }

444 {
445  DEBUG_LOG("Received opcode CMSG_PETITION_SIGN"); // ok
446  // recv_data.hexlike();
447 
448  ObjectGuid petitionGuid;
449  uint8 unk;
450  recv_data >> petitionGuid; // petition guid
451  recv_data >> unk;
452 
453  uint32 petitionLowGuid = petitionGuid.GetCounter();
454 
455  QueryResult* result = CharacterDatabase.PQuery(
456  "SELECT ownerguid, "
457  " (SELECT COUNT(playerguid) FROM petition_sign WHERE petition_sign.petitionguid = '%u') AS signs, "
458  " type "
459  "FROM petition WHERE petitionguid = '%u'", petitionLowGuid, petitionLowGuid);
460 
461  if (!result)
462  {
463  sLog.outError("any petition on server...");
464  return;
465  }
466 
467  Field* fields = result->Fetch();
468  uint32 ownerLowGuid = fields[0].GetUInt32();
469  ObjectGuid ownerGuid = ObjectGuid(HIGHGUID_PLAYER, ownerLowGuid);
470  uint8 signs = fields[1].GetUInt8();
471  uint32 type = fields[2].GetUInt32();
472 
473  delete result;
474 
475  if (ownerGuid == _player->GetObjectGuid())
476  return;
477 
478  // not let enemies sign guild charter
479  if (!sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_GUILD) &&
480  GetPlayer()->GetTeam() != sObjectMgr.GetPlayerTeamByGUID(ownerGuid))
481  {
482  if (type != 9)
483  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", "", ERR_ARENA_TEAM_NOT_ALLIED);
484  else
485  SendGuildCommandResult(GUILD_CREATE_S, "", ERR_GUILD_NOT_ALLIED);
486  return;
487  }
488 
489  if (type != 9)
490  {
491  if (_player->GetLevel() < sWorld.getConfig(CONFIG_UINT32_MAX_PLAYER_LEVEL))
492  {
493  SendArenaTeamCommandResult(ERR_ARENA_TEAM_CREATE_S, "", _player->GetName(), ERR_ARENA_TEAM_TARGET_TOO_LOW_S);
494  return;
495  }
496 
497  if (!IsArenaTypeValid(ArenaType(type)))
498  return;
499 
500  uint8 slot = ArenaTeam::GetSlotByType(ArenaType(type));
501  if (slot >= MAX_ARENA_SLOT)
502  return;
503 
504  if (_player->GetArenaTeamId(slot))
505  {
506  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", _player->GetName(), ERR_ALREADY_IN_ARENA_TEAM_S);
507  return;
508  }
509 
510  if (_player->GetArenaTeamIdInvited())
511  {
512  SendArenaTeamCommandResult(ERR_ARENA_TEAM_INVITE_SS, "", _player->GetName(), ERR_ALREADY_INVITED_TO_ARENA_TEAM_S);
513  return;
514  }
515  }
516  else
517  {
518  if (_player->GetGuildId())
519  {
520  SendGuildCommandResult(GUILD_INVITE_S, _player->GetName(), ERR_ALREADY_IN_GUILD_S);
521  return;
522  }
523  if (_player->GetGuildIdInvited())
524  {
525  SendGuildCommandResult(GUILD_INVITE_S, _player->GetName(), ERR_ALREADY_INVITED_TO_GUILD_S);
526  return;
527  }
528  }
529 
530  if (++signs > type) // client signs maximum
531  return;
532 
533  // client doesn't allow to sign petition two times by one character, but not check sign by another character from same account
534  // not allow sign another player from already sign player account
535  result = CharacterDatabase.PQuery("SELECT playerguid FROM petition_sign WHERE player_account = '%u' AND petitionguid = '%u'", GetAccountId(), petitionLowGuid);
536 
537  if (result)
538  {
539  delete result;
540  WorldPacket data(SMSG_PETITION_SIGN_RESULTS, (8 + 8 + 4));
541  data << ObjectGuid(petitionGuid);
542  data << ObjectGuid(_player->GetObjectGuid());
543  data << uint32(PETITION_SIGN_ALREADY_SIGNED);
544 
545  // close at signer side
546  SendPacket(data);
547 
548  // update for owner if online
549  if (Player* owner = sObjectMgr.GetPlayer(ownerGuid))
550  owner->GetSession()->SendPacket(data);
551  return;
552  }
553 
554  CharacterDatabase.PExecute("INSERT INTO petition_sign (ownerguid,petitionguid, playerguid, player_account) VALUES ('%u', '%u', '%u','%u')",
555  ownerLowGuid, petitionLowGuid, _player->GetGUIDLow(), GetAccountId());
556 
557  DEBUG_LOG("PETITION SIGN: %s by %s", petitionGuid.GetString().c_str(), _player->GetGuidStr().c_str());
558 
559  WorldPacket data(SMSG_PETITION_SIGN_RESULTS, (8 + 8 + 4));
560  data << ObjectGuid(petitionGuid);
561  data << ObjectGuid(_player->GetObjectGuid());
562  data << uint32(PETITION_SIGN_OK);
563 
564  // close at signer side
565  SendPacket(data);
566 
567  // update signs count on charter, required testing...
568  // Item *item = _player->GetItemByGuid(petitionguid));
569  // if(item)
570  // item->SetUInt32Value(ITEM_FIELD_ENCHANTMENT_1_1+1, signs);
571 
572  // update for owner if online
573  if (Player* owner = sObjectMgr.GetPlayer(ownerGuid))
574  owner->GetSession()->SendPacket(data);
575 }
