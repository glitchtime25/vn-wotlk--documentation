217  void ExecuteAction(uint32 action) override
218  {
219  switch (action)
220  {
221  case OURO_SPAWN:
222  {
223  DoCastSpellIfCan(nullptr, SPELL_BIRTH_SPAWN);
224  if (m_creature->GetUInt32Value(UNIT_CREATED_BY_SPELL))
225  {
226  if (Creature* trigger = GetClosestCreatureWithEntry(m_creature, NPC_OURO_TRIGGER, 100.f)) // if not sufficient in the future, do using SPELL_SET_OURO_HEALTH
227  m_creature->SetHealthPercent(static_cast<npc_ouro_triggerAI*>(trigger->AI())->m_ouroHealth);
228  }
229  else
230  {
231  if (m_instance)
232  m_instance->SetData(TYPE_OURO, IN_PROGRESS);
233  }
234  SetCombatScriptStatus(true);
235  if (GameObject* go = m_creature->GetMap()->GetGameObject(m_base))
236  go->SendGameObjectCustomAnim(m_base);
237  ResetTimer(OURO_SPAWN_DELAY_ATTACK, 4000);
238  DisableCombatAction(action);
239  break;
240  }
241  case OURO_ENRAGE:
242  {
243  if (m_creature->GetHealthPercent() < 20.0f)
244  {
245  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
246  {
247  m_firstMound = false; // in order to avoid spawning more ouros
248  ResetCombatAction(OURO_BOULDER, 500);
249  SetActionReadyStatus(action, false);
250  DisableCombatAction(OURO_SUBMERGE);
251  DisableCombatAction(OURO_NO_MELEE_BURROW);
252  return;
253  }
254  }
255  break;
256  }
257  case OURO_SUBMERGE:
258  {
259  Burrow();
260  break;
261  }
262  case OURO_SAND_BLAST:
263  {
264  if (DoCastSpellIfCan(nullptr, SPELL_SANDBLAST) == CAST_OK)
265  ResetCombatAction(action, 25000);
266  break;
267  }
268  case OURO_SWEEP:
269  {
270  if (DoCastSpellIfCan(nullptr, SPELL_SWEEP) == CAST_OK)
271  ResetCombatAction(action, 20000);
272  break;
273  }
274  case OURO_BOULDER:
275  {
276  uint32 timer = 500;
277  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
278  m_rangeCheckState = -1;
279  else
280  {
281  ++m_rangeCheckState;
282  if (m_rangeCheckState > 1)
283  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_BOULDER, SELECT_FLAG_PLAYER))
284  if (DoCastSpellIfCan(target, SPELL_BOULDER) == CAST_OK)
285  timer = 2500;
286  }
287  ResetCombatAction(action, timer);
288  break;
289  }
290  case OURO_NO_MELEE_BURROW:
291  {
292  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
293  m_burrowCounter = 0;
294  else
295  {
296  ++m_burrowCounter;
297  if (m_burrowCounter >= 10)
298  Burrow();
299  }
300  ResetCombatAction(action, 1000);
301  break;
302  }
303  }
304  }
