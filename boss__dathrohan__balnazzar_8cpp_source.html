94  return;
95 
96  // START NOT TRANSFORMED

112  }

126  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_HOLYSTRIKE);
127  m_uiHolyStrike_Timer = 15000;
128  }
129  else m_uiHolyStrike_Timer -= uiDiff;
130 
131  // BalnazzarTransform
132  if (m_creature->GetHealthPercent() < 40.0f)
133  {
134  // restore hp, mana and stun
135  if (DoCastSpellIfCan(m_creature, SPELL_BALNAZZARTRANSFORM) == CAST_OK)
136  {

138  DoScriptText(SAY_TRANSFORM, m_creature);
139  // Dathrohan wield a weapon but Balnazzar does not need one
140  SetEquipmentSlots(true);

142  m_bTransformed = true;

144  }
145  }
146  else

148  // MindBlast
149  if (m_uiMindBlast_Timer < uiDiff)
150  {
151  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MINDBLAST);

166  {

168  DoCastSpellIfCan(pTarget, SPELL_PSYCHICSCREAM);

173
