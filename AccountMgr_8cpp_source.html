39 AccountOpResult AccountMgr::CreateAccount(std::string username, std::string password) const
40 {
41  if (utf8length(username) > MAX_ACCOUNT_STR)
42  return AOR_NAME_TOO_LONG; // username's too long
43 
44  normalizeString(username);
45  normalizeString(password);
46 
47  if (GetId(username))
48  {
49  return AOR_NAME_ALREADY_EXIST; // username does already exist
50  }
51 
52  SRP6 srp;
53 
54  srp.CalculateVerifier(CalculateShaPassHash(username, password));

56  const char* v_hex = srp.GetVerifier().AsHexStr();

58  bool update_sv = LoginDatabase.PExecute(

60  username.c_str(), v_hex, s_hex);
61 
62  OPENSSL_free((void*)s_hex);
63  OPENSSL_free((void*)v_hex);
64 
65  if (!update_sv)
66  return AOR_DB_INTERNAL_ERROR; // unexpected error
67  LoginDatabase.Execute(
68  "INSERT INTO realmcharacters (realmid, acctid, numchars) SELECT realmlist.id, account.id, 0 FROM realmlist,account LEFT JOIN realmcharacters ON acctid=account.id WHERE acctid IS NULL");

106 AccountOpResult AccountMgr::DeleteAccount(uint32 accid) const
107 {
108  QueryResult* result = LoginDatabase.PQuery("SELECT 1 FROM account WHERE id='%u'", accid);
109  if (!result)
110  return AOR_NAME_NOT_EXIST; // account doesn't exist
111  delete result;
112 
113  // existing characters list
114  result = CharacterDatabase.PQuery("SELECT guid FROM characters WHERE account='%u'", accid);
115  if (result)
116  {
117  do
118  {
119  Field* fields = result->Fetch();
120  uint32 guidlo = fields[0].GetUInt32();
121  ObjectGuid guid = ObjectGuid(HIGHGUID_PLAYER, guidlo);
122 
123  // kick if player currently
124  ObjectAccessor::KickPlayer(guid);
125  Player::DeleteFromDB(guid, accid, false); // no need to update realm characters
126  }
127  while (result->NextRow());
128 
129  delete result;
130  }
131 
132  // table realm specific but common for all characters of account for realm
133  CharacterDatabase.PExecute("DELETE FROM character_tutorial WHERE account = '%u'", accid);
134 
135  LoginDatabase.BeginTransaction();
136 
137  bool res =
138  LoginDatabase.PExecute("DELETE FROM account WHERE id='%u'", accid) &&
139  LoginDatabase.PExecute("DELETE FROM realmcharacters WHERE acctid='%u'", accid);
140 
141  LoginDatabase.CommitTransaction();
142 
143  if (!res)
144  return AOR_DB_INTERNAL_ERROR; // unexpected error;
145 
146  return AOR_OK;
