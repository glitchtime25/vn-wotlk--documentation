262  if (opHandle.packetProcessing == PROCESS_IMMEDIATE)

264  try

266  (this->*opHandle.handler)(*new_packet);

352 bool WorldSession::Update(uint32 /*diff*/)
353 {
354  GetMessager().Execute(this);
355 
356  std::deque<std::unique_ptr<WorldPacket>> recvQueueCopy;
357  {
358  std::lock_guard<std::mutex> guard(m_recvQueueLock);
359  std::swap(recvQueueCopy, m_recvQueue);
360  }
361 
362  if (m_Socket && !m_Socket->IsClosed() && m_anticheat)
363  {
364  auto const now = WorldTimer::getMSTime();
365  m_anticheat->Update(WorldTimer::getMSTimeDiff(m_lastAnticheatUpdate, now));
366  m_lastAnticheatUpdate = now;
367  }
368 
371  while (m_Socket && !m_Socket->IsClosed() && !recvQueueCopy.empty())
372  {
373  // sLog.outError("MOEP: %s (0x%.4X)", packet->GetOpcodeName(), packet->GetOpcode());
374 
375  auto const packet = std::move(recvQueueCopy.front());
376  recvQueueCopy.pop_front();
377 
378  OpcodeHandler const& opHandle = opcodeTable[packet->GetOpcode()];
379  switch (opHandle.status)
380  {
381  case STATUS_LOGGEDIN:
382  if (!_player)
383  {
384  // skip STATUS_LOGGEDIN opcode unexpected errors if player logout sometime ago - this can be network lag delayed packets
385  if (!m_playerRecentlyLogout)
386  LogUnexpectedOpcode(*packet, "the player has not logged in yet");
387  }
388  else if (_player->IsInWorld())
389  ExecuteOpcode(opHandle, *packet);
390 
391  // lag can cause STATUS_LOGGEDIN opcodes to arrive after the player started a transfer
392 
393 #ifdef BUILD_PLAYERBOT
394  if (_player && _player->GetPlayerbotMgr())
395  _player->GetPlayerbotMgr()->HandleMasterIncomingPacket(*packet);
396 #endif
397  break;
398  case STATUS_LOGGEDIN_OR_RECENTLY_LOGGEDOUT:
399  if (!_player && !m_playerRecentlyLogout)
400  {
401  LogUnexpectedOpcode(*packet, "the player has not logged in yet and not recently logout");
402  }
403  else
404  // not expected _player or must checked in packet hanlder
405  ExecuteOpcode(opHandle, *packet);
406  break;
407  case STATUS_TRANSFER:
408  if (!_player)
409  LogUnexpectedOpcode(*packet, "the player has not logged in yet");
410  else if (_player->IsInWorld())
411  LogUnexpectedOpcode(*packet, "the player is still in world");
412  else
413  ExecuteOpcode(opHandle, *packet);
414  break;
415  case STATUS_AUTHED:
416  // prevent cheating with skip queue wait
417  if (m_inQueue && packet->GetOpcode() != CMSG_WARDEN_DATA)
418  {
419  LogUnexpectedOpcode(*packet, "the player not pass queue yet");
420  break;
421  }
422 
423  // single from authed time opcodes send in to after logout time
424  // and before other STATUS_LOGGEDIN_OR_RECENTLY_LOGGOUT opcodes.
425  if (packet->GetOpcode() != CMSG_SET_ACTIVE_VOICE_CHANNEL)
426  m_playerRecentlyLogout = false;
427 
428  ExecuteOpcode(opHandle, *packet);
429  break;
430  case STATUS_NEVER:
431  sLog.outError("SESSION: received not allowed opcode %s (0x%.4X)",
432  packet->GetOpcodeName(),
433  packet->GetOpcode());
434  break;
435  case STATUS_UNHANDLED:
436  DEBUG_LOG("SESSION: received not handled opcode %s (0x%.4X)",
437  packet->GetOpcodeName(),
438  packet->GetOpcode());
439  break;
440  default:
441  sLog.outError("SESSION: received wrong-status-req opcode %s (0x%.4X)",
442  packet->GetOpcodeName(),
443  packet->GetOpcode());
444  break;
445  }
446  }
447 
448 #ifdef BUILD_PLAYERBOT
449  // Process player bot packets
450  // The PlayerbotAI class adds to the packet queue to simulate a real player
451  // since Playerbots are known to the World obj only by its master's WorldSession object
452  // we need to process all master's bot's packets.
453  if (GetPlayer() && GetPlayer()->GetPlayerbotMgr())
454  {
455  for (PlayerBotMap::const_iterator itr = GetPlayer()->GetPlayerbotMgr()->GetPlayerBotsBegin();
456  itr != GetPlayer()->GetPlayerbotMgr()->GetPlayerBotsEnd(); ++itr)
457  {
458  Player* const botPlayer = itr->second;
459  WorldSession* const pBotWorldSession = botPlayer->GetSession();
460  while(!pBotWorldSession->m_recvQueue.empty())
461  {
462  auto const botpacket = std::move(pBotWorldSession->m_recvQueue.front());
463  pBotWorldSession->m_recvQueue.pop_front();
464 
465  OpcodeHandler const& opHandle = opcodeTable[botpacket->GetOpcode()];
466  pBotWorldSession->ExecuteOpcode(opHandle, *botpacket);
467  }
468  }
469  GetPlayer()->GetPlayerbotMgr()->RemoveBots();
470  }
471 #endif
472 
473  // check if we are safe to proceed with logout
474  // logout procedure should happen only in World::UpdateSessions() method!!!
475  switch (m_sessionState)
476  {
477  case WORLD_SESSION_STATE_CREATED:
478  {
479  if (m_requestSocket)
480  {
481  if (!IsOffline())
482  SetOffline();
483 
484  m_Socket = m_requestSocket;
485  m_requestSocket = nullptr;
486  sLog.outDetail("New Session key %s", m_Socket->GetSessionKey().AsHexStr());
487  SendAuthOk();
488  }
489  else
490  {
491  if (m_inQueue)
492  SendAuthQueued();
493  else
494  SendAuthOk();
495  }
496  SetInCharSelection();
497  return true;
498  }
499  case WORLD_SESSION_STATE_CHAR_SELECTION:
500  {
501  // waiting to go online
502  // TODO:: Maybe check if have to send queue update?
503  if (!m_Socket || (m_Socket && m_Socket->IsClosed()))
504  {
505  // directly remove this session
506  return false;
507  }
508 
509  if (ShouldLogOut(time(nullptr)) && !m_playerLoading) // check if delayed logout is fired
510  LogoutPlayer();
511 
512  if (m_kickTime && m_kickTime <= time(nullptr))
513  {
514  KickPlayer(true, true);
515  return false;
516  }
517 
518  return true;
519  }
520  case WORLD_SESSION_STATE_READY:
521  {
522  if (m_Socket && m_Socket->IsClosed())
523  {
524  if (!_player)
525  return false;
526 
527  // give the opportunity for this player to reconnect within 20 sec
528  SetOffline();
529  }
530  else if (ShouldLogOut(time(nullptr)) && !m_playerLoading) // check if delayed logout is fired
531  LogoutPlayer();
532 
533  return true;
534  }
535 
536  case WORLD_SESSION_STATE_OFFLINE:
537  {
538  if (ShouldDisconnect(time(nullptr))) // check if delayed logout is fired
539  {
540  LogoutPlayer();
541  if (!m_requestSocket && (!m_Socket || m_Socket->IsClosed()))
542  return false;
543  }
544 
545  return true;
546  }
547  default:
548  break;
549  }
550 
551  return true;
552 }
