108  void PhaseEquipStart()
109  {
110  ScriptedInstance* pInstance = (ScriptedInstance*)m_creature->GetInstanceData();
111 
112  if (!pInstance)
113  return;
114 
115  GameObject* pChest = pInstance->GetSingleGameObjectFromStorage(GO_SMITE_CHEST);
116 
117  if (!pChest)
118  return;
119 
120  m_uiPhase = PHASE_EQUIP_NULL;
121 
122  float fX, fY, fZ;
123  pChest->GetContactPoint(m_creature, fX, fY, fZ, CONTACT_DISTANCE);
124 
125  m_creature->GetMotionMaster()->Clear();
126  m_creature->SetFacingToObject(pChest);
127  m_creature->GetMotionMaster()->MovePoint(0, fX, fY, fZ, FORCED_MOVEMENT_RUN);
128  }
129 
130  void PhaseEquipProcess()
131  {
132  if (m_creature->GetHealthPercent() < 33.0f)
133  {
134  // It's Hammer, go Hammer!
135  SetEquipmentSlots(false, EQUIP_ID_HAMMER, EQUIP_UNEQUIP);
136  DoCastSpellIfCan(m_creature, SPELL_SMITE_HAMMER);
137  }
138  else
139  SetEquipmentSlots(false, EQUIP_ID_AXE, EQUIP_ID_AXE);
140 
141  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
142  m_uiPhase = PHASE_EQUIP_END;
143  m_uiEquipTimer = 1000;
144  }
145 
146  void PhaseEquipEnd()
147  {
148  // We don't have GetVictim, so select from threat list
149  Unit* pVictim = m_creature->SelectAttackingTarget(ATTACKING_TARGET_TOPAGGRO, 0);
150 
151  if (!pVictim)
152  {
153  EnterEvadeMode();
154  return;
155  }
156 
157  m_creature->SetSheath(SHEATH_STATE_MELEE);
158 
159  m_uiPhase = m_creature->GetHealthPercent() < 33.0f ? PHASE_3 : PHASE_2;
160 
161  if (m_uiPhase == PHASE_2)
162  DoCastSpellIfCan(m_creature, SPELL_THRASH, CAST_TRIGGERED);
163 
164  AttackStart(pVictim);
165  }

167  void UpdateAI(const uint32 uiDiff) override
168  {
169  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
170  {
171  if (m_uiEquipTimer)
172  {
173  // decrease the cooldown in between equipment change phases
174  if (m_uiEquipTimer > uiDiff)
175  {
176  m_uiEquipTimer -= uiDiff;
177  return;
178  }
179  m_uiEquipTimer = 0;
180  }
181 
182  switch (m_uiPhase)
183  {
184  case PHASE_EQUIP_START:
185  PhaseEquipStart();
186  break;
187  case PHASE_EQUIP_PROCESS:
188  PhaseEquipProcess();
189  break;
190  case PHASE_EQUIP_END:
191  PhaseEquipEnd();
192  break;
193  }
194 
195  return;
196  }
197 
198  // the normal combat phases
199  switch (m_uiPhase)
200  {
201  case PHASE_1:
202  {
203  if (m_creature->GetHealthPercent() < 66.0f)
204  {
205  if (DoCastSpellIfCan(m_creature, SPELL_SMITE_STOMP) == CAST_OK)
206  {
207  DoScriptText(m_creature->GetHealthPercent() < 33.0f ? SAY_PHASE_3 : SAY_PHASE_2, m_creature);
208  m_uiPhase = PHASE_EQUIP_START;
209  m_uiEquipTimer = 2500;
210 
211  // will clear GetVictim (m_attacking)
212  m_creature->AttackStop(true);
213  m_creature->RemoveAurasDueToSpell(SPELL_NIBLE_REFLEXES);
214  }
215  return;
216  }
217  break;
218  }
219  case PHASE_2:
220  {
221  if (m_creature->GetHealthPercent() < 33.0f)
222  {
223  if (DoCastSpellIfCan(m_creature, SPELL_SMITE_STOMP) == CAST_OK)
224  {
225  DoScriptText(SAY_PHASE_3, m_creature);
226  m_uiPhase = PHASE_EQUIP_START;
227  m_uiEquipTimer = 2500;
228 
229  // will clear GetVictim (m_attacking)
230  m_creature->AttackStop(true);
231  m_creature->RemoveAurasDueToSpell(SPELL_THRASH);
232  }
233  return;
234  }
235  break;
236  }
237  case PHASE_3:
238  {
239  if (m_uiSlamTimer < uiDiff)
240  {
241  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SMITE_SLAM) == CAST_OK)
242  m_uiSlamTimer = 11000;
243  }
244  else
245  m_uiSlamTimer -= uiDiff;
246 
247  break;
248  }
249  }
250 
251  DoMeleeAttackIfReady();
252  }
