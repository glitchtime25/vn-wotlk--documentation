39  while (((i = m_events.begin()) != m_events.end()) && i->first <= m_time)
40  {
41  // get and remove event from queue
42  BasicEvent* Event = i->second;
43  m_events.erase(i);
44

52  }

61 void EventProcessor::KillAllEvents(bool force)
62 {

65 
66  // first, abort all existing events
