121  void UpdateAI(const uint32 uiDiff) override
122  {
123  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
124  return;
125 
126  // War Stomp
127  if (m_uiWarStompTimer < uiDiff)
128  {
129  if (DoCastSpellIfCan(m_creature, SPELL_WAR_STOMP) == CAST_OK)
130  m_uiWarStompTimer = urand(17000, 38000);
131  }
132  else
133  m_uiWarStompTimer -= uiDiff;
134 
135  // Arcane Volley
136  if (m_uiArcaneVolleyTimer < uiDiff)
137  {
138  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ARCANE_VOLLEY : SPELL_ARCANE_VOLLEY_H) == CAST_OK)
139  m_uiArcaneVolleyTimer = urand(16000, 38000);
140  }
141  else
142  m_uiArcaneVolleyTimer -= uiDiff;
143 
144  // Summon Treants
145  if (m_uiSummonTreantsTimer < uiDiff)
146  {
147  SummonTreants();
148  m_uiSummonTreantsTimer = urand(37000, 55000);
149  }
150  else
151  m_uiSummonTreantsTimer -= uiDiff;
152 
153  DoMeleeAttackIfReady();
154  }
