46 void LoadRandomEnchantmentsTable()
47 {
48  RandomItemEnch.clear(); // for reload case
49 
50  uint32 count = 0;
51  QueryResult* result = WorldDatabase.Query("SELECT entry, ench, chance FROM item_enchantment_template");
52 
53  if (result)
54  {
55  BarGoLink bar(result->GetRowCount());
56 
57  do
58  {
59  Field* fields = result->Fetch();
60  bar.step();
61 
62  uint32 entry = fields[0].GetUInt32();
63  uint32 ench = fields[1].GetUInt32();
64  float chance = fields[2].GetFloat();
65 
66  if (chance > 0.000001f && chance <= 100.0f)
67  RandomItemEnch[entry].push_back(EnchStoreItem(ench, chance));
68 
69  ++count;
70  }
71  while (result->NextRow());
72 
73  delete result;
74 
75  sLog.outString(">> Loaded %u Item Enchantment definitions", count);
76  }
77  else
78  sLog.outErrorDb(">> Loaded 0 Item Enchantment definitions. DB table `item_enchantment_template` is empty.");
79 
80  sLog.outString();
81 }

83 uint32 GetItemEnchantMod(uint32 entry)
84 {
85  if (!entry) return 0;
86 
87  EnchantmentStore::const_iterator tab = RandomItemEnch.find(entry);
88 
89  if (tab == RandomItemEnch.end())
90  {
91  sLog.outErrorDb("Item RandomProperty / RandomSuffix id #%u used in `item_template` but it doesn't have records in `item_enchantment_template` table.", entry);
92  return 0;
93  }
94 
95  double dRoll = rand_chance();
96  float fCount = 0;
97 
98  const EnchStoreList& enchantList = tab->second;
99  for (auto ench_iter : enchantList)
100  {
101  fCount += ench_iter.chance;
102 
103  if (fCount > dRoll) return ench_iter.ench;
104  }
105 
106  // we could get here only if sum of all enchantment chances is lower than 100%
107  dRoll = (irand(0, (int)floor(fCount * 100) + 1)) / 100;
108  fCount = 0;
109 
110  for (auto ench_iter : enchantList)
111  {
112  fCount += ench_iter.chance;
113 
114  if (fCount > dRoll) return ench_iter.ench;
115  }
116 
117  return 0;
118 }

120 uint32 GenerateEnchSuffixFactor(uint32 item_id)
121 {
122  ItemPrototype const* itemProto = ObjectMgr::GetItemPrototype(item_id);
123 
124  if (!itemProto)
125  return 0;
126  if (!itemProto->RandomSuffix)
127  return 0;
128 
129  RandomPropertiesPointsEntry const* randomProperty = sRandomPropertiesPointsStore.LookupEntry(itemProto->ItemLevel);
130  if (!randomProperty)
131  return 0;
132 
133  uint32 suffixFactor;
134  switch (itemProto->InventoryType)
135  {
136  // Items of that type don`t have points
137  case INVTYPE_NON_EQUIP:
138  case INVTYPE_BAG:
139  case INVTYPE_TABARD:
140  case INVTYPE_AMMO:
141  case INVTYPE_QUIVER:
142  case INVTYPE_RELIC:
143  return 0;
144  // Select point coefficient
145  case INVTYPE_HEAD:
146  case INVTYPE_BODY:
147  case INVTYPE_CHEST:
148  case INVTYPE_LEGS:
149  case INVTYPE_2HWEAPON:
150  case INVTYPE_ROBE:
151  suffixFactor = 0;
152  break;
153  case INVTYPE_SHOULDERS:
154  case INVTYPE_WAIST:
155  case INVTYPE_FEET:
156  case INVTYPE_HANDS:
157  case INVTYPE_TRINKET:
158  suffixFactor = 1;
159  break;
160  case INVTYPE_NECK:
161  case INVTYPE_WRISTS:
162  case INVTYPE_FINGER:
163  case INVTYPE_SHIELD:
164  case INVTYPE_CLOAK:
165  case INVTYPE_HOLDABLE:
166  suffixFactor = 2;
167  break;
168  case INVTYPE_WEAPON:
169  case INVTYPE_WEAPONMAINHAND:
170  case INVTYPE_WEAPONOFFHAND:
171  suffixFactor = 3;
172  break;
173  case INVTYPE_RANGED:
174  case INVTYPE_THROWN:
175  case INVTYPE_RANGEDRIGHT:
176  suffixFactor = 4;
177  break;
178  default:
179  return 0;
180  }
181  // Select rare/epic modifier
182  switch (itemProto->Quality)
183  {
184  case ITEM_QUALITY_UNCOMMON:
185  return randomProperty->UncommonPropertiesPoints[suffixFactor];
186  case ITEM_QUALITY_RARE:
187  return randomProperty->RarePropertiesPoints[suffixFactor];
188  case ITEM_QUALITY_EPIC:
189  return randomProperty->EpicPropertiesPoints[suffixFactor];
190  case ITEM_QUALITY_LEGENDARY:
191  case ITEM_QUALITY_ARTIFACT:
192  return 0; // not have random properties
193  default:
194  break;
195  }
196  return 0;
197 }
