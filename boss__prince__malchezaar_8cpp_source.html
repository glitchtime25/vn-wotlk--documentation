185  void ExecuteAction(uint32 action) override
186  {
187  switch (action)
188  {
189  case MALCHEZAAR_PHASE_2:
190  {
191  // transition to phase 2
192  if (m_creature->GetHealthPercent() < 60.0f)
193  {
194  if (DoCastSpellIfCan(nullptr, SPELL_EQUIP_AXES, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
195  {
196  DoScriptText(SAY_AXE_TOSS1, m_creature);
197 
198  SetEquipmentSlots(false, EQUIP_ID_AXE, EQUIP_ID_AXE, EQUIP_NO_CHANGE);
199  m_creature->SetAttackTime(BASE_ATTACK, ATTACK_TIMER_AXES);
200  ResetCombatAction(MALCHEZAAR_THRASH, urand(6000, 8000));
201  DisableCombatAction(MALCHEZAAR_SHADOW_WORD_PAIN);
202  ResetCombatAction(MALCHEZAAR_SUNDER_ARMOR, urand(5000, 10000));
203  SetActionReadyStatus(action, false);
204  m_creature->SetCanParry(true);
205  }
206  }
207  return;
208  }
209  case MALCHEZAAR_PHASE_3:
210  {
211  // transition to phase 3
212  if (m_creature->GetHealthPercent() < 30.0f)
213  {
214  if (DoCastSpellIfCan(nullptr, SPELL_SUMMON_AXES) == CAST_OK)
215  {
216  DoCastSpellIfCan(nullptr, SPELL_ENFEEBLE_EFFECT, CAST_TRIGGERED);
217  DoScriptText(SAY_SPECIAL3, m_creature);
218 
219  SetEquipmentSlots(false, EQUIP_UNEQUIP, EQUIP_UNEQUIP, EQUIP_NO_CHANGE);
220  m_creature->SetAttackTime(BASE_ATTACK, ATTACK_TIMER_DEFAULT);
221 
222  DisableCombatAction(MALCHEZAAR_THRASH);
223  DisableCombatAction(MALCHEZAAR_ENFEEBLE);
224  DisableCombatAction(MALCHEZAAR_SUNDER_ARMOR);
225  ResetCombatAction(MALCHEZAAR_SHADOW_NOVA, 10000);
226  ResetCombatAction(MALCHEZAAR_AMPLIFY_DAMAGE, 5000);
227  ResetCombatAction(MALCHEZAAR_INFERNAL, 15000);
228  ResetCombatAction(MALCHEZAAR_SHADOW_WORD_PAIN, 5000);
229  m_creature->SetCanParry(false);
230  SetActionReadyStatus(action, false);
231  }
232  }
233  return;
234  }
235  case MALCHEZAAR_SUNDER_ARMOR:
236  {
237  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SUNDER_ARMOR) == CAST_OK)
238  ResetCombatAction(action, urand(10000, 18000));
239  break;
240  }
241  case MALCHEZAAR_AMPLIFY_DAMAGE:
242  {
243  if (DoCastSpellIfCan(nullptr, SPELL_AMPLIFY_DAMAGE) == CAST_OK)
244  ResetCombatAction(action, urand(20000, 30000));
245  break;
246  }
247  case MALCHEZAAR_INFERNAL:
248  {
249  if (Creature* relayClose = m_creature->GetMap()->GetCreature(m_uiRelayGuidClose))
250  {
251  if (Creature* relayFar = m_creature->GetMap()->GetCreature(m_uiRelayGuidFar))
252  {
253  relayFar->CastSpell(relayClose, SPELL_INFERNAL_RELAY, TRIGGERED_NONE);
254  DoScriptText(urand(0, 1) ? SAY_SUMMON1 : SAY_SUMMON2, m_creature);
255  ResetCombatAction(action, GetActionReadyStatus(MALCHEZAAR_PHASE_3) ? 45000 : 17000);
256  }
257  }
258  break;
259  }
260  case MALCHEZAAR_SHADOW_NOVA:
261  {
262  if (DoCastSpellIfCan(nullptr, SPELL_SHADOW_NOVA) == CAST_OK)
263  {
264  if (GetActionReadyStatus(MALCHEZAAR_PHASE_3))
265  ResetCombatAction(action, 30000);
266  else
267  DisableCombatAction(action);
268  }
269  break;
270  }
271  case MALCHEZAAR_SHADOW_WORD_PAIN:
272  {
273  if (DoCastSpellIfCan(GetActionReadyStatus(MALCHEZAAR_PHASE_2) ? m_creature->GetVictim() : nullptr, GetActionReadyStatus(MALCHEZAAR_PHASE_2) ? SPELL_SW_PAIN_PHASE1 : SPELL_SW_PAIN_PHASE3) == CAST_OK)
274  ResetCombatAction(action, 20000);
275  break;
276  }
277  case MALCHEZAAR_ENFEEBLE:
278  {
279  if (DoCastSpellIfCan(m_creature, SPELL_ENFEEBLE) == CAST_OK)
280  {
281  ResetCombatAction(action, 30000);
282  ResetCombatAction(MALCHEZAAR_SHADOW_NOVA, 5000);
283  }
284  break;
285  }
286  case MALCHEZAAR_THRASH:
287  {
288  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_THRASH) == CAST_OK)
289  ResetCombatAction(action, urand(6000, 8000));
290  break;
291  }
292  }
293  }
