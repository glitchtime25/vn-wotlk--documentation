52  * 0 1 2 3 4

54  * 0 1 2
55  * **************************************
56  * entry/guid: creature_template.entry/guid

58  * master_entry/master_guid creature_template.entry of the npc, that shall trigger the actions

60  * search_radius: radius, in which master and slave must be spawned so that they are linked together

64 void CreatureLinkingMgr::LoadFromDB()
65 {
66  // Clear maps
67  m_creatureLinkingMap.clear();
68  m_creatureLinkingGuidMap.clear();
69  m_eventTriggers.clear(); // master
70  m_eventGuidTriggers.clear();
71 
72  // Load `creature_linking_template`
73  sLog.outString("> Loading table `creature_linking_template`");
74  uint32 count = 0;
75  QueryResult* result = WorldDatabase.Query("SELECT entry, map, master_entry, flag, search_range FROM creature_linking_template");
76  if (!result)
77  {
78  BarGoLink bar(1);
79  bar.step();
80  sLog.outString(">> Table creature_linking_template is empty.");
81  sLog.outString();
82  }
83  else
84  {
85  BarGoLink bar((int)result->GetRowCount());
86  do
87  {
88  bar.step();
89 
90  Field* fields = result->Fetch();
91  CreatureLinkingInfo tmp;
92 
93  uint32 entry = fields[0].GetUInt32();
94  tmp.mapId = fields[1].GetUInt32();
95  tmp.masterId = fields[2].GetUInt32();
96  tmp.linkingFlag = fields[3].GetUInt16();
97  tmp.searchRange = fields[4].GetUInt16();
98  tmp.masterDBGuid = 0; // Will be initialized for unique mobs in IsLinkingEntryValid (only for spawning dependend)
99 
100  if (!IsLinkingEntryValid(entry, &tmp, true))
101  continue;
102 
103  ++count;
104 
105  // Add it to the map
106  m_creatureLinkingMap.insert(CreatureLinkingMap::value_type(entry, tmp));
107 
108  // Store master_entry
109  m_eventTriggers.insert(tmp.masterId);
110  }
111  while (result->NextRow());
112 
113  sLog.outString(">> Loaded creature linking for %u creature-entries", count);
114  sLog.outString();
115 
116  delete result;
117  }
118 
119  // Load `creature_linking`
120  sLog.outString("> Loading table `creature_linking`");
121  count = 0;
122  result = WorldDatabase.Query("SELECT guid, master_guid, flag FROM creature_linking");
123  if (!result)
124  {
125  BarGoLink bar(1);
126  bar.step();
127 
128  sLog.outString(">> Table creature_linking is empty.");
129  sLog.outString();
130 
131  return;
132  }
133 
134  BarGoLink guidBar((int)result->GetRowCount());
135  do
136  {
137  guidBar.step();
138 
139  Field* fields = result->Fetch();
140  CreatureLinkingInfo tmp;
141 
142  uint32 guid = fields[0].GetUInt32();
143  tmp.mapId = INVALID_MAP_ID; // some invalid value, this marks the guid-linking
144  tmp.masterId = fields[1].GetUInt32();
145  tmp.linkingFlag = fields[2].GetUInt16();
146  tmp.masterDBGuid = tmp.masterId;
147  tmp.searchRange = 0;
148 
149  if (!IsLinkingEntryValid(guid, &tmp, false))
150  continue;
151 
152  ++count;
153 
154  // Add it to the map
155  m_creatureLinkingGuidMap.insert(CreatureLinkingMap::value_type(guid, tmp));
156 
157  // Store master_guid
158  m_eventGuidTriggers.insert(tmp.masterId);
159  }
160  while (result->NextRow());
161 
162  sLog.outString(">> Loaded creature linking for %u creature-Guids", count);
163  sLog.outString();
164 
165  delete result;
166 }

176 bool CreatureLinkingMgr::IsLinkingEntryValid(uint32 slaveEntry, CreatureLinkingInfo* pTmp, bool byEntry)
177 {
178  // Basic checks first
179  if (byEntry) // Entry given
180  {
181  CreatureInfo const* pInfo = ObjectMgr::GetCreatureTemplate(slaveEntry);
182  CreatureInfo const* pMasterInfo = ObjectMgr::GetCreatureTemplate(pTmp->masterId);
183 
184  if (!pInfo)
185  {
186  sLog.outErrorDb("`creature_linking_template` has a non existing slave_entry (slave: %u, master %u), skipped.", slaveEntry, pTmp->masterId);
187  return false;
188  }
189  if (!pMasterInfo)
190  {
191  sLog.outErrorDb("`creature_linking_template` has a non existing master_entry (slave: %u, master %u), skipped", slaveEntry, pTmp->masterId);
192  return false;
193  }
194  if (pTmp->mapId && !sMapStore.LookupEntry(pTmp->mapId))
195  {
196  sLog.outErrorDb("`creature_linking_template` has a non existing map %u (slave %u, master %u), skipped", pTmp->mapId, slaveEntry, pTmp->masterId);
197  return false;
198  }
199  }
200  else // guid given
201  {
202  CreatureData const* slaveData = sObjectMgr.GetCreatureData(slaveEntry);
203  CreatureData const* masterData = sObjectMgr.GetCreatureData(pTmp->masterId);
204 
205  if (!slaveData)
206  {
207  sLog.outErrorDb("`creature_linking` has a non existing slave (guid: %u, master_guid %u), skipped", slaveEntry, pTmp->masterId);
208  return false;
209  }
210  if (!masterData)
211  {
212  sLog.outErrorDb("`creature_linking` has a non existing master (guid: %u,, master_guid: %u), skipped", slaveEntry, pTmp->masterId);
213  return false;
214  }
215  if (slaveData->mapid != masterData->mapid)
216  {
217  sLog.outErrorDb("`creature_linking` has a slave and master on different maps (guid: %u, master_guid: %u), skipped", slaveEntry, pTmp->masterId);
218  return false;
219  }
220  }
221 
222  if (pTmp->linkingFlag & ~(LINKING_FLAG_INVALID - 1) || pTmp->linkingFlag == 0)
223  {
224  sLog.outErrorDb("`creature_linking%s` has invalid flag, (entry: %u, map: %u, flags: %u), skipped", byEntry ? "_template" : "", slaveEntry, pTmp->mapId, pTmp->linkingFlag);
225  return false;
226  }
227 
228  // Additional checks, depending on flags
229  if (pTmp->linkingFlag & FLAG_DESPAWN_ON_RESPAWN && slaveEntry == pTmp->masterId)
230  {
231  sLog.outErrorDb("`creature_linking%s` has pointless FLAG_DESPAWN_ON_RESPAWN for self, (entry: %u, map: %u), skipped", byEntry ? "_template" : "", slaveEntry, pTmp->mapId);
232  return false;
233  }
234 
235  if (byEntry)
236  {
237  // Check for uniqueness of mob whom is followed, on whom spawning is dependend
238  if (pTmp->searchRange == 0 && pTmp->linkingFlag & (FLAG_FOLLOW | FLAG_CANT_SPAWN_IF_BOSS_DEAD | FLAG_CANT_SPAWN_IF_BOSS_ALIVE))
239  {
240  QueryResult* result = WorldDatabase.PQuery("SELECT guid FROM creature WHERE id=%u AND map=%u LIMIT 2", pTmp->masterId, pTmp->mapId);
241  if (!result)
242  {
243  sLog.outErrorDb("`creature_linking_template` has FLAG_FOLLOW, but no master, (entry: %u, map: %u, master: %u)", slaveEntry, pTmp->mapId, pTmp->masterId);
244  return false;
245  }
246 
247  if (result->GetRowCount() > 1)
248  {
249  sLog.outErrorDb("`creature_linking_template` has FLAG_FOLLOW, but non unique master, (entry: %u, map: %u, master: %u)", slaveEntry, pTmp->mapId, pTmp->masterId);
250  delete result;
251  return false;
252  }
253  Field* fields = result->Fetch();
254  pTmp->masterDBGuid = fields[0].GetUInt32();
255  delete result;
256  }
257  }
258 
259  // All checks are passed, entry/guid is valid
260  return true;
261 }

264 enum EventMask
265 {
266  EVENT_MASK_ON_AGGRO = FLAG_AGGRO_ON_AGGRO,
267  EVENT_MASK_ON_EVADE = FLAG_RESPAWN_ON_EVADE | FLAG_DESPAWN_ON_EVADE | FLAG_EVADE_ON_EVADE,
268  EVENT_MASK_ON_DIE = FLAG_DESPAWN_ON_DEATH | FLAG_SELFKILL_ON_DEATH | FLAG_RESPAWN_ON_DEATH | FLAG_FOLLOW,
269  EVENT_MASK_ON_RESPAWN = FLAG_RESPAWN_ON_RESPAWN | FLAG_DESPAWN_ON_RESPAWN | FLAG_FOLLOW,
270  EVENT_MASK_TRIGGER_TO = FLAG_TO_AGGRO_ON_AGGRO | FLAG_TO_RESPAWN_ON_EVADE | FLAG_FOLLOW,
271  EVENT_MASK_ON_DESPAWN = FLAG_DESPAWN_ON_DESPAWN,
272 };

275 bool CreatureLinkingMgr::IsLinkedEventTrigger(Creature* pCreature) const
276 {
277  // Entry case
278  if (m_eventTriggers.find(pCreature->GetEntry()) != m_eventTriggers.end())
279  return true;
280 
281  // Guid case
282  if (m_eventGuidTriggers.find(pCreature->GetDbGuid()) != m_eventGuidTriggers.end())
283  return true;
284 
285  // Also return true for npcs that trigger reverse actions, or for followers(needed in respawn)
286  if (CreatureLinkingInfo const* pInfo = GetLinkedTriggerInformation(pCreature))
287  return (pInfo->linkingFlag & EVENT_MASK_TRIGGER_TO) != 0;
288 
289  return false;

333 // Function to add slave-NPCs to the holder
334 void CreatureLinkingHolder::AddSlaveToHolder(Creature* pCreature)
335 {
336  CreatureLinkingInfo const* pInfo = sCreatureLinkingMgr.GetLinkedTriggerInformation(pCreature);
337  if (!pInfo)
338  return;
339 
340  if (pInfo->mapId == INVALID_MAP_ID) // Guid case, store master->slaves for fast access
341  {
342  HolderMapBounds bounds = m_holderGuidMap.equal_range(pInfo->masterId);
343  for (HolderMap::iterator itr = bounds.first; itr != bounds.second; ++itr)
344  {
345  if (itr->second.linkingFlag == pInfo->linkingFlag)
346  {
347  itr->second.linkedGuids.push_back(std::make_pair(pCreature->GetDbGuid(), pCreature->GetObjectGuid()));
348  pCreature = nullptr; // Store that is was handled
349  break;
350  }
351  }
352 
353  // If this is a new flag, insert new entry
354  if (pCreature)
355  {
356  InfoAndGuids tmp;
357  tmp.linkedGuids.push_back(std::make_pair(pCreature->GetDbGuid(), pCreature->GetObjectGuid()));
358  tmp.linkingFlag = pInfo->linkingFlag;
359  tmp.searchRange = 0;
360  m_holderGuidMap.insert(HolderMap::value_type(pInfo->masterId, tmp));
361  }
362  return;
363  }
364 
365  // First try to find holder with same flag
366  HolderMapBounds bounds = m_holderMap.equal_range(pInfo->masterId);
367  for (HolderMap::iterator itr = bounds.first; itr != bounds.second; ++itr)
368  {
369  if (itr->second.linkingFlag == pInfo->linkingFlag && itr->second.searchRange == pInfo->searchRange)
370  {
371  itr->second.linkedGuids.push_back(std::make_pair(pCreature->GetDbGuid(), pCreature->GetObjectGuid()));
372  pCreature = nullptr; // Store that is was handled
373  break;
374  }
375  }
376 
377  // If this is a new flag, insert new entry
378  if (pCreature)
379  {
380  InfoAndGuids tmp;
381  tmp.linkedGuids.push_back(std::make_pair(pCreature->GetDbGuid(), pCreature->GetObjectGuid()));
382  tmp.linkingFlag = pInfo->linkingFlag;
383  tmp.searchRange = pInfo->searchRange;
384  m_holderMap.insert(HolderMap::value_type(pInfo->masterId, tmp));
385  }
386 }

407 // Function to process actions for linked NPCs
408 void CreatureLinkingHolder::DoCreatureLinkingEvent(CreatureLinkingEvent eventType, Creature* pSource, Unit* pEnemy /* = nullptr*/)
409 {
410  // This check will be needed in reload case
411  if (!sCreatureLinkingMgr.IsLinkedEventTrigger(pSource))
412  return;
413 
414  // Ignore atypic behaviour
415  if (pSource->IsControlledByPlayer())
416  return;
417 
418  if (eventType == LINKING_EVENT_AGGRO && !pEnemy)
419  return;
420 
421  uint32 eventFlagFilter = 0;
422  uint32 reverseEventFlagFilter = 0;
423 
424  switch (eventType)
425  {
426  case LINKING_EVENT_AGGRO: eventFlagFilter = EVENT_MASK_ON_AGGRO; reverseEventFlagFilter = FLAG_TO_AGGRO_ON_AGGRO; break;
427  case LINKING_EVENT_EVADE: eventFlagFilter = EVENT_MASK_ON_EVADE; reverseEventFlagFilter = FLAG_TO_RESPAWN_ON_EVADE; break;
428  case LINKING_EVENT_DIE: eventFlagFilter = EVENT_MASK_ON_DIE; reverseEventFlagFilter = 0; break;
429  case LINKING_EVENT_RESPAWN: eventFlagFilter = EVENT_MASK_ON_RESPAWN; reverseEventFlagFilter = FLAG_FOLLOW; break;
430  case LINKING_EVENT_DESPAWN: eventFlagFilter = EVENT_MASK_ON_DESPAWN; reverseEventFlagFilter = 0; break;
431  }
432 
433  // Process Slaves (by entry)
434  HolderMapBounds bounds = m_holderMap.equal_range(pSource->GetEntry());
435  for (HolderMap::iterator itr = bounds.first; itr != bounds.second; ++itr)
436  {
437  if (!itr->second.inUse)
438  {
439  itr->second.inUse = true;
440  ProcessSlaveGuidList(eventType, pSource, itr->second.linkingFlag & eventFlagFilter, itr->second.searchRange, itr->second.linkedGuids, pEnemy);
441  itr->second.inUse = false;
442  }
443  }
444 
445  // Process Slaves (by guid)
446  bounds = m_holderGuidMap.equal_range(pSource->GetDbGuid());
447  for (HolderMap::iterator itr = bounds.first; itr != bounds.second; ++itr)
448  {
449  if (!itr->second.inUse)
450  {
451  itr->second.inUse = true;
452  ProcessSlaveGuidList(eventType, pSource, itr->second.linkingFlag & eventFlagFilter, itr->second.searchRange, itr->second.linkedGuids, pEnemy);
453  itr->second.inUse = false;
454  }
455  }
456 
457  // Process Master
458  if (CreatureLinkingInfo const* pInfo = sCreatureLinkingMgr.GetLinkedTriggerInformation(pSource))
459  {
460  if (pInfo->linkingFlag & reverseEventFlagFilter)
461  {
462  Creature* pMaster = nullptr;
463  if (pInfo->mapId != INVALID_MAP_ID) // entry case
464  {
465  BossGuidMapBounds finds = m_masterGuid.equal_range(pInfo->masterId);
466  for (BossGuidMap::const_iterator itr = finds.first; itr != finds.second; ++itr)
467  {
468  Creature* master = pSource->GetMap()->GetCreature(itr->second);
469  if (master && IsSlaveInRangeOfMaster(pSource, master, pInfo->searchRange))
470  {
471  pMaster = master;
472  break;
473  }
474  }
475  }
476  else // guid case
477  {
478  CreatureData const* masterData = sObjectMgr.GetCreatureData(pInfo->masterDBGuid);
479  CreatureInfo const* cInfo = ObjectMgr::GetCreatureTemplate(masterData->id);
480  pMaster = pSource->GetMap()->GetCreature(pInfo->masterDBGuid);
481  }
482 
483  if ((!pMaster || pMaster->IsCorpse()) && eventType == LINKING_EVENT_EVADE && pSource->IsUsingNewSpawningSystem())
484  pSource->GetMap()->GetSpawnManager().RespawnCreature(pInfo->masterDBGuid);
485  else if (pMaster)
486  {
487  switch (eventType)
488  {
489  case LINKING_EVENT_AGGRO:
490  if (pMaster->IsControlledByPlayer())
491  return;
492 
493  if (pMaster->IsInCombat()) // TODO: Add group leashing
494  {
495  pMaster->AddThreat(pEnemy);
496  pEnemy->AddThreat(pMaster);
497  pEnemy->SetInCombatWith(pMaster);
498  pEnemy->GetCombatManager().TriggerCombatTimer(pMaster);
499  }
500  else
501  pMaster->AI()->AttackStart(pEnemy);
502  break;
503  case LINKING_EVENT_EVADE:
504  if (!pMaster->IsAlive())
505  pMaster->Respawn();
506  break;
507  case LINKING_EVENT_RESPAWN:
508  if (pMaster->IsAlive())
509  SetFollowing(pSource, pMaster);
510  break;
511  case LINKING_EVENT_DIE: // Nothing linked for this case
512  case LINKING_EVENT_DESPAWN: // Nothing linked for this case
513  break;
514  }
515  }
516  }
517  }
518 }

520 // Helper function, to process a slave list
521 void CreatureLinkingHolder::ProcessSlaveGuidList(CreatureLinkingEvent eventType, Creature* pSource, uint32 flag, uint16 searchRange, std::list<std::pair<uint32, ObjectGuid>>& slaveGuidList, Unit* pEnemy)
522 {
523  if (!flag)
524  return;
525 
526  uint32 preprocessFlag = 0;
527  uint32 postprocessFlag = flag;
528  if (pSource->IsUsingNewSpawningSystem())
529  {
530  preprocessFlag = (postprocessFlag & (FLAG_RESPAWN_ON_EVADE | FLAG_RESPAWN_ON_DEATH | FLAG_RESPAWN_ON_RESPAWN));
531  postprocessFlag = (postprocessFlag & ~(FLAG_RESPAWN_ON_EVADE | FLAG_RESPAWN_ON_DEATH | FLAG_RESPAWN_ON_RESPAWN));
532  }
533 
534  for (auto slave_itr = slaveGuidList.begin(); slave_itr != slaveGuidList.end();)
535  {
536  Creature* pSlave;
537  if ((*slave_itr).first)
538  pSlave = pSource->GetMap()->GetCreature((*slave_itr).first);
539  else
540  pSlave = pSource->GetMap()->GetCreature((*slave_itr).second);
541  if ((!pSlave || pSlave->IsCorpse()) && preprocessFlag) // dynguid respawning
542  pSource->GetMap()->GetSpawnManager().RespawnCreature((*slave_itr).first);
543  if (!pSlave)
544  {
545  // Remove old guid first
546  slave_itr = slaveGuidList.erase(slave_itr);
547  continue;
548  }
549 
550  ++slave_itr;
551 
552  // Ignore Pets
553  if (pSlave->IsPet())
554  continue;
555 
556  // Handle single slave
557  if (IsSlaveInRangeOfMaster(pSlave, pSource, searchRange))
558  ProcessSlave(eventType, pSource, postprocessFlag, pSlave, pEnemy);
559  }
560 }

562 // Helper function, to process a single slave
563 void CreatureLinkingHolder::ProcessSlave(CreatureLinkingEvent eventType, Creature* pSource, uint32 flag, Creature* pSlave, Unit* pEnemy)
564 {
565  switch (eventType)
566  {
567  case LINKING_EVENT_AGGRO:
568  if (flag & FLAG_AGGRO_ON_AGGRO)
569  {
570  if (pSlave->IsControlledByPlayer())
571  return;
572 
573  if (pSlave->IsInCombat())
574  {
575  pSlave->AddThreat(pEnemy);
576  pEnemy->AddThreat(pSlave);
577  pEnemy->SetInCombatWith(pSlave);
578  pEnemy->GetCombatManager().TriggerCombatTimer(pSlave);
579  }
580  else
581  pSlave->AI()->AttackStart(pEnemy);
582  }
583  break;
584  case LINKING_EVENT_EVADE:
585  if (flag & FLAG_DESPAWN_ON_EVADE && pSlave->IsAlive())
586  pSlave->ForcedDespawn();
587  if (flag & FLAG_RESPAWN_ON_EVADE && !pSlave->IsAlive())
588  pSlave->Respawn();
589  if (flag & FLAG_EVADE_ON_EVADE && pSlave->IsAlive())
590  pSlave->AI()->EnterEvadeMode();
591  break;
592  case LINKING_EVENT_DIE:
593  if (flag & FLAG_SELFKILL_ON_DEATH && pSlave->IsAlive())
594  pSlave->Suicide();
595  if (flag & FLAG_DESPAWN_ON_DEATH && pSlave->IsAlive())
596  pSlave->ForcedDespawn();
597  if (flag & FLAG_RESPAWN_ON_DEATH && !pSlave->IsAlive())
598  pSlave->Respawn();
599  break;
600  case LINKING_EVENT_RESPAWN:
601  if (flag & FLAG_RESPAWN_ON_RESPAWN)
602  {
603  // Additional check to prevent endless loops (in case whole group respawns on first respawn)

707 bool CreatureLinkingHolder::CanSpawn(uint32 dbGuid, Map* _map, CreatureLinkingInfo const* pInfo, float sx, float sy) const
708 {
709  if (!pInfo) // Prepare data for recursive use
710  {
711  CreatureData const* data = sObjectMgr.GetCreatureData(dbGuid);
712  if (!data)
713  return true;
714  if (data->spawnMask && (data->spawnMask & (1 << _map->GetDifficulty())) == 0)
715  return false;
716  pInfo = sCreatureLinkingMgr.GetLinkedTriggerInformation(data->id, dbGuid, data->mapid);
717  if (!pInfo)
718  return true;
719  // Has dbGuid npc actually spawning linked?
720  if (!sCreatureLinkingMgr.IsSpawnedByLinkedMob(pInfo))
721  return true;
722 
723  sx = data->posX; // Fill position data
724  sy = data->posY;
725  }
726 
727  if (pInfo->searchRange == 0) // Map wide case
728  {
729  if (!pInfo->masterDBGuid)
730  return false; // This should never happen
731 
732  if (pInfo->linkingFlag & FLAG_CANT_SPAWN_IF_BOSS_DEAD)
733  {
734  if (InstanceData* data = _map->GetInstanceData())
735  if (data->IsEncounterInProgress())
736  return false;
737 
738  return IsRespawnReady(pInfo->masterDBGuid, _map);
739  }
740  if (pInfo->linkingFlag & FLAG_CANT_SPAWN_IF_BOSS_ALIVE)
741  return !IsRespawnReady(pInfo->masterDBGuid, _map);
742  return true;
743  }
744 
745  // Search for nearby master
746  BossGuidMapBounds finds = m_masterGuid.equal_range(pInfo->masterId);
747  for (BossGuidMap::const_iterator itr = finds.first; itr != finds.second; ++itr)
748  {
749  Creature* pMaster = _map->GetCreature(itr->second);
750  if (pMaster && IsSlaveInRangeOfMaster(pMaster, sx, sy, pInfo->searchRange))
751  {
752  if (pInfo->linkingFlag & FLAG_CANT_SPAWN_IF_BOSS_DEAD)
753  return pMaster->IsAlive();
754  if (pInfo->linkingFlag & FLAG_CANT_SPAWN_IF_BOSS_ALIVE)
755  return !pMaster->IsAlive();
756  return true;
757  }
758  }
759 
760  return true; // local boss does not exist - spawn
761 }
