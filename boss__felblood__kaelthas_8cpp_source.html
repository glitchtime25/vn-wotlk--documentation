356  void ExecuteAction(uint32 action) override
357  {
358  switch (action)

362  DoScriptText(SAY_TIRED, m_creature);
363  DoCastSpellIfCan(nullptr, m_isRegularMode ? SPELL_POWER_FEEDBACK : SPELL_POWER_FEEDBACK_H);
364  SetActionReadyStatus(action, false);
365  return;
366  }
367  case KAEL_ACTION_GRAVITY_LAPSE:
368  {
369  // Cast Gravity Lapse on Players
370  if (DoCastSpellIfCan(nullptr, SPELL_GRAVITY_LAPSE) == CAST_OK)
371  {
372  if (m_firstGravityLapse)
373  {
374  DoScriptText(SAY_GRAVITY_LAPSE, m_creature);
375  m_firstGravityLapse = false;
376  }
377  else
378  DoScriptText(SAY_RECAST_GRAVITY, m_creature);
379 
380  ResetTimer(KAEL_GRAVITY_LAPSE_SCRIPT, 4500);
381  m_gravityLapseStage = 0;
382  SetCombatScriptStatus(true);
383  m_creature->SetTarget(nullptr);
384  m_creature->SetFacingTo(m_creature->GetRespawnPosition().o);
385  }
386  return;
387  }
388  case KAEL_ACTION_PHASE_TRANSITION:
389  {
390  // Below 50%
391  if (m_creature->GetHealthPercent() < 50.0f)
392  {
393  if (DoCastSpellIfCan(nullptr, SPELL_TELEPORT_CENTER) == CAST_OK)
394  {
395  SetCombatMovement(false);
396  SetMeleeEnabled(false);
397 
398  m_isFirstPhase = false;
399  SetActionReadyStatus(action, false);
400  DisableCombatAction(KAEL_ACTION_PYROBLAST);
401  DisableCombatAction(KAEL_ACTION_SHOCK_BARRIER);
402  DisableCombatAction(KAEL_ACTION_FLAMESTRIKE);
403  DisableCombatAction(KAEL_ACTION_PHOENIX);
404  DisableCombatAction(KAEL_ACTION_FIREBALL);
405  SetActionReadyStatus(KAEL_ACTION_GRAVITY_LAPSE, true);
406  }
407  }
408  return;
409  }
410  case KAEL_ACTION_PYROBLAST:
411  {
412  if (DoCastSpellIfCan(nullptr, SPELL_PYROBLAST) == CAST_OK)
413  {
414  DoScriptText(SAY_PYROBLAST, m_creature);
415  DisableCombatAction(action);
416  }
417  return;
418  }
419  case KAEL_ACTION_SHOCK_BARRIER:
420  {
421  if (DoCastSpellIfCan(nullptr, SPELL_SHOCK_BARRIER) == CAST_OK)
422  {
423  ResetCombatAction(KAEL_ACTION_PYROBLAST, 2000);
424  ResetCombatAction(action, 60000);
425  }
426  return;
427  }
428  case KAEL_ACTION_FLAMESTRIKE:
429  {
430  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_FLAME_STRIKE, SELECT_FLAG_PLAYER))
431  {
432  if (DoCastSpellIfCan(target, m_isRegularMode ? SPELL_FLAME_STRIKE : SPELL_FLAME_STRIKE_H) == CAST_OK)
433  {
434  DoScriptText(SAY_FLAMESTRIKE, m_creature);

439  }

442  if (DoCastSpellIfCan(nullptr, SPELL_PHOENIX) == CAST_OK)

444  DoScriptText(SAY_PHOENIX, m_creature);
