76  void UpdateAI(const uint32 uiDiff) override
77  {
78  // Return since we have no target
79  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
80  return;
81 
82  // Chromatic Chaos at 50%
83  if (!m_bHasChromaticChaos && m_creature->GetHealthPercent() < 50.0f)
84  {
85  if (m_pInstance)
86  {
87  if (Creature* pNefarius = m_pInstance->GetSingleCreatureFromStorage(NPC_LORD_VICTOR_NEFARIUS))
88  {
89  pNefarius->CastSpell(m_creature, SPELL_CHROMATIC_CHAOS, TRIGGERED_OLD_TRIGGERED);
90  DoScriptText(SAY_NEFARIUS_BUFF_GYTH, pNefarius);
91  m_bHasChromaticChaos = true;
92  }
93  }
94  }
95 
96  // CorrosiveAcid_Timer
97  if (uiCorrosiveAcidTimer < uiDiff)
98  {
99  if (DoCastSpellIfCan(m_creature, SPELL_CORROSIVE_ACID) == CAST_OK)
100  uiCorrosiveAcidTimer = 7000;
101  }
102  else
103  uiCorrosiveAcidTimer -= uiDiff;
104 
105  // Freeze_Timer
106  if (uiFreezeTimer < uiDiff)
107  {
108  if (DoCastSpellIfCan(m_creature, SPELL_FREEZE) == CAST_OK)
109  uiFreezeTimer = 16000;
110  }
111  else
112  uiFreezeTimer -= uiDiff;
113 
114  // Flamebreath_Timer
115  if (uiFlamebreathTimer < uiDiff)
116  {
117  if (DoCastSpellIfCan(m_creature, SPELL_FLAME_BREATH) == CAST_OK)
118  uiFlamebreathTimer = 10500;
119  }
120  else
121  uiFlamebreathTimer -= uiDiff;
122 
123  if (uiKnockAwayTimer < uiDiff)
124  {
125  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_KNOCK_AWAY) == CAST_OK)
126  uiKnockAwayTimer = 23000;
127  }
128  else
129  uiKnockAwayTimer -= uiDiff;
130 
131  // Summon Rend
132  if (!m_bSummonedRend && m_creature->GetHealthPercent() < 11.0f)
133  {
134  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_REND) == CAST_OK)
135  {
136  SetDeathPrevention(false);
137  m_creature->RemoveAurasDueToSpell(SPELL_REND_MOUNTS);
138  m_bSummonedRend = true;
139  }
140  }
141 
142  DoMeleeAttackIfReady();
143  }
144 };
