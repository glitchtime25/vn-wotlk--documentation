128  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
129  {
130  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
131  {
132  if (PlayerbotAI::ORDERS_HEAL & m_ai.GetCombatOrder())
133  return HealPlayer(GetHealTarget());
134  else
135  return RETURN_NO_ACTION_OK; // wait it out
136  }
137  else

139  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
140  }
141  }
142 
143  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
144  {
145  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
146  return RETURN_NO_ACTION_OK; // wait it out
147  else
148  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
149  }
150 
151  switch (m_ai.GetScenarioType())
152  {
153  case PlayerbotAI::SCENARIO_PVP_DUEL:
154  case PlayerbotAI::SCENARIO_PVP_BG:
155  case PlayerbotAI::SCENARIO_PVP_ARENA:
156  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:

158  case PlayerbotAI::SCENARIO_PVE:
159  case PlayerbotAI::SCENARIO_PVE_ELITE:

193 CombatManeuverReturns PlayerbotShamanAI::DoNextCombatManeuverPVE(Unit* pTarget)
194 {
195  uint32 spec = m_bot.GetSpec();
196 
197  // Make sure healer stays put, don't even melee (aggro) if in range.
198  if (m_ai.IsHealer() && m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_RANGED)
199  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
200  else if (!m_ai.IsHealer() && m_ai.GetCombatStyle() != PlayerbotAI::COMBAT_MELEE)
201  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
202 
203  // Dispel disease/poison
204  if (m_ai.HasDispelOrder() && DispelPlayer() & RETURN_CONTINUE)
205  return RETURN_CONTINUE;
206 
207  // Heal (try to pick a target by on common rules, than heal using each PlayerbotClassAI HealPlayer() method)
208  if (FindTargetAndHeal())
209  return RETURN_CONTINUE;
210 
211  // Damage Spells
212  DropTotems();
213  CheckShields();
214  UseCooldowns();
215  switch (spec)
216  {
217  case SHAMAN_SPEC_ENHANCEMENT:
218  if (STORMSTRIKE > 0 && (m_bot.IsSpellReady(STORMSTRIKE)) && m_ai.CastSpell(STORMSTRIKE, *pTarget) == SPELL_CAST_OK)
219  return RETURN_CONTINUE;

221  return RETURN_CONTINUE;
222  if (EARTH_SHOCK > 0 && (m_bot.IsSpellReady(EARTH_SHOCK)) && m_ai.CastSpell(EARTH_SHOCK, *pTarget) == SPELL_CAST_OK)

227  return RETURN_CONTINUE;
228  /*if (FOCUSED > 0 && m_ai.CastSpell(FOCUSED, *pTarget) == SPELL_CAST_OK)
229  return RETURN_CONTINUE;*/

231 
232  case SHAMAN_SPEC_RESTORATION:
233  // fall through to elemental
234 
235  case SHAMAN_SPEC_ELEMENTAL:
236  if (FLAME_SHOCK > 0 && (!pTarget->HasAura(FLAME_SHOCK)) && m_ai.CastSpell(FLAME_SHOCK, *pTarget) == SPELL_CAST_OK)
237  return RETURN_CONTINUE;
238  if (LAVA_BURST > 0 && (pTarget->HasAura(FLAME_SHOCK)) && (m_bot.IsSpellReady(LAVA_BURST)) && m_ai.CastSpell(LAVA_BURST, *pTarget) == SPELL_CAST_OK)

241  return RETURN_CONTINUE;
242  /*if (WIND_SHOCK > 0 && m_ai.CastSpell(WIND_SHOCK, *pTarget) == SPELL_CAST_OK)
243  return RETURN_CONTINUE;*/
244  /*if (FROST_SHOCK > 0 && !pTarget->HasAura(FROST_SHOCK, EFFECT_INDEX_0) && m_ai.CastSpell(FROST_SHOCK, *pTarget) == SPELL_CAST_OK)
245  return RETURN_CONTINUE;*/
246  /*if (CHAIN_LIGHTNING > 0 && m_ai.CastSpell(CHAIN_LIGHTNING, *pTarget) == SPELL_CAST_OK)

270 CombatManeuverReturns PlayerbotShamanAI::HealPlayer(Player* target)
271 {
272  CombatManeuverReturns r = PlayerbotClassAI::HealPlayer(target);
273  if (r != RETURN_NO_ACTION_OK)
274  return r;
275 
276  // If target is out of range (40 yards) and is a tank: move towards it
277  // if bot is not asked to stay
278  // Other classes have to adjust their position to the healers
279  // TODO: This code should be common to all healers and will probably
280  // move to a more suitable place like PlayerbotAI::DoCombatMovement()
281  if ((GetTargetJob(target) == JOB_TANK || GetTargetJob(target) == JOB_MAIN_TANK)
282  && m_bot.GetPlayerbotAI()->GetMovementOrder() != PlayerbotAI::MOVEMENT_STAY
283  && !m_ai.In_Reach(target, HEALING_WAVE))
284  {
285  m_bot.GetMotionMaster()->MoveFollow(target, 39.0f, m_bot.GetOrientation());
286  return RETURN_CONTINUE;
287  }
288 
289  // Everyone is healthy enough, return OK. MUST correlate to highest value below (should be last HP check)
290  if (target->GetHealthPercent() >= 80)
291  return RETURN_NO_ACTION_OK;
292 
293  // Technically the best rotation is CHAIN + LHW + LHW, or RIPTIDE + LHW + LHW (proc Tidal Waves then two short LHW), subbing in HW for trouble (bad mana efficiency)
294  if (target->GetHealthPercent() < 30 && HEALING_WAVE > 0 && m_ai.CastSpell(HEALING_WAVE, *target) == SPELL_CAST_OK)
295  return RETURN_CONTINUE;
296  if (target->GetHealthPercent() < 50 && LESSER_HEALING_WAVE > 0 && m_ai.CastSpell(LESSER_HEALING_WAVE, *target) == SPELL_CAST_OK)
297  return RETURN_CONTINUE;
298  if (target->GetHealthPercent() < 60 && RIPTIDE > 0 && !target->HasAura(RIPTIDE, EFFECT_INDEX_0) && m_ai.CastSpell(RIPTIDE, *target) == SPELL_CAST_OK)
299  return RETURN_CONTINUE;
300  if (target->GetHealthPercent() < 80 && CHAIN_HEAL > 0 && m_ai.CastSpell(CHAIN_HEAL, *target) == SPELL_CAST_OK)
301  return RETURN_CONTINUE;
302 
303  return RETURN_NO_ACTION_UNKNOWN;
304 } // end HealTarget

364 void PlayerbotShamanAI::DropTotems()
365 {
366  uint32 spec = m_bot.GetSpec();
367 
368  Totem* earth = m_bot.GetTotem(TOTEM_SLOT_EARTH);
369  Totem* fire = m_bot.GetTotem(TOTEM_SLOT_FIRE);
370  Totem* water = m_bot.GetTotem(TOTEM_SLOT_WATER);
371  Totem* air = m_bot.GetTotem(TOTEM_SLOT_AIR);
372 
373  // Earth Totems
374  if ((earth == nullptr) || (m_bot.GetDistance(earth) > 30))
375  {
376  if (STRENGTH_OF_EARTH_TOTEM > 0 && m_ai.CastSpell(STRENGTH_OF_EARTH_TOTEM) == SPELL_CAST_OK)
377  return;
378  }
379 
380  // Fire Totems
381  if ((fire == nullptr) || (m_bot.GetDistance(fire) > 30))
382  {
383  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_RESIST_FROST && FROST_RESISTANCE_TOTEM > 0 && m_ai.CastSpell(FROST_RESISTANCE_TOTEM) == SPELL_CAST_OK)
384  return;
385  else if (spec == SHAMAN_SPEC_ELEMENTAL && TOTEM_OF_WRATH > 0 && m_ai.CastSpell(TOTEM_OF_WRATH) == SPELL_CAST_OK)
386  return;
387  // If the spec didn't take totem of wrath, use flametongue
388  else if ((spec != SHAMAN_SPEC_ELEMENTAL || TOTEM_OF_WRATH == 0) && FLAMETONGUE_TOTEM > 0 && m_ai.CastSpell(FLAMETONGUE_TOTEM) == SPELL_CAST_OK)
389  return;
390  }
391 
392  // Air totems
393  if ((air == nullptr) || (m_bot.GetDistance(air) > 30))
394  {
395  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_RESIST_NATURE && NATURE_RESISTANCE_TOTEM > 0 && m_ai.CastSpell(NATURE_RESISTANCE_TOTEM) == SPELL_CAST_OK)
396  return;
397  else if (spec == SHAMAN_SPEC_ENHANCEMENT)
398  {
399  if (WIND_FURY_TOTEM > 0 /*&& !m_bot.HasAura(IMPROVED_ICY_TALONS)*/ && m_ai.CastSpell(WIND_FURY_TOTEM) == SPELL_CAST_OK)
400  return;
401  }
402  else
403  {
404  if (WRATH_OF_AIR_TOTEM > 0 && m_ai.CastSpell(WRATH_OF_AIR_TOTEM) == SPELL_CAST_OK)
405  return;
406  }
407  }
408 
409  // Water Totems
