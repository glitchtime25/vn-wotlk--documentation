133  void JustSummoned(Creature* pSummoned) override
134  {
135  switch (pSummoned->GetEntry())
136  {
137  case NPC_CRYSTALLINE_TANGLER:
138  pSummoned->CastSpell(pSummoned, SPELL_CRYSTALLINE_TANGLER, TRIGGERED_OLD_TRIGGERED);
139 
140  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
141  pSummoned->AI()->AttackStart(pTarget);
142  break;
143  case NPC_CRYSTAL_SPIKE_TRIGGER:
144  pSummoned->CastSpell(pSummoned, SPELL_CRYSTAL_SPIKE_PRE, TRIGGERED_OLD_TRIGGERED);
145  ++m_uiSpikeCount;
146  // no break;
147  case NPC_CRYSTAL_SPIKE_INITIAL:
148  // make creature passive
149  pSummoned->AI()->SetReactState(REACT_PASSIVE);
150  pSummoned->SetCanEnterCombat(false);
151 
152  // Update orientation so we can always face the boss
153  pSummoned->SetFacingToObject(m_creature);
154 
155  // allow continuous summoning only until we reach the limit
156  if (m_uiSpikeCount < MAX_ALLOWED_SPIKES)
157  pSummoned->CastSpell(pSummoned, SPELL_CRYSTAL_SPIKE_AURA, TRIGGERED_OLD_TRIGGERED);
158  break;
159  }
160  }

162  void ExecuteAction(uint32 action) override
163  {
164  switch (action)
165  {
166  case ORMOROK_ACTION_TRAMPLE:
167  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_TRAMPLE : SPELL_TRAMPLE_H) == CAST_OK)
168  ResetCombatAction(action, 10000);
169  break;
170  case ORMOROK_ACTION_REFLECTION:
171  if (DoCastSpellIfCan(m_creature, SPELL_REFLECTION) == CAST_OK)
172  ResetCombatAction(action, 30000);
173  break;
174  case ORMOROK_ACTION_CRYSTAL_SPIKE:
175  {
176  uint32 uiSpikeSpell = SPELL_CRYSTAL_SPIKES;
177  if (!m_isRegularMode)
178  uiSpikeSpell = urand(0, 1) ? SPELL_CRYSTAL_SPIKES_H1 : SPELL_CRYSTAL_SPIKES_H2;
179 
180  if (DoCastSpellIfCan(m_creature, uiSpikeSpell) == CAST_OK)
181  {
182  DoScriptText(SAY_ICESPIKE, m_creature);
183  m_uiSpikeCount = 0;
184  ResetCombatAction(action, 30000);
185  }
186  break;
187  }
188  case ORMOROK_ACTION_TANGLER:
189  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_TANGLER_H) == CAST_OK)
190  ResetCombatAction(action, urand(15000, 20000));
191  break;
192  case ORMOROK_ACTION_ENRAGE:
193  if (m_creature->GetHealthPercent() < 25.0f)
194  {
195  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_FRENZY : SPELL_FRENZY_H) == CAST_OK)
196  {
197  DoScriptText(EMOTE_BOSS_GENERIC_FRENZY, m_creature);
198  SetActionReadyStatus(action, false);
199  }
200  }
201  break;
202  }
203  }

243 struct spell_crystal_spike_aura : public AuraScript
244 {
245  void OnPeriodicDummy(Aura* aura) const override
246  {
247  Unit* target = aura->GetTarget();
248  if (!target)
249  return;
250 
251  instance_nexus* pInstance = static_cast<instance_nexus*>(target->GetInstanceData());
252  if (!pInstance)
253  return;
254 
255  Creature* pOrmorok = pInstance->GetSingleCreatureFromStorage(NPC_ORMOROK);
256  if (!pOrmorok)
257  return;
258 
259  // The following spells define the direction of the spike line
260  // All of the spells are targeting the back of the caster, but some take a small turn to left or right
261  // The exact algorithm is unk but we know that the chances of getting a straight line are about 75%. The other two directions are about 12.5% each
262  uint32 castSpellId;
263  if (roll_chance_i(75))
264  castSpellId = SPELL_CRYSTAL_SPIKE_BACK;
265  else
266  castSpellId = urand(0, 1) ? SPELL_CRYSTAL_SPIKE_LEFT : SPELL_CRYSTAL_SPIKE_RIGHT;
267 
268  target->CastSpell(target, castSpellId, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, pOrmorok->GetObjectGuid());
269  }
