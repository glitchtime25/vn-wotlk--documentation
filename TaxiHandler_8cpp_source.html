142 {
143  DEBUG_LOG("WORLD: Received opcode CMSG_ACTIVATETAXIEXPRESS");
144 
145  ObjectGuid guid;
146  uint32 node_count;
147 
148  recv_data >> guid >> node_count;
149 
150  Creature* npc = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_FLIGHTMASTER);
151  if (!npc)
152  {
153  DEBUG_LOG("WORLD: HandleActivateTaxiExpressOpcode - %s not found or you can't interact with it.", guid.GetString().c_str());
154  return;
155  }
156  std::vector<uint32> nodes;
157 
158  for (uint32 i = 0; i < node_count; ++i)
159  {
160  uint32 node;
161  recv_data >> node;
162 
163  if (!_player->m_taxi.IsTaximaskNodeKnown(node) && !_player->isTaxiCheater())
164  {
165  SendActivateTaxiReply(ERR_TAXINOTVISITED);
166  recv_data.rpos(recv_data.wpos()); // prevent additional spam at rejected packet
167  return;
168  }
169 
170  nodes.push_back(node);
171  }
172 
173  if (nodes.empty())
174  return;
175 
176  DEBUG_LOG("WORLD: Received opcode CMSG_ACTIVATETAXIEXPRESS from %d to %d", nodes.front(), nodes.back());
177 
178  GetPlayer()->ActivateTaxiPathTo(nodes, npc);
179 }
