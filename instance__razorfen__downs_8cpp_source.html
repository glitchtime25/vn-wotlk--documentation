109 void instance_razorfen_downs::OnCreatureDeath(Creature* pCreature)
110 {
111  // Only use this function if gong event is in progress
112  if (GetData(TYPE_TUTEN_KASH) != IN_PROGRESS)
113  return;
114 
115  switch (pCreature->GetEntry())
116  {
117  case NPC_TOMB_FIEND:
118  case NPC_TOMB_REAVER:
119  m_lSpawnedMobsList.remove(pCreature->GetObjectGuid());
120 
121  // No more wave-mobs around, enable the gong for the next wave
122  if (m_lSpawnedMobsList.empty())
123  {
124  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_GONG))
125  {
126  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
127 
128  // Workaround - GO need to be respawned - requires core fix, GO shouldn't despawn in the first place
129  pGo->Respawn();
130  }
131  }
132  break;
133  case NPC_TUTEN_KASH:
134  SetData(TYPE_TUTEN_KASH, DONE);
135  break;
136  }
137 }

139 void instance_razorfen_downs::DoSpawnWaveIfCan(GameObject* pGo)
140 {
141  // safety checks
142  if (GetData(TYPE_TUTEN_KASH) == DONE)
143  return;
144 
145  if (!m_lSpawnedMobsList.empty())
146  return;
147 
148  pGo->SendGameObjectCustomAnim(pGo->GetObjectGuid());
149 
150  if (m_uiWaveCounter >= MAX_WAVES)
151  return;
152 
153  for (uint8 i = 0; i < aWaveSummonInformation[m_uiWaveCounter].m_uiNPCperWave; ++i)
154  {
155  uint8 uiPos = i % 2; // alternate spawn between the left and right corridor
156  float fPosX, fPosY, fPosZ;
157  float fTargetPosX, fTargetPosY, fTargetPosZ;
158 
159  pGo->GetRandomPoint(aSpawnLocations[uiPos].x, aSpawnLocations[uiPos].y, aSpawnLocations[uiPos].z, 5.0f, fPosX, fPosY, fPosZ);
160 
161  // move the summoned NPC toward the gong
162  if (Creature* pSummoned = pGo->SummonCreature(aWaveSummonInformation[m_uiWaveCounter].m_uiNpcEntry, fPosX, fPosY, fPosZ, aSpawnLocations[uiPos].o, TEMPSPAWN_DEAD_DESPAWN, 0))
163  {
164  pSummoned->SetWalk(false);
165  pGo->GetContactPoint(pSummoned, fTargetPosX, fTargetPosY, fTargetPosZ);
166  pSummoned->GetMotionMaster()->MovePoint(0, fTargetPosX, fTargetPosY, fTargetPosZ);
167  }
168  }
169 
170  // Will increase m_uiWaveCounter, hence after the wave is summoned
171  SetData(TYPE_TUTEN_KASH, IN_PROGRESS);
172  pGo->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
173 }
