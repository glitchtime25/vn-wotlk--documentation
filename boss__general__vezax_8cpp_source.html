142  void JustSummoned(Creature* pSummoned) override
143  {
144  if (pSummoned->GetEntry() == NPC_SARONITE_VAPOR)
145  {
146  m_lVaporsGuids.push_back(pSummoned->GetObjectGuid());
147 
148  // if vapors have reached the max number for hard mode then summon animus
149  if (m_lVaporsGuids.size() == MAX_HARD_MODE_VAPORS)
150  DoPrepareAnimusIfCan();
151  }
152  else if (pSummoned->GetEntry() == NPC_SARONITE_ANIMUS)

156  void ReceiveAIEvent(AIEventType eventType, Unit* pSender, Unit* /*pInvoker*/, uint32 /*uiMiscValue*/) override
157  {
158  if (pSender->GetEntry() == NPC_SARONITE_VAPOR)
159  {
160  // decrease the number of vapors when they die
161  if (eventType == AI_EVENT_CUSTOM_A)
162  m_lVaporsGuids.remove(pSender->GetObjectGuid());
163  else if (eventType == AI_EVENT_CUSTOM_B)
164  {
165  ++m_uiVaporsGathered;
166 
167  // Cast the saronite formation aura when all vapors arive
168  if (m_uiVaporsGathered == MAX_HARD_MODE_VAPORS)
169  {
170  // visual on the summoning bunny
171  if (m_pInstance)
172  {
173  if (Creature* pBunny = m_creature->GetMap()->GetCreature(m_pInstance->GetVezaxBunnyGuid(true)))
174  pBunny->CastSpell(pBunny, SPELL_ANIMUS_FORMATION, TRIGGERED_OLD_TRIGGERED);
175  }
176 
177  // Despawn the vapors
178  for (GuidList::const_iterator itr = m_lVaporsGuids.begin(); itr != m_lVaporsGuids.end(); ++itr)
179  {
180  if (Creature* pVapor = m_creature->GetMap()->GetCreature(*itr))
181  pVapor->ForcedDespawn();
182  }
183 
184  DoScriptText(EMOTE_ANIMUS, m_creature);
185  m_uiHardModeTimer = 4000;
186  }
187  }
188  }
189  // remove saronite barrier when animus dies
190  else if (pSender->GetEntry() == NPC_SARONITE_ANIMUS && eventType == AI_EVENT_CUSTOM_C)
191  m_creature->RemoveAurasDueToSpell(SPELL_SARONITE_BARRIER);
192  }

214  // Gather the vapors to the spawn point

216  {
217  if (Creature* pVapor = m_creature->GetMap()->GetCreature(*itr))

219  pVapor->SetWalk(false);

221  }
222  }

225  void UpdateAI(const uint32 uiDiff) override
226  {
227  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
228  return;
229 
230  if (m_uiHardModeTimer)
231  {
232  if (m_uiHardModeTimer <= uiDiff)
233  {
234  switch (m_uiHardModeStage)
235  {
236  case 0:
237  {
238  if (DoCastSpellIfCan(m_creature, SPELL_SARONITE_BARRIER, CAST_TRIGGERED | CAST_INTERRUPT_PREVIOUS) == CAST_OK)
239  {
240  DoScriptText(SAY_HARD_MODE, m_creature);
241  m_uiHardModeTimer = 2000;
242  }
243  break;
244  }
245  case 1:
246  {
247  if (m_pInstance)
248  {
249  if (Creature* pBunny = m_creature->GetMap()->GetCreature(m_pInstance->GetVezaxBunnyGuid(true)))
250  {
251  pBunny->RemoveAurasDueToSpell(SPELL_ANIMUS_FORMATION);
252  pBunny->CastSpell(pBunny, SPELL_SUMMON_ANIMUS, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
253  }
254  }
255  m_uiHardModeTimer = 0;
256  break;
257  }
258  }
259  ++m_uiHardModeStage;
260  }
261  else
262  m_uiHardModeTimer -= uiDiff;
263  }
264 
265  // summon saronite vapors before the hard mode
266  if (m_pInstance && m_pInstance->GetData(TYPE_VEZAX_HARD) == NOT_STARTED)
267  {
268  if (m_uiSaroniteVaporTimer < uiDiff)
269  {
270  // get a bunny that will summon the vapors
271  if (m_pInstance)
272  {
273  if (Creature* pBunny = m_creature->GetMap()->GetCreature(m_pInstance->GetVezaxBunnyGuid(false)))
274  {
275  pBunny->CastSpell(pBunny, SPELL_SUMMON_VAPORS, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
276  DoScriptText(EMOTE_VAPOR, m_creature);
277 
278  m_uiSaroniteVaporTimer = 30000;
279  }
280  }
281  }
282  else
283  m_uiSaroniteVaporTimer -= uiDiff;
284  }
285 
286  // Searing flames only while animus is not around
287  if (m_pInstance && (m_pInstance->GetData(TYPE_VEZAX_HARD) != IN_PROGRESS || !m_bIsRegularMode))
288  {
289  if (m_uiFlamesTimer < uiDiff)
290  {
291  if (DoCastSpellIfCan(m_creature, SPELL_SEARING_FLAMES) == CAST_OK)
292  m_uiFlamesTimer = urand(9000, 16000);
293  }
294  else
295  m_uiFlamesTimer -= uiDiff;
296  }
297 
298  if (m_uiSurgeTimer < uiDiff)
299  {
300  if (DoCastSpellIfCan(m_creature, SPELL_SURGE_OF_DARKNESS) == CAST_OK)
301  {
302  DoScriptText(SAY_SURGE, m_creature);
303  DoScriptText(EMOTE_SURGE, m_creature);
304  m_uiSurgeTimer = 62000;
305  }
306  }
307  else
308  m_uiSurgeTimer -= uiDiff;
309 
310  if (m_uiMarkTimer < uiDiff)
311  {
312  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
313  {
314  if (DoCastSpellIfCan(pTarget, SPELL_MARK_OF_FACELESS) == CAST_OK)
315  m_uiMarkTimer = urand(25000, 30000);
316  }
317  }
318  else
319  m_uiMarkTimer -= uiDiff;
320 
321  if (m_uiCrashTimer < uiDiff)
322  {
323  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
324  {
325  if (DoCastSpellIfCan(pTarget, SPELL_SHADOW_CRASH) == CAST_OK)
326  m_uiCrashTimer = 15000;
327  }
328  }
329  else
330  m_uiCrashTimer -= uiDiff;
331 
332  if (m_uiEnrageTimer < uiDiff)
333  {
334  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
335  {
336  DoScriptText(SAY_ENRAGE, m_creature);
337  m_uiEnrageTimer = 30000;
338  }
339  }
340  else
341  m_uiEnrageTimer -= uiDiff;
342 
343  DoMeleeAttackIfReady();
344  }
