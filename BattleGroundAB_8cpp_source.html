37 void BattleGroundAB::Update(uint32 diff)
38 {
39  BattleGround::Update(diff);
40 
41  if (GetStatus() != STATUS_IN_PROGRESS)
42  return;
43 
44  // start counting the team points on every update tick;
45  // every time we start from 0 for each team
46  int team_points[PVP_TEAM_COUNT] = { 0, 0 };
47 
48  for (uint8 node = 0; node < BG_AB_MAX_NODES; ++node)
49  {
50  // 3 sec delay to spawn new banner instead previous despawned one
51  if (m_bannerTimers[node].timer)
52  {
53  if (m_bannerTimers[node].timer <= diff)
54  {
55  DoUpdateBanner(node, m_bannerTimers[node].status, false);
56  m_bannerTimers[node].timer = 0;
57  }
58  else
59  m_bannerTimers[node].timer -= diff;
60  }
61 
62  // 1-minute to occupy a node from contested state
63  if (m_nodeTimers[node])
64  {
65  if (m_nodeTimers[node] <= diff)
66  {
67  m_nodeTimers[node] = 0;
68 
69  // Process the changes from contested to occupied
70  PvpTeamIndex teamIndex = PvpTeamIndex(m_nodeStatus[node] - 1);
71  Team team = GetTeamIdByTeamIndex(PvpTeamIndex(teamIndex));
72 
73  m_prevNodeStatus[node] = m_nodeStatus[node];
74  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED);
75 
76  // create new occupied banner
77  DoUpdateBanner(node, m_nodeStatus[node], true);
78  ProcessNodeCapture(node, teamIndex);
79  DoUpdateNodeWorldstate(node, abOccupiedStates[teamIndex][node]);
80 
81  // Message to chatlog
82  ChatMsg chatSystem = teamIndex == TEAM_INDEX_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE;
83  uint32 factionStrig = teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_ALLY : LANG_BG_HORDE;
84 
85  SendMessage2ToAll(LANG_BG_AB_NODE_TAKEN, chatSystem, nullptr, factionStrig, GetNodeMessageId(node));
86  PlaySoundToAll(teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_SOUND_NODE_CAPTURED_ALLIANCE : BG_AB_SOUND_NODE_CAPTURED_HORDE);
87  }
88  else
89  m_nodeTimers[node] -= diff;
90  }
91 
92  // for each occupied node we add team points;
93  // this influences the abTickPoints and the abTickIntervals values
94  for (uint8 teamIndex = 0; teamIndex < PVP_TEAM_COUNT; ++teamIndex)
95  if (m_nodeStatus[node] == ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED))
96  ++team_points[teamIndex];
97  }
98 
99  // Accumulate points
100  for (uint8 teamIndex = 0; teamIndex < PVP_TEAM_COUNT; ++teamIndex)
101  {
102  int points = team_points[teamIndex];
103  if (!points)
104  continue;
105 
106  m_lastTick[teamIndex] += diff;
107 
108  if (m_lastTick[teamIndex] > abTickIntervals[points])
109  {
110  m_lastTick[teamIndex] -= abTickIntervals[points];
111 
112  int32 worldstateId = teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_OP_RESOURCES_ALLY : BG_AB_OP_RESOURCES_HORDE;
113  int32 newValue = GetBgMap()->GetVariableManager().GetVariable(worldstateId) + abTickPoints[points];
114 
115  if (newValue > BG_AB_MAX_TEAM_SCORE)
116  newValue = BG_AB_MAX_TEAM_SCORE;
117 
118  m_honorScoreTicks[teamIndex] += abTickPoints[points];
119  m_reputationScoreTics[teamIndex] += abTickPoints[points];
120 
121  // update reputation
122  if (m_reputationScoreTics[teamIndex] >= m_reputationTics)
123  {
124  (teamIndex == TEAM_INDEX_ALLIANCE) ? RewardReputationToTeam(509, 10, ALLIANCE) : RewardReputationToTeam(510, 10, HORDE);
125  m_reputationScoreTics[teamIndex] -= m_reputationTics;
126  }
127 
128  // update honor
129  if (m_honorScoreTicks[teamIndex] >= m_honorTicks)
130  {
131  RewardHonorToTeam(GetBonusHonorFromKill(1), (teamIndex == TEAM_INDEX_ALLIANCE) ? ALLIANCE : HORDE);
132  m_honorScoreTicks[teamIndex] -= m_honorTicks;
133  }
134 
135  // send info for near victory
136  if (!m_isInformedNearVictory && newValue > BG_AB_WARNING_NEAR_VICTORY_SCORE)
137  {
138  SendMessageToAll(teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_AB_A_NEAR_VICTORY : LANG_BG_AB_H_NEAR_VICTORY, CHAT_MSG_BG_SYSTEM_NEUTRAL);
139  PlaySoundToAll(teamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_SOUND_NEAR_VICTORY_ALLIANCE : BG_AB_SOUND_NEAR_VICTORY_HORDE);
140  m_isInformedNearVictory = true;
141  }
142 
143  // update resource world state
144  GetBgMap()->GetVariableManager().SetVariable(worldstateId, newValue);
145 
146  // update achievement flags
147  // we increased m_TeamScores[team] so we just need to check if it is 500 more than other teams resources
148  // horde will be a bit disadvantaged, but we can assume that points aren't updated for both team in same Update() call
149  PvpTeamIndex otherTeamIdx = GetOtherTeamIndex(PvpTeamIndex(teamIndex));
150  if (newValue > GetBgMap()->GetVariableManager().GetVariable(otherTeamIdx == TEAM_INDEX_ALLIANCE ? BG_AB_OP_RESOURCES_ALLY : BG_AB_OP_RESOURCES_HORDE) + 500)
151  m_teamScores500Disadvantage[otherTeamIdx] = true;
152  }
153  }
154 
155  // Test win condition
156  if (GetBgMap()->GetVariableManager().GetVariable(BG_AB_OP_RESOURCES_ALLY) >= BG_AB_MAX_TEAM_SCORE)
157  EndBattleGround(ALLIANCE);
158  if (GetBgMap()->GetVariableManager().GetVariable(BG_AB_OP_RESOURCES_HORDE) >= BG_AB_MAX_TEAM_SCORE)
159  EndBattleGround(HORDE);
160 }

260 // Method that handles the node capture by a team
261 void BattleGroundAB::ProcessNodeCapture(uint8 node, PvpTeamIndex teamIdx)
262 {
263  Team team = GetTeamIdByTeamIndex(teamIdx);
264  int32 worldstateId = teamIdx == TEAM_INDEX_ALLIANCE ? BG_AB_OP_OCCUPIED_BASES_ALLY : BG_AB_OP_OCCUPIED_BASES_HORDE;
265  int32 newValue = GetBgMap()->GetVariableManager().GetVariable(worldstateId) + 1;
266  GetBgMap()->GetVariableManager().SetVariable(worldstateId, newValue);
267 
268  DEBUG_LOG("BattleGroundAB: Node with id %u was captured by team %u.", node, team);
269 
270  if (newValue >= 5)
271  CastSpellOnTeam(BG_AB_SPELL_QUEST_REWARD_5_BASES, team);
272  else if (newValue >= 4)
273  CastSpellOnTeam(BG_AB_SPELL_QUEST_REWARD_4_BASES, team);
274 
275  // setup graveyards
276  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(abGraveyardData[node].id, BG_AB_ZONE_MAIN, team);
277 
278  uint32 healerEntry = teamIdx == TEAM_INDEX_ALLIANCE ? BG_NPC_SPIRIT_GUIDE_ALLIANCE : BG_NPC_SPIRIT_GUIDE_HORDE;
279  uint32 defenderEntry = teamIdx == TEAM_INDEX_ALLIANCE ? BG_NPC_HON_DEFENDER_TRIGGER_25_A : BG_NPC_HON_DEFENDER_TRIGGER_25_H;
280 
281  // get an object as a reference to spawn for summoning
282  GameObject* object = GetSingleGameObjectFromStorage(abNodeData[node].id);
283  if (!object)
284  {
285  sLog.outError("BattleGroundAB: Cannot find gameobject entry %u used as a reference in node capture.", abNodeData[node].id);
286  return;
287  }
288 
289  // spawn spirit healer and defender
290  if (Creature* healer = object->SummonCreature(healerEntry, abGraveyardData[node].x, abGraveyardData[node].y, abGraveyardData[node].z, abGraveyardData[node].o, TEMPSPAWN_DEAD_DESPAWN, 0))
291  m_spiritHealers[node] = healer->GetObjectGuid();
292  if (Creature* defender = object->SummonCreature(defenderEntry, abNodeData[node].x, abNodeData[node].y, abNodeData[node].z, abNodeData[node].o, TEMPSPAWN_DEAD_DESPAWN, 0))
293  m_honorableDefender[node] = defender->GetObjectGuid();
294 }

296 // Method that handles the banner click
297 void BattleGroundAB::HandlePlayerClickedOnFlag(Player* player, GameObject* go)
298 {
299  if (GetStatus() != STATUS_IN_PROGRESS)
300  return;
301 
302  DEBUG_LOG("BattleGroundAB: Handle flag clicked by team %u for gameobject entry %u.", player->GetTeam(), go->GetEntry());
303 
304  PvpTeamIndex teamIndex = GetTeamIndexByTeamId(player->GetTeam());
305  uint32 scoreType = 0;
306  uint32 soundId = 0;
307  uint32 worldState = 0;
308  int32 textEntry = 0;
309  uint32 factionStrig = 0;
310 
311  // process battleground event
312  uint8 event = (sBattleGroundMgr.GetGameObjectEventIndex(go->GetDbGuid())).event1;
313  if (event >= BG_AB_MAX_NODES) // not a node
314  return;
315 
316  ABNodes node = ABNodes(event);
317 
318  player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_PVP_ACTIVE_CANCELS);
319 
320  // Node is NEUTRAL -> change to CONTESTED
321  if (m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_TYPE_NEUTRAL))
322  {
323  scoreType = SCORE_BASES_ASSAULTED;
324  worldState = abContestedStates[teamIndex][node];
325 
326  m_prevNodeStatus[node] = m_nodeStatus[node];
327  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
328 
329  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
330 
331  textEntry = LANG_BG_AB_NODE_CLAIMED;
332  factionStrig = teamIndex == TEAM_INDEX_ALLIANCE ? LANG_BG_ALLY : LANG_BG_HORDE;
333  soundId = BG_AB_SOUND_NODE_CLAIMED;
334 
335  // update achievements
336  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 122);
337  }
338  // Node is CONTESTED -> change depends on previous status
339  else if (m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_STATUS_ALLY_CONTESTED) || m_nodeStatus[node] == ABNodeStatus(BG_AB_NODE_STATUS_HORDE_CONTESTED))
340  {
341  // Last state is CONTESTED -> change to CONTESTED (change owner team)
342  if (m_prevNodeStatus[node] < ABNodeStatus(BG_AB_NODE_TYPE_OCCUPIED))
343  {
344  scoreType = SCORE_BASES_ASSAULTED;
345  worldState = abContestedStates[teamIndex][node];
346 
347  m_prevNodeStatus[node] = m_nodeStatus[node];
348  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
349 
350  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
351 
352  textEntry = LANG_BG_AB_NODE_ASSAULTED;
353 
354  // update achievements
355  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 122);
356  }
357  // Last state is OCCUPIED -> change back to OCCUPIED (defend the node)
358  else
359  {
360  scoreType = SCORE_BASES_DEFENDED;
361  worldState = abOccupiedStates[teamIndex][node];
362 
363  m_prevNodeStatus[node] = m_nodeStatus[node];
364  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_OCCUPIED);
365 
366  m_nodeTimers[node] = 0;
367 
368  textEntry = LANG_BG_AB_NODE_DEFENDED;
369 
370  // update achievements
371  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 123);
372 
373  // apply the node capture benefits
374  ProcessNodeCapture(node, teamIndex);
375  }
376 
377  soundId = (teamIndex == TEAM_INDEX_ALLIANCE) ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
378  }
379  // Node is OCCUPIED -> change to CONTESTED
380  else
381  {
382  scoreType = SCORE_BASES_ASSAULTED;
383  worldState = abContestedStates[teamIndex][node];
384 
385  m_prevNodeStatus[node] = m_nodeStatus[node];
386  m_nodeStatus[node] = ABNodeStatus(teamIndex + BG_AB_NODE_TYPE_CONTESTED);
387 
388  m_nodeTimers[node] = BG_AB_FLAG_CAPTURING_TIME;
389 
390  textEntry = LANG_BG_AB_NODE_ASSAULTED;
391  soundId = (teamIndex == TEAM_INDEX_ALLIANCE) ? BG_AB_SOUND_NODE_ASSAULTED_ALLIANCE : BG_AB_SOUND_NODE_ASSAULTED_HORDE;
392 
393  PvpTeamIndex otherTeamIndex = GetOtherTeamIndex(teamIndex);
394  int32 worldstateId = otherTeamIndex == TEAM_INDEX_ALLIANCE ? BG_AB_OP_OCCUPIED_BASES_ALLY : BG_AB_OP_OCCUPIED_BASES_HORDE;
395  int32 newValue = GetBgMap()->GetVariableManager().GetVariable(worldstateId) - 1;
396  GetBgMap()->GetVariableManager().SetVariable(worldstateId, newValue);
397 
398  // unlink graveyard
399  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(abGraveyardData[node].id, BG_AB_ZONE_MAIN, TEAM_INVALID);
400 
401  // despawn spirit healer and defender
402  if (Creature* healer = GetBgMap()->GetCreature(m_spiritHealers[node]))
403  healer->ForcedDespawn();
404  if (Creature* defender = GetBgMap()->GetCreature(m_honorableDefender[node]))
405  defender->ForcedDespawn();
406 
407  // update achievements
408  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 122);
409  }
410 
411  // update score
412  UpdatePlayerScore(player, scoreType, 1);
413 
414  // update banner object
415  DoUpdateBanner(node, m_nodeStatus[node], true);
416 
417  // update world state
418  DoUpdateNodeWorldstate(node, worldState);
419 
420  // send message; this depends on the node status
421  ChatMsg chatSystem = teamIndex == TEAM_INDEX_ALLIANCE ? CHAT_MSG_BG_SYSTEM_ALLIANCE : CHAT_MSG_BG_SYSTEM_HORDE;
422 
423  if (factionStrig)
424  SendMessage2ToAll(textEntry, chatSystem, player, GetNodeMessageId(node), factionStrig);
425  else
426  SendMessage2ToAll(textEntry, chatSystem, player, GetNodeMessageId(node));
427 
428  // play sound to all players
429  PlaySoundToAll(soundId);
430 }

498 void BattleGroundAB::EndBattleGround(Team winner)
