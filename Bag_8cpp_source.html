58 bool Bag::Create(uint32 guidlow, uint32 itemid, Player const* owner)
59 {
60  ItemPrototype const* itemProto = ObjectMgr::GetItemPrototype(itemid);
61 
62  if (!itemProto || itemProto->ContainerSlots > MAX_BAG_SIZE)
63  return false;
64 
65  Object::_Create(guidlow, guidlow, 0, HIGHGUID_CONTAINER);
66 
67  SetEntry(itemid);
68  SetObjectScale(DEFAULT_OBJECT_SCALE);
69 
70  SetGuidValue(ITEM_FIELD_OWNER, owner ? owner->GetObjectGuid() : ObjectGuid());
71  SetGuidValue(ITEM_FIELD_CONTAINED, owner ? owner->GetObjectGuid() : ObjectGuid());
72 
73  SetUInt32Value(ITEM_FIELD_MAXDURABILITY, itemProto->MaxDurability);
74  SetUInt32Value(ITEM_FIELD_DURABILITY, itemProto->MaxDurability);
75  SetUInt32Value(ITEM_FIELD_STACK_COUNT, 1);
76 
77  // Setting the number of Slots the Container has
78  SetUInt32Value(CONTAINER_FIELD_NUM_SLOTS, itemProto->ContainerSlots);
79 
80  // Cleaning 20 slots
81  for (uint8 i = 0; i < MAX_BAG_SIZE; ++i)
82  {
83  SetGuidValue(CONTAINER_FIELD_SLOT_1 + (i * 2), ObjectGuid());
84  m_bagslot[i] = nullptr;
85  }
86 
87  return true;
88 }
