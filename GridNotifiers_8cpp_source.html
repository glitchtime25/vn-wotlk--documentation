41 void VisibleNotifier::Notify()
42 {
43  Player& player = *i_camera.GetOwner();
44  // at this moment i_clientGUIDs have guids that not iterate at grid level checks
45  // but exist one case when this possible and object not out of range: transports
46  if (GenericTransport* transport = player.GetTransport())
47  {
48  for (auto itr : transport->GetPassengers())
49  {
50  if (i_clientGUIDs.find(itr->GetObjectGuid()) != i_clientGUIDs.end())
51  {
52  // ignore far sight case
53  if (itr->IsPlayer())
54  static_cast<Player*>(itr)->UpdateVisibilityOf(static_cast<Player*>(itr), &player);
55  player.UpdateVisibilityOf(&player, itr, i_data, i_visibleNow);
56  i_clientGUIDs.erase(itr->GetObjectGuid());
57  }
58  }
59  }
60 
61  // Far objects update on player notify
62  for (GuidSet::iterator itr = i_clientGUIDs.begin(); itr != i_clientGUIDs.end();)
63  {
64  GuidSet::iterator current = itr++;
65  if (WorldObject* obj = player.GetMap()->GetWorldObject(*current))
66  {
67  if (!obj->GetVisibilityData().IsVisibilityOverridden())
68  continue;
69 
70  player.UpdateVisibilityOf(&player, obj);
71  i_clientGUIDs.erase(current);
72  }
73  }
74 
75  for (GuidSet::iterator itr = i_clientGUIDs.begin(); itr != i_clientGUIDs.end();)
76  {
77  if ((*itr).IsMOTransport())
78  {
79  itr = i_clientGUIDs.erase(itr);
80  continue;
81  }
82  ++itr;
83  }
84 
85  // generate outOfRange for not iterate objects
86  i_data.AddOutOfRangeGUID(i_clientGUIDs);
87  for (GuidSet::iterator itr = i_clientGUIDs.begin(); itr != i_clientGUIDs.end(); ++itr)
88  {
89  if (WorldObject* target = player.GetMap()->GetWorldObject(*itr))
90  {
91  if (target->GetTypeId() == TYPEID_UNIT)
92  player.BeforeVisibilityDestroy(static_cast<Creature*>(target));
93  player.RemoveAtClient(target);
94  }
95  else
96  sLog.outCustomLog("Object was %s in current map.", player.GetMap()->m_objRemoveList.find(*itr) == player.GetMap()->m_objRemoveList.end() ? "not found" : "found");
97 
98 
99  DEBUG_FILTER_LOG(LOG_FILTER_VISIBILITY_CHANGES, "%s is out of range (no in active cells set) now for %s",
100  itr->GetString().c_str(), player.GetGuidStr().c_str());
101  }
102 
103  if (i_data.HasData())
104  {
105  // send create/outofrange packet to player (except player create updates that already sent using SendUpdateToPlayer)
106  for (size_t i = 0; i < i_data.GetPacketCount(); ++i)
107  {
108  WorldPacket packet = i_data.BuildPacket(i);
109  player.GetSession()->SendPacket(packet);
110  }
111 
112  // send out of range to other players if need
113  GuidSet const& oor = i_data.GetOutOfRangeGUIDs();
114  for (auto iter : oor)
115  {
116  if (!iter.IsPlayer())
117  continue;
118 
119  if (Player* plr = ObjectAccessor::FindPlayer(iter))
120  plr->UpdateVisibilityOf(plr->GetCamera().GetBody(), &player);
121  }
122  }
123 
124  // Now do operations that required done at object visibility change to visible
125 
126  // send data at target visibility change (adding to client)
127  for (auto vItr : i_visibleNow)
128  {
129  // target aura duration for caster show only if target exist at caster client
130  if (vItr != &player && vItr->isType(TYPEMASK_UNIT))
131  player.SendAurasForTarget((Unit*)vItr);
132  }
133 }

267 void MaNGOS::RespawnDo::operator()(Creature* u) const
268 {
269  // prevent respawn creatures for not active BG event
270  Map* map = u->GetMap();
271  if (map->IsBattleGroundOrArena())
272  {
273  BattleGroundEventIdx eventId = sBattleGroundMgr.GetCreatureEventIndex(u->GetDbGuid());
274  if (!((BattleGroundMap*)map)->GetBG()->IsActiveEvent(eventId.event1, eventId.event2))
275  return;
276  }
277 
278  if (u->IsUsingNewSpawningSystem())
279  {
280  if (u->GetMap()->GetMapDataContainer().GetSpawnGroupByGuid(u->GetDbGuid(), TYPEID_UNIT))
281  u->GetMap()->GetPersistentState()->SaveCreatureRespawnTime(u->GetDbGuid(), time(nullptr));
282  else
283  u->GetMap()->GetSpawnManager().RespawnCreature(u->GetDbGuid(), 0);
284  }
285  else
286  u->Respawn();
287 }
