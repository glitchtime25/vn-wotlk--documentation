78  void ExecuteAction(uint32 action) override
79  {
80  switch (action)
81  {
82  case PATCHWERK_ENRAGE_LOW:
83  if (m_creature->GetHealthPercent() <= 5.f)
84  {
85  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
86  {
87  DoBroadcastText(EMOTE_ENRAGE, m_creature);
88  DisableCombatAction(action);
89  }
90  }
91  break;
92  case PATCHWERK_BERSERK:
93  DoBroadcastText(EMOTE_BERSERK, m_creature);
94  m_creature->CastSpell(nullptr, SPELL_BERSERK, TRIGGERED_OLD_TRIGGERED);
95  m_creature->SetSpellList(SPELLSET_BERSERK);
96  break;
97  }
98  }
