129  void ExecuteAction(uint32 action) override
130  {
131  switch (action)
132  {
133  case FANKRISS_MORTAL_WOUND:
134  {
135  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_MORTAL_WOUND) == CAST_OK)
136  ResetCombatAction(action, urand(7000, 14000));
137  break;
138  }
139  case FANKRISS_SUMMON_WORM:
140  {
141  if (DoCastSpellIfCan(nullptr, m_summonWormSpells[m_wormsSpawnPerWave - 1]) == CAST_OK)
142  {
143  --m_wormsSpawnPerWave;
144  // Initialise the next wave now that all Spawn of Fankriss for the current wave were spawned
145  if (!m_wormsSpawnPerWave)
146  {
147  // Determine how many Spawn of Fankriss will be spawned in the next wave
148  m_wormsSpawnPerWave = urand(1, 3);
149  // Randomise the summoning spells so each NPC spawns in a different location
150  std::shuffle(m_summonWormSpells.begin(), m_summonWormSpells.end(), *GetRandomGenerator());
151  ResetCombatAction(action, urand(22000, 70000));
152  }
153  else // Summon another Spawn of Fankriss in a few seconds
154  ResetCombatAction(action, urand(5000, 7000));
155  }
156  break;
157  }
158  case FANKRISS_ENTANGLE:
159  {
160  uint32 entangleIndex = urand(0, 2);
161  // Teleport and untangle one random player into one of the three alcoves
162  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, uint32(0), SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK))
163  {
164  if (DoCastSpellIfCan(target, aEntangleSpells[entangleIndex]) == CAST_OK)
165  {
166  m_EntangleTargetGuid = target->GetObjectGuid();
167  ResetCombatAction(action, urand(25000, 55000));
168  }
169  }
170 
171  // At combat start, Vekniss Hatchlings are summoned in all three alcoves
172  if (!m_isInitialEntangleDone)
173  {
174  DoCastSpellIfCan(nullptr, aSpawnHatchlingsSpells[(entangleIndex + 1) % 3], CAST_TRIGGERED);
175  DoCastSpellIfCan(nullptr, aSpawnHatchlingsSpells[(entangleIndex + 2) % 3], CAST_TRIGGERED);
176  m_isInitialEntangleDone = true;
177  }
178  break;
179  }
180  }
181  }
