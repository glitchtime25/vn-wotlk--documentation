75 // process the capture events
76 bool OutdoorPvPGH::HandleEvent(uint32 eventId, Object* source, Object* target)
77 {
78  if (!source->IsGameObject())
79  return false;
80 
81  // If we are not using the lighthouse return
82  if (source->GetEntry() != GO_VENTURE_BAY_LIGHTHOUSE)
83  return false;
84 
85  GameObject* go = static_cast<GameObject*>(source);
86 
87  bool eventHandled = true;
88 
89  switch (eventId)
90  {
91  case EVENT_LIGHTHOUSE_WIN_ALLIANCE:
92  // Ignore the event if the zone is already in alliance control
93  if (m_zoneOwner == ALLIANCE)
94  return true;
95 
96  // Spawn the npcs only when the tower is fully controlled. Also allow the event to handle summons in DB.
97  m_zoneOwner = ALLIANCE;
98  LockLighthouse(go);
99  DespawnVendors(go);
100  eventHandled = false;
101  break;
102  case EVENT_LIGHTHOUSE_WIN_HORDE:
103  // Ignore the event if the zone is already in horde control
104  if (m_zoneOwner == HORDE)
105  return true;
106 
107  // Spawn the npcs only when the tower is fully controlled. Also allow the event to handle summons in DB.
108  m_zoneOwner = HORDE;
109  LockLighthouse(go);
110  DespawnVendors(go);
111  eventHandled = false;
112  break;
113  case EVENT_LIGHTHOUSE_PROGRESS_ALLIANCE:
114  SetBannerVisual(go, CAPTURE_ARTKIT_ALLIANCE, CAPTURE_ANIM_ALLIANCE);
115  break;
116  case EVENT_LIGHTHOUSE_PROGRESS_HORDE:
117  SetBannerVisual(go, CAPTURE_ARTKIT_HORDE, CAPTURE_ANIM_HORDE);
118  break;
119  case EVENT_LIGHTHOUSE_NEUTRAL_ALLIANCE:
120  case EVENT_LIGHTHOUSE_NEUTRAL_HORDE:
121  m_zoneOwner = TEAM_NONE;
122  SetBannerVisual(go, CAPTURE_ARTKIT_NEUTRAL, CAPTURE_ANIM_NEUTRAL);
123  break;
124  }
125 
126  // there are some events which required further DB script
127  return eventHandled;
128 }

131 void OutdoorPvPGH::DespawnVendors(const WorldObject* objRef)
