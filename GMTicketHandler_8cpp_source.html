35 void WorldSession::SendGMTicket(const GMTicket& ticket, time_t now/* = time(nullptr)*/) const
36 {
37  float daysTicketAge = sTicketMgr.GetTicketAgeDays(now, ticket.GetCreatedAt());

41  WorldPacket data(SMSG_GMTICKET_GETTICKET, 4);
42  data << uint32(GMTICKET_STATUS_HASTEXT);

70 void WorldSession::HandleGMTicketCreateOpcode(WorldPacket& recv_data)
71 {
72  if (sTicketMgr.GetSystemStatus() != GMTICKET_QUEUE_STATUS_ENABLED || !m_ticketSquelchTimer.Passed())
73  {
74  recv_data.read_skip(recv_data.size());
75  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
76  return;
77  }
78 
79  uint8 category = 0; // TBC+: "<Uncategorized>"
80  uint32 map;
81  float x, y, z;
82  std::string message;
83 
84  recv_data >> map >> x >> y >> z;
85  recv_data >> message;
86 
87  recv_data.read_skip<uint32>(); // TBC+: need response?
88 
89  recv_data.read_skip<uint8>(); // WotLK: unknown bool
90 
91  uint32 chatDataLineCount;
92  recv_data >> chatDataLineCount;
93 
94  for (uint32 i = 0; i < chatDataLineCount; i++)
95  recv_data.read_skip<uint32>(); // WotLK: chat timestamps
96 
97  uint32 chatDataSizeInflated;
98  recv_data >> chatDataSizeInflated;
99 
100  if (size_t chatDataSizeDeflated = (recv_data.size() - recv_data.rpos()))
101  recv_data.read_skip(chatDataSizeDeflated); // Compressed chat data
102 
103  Player* player = GetPlayer();
104  const ObjectGuid& guid = player->GetObjectGuid();
105 
106  if (!CheckChatMessage(message))
107  {
108  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
109  return;
110  }
111 
112  // Check if open ticket already exists
113  if (GMTicket* existing = sTicketMgr.GetTicketByPlayer(guid))
114  {
115  m_ticketSquelchTimer.Reset(5000);
116  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_ALREADY_EXIST);
117  }
118  else
119  {
120  // First, try to recycle stored abandoned ticket as new: to avoid new/abandon db spam
121  if (GMTicket* abandoned = sTicketMgr.GetTicketByPlayer(player->GetObjectGuid(), GMTICKET_STATE_ABANDONED))
122  {
123  abandoned->Recycle(GetPlayer(), message, category);
124 
125  if (sTicketMgr.Recycle(abandoned))
126  {
127  m_ticketSquelchTimer.Reset(5000);
128  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_SUCCESS);
129  }
130  else
131  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
132  }
133  else
134  {
135  GMTicket* ticket = new GMTicket(GetPlayer(), message, category);
136 
137  if (sTicketMgr.Add(ticket))
138  {
139  m_ticketSquelchTimer.Reset(5000);
140  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_SUCCESS);
141  }
142  else
143  {
144  delete ticket;
145  SendGMTicketResult(SMSG_GMTICKET_CREATE, GMTICKET_RESPONSE_CREATE_ERROR);
146  }
147  }
148  }
149 }

192 void WorldSession::HandleGMSurveySubmitOpcode(WorldPacket& recv_data)
193 {
194  if (sTicketMgr.GetSystemStatus() != GMTICKET_QUEUE_STATUS_ENABLED || !m_ticketSquelchTimer.Passed())
195  {
196  recv_data.read_skip(recv_data.size());
197  return;
198  }
199 
200  m_ticketSquelchTimer.Reset(5000);
201 
202  // GM survey is shown after SMSG_GM_TICKET_STATUS_UPDATE with status 3
203 
204  uint32 surveyId;
205  recv_data >> surveyId; // Survey ID: found in GMSurveySurveys.dbc
206 
207  // Amount of question-answer pairs has hardcoded a limit of 10
208  std::map<uint32, uint8> answers;
209 
210  for (uint8 i = 0; i < MAX_GMSURVEY_QUESTIONS; ++i)
211  {
212  uint32 questionId;
213  recv_data >> questionId; // Question ID: questions found in GMSurveyQuestions.dbc
214 
215  // Detect end of sequence
216  if (!questionId)
217  break;
218 
219  uint8 answer;
220  recv_data >> answer; // Rating: hardcoded limit of 0-5 in pre-Wrath, ranges defined in GMSurveyAnswers.dbc Wrath+
221 
222  answers.insert({questionId, answer});
223 
224  std::string unused;
225  recv_data >> unused; // Answer comment: Unused in stock UI, can be only set by calling Lua function
226 
227  // utf8limit(unused, 11458); // Answer comment max sizes in bytes: Vanilla - 8106:8110, TBC - 11459:11463, Wrath - 582:586
228  }
229 
230  std::string comment;
231  recv_data >> comment; // Comment
232 
233  if (!CheckChatMessage(comment))
234  return;
235 
236  utf8limit(comment, 11469); // Comment max sizes in bytes: Vanilla - 8117:8121, TBC - 11470:11474, Wrath - 593:597
237 
238  // Perform some sanity checks on the input:
239 
240  GMSurveyCurrentSurveyEntry const* currentSurvey = sGMSurveyCurrentSurveyStore.LookupEntry(uint32(m_sessionDbcLocale));
241 
242  // Check if survey id is correct
243  if (!currentSurvey || currentSurvey->surveyID != surveyId)
244  return;
245 
246  GMSurveyEntry const* survey = sGMSurveySurveysStore.LookupEntry(surveyId);
247 
248  // Check if survey entry with provided id exists
249  if (!survey)
250  return;
251 
252  // Check if all provided question ids match the survey entry
253  {
254  for (auto itr = answers.begin(); itr != answers.end(); ++itr)
255  {
256  bool exists = false;
257 
258  for (uint8 j = 0; (j < MAX_GMSURVEY_QUESTIONS && survey->questionID[j] && !exists); ++j)
259  exists = (survey->questionID[j] == itr->first);
260 
261  if (!exists)
262  return;
263 
264  // WotLK: Check if provided answer is allowed for this question
265  bool allowed = false;
266 
267  for (uint32 k = 0; k < sGMSurveyAnswersStore.GetNumRows(); ++k)
268  {
269  if (GMSurveyAnswersEntry const* answer = sGMSurveyAnswersStore.LookupEntry(k))
270  allowed = (answer->questionID == itr->first && answer->sortIndex == itr->second);
271  }
272 
273  if (!allowed)
274  return;
275  }
276  }
277 
278  // We only store closed tickets in memory if they are surveyed, so we lookup one:
279  if (GMTicket* surveyed = sTicketMgr.GetTicketByPlayer(GetPlayer()->GetObjectGuid(), GMTICKET_STATE_CLOSED))
280  {
281  GMSurveyResult* result = new GMSurveyResult(*survey, surveyed->GetId(), answers, comment);
282  sTicketMgr.Survey(surveyed, result);
283  delete result;
284  }
285 }
