122  void SpellHitTarget(Unit* pTarget, SpellEntry const* pSpellEntry) override
123  {
124  if (pTarget == m_creature && pSpellEntry->Id == SPELL_TELEPORT)
125  {
126  // Say Text and cast Soulstorm
127  DoScriptText(SAY_SOULSTORM, m_creature);
128  DoCastSpellIfCan(m_creature, SPELL_SOULSTORM_VISUAL, CAST_TRIGGERED | CAST_INTERRUPT_PREVIOUS);
129  DoCastSpellIfCan(m_creature, SPELL_SOULSTORM, CAST_INTERRUPT_PREVIOUS);
130 
131  // enable phase 2 spells
132  SetCombatMovement(false);
133  ResetCombatAction(BRONJAHM_ACTION_FEAR, urand(1000, 2000));
134  }
135  }
136 
137  void ExecuteAction(uint32 action) override
138  {
139  switch (action)
140  {
141  case BRONJAHM_ACTION_MAGIC_BANE:
142  if (DoCastSpellIfCan(m_creature, SPELL_MAGICS_BANE) == CAST_OK)
143  ResetCombatAction(action, urand(7000, 15000));
144  break;
145  case BRONJAHM_ACTION_CORRUPT_SOUL:
146  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
147  {
148  if (DoCastSpellIfCan(pTarget, SPELL_CORRUPT_SOUL) == CAST_OK)
149  {
150  DoScriptText(SAY_CORRUPT_SOUL, m_creature);
151  ResetCombatAction(action, urand(20000, 30000));
152  }
153  }
154  break;
155  case BRONJAHM_ACTION_SHADOW_BOLT:
156  if (m_creature->GetHealthPercent() < 30.0f)
157  {
158  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
159  {
160  if (DoCastSpellIfCan(pTarget, SPELL_SHADOW_BOLT) == CAST_OK)
161  ResetCombatAction(action, urand(1000, 2000));
162  }
163  }
164  else
165  {
166  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SHADOW_BOLT) == CAST_OK)
167  ResetCombatAction(action, urand(1000, 2000));
168  }
169  break;
170  case BRONJAHM_ACTION_FEAR:
171  if (DoCastSpellIfCan(m_creature, SPELL_FEAR) == CAST_OK)
172  ResetCombatAction(action, urand(10000, 15000));
173  break;
174  case BRONJAHM_ACTION_TELEPORT:
175  if (m_creature->GetHealthPercent() < 30.0f)
176  {
177  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT) == CAST_OK)
178  {
179  // prepare phase 2; disable phase 1 spells
180  SetActionReadyStatus(action, false);
181 
182  DisableCombatAction(BRONJAHM_ACTION_MAGIC_BANE);
183  DisableCombatAction(BRONJAHM_ACTION_CORRUPT_SOUL);
184  }
185  }
186  break;
187  }
188  }
189 };
