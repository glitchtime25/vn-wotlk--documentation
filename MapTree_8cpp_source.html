154  bool StaticMapTree::isInLineOfSight(const Vector3& pos1, const Vector3& pos2, bool ignoreM2Model) const
155  {
156  float maxDist = (pos2 - pos1).magnitude();
157  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too:
158  MANGOS_ASSERT(maxDist < std::numeric_limits<float>::max());
159  // prevent NaN values which can cause BIH intersection to enter infinite loop
160  if (maxDist < 1e-10f)
161  return true;
162  // direction with length of 1
163  G3D::Ray ray = G3D::Ray::fromOriginAndDirection(pos1, (pos2 - pos1) / maxDist);
164  return !getIntersectionTime(ray, maxDist, true, ignoreM2Model);
165  }

172  bool StaticMapTree::getObjectHitPos(Vector3 const& pPos1, Vector3 const& pPos2, Vector3& pResultHitPos, float pModifyDist) const
173  {
174  float maxDist = (pPos2 - pPos1).magnitude();
175  // valid map coords should *never ever* produce float overflow, but this would produce NaNs too:
176  MANGOS_ASSERT(maxDist < std::numeric_limits<float>::max());
177  // prevent NaN values which can cause BIH intersection to enter infinite loop
178  if (maxDist < 1e-10f)
179  {
180  pResultHitPos = pPos2;
181  return false;
182  }
183 
184  Vector3 dir = (pPos2 - pPos1) / maxDist; // direction with length of 1
185  G3D::Ray ray(pPos1, dir);
186  float dist = maxDist;
187  if (getIntersectionTime(ray, dist))
188  {
189  pResultHitPos = pPos1 + dir * dist;
190  if (pModifyDist < 0)
191  {
192  if ((pResultHitPos - pPos1).magnitude() > -pModifyDist)
193  pResultHitPos = pResultHitPos + dir * pModifyDist;
194  else
195  pResultHitPos = pPos1;
196  }
197  else
198  pResultHitPos = pResultHitPos + dir * pModifyDist;
199 
200  return true;
201  }
202 
203  pResultHitPos = pPos2;
204  return false;
205  }

209  float StaticMapTree::getHeight(Vector3 const& pPos, float maxSearchDist) const
210  {
211  float height = G3D::inf();
212  Vector3 dir;
213  if (maxSearchDist >= 0.f)
214  dir = Vector3(0, 0, -1);
215  else
216  dir = Vector3(0, 0, 1);
217  G3D::Ray ray(pPos, dir); // direction with length of 1
218  float maxDist = std::abs(maxSearchDist);
219  if (getIntersectionTime(ray, maxDist))
220  {
221  if (maxSearchDist >= 0.f)
222  height = pPos.z - maxDist;
223  else
224  height = pPos.z + maxDist;
225  }
226  return height;
227  }

270  bool StaticMapTree::InitMap(std::string const& fname, VMapManager2* vm)
271  {
272  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "Initializing StaticMapTree '%s'", fname.c_str());
273 
274  FILE* rf = nullptr;
275  {
276  std::string fullname = iBasePath + fname;
277  rf = fopen(fullname.c_str(), "rb");
278  if (!rf)
279  return false;
280  }
281 
282  bool success = true;
283  char chunk[8];
284 
285  // general info
286  if (!readChunk(rf, chunk, VMAP_MAGIC, 8))
287  success = false;
288 
289  iIsTiled = false;
290  if (success)
291  {
292  uint8 tiled;
293  if (fread(&tiled, sizeof(uint8), 1, rf) == 1)
294  iIsTiled = tiled != 0;
295  else
296  success = false;
297  }
298 
299  // Nodes
300  if (success && !readChunk(rf, chunk, "NODE", 4))
301  success = false;
302 
303  if (success)
304  success = iTree.readFromFile(rf);
305 
306  if (success && !readChunk(rf, chunk, "GOBJ", 4))
307  success = false;
308 
309  if (success)
310  {
311  iNTreeValues = iTree.primCount();
312  iTreeValues = new ModelInstance[iNTreeValues];
313  }
314 
315  // global model spawns
316  // only non-tiled maps have them, and if so exactly one (so far at least...)
317 #ifdef VMAP_DEBUG
318  DEBUG_LOG("Map isTiled: %u", static_cast<uint32>(iIsTiled));
319 #endif
320  if (success)
321  {
322  if (!iIsTiled)
323  {
324  // read model spawns
325  ModelSpawn spawn;
326  while (ModelSpawn::readFromFile(rf, spawn))
327  {
328  // acquire model instance
329  WorldModel* model = vm->acquireModelInstance(iBasePath, spawn.name);
330  if (model)
331  model->setModelFlags(spawn.flags);
332  else
333  ERROR_LOG("StaticMapTree::LoadMapTile() could not acquire WorldModel pointer for '%s'!", spawn.name.c_str());
334 
335  // update tree
336  uint32 referencedVal;
337 
338  fread(&referencedVal, sizeof(uint32), 1, rf);
339  if (!iLoadedSpawns.count(referencedVal))
340  {
341  if (referencedVal > iNTreeValues)
342  {
343  ERROR_LOG("invalid tree element! (%u/%u)", referencedVal, iNTreeValues);
344  continue;
345  }
346 
347  iTreeValues[referencedVal] = ModelInstance(spawn, model);
348  iLoadedSpawns[referencedVal] = 1;
349  }
350  else
351  {
352  ++iLoadedSpawns[referencedVal];
353 #ifdef VMAP_DEBUG
354  if (iTreeValues[referencedVal].ID != spawn.ID)
355  DEBUG_LOG("Error: trying to load wrong spawn in node!");
356  else if (iTreeValues[referencedVal].name != spawn.name)
357  DEBUG_LOG("Error: name mismatch on GUID=%u", spawn.ID);
358 #endif
359  }
360  }
361  }
362  }
363 
364  if (!success && iTreeValues)
365  {
366  delete[] iTreeValues;
367  iTreeValues = nullptr;
368  }
369 
370  fclose(rf);
371  return success;
372  }
