112  uint32 GetInitialActionTimer(YorActions id)
113  {
114  switch (id)
115  {
116  case YOR_ACTION_STOMP: return 15000;
117  case YOR_ACTION_DOUBLE_BREATH: return 8000;
118  default: return 0;
119  }
120  }

122  uint32 GetSubsequentActionTimer(YorActions id)
123  {
124  switch (id)
125  {
126  case YOR_ACTION_STOMP: return 20000;
127  case YOR_ACTION_DOUBLE_BREATH: return urand(9000, 14000);
128  default: return 0;
129  }
130  }

132  void ExecuteActions() override
133  {
134  if (!CanExecuteCombatAction())
135  return;
136 
137  for (uint32 i = 0; i < YOR_COMBAT_ACTION_MAX; ++i)
138  {
139  if (!GetActionReadyStatus(i))
140  continue;
141 
142  switch (i)
143  {
144  case YOR_ACTION_STOMP:
145  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_NEAREST_BY, 0, SPELL_STOMP, SELECT_FLAG_USE_EFFECT_RADIUS))
146  {
147  if (DoCastSpellIfCan(nullptr, SPELL_STOMP) == CAST_OK)
148  {
149  SetActionReadyStatus(i, false);
150  ResetTimer(i, GetSubsequentActionTimer(YorActions(i)));
151  return;
152  }
153  }
154  continue;
155  case YOR_ACTION_DOUBLE_BREATH:
156  if (DoCastSpellIfCan(nullptr, SPELL_DOUBLE_BREATH) == CAST_OK)
157  {
158  SetActionReadyStatus(i, false);
159  ResetTimer(i, GetSubsequentActionTimer(YorActions(i)));
160  return;
161  }
162  continue;
163  }
164  }
165  }
