52  void MoveInLineOfSight(Unit* who) override
53  {
54  if (m_creature->GetDistanceZ(who) > CREATURE_Z_ATTACK_RANGE_MELEE)
55  return;
56 
57  if (m_creature->IsEnemy(who))
58  {
59  // exception for quests 20439 and 24451
60  if (who->HasAura(SPELL_COVENANT_DISGUISE_1) || who->HasAura(SPELL_COVENANT_DISGUISE_2) ||
61  who->HasAura(SPELL_SUNREAVER_DISGUISE_1) || who->HasAura(SPELL_SUNREAVER_DISGUISE_2))
62  return;
63 
64  if (m_creature->IsWithinDistInMap(who, m_creature->GetAttackDistance(who)) && m_creature->IsWithinLOSInMap(who))
65  {
66  if (Player* pPlayer = who->GetBeneficiaryPlayer())
67  {
68  // it's mentioned that pet may also be teleported, if so, we need to tune script to apply to those in addition.
69 
70  if (pPlayer->GetAreaId() == AREA_ID_SILVER_ENCLAVE)
71  DoCastSpellIfCan(pPlayer, SPELL_TRESPASSER_A);
72  else if (pPlayer->GetAreaId() == AREA_ID_SUNREAVER)
73  DoCastSpellIfCan(pPlayer, SPELL_TRESPASSER_H);

125 bool AreaTrigger_at_underbelly_h(Player* player, AreaTriggerEntry const* /*pAt*/)
126 {
127  // Player is deaed, a GM, no quest or hasn't got item: do nothing
128  if (!player->IsAlive() || player->IsGameMaster() ||
129  player->GetQuestStatus(QUEST_THE_SILVER_COVENANTS_SCHEME) == QUEST_STATUS_NONE ||
130  !player->HasItemCount(ITEM_SILVER_COVENANT_ORDERS, 1))
131  return false;
132 
133  if (world_map_northrend* northrend = dynamic_cast<world_map_northrend*>(player->GetInstanceData()))
134  if (northrend->IsDalaranCooldownForPlayer(player->GetObjectGuid()))
135  return false;
136 
137  CreatureList agents;
138  GetCreatureListWithEntryInGrid(agents, player, NPC_SILVER_COVENANT_AGENT, 50.0f);
139  bool found = false;
140  for (Creature* agent : agents)
141  {
142  if (agent->GetSpawnerGuid() == player->GetObjectGuid())
143  {
144  found = true;
145  break;
146  }
147  }
148  if (!found) // Summon Sunreaver Agent
149  {
150  if (Creature* silvercovenantagent = player->SummonCreature(NPC_SILVER_COVENANT_AGENT, m_silvercovenantagent.x, m_silvercovenantagent.y, m_silvercovenantagent.z, m_silvercovenantagent.o, TEMPSPAWN_TIMED_OOC_DESPAWN, 5 * MINUTE * IN_MILLISECONDS))
151  {
152  if (world_map_northrend* northrend = dynamic_cast<world_map_northrend*>(player->GetInstanceData()))
153  northrend->SetDalaranCooldownTime(player->GetObjectGuid());
154  silvercovenantagent->AI()->SendAIEvent(AI_EVENT_CUSTOM_EVENTAI_B, player, silvercovenantagent);
155  return true;
156  }
157  }
158  return false;
159 }
