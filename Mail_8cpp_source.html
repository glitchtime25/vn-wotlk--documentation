123 {
124  if (!m_mailTemplateId || !m_mailTemplateItemsNeed)
125  return false;
126 
127  m_mailTemplateItemsNeed = false;
128 
129  // mailLoot can be empty
130  Loot mailLoot(receiver, m_mailTemplateId, LOOT_MAIL);
131  LootItemList lootList;

133  mailLoot.GetLootItemsListFor(receiver, lootList);
134  for (LootItemList::const_iterator lootItr = lootList.begin(); lootItr != lootList.end(); ++lootItr)

145  }
146

201 void MailDraft::SendReturnToSender(uint32 sender_acc, ObjectGuid sender_guid, ObjectGuid receiver_guid)
202 {
203  Player* receiver = sObjectMgr.GetPlayer(receiver_guid);
204 
205  uint32 rc_account = 0;
206  if (!receiver)
207  rc_account = sObjectMgr.GetPlayerAccountIdByGUID(receiver_guid);
208 
209  if (!receiver && !rc_account) // sender not exist
210  {
211  deleteIncludedItems(true);
212  return;
213  }
214 
215  // prepare mail and send in other case
216  bool needItemDelay = false;
217 
218  if (!m_items.empty())
219  {
220  // if item send to character at another account, then apply item delivery delay
221  needItemDelay = sender_acc != rc_account;
222 
223  // set owner to new receiver (to prevent delete item with sender char deleting)
224  CharacterDatabase.BeginTransaction();
225  for (auto& m_item : m_items)
226  {
227  Item* item = m_item.second;
228  item->SaveToDB(); // item not in inventory and can be save standalone
229  // owner in data will set at mail receive and item extracting
230  CharacterDatabase.PExecute("UPDATE item_instance SET owner_guid = '%u' WHERE guid='%u'", receiver_guid.GetCounter(), item->GetGUIDLow());
231  }
232  CharacterDatabase.CommitTransaction();
233  }
234 
235  // If theres is an item, there is a one hour delivery delay.
236  uint32 deliver_delay = needItemDelay ? sWorld.getConfig(CONFIG_UINT32_MAIL_DELIVERY_DELAY) : 0;
237 
238  // will delete item or place to receiver mail list
239  SendMailTo(MailReceiver(receiver, receiver_guid), MailSender(MAIL_NORMAL, sender_guid.GetCounter()), MAIL_CHECK_MASK_RETURNED, deliver_delay);
240 }
249 void MailDraft::SendMailTo(MailReceiver const& receiver, MailSender const& sender, MailCheckMask checked, uint32 deliver_delay)
250 {
251  Player* pReceiver = receiver.GetPlayer(); // can be nullptr
252 
253  uint32 pReceiverAccount = 0;
254  if (!pReceiver)
255  pReceiverAccount = sObjectMgr.GetPlayerAccountIdByGUID(receiver.GetPlayerGuid());
256 
257  if (!pReceiver && !pReceiverAccount) // receiver not exist
258  {
259  deleteIncludedItems(true);
260  return;
261  }
262 
263  bool has_items = !m_items.empty();
264 
265  // generate mail template items for online player, for offline player items will generated at open
266  if (pReceiver)
267  {
268  if (prepareItems(pReceiver))
269  has_items = true;
270  }
271 
272  uint32 mailId = sObjectMgr.GenerateMailID();
273 
274  time_t deliver_time = time(nullptr) + deliver_delay;
275 
276  uint32 expire_delay;
277  // auction mail without any items and money (auction sale note) pending 1 hour
278  if (sender.GetMailMessageType() == MAIL_AUCTION && m_items.empty() && !m_money)
279  expire_delay = HOUR;
280  // default case: expire time if COD 3 days, if no COD 30 days
281  else
282  expire_delay = (m_COD > 0) ? 3 * DAY : 30 * DAY;
283 
284  time_t expire_time = deliver_time + expire_delay;
285 
286  // Add to DB
287  std::string safe_subject = GetSubject();
288  CharacterDatabase.escape_string(safe_subject);
289 
290  std::string safe_body = GetBody();
291  CharacterDatabase.escape_string(safe_body);
292 
293  CharacterDatabase.BeginTransaction();
294  CharacterDatabase.PExecute("INSERT INTO mail (id,messageType,stationery,mailTemplateId,sender,receiver,subject,body,has_items,expire_time,deliver_time,money,cod,checked) "
295  "VALUES ('%u', '%u', '%u', '%u', '%u', '%u', '%s', '%s', '%u', '" UI64FMTD "','" UI64FMTD "', '%u', '%u', '%u')",
296  mailId, sender.GetMailMessageType(), sender.GetStationery(), GetMailTemplateId(), sender.GetSenderId(), receiver.GetPlayerGuid().GetCounter(), safe_subject.c_str(), safe_body.c_str(), (has_items ? 1 : 0), (uint64)expire_time, (uint64)deliver_time, m_money, m_COD, checked);
297 
298  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
299  {
300  Item* item = mailItemIter->second;
301  CharacterDatabase.PExecute("INSERT INTO mail_items (mail_id,item_guid,item_template,receiver) VALUES ('%u', '%u', '%u','%u')",
302  mailId, item->GetGUIDLow(), item->GetEntry(), receiver.GetPlayerGuid().GetCounter());
303  }
304  CharacterDatabase.CommitTransaction();
305 
306  // For online receiver update in game mail status and data
307  if (pReceiver)
308  {
309  pReceiver->AddNewMailDeliverTime(deliver_time);
310 
311  Mail* m = new Mail;
312  m->messageID = mailId;
313  m->mailTemplateId = GetMailTemplateId();
314  m->subject = GetSubject();
315  m->body = GetBody();
316  m->money = GetMoney();
317  m->COD = GetCOD();
318 
319  for (MailItemMap::const_iterator mailItemIter = m_items.begin(); mailItemIter != m_items.end(); ++mailItemIter)
320  {
321  Item* item = mailItemIter->second;
322  m->AddItem(item->GetGUIDLow(), item->GetEntry());
323  }
324 
325  m->messageType = sender.GetMailMessageType();
326  m->stationery = sender.GetStationery();
327  m->sender = sender.GetSenderId();
328  m->receiverGuid = receiver.GetPlayerGuid();
329  m->expire_time = expire_time;
330  m->deliver_time = deliver_time;
331  m->checked = checked;
332  m->state = MAIL_STATE_UNCHANGED;
333 
334  pReceiver->AddMail(m); // to insert new mail to beginning of maillist
335 
336  if (!m_items.empty())
337  {
338  for (auto& m_item : m_items)
339  pReceiver->AddMItem(m_item.second);
340  }
341  }
342  else if (!m_items.empty())
343  deleteIncludedItems();
344 }
