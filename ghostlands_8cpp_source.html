156  DoBroadcastText(SAY_START, pCreature, pPlayer);

159  pCreature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_QUESTGIVER);
160  if (npc_ranger_lilathaAI* pEscortAI = dynamic_cast<npc_ranger_lilathaAI*>(pCreature->AI()))
161  pEscortAI->Start(false, pPlayer, pQuest, false, false, WAYPOINT_PATH_ID);
