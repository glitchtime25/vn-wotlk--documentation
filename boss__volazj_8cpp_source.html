194  void SpellHitTarget(Unit* pTarget, const SpellEntry* pSpell) override
195  {
196  if (pSpell->Id == SPELL_INSANITY && pTarget->GetTypeId() == TYPEID_PLAYER)
197  {
198  // Apply this only for the first target hit
199  if (!m_creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE))
200  {
201  DoCastSpellIfCan(m_creature, SPELL_INSANITY_VISUAL, CAST_TRIGGERED);
202  DoCastSpellIfCan(m_creature, SPELL_WHISPER_INSANITY, CAST_TRIGGERED);
203 
204  DoScriptText(SAY_INSANITY, m_creature);
205 
206  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
207  SetCombatMovement(false);
208 
209  if (m_pInstance)
210  m_pInstance->SetData(TYPE_VOLAZJ, SPECIAL);
211 
212  m_bIsInsanityInProgress = true;
213  }
214 
215  // Store the players in the instance, in order to better handle phasing
216  if (m_pInstance)
217  m_pInstance->SetData64(DATA_INSANITY_PLAYER, pTarget->GetObjectGuid());
218 
219  // Phase and summon a Visage for each player
220  pTarget->CastSpell(pTarget, aInsanityPhaseSpells[m_uiInsanityIndex], TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
221  pTarget->CastSpell(pTarget, aSpawnVisageSpells[m_uiInsanityIndex], TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
222  ++m_uiInsanityIndex;
223  }
224  }

226  void UpdateAI(const uint32 uiDiff) override
227  {
228  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
229  return;
230 
231  // Check for Insanity
232  if (m_bIsInsanityInProgress)
233  {
234  if (!m_creature->HasAura(SPELL_INSANITY_VISUAL))
235  {
236  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
237  SetCombatMovement(true);
238  m_bIsInsanityInProgress = false;
239  }
240 
241  // No other actions during insanity
242  return;
243  }
244 
245  if (m_creature->GetHealthPercent() < 100.0f - (float)m_uiCombatPhase * 33.4f)
246  {
247  if (DoCastSpellIfCan(m_creature, SPELL_INSANITY) == CAST_OK)
248  {
249  m_uiInsanityIndex = 0;
250  ++m_uiCombatPhase;
251  }
252  }
253 
254  if (m_uiMindFlayTimer < uiDiff)
255  {
256  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_MIND_FLAY : SPELL_MIND_FLAY_H) == CAST_OK)
257  m_uiMindFlayTimer = urand(10000, 20000);
258  }
259  else
260  m_uiMindFlayTimer -= uiDiff;
261 
262  if (m_uiShadowBoltTimer < uiDiff)
263  {
264  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SHADOW_BOLT : SPELL_SHADOW_BOLT_H) == CAST_OK)
265  m_uiShadowBoltTimer = urand(8000, 13000);
266  }
267  else
268  m_uiShadowBoltTimer -= uiDiff;
269 
270  if (m_uiShiverTimer < uiDiff)
271  {
272  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
273  {
274  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_SHIVER : SPELL_SHIVER_H) == CAST_OK)
275  m_uiShiverTimer = 30000;
276  }
277  }
278  else
279  m_uiShiverTimer -= uiDiff;
280 
281  DoMeleeAttackIfReady();
282  }
283 };
