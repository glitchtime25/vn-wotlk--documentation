100 void instance_oculus::SetData(uint32 uiType, uint32 uiData)
101 {
102  switch (uiType)
103  {
104  case TYPE_DRAKOS:
105  m_auiEncounter[TYPE_DRAKOS] = uiData;
106  if (uiData == DONE)
107  {
108  // Open all cages
109  for (GuidList::const_iterator itr = m_lCageDoorGUIDs.begin(); itr != m_lCageDoorGUIDs.end(); ++itr)
110  DoUseDoorOrButton(*itr);
111 
112  // Notes: The dialogue is handled by DB script
113  // Also the Centrifuge Constructs and the related npcs should be summoned - requires additional research
114 
115  // Activate the world state - the Centrifuge contructs should be loaded by now
116  DoUpdateWorldState(WORLD_STATE_CONSTRUCTS, 1);
117  DoUpdateWorldState(WORLD_STATE_CONSTRUCTS_COUNT, m_sConstructsAliveGUIDSet.size());
118 
119  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_EREGOS_ID);
120  }
121  break;
122  case TYPE_VAROS:
123  m_auiEncounter[TYPE_VAROS] = uiData;
124  if (uiData == DONE)
125  {
126  // Note: Image of Belgaristrasz dialogue is handled by DB script
127  DoSpawnNextBossIfCan();
128  DoUpdateWorldState(WORLD_STATE_CONSTRUCTS, 0);
129  }
130  break;
131  case TYPE_UROM:
132  m_auiEncounter[TYPE_UROM] = uiData;
133  // Note: Image of Belgaristrasz dialogue is handled by DB script
134  if (uiData == DONE)
135  DoSpawnNextBossIfCan();
136  break;
137  case TYPE_EREGOS:
138  m_auiEncounter[TYPE_EREGOS] = uiData;
139  // Note: Image of Belgaristrasz teleports to the Cache location and does more dialogue - requires additional research
140  if (uiData == DONE)
141  {
142  // The data about the cache isn't consistent, so it's better to handle both cases
143  DoToggleGameObjectFlags(instance->IsRegularDifficulty() ? GO_CACHE_EREGOS : GO_CACHE_EREGOS_H, GO_FLAG_NO_INTERACT, false);
144  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_CACHE_EREGOS : GO_CACHE_EREGOS_H, 30 * MINUTE);
145  }
146  break;
147  }
148 
149  if (uiData == DONE)
150  {
151  OUT_SAVE_INST_DATA;
152 
153  std::ostringstream saveStream;
154  saveStream << m_auiEncounter[TYPE_DRAKOS] << " " << m_auiEncounter[TYPE_VAROS] << " " << m_auiEncounter[TYPE_UROM] << " " << m_auiEncounter[TYPE_EREGOS];
155 
156  strInstData = saveStream.str();
157 
158  SaveToDB();
159  OUT_SAVE_INST_DATA_COMPLETE;
160  }
161 }

220 void instance_oculus::DoSpawnNextBossIfCan()
221 {
222  Player* pPlayer = GetPlayerInMap();
223  if (!pPlayer)
224  return;
225 
226  if (GetData(TYPE_UROM) == DONE)
227  {
228  // return if already summoned
229  if (GetSingleCreatureFromStorage(NPC_EREGOS, true))
230  return;
231 
232  pPlayer->SummonCreature(NPC_EREGOS, aOculusBossSpawnLocs[1][0], aOculusBossSpawnLocs[1][1], aOculusBossSpawnLocs[1][2], aOculusBossSpawnLocs[1][3], TEMPSPAWN_DEAD_DESPAWN, 0);
233  }
234  else if (GetData(TYPE_VAROS) == DONE)
235  {
236  // return if already summoned
237  if (GetSingleCreatureFromStorage(NPC_UROM, true))
238  return;
239 
240  pPlayer->SummonCreature(NPC_UROM, aOculusBossSpawnLocs[0][0], aOculusBossSpawnLocs[0][1], aOculusBossSpawnLocs[0][2], aOculusBossSpawnLocs[0][3], TEMPSPAWN_DEAD_DESPAWN, 0);

244 void instance_oculus::Load(const char* chrIn)
245 {
