43 bool ProcessEventId_event_spells_warlock_dreadsteed(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool bIsStart)
44 {
45  if (bIsStart && pSource->GetTypeId() == TYPEID_PLAYER)
46  {
47  if (instance_dire_maul* pInstance = (instance_dire_maul*)((Player*)pSource)->GetInstanceData())
48  {
49  // summon J'eevee and start event
50  if (uiEventId == EVENT_ID_SUMMON_JEEVEE)
51  {
52  if (pInstance->GetData(TYPE_DREADSTEED) == NOT_STARTED || pInstance->GetData(TYPE_DREADSTEED) == FAIL)
53  {
54  // start event: summon the dummy infernal controller and set in progress
55  ((Player*)pSource)->SummonCreature(NPC_WARLOCK_DUMMY_INFERNAL, -37.9392f, 812.805f, -29.4525f, 4.81711f, TEMPSPAWN_DEAD_DESPAWN, 0);
56 
57  // J'eevee starts on path 1 here. Path 0 is for Scholomance event
58  if (Creature* pImp = ((Player*)pSource)->SummonCreature(NPC_JEEVEE, -37.9392f, 812.805f, -29.4525f, 4.81711f, TEMPSPAWN_DEAD_DESPAWN, 0, true, true))
59  pImp->GetMotionMaster()->MoveWaypoint(1);
60 
61  pInstance->SetData(TYPE_DREADSTEED, IN_PROGRESS);
62 
63  // allow the rest to be handled by dbscript
64  return false;
65  }
66  }
67  // summon Dreadsteed
68  else if (uiEventId == EVENT_ID_SUMMON_DREADSTEED)
69  {
70  if (pInstance->GetData(TYPE_DREADSTEED) == SPECIAL)
71  {
72  pInstance->SetData(TYPE_DREADSTEED, DONE);
73 
74  // rest is done by DBscript
75  return false;
76  }
77  }
78  }
79  }
80  return true;
81 }
