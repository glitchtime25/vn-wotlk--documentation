48 void WorldSession::HandleMoveWorldportAckOpcode()
49 {
50  // ignore unexpected far teleports
51  if (!GetPlayer()->IsBeingTeleportedFar())
52  return;
53 
54  // get start teleport coordinates (will used later in fail case)
55  WorldLocation old_loc;
56  GetPlayer()->GetPosition(old_loc);
57 
58  // get the teleport destination
59  WorldLocation& loc = GetPlayer()->GetTeleportDest();
60 
61  // possible errors in the coordinate validity check (only cheating case possible)
62  if (!MapManager::IsValidMapCoord(loc.mapid, loc.coord_x, loc.coord_y, loc.coord_z, loc.orientation))
63  {
64  sLog.outError("WorldSession::HandleMoveWorldportAckOpcode: %s was teleported far to a not valid location "
65  "(map:%u, x:%f, y:%f, z:%f) We port him to his homebind instead..",
66  GetPlayer()->GetGuidStr().c_str(), loc.mapid, loc.coord_x, loc.coord_y, loc.coord_z);
67  // stop teleportation else we would try this again and again in LogoutPlayer...
68  GetPlayer()->SetSemaphoreTeleportFar(false);
69  // and teleport the player to a valid place
70  GetPlayer()->TeleportToHomebind();
71  return;
72  }
73 
74  // get the destination map entry, not the current one, this will fix homebind and reset greeting
75  MapEntry const* mEntry = sMapStore.LookupEntry(loc.mapid);
76 
77  auto returnHomeFunc = [this, player = GetPlayer(), old_loc, loc]()
78  {
79  Map* map = nullptr;
80  // must have map in teleport
81  if (!map)
82  map = sMapMgr.CreateMap(loc.mapid, player);
83  if (!map)
84  map = sMapMgr.CreateMap(old_loc.mapid, player);
85 
86  if (!map)
87  {
88  KickPlayer();
89  return;
90  }
91 
92  player->SetMap(map);
93 
94  player->SetSemaphoreTeleportFar(false);
95 
96  // Teleport to previous place, if cannot be ported back TP to homebind place
97  if (!player->TeleportTo(old_loc))
98  {
99  DETAIL_LOG("WorldSession::HandleMoveWorldportAckOpcode: %s cannot be ported to his previous place, teleporting him to his homebind place...",
100  player->GetGuidStr().c_str());
101  player->TeleportToHomebind();
102  }
103  };
104 
105  Map* map = nullptr;
106 
107  // prevent crash at attempt landing to not existed battleground instance
108  if (mEntry->IsBattleGroundOrArena())
109  {
110  if (GetPlayer()->GetBattleGroundId())
111  map = sMapMgr.FindMap(loc.mapid, GetPlayer()->GetBattleGroundId());
112 
113  if (!map)
114  {
115  DETAIL_LOG("WorldSession::HandleMoveWorldportAckOpcode: %s was teleported far to nonexisten battleground instance "
116  " (map:%u, x:%f, y:%f, z:%f) Trying to port him to his previous place..",
117  GetPlayer()->GetGuidStr().c_str(), loc.mapid, loc.coord_x, loc.coord_y, loc.coord_z);
118 
119  returnHomeFunc();
120  return;
121  }
122  }
123 
124  Difficulty diff = GetPlayer()->GetGroup() ? GetPlayer()->GetGroup()->GetDifficulty(mEntry->IsRaid()) : GetPlayer()->GetDifficulty(mEntry->IsRaid());
125  uint32 miscRequirement = 0;
126  if (AreaTrigger const* at = sObjectMgr.GetMapEntranceTrigger(loc.mapid))
127  {
128  if (AREA_LOCKSTATUS_OK != GetPlayer()->GetAreaTriggerLockStatus(at, diff, miscRequirement))
129  {
130  returnHomeFunc();
131  return;
132  }
133  }
134 
135  InstanceTemplate const* mInstance = ObjectMgr::GetInstanceTemplate(loc.mapid);
136 
137  // reset instance validity, except if going to an instance inside an instance
138  if (!GetPlayer()->m_InstanceValid && !mInstance)
139  GetPlayer()->m_InstanceValid = true;
140 
141  GetPlayer()->SetSemaphoreTeleportFar(false);
142 
143  // relocate the player to the teleport destination
144  if (!map)
145  map = sMapMgr.CreateMap(loc.mapid, GetPlayer());
146 
147  GetPlayer()->SetMap(map);
148 
149  // must be in wotlk before AlterTeleportLocation - TODO: Refactor BG entering so that BG script can alter transport movement info on bg enter and
150  // change phase mask without interrupt flag removing it
151  // SOTA must already send player inside bg on ship and in already correct phase mask else player falls through the ship on SOTA entry when attacker
152  _player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_ENTER_WORLD);
153 
154  if (_player->InBattleGround())
155  if (BattleGround* bg = _player->GetBattleGround())
156  if (_player->IsInvitedForBattleGroundInstance(_player->GetBattleGroundId()))
157  bg->AlterTeleportLocation(GetPlayer(), GetPlayer()->m_teleportTransport, loc.coord_x, loc.coord_y, loc.coord_z, loc.orientation);
158 
159  bool found = true;
160  if (GetPlayer()->m_teleportTransport)
161  {
162  found = false;
163  GetPlayer()->m_movementInfo.t_pos = Position(loc.coord_x, loc.coord_y, loc.coord_z, loc.orientation); // when teleporting onto transport, position is local coords
164  if (GenericTransport* transport = map->GetTransport(GetPlayer()->m_teleportTransport))
165  if (transport->GetMapId() == loc.mapid)
166  found = true;
167  }
168  else
169  GetPlayer()->Relocate(loc.coord_x, loc.coord_y, loc.coord_z, loc.orientation);
170  auto lambda = [this, loc, old_loc, mEntry, mInstance](Map* map)
171  {
172  if (GenericTransport* transport = map->GetTransport(GetPlayer()->m_teleportTransport))
173  {
174  if (transport->GetMapId() == loc.mapid)
175  {
176  transport->AddPassenger(GetPlayer(), false);
177  transport->UpdatePassengerPosition(GetPlayer());
178  if (GetPlayer()->IsVehicle()) // when teleporting onto transport, protocol doesnt support summoning accessories
179  GetPlayer()->GetVehicleInfo()->DisableAccessoryInit();
180  }
181  }
182  GetPlayer()->m_teleportTransport = ObjectGuid();
183 
184  GetPlayer()->SendInitialPacketsBeforeAddToMap();
185  // the CanEnter checks are done in TeleporTo but conditions may change
186  // while the player is in transit, for example the map may get full
187  if (!GetPlayer()->GetMap()->Add(GetPlayer()))
188  {
189  // if player wasn't added to map, reset his map pointer!
190  GetPlayer()->ResetMap();
191 
192  DETAIL_LOG("WorldSession::HandleMoveWorldportAckOpcode: %s was teleported far but couldn't be added to map "
193  " (map:%u, x:%f, y:%f, z:%f) Trying to port him to his previous place..",
194  GetPlayer()->GetGuidStr().c_str(), loc.mapid, loc.coord_x, loc.coord_y, loc.coord_z);
195 
196  // Teleport to previous place, if cannot be ported back TP to homebind place
197  if (!GetPlayer()->TeleportTo(old_loc))
198  {
199  DETAIL_LOG("WorldSession::HandleMoveWorldportAckOpcode: %s cannot be ported to his previous place, teleporting him to his homebind place...",
200  GetPlayer()->GetGuidStr().c_str());
201  GetPlayer()->TeleportToHomebind();
202  }
203  return;
204  }
205 
206  // battleground state prepare (in case join to BG), at relogin/tele player not invited
207  // only add to bg group and object, if the player was invited (else he entered through command)
208  if (_player->InBattleGround())
209  {
210  // cleanup setting if outdated
211  if (!mEntry->IsBattleGroundOrArena())
212  {
213  // We're not in BG
214  _player->SetBattleGroundId(0, BATTLEGROUND_TYPE_NONE);
215  // reset destination bg team
216  _player->SetBGTeam(TEAM_NONE);
217  }
218  // join to bg case
219  else if (BattleGround* bg = _player->GetBattleGround())
220  {
221  if (_player->IsInvitedForBattleGroundInstance(_player->GetBattleGroundId()))
222  bg->AddPlayer(_player);
223  }
224  }
225 
226  m_anticheat->Teleport({ loc.coord_x, loc.coord_y, loc.coord_z, loc.orientation });
227 
228  GetPlayer()->SendInitialPacketsAfterAddToMap();
229 
230  // flight fast teleport case
231  if (_player->InBattleGround())
232  _player->TaxiFlightInterrupt(false);
233  else
234  _player->TaxiFlightResume();
235 
236  if (mInstance)
237  {
238  // mount allow check
239  if (!mInstance->mountAllowed)
240  _player->RemoveSpellsCausingAura(SPELL_AURA_MOUNTED);
241  }
242 
243  // honorless target
244  if (GetPlayer()->pvpInfo.inPvPEnforcedArea)
245  GetPlayer()->CastSpell(GetPlayer(), 2479, TRIGGERED_OLD_TRIGGERED);
246 
247  _player->ResummonPetTemporaryUnSummonedIfAny();
248 
249  // lets process all delayed operations on successful teleport
250  GetPlayer()->ProcessDelayedOperations();
251 
252  // notify group after successful teleport
253  if (_player->GetGroup())
254  _player->SetGroupUpdateFlag(GROUP_UPDATE_FULL);
255  };
256  if (found)
257  lambda(map);
258  else
259  map->GetMessager().AddMessage(lambda);
260 }

262 void WorldSession::HandleMoveTeleportAckOpcode(WorldPacket& recv_data)
263 {
264  DEBUG_LOG("MSG_MOVE_TELEPORT_ACK");
265 
266  ObjectGuid guid;
267 
268  recv_data >> guid.ReadAsPacked();
269 
270  uint32 counter, time;
271  recv_data >> counter >> time;
272  DEBUG_LOG("Guid: %s", guid.GetString().c_str());
273  DEBUG_LOG("Counter %u, time %u", counter, time / IN_MILLISECONDS);
274 
275  Unit* mover = _player->GetMover();
276  Player* plMover = mover->GetTypeId() == TYPEID_PLAYER ? (Player*)mover : nullptr;
277 
278  if (!plMover || !plMover->IsBeingTeleportedNear())
279  return;
280 
281  if (guid != plMover->GetObjectGuid())
282  return;
283 
284  m_anticheat->OrderAck(recv_data.GetOpcode(), counter);
285 
286  plMover->SetSemaphoreTeleportNear(false);
287 
288  uint32 old_zone = plMover->GetZoneId();
289 
290  WorldLocation const& dest = plMover->GetTeleportDest();
291 
292  plMover->SetDelayedZoneUpdate(false, 0);
293 
294  plMover->SetPosition(dest.coord_x, dest.coord_y, dest.coord_z, dest.orientation, true);
295 
296  plMover->SetFallInformation(0, dest.coord_z);
297 
298  GenericTransport* currentTransport = nullptr;
299  if (plMover->m_teleportTransport)
300  currentTransport = plMover->GetMap()->GetTransport(plMover->m_teleportTransport);
301  if (currentTransport)
302  currentTransport->AddPassenger(plMover);
303  plMover->m_teleportTransport = ObjectGuid();
304 
305  uint32 newzone, newarea;
306  plMover->GetZoneAndAreaId(newzone, newarea);
307  plMover->UpdateZone(newzone, newarea);
308 
309  // new zone
310  if (old_zone != newzone)
311  {
312  // honorless target
313  if (plMover->pvpInfo.inPvPEnforcedArea)
314  plMover->CastSpell(plMover, 2479, TRIGGERED_OLD_TRIGGERED);
315  }
316 
317  m_anticheat->Teleport({ dest.coord_x, dest.coord_y, dest.coord_z, dest.orientation });
318 
319  // resummon pet
320  GetPlayer()->ResummonPetTemporaryUnSummonedIfAny();
321 
322  // lets process all delayed operations on successful teleport
323  GetPlayer()->ProcessDelayedOperations();
324 }

326 void WorldSession::HandleMovementOpcodes(WorldPacket& recv_data)
327 {
328  Opcodes opcode = recv_data.GetOpcode();
329  if (!sLog.HasLogFilter(LOG_FILTER_PLAYER_MOVES))
330  {
331  DEBUG_LOG("WORLD: Received opcode %s (%u, 0x%X)", LookupOpcodeName(opcode), opcode, opcode);
332  recv_data.hexlike();
333  }
334 
335  Unit* mover = _player->GetMover();
336  Player* plMover = mover->GetTypeId() == TYPEID_PLAYER ? (Player*)mover : nullptr;
337 
338  /* extract packet */
339  ObjectGuid guid;
340  MovementInfo movementInfo;
341 
342  recv_data >> guid.ReadAsPacked();
343  recv_data >> movementInfo;
344  /*----------------*/
345 
346  if (opcode == CMSG_MOVE_CHNG_TRANSPORT) // TODO: Should mark down pending transport change for anticheat purpose
347  {
348  if (guid != mover->GetObjectGuid() && _player->GetObjectGuid() == guid)
349  mover = _player;
350  }
351 
352  if (!ProcessMovementInfo(movementInfo, mover, plMover, recv_data))
353  return;
354 
355  // CMSG opcode has no handler in client, should not be sent to others.
356  // It is sent by client when you jump and hit something on the way up,
357  // thus stopping upward movement and causing you to descend sooner.
358  if (opcode == CMSG_MOVE_FALL_RESET)
359  return;
360 
361  WorldPacket data(opcode, recv_data.size());
362  data << mover->GetPackGUID(); // write guid
363  movementInfo.Write(data); // write data
364  mover->SendMessageToSetExcept(data, _player);
365 }

367 void WorldSession::HandleForceSpeedChangeAckOpcodes(WorldPacket& recv_data)
368 {
369  Opcodes opcode = recv_data.GetOpcode();
370  DEBUG_LOG("WORLD: Received %s (%u, 0x%X) opcode", recv_data.GetOpcodeName(), opcode, opcode);
371 
372  /* extract packet */
373  ObjectGuid guid;
374  uint32 counter;
375  MovementInfo movementInfo;
376  float newspeed;
377 
378  recv_data >> guid.ReadAsPacked();
379  recv_data >> counter; // counter or moveEvent
380  recv_data >> movementInfo;
381  recv_data >> newspeed;
382 
383  Unit* mover = _player->GetMover();
384 
385  if (!ProcessMovementInfo(movementInfo, mover, _player, recv_data))
386  return;
387 
388  m_anticheat->OrderAck(opcode, counter);
389 
390  if (opcode == CMSG_MOVE_SET_COLLISION_HGT_ACK)
391  {
392  WorldPacket data(MSG_MOVE_SET_COLLISION_HGT, 18);
393  data << guid.WriteAsPacked();
394  data << movementInfo;
395  data << newspeed; // new collision height
396  mover->SendMessageToSetExcept(data, _player);
397  return;
398  }
399 
400  /*----------------*/
401 
402  // Process anticheat checks, remember client-side speed ...
403  if (_player->IsSelfMover() && !m_anticheat->SpeedChangeAck(movementInfo, recv_data, newspeed))
404  return;
405 
406  // client ACK send one packet for mounted/run case and need skip all except last from its
407  // in other cases anti-cheat check can be fail in false case
408  UnitMoveType move_type;
409  UnitMoveType force_move_type;
410 
411  static char const* move_type_name[MAX_MOVE_TYPE] = { "Walk", "Run", "RunBack", "Swim", "SwimBack", "TurnRate", "Flight", "FlightBack", "PitchRate" };
412 
413  switch (opcode)
414  {
415  case CMSG_FORCE_WALK_SPEED_CHANGE_ACK: move_type = MOVE_WALK; force_move_type = MOVE_WALK; break;
416  case CMSG_FORCE_RUN_SPEED_CHANGE_ACK: move_type = MOVE_RUN; force_move_type = MOVE_RUN; break;
417  case CMSG_FORCE_RUN_BACK_SPEED_CHANGE_ACK: move_type = MOVE_RUN_BACK; force_move_type = MOVE_RUN_BACK; break;
418  case CMSG_FORCE_SWIM_SPEED_CHANGE_ACK: move_type = MOVE_SWIM; force_move_type = MOVE_SWIM; break;
419  case CMSG_FORCE_SWIM_BACK_SPEED_CHANGE_ACK: move_type = MOVE_SWIM_BACK; force_move_type = MOVE_SWIM_BACK; break;
420  case CMSG_FORCE_TURN_RATE_CHANGE_ACK: move_type = MOVE_TURN_RATE; force_move_type = MOVE_TURN_RATE; break;
421  case CMSG_FORCE_FLIGHT_SPEED_CHANGE_ACK: move_type = MOVE_FLIGHT; force_move_type = MOVE_FLIGHT; break;
422  case CMSG_FORCE_FLIGHT_BACK_SPEED_CHANGE_ACK: move_type = MOVE_FLIGHT_BACK; force_move_type = MOVE_FLIGHT_BACK; break;
423  case CMSG_FORCE_PITCH_RATE_CHANGE_ACK: move_type = MOVE_PITCH_RATE; force_move_type = MOVE_PITCH_RATE; break;
424  default:
425  sLog.outError("WorldSession::HandleForceSpeedChangeAck: Unknown move type opcode: %u", opcode);
426  return;
427  }
428 
429  const SpeedOpcodePair& speedOpcodes = SetSpeed2Opc_table[move_type];
430  WorldPacket data(speedOpcodes[2], 18);
431  data << guid.WriteAsPacked();
432  data << movementInfo;
433  data << newspeed;
434  mover->SendMessageToSetExcept(data, _player);
435 
436  // skip all forced speed changes except last and unexpected
437  // in run/mounted case used one ACK and it must be skipped.m_forced_speed_changes[MOVE_RUN} store both.
438  if (_player->m_forced_speed_changes[force_move_type] > 0)
439  {
440  --_player->m_forced_speed_changes[force_move_type];
441  if (_player->m_forced_speed_changes[force_move_type] > 0)
442  return;
443  }
444 
445  if (!_player->GetTransport() && fabs(mover->GetSpeed(move_type) - newspeed) > 0.01f)
446  {
447  if (mover->GetSpeed(move_type) > newspeed) // must be greater - just correct
448  {
449  sLog.outError("%sSpeedChange player %s is NOT correct (must be %f instead %f), force set to correct value",
450  move_type_name[move_type], _player->GetName(), mover->GetSpeed(move_type), newspeed);
451  mover->SetSpeedRate(move_type, _player->GetSpeedRate(move_type), true);
452  }
453  else // must be lesser - cheating
454  {
455  BASIC_LOG("Player %s from account id %u kicked for incorrect speed (must be %f instead %f)",
456  _player->GetName(), _player->GetSession()->GetAccountId(), mover->GetSpeed(move_type), newspeed);
457  _player->GetSession()->KickPlayer();
458  }
459  }
460 }

514 void WorldSession::HandleMoveKnockBackAck(WorldPacket& recv_data)
515 {
516  DEBUG_LOG("CMSG_MOVE_KNOCK_BACK_ACK");
517 
518  Unit* mover = _player->GetMover();
519 
520  // ignore, waiting processing in WorldSession::HandleMoveWorldportAckOpcode and WorldSession::HandleMoveTeleportAck
521  if (mover->IsPlayer() && static_cast<Player*>(mover)->IsBeingTeleported())
522  {
523  recv_data.rpos(recv_data.wpos()); // prevent warnings spam
524  return;
525  }
526 
527  ObjectGuid guid;
528  uint32 counter;
529  MovementInfo movementInfo;
530 
531  recv_data >> guid.ReadAsPacked();
532  recv_data >> counter; // knockback packets counter
533  recv_data >> movementInfo;
534 
535  m_anticheat->OrderAck(recv_data.GetOpcode(), counter);
536 
537  if (!ProcessMovementInfo(movementInfo, mover, _player, recv_data))
538  return;
539 
540  if (mover->IsPlayer() && static_cast<Player*>(mover)->IsFreeFlying())
541  mover->SetCanFly(true);
542 
543  WorldPacket data(MSG_MOVE_KNOCK_BACK, recv_data.size() + 15);
544  data << mover->GetPackGUID();
545  data << movementInfo;
546  data << movementInfo.jump.cosAngle;
547  data << movementInfo.jump.sinAngle;
548  data << movementInfo.jump.xyspeed;
549  data << movementInfo.jump.zspeed;
550  mover->SendMessageToSetExcept(data, _player);

822 void WorldSession::HandleTimeSyncResp(WorldPacket& recvData)
823 {
824  DEBUG_LOG("CMSG_TIME_SYNC_RESP");
825 
826  uint32 counter, clientTimestamp;
827  recvData >> counter >> clientTimestamp;
828 
829  if (m_pendingTimeSyncRequests.count(counter) == 0)
830  return;
831 
832  uint32 serverTimeAtSent = m_pendingTimeSyncRequests.at(counter);
833  m_pendingTimeSyncRequests.erase(counter);
834 
835  // time it took for the request to travel to the client, for the client to process it and reply and for response to travel back to the server.
836  // we are going to make 2 assumptions:
837  // 1) we assume that the request processing time equals 0.
838  // 2) we assume that the packet took as much time to travel from server to client than it took to travel from client to server.
839  uint32 roundTripDuration = WorldTimer::getMSTimeDiff(serverTimeAtSent, recvData.GetReceivedTime());
840  uint32 lagDelay = roundTripDuration / 2;
841 
842  /*
843  clockDelta = serverTime - clientTime
844  where
845  serverTime: time that was displayed on the clock of the SERVER at the moment when the client processed the SMSG_TIME_SYNC_REQUEST packet.
846  clientTime: time that was displayed on the clock of the CLIENT at the moment when the client processed the SMSG_TIME_SYNC_REQUEST packet.
847  Once clockDelta has been computed, we can compute the time of an event on server clock when we know the time of that same event on the client clock,
848  using the following relation:
849  serverTime = clockDelta + clientTime
850  */
851  int64 clockDelta = (int64)(serverTimeAtSent + lagDelay) - (int64)clientTimestamp;
852  m_timeSyncClockDeltaQueue.push_back(std::pair<int64, uint32>(clockDelta, roundTripDuration));
853  ComputeNewClockDelta();
854 }

856 void WorldSession::ComputeNewClockDelta()
857 {
858  // implementation of the technique described here: https://web.archive.org/web/20180430214420/http://www.mine-control.com/zack/timesync/timesync.html
859  // to reduce the skew induced by dropped TCP packets that get resent.
860 
861  using namespace boost::accumulators;
862 
863  accumulator_set<uint32, features<tag::mean, tag::median, tag::variance(lazy)> > latencyAccumulator;
864 
865  for (auto pair : m_timeSyncClockDeltaQueue)
866  latencyAccumulator(pair.second);
867 
868  uint32 latencyMedian = static_cast<uint32>(std::round(median(latencyAccumulator)));
869  uint32 latencyStandardDeviation = static_cast<uint32>(std::round(sqrt(variance(latencyAccumulator))));
870 
871  accumulator_set<int64, features<tag::mean> > clockDeltasAfterFiltering;
872  uint32 sampleSizeAfterFiltering = 0;
873  for (auto pair : m_timeSyncClockDeltaQueue)
874  {
875  if (pair.second < latencyStandardDeviation + latencyMedian)
876  {
877  clockDeltasAfterFiltering(pair.first);
878  sampleSizeAfterFiltering++;
879  }
880  }
881 
882  if (sampleSizeAfterFiltering != 0)
883  {
884  int64 meanClockDelta = static_cast<int64>(std::round(mean(clockDeltasAfterFiltering)));
885  if (std::abs(meanClockDelta - m_timeSyncClockDelta) > 25)
886  m_timeSyncClockDelta = meanClockDelta;
887  }
888  else if (m_timeSyncClockDelta == 0)
889  {
890  std::pair<int64, uint32> back = m_timeSyncClockDeltaQueue.back();
891  m_timeSyncClockDelta = back.first;
892  }
893 }
