63 void UpdateData::Compress(void* dst, uint32* dst_size, void* src, int src_size)
64 {
65  z_stream c_stream;
66 
67  c_stream.zalloc = (alloc_func)nullptr;
68  c_stream.zfree = (free_func)nullptr;
69  c_stream.opaque = (voidpf)nullptr;
70 
71  // default Z_BEST_SPEED (1)
72  int z_res = deflateInit(&c_stream, sWorld.getConfig(CONFIG_UINT32_COMPRESSION));
73  if (z_res != Z_OK)
74  {
75  sLog.outError("Can't compress update packet (zlib: deflateInit) Error code: %i (%s)", z_res, zError(z_res));
76  *dst_size = 0;
77  return;
78  }
79 
80  c_stream.next_out = (Bytef*)dst;
81  c_stream.avail_out = *dst_size;
82  c_stream.next_in = (Bytef*)src;
83  c_stream.avail_in = (uInt)src_size;
84 
85  z_res = deflate(&c_stream, Z_NO_FLUSH);
86  if (z_res != Z_OK)
87  {
88  sLog.outError("Can't compress update packet (zlib: deflate) Error code: %i (%s)", z_res, zError(z_res));
89  *dst_size = 0;
90  return;
91  }
92 
93  if (c_stream.avail_in != 0)
94  {
95  sLog.outError("Can't compress update packet (zlib: deflate not greedy)");
96  *dst_size = 0;
97  return;
98  }
99 
100  z_res = deflate(&c_stream, Z_FINISH);
101  if (z_res != Z_STREAM_END)
102  {
103  sLog.outError("Can't compress update packet (zlib: deflate should report Z_STREAM_END instead %i (%s)", z_res, zError(z_res));
104  *dst_size = 0;
105  return;
106  }
107 
108  z_res = deflateEnd(&c_stream);
109  if (z_res != Z_OK)
110  {
111  sLog.outError("Can't compress update packet (zlib: deflateEnd) Error code: %i (%s)", z_res, zError(z_res));
112  *dst_size = 0;
113  return;
114  }
115 
116  *dst_size = c_stream.total_out;
117 }

119 WorldPacket UpdateData::BuildPacket(size_t index)
120 {
121  WorldPacket packet;
122  MANGOS_ASSERT(packet.empty()); // shouldn't happen
123 
124  ByteBuffer buf(4 + (m_outOfRangeGUIDs.empty() ? 0 : 1 + 4 + 9 * m_outOfRangeGUIDs.size()) + m_data[index].m_buffer.wpos());
125 
126  buf << (uint32)(!m_outOfRangeGUIDs.empty() ? m_data[index].m_blockCount + 1 : m_data[index].m_blockCount);
127 
128  if (!m_outOfRangeGUIDs.empty())
129  {
130  buf << (uint8) UPDATETYPE_OUT_OF_RANGE_OBJECTS;
131  buf << (uint32) m_outOfRangeGUIDs.size();
132 
133  for (auto m_outOfRangeGUID : m_outOfRangeGUIDs)
134  buf << m_outOfRangeGUID.WriteAsPacked();
135  }
136 
137  buf.append(m_data[index].m_buffer);
138 
139  size_t pSize = buf.wpos(); // use real used data size
140 
141  if (pSize > 100) // compress large packets
142  {
143  uint32 destsize = compressBound(pSize);
144  packet.resize(destsize + sizeof(uint32));
145 
146  packet.put<uint32>(0, pSize);
147  Compress(const_cast<uint8*>(packet.contents()) + sizeof(uint32), &destsize, (void*)buf.contents(), pSize);
148  if (destsize == 0)
149  return packet;
150 
151  packet.resize(destsize + sizeof(uint32));
152  packet.SetOpcode(SMSG_COMPRESSED_UPDATE_OBJECT);
153  }
154  else // send small packets without compression
155  {
156  packet.append(buf);
157  packet.SetOpcode(SMSG_UPDATE_OBJECT);
158  }
159 
160  return packet;
161 }
