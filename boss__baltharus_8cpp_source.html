163  void ExecuteAction(uint32 action) override
164  {
165  switch (action)
166  {
167  case BALTHARUS_REPELLING_WAVE:
168  if (m_creature->GetHealthPercent() < 100 - m_fHealthPercentCheck * m_uiPhase)
169  {
170  if (DoCastSpellIfCan(m_creature, SPELL_REPELLING_WAVE) == CAST_OK)
171  {
172  ResetCombatAction(BALTHARUS_SUMMON_CLONE, 2000);
173  ++m_uiPhase;
174  }
175  }
176  break;
177  case BALTHARUS_ENERVATING_BRAND:
178  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
179  {
180  if (DoCastSpellIfCan(target, SPELL_ENERVATING_BRAND) == CAST_OK)
181  ResetCombatAction(action, 25000);
182  }
183  break;
184  case BALTHARUS_BLADE_TEMPEST:
185  if (DoCastSpellIfCan(m_creature, SPELL_BLADE_TEMPEST) == CAST_OK)
186  ResetCombatAction(action, 25000);
187  break;
188  case BALTHARUS_CLEAVE:
189  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
190  ResetCombatAction(action, 20000);
191  break;
192  case BALTHARUS_SUMMON_CLONE:
193  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_CLONE) == CAST_OK)
194  {
195  DoScriptText(SAY_SPLIT, m_creature);
196  DisableCombatAction(action);
197  }
198  break;
199  }
200  }
