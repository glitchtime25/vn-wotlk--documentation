79 void instance_halls_of_lightning::SetData(uint32 uiType, uint32 uiData)
80 {
81  switch (uiType)
82  {
83  case TYPE_BJARNGRIM:
84  if (uiData == SPECIAL)
85  m_bLightningStruck = true;
86  else if (uiData == FAIL)
87  m_bLightningStruck = false;
88  m_auiEncounter[uiType] = uiData;
89  break;
90  case TYPE_VOLKHAN:
91  if (uiData == DONE)
92  DoUseDoorOrButton(GO_VOLKHAN_DOOR);
93  else if (uiData == IN_PROGRESS)
94  m_bIsShatterResistant = true;
95  else if (uiData == SPECIAL)
96  m_bIsShatterResistant = false;
97  m_auiEncounter[uiType] = uiData;
98  break;
99  case TYPE_IONAR:
100  if (uiData == DONE)
101  DoUseDoorOrButton(GO_IONAR_DOOR);
102  m_auiEncounter[uiType] = uiData;
103  break;
104  case TYPE_LOKEN:
105  if (uiData == IN_PROGRESS)
106  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_LOKEN_ID);
107  else if (uiData == DONE)
108  {
109  if (GameObject* pGlobe = GetSingleGameObjectFromStorage(GO_LOKEN_THRONE))
110  pGlobe->SendGameObjectCustomAnim(pGlobe->GetObjectGuid());
111  }
112  m_auiEncounter[uiType] = uiData;
113  break;
114  }
115 
116  if (uiData == DONE)
117  {
118  OUT_SAVE_INST_DATA;
119 
120  std::ostringstream saveStream;
121  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
122 
123  m_strInstData = saveStream.str();
124 
125  SaveToDB();
126  OUT_SAVE_INST_DATA_COMPLETE;
127  }
128 }
