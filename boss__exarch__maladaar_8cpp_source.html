180  void Reset() override
181  {
182  m_targetGuid.Clear();
183  if (Creature* avatar = m_creature->GetMap()->GetCreature(m_avatar))
184  avatar->ForcedDespawn();
185 
186  m_uiFearTimer = urand(11000, 29000);
187  m_uiRibbonOfSoulsTimer = urand(4000, 8000);
188  m_uiStolenSoulTimer = urand(19000, 31000);

193  void MoveInLineOfSight(Unit* pWho) override
194  {
195  if (!m_bHasTaunted && pWho->GetTypeId() == TYPEID_PLAYER && m_creature->IsWithinDistInMap(pWho, 150.0f) && m_creature->IsWithinLOSInMap(pWho))
196  {
197  DoScriptText(SAY_INTRO, m_creature);
198  m_bHasTaunted = true;
199  }

201  ScriptedAI::MoveInLineOfSight(pWho);
202  }

204  void Aggro(Unit* /*pWho*/) override
205  {
206  switch (urand(0, 2))
207  {
208  case 0: DoScriptText(SAY_AGGRO_1, m_creature); break;
209  case 1: DoScriptText(SAY_AGGRO_2, m_creature); break;
210  case 2: DoScriptText(SAY_AGGRO_3, m_creature); break;
211  }
212  }
213 
214  void JustSummoned(Creature* pSummoned) override
215  {
216  switch (pSummoned->GetEntry())
217  {
218  case NPC_STOLEN_SOUL:
219  {
220  // SPELL_STOLEN_SOUL_VISUAL has shapeshift effect, but not implemented feature in mangos for this spell.
221  pSummoned->CastSpell(pSummoned, SPELL_STOLEN_SOUL_VISUAL, TRIGGERED_NONE);
222 
223  if (Player* pTarget = m_creature->GetMap()->GetPlayer(m_targetGuid))
224  {
225  if (mob_stolen_soulAI* pSoulAI = dynamic_cast<mob_stolen_soulAI*>(pSummoned->AI()))
226  pSoulAI->SetSoulInfo(pTarget);
227 
228  pSummoned->AI()->AttackStart(pTarget);
229  }
230  }
231  case NPC_AVATAR_MARTYRED:
232  {
233  pSummoned->CastSpell(pSummoned, SPELL_PHASE_IN, TRIGGERED_NONE);
234  m_avatar = pSummoned->GetObjectGuid();
235  }
236  }
237  }

255  void SpellHitTarget(Unit* pTarget, const SpellEntry* pSpell) override

261  void UpdateAI(const uint32 uiDiff) override
262  {
263  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
264  return;
265 
266  if (!m_bHasSummonedAvatar && m_creature->GetHealthPercent() < 25.0f)
267  {
268  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_AVATAR) == CAST_OK)
269  {
270  DoScriptText(SAY_SUMMON, m_creature);
271  m_bHasSummonedAvatar = true;
272  m_uiStolenSoulTimer = urand(15000, 30000);
273  }
274  }
275 
276  if (m_uiStolenSoulTimer < uiDiff)
277  {
278  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_STOLEN_SOUL, SELECT_FLAG_PLAYER))
279  {
280  if (DoCastSpellIfCan(pTarget, SPELL_STOLEN_SOUL) == CAST_OK)
281  {
282  if (urand(0, 1))
283  DoScriptText(urand(0, 1) ? SAY_ROAR : SAY_SOUL_CLEAVE, m_creature);
284 
285  m_targetGuid = pTarget->GetObjectGuid();
286 
287  m_uiStolenSoulTimer = urand(35000, 67000);
288  }
289  }
290  }
291  else
292  m_uiStolenSoulTimer -= uiDiff;
293 
294  if (m_uiRibbonOfSoulsTimer < uiDiff)
295  {
296  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
297  {
298  if (DoCastSpellIfCan(pTarget, SPELL_RIBBON_OF_SOULS) == CAST_OK)
299  m_uiRibbonOfSoulsTimer = urand(4000, 18000);
300  }
301  }
302  else
303  m_uiRibbonOfSoulsTimer -= uiDiff;
304 
305  if (m_uiFearTimer < uiDiff)
306  {
307  if (DoCastSpellIfCan(m_creature, SPELL_SOUL_SCREAM) == CAST_OK)
308  m_uiFearTimer = urand(13000, 30000);
309  }
310  else
311  m_uiFearTimer -= uiDiff;
312 
313  DoMeleeAttackIfReady();
314  }
315 };
