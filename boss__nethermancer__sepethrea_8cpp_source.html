102  void UpdateAI(const uint32 uiDiff) override
103  {
104  // Return since we have no target
105  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
106  return;
107 
108  // Arcane Blast
109  if (m_uiArcaneBlastTimer < uiDiff)
110  {
111  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ARCANE_BLAST) == CAST_OK)
112  {
113  m_uiArcaneBlastTimer = urand(15000, 30000);
114  m_creature->getThreatManager().modifyThreatPercent(m_creature->GetVictim(), -50.0f);
115  }
116  }
117  else
118  m_uiArcaneBlastTimer -= uiDiff;
119 
120  // Dragons Breath
121  if (m_uiDragonsBreathTimer < uiDiff)
122  {
123  if (DoCastSpellIfCan(m_creature, SPELL_DRAGONS_BREATH) == CAST_OK)
124  {
125  if (urand(0, 1))
126  DoScriptText(urand(0, 1) ? SAY_DRAGONS_BREATH_1 : SAY_DRAGONS_BREATH_2, m_creature);
127 
128  m_uiDragonsBreathTimer = urand(20000, 35000);
129  }
130  }
131  else
132  m_uiDragonsBreathTimer -= uiDiff;
133 
134  DoMeleeAttackIfReady();
135  }
