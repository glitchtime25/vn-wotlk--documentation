117  void JustDidDialogueStep(int32 entry) override
118  {
119  // Start combat after the dialogue is finished
120  if (entry == SPELL_SHADOWBLINK)
121  {
122  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
123  m_creature->SetFactionTemporary(FACTION_BLACK_DRAGON, TEMPFACTION_RESTORE_REACH_HOME | TEMPFACTION_RESTORE_RESPAWN);
124  DoCastSpellIfCan(nullptr, SPELL_NEFARIUS_BARRIER, CAST_TRIGGERED);
125  DoCastSpellIfCan(nullptr, SPELL_SHADOWBLINK);
126  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
127  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
128  m_creature->SetInCombatWithZone();
129  SetCombatMovement(false); // Nefarius teleports via Shadowblink, he does not move
130 
131  // Spawn the two spawner NPCs that will handle the drakonids spawning (both brood and chromatic)
132  if (m_instance)
133  {
134  // Left
135  m_creature->SummonCreature(m_instance->GetData(TYPE_NEFA_LTUNNEL), aNefarianLocs[0].m_fX, aNefarianLocs[0].m_fY, aNefarianLocs[0].m_fZ, 5.000f, TEMPSPAWN_TIMED_OOC_OR_CORPSE_DESPAWN, HOUR * IN_MILLISECONDS);
136  // Right
137  m_creature->SummonCreature(m_instance->GetData(TYPE_NEFA_RTUNNEL), aNefarianLocs[1].m_fX, aNefarianLocs[1].m_fY, aNefarianLocs[1].m_fZ, 5.000f, TEMPSPAWN_TIMED_OOC_OR_CORPSE_DESPAWN, HOUR * IN_MILLISECONDS);
138  }

147  void ExecuteAction(uint32 action) override
148  {
149  switch (action)
150  {
151  case NEFARIUS_SHADOW_BOLT:
152  {
153  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SHADOWBOLT, SELECT_FLAG_PLAYER))
154  if (DoCastSpellIfCan(target, SPELL_SHADOWBOLT) == CAST_OK)
155  ResetCombatAction(action, urand(2 * IN_MILLISECONDS, 4 * IN_MILLISECONDS));
156  break;
157  }
158  case NEFARIUS_FEAR:
159  {
160  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_FEAR, SELECT_FLAG_PLAYER))
161  if (DoCastSpellIfCan(target, SPELL_FEAR) == CAST_OK)
162  ResetCombatAction(action, urand(10 * IN_MILLISECONDS, 20 * IN_MILLISECONDS));
163  break;
164  }
165  case NEFARIUS_SHADOWBOLT_VOLLEY:
166  {
167  if (DoCastSpellIfCan(nullptr, SPELL_SHADOWBOLT_VOLLEY) == CAST_OK)
168  ResetCombatAction(action, urand(19 * IN_MILLISECONDS, 28 * IN_MILLISECONDS));
169  break;
170  }
171  case NEFARIUS_SILENCE:
172  {
173  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SILENCE, SELECT_FLAG_PLAYER))
174  if (DoCastSpellIfCan(target, SPELL_SILENCE) == CAST_OK)
175  ResetCombatAction(action, urand(14 * IN_MILLISECONDS, 23 * IN_MILLISECONDS));
176  break;
177  }
178  case NEFARIUS_SHADOW_COMMAND:
179  {
180  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_SHADOW_COMMAND, SELECT_FLAG_PLAYER))
181  if (DoCastSpellIfCan(target, SPELL_SHADOW_COMMAND) == CAST_OK)
182  ResetCombatAction(action, urand(24 * IN_MILLISECONDS, 30 * IN_MILLISECONDS));
183  break;
184  }
185  case NEFARIUS_SHADOW_BLINK:
186  {
187  if (DoCastSpellIfCan(nullptr, SPELL_SHADOWBLINK) == CAST_OK)
188  ResetCombatAction(action, urand(30 * IN_MILLISECONDS, 40 * IN_MILLISECONDS));
189  break;
190  }
191  }

214 bool GossipSelect_boss_victor_nefarius(Player* player, Creature* creature, uint32 /*uiSender*/, uint32 action)
215 {
216  if (creature->GetMapId() != MAP_ID_BWL)
217  return true;
218 
219  switch (action)
220  {
221  case GOSSIP_ACTION_INFO_DEF+1:
222  creature->HandleEmote(EMOTE_ONESHOT_TALK);
223  player->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, GOSSIP_ITEM_NEFARIUS_2, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
224  player->SEND_GOSSIP_MENU(GOSSIP_TEXT_NEFARIUS_2, creature->GetObjectGuid());
225  break;
226  case GOSSIP_ACTION_INFO_DEF+2:
227  creature->HandleEmote(EMOTE_ONESHOT_TALK);
228  player->ADD_GOSSIP_ITEM_ID(GOSSIP_ICON_CHAT, GOSSIP_ITEM_NEFARIUS_3, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 3);
229  player->SEND_GOSSIP_MENU(GOSSIP_TEXT_NEFARIUS_3, creature->GetObjectGuid());
230  break;
231  case GOSSIP_ACTION_INFO_DEF+3:
232  creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
233  player->CLOSE_GOSSIP_MENU();
234  // Start the intro event
235  if (boss_victor_nefariusAI* bossAI = dynamic_cast<boss_victor_nefariusAI*>(creature->AI()))
236  bossAI->DoStartIntro();
237  break;
238  }
239  return true;
240 }
