84 {

86 
87  std::unique_ptr<QueryResult> result(WorldDatabase.Query("SELECT dungeonId, maxLevel, firstQuestId, otherQuestId FROM lfg_dungeon_rewards ORDER BY dungeonId, maxLevel ASC"));
88 
89  uint32 count = 0;

94  sLog.outString(">> Loaded %u lfg_dungeon_rewards", count);
95  sLog.outString();

97  }

99  BarGoLink bar(result->GetRowCount());
100 
101  do
102  {
103  bar.step();
104  Field* fields = result->Fetch();
105  uint32 dungeonId = fields[0].GetUInt32();
106  uint32 maxLevel = fields[1].GetUInt8();
107  uint32 firstQuestId = fields[2].GetUInt32();
108  uint32 otherQuestId = fields[3].GetUInt32();
109 
110  if (!GetLFGDungeonEntry(dungeonId))
111  {
112  sLog.outErrorDb("Dungeon %u specified in table `lfg_dungeon_rewards` does not exist!", dungeonId);

114  }
115

119  maxLevel = sWorld.getConfig(CONFIG_UINT32_MAX_PLAYER_LEVEL);
120  }
121

126  }

129  {

131  otherQuestId = 0;
132  }
133

135  reward.maxLevel = maxLevel;
136  reward.firstQuest = firstQuestId;
137  reward.otherQuest = otherQuestId;
138  m_lfgRewards.emplace(dungeonId, reward);
139  ++count;
140  } while (result->NextRow());
141 
142  sLog.outString(">> Loaded %u lfg_dungeon_rewards", count);
143  sLog.outString();

151  for (uint32 i = 0; i < sLFGDungeonStore.GetNumRows(); ++i)
152  {

155  continue;

164  break;
165  }

169

173  BarGoLink bar(1);
174  bar.step();
175  sLog.outString(">> Loaded %u lfg_dungeon_template", count);

178  }
179

181 
182  do
183  {
184  bar.step();

188  if (itr == m_lfgDungeons.end())

195  data.x = fields[1].GetFloat();
196  data.y = fields[2].GetFloat();
197  data.z = fields[3].GetFloat();
198  data.o = fields[4].GetFloat();
199 
200  ++count;
201  } while (result->NextRow());
202 
203  // Fill all other teleport coords from areatriggers
204  for (auto itr = m_lfgDungeons.begin(); itr != m_lfgDungeons.end(); ++itr)

206  LFGDungeonData& dungeon = itr->second;
207 
208  // No teleport coords in database, load from areatriggers

211  AreaTrigger const* at = sObjectMgr.GetMapEntranceTrigger(dungeon.map);
212  if (!at)
213  {
214  sLog.outErrorDb("Failed to load dungeon %s, cant find areatrigger for map %u", dungeon.name.c_str(), dungeon.map);
215  continue;
216  }
217 
218  dungeon.map = at->target_mapId;
219  dungeon.x = at->target_X;
220  dungeon.y = at->target_Y;
221  dungeon.z = at->target_Z;
222  dungeon.o = at->target_Orientation;
223  }
224 
225  if (dungeon.type != LFG_TYPE_RANDOM_DUNGEON)
226  m_cachedDungeonMapsPerGroup[dungeon.group].insert(dungeon.id);
227  m_cachedDungeonMapsPerGroup[0].insert(dungeon.id);

257 void LFGMgr::FinishDungeon(Group* group, uint32 dungeonId, Map const* currMap)
258 {
259  LFGData& groupLfgData = group->GetLfgData();
260  uint32 gDungeonId = groupLfgData.GetDungeon();
261  if (gDungeonId != dungeonId)
262  {
263  sLog.outDebug("Group %s finished dungeon %u but queued for %u", group->GetObjectGuid().GetString().c_str(), dungeonId, gDungeonId);
264  return;
265  }
266 
267  if (groupLfgData.GetState() == LFG_STATE_FINISHED_DUNGEON) // Shouldn't happen. Do not reward multiple times
268  {
269  sLog.outDebug("Group: %s already rewarded", group->GetObjectGuid().GetString().c_str());
270  return;
271  }
272 
273  groupLfgData.SetState(LFG_STATE_FINISHED_DUNGEON);
274 
275  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
276  {
277  Player* player = itr->getSource();
278  ObjectGuid guid = player->GetObjectGuid();
279  LFGData& playerLfgData = player->GetLfgData();
280  if (playerLfgData.GetState() == LFG_STATE_FINISHED_DUNGEON)
281  {
282  sLog.outDebug("Group: %s, Player: %s already rewarded", group->GetObjectGuid().GetString().c_str(), guid.GetString().c_str());
283  continue;
284  }
285 
286  uint32 rDungeonId = playerLfgData.GetDungeon();
287 
288  playerLfgData.SetState(LFG_STATE_FINISHED_DUNGEON);
289 
290  // Give rewards only if its a random dungeon
291  LFGDungeonData const* dungeon = GetLFGDungeon(rDungeonId);
292 
293  if (!dungeon || (dungeon->type != LFG_TYPE_RANDOM_DUNGEON && !dungeon->seasonal))
294  {
295  sLog.outDebug("Group: %s, Player: %s dungeon %u is not random or seasonal", group->GetObjectGuid().GetString().c_str(), guid.GetString().c_str(), rDungeonId);
296  continue;
297  }
298 
299  if (player->GetMap() != currMap)
300  {
301  sLog.outDebug("Group: %s, Player: %s is in a different map", group->GetObjectGuid().GetString().c_str(), guid.GetString().c_str());
302  continue;
303  }
304 
305  player->RemoveAurasDueToSpell(LFG_SPELL_DUNGEON_COOLDOWN);
306 
307  LFGDungeonData const* dungeonDone = GetLFGDungeon(dungeonId);
308  uint32 mapId = dungeonDone ? uint32(dungeonDone->map) : 0;
309 
310  if (player->GetMapId() != mapId)
311  {
312  sLog.outDebug("Group: %s, Player: %s is in map %u and should be in %u to get reward", group->GetObjectGuid().GetString().c_str(), guid.GetString().c_str(), player->GetMapId(), mapId);
313  continue;
314  }
315 
316  // Update achievements
317  if (dungeon->difficulty == DUNGEON_DIFFICULTY_HEROIC)
318  {
319  uint8 lfdRandomPlayers = 0;
320  if (uint8 numParty = playerLfgData.GetCountAtJoin())
321  lfdRandomPlayers = 5 - numParty;
322  else
323  lfdRandomPlayers = 4;
324  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_USE_LFD_TO_GROUP_WITH_PLAYERS, lfdRandomPlayers);
325  }
326 
327  LfgReward const* reward = GetRandomDungeonReward(rDungeonId, player->GetLevel());
328  if (!reward)
329  continue;
330 
331  bool done = false;
332  Quest const* quest = sObjectMgr.GetQuestTemplate(reward->firstQuest);
333  if (!quest)
334  continue;
335 
336  // if we can take the quest, means that we haven't done this kind of "run", IE: First Heroic Random of Day.
337  if (player->CanRewardQuest(quest, false))
338  player->RewardQuest(quest, 0, nullptr, false);
339  else
340  {
341  done = true;
342  quest = sObjectMgr.GetQuestTemplate(reward->otherQuest);
343  if (!quest)
344  continue;
345  // we give reward without informing client (retail does this)
346  player->RewardQuest(quest, 0, nullptr, false);
347  }
348 
349  // Give rewards
350  sLog.outDebug("Group: %s, Player: %s done dungeon %u, %s previously done.", group->GetObjectGuid().GetString().c_str(), guid.GetString().c_str(), gDungeonId, done ? " " : " not");
351  LfgPlayerRewardData data = LfgPlayerRewardData(dungeon->Entry(), dungeonDone->Entry(), done, quest);
352  player->GetSession()->SendLfgPlayerReward(data);
353  }
354 }

363 void LFGMgr::GetCompatibleDungeons(LfgDungeonSet& dungeons, GuidSet const& players, LfgLockPartyMap& lockMap, LfgInstanceSaveMap& savedMap, bool isContinue)
364 {
365  lockMap.clear();
366 
367  for (GuidSet::const_iterator it = players.begin(); it != players.end() && !dungeons.empty(); ++it)
368  {
369  ObjectGuid guid = (*it);
370  Player* player = ObjectAccessor::FindPlayer(guid);
371  if (!player)
372  continue;
373  LfgLockMap const& cachedLockMap = GetLockedDungeons(player);
374  for (LfgLockMap::const_iterator it2 = cachedLockMap.begin(); it2 != cachedLockMap.end() && !dungeons.empty(); ++it2)
375  {
376  uint32 dungeonId = (it2->first & 0x00FFFFFF); // Compare dungeon ids
377  LfgDungeonSet::iterator itDungeon = dungeons.find(dungeonId);
378  if (itDungeon != dungeons.end())
379  {
380  bool eraseDungeon = true;
381 
382  // Don't remove the dungeon if team members are trying to continue a locked instance
383  if (it2->second == LFG_LOCKSTATUS_RAID_LOCKED && isContinue)
384  {
385  LFGDungeonData const* dungeon = GetLFGDungeon(dungeonId);
386  MANGOS_ASSERT(dungeon);
387  MANGOS_ASSERT(player);
388  if (InstancePlayerBind* playerBind = player->GetBoundInstance(dungeon->map, Difficulty(dungeon->difficulty)))
389  {
390  if (DungeonPersistentState* state = playerBind->state)
391  {
392  ObjectGuid dungeonInstanceGuid = state->GetInstanceGuid();
393  auto itLockedDungeon = savedMap.find(dungeonId);
394  if (itLockedDungeon == savedMap.end() || itLockedDungeon->second.instanceGuid == dungeonInstanceGuid)
395  eraseDungeon = false;
396 
397  auto& lockedDungeon = savedMap[dungeonId];
398  lockedDungeon.instanceGuid = dungeonInstanceGuid;
399  lockedDungeon.completedEncountersMask = state->GetCompletedEncountersMask();
400  }
401  }
402  }
403 
404  if (eraseDungeon)
405  dungeons.erase(itDungeon);
406 
407  lockMap[guid][dungeonId] = it2->second;
408  }
409  }
410  }
411  if (!dungeons.empty())
412  lockMap.clear();
413 }

499  lockData = LFG_LOCKSTATUS_MISSING_ITEM;
500  }

507  lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_LOW_LEVEL;
508  lockData = LFG_LOCKSTATUS_ATTUNEMENT_TOO_HIGH_LEVEL;
509  */
510 
511  if (lockData)
512  lock[dungeon->Entry()] = lockData;
513  }
514 
515  return lock;
516 
517  return lock;

541  // At least 1 role must be selected

547
