89  void MovementInform(uint32 uiType, uint32 uiPointId)
90  {
91  if (uiType != POINT_MOTION_TYPE || !uiPointId)
92  return;
93 
94  if (uiPointId == POINT_MOVE_SUNBLADE_PROTECTOR)
95  {
96  SetCombatScriptStatus(false);

98  SetMeleeEnabled(true);

104  void ExecuteAction(uint32 action)
105  {
106  switch (action)
107  {
108  case SUNBLADE_SCOUT_ACTIVATE_SUNBLADE_PROTECTOR:
109  {
110  // Move to closest Sunblade Protector
111  if (Creature* Protector = GetClosestCreatureWithEntry(m_creature, NPC_SUNBLADE_PROTECTOR, 150.0f, true))
112  if (Protector->HasAura(SPELL_STONED))
113  {
114  DoScriptText(SAY_SCOUT_AGGRO, m_creature);
115  SetCombatScriptStatus(true);
116  SetCombatMovement(false);
117  SetMeleeEnabled(false);
118 
119  if (m_creature->IsWithinDist(Protector, CAST_DISTANCE))
120  {
121  SetCombatScriptStatus(false);
122  SetCombatMovement(true);
123  SetMeleeEnabled(true);
124  DoStartMovement(m_creature->GetVictim());
125  m_creature->SetInFront(Protector);
126  DoCastSpellIfCan(nullptr, SPELL_ACTIVATE_SUNBLADE_PROTECTOR);
127  }
128  else
129  {
130  float fX, fY, fZ;
131  Protector->GetContactPoint(m_creature, fX, fY, fZ, CAST_DISTANCE);
132  m_creature->GetMotionMaster()->MovePoint(POINT_MOVE_SUNBLADE_PROTECTOR, fX, fY, fZ, FORCED_MOVEMENT_RUN, true);
133  }
134  SetActionReadyStatus(SUNBLADE_SCOUT_ACTIVATE_SUNBLADE_PROTECTOR, false);
135  break;
136  }
137  break;
138  }
139  case SUNBLADE_SCOUT_SINISTER_STRIKE:
140  {
141  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SINISTER_STRIKE) == CAST_OK)
142  ResetCombatAction(action, urand(10000, 19000));
143  break;
144  }
145  }
146  }
