104 struct ZombieChowSearch : AuraScript
105 {
106  void OnPeriodicTrigger(Aura* aura, PeriodicTriggerData& data) const override
107  {
108  Creature* caster = static_cast<Creature*>(aura->GetCaster());
109  if (!caster)
110  return;
111  if (!caster->AI())
112  return;
113  CreatureList zombiesInRange;
114  GetCreatureListWithEntryInGrid(zombiesInRange, caster, NPC_ZOMBIE_CHOW, 10.f);
115  for (auto zombieItr = zombiesInRange.begin();zombieItr != zombiesInRange.end();)
116  {
117  if (!(*zombieItr) || !(*zombieItr)->IsAlive())
118  zombieItr = zombiesInRange.erase(zombieItr);
119  else
120  zombieItr++;
121  }
122 
123  if (zombiesInRange.empty())
124  return;
125  else if(zombiesInRange.size() == 1)
126  {
127  Creature* pZombie = zombiesInRange.front();
128  if (!pZombie)
129  return;
130  caster->GetMotionMaster()->MoveCharge(*pZombie, 30.f, EVENT_CHARGE);
131  caster->AI()->DoCastSpellIfCan(pZombie, SPELL_ZOMBIE_CHOW_SEARCH_INSTAKILL_TARGETED);
132  DoBroadcastText(EMOTE_ZOMBIE, caster);
133  return;
134  }
135  caster->CastSpell(nullptr, SPELL_ZOMBIE_CHOW_SEARCH_INSTAKILL_AOE, TRIGGERED_OLD_TRIGGERED);
136  DoBroadcastText(EMOTE_ZOMBIES, caster);
137  }
138 };

150 
151  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
152  {
153  if (effIdx != EFFECT_INDEX_0)
154  return;
155  Unit* unitTarget = spell->GetUnitTarget();
156  if (!unitTarget)
157  return;
158  if (!unitTarget->IsAlive())
159  return;
160  int32 damage = unitTarget->GetHealth() - unitTarget->GetMaxHealth() * 0.05f;
161  if (damage > 0)
162  spell->GetCaster()->CastCustomSpell(unitTarget, SPELL_DECIMATE_DAMAGE, &damage, nullptr, nullptr, TRIGGERED_INSTANT_CAST);
163  if (unitTarget->IsPlayer() || unitTarget->IsControlledByPlayer())
164  return;
165  if (!unitTarget->AI())
166  return;
167  unitTarget->AttackStop(true, true);
168  unitTarget->AI()->SetReactState(REACT_PASSIVE);
169  unitTarget->GetMotionMaster()->MovePoint(0, spell->GetCaster()->GetPosition(), FORCED_MOVEMENT_WALK);
170  }

173 struct SummonZombieChow : AuraScript
174 {
175  void OnPeriodicTrigger(Aura* aura, PeriodicTriggerData& data) const override
176  {
177  data.spellInfo = nullptr;
178  Creature* caster = static_cast<Creature*>(aura->GetCaster());
179  if (!caster)
180  return;
181  instance_naxxramas* instance = dynamic_cast<instance_naxxramas*>(caster->GetInstanceData());
182  if (!instance)
183  return;
184 
185  GuidVector gluthTriggers = instance->GetGluthTriggers();
186  if (gluthTriggers.size() < 3)
187  return;
188 
189  std::shuffle(gluthTriggers.begin(), gluthTriggers.end(), *GetRandomGenerator());
190 
191  if (caster->GetMap()->GetDifficulty() == RAID_DIFFICULTY_10MAN_NORMAL)
192  gluthTriggers.resize(1);
193  else
194  gluthTriggers.resize(2);
195 
196  for (auto& guid : gluthTriggers)
197  {
198  if (Creature* triggerNpc = caster->GetMap()->GetCreature(guid))
199  triggerNpc->CastSpell(triggerNpc, SPELL_SUMMON_ZOMBIE_CHOW_TR, TRIGGERED_INSTANT_CAST);
