216  void UpdateAI(const uint32 uiDiff) override
217  {
218  if (m_uiEpilogueTimer)
219  {
220  if (m_uiEpilogueTimer <= uiDiff)
221  {
222  switch (m_uiEpilogueStage)
223  {
224  case 0:
225  if (m_pInstance)
226  m_pInstance->SetData(TYPE_HODIR, DONE);
227 
228  DoScriptText(SAY_EPILOGUE, m_creature);
229  m_uiEpilogueTimer = 10000;
230  break;
231  case 1:
232  if (DoCastSpellIfCan(m_creature, SPELL_TELEPORT) == CAST_OK)
233  {
234  m_creature->ForcedDespawn(2000);
235  m_uiEpilogueTimer = 0;
236  }
237  break;
238  }
239  ++m_uiEpilogueStage;
240  }
241  else
242  m_uiEpilogueTimer -= uiDiff;
243  }
244 
245  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
246  return;
247 
248  if (m_uiBerserkTimer)
249  {
250  if (m_uiBerserkTimer <= uiDiff)
251  {
252  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
253  {
254  DoScriptText(SAY_BERSERK, m_creature);
255  m_uiBerserkTimer = 0;
256  }
257  }
258  else
259  m_uiBerserkTimer -= uiDiff;
260  }
261 
262  if (m_uiFlashFreezeTimer < uiDiff)
263  {
264  if (DoCastSpellIfCan(m_creature, SPELL_FLASH_FREEZE) == CAST_OK)
265  {
266  DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ICICLE_SNOWPACK : SPELL_ICICLE_SNOWPACK_H, CAST_TRIGGERED);
267  DoScriptText(EMOTE_FLASH_FREEZE, m_creature);
268  DoScriptText(SAY_FLASH_FREEZE, m_creature);
269  m_uiFlashFreezeTimer = 50000;
270  }
271  }
272  else
273  m_uiFlashFreezeTimer -= uiDiff;
274 
275  if (m_uiFrozenBlowsTimer < uiDiff)
276  {
277  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_FROZEN_BLOWS : SPELL_FROZEN_BLOWS_H) == CAST_OK)
278  {
279  DoScriptText(SAY_FROZEN_BLOWS, m_creature);
280  DoScriptText(EMOTE_FROZEN_BLOWS, m_creature);
281  m_uiFrozenBlowsTimer = 60000;
282  }
283  }
284  else
285  m_uiFrozenBlowsTimer -= uiDiff;
286 
287  if (m_uiFreezeTimer < uiDiff)
288  {
289  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
290  {
291  if (DoCastSpellIfCan(pTarget, SPELL_FREEZE) == CAST_OK)
292  m_uiFreezeTimer = 15000;
293  }
294  }
295  else
296  m_uiFreezeTimer -= uiDiff;
297 
298  DoMeleeAttackIfReady();
299  }

335  {
336  if (Creature* pHodir = m_pInstance->GetSingleCreatureFromStorage(NPC_HODIR))

348  void UpdateAI(const uint32 /*uiDiff*/) override
349  {
350  // Flash Freeze npcs should be always be summoned
351  if (!m_creature->IsTemporarySummon())
352  return;
353 
354  // do the freezing on the first update tick
355  if (!m_bFreezeInit)
356  {
357  // Flash Freeze npc will always stun or kill the summoner
358  if (m_creature->GetEntry() == NPC_FLASH_FREEZE_NPC)
359  {
360  if (DoCastSpellIfCan(m_creature, SPELL_FLASH_FREEZE_AURA_NPC) == CAST_OK)
361  DoCastSpellIfCan(m_creature, SPELL_FLASH_FREEZE_INITIAL, CAST_TRIGGERED);
362  }
363  else if (m_creature->GetEntry() == NPC_FLASH_FREEZE)
364  {
365  if (Unit* pSummoner = m_creature->GetMap()->GetUnit(m_creature->GetSpawnerGuid()))
366  {
367  // kill frozen players
368  if (pSummoner->HasAura(SPELL_FREEZE))
369  DoCastSpellIfCan(pSummoner, SPELL_FLASH_FREEZE_KILL);
370  else
371  DoCastSpellIfCan(m_creature, SPELL_FLASH_FREEZE_AURA);
372 
373  if (pSummoner->GetTypeId() == TYPEID_PLAYER && m_pInstance)
374  m_pInstance->SetSpecialAchievementCriteria(TYPE_ACHIEV_CHEESE_FREEZE, false);
375  }
376  }
377 
378  m_bFreezeInit = true;
379  }
380  }

392 bool ProcessEventId_event_boss_hodir(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool /*bIsStart*/)
393 {
394  if (pSource->GetTypeId() == TYPEID_UNIT)
395  {
396  instance_ulduar* pInstance = (instance_ulduar*)((Creature*)pSource)->GetInstanceData();
397  if (!pInstance)
398  return true;
399 
400  if (uiEventId == EVENT_ID_SHATTER_CHEST)
401  {
402  // Mark hard mode as failed and despawn the Rare cache
403  pInstance->SetData(TYPE_HODIR_HARD, FAIL);
404 
405  if (GameObject* pChest = pInstance->GetSingleGameObjectFromStorage(pInstance->instance->IsRegularDifficulty() ? GO_CACHE_OF_RARE_WINTER_10 : GO_CACHE_OF_RARE_WINTER_25))
406  pChest->SetLootState(GO_JUST_DEACTIVATED);
407  }
408  else if (uiEventId == EVENT_ID_ATTACK_START)
409  {
410  // Start encounter
411  if (Creature* pHodir = pInstance->GetSingleCreatureFromStorage(NPC_HODIR))
412  {
413  // ignore if event already completed
414  if (pHodir->GetFaction() == FACTION_ID_FRIENDLY)
415  return true;
416 
417  pHodir->SetInCombatWithZone();
418  }
419  }
420 
421  return true;
422  }
423 
424  return false;
425 }
