58 
59 bool GOUse_go_gauntlet_gate(Player* pPlayer, GameObject* pGo)
60 {
61  ScriptedInstance* pInstance = (ScriptedInstance*)pGo->GetInstanceData();
62 
63  if (!pInstance)
64  return false;
65 
66  if (pInstance->GetData(TYPE_BARON_RUN) != NOT_STARTED)
67  return false;
68 
69  if (Group* pGroup = pPlayer->GetGroup())
70  {
71  for (GroupReference* itr = pGroup->GetFirstMember(); itr != nullptr; itr = itr->next())
72  {
73  Player* pGroupie = itr->getSource();
74  if (!pGroupie)
75  continue;
76 
77  if (!pGroupie->HasAura(SPELL_BARON_ULTIMATUM))
78  pGroupie->CastSpell(pGroupie, SPELL_BARON_ULTIMATUM, TRIGGERED_OLD_TRIGGERED);
79  }
80  }
81  else
82  {
83  if (!pPlayer->HasAura(SPELL_BARON_ULTIMATUM))
84  pPlayer->CastSpell(pPlayer, SPELL_BARON_ULTIMATUM, TRIGGERED_OLD_TRIGGERED);
85  }
86 
87  pInstance->SetData(TYPE_BARON_RUN, IN_PROGRESS);
88  return false;
89 }

94 
95 bool GOUse_go_stratholme_postbox(Player* pPlayer, GameObject* pGo)
96 {
97  ScriptedInstance* pInstance = (ScriptedInstance*)pGo->GetInstanceData();
98 
99  if (!pInstance)
100  return false;
101 
102  if (pInstance->GetData(TYPE_POSTMASTER) == DONE)
103  return false;
104 
105  // When the data is Special, spawn the postmaster
106  if (pInstance->GetData(TYPE_POSTMASTER) == SPECIAL)
107  {
108  pPlayer->CastSpell(pPlayer, SPELL_SUMMON_POSTMASTER, TRIGGERED_OLD_TRIGGERED);
109  pInstance->SetData(TYPE_POSTMASTER, DONE);
110  }
111  else
112  pInstance->SetData(TYPE_POSTMASTER, IN_PROGRESS);
113 
114  // Summon 3 postmen for each postbox
115  float fX, fY, fZ;
116  for (uint8 i = 0; i < 3; ++i)
117  {
118  pPlayer->GetRandomPoint(pPlayer->GetPositionX(), pPlayer->GetPositionY(), pPlayer->GetPositionZ(), 3.0f, fX, fY, fZ);
119  pPlayer->SummonCreature(NPC_UNDEAD_POSTMAN, fX, fY, fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 0);
120  }
121 
122  return false;
123 }
