282  void DoInitializeAdds()
283  {
284  // not if m_creature are dead, so avoid
285  if (!m_creature->IsAlive())
286  return;
287 
288  auto& addsList = m_instance->GetMalacrassAddsEntryList();
289 
290  // it's empty, so first time
291  if (addsList.empty())
292  {
293  addsList.resize(MAX_ACTIVE_ADDS);
294 
295  for (uint8 i = 0; i < MAX_ACTIVE_ADDS; ++i)
296  {
297  uint8 addVersion = urand(0, 1);
298  addsList[i] = aSpawnEntries[i][addVersion];
299  Creature* creature = m_creature->SummonCreature(aSpawnEntries[i][addVersion], m_aAddPositions[i][0], m_aAddPositions[i][1], m_aAddPositions[i][2], m_aAddPositions[i][3], TEMPSPAWN_DEAD_DESPAWN, 0);
300  creature->SetCorpseDelay(5);
301  }
302  }
303  // Resummon the killed adds
304  else
305  {
306  if (!m_instance)
307  return;
308 
309  for (uint8 i = 0; i < MAX_ACTIVE_ADDS; ++i)
310  {
311  // If we already have the creature on the map, then don't summon it
312  if (m_instance->GetSingleCreatureFromStorage(addsList[i], true))
313  continue;
314 
315  m_creature->SummonCreature(addsList[i], m_aAddPositions[i][0], m_aAddPositions[i][1], m_aAddPositions[i][2], m_aAddPositions[i][3], TEMPSPAWN_DEAD_DESPAWN, 0);
316  }
317  }
318  }

346  void SpellHitTarget(Unit* target, const SpellEntry* spellInfo) override
347  {
348  // Set the player's class when hit with soul siphon
349  if (target->GetTypeId() == TYPEID_PLAYER && spellInfo->Id == SPELL_SIPHON_SOUL)
350  {
351  m_playerClass = target->getClass();
352 
353  if (m_playerClass == CLASS_WARRIOR || m_playerClass == CLASS_ROGUE)
354  m_creature->ApplySpellImmune(nullptr, IMMUNITY_MECHANIC, MECHANIC_DISARM, false);
355 
356  for (uint8 i = 0; i < 3; ++i)
357  {
358  switch (aMalacrassStolenAbility[m_playerClass][i].m_uiSpellId)
359  {
360  case SPELL_HU_EXPLOSIVE_TRAP:
361  case SPELL_HU_FREEZING_TRAP:
362  case SPELL_HU_SNAKE_TRAP:
363  ResetCombatAction(MALACRASS_PLAYER_ABILITY_1 + i, urand(1000, 25000));
364  break;
365  default:
366  ResetCombatAction(MALACRASS_PLAYER_ABILITY_1 + i, aMalacrassStolenAbility[m_playerClass][i].m_uiInitialTimer);
367  break;
368  }
369  }
370  }

373  bool CanUseSpecialAbility(uint32 spellIndex)
374  {
375  Unit* target = nullptr;
376  bool requireTarget = true;
377 
378  switch (aMalacrassStolenAbility[m_playerClass][spellIndex].m_uiTargetType)
379  {
380  case TARGET_TYPE_SELF:
381  target = nullptr;
382  requireTarget = false;
383  break;
384  case TARGET_TYPE_VICTIM:
385  target = m_creature->GetVictim();
386  break;
387  case TARGET_TYPE_RANDOM:
388  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER);
389  break;
390  case TARGET_TYPE_FRIENDLY:
391  target = DoSelectLowestHpFriendly(50.0f);
392  break;
393  case TARGET_TYPE_RANDOM_NOT_MAIN:
394  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK);
395  break;
396  }
397 
398  if (!requireTarget || target)
399  {
400  if (DoCastSpellIfCan(target, aMalacrassStolenAbility[m_playerClass][spellIndex].m_uiSpellId) == CAST_OK)
401  return true;
402  }
403 
404  return false;
405  }

407  void ExecuteAction(uint32 action) override
408  {
409  switch (action)
410  {
411  case MALACRASS_SPIRIT_BOLTS:
412  if (DoCastSpellIfCan(nullptr, SPELL_SPIRIT_BOLTS) == CAST_OK)
413  {
414  DoScriptText(SAY_SPIRIT_BOLTS, m_creature);
415  DisableCombatAction(MALACRASS_PLAYER_ABILITY_1);
416  DisableCombatAction(MALACRASS_PLAYER_ABILITY_2);
417  DisableCombatAction(MALACRASS_PLAYER_ABILITY_3);
418  DisableCombatAction(MALACRASS_PLAYER_ABILITY_4);
419  DisableCombatAction(action);
420  ResetCombatAction(MALACRASS_SIPHON_SOUL, 0);
421  if (m_playerClass == CLASS_WARRIOR || m_playerClass == CLASS_ROGUE)
422  m_creature->ApplySpellImmune(nullptr, IMMUNITY_MECHANIC, MECHANIC_DISARM, true);
423  }
424  break;
425  case MALACRASS_DRAIN_POWER_ENABLE:
426  if (m_creature->GetHealthPercent() < 80.0f && DoCastSpellIfCan(nullptr, SPELL_DRAIN_POWER) == CAST_OK)
427  {
428  SetActionReadyStatus(action, false);
429  DoScriptText(SAY_DRAIN_POWER, m_creature);
430  ResetCombatAction(MALACRASS_DRAIN_POWER, 30000);
431  }
432  break;
433  case MALACRASS_DRAIN_POWER:
434  if (DoCastSpellIfCan(nullptr, SPELL_DRAIN_POWER) == CAST_OK)
435  {
436  DoScriptText(SAY_DRAIN_POWER, m_creature);
437  ResetCombatAction(action, 30000);
438  }
439  break;
440  case MALACRASS_SIPHON_SOUL:
441  if (DoCastSpellIfCan(nullptr, SPELL_SIPHON_SOUL_DUMMY) == CAST_OK)
442  {
443  DoScriptText(SAY_SOUL_SIPHON, m_creature);
444  DisableCombatAction(action);
445  ResetCombatAction(MALACRASS_SPIRIT_BOLTS, 30000);
446  }
447  break;
448  case MALACRASS_PLAYER_ABILITY_1:
449  case MALACRASS_PLAYER_ABILITY_2:
450  case MALACRASS_PLAYER_ABILITY_3:
451  case MALACRASS_PLAYER_ABILITY_4:
452  if (CanUseSpecialAbility(action - MALACRASS_PLAYER_ABILITY_1))
453  ResetCombatAction(action, aMalacrassStolenAbility[m_playerClass][action - MALACRASS_PLAYER_ABILITY_1].m_uiCooldown);
454  break;
455  }
456  }
