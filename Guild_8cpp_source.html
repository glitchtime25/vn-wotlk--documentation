222  !((1 << (newmember.Class - 1)) & CLASSMASK_ALL_PLAYABLE))
223  {
224  sLog.outError("%s has a broken data in field `characters` table, cannot add him to guild.", plGuid.GetString().c_str());

227  }
228 
229  newmember.RankId = plRank;
230  newmember.OFFnote = (std::string)"";
231  newmember.Pnote = (std::string)"";
232  newmember.LogoutTime = time(nullptr);
233  newmember.BankResetTimeMoney = 0; // this will force update at first query
234  for (unsigned int& i : newmember.BankResetTimeTab)
235  i = 0;
236  members[lowguid] = newmember;
237 
238  std::string dbPnote = newmember.Pnote;
239  std::string dbOFFnote = newmember.OFFnote;
240  CharacterDatabase.escape_string(dbPnote);

244  m_Id, lowguid, newmember.RankId, dbPnote.c_str(), dbOFFnote.c_str());
245 
246  // If player not in game data in data field will be loaded from guild tables, no need to update it!!
247  if (pl)
248  {
249  pl->SetInGuild(m_Id);
250  pl->SetRank(newmember.RankId);
251  pl->SetGuildIdInvited(0);
252  }
253 
254  UpdateAccountsNumber();
255 
256  return true;

332 bool Guild::LoadRanksFromDB(QueryResult* guildRanksResult)
333 {
334  if (!guildRanksResult)
335  {
336  sLog.outError("Guild %u has broken `guild_rank` data, creating new...", m_Id);
337  CreateDefaultGuildRanks(0);
338  return true;
339  }
340 
341  bool broken_ranks = false;
342 
343  // GUILD RANKS are sequence starting from 0 = GUILD_MASTER (ALL PRIVILEGES) to max 9 (lowest privileges)
344  // the lower rank id is considered higher rank - so promotion does rank-- and demotion does rank++
345  // between ranks in sequence cannot be gaps - so 0,1,2,4 cannot be
346  // min ranks count is 5 and max is 10.
347 
348  do
349  {
350  Field* fields = guildRanksResult->Fetch();
351  // condition that would be true when all ranks in QueryResult will be processed and guild without ranks is being processed
352  if (!fields)
353  break;
354 
355  uint32 guildId = fields[0].GetUInt32();
356  if (guildId < m_Id)
357  {
358  // there is in table guild_rank record which doesn't have guildid in guild table, report error
359  sLog.outErrorDb("Guild %u does not exist but it has a record in guild_rank table, deleting it!", guildId);
360  CharacterDatabase.PExecute("DELETE FROM guild_rank WHERE guildid = '%u'", guildId);
361  continue;
362  }
363 
364  if (guildId > m_Id) // we loaded all ranks for this guild already, break cycle
365  break;
366 
367  uint32 rankID = fields[1].GetUInt32();
368  std::string rankName = fields[2].GetCppString();
369  uint32 rankRights = fields[3].GetUInt32();
370  uint32 rankMoney = fields[4].GetUInt32();
371 
372  if (rankID != m_Ranks.size()) // guild_rank.ids are sequence 0,1,2,3..
373  broken_ranks = true;
374 
375  // first rank is guildmaster, prevent loss leader rights
376  if (m_Ranks.empty())
377  rankRights |= GR_RIGHT_ALL;
378 
379  AddRank(rankName, rankRights, rankMoney);
380  }
381  while (guildRanksResult->NextRow());
382 
383  if (m_Ranks.size() < GUILD_RANKS_MIN_COUNT) // if too few ranks, renew them
384  {
385  m_Ranks.clear();
386  sLog.outError("Guild %u has broken `guild_rank` data, creating new...", m_Id);
387  CreateDefaultGuildRanks(0); // 0 is default locale_idx
388  broken_ranks = false;
389  }
390  // guild_rank have wrong numbered ranks, repair
391  if (broken_ranks)
392  {
393  sLog.outError("Guild %u has broken `guild_rank` data, repairing...", m_Id);
394  CharacterDatabase.BeginTransaction();
395  CharacterDatabase.PExecute("DELETE FROM guild_rank WHERE guildid='%u'", m_Id);
396  for (size_t i = 0; i < m_Ranks.size(); ++i)
397  {
398  std::string name = m_Ranks[i].Name;
399  uint32 rights = m_Ranks[i].Rights;
400  CharacterDatabase.escape_string(name);
401  CharacterDatabase.PExecute("INSERT INTO guild_rank (guildid,rid,rname,rights) VALUES ('%u', '%u', '%s', '%u')", m_Id, uint32(i), name.c_str(), rights);
402  }
403  CharacterDatabase.CommitTransaction();
404  }
405 
406  return true;
407 }
