195 template <class T>
196 PoolObject* PoolGroup<T>::RollOne(SpawnedPoolData& spawns, uint32 triggerFrom, MapPersistentState& mapState)
197 {
198  PoolObject* explicitlyObjFound = nullptr;
199  auto gen = GetRandomGenerator();
200 
201  if (!ExplicitlyChanced.empty())
202  {
203  float roll = (float)rand_chance();
204 
205  std::vector <PoolObject*> explicitlyChancedVector;
206 
207  // call memory manager once to reserve enough memory for performance
208  explicitlyChancedVector.reserve(ExplicitlyChanced.size());
209 
210  // fill new vector with address of object in EqualChanced list
211  std::transform(ExplicitlyChanced.begin(), ExplicitlyChanced.end(), std::back_inserter(explicitlyChancedVector), [](PoolObject& objPtr) { return &objPtr; });
212 
213  // randomize the new vector
214  std::shuffle(explicitlyChancedVector.begin(), explicitlyChancedVector.end(), *gen);
215 
216  for (auto obj : explicitlyChancedVector)
217  {
218  if (obj->exclude)
219  continue;
220 
221  if (!CanSpawn(obj, mapState))
222  continue;
223 
224  if (obj->guid != triggerFrom && spawns.IsSpawnedObject<T>(obj->guid))
225  continue;
226 
227  // we found an object but it may have no luck to be picked so we save it in case of no other object have been picked.
228  if (!explicitlyObjFound)
229  explicitlyObjFound = obj;
230 
231  if (roll < obj->chance)
232  return obj;
233  }
234  }
235 
236  if (!EqualChanced.empty())
237  {
238  std::vector <PoolObject*> equalyChancedVector;
239 
240  // call memory manager once to reserve enough memory for performance
241  equalyChancedVector.reserve(EqualChanced.size());
242 
243  // fill new vector with address of object in EqualChanced list
244  std::transform(EqualChanced.begin(), EqualChanced.end(), std::back_inserter(equalyChancedVector), [](PoolObject& objPtr) { return &objPtr; });
245 
246  // randomize the new vector
247  std::shuffle(equalyChancedVector.begin(), equalyChancedVector.end(), *gen);
248 
249  for (auto obj : equalyChancedVector)
250  {
251  if (obj->exclude)
252  continue;
253 
254  if (!CanSpawn(obj, mapState))
255  continue;
256 
257  if (obj->guid != triggerFrom && spawns.IsSpawnedObject<T>(obj->guid))
258  continue;
259 
260  return obj;
261  }
262  }
263 
264  // return first explicitly chanced object if there is one
265  return explicitlyObjFound;
266 }

390 template <class T>
391 void PoolGroup<T>::SpawnObject(MapPersistentState& mapState, uint32 limit, uint32 triggerFrom, bool instantly)
392 {
393  SpawnedPoolData& spawns = mapState.GetSpawnedPoolData();
394 
395  uint32 lastDespawned = 0;
396  int count = limit - spawns.GetSpawnedObjects(poolId);
397 
398  // If triggered from some object respawn this object is still marked as spawned
399  // and also counted into m_SpawnedPoolAmount so we need increase count to be
400  // spawned by 1
401  if (triggerFrom)
402  {
403  if (spawns.IsSpawnedObject<T>(triggerFrom))
404  ++count;
405  else
406  triggerFrom = 0;
407  }
408 
409  // This will try to spawn the rest of pool, not guaranteed
410  for (int i = 0; i < count; ++i)
411  {
412  PoolObject* obj = RollOne(spawns, triggerFrom, mapState);
413  if (!obj)
414  continue;
415  if (obj->guid == lastDespawned)
416  continue;
417 
418  if (obj->guid == triggerFrom)
419  {
420  MANGOS_ASSERT(spawns.IsSpawnedObject<T>(obj->guid));
421  MANGOS_ASSERT(spawns.GetSpawnedObjects(poolId) > 0);
422  ReSpawn1Object(mapState, obj);
423  triggerFrom = 0;
424  continue;
425  }
426 
427  spawns.AddSpawn<T>(obj->guid, poolId);
428  Spawn1Object(mapState, obj, instantly);
429 
430  if (triggerFrom)
431  {
432  // One spawn one despawn no count increase
433  DespawnObject(mapState, triggerFrom);
434  lastDespawned = triggerFrom;
435  triggerFrom = 0;
436  }
437  }
438 }

603 void PoolManager::LoadFromDB()
604 {

606  if (!result)

613  max_pool_id = fields[0].GetUInt16();
614  delete result;

619  if (!result)

621  mPoolTemplate.clear();
622  sLog.outString(">> Table pool_template is empty:");
623  sLog.outString();
624  return;
625  }

628

630  do

632  ++count;
