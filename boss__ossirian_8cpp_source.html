134  void DoSpawnNextCrystal(uint32 spawnCount)
135  {
136  if (!m_instance)
137  return;
138 
139  GuidVector vector;
140  m_instance->GetCreatureGuidVectorFromStorage(NPC_OSSIRIAN_TRIGGER, vector);
141  if (vector.size() < 2) // wrong db data
142  return;
143 
144  vector.erase(vector.begin()); // remove first
145  std::shuffle(vector.begin(), vector.end(), *GetRandomGenerator());
146 
147  uint32 spawned = 0;
148  for (uint32 i = 0; i < vector.size() - 1; ++i) // try to find at least one
149  {
150  // iterate from random roll until either one (should always occur) is found or we run out of crystals
151  if (Creature* creature = m_creature->GetMap()->GetCreature(vector[i]))
152  {
153  if (!creature->IsAlive())
154  {
155  creature->Respawn();
156  ++spawned;
157  if (spawned >= spawnCount)
158  break;
159  }
160  }
161  }
162  }

249  void ExecuteAction(uint32 action) override
250  {
251  switch (action)
252  {
253  case OSSIRIAN_INITIAL_SPAWN:
254  {
255  DoSpawnNextCrystal(5);
256  DisableCombatAction(action);
257  break;
258  }
259  case OSSIRIAN_SUPREME:
260  {
261  if (DoCastSpellIfCan(nullptr, SPELL_SUPREME, CAST_AURA_NOT_PRESENT) == CAST_OK)
262  {
263  switch (urand(0, 2))
264  {
265  case 0: DoScriptText(SAY_SUPREME_1, m_creature); break;
266  case 1: DoScriptText(SAY_SUPREME_2, m_creature); break;
267  case 2: DoScriptText(SAY_SUPREME_3, m_creature); break;
268  }
269  ResetCombatAction(action, 45000);
270  }
271  break;
272  }
273  case OSSIRIAN_CYCLONE:
274  {
275  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CYCLONE) == CAST_OK)
276  ResetCombatAction(action, 20000);
277  break;
278  }
279  case OSSIRIAN_STOMP:
280  {
281  if (DoCastSpellIfCan(nullptr, SPELL_STOMP) == CAST_OK)
282  ResetCombatAction(action, 30000);
283  break;
284  }
285  case OSSIRIAN_CURSE_OF_TONGUES:
286  {
287  if (DoCastSpellIfCan(nullptr, SPELL_CURSE_OF_TONGUES) == CAST_OK)
288  ResetCombatAction(action, urand(20000, 30000));
289  break;
290  }
291  case OSSIRIAN_SPEEDUP:
292  {
293  m_creature->UpdateSpeed(MOVE_RUN, false, 2.2f);
294  DisableCombatAction(action);
295  break;
296  }
297  }
