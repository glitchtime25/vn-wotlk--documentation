62  void UpdateAI(const uint32 uiDiff) override
63  {
64  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
65  return;
66 
67  if (m_uiDetonationTimer)
68  {
69  if (m_uiDetonationTimer <= uiDiff)
70  {
71  if (DoCastSpellIfCan(m_creature, SPELL_DETONATION) == CAST_OK)
72  {
73  DoScriptText(SAY_SPECIALAE, m_creature);
74  m_uiDetonationTimer = 0;
75  }
76  }
77  else
78  m_uiDetonationTimer -= uiDiff;
79  }
80 
81  // Do not attack while having the bubble active
82  if (m_creature->HasAura(SPELL_ARCANE_BUBBLE))
83  return;
84 
85  // If we are <50% hp cast Arcane Bubble
86  if (!bShielded && m_creature->GetHealthPercent() <= 50.0f)
87  {
88  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_BUBBLE) == CAST_OK)
89  {
90  m_uiDetonationTimer = 1000;
91  bShielded = true;
92  }
93  }
94 
95  if (m_uiPolymorphTimer < uiDiff)
96  {
97  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
98  {
99  if (DoCastSpellIfCan(pTarget, SPELL_POLYMORPH) == CAST_OK)
100  m_uiPolymorphTimer = 20000;
101  }
102  }
103  else
104  m_uiPolymorphTimer -= uiDiff;
105 
106  // Silence_Timer
107  if (m_uiSilenceTimer < uiDiff)
108  {
109  if (DoCastSpellIfCan(m_creature, SPELL_SILENCE) == CAST_OK)
110  m_uiSilenceTimer = urand(15000, 22000);
111  }
112  else
113  m_uiSilenceTimer -= uiDiff;
114 
115  // ArcaneExplosion_Timer
116  if (m_uiArcaneExplosionTimer < uiDiff)
117  {
118  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_EXPLOSION) == CAST_OK)
119  m_uiArcaneExplosionTimer = urand(2500, 8500);
120  }
121  else
122  m_uiArcaneExplosionTimer -= uiDiff;
123 
124  DoMeleeAttackIfReady();
125  }
