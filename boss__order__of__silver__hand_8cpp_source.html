115  bool CanUseSpecialAbility(uint32 uiIndex)
116  {
117  Unit* pTarget = nullptr;
118 
119  switch (m_aSilverHandAbility[uiIndex].m_uiTargetType)
120  {
121  case TARGET_TYPE_SELF:
122  pTarget = m_creature;
123  break;
124  case TARGET_TYPE_VICTIM:
125  pTarget = m_creature->GetVictim();
126  break;
127  case TARGET_TYPE_RANDOM:
128  pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, m_aSilverHandAbility[uiIndex].m_uiSpellId, SELECT_FLAG_IN_LOS);
129  break;
130  case TARGET_TYPE_FRIENDLY:
131  pTarget = DoSelectLowestHpFriendly(10.0f);
132  break;
133  }
134 
135  if (pTarget)
136  {
137  if (DoCastSpellIfCan(pTarget, m_aSilverHandAbility[uiIndex].m_uiSpellId) == CAST_OK)
138  return true;
139  }
140 
141  return false;
142  }

144  void UpdateAI(const uint32 uiDiff) override
145  {
146  // Return since we have no target
147  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
148  return;
149 
150  for (auto& m_mSpellTimer : m_mSpellTimers)
151  {
152  if (m_mSpellTimer.second < uiDiff)
153  {
154  if (CanUseSpecialAbility(m_mSpellTimer.first))
155  {
156  m_mSpellTimer.second = m_aSilverHandAbility[m_mSpellTimer.first].m_uiCooldown;
157  break;
158  }
159  }
160  else
161  m_mSpellTimer.second -= uiDiff;
162  }
163 
164  DoMeleeAttackIfReady();
165  }
