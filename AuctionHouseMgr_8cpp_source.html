268

315 void AuctionHouseMgr::LoadAuctions()
316 {
317  QueryResult* result = CharacterDatabase.Query("SELECT COUNT(*) FROM auction");
318  if (!result)
319  {
320  BarGoLink bar(1);
321  bar.step();
322  sLog.outString(">> Loaded 0 auctions. DB table `auction` is empty.");
323  return;
324  }
325 
326  Field* fields = result->Fetch();
327  uint32 AuctionCount = fields[0].GetUInt32();
328  delete result;
329 
330  if (!AuctionCount)
331  {
332  BarGoLink bar(1);
333  bar.step();
334  sLog.outString(">> Loaded 0 auctions. DB table `auction` is empty.");
335  return;
336  }
337 
338  result = CharacterDatabase.Query("SELECT id,houseid,itemguid,item_template,item_count,item_randompropertyid,itemowner,buyoutprice,time,moneyTime,buyguid,lastbid,startbid,deposit FROM auction");
339  if (!result)
340  {
341  BarGoLink bar(1);
342  bar.step();
343  sLog.outString(">> Loaded 0 auctions. DB table `auction` is empty.");
344  return;
345  }
346 
347  BarGoLink bar(AuctionCount);
348 
349  typedef std::map<uint32, std::wstring> PlayerNames;
350  PlayerNames playerNames; // caching for load time
351 
352  do
353  {
354  fields = result->Fetch();
355 
356  bar.step();
357 
358  AuctionEntry* auction = new AuctionEntry;
359  auction->Id = fields[0].GetUInt32();
360  uint32 houseid = fields[1].GetUInt32();
361  auction->itemGuidLow = fields[2].GetUInt32();
362  auction->itemTemplate = fields[3].GetUInt32();
363  auction->itemCount = fields[4].GetUInt32();
364  auction->itemRandomPropertyId = fields[5].GetUInt32();
365 
366  auction->owner = fields[6].GetUInt32();
367 
368  if (auction->owner)
369  {
370  std::wstring& plWName = playerNames[auction->owner];
371  if (plWName.empty())
372  {
373  std::string plName;
374  if (!sObjectMgr.GetPlayerNameByGUID(ObjectGuid(HIGHGUID_PLAYER, auction->owner), plName))
375  plName = sObjectMgr.GetMangosStringForDbcLocale(LANG_UNKNOWN);
376 
377  Utf8toWStr(plName, plWName);
378  }
379 
380  auction->ownerName = plWName;
381  }
382 
383  auction->buyout = fields[7].GetUInt32();
384  auction->expireTime = time_t(fields[8].GetUInt64());
385  auction->moneyDeliveryTime = time_t(fields[9].GetUInt64());
386  auction->bidder = fields[10].GetUInt32();
387  auction->bid = fields[11].GetUInt32();
388  auction->startbid = fields[12].GetUInt32();
389  auction->deposit = fields[13].GetUInt32();
390  auction->auctionHouseEntry = nullptr; // init later
391 
392  if (auction->moneyDeliveryTime)
393  auction->itemGuidLow = 0; // must be 0 if auction delivery pending
394  else
395  {
396  // check if sold item exists for guid
397  // and item_template in fact (GetAItem will fail if problematic in result check in AuctionHouseMgr::LoadAuctionItems)
398  Item* pItem = GetAItem(auction->itemGuidLow);
399  if (!pItem)
400  {
401  auction->DeleteFromDB();
402  sLog.outError("Auction %u has not a existing item : %u, deleted", auction->Id, auction->itemGuidLow);
403  delete auction;
404  continue;
405  }
406 
407  // overwrite by real item data
408  if ((auction->itemTemplate != pItem->GetEntry()) ||
409  (auction->itemCount != pItem->GetCount()) ||
410  (auction->itemRandomPropertyId != pItem->GetItemRandomPropertyId()))
411  {
412  auction->itemTemplate = pItem->GetEntry();
413  auction->itemCount = pItem->GetCount();
414  auction->itemRandomPropertyId = pItem->GetItemRandomPropertyId();
415 
416  // No SQL injection (no strings)
417  CharacterDatabase.PExecute("UPDATE auction SET item_template = %u, item_count = %u, item_randompropertyid = %i WHERE itemguid = %u",
418  auction->itemTemplate, auction->itemCount, auction->itemRandomPropertyId, auction->itemGuidLow);
419  }
420  }
421 
422  auction->auctionHouseEntry = sAuctionHouseStore.LookupEntry(houseid);
423 
424  if (!auction->auctionHouseEntry)
425  {
426  // need for send mail, use goblin auctionhouse
427  auction->auctionHouseEntry = sAuctionHouseStore.LookupEntry(7);
428 
429  // Attempt send item back to owner
430  std::ostringstream msgAuctionCanceledOwner;
431  msgAuctionCanceledOwner << auction->itemTemplate << ":" << auction->itemRandomPropertyId << ":" << AUCTION_CANCELED << ":" << auction->Id << ":" << auction->itemCount;
432 
433  if (auction->itemGuidLow)
434  {
435  Item* pItem = GetAItem(auction->itemGuidLow);
436 
437  RemoveAItem(auction->itemGuidLow);
438  auction->itemGuidLow = 0;
439 
440  // item will deleted or added to received mail list
441  MailDraft(msgAuctionCanceledOwner.str(), "")// TODO: fix body
442  .AddItem(pItem)
443  .SendMailTo(MailReceiver(ObjectGuid(HIGHGUID_PLAYER, auction->owner)), auction, MAIL_CHECK_MASK_COPIED);
444  }
445 
446  auction->DeleteFromDB();
447  delete auction;
448 
449  continue;
450  }
451 
452  GetAuctionsMap(auction->auctionHouseEntry)->AddAuction(auction);
453  }
454  while (result->NextRow());
455  delete result;
456 
457  sLog.outString(">> Loaded %u auctions", AuctionCount);
458  sLog.outString();
459 }

502 AuctionHouseEntry const* AuctionHouseMgr::GetAuctionHouseEntry(Unit* unit)
503 {
504  uint32 houseid = 1; // dwarf auction house (used for normal cut/etc percents)
505 
506  if (!sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_AUCTION))
507  {
508  if (unit->GetTypeId() == TYPEID_UNIT)
509  {
510  // FIXME: found way for proper auctionhouse selection by another way
511  // AuctionHouse.dbc have faction field with _player_ factions associated with auction house races.
512  // but no easy way convert creature faction to player race faction for specific city
513  uint32 factionTemplateId = unit->GetFaction();
514  switch (factionTemplateId)
515  {
516  case 12: houseid = 1; break; // human
517  case 29: houseid = 6; break; // orc, and generic for horde
518  case 55: houseid = 2; break; // dwarf/gnome, and generic for alliance
519  case 68: houseid = 4; break; // undead
520  case 80: houseid = 3; break; // n-elf
521  case 104: houseid = 5; break; // trolls
522  case 120: houseid = 7; break; // booty bay, neutral
523  case 474: houseid = 7; break; // gadgetzan, neutral
524  case 534: houseid = 2; break; // Alliance Generic
525  case 855: houseid = 7; break; // everlook, neutral
526  case 1604: houseid = 6; break; // b-elfs,
527  case 1638: houseid = 2; break; // exodar, alliance
528  default: // for unknown case
529  {
530  FactionTemplateEntry const* u_entry = sFactionTemplateStore.LookupEntry(factionTemplateId);
531  if (!u_entry)
532  houseid = 7; // goblin auction house
533  else if (u_entry->factionGroupMask & FACTION_GROUP_MASK_ALLIANCE)
534  houseid = 1; // human auction house
535  else if (u_entry->factionGroupMask & FACTION_GROUP_MASK_HORDE)
536  houseid = 6; // orc auction house
537  else
538  houseid = 7; // goblin auction house
539  break;
540  }
541  }
542  }
543  else
544  {
545  Player* player = (Player*)unit;
546  if (player->GetAuctionAccessMode() > 0)
547  houseid = 7;
548  else
549  {
550  switch (((Player*)unit)->GetTeam())
551  {
552  case ALLIANCE: houseid = player->GetAuctionAccessMode() == 0 ? 1 : 6; break;
553  case HORDE: houseid = player->GetAuctionAccessMode() == 0 ? 6 : 1; break;
554  default: break;
555  }
556  }
557  }
558  }
559 
560  return sAuctionHouseStore.LookupEntry(houseid);
561 }

563 void AuctionHouseObject::Update()
564 {
565  time_t curTime = sWorld.GetGameTime();
567  for (AuctionEntryMap::iterator itr = AuctionsMap.begin(); itr != AuctionsMap.end();)
568  {
569  if (itr->second->moneyDeliveryTime) // pending auction
570  {
571  if (curTime > itr->second->moneyDeliveryTime)
572  {
573  sAuctionMgr.SendAuctionSuccessfulMail(itr->second);
574 
575  itr->second->DeleteFromDB();
576  MANGOS_ASSERT(!itr->second->itemGuidLow); // already removed or send in mail at won
577  delete itr->second;
578  AuctionsMap.erase(itr++);
579  continue;
580  }
581  }
582  else // active auction
583  {
584  if (curTime > itr->second->expireTime)
585  {
587  if (itr->second->bid)
588  itr->second->AuctionBidWinning();
590  else
591  {
592  sAuctionMgr.SendAuctionExpiredMail(itr->second);
593 
594  itr->second->DeleteFromDB();
595  delete itr->second;
596  AuctionsMap.erase(itr++);
597  continue;
598  }
599  }
600  }
601 
602  ++itr;
603  }
604 }

785 bool AuctionSorter::operator()(const AuctionEntry* auc1, const AuctionEntry* auc2) const
786 {
787  if (m_sort[0] == MAX_AUCTION_SORT) // not sorted
788  return false;
789 
790  for (uint32 i = 0; i < MAX_AUCTION_SORT; ++i)
791  {
792  if (m_sort[i] == MAX_AUCTION_SORT) // end of sort
793  return false;
794 
795  int res = auc1->CompareAuctionEntry(m_sort[i] & ~AUCTION_SORT_REVERSED, auc2, m_viewPlayer);
796  // "equal" by used column
797  if (res == 0)
798  continue;
799  // less/greater and normal/reversed ordered
800  return (res < 0) == ((m_sort[i] & AUCTION_SORT_REVERSED) == 0);
801  }
802 
803  return false; // "equal" by all sorts
804 }
