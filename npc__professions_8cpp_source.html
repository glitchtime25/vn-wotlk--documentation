128 bool EquippedOk(Player* pPlayer, uint32 spellId)
129 {
130  SpellEntry const* spell = GetSpellStore()->LookupEntry<SpellEntry>(spellId);
131 
132  if (!spell)
133  return false;
134 
135  for (unsigned int reqSpell : spell->EffectTriggerSpell)
136  {
137  if (!reqSpell)
138  continue;
139 
140  for (int j = EQUIPMENT_SLOT_START; j < EQUIPMENT_SLOT_END; ++j)
141  {
142  Item* pItem = pPlayer->GetItemByPos(INVENTORY_SLOT_BAG_0, j);
143  if (pItem)
144  if (pItem->GetProto()->RequiredSpell == reqSpell)
145  {
146  // pPlayer has item equipped that require specialty. Not allow to unlearn, player has to unequip first
147  debug_log("SD2: player attempt to unlearn spell %u, but item %u is equipped.", reqSpell, pItem->GetProto()->ItemId);
148  return false;
149  }
150  }
151  }
152  return true;
153 }

205 bool IsEligibleSpecializeLW(Player* pPlayer)
206 {
207  // Allow re-learn specialization as long as player is not specialized currently and has completed any specialization quest
208  if ((!pPlayer->HasSpell(S_DRAGON) && !pPlayer->HasSpell(S_ELEMENTAL) && !pPlayer->HasSpell(S_TRIBAL))
209  && (pPlayer->GetQuestRewardStatus(Q_DRAGON_ALLY)
210  || pPlayer->GetQuestRewardStatus(Q_DRAGON_HORDE)
211  || pPlayer->GetQuestRewardStatus(Q_ELEMENTAL_ALLY)
212  || pPlayer->GetQuestRewardStatus(Q_ELEMENTAL_HORDE)
213  || pPlayer->GetQuestRewardStatus(Q_TRIBAL_ALLY)
214  || pPlayer->GetQuestRewardStatus(Q_TRIBAL_HORDE)))
215  return true;
216  return false;
217 }

219 bool GossipHello_npc_prof_leather(Player* pPlayer, Creature* pCreature)
220 {
221  if (pCreature->isQuestGiver())
222  pPlayer->PrepareQuestMenu(pCreature->GetObjectGuid());
223  if (pCreature->isVendor())
224  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_VENDOR, GOSSIP_TEXT_BROWSE_GOODS, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRADE);
225 
226  uint32 eCreature = pCreature->GetEntry();
227 
228  if (pPlayer->HasSkill(SKILL_LEATHERWORKING) && pPlayer->GetSkillValueBase(SKILL_LEATHERWORKING) >= 225 && pPlayer->GetLevel() > 39)
229  {
230  switch (eCreature)
231  {
232  case N_TRAINER_DRAGON1:
233  case N_TRAINER_DRAGON2:
234  if (pPlayer->HasSpell(S_DRAGON))
235  {
236  pPlayer->ADD_GOSSIP_ITEM_EXTENDED(GOSSIP_ICON_CHAT, GOSSIP_UNLEARN_DRAGON, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 1, BOX_UNLEARN_DRAGON, GetUnlearnCostMedium(pPlayer), false);
237  if (pCreature->isTrainer())
238  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_TRAINER, GOSSIP_TEXT_TRAIN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRAIN);
239  }
240  else if (IsEligibleSpecializeLW(pPlayer))
241  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_LEARN_DRAGON, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 2);
242  break;
243  case N_TRAINER_ELEMENTAL1:
244  case N_TRAINER_ELEMENTAL2:
245  if (pPlayer->HasSpell(S_ELEMENTAL))
246  {
247  pPlayer->ADD_GOSSIP_ITEM_EXTENDED(GOSSIP_ICON_CHAT, GOSSIP_UNLEARN_ELEMENTAL, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 3, BOX_UNLEARN_ELEMENTAL, GetUnlearnCostMedium(pPlayer), false);
248  if (pCreature->isTrainer())
249  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_TRAINER, GOSSIP_TEXT_TRAIN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRAIN);
250  }
251  else if (IsEligibleSpecializeLW(pPlayer))
252  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_LEARN_ELEMENTAL, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 4);
253  break;
254  case N_TRAINER_TRIBAL1:
255  case N_TRAINER_TRIBAL2:
256  if (pPlayer->HasSpell(S_TRIBAL))
257  {
258  pPlayer->ADD_GOSSIP_ITEM_EXTENDED(GOSSIP_ICON_CHAT, GOSSIP_UNLEARN_TRIBAL, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 5, BOX_UNLEARN_TRIBAL, GetUnlearnCostMedium(pPlayer), false);
259  if (pCreature->isTrainer())
260  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_TRAINER, GOSSIP_TEXT_TRAIN, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_TRAIN);
261  }
262  else if (IsEligibleSpecializeLW(pPlayer))
263  pPlayer->ADD_GOSSIP_ITEM(GOSSIP_ICON_CHAT, GOSSIP_LEARN_TRIBAL, GOSSIP_SENDER_CHECK, GOSSIP_ACTION_INFO_DEF + 6);
264  break;
265  }
266  }
267 
268  pPlayer->SEND_GOSSIP_MENU(pPlayer->GetGossipTextId(pCreature), pCreature->GetObjectGuid());
269  return true;

272 void SendActionMenu_npc_prof_leather(Player* pPlayer, Creature* pCreature, uint32 uiAction)
273 {
274  switch (uiAction)
275  {
276  case GOSSIP_ACTION_TRADE:
277  pPlayer->SEND_VENDORLIST(pCreature->GetObjectGuid());
278  break;
279  case GOSSIP_ACTION_TRAIN:
280  pPlayer->SEND_TRAINERLIST(pCreature->GetObjectGuid());
281  break;
282  // Unlearn Dragonscale Leatherworking
283  case GOSSIP_ACTION_INFO_DEF + 1:
284  if (EquippedOk(pPlayer, S_UNLEARN_DRAGON))
285  {
286  if (pPlayer->GetMoney() >= uint32(GetUnlearnCostMedium(pPlayer)))
287  {
288  pPlayer->CastSpell(pPlayer, S_UNLEARN_DRAGON, TRIGGERED_OLD_TRIGGERED);
289  ProfessionUnlearnSpells(pPlayer, S_UNLEARN_DRAGON);
290  pPlayer->ModifyMoney(-GetUnlearnCostMedium(pPlayer));
291  }
292  else
293  pPlayer->SendBuyError(BUY_ERR_NOT_ENOUGHT_MONEY, pCreature, 0, 0);
294  }
295  else
296  pPlayer->SendEquipError(EQUIP_ERR_CANT_DO_RIGHT_NOW, nullptr, nullptr);
297  pPlayer->CLOSE_GOSSIP_MENU();
298  break;
299  case GOSSIP_ACTION_INFO_DEF + 2: // Learn Dragonscale Leatherworking
300  pPlayer->CastSpell(pPlayer, S_LEARN_DRAGON, TRIGGERED_OLD_TRIGGERED);
301  pPlayer->CLOSE_GOSSIP_MENU();
302  break;
303  case GOSSIP_ACTION_INFO_DEF + 3: // Unlearn Elemental Leatherworking
304  if (EquippedOk(pPlayer, S_UNLEARN_ELEMENTAL))
305  {
306  if (pPlayer->GetMoney() >= uint32(GetUnlearnCostMedium(pPlayer)))
