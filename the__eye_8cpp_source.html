116 void instance_the_eye::SetData(uint32 uiType, uint32 uiData)
117 {
118  switch (uiType)
119  {
120  case TYPE_ALAR:
121  case TYPE_SOLARIAN:
122  case TYPE_VOIDREAVER:
123  m_auiEncounter[uiType] = uiData;
124 #ifdef PRENERF_2_0_3
125  if (CheckDoorOpening())
126  OpenDoors();
127 #endif
128  break;
129  case TYPE_KAELTHAS:
130  // Don't set the same data twice
131  if (m_auiEncounter[uiType] == uiData)
132  break;
133  DoUseDoorOrButton(GO_ARCANE_DOOR_HORIZ_3);
134  DoUseDoorOrButton(GO_ARCANE_DOOR_HORIZ_4);
135  if (uiData == FAIL)
136  {
137  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_KAEL_STATUE_LEFT))
138  pGo->ResetDoorOrButton();
139  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_KAEL_STATUE_RIGHT))
140  pGo->ResetDoorOrButton();
141  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_BRIDGE_WINDOW))
142  pGo->ResetDoorOrButton();
143 
144  // despawn the advisors
145  for (unsigned int aAdvisor : aAdvisors)
146  if (Creature* add = GetSingleCreatureFromStorage(aAdvisor))
147  add->ForcedDespawn();
148 
149  if (Creature* kael = GetSingleCreatureFromStorage(NPC_KAELTHAS))
150  {
151  kael->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, kael, kael);
152  kael->SetRespawnDelay(30, true);
153  kael->ForcedDespawn();
154  }
155  }
156  m_auiEncounter[uiType] = uiData;
157  break;
158  }
159 
160  if (uiData == DONE || uiData == SPECIAL)
161  {
162  OUT_SAVE_INST_DATA;
163 
164  std::ostringstream saveStream;
165  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
166  << m_auiEncounter[3];
167 
168  m_strInstData = saveStream.str();
169 
170  SaveToDB();
171  OUT_SAVE_INST_DATA_COMPLETE;
172  }
173 }
