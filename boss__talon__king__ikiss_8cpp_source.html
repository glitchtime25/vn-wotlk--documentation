154  void HandleBlink()
155  {
156  SetCombatMovement(true);
157  SetMeleeEnabled(true);
158  SetCombatScriptStatus(false);
159  DoCastSpellIfCan(nullptr, m_isRegularMode ? SPELL_ARCANE_EXPLOSION : SPELL_ARCANE_EXPLOSION_H);

164  void ExecuteAction(uint32 action)
165  {
166  switch (action)
167  {
168  case TALON_KING_IKISS_MANA_SHIELD:
169  if (m_creature->GetHealthPercent() < 15.0f)
170  if (DoCastSpellIfCan(nullptr, SPELL_MANA_SHIELD) == CAST_OK)
171  DisableCombatAction(action);
172  break;
173  case TALON_KING_IKISS_BLINK_START:
174  if (m_creature->GetHealthPercent() < m_HealthPercent)
175  {
176  if (DoCastSpellIfCan(m_creature, SPELL_BLINK, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
177  {
178  SetMeleeEnabled(false);
179  SetCombatMovement(false);
180  SetCombatScriptStatus(true);
181  ResetTimer(TALON_KING_IKISS_BLINK, 1000);
182  DoScriptText(EMOTE_ARCANE_EXP, m_creature);
183 
184  // There is no relationship between the health percentages
185  switch (m_uiBlinkPhase)
186  {
187  case 0: m_HealthPercent = 50.0f; break;
188  case 1: m_HealthPercent = 25.0f; break;
189  case 2: m_HealthPercent = 0.0f; break;
190  }
191 
192  ++m_uiBlinkPhase;
193  }
194  }
195  break;
196  case TALON_KING_IKISS_ACTION_SLOW:
197  {
198  if (!m_isRegularMode)
199  if (DoCastSpellIfCan(nullptr, SPELL_SLOW_H) == CAST_OK)
200  ResetCombatAction(action, GetSubsequentActionTimer(action));
201  break;
202  }
203  case TALON_KING_IKISS_ACTION_POLYMORPH:
204  {
205  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA | SELECT_FLAG_SKIP_TANK))
206  if (DoCastSpellIfCan(target, m_isRegularMode ? SPELL_POLYMORPH : SPELL_POLYMORPH_H) == CAST_OK)
207  ResetCombatAction(action, GetSubsequentActionTimer(action));
208  break;
209  }
210  case TALON_KING_IKISS_ACTION_ARCANE_VOLLEY:
211  {
212  if (DoCastSpellIfCan(nullptr, m_isRegularMode ? SPELL_ARCANE_VOLLEY : SPELL_ARCANE_VOLLEY_H) == CAST_OK)
213  ResetCombatAction(action, GetSubsequentActionTimer(action));
214  }
215  break;
216  }
217  }
