211 bool EffectAuraDummy_spell_aura_dummy_npc_crates_dummy(const Aura* pAura, bool bApply)
212 {
213  if (pAura->GetId() == SPELL_ARCANE_DISRUPTION && pAura->GetEffIndex() == EFFECT_INDEX_0 && bApply)
214  {
215  if (Creature* pTarget = (Creature*)pAura->GetTarget())
216  {
217  if (pTarget->GetEntry() != NPC_GRAIN_CRATE_HELPER)
218  return true;
219 
220  CreatureList lCrateBunnyList;
221  if (instance_culling_of_stratholme* pInstance = (instance_culling_of_stratholme*)pTarget->GetInstanceData())
222  {
223  pInstance->GetCratesBunnyOrderedList(lCrateBunnyList);
224  uint8 i = 0;
225  for (CreatureList::const_iterator itr = lCrateBunnyList.begin(); itr != lCrateBunnyList.end(); ++itr)
226  {
227  ++i;
228  if (*itr == pTarget)
229  {
230  // check if the event can proceed
231  if (!pInstance->CanGrainEventProgress(pTarget))
232  return true;
233 
234  break;
235  }
236  }
237 
238  switch (i)
239  {
240  case 1:
241  // Start NPC_ROGER_OWENS Event
242  if (Creature* pRoger = pInstance->GetSingleCreatureFromStorage(NPC_ROGER_OWENS))
243  {
244  pRoger->SetStandState(UNIT_STAND_STATE_STAND);
245  pRoger->GetMotionMaster()->MoveWaypoint();
246  }
247  break;
248  case 2:
249  // Start NPC_SERGEANT_MORIGAN Event
250  if (Creature* pMorigan = pInstance->GetSingleCreatureFromStorage(NPC_SERGEANT_MORIGAN))
251  pMorigan->GetMotionMaster()->MoveWaypoint();
252  break;
253  case 3:
254  // Start NPC_JENA_ANDERSON Event
255  if (Creature* pJena = pInstance->GetSingleCreatureFromStorage(NPC_JENA_ANDERSON))
256  pJena->GetMotionMaster()->MoveWaypoint();
257  break;
258  case 4:
259  // Start NPC_MALCOM_MOORE Event
260  pTarget->SummonCreature(NPC_MALCOM_MOORE, 1605.452f, 804.9279f, 122.961f, 5.19f, TEMPSPAWN_DEAD_DESPAWN, 0);
261  break;
262  case 5:
263  // Start NPC_BARTLEBY_BATTSON Event
264  if (Creature* pBartleby = pInstance->GetSingleCreatureFromStorage(NPC_BARTLEBY_BATTSON))
265  pBartleby->GetMotionMaster()->MoveWaypoint();
266  break;
267  }
268 
269  // Finished event, give killcredit
270  if (pInstance->GetData(TYPE_GRAIN_EVENT) == DONE)
271  {
272  pTarget->CastSpell(pTarget, SPELL_CRATES_KILL_CREDIT, TRIGGERED_OLD_TRIGGERED);
273  pInstance->DoOrSimulateScriptTextForThisInstance(SAY_SOLDIERS_REPORT, NPC_LORDAERON_CRIER);
274  }
275 
276  // despawn the GO visuals and spanw the plague crate
277  if (GameObject* pCrate = GetClosestGameObjectWithEntry(pTarget, GO_SUSPICIOUS_GRAIN_CRATE, 5.0f))
278  pCrate->SetLootState(GO_JUST_DEACTIVATED);
279  if (GameObject* pHighlight = GetClosestGameObjectWithEntry(pTarget, GO_CRATE_HIGHLIGHT, 5.0f))
280  pHighlight->SetLootState(GO_JUST_DEACTIVATED);
281  if (GameObject* pCrate = GetClosestGameObjectWithEntry(pTarget, GO_PLAGUE_GRAIN_CRATE, 5.0f))
282  {
283  pCrate->SetRespawnTime(6 * HOUR * IN_MILLISECONDS);
284  pCrate->Refresh();
285  }
286  }
287  }
288  }
289  return true;
290 }
