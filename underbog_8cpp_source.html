89 bool AreaTrigger_at_ghazan_surface(Player* player, AreaTriggerEntry const* /*pAt*/)
90 {
91  ScriptedInstance* instance = (ScriptedInstance*)player->GetMap()->GetInstanceData();
92  if (instance->GetData(DATA_GHAZAN_SURFACE) == 0)
