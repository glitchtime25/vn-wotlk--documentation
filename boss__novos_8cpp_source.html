206  void UpdateAI(const uint32 uiDiff) override
207  {
208  switch (m_uiPhase)
209  {
210  case PHASE_IDLE:
211  // no actions during this phase
212  return;
213  case PHASE_SHIELDED: // Event Phase, only summoning of mobs
214  if (m_uiSummonHandlerTimer < uiDiff)
215  {
216  float fX, fY, fZ;
217  ++m_uiSummonedHandlers;
218  m_creature->GetRandomPoint(aHandlerSummonPos[m_uiSummonedHandlers % 2][0], aHandlerSummonPos[m_uiSummonedHandlers % 2][1], aHandlerSummonPos[m_uiSummonedHandlers % 2][2], 10.0f, fX, fY, fZ);
219  m_creature->SummonCreature(NPC_CRYSTAL_HANDLER, fX, fY, fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 0);
220 
221  DoScriptText(SAY_ADDS, m_creature);
222  DoScriptText(EMOTE_ASSISTANCE, m_creature);
223 
224  m_uiSummonHandlerTimer = 40000;
225  }
226  else
227  m_uiSummonHandlerTimer -= uiDiff;
228 
229  if (m_uiSummonShadowcasterTimer < uiDiff)
230  {
231  if (m_pInstance)
232  if (Creature* pSummoner = m_pInstance->GetSummonDummy())
233  pSummoner->CastSpell(pSummoner, SPELL_SUMMON_RISON_SHADOWCASTER, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
234  m_uiSummonShadowcasterTimer = 25000;
235  }
236  else
237  m_uiSummonShadowcasterTimer -= uiDiff;
238 
239  if (m_uiSummonFetidTrollTimer < uiDiff)
240  {
241  if (m_pInstance)
242  if (Creature* pSummoner = m_pInstance->GetSummonDummy())
243  pSummoner->CastSpell(pSummoner, SPELL_SUMMON_FETID_TROLL_CORPSE, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
244  m_uiSummonFetidTrollTimer = 5000;
245  }
246  else
247  m_uiSummonFetidTrollTimer -= uiDiff;
248 
249  if (m_uiSummonHulkingCorpseTimer < uiDiff)
250  {
251  if (m_pInstance)
252  if (Creature* pSummoner = m_pInstance->GetSummonDummy())
253  pSummoner->CastSpell(pSummoner, SPELL_SUMMON_HULKING_CORPSE, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
254  m_uiSummonHulkingCorpseTimer = 30000;
255  }
256  else
257  m_uiSummonHulkingCorpseTimer -= uiDiff;
258 
259  break;
260 
261  case PHASE_WAITING: // Short delay between last destroyed crystal and entering combat
262  if (m_uiPhaseTimer < uiDiff)
263  {
264  m_uiPhase = PHASE_NORMAL;
265  // Remove Immunity and Shield Aura
266  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
267  m_creature->RemoveAllAuras();
268 
269  if (!m_bIsRegularMode)
270  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MINIONS_H, CAST_INTERRUPT_PREVIOUS);
271  else
272  m_creature->InterruptNonMeleeSpells(true);
273  }
274  else
275  m_uiPhaseTimer -= uiDiff;
276 
277  break;
278 
279  case PHASE_NORMAL: // Normal Phase, attack enemies
280  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
281  return;
282 
283  if (m_uiArcaneBlastTimer < uiDiff)
284  {
285  // TODO - might be possible that this spell is only casted, when there is an enemy in range
286  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ARCANE_BLAST : SPELL_ARCANE_BLAST_H) == CAST_OK)
287  m_uiArcaneBlastTimer = urand(7000, 9000);
288  }
289  else
290  m_uiArcaneBlastTimer -= uiDiff;
291 
292  if (m_uiBlizzardTimer < uiDiff)
293  {
294  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
295  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_BLIZZARD : SPELL_BLIZZARD_H) == CAST_OK)
296  m_uiBlizzardTimer = urand(9000, 13500);
297  }
298  else
299  m_uiBlizzardTimer -= uiDiff;
300 
301  if (m_uiWrathTimer < uiDiff)
302  {
303  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
304  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_WRATH_OF_MISERY : SPELL_WRATH_OF_MISERY_H) == CAST_OK)
305  m_uiWrathTimer = urand(12500, 17200);
306  }
307  else
308  m_uiWrathTimer -= uiDiff;
309 
310  if (!m_creature->IsNonMeleeSpellCasted(true)) // TODO Use this additional check, because might want to change the random target to be a target that is in LoS (which then is expensive)
311  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
312  DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_FROSTBOLT : SPELL_FROSTBOLT_H);
313 
314  break;
315  }
316  }

376 // Handling of the dummy auras of Crystal Handler Death spells, on remove the Crystal needs to be opened
377 bool EffectAuraDummy_npc_crystal_channel_target(const Aura* pAura, bool bApply)
378 {
379  for (uint8 i = 0; i < MAX_CRYSTALS; ++i)
380  {
381  if (pAura->GetId() == aCrystalHandlerDeathSpells[i])
382  {
383  if (pAura->GetEffIndex() == EFFECT_INDEX_0 && !bApply)
384  {
385  if (Creature* pCreature = (Creature*)pAura->GetTarget())
386  {
387  if (instance_draktharon_keep* pInstance = (instance_draktharon_keep*)pCreature->GetInstanceData())
388  {
389  if (pInstance->GetData(TYPE_NOVOS) == NOT_STARTED || pInstance->GetData(TYPE_NOVOS) == FAIL)
390  return true;
391 
392  pInstance->DoHandleCrystal(i);
393 
394  // Inform Novos about removed
395  if (Creature* pNovos = pInstance->GetSingleCreatureFromStorage(NPC_NOVOS))
396  if (boss_novosAI* pNovosAI = dynamic_cast<boss_novosAI*>(pNovos->AI()))
397  pNovosAI->LostOneCrystal();
398  }
399  }
400  }
401 
402  return true;
403  }
404  }
405 
406  return false;
407 }
