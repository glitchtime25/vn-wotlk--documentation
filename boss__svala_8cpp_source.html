147  void ExecuteAction(uint32 action) override
148  {
149  switch (action)
150  {
151  case SVALA_ACTION_SINISTER_STRIKE:
152  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_SINISTER_STRIKE : SPELL_SINISTER_STRIKE_H) == CAST_OK)
153  ResetCombatAction(action, 10000);
154  break;
155  case SVALA_ACTION_CALL_FLAMES:
156  if (DoCastSpellIfCan(m_creature, SPELL_CALL_FLAMES) == CAST_OK)
157  ResetCombatAction(action, urand(15000, 20000));
158  break;
159  case SVALA_ACTION_RITUAL_STRIKE:
160  if (DoCastSpellIfCan(m_creature, SPELL_RITUAL_STRIKE, CAST_TRIGGERED) == CAST_OK)
161  DisableCombatAction(action);
162  break;
163  case SVALA_ACTION_RITUAL_OF_SWORD:
164  if (m_creature->GetHealthPercent() < 50.0f)
165  {
166  if (DoCastSpellIfCan(m_creature, SPELL_RITUAL_SELECTION) == CAST_OK)
167  {
168  // summon channelers
169  DoCastSpellIfCan(m_creature, SPELL_SUMMON_CHANNELER_1, CAST_TRIGGERED);
170  DoCastSpellIfCan(m_creature, SPELL_SUMMON_CHANNELER_2, CAST_TRIGGERED);
171  DoCastSpellIfCan(m_creature, SPELL_SUMMON_CHANNELER_3, CAST_TRIGGERED);
172 
173  // disarm and teleport the boss
174  DoCastSpellIfCan(m_creature, SPELL_RITUAL_DISARM, CAST_TRIGGERED);
175  DoCastSpellIfCan(m_creature, SPELL_RITUAL_OF_SWORD, CAST_TRIGGERED);
176 
177  switch (urand(0, 3))
178  {
179  case 0: DoScriptText(SAY_SACRIFICE_1, m_creature); break;
180  case 1: DoScriptText(SAY_SACRIFICE_2, m_creature); break;
181  case 2: DoScriptText(SAY_SACRIFICE_3, m_creature); break;
182  case 3: DoScriptText(SAY_SACRIFICE_4, m_creature); break;
183  }
184 
185  SetActionReadyStatus(action, false);
186  ResetCombatAction(SVALA_ACTION_RITUAL_STRIKE, 1000);
187  }
188  }
189  break;
190  }
191  }
