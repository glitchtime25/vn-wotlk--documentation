172 ObjectAccessor::RemoveCorpse(Corpse* corpse)
173 {
174  MANGOS_ASSERT(corpse && corpse->GetType() != CORPSE_BONES);
175 
176  Guard guard(i_corpseGuard);
177  Player2CorpsesMapType::iterator iter = i_player2corpse.find(corpse->GetOwnerGuid());
178  if (iter == i_player2corpse.end())
179  return;
180 
181  // build mapid*cellid -> guid_set map
182  CellPair cell_pair = MaNGOS::ComputeCellPair(corpse->GetPositionX(), corpse->GetPositionY());
183  uint32 cell_id = (cell_pair.y_coord * TOTAL_NUMBER_OF_CELLS_PER_MAP) + cell_pair.x_coord;
184 
185  sObjectMgr.DeleteCorpseCellData(corpse->GetMapId(), cell_id, corpse->GetOwnerGuid().GetCounter());
186  corpse->RemoveFromWorld();
187 
188  i_player2corpse.erase(iter);
189 }

229 Corpse*
230 ObjectAccessor::ConvertCorpseForPlayer(ObjectGuid player_guid, bool insignia)
231 {
232  Corpse* corpse = GetCorpseForPlayerGUID(player_guid);
233  if (!corpse)
234  {
235  // in fact this function is called from several places
236  // even when player doesn't have a corpse, not an error
237  // sLog.outError("Try remove corpse that not in map for GUID %ul", player_guid);
238  return nullptr;
239  }
240 
241  DEBUG_LOG("Deleting Corpse and spawning bones.");
242 
243  // remove corpse from player_guid -> corpse map
244  RemoveCorpse(corpse);
245 
246  // remove resurrectable corpse from grid object registry (loaded state checked into call)
247  // do not load the map if it's not loaded
248  Map* map = sMapMgr.FindMap(corpse->GetMapId(), corpse->GetInstanceId());
249  if (map)
250  map->Remove(corpse, false);
251 
252  // remove corpse from DB
253  corpse->DeleteFromDB();
254 
255  Corpse* bones = nullptr;
256  // create the bones only if the map and the grid is loaded at the corpse's location
257  // ignore bones creating option in case insignia
258  if (map && (insignia ||
259  (map->IsBattleGroundOrArena() ? sWorld.getConfig(CONFIG_BOOL_DEATH_BONES_BG_OR_ARENA) : sWorld.getConfig(CONFIG_BOOL_DEATH_BONES_WORLD))) &&
260  !map->IsRemovalGrid(corpse->GetPositionX(), corpse->GetPositionY()))
261  {
262  // Create bones, don't change Corpse
263  bones = new Corpse;
264  bones->Create(corpse->GetGUIDLow());
265 
266  for (int i = 3; i < CORPSE_END; ++i) // don't overwrite guid and object type
267  bones->SetUInt32Value(i, corpse->GetUInt32Value(i));
268 
269  bones->SetGrid(corpse->GetGrid());
270  // bones->m_time = m_time; // don't overwrite time
271  // bones->m_inWorld = m_inWorld; // don't overwrite world state
272  // bones->m_type = m_type; // don't overwrite type
273  bones->Relocate(corpse->GetPositionX(), corpse->GetPositionY(), corpse->GetPositionZ(), corpse->GetOrientation());
274  bones->SetPhaseMask(corpse->GetPhaseMask(), false);
275 
276  bones->SetUInt32Value(CORPSE_FIELD_FLAGS, CORPSE_FLAG_UNK2 | CORPSE_FLAG_BONES);
277  bones->SetOwnerGuid(player_guid);
278 
279  for (int i = 0; i < EQUIPMENT_SLOT_END; ++i)
280  {
281  if (corpse->GetUInt32Value(CORPSE_FIELD_ITEM + i))
282  bones->SetUInt32Value(CORPSE_FIELD_ITEM + i, 0);
283  }
284 
285  // add bones in grid store if grid loaded where corpse placed
286  map->Add(bones);
287  }
288 
289  // all references to the corpse should be removed at this point
290  delete corpse;
291 
292  return bones;
293 }
