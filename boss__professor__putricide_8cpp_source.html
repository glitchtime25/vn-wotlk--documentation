314  void UpdateAI(const uint32 uiDiff) override
315  {
316  // no attacking during the Festergut / Rotface encounters
317  if (m_creature->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE))
318  return;
319 
320  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
321  return;
322 
323  if (!m_pInstance)
324  return;
325 
326  // Enrage
327  if (m_uiEnrageTimer)
328  {
329  if (m_uiEnrageTimer <= uiDiff)
330  {
331  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
332  {
333  DoScriptText(SAY_BERSERK, m_creature);
334  m_uiEnrageTimer = 0;
335  }
336  }
337  else
338  m_uiEnrageTimer -= uiDiff;
339  }
340 
341  switch (m_uiPhase)
342  {
343  case PHASE_TWO:
344  case PHASE_THREE:
345  {
346  // Choking Gas
347  if (m_uiChokingGasBombTimer < uiDiff)
348  {
349  if (DoCastSpellIfCan(m_creature, SPELL_CHOKING_GAS_BOMB) == CAST_OK)
350  m_uiChokingGasBombTimer = 35000;
351  }
352  else
353  m_uiChokingGasBombTimer -= uiDiff;
354 
355  // Malleable Goo
356  if (m_uiMalleableGooTimer < uiDiff)
357  {
358  if (DoCastSpellIfCan(m_creature, SPELL_MALLEABLE_GOO) == CAST_OK)
359  m_uiMalleableGooTimer = 25000;
360  }
361  else
362  m_uiMalleableGooTimer -= uiDiff;
363 
364  if (m_uiPhase == PHASE_THREE)
365  break;
366  // else no break;
367  }
368  case PHASE_ONE:
369  {
370  if (m_uiTransitionTimer)
371  {
372  if (m_uiTransitionTimer <= uiDiff)
373  {
374  DoCastSpellIfCan(m_creature, SPELL_TEAR_GAS_PERIODIC, CAST_TRIGGERED);
375 
376  float fX, fY, fZ;
377  SetCombatMovement(false);
378 
379  m_creature->GetMotionMaster()->Clear();
380  m_creature->GetRespawnCoord(fX, fY, fZ);
381  m_creature->GetMotionMaster()->MovePoint(POINT_PUTRICIDE_SPAWN, fX, fY, fZ);
382 
383  m_uiPhase = (m_uiPhase == PHASE_ONE ? PHASE_RUNNING_ONE : PHASE_RUNNING_TWO);
384  m_uiTransitionTimer = 0;
385  }
386  else
387  m_uiTransitionTimer -= uiDiff;
388 
389  // Preparing for phase transition
390  return;
391  }
392 
393  // Unbound Plague
394  if (m_pInstance->IsHeroicDifficulty())
395  {
396  if (m_uiUnboundPlagueTimer < uiDiff)
397  {
398  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_UNBOUND_PLAGUE, SELECT_FLAG_PLAYER))
399  {
400  if (DoCastSpellIfCan(pTarget, SPELL_UNBOUND_PLAGUE) == CAST_OK)
401  m_uiUnboundPlagueTimer = 70000;
402  }
403  }
404  else
405  m_uiUnboundPlagueTimer -= uiDiff;
406  }
407 
408  // Unstable Experiment
409  if (m_uiUnstableExperimentTimer < uiDiff)
410  {
411  if (DoCastSpellIfCan(m_creature, SPELL_UNSTABLE_EXPERIMENT) == CAST_OK)
412  {
413  DoScriptText(EMOTE_EXPERIMENT, m_creature);
414  m_uiUnstableExperimentTimer = 37000;
415  }
416  }
417  else
418  m_uiUnstableExperimentTimer -= uiDiff;
419 
420  // health check
421  if (m_creature->GetHealthPercent() <= (m_uiPhase == PHASE_ONE ? 80.0f : 35.0f))
422  {
423  uint32 spellId = (m_pInstance->IsHeroicDifficulty() ? SPELL_VOLATILE_EXPERIMENT : SPELL_TEAR_GAS);
424 
425  if (DoCastSpellIfCan(m_creature, spellId, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
426  {
427  m_uiTransitionTimer = 2500;
428  return;
429  }
430  }
431 
432  break;
433  }
434  // Transition phases
435  case PHASE_TRANSITION_ONE:
436  case PHASE_TRANSITION_TWO:
437  {
438  if (m_uiTransitionTimer < uiDiff)
439  {
440  SetCombatMovement(true);
441  m_creature->GetMotionMaster()->Clear();
442  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
443 
444  DoScriptText((m_uiPhase == PHASE_TRANSITION_ONE ? SAY_TRANSFORM_1 : SAY_TRANSFORM_2), m_creature);
445 
446  // in heroic it changes form at the end of the transition
447  if (!m_pInstance->IsHeroicDifficulty())
448  DoCastSpellIfCan(m_creature, SPELL_TEAR_GAS_CANCEL, CAST_TRIGGERED);
449 
450  m_uiPhase = (m_uiPhase == PHASE_TRANSITION_ONE ? PHASE_TWO : PHASE_THREE);
451  m_uiTransitionTimer = 0;
452  }
453  else
454  m_uiTransitionTimer -= uiDiff;
455 
456  return;
457  }
458  // Movement transition phases
459  case PHASE_RUNNING_ONE:
460  case PHASE_RUNNING_TWO:
461  {
462  // wait for arriving at the table (during phase transition)
463  return;
464  }
465  }
466 
467  // Slime Puddle (all phases, except for transition
468  if (m_uiPuddleTimer < uiDiff)
469  {
470  if (DoCastSpellIfCan(m_creature, SPELL_SLIME_PUDDLE_TRIGGER) == CAST_OK)
471  m_uiPuddleTimer = 35000;
472  }
473  else
474  m_uiPuddleTimer -= uiDiff;
475 
476  DoMeleeAttackIfReady();
477  }

485 bool EffectScriptEffectCreature_npc_putricide(Unit* pCaster, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
486 {
487  if (pCreatureTarget->GetEntry() != NPC_PROFESSOR_PUTRICIDE)
488  return false;
489 
490  // Ooze flood trigger - used in Rotface encounter
491  if (uiSpellId == SPELL_OOZE_FLOOD_TRIGGER && uiEffIndex == EFFECT_INDEX_0)
492  {
493  instance_icecrown_citadel* pInstance = (instance_icecrown_citadel*)pCreatureTarget->GetInstanceData();
494  if (!pInstance)
495  return false;
496 
497  // Defind the target as one of the stalkers from the top of the taps
498  GuidList lStalkersGuidList;
499  pInstance->GetRotfaceStalkersList(lStalkersGuidList);
500 
501  std::vector<Creature*> vTapStalkers;
502  vTapStalkers.reserve(lStalkersGuidList.size());
503 
504  for (GuidList::const_iterator itr = lStalkersGuidList.begin(); itr != lStalkersGuidList.end(); ++itr)
505  {
506  if (Creature* pStalker = pCreatureTarget->GetMap()->GetCreature(*itr))
507  vTapStalkers.push_back(pStalker);
508  }
509 
510  if (vTapStalkers.empty())
511  {
512  script_error_log("Instance Icecrown Citadel: ERROR Failed to properly find creature %u for Ooze Flood event.", NPC_PUDDLE_STALKER);
513  return false;
514  }
515 
516  // pick random target of the tap stalkers
517  Creature* pTarget = vTapStalkers[urand(0, vTapStalkers.size() - 1)];
518  if (!pTarget)
519  return false;
520 
521  // get the nearest twin tap stalker
522  Creature* pNearTarget = nullptr;
523  CreatureList lTargetsInRange;
524  GetCreatureListWithEntryInGrid(lTargetsInRange, pTarget, pTarget->GetEntry(), 30.0f);
525 
526  if (lTargetsInRange.empty())
527  return false;
528 
529  // find only the nearest tap trigger
530  for (CreatureList::const_iterator itr = lTargetsInRange.begin(); itr != lTargetsInRange.end(); ++itr)
531  {
532  if ((*itr)->GetPositionZ() > 370.0f)
533  pNearTarget = *itr;
534  }
535 
536  if (!pNearTarget)
537  return false;
538 
539  // cast the triggered spell on each target
540  pCreatureTarget->CastSpell(pTarget, GetSpellStore()->LookupEntry<SpellEntry>(uiSpellId)->CalculateSimpleValue(uiEffIndex), TRIGGERED_OLD_TRIGGERED);
541  pCreatureTarget->CastSpell(pNearTarget, GetSpellStore()->LookupEntry<SpellEntry>(uiSpellId)->CalculateSimpleValue(uiEffIndex), TRIGGERED_OLD_TRIGGERED);
542  DoScriptText(urand(0, 1) ? SAY_SLIME_FLOW_1 : SAY_SLIME_FLOW_2, pCreatureTarget);
543 
544  return true;
545  }
546  if (uiSpellId == SPELL_UNSTABLE_EXPERIMENT && uiEffIndex == EFFECT_INDEX_0)
547  {
548  pCreatureTarget->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, pCaster, pCreatureTarget);
549  return true;
550  }
551 
552  return false;
553 }
