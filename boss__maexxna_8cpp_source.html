229 struct WebWrapMaexxna : public SpellScript

231  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
232  {
233  uint32 targetCount = spell->m_spellInfo->Id == SPELL_WEBWRAP ? MAX_PLAYERS_WEB_WRAP : MAX_PLAYERS_WEB_WRAP_H;
234  std::vector<uint32> targetSpells{ 28617, 29280, 29281, 29282, 29283, 29285, 29287 };
235  std::vector<Unit*> unitList;
236  spell->GetCaster()->SelectAttackingTargets(unitList, ATTACKING_TARGET_ALL_SUITABLE, 1, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK);
237  std::shuffle(unitList.begin(), unitList.end(), *GetRandomGenerator());
238  unitList.resize(targetCount);
239  for (Unit* target : unitList)
240  {
241  uint32 spellId = targetSpells[urand(0, targetSpells.size() - 1)];
242  targetSpells.erase(std::remove(targetSpells.begin(), targetSpells.end(), spellId), targetSpells.end());
243  target->CastSpell(nullptr, spellId, TRIGGERED_OLD_TRIGGERED);
244  }
245  }
