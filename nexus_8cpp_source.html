27 bool GOUse_go_containment_sphere(Player* /*pPlayer*/, GameObject* pGo)
28 {
29  ScriptedInstance* pInstance = (ScriptedInstance*)pGo->GetInstanceData();
30 
31  if (!pInstance)
32  return false;
33 
34  switch (pGo->GetEntry())
35  {
36  case GO_CONTAINMENT_SPHERE_TELESTRA: pInstance->SetData(TYPE_TELESTRA, SPECIAL); break;
37  case GO_CONTAINMENT_SPHERE_ANOMALUS: pInstance->SetData(TYPE_ANOMALUS, SPECIAL); break;

43 
44  if (Creature* pCaster = GetClosestCreatureWithEntry(pGo, NPC_BREATH_CASTER, 20.0f))

46 
47  return false;
48 }

130 void instance_nexus::SetData(uint32 uiType, uint32 uiData)
131 {
132  switch (uiType)
133  {
134  case TYPE_TELESTRA:
135  m_auiEncounter[uiType] = uiData;
136  if (uiData == IN_PROGRESS)
137  SetSpecialAchievementCriteria(TYPE_ACHIEV_SPLIT_PERSONALITY, true);
138  if (uiData == DONE)
139  DoToggleGameObjectFlags(GO_CONTAINMENT_SPHERE_TELESTRA, GO_FLAG_NO_INTERACT, false);
140  break;
141  case TYPE_ANOMALUS:
142  m_auiEncounter[uiType] = uiData;
143  if (uiData == IN_PROGRESS)
144  SetSpecialAchievementCriteria(TYPE_ACHIEV_CHAOS_THEORY, true);
145  if (uiData == DONE)
146  DoToggleGameObjectFlags(GO_CONTAINMENT_SPHERE_ANOMALUS, GO_FLAG_NO_INTERACT, false);
147  break;
148  case TYPE_ORMOROK:
149  m_auiEncounter[uiType] = uiData;
150  if (uiData == DONE)
151  DoToggleGameObjectFlags(GO_CONTAINMENT_SPHERE_ORMOROK, GO_FLAG_NO_INTERACT, false);
152  break;
153  case TYPE_KERISTRASZA:
154  m_auiEncounter[uiType] = uiData;
155  if (uiData == IN_PROGRESS)
156  m_sIntenseColdFailPlayers.clear();
157  break;
158  case TYPE_INTENSE_COLD_FAILED:
159  // Insert the players who fail the achiev and haven't been already inserted in the set
160  if (m_sIntenseColdFailPlayers.find(uiData) == m_sIntenseColdFailPlayers.end())
161  m_sIntenseColdFailPlayers.insert(uiData);
162  break;
163  default:
164  script_error_log("Instance Nexus: ERROR SetData = %u for type %u does not exist/not implemented.", uiType, uiData);
165  return;
166  }
167 
168  if (m_auiEncounter[TYPE_TELESTRA] == SPECIAL && m_auiEncounter[TYPE_ANOMALUS] == SPECIAL && m_auiEncounter[TYPE_ORMOROK] == SPECIAL && uiType != TYPE_KERISTRASZA)
169  {
170  // release Keristrasza from her prison here
171  SetData(TYPE_KERISTRASZA, SPECIAL);
172 
173  Creature* pCreature = GetSingleCreatureFromStorage(NPC_KERISTRASZA);
174  if (pCreature && pCreature->IsAlive())
175  {
176  pCreature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC | UNIT_FLAG_IMMUNE_TO_PLAYER);
177  pCreature->RemoveAurasDueToSpell(SPELL_FROZEN_PRISON);
178  }
179  }
180 
181  if (uiData == DONE || uiData == SPECIAL)
182  {
183  OUT_SAVE_INST_DATA;
184 
185  std::ostringstream saveStream;
186  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
187 
188  m_strInstData = saveStream.str();
189 
190  SaveToDB();
191  OUT_SAVE_INST_DATA_COMPLETE;
192  }
193 }
