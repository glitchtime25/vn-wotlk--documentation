108 void instance_scarlet_monastery::SetData(uint32 uiType, uint32 uiData)
109 {
110  if (uiType == TYPE_MOGRAINE_AND_WHITE_EVENT)
111  {
112  if (uiData == IN_PROGRESS)
113  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
114  if (uiData == FAIL)
115  {
116  Creature* pWhitemane = GetSingleCreatureFromStorage(NPC_WHITEMANE);
117  if(!pWhitemane)
118  return;
119  Creature* pMograine = GetSingleCreatureFromStorage(NPC_MOGRAINE);
120  if(!pMograine)
121  return;
122  if(pWhitemane->IsAlive() && pMograine->IsAlive())
123  {
124  pWhitemane->SetRespawnDelay(30, true);
125  pWhitemane->ForcedDespawn();
126  pMograine->SetRespawnDelay(30, true);
127  pMograine->ForcedDespawn();
128  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
129  m_auiEncounter[0] = NOT_STARTED;
130  return;
131  }
132  if(!pMograine->IsAlive() && pWhitemane->IsAlive())
133  {
134  pWhitemane->SetRespawnDelay(30, true);
135  pWhitemane->ForcedDespawn();
136  DoUseDoorOrButton(GO_WHITEMANE_DOOR);
137  m_auiEncounter[0] = uiData;
138  return;
139  }
140  if(!pWhitemane->IsAlive() && pMograine->IsAlive())
141  {
142  pMograine->ForcedDespawn();
143  m_auiEncounter[0] = uiData;
144  return;
145  }
146  }
147 
148  m_auiEncounter[0] = uiData;
149  }
150  else if (uiType == TYPE_ASHBRINGER_EVENT)
151  {
152  if (uiData == IN_PROGRESS)
153  {
154  DoUseDoorOrButton(GO_CHAPEL_DOOR);
155 
156  Creature* whitemane = GetSingleCreatureFromStorage(NPC_WHITEMANE, true);
157  if (whitemane && whitemane->IsAlive() && !whitemane->IsInCombat())
158  whitemane->ForcedDespawn();
159 
160  for (auto scarletCathedralNpcGuid : m_sAshbringerFriendlyGuids)
161  if (Creature* scarletNpc = instance->GetCreature(scarletCathedralNpcGuid))
162  if (scarletNpc->IsAlive() && !scarletNpc->IsInCombat())
163  scarletNpc->setFaction(35);
164  }
165  m_auiEncounter[1] = uiData;
166  }
167 }

202
