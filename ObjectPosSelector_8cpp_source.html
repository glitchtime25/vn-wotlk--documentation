55 void ObjectPosSelector::AddUsedArea(WorldObject const* obj, float angle, float dist)
56 {
57  MANGOS_ASSERT(obj);
58 
59  // skip some unexpected results.
60  if (dist == 0.0f)
61  return;
62 
63  // (half) angle that obj occupies
64  float sr_angle = atan(OCCUPY_POS_ANGLE_ATAN_FACTOR * obj->GetObjectBoundingRadius() / dist);
65 
66  if (angle >= 0)
67  m_UsedAreaLists[USED_POS_PLUS].insert(UsedArea(angle, OccupiedArea(sr_angle, obj)));

113 void ObjectPosSelector::InitializeAngle(UsedAreaSide side)
114 {
115  m_nextUsedAreaItr[side] = m_UsedAreaLists[side].begin();
116 
117  // if another side not alow use 0.0f angle calculate possible value in 0..m_searchedForReqHAngle range
118  if (!m_UsedAreaLists[~side].empty())
119  {
120  UsedArea const& otherArea = *m_UsedAreaLists[~side].begin();
121  m_stepAngle[side] = std::max(m_searchedForReqHAngle + otherArea.second.angleOffset - otherArea.first, 0.0f);
122  }
123  else // Other side empty. start from 0
124  m_stepAngle[side] = 0.0f;
125 
126  // As m_stepAngle will be incremented first in ::NextSideAngle
127  m_stepAngle[side] -= m_searchedForReqHAngle;
128 }

137 bool ObjectPosSelector::NextAngle(float& angle)
138 {
139  // loop until both side fail and leave 0..PI
140  for (;;)
141  {
142  // ++ direction less updated
143  if (m_stepAngle[USED_POS_PLUS] < M_PI_F && m_stepAngle[USED_POS_PLUS] <= m_stepAngle[USED_POS_MINUS])
144  {
145  if (NextSideAngle(USED_POS_PLUS, angle))
146  return true;
147  }
148  // -- direction less updated
149  else if (m_stepAngle[USED_POS_MINUS] < M_PI_F)
150  {
151  if (NextSideAngle(USED_POS_MINUS, angle))
152  return true;
153  }
154  // both sides finishes
155  else
156  break;
157  }
158 
159  // no angles
160  return false;
161 }

172 bool ObjectPosSelector::NextSideAngle(UsedAreaSide side, float& angle)
173 {
174  // next possible angle
175  m_stepAngle[side] += (m_searchedForReqHAngle + 0.01);
176 
177  // prevent jump to another side
178  if (m_stepAngle[side] > M_PI_F)
179  return false;
180 
181  // no used area anymore on this side
182  if (m_nextUsedAreaItr[side] == m_UsedAreaLists[side].end())
183  {
184  angle = m_stepAngle[side] * SignOf(side);
185  return true;
186  }
187 
188  // Already occupied and no better found
189  if ((m_searchPosFor && m_nextUsedAreaItr[side]->second.occupyingObj == m_searchPosFor) ||
190  // Next occupied is too far away
191  (m_stepAngle[side] + m_searchedForReqHAngle < m_nextUsedAreaItr[side]->first - m_nextUsedAreaItr[side]->second.angleOffset))
192  {
193  angle = m_stepAngle[side] * SignOf(side);
194  return true;
195  }
196 
197  // angle set at first possible pos after passed m_nextUsedAreaItr
198  m_stepAngle[side] = m_nextUsedAreaItr[side]->first + m_nextUsedAreaItr[side]->second.angleOffset;
199 
200  ++m_nextUsedAreaItr[side];
201 
202  return false;
203 }
