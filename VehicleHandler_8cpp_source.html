70 void WorldSession::HandleRequestVehicleNextSeat(WorldPacket& recvPacket)
71 {
72  if (recvPacket.GetOpcode() == CMSG_REQUEST_VEHICLE_PREV_SEAT)
73  DEBUG_LOG("WORLD: Received opcode CMSG_REQUEST_VEHICLE_PREV_SEAT");
74  else
75  DEBUG_LOG("WORLD: Received opcode CMSG_REQUEST_VEHICLE_NEXT_SEAT");
76 
77  TransportInfo* transportInfo = _player->GetTransportInfo();
78  if (!transportInfo || !transportInfo->IsOnVehicle())
79  return;
80 
81  Unit* vehicle = static_cast<Unit*>(transportInfo->GetTransport());
82 
83  switch (recvPacket.GetOpcode())
84  {
85  case CMSG_REQUEST_VEHICLE_PREV_SEAT:
86  vehicle->GetVehicleInfo()->ChangeSeat(_player, transportInfo->GetTransportSeat(), false);
87  break;
88  case CMSG_REQUEST_VEHICLE_NEXT_SEAT:
89  vehicle->GetVehicleInfo()->ChangeSeat(_player, transportInfo->GetTransportSeat(), true);
90  break;
91 
92  default:
93  break;
94  }
95 }

97 void WorldSession::HandleRequestVehicleSwitchSeat(WorldPacket& recvPacket)
98 {
99  DEBUG_LOG("WORLD: Received opcode CMSG_REQUEST_VEHICLE_SWITCH_SEAT");
100  recvPacket.hexlike();
101 
102  ObjectGuid vehicleGuid;
103  uint8 seat;
104 
105  recvPacket >> vehicleGuid.ReadAsPacked();
106  recvPacket >> seat;
107 
108  TransportInfo* transportInfo = _player->GetTransportInfo();
109  if (!transportInfo || !transportInfo->IsOnVehicle())
110  return;
111 
112  Unit* vehicle = (Unit*)transportInfo->GetTransport();
113 
114  if (vehicleGuid != vehicle->GetObjectGuid())
115  {
116  Unit* destVehicle = _player->GetMap()->GetUnit(vehicleGuid);
117 
118  if (!destVehicle || !destVehicle->IsVehicle())
119  return;
120 
121  // Change vehicle is not possible
122  if (destVehicle->GetVehicleInfo()->GetVehicleEntry()->m_flags & VEHICLE_FLAG_DISABLE_SWITCH)
123  return;
124 
125  SpellClickInfoMapBounds clickPair = sObjectMgr.GetSpellClickInfoMapBounds(destVehicle->GetEntry());
126  for (SpellClickInfoMap::const_iterator itr = clickPair.first; itr != clickPair.second; ++itr)
127  if (itr->second.IsFitToRequirements(_player, destVehicle->GetTypeId() == TYPEID_UNIT ? (Creature*)destVehicle : nullptr))
128  _player->CastSpell(destVehicle, itr->second.spellId, TRIGGERED_OLD_TRIGGERED);
129  }
130  else
131  vehicle->GetVehicleInfo()->SwitchSeat(_player, seat);
132 }
