369  {
370  UpdateTimers(uiDiff, m_creature->IsInCombat());
371 
372  // Return since we have no target
373  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
374  return;
375 
376  // spell will target dragons, if they are still alive at 35%
377  if (!m_bIsBerserk && m_creature->GetHealthPercent() < 35.0f)
378  {
379  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
380  {
381  DoScriptText(SAY_SARTHARION_BERSERK, m_creature);
382  m_bIsBerserk = true;
383  }
384  }
385 
386  // soft enrage
387  if (!m_bIsSoftEnraged && m_creature->GetHealthPercent() <= 10.0f)
388  {
389  // TODO
390  m_bIsSoftEnraged = true;
391  }
392 
393  // hard enrage
394  if (!m_bIsHardEnraged)
395  {
396  if (m_uiEnrageTimer < uiDiff)
397  {
398  if (DoCastSpellIfCan(m_creature, SPELL_PYROBUFFET, CAST_TRIGGERED) == CAST_OK)
399  m_bIsHardEnraged = true;
400  }
401  else
402  m_uiEnrageTimer -= uiDiff;
403  }
404 
405  // flame tsunami

407  {
408  SendFlameTsunami();
409  m_uiFlameTsunamiTimer = 33000;
410  }

413
