106 struct Icecrown25MeleeTrinket : public AuraScript
107 {
108  void OnAuraInit(Aura* aura) const override
109  {
110  Unit* target = aura->GetTarget();
111  aura->SetScriptValue(classMap.find(Classes(target->getClass())) == classMap.end());
112  if (aura->GetScriptValue() == 1)
113  sLog.outError("Icecrown25MeleeTrinket called for unknown class %u", target->getClass());
114  }
115 
116  SpellAuraProcResult OnProc(Aura* aura, ProcExecutionData& procData) const override
117  {
118  if (aura->GetScriptValue() == 1)
119  return SPELL_AURA_PROC_FAILED;
120 
121  bool heroic = false;
122  if (aura->GetSpellProto()->Id == 71562)
123  heroic = true;
124  Unit* caster = aura->GetCaster();
125  if (!caster)
126  return SPELL_AURA_PROC_FAILED;
127  Classes characterClass = Classes(caster->getClass());
128  std::unordered_map<Classes, const std::vector<uint32>>::const_iterator itr;
129  if (heroic)
130  itr = classMapHeroic.find(characterClass);
131  else
132  itr = classMap.find(characterClass);
133  const std::vector<uint32>& spellArray = itr->second;
134  if (!caster->IsAlive() || spellArray.empty())
135  return SPELL_AURA_PROC_FAILED;
136  uint32 spellToCast = spellArray[urand(0, spellArray.size() - 1)];
137  procData.triggeredSpellId = spellToCast;
138  procData.cooldown = std::chrono::seconds(1min + 45s).count();
139  return SPELL_AURA_PROC_OK;
140  }
141 };
