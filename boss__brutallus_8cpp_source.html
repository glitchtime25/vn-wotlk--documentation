301  if (Creature* madrigosa = m_instance->GetSingleCreatureFromStorage(NPC_MADRIGOSA))

303  madrigosa->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC);

306  DisableCombatAction(BRUTALLUS_SLASH);
307  DisableCombatAction(BRUTALLUS_STOMP);
308  DisableCombatAction(BRUTALLUS_BURN);
309  DisableCombatAction(BRUTALLUS_LOVE);
310  }
311  break;
312  case SPELL_FROST_BREATH:

333  madrigosa->CastSpell(m_creature, SPELL_FROST_BLAST, TRIGGERED_OLD_TRIGGERED);

342  break;
343  case SPELL_FEL_FIREBALL:
344  // Spell has script target Madrigosa

347  case POINT_MOVE_GROUND:

349  madrigosa->GetMotionMaster()->MovePoint(POINT_MOVE_FEL_FIREBALL, aMadrigosaLoc[2]);

353  m_creature->SetImmobilizedState(false);

355  m_creature->SetLevitate(true);
356  SetCombatMovement(false);

363  break;

366  m_creature->GetMotionMaster()->MovePoint(POINT_MOVE_ENCAPSULATE, Position(1463.13f, 578.7533f, 44.51308f, 0.f), FORCED_MOVEMENT_RUN);

370  m_creature->GetMotionMaster()->MoveFall();

373  m_creature->SetIgnoreMMAP(false);
374  m_creature->SetHover(false);
375  m_creature->SetLevitate(false);

430  void ExecuteAction(uint32 action) override
431  {
432  switch (action)
433  {
434  case BRUTALLUS_BERSERK:
435  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
436  {
437  DoScriptText(YELL_BERSERK, m_creature);
438  DisableCombatAction(action);
439  }
440  break;
441  case BRUTALLUS_SLASH:
442  if (DoCastSpellIfCan(nullptr, SPELL_METEOR_SLASH) == CAST_OK)
443  ResetCombatAction(action, 11000);
444  break;
445  case BRUTALLUS_STOMP:
446  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_STOMP) == CAST_OK)
447  ResetCombatAction(action, 30000);
448  break;
449  case BRUTALLUS_BURN:
450  // different spell in choice due to requiring no aura check
451  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_BURN_DAMAGE, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA | SELECT_FLAG_SKIP_TANK))
452  if (DoCastSpellIfCan(target, SPELL_BURN) == CAST_OK)
453  ResetCombatAction(action, 20000);
454  break;
455  case BRUTALLUS_LOVE:
456  switch (urand(0, 4))
457  {
458  case 0: DoScriptText(YELL_LOVE1, m_creature); break;
459  case 1: DoScriptText(YELL_LOVE2, m_creature); break;
460  case 2: DoScriptText(YELL_LOVE3, m_creature); break;
461  default: break;
462  }
463  ResetCombatAction(action, urand(15000, 23000));
464  break;
465  }
