132  void UpdateEscortAI(const uint32 uiDiff) override
133  {
134  // Combat check
135  if (m_creature->SelectHostileTarget() && m_creature->GetVictim())
136  {
137  if (m_uiDeathCoilTimer < uiDiff)
138  {
139  DoScriptText(SAY_SPELL, m_creature);
140  m_creature->CastSpell(m_creature->GetVictim(), SPELL_DEATH_COIL, TRIGGERED_NONE);
141  m_uiDeathCoilTimer = 4000;
142  }
143  else
144  m_uiDeathCoilTimer -= uiDiff;
145 
146  DoMeleeAttackIfReady();
147  }
148  else
149  {
150  // Out of combat but being escorted
151  if (HasEscortState(STATE_ESCORT_ESCORTING))
152  {
153  if (m_uiChatTimer < uiDiff)
154  {
155  if (m_creature->HasAura(SPELL_ELIXIR_OF_FORTITUDE, EFFECT_INDEX_0))
156  {
157  DoScriptText(SAY_RAND_1, m_creature);
158  m_creature->CastSpell(m_creature, SPELL_BLUE_FIREWORK, TRIGGERED_NONE);
159  }
160  else
161  {
162  DoScriptText(SAY_RAND_2, m_creature);
163  m_creature->CastSpell(m_creature, SPELL_ELIXIR_OF_FORTITUDE, TRIGGERED_NONE);
164  }
165 
166  m_uiChatTimer = 12000;
167  }
168  else
169  m_uiChatTimer -= uiDiff;
170  }
171  }
172  }

195 {

197 
198  switch (uiAction)

202 
203  if (pEscortAI)

206  case GOSSIP_ACTION_INFO_DEF+2:
207  pPlayer->CLOSE_GOSSIP_MENU();
208 
209  if (pEscortAI)

212  case GOSSIP_ACTION_INFO_DEF+3:

214 
215  if (pEscortAI)

218  default:
