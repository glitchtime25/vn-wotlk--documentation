170  void ExecuteAction(uint32 action) override
171  {
172  switch (action)
173  {
174  case MOROGRIM_PHASE_2:
175  {
176  if (m_creature->GetHealthPercent() < 25.0f)
177  {
178  DisableCombatAction(MOROGRIM_WATERY_GRAVE);
179  ResetCombatAction(MOROGRIM_WATER_GLOBULES, 1);
180  SetActionReadyStatus(action, false);
181  }
182  break;
183  }
184  case MOROGRIM_EARTHQUAKE:
185  {
186  if (DoCastSpellIfCan(nullptr, SPELL_EARTHQUAKE) == CAST_OK)
187  {
188  DoScriptText(EMOTE_EARTHQUAKE, m_creature);
189  DoScriptText(urand(0, 1) ? SAY_SUMMON1 : SAY_SUMMON2, m_creature);
190 
191  // summon murlocs - north
192  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A1, CAST_TRIGGERED);
193  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A3, CAST_TRIGGERED);
194  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A5, CAST_TRIGGERED);
195  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A7, CAST_TRIGGERED);
196  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A9, CAST_TRIGGERED);
197  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_A10, CAST_TRIGGERED);
198 
199  // summon murlocs - south
200  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_B2, CAST_TRIGGERED);
201  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_B4, CAST_TRIGGERED);
202  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_B6, CAST_TRIGGERED);

204  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_B9, CAST_TRIGGERED);
205  DoCastSpellIfCan(m_creature, SPELL_SUMMON_MURLOC_B10, CAST_TRIGGERED);
206 
207  ResetCombatAction(action, 50000);
208  }
209  break;
210  }
211  case MOROGRIM_TIDAL_WAVE:
212  {
213  if (DoCastSpellIfCan(nullptr, SPELL_TIDAL_WAVE) == CAST_OK)
214  ResetCombatAction(action, urand(20000, 25000));
215  break;
216  }
217  case MOROGRIM_WATERY_GRAVE:
218  {
219  if (DoCastSpellIfCan(nullptr, SPELL_WATERY_GRAVE) == CAST_OK)
220  {
221  DoScriptText(EMOTE_WATERY_GRAVE, m_creature);
222  ResetCombatAction(action, 30000);
223  }
224  break;
225  }
226  case MOROGRIM_WATER_GLOBULES:
227  {
228  if (!CanExecuteCombatAction())
229  return;
230 
231  for (uint8 i = 0; i < 4; ++i)
232  m_creature->CastSpell(nullptr, m_auiSpellSummonGlobule[i], TRIGGERED_OLD_TRIGGERED);
233 
234  DoScriptText(urand(0, 1) ? SAY_SUMMON_BUBL1 : SAY_SUMMON_BUBL2, m_creature);
235  DoScriptText(EMOTE_WATERY_GLOBULES, m_creature);
236  ResetCombatAction(action, 25000);
237  break;
238  }
239  }
240  }
