30 struct HunterPetScaling1 : public AuraScript
31 {
32  int32 OnAuraValueCalculate(AuraCalcData& data, int32 value) const override
33  {
34  Unit* target = data.target;
35  switch (data.effIdx)
36  {
37  case EFFECT_INDEX_0: // stamina
38  if (Unit* owner = data.caster->GetOwner())
39  {
40  float coeff = 1.f;
41  if (target->HasSpell(62762))
42  coeff = 1.4f;
43  else if (target->HasSpell(62758))
44  coeff = 1.2f;
45  value = float(owner->GetStat(STAT_STAMINA)) * 0.3f * coeff;
46  if (owner->HasAura(SPELL_INCREASED_ATTACK_POWER))
47  value += 52;
48  }
49  break;
50  case EFFECT_INDEX_1: // attack power
51  if (Unit* owner = data.caster->GetOwner())
52  {
53  float coeff = 1.f;
54  if (target->HasSpell(62762))
55  coeff = 1.3f;
56  else if (target->HasSpell(62758))
57  coeff = 1.15f;
58  value = owner->GetTotalAttackPowerValue(RANGED_ATTACK) * 0.25f * coeff;
59  if (owner->HasAura(SPELL_INCREASED_ATTACK_POWER))
60  value += 70;
61  }
62  break;
63  case EFFECT_INDEX_2: // spell power
64  if (Unit* owner = data.caster->GetOwner())
65  {
66  float coeff = 1.f;
67  if (target->HasSpell(62762))
68  coeff = 1.3f;
69  else if (target->HasSpell(62758))
70  coeff = 1.15f;
71  value = owner->GetTotalAttackPowerValue(RANGED_ATTACK) * 0.125f * coeff;
72  }
73  break;
74  }
75  return value;
76  }
