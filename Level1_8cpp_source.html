398 bool ChatHandler::HandleNamegoCommand(char* args)
399 {
400  Player* target;
401  ObjectGuid target_guid;
402  std::string target_name;
403  if (!ExtractPlayerTarget(&args, &target, &target_guid, &target_name))
404  return false;
405 
406  Player* player = m_session->GetPlayer();
407  if (target == player || target_guid == player->GetObjectGuid())
408  {
409  PSendSysMessage(LANG_CANT_TELEPORT_SELF);
410  SetSentErrorMessage(true);
411  return false;
412  }
413 
414  if (target)
415  {
416  std::string nameLink = playerLink(target_name);
417  // check online security
418  if (HasLowerSecurity(target))
419  return false;
420 
421  // make sure player is in world
422  if (!target->IsInWorld() || target->IsBeingTeleported())
423  {
424  PSendSysMessage(LANG_IS_TELEPORTED, nameLink.c_str());
425  SetSentErrorMessage(true);
426  return false;
427  }
428 
429  Map* pMap = player->GetMap();
430 
431  if (!pMap)
432  return false;
433 
434  if (pMap->IsBattleGroundOrArena())
435  {
436  // only allow if gm mode is on
437  if (!player->IsGameMaster())
438  {
439  PSendSysMessage(LANG_CANNOT_GO_TO_BG_GM, nameLink.c_str());
440  SetSentErrorMessage(true);
441  return false;
442  }
443  // if both players are in different bgs
444  if (target->GetBattleGroundId() && player->GetBattleGroundId() != target->GetBattleGroundId())
445  {
446  PSendSysMessage(LANG_CANNOT_GO_TO_BG_FROM_BG, nameLink.c_str());
447  SetSentErrorMessage(true);
448  return false;
449  }
450  // all's well, set bg id
451  // when porting out from the bg, it will be reset to 0
452  target->SetBattleGroundId(player->GetBattleGroundId(), player->GetBattleGroundTypeId());
453  // remember current position as entry point for return at bg end teleportation
454  if (!target->GetMap()->IsBattleGroundOrArena())
455  target->SetBattleGroundEntryPoint();
456  }
457  else if (pMap->IsDungeon())
458  {
459  Map* cMap = target->GetMap();
460  if (cMap->Instanceable() && cMap->GetInstanceId() != pMap->GetInstanceId())
461  {
462  // cannot summon from instance to instance
463  PSendSysMessage(LANG_CANNOT_SUMMON_TO_INST, nameLink.c_str());
464  SetSentErrorMessage(true);
465  return false;
466  }
467 
468  // we are in instance, and can summon only player in our group with us as lead
469  if (!player->GetGroup() || !target->GetGroup() ||
470  (target->GetGroup()->GetLeaderGuid() != player->GetObjectGuid()) ||
471  (player->GetGroup()->GetLeaderGuid() != player->GetObjectGuid()))
472  // the last check is a bit excessive, but let it be, just in case
473  {
474  PSendSysMessage(LANG_CANNOT_SUMMON_TO_INST, nameLink.c_str());
475  SetSentErrorMessage(true);
476  return false;
477  }
478  }
479 
480  PSendSysMessage(LANG_SUMMONING, nameLink.c_str(), "");
481  if (needReportToTarget(target))
482  ChatHandler(target).PSendSysMessage(LANG_SUMMONED_BY, playerLink(player->GetName()).c_str());
483 
484  if (!target->TaxiFlightInterrupt(!pMap->IsBattleGroundOrArena()))
485  target->SaveRecallPosition();
486 
487  // before GM
488  float x, y, z;
489  player->GetClosePoint(x, y, z, target->GetObjectBoundingRadius());
490  target->TeleportTo(player->GetMapId(), x, y, z, target->GetOrientation());
491  }
492  else
493  {
494  // check offline security
495  if (HasLowerSecurity(nullptr, target_guid))
496  return false;
497 
498  std::string nameLink = playerLink(target_name);
499 
500  PSendSysMessage(LANG_SUMMONING, nameLink.c_str(), GetMangosString(LANG_OFFLINE));
501 
502  // in point where GM stay
503  Player::SavePositionInDB(target_guid, player->GetMapId(),
504  player->GetPositionX(),
505  player->GetPositionY(),
506  player->GetPositionZ(),
507  player->GetOrientation(),
508  player->GetZoneId());
509  }
510 
511  return true;
512 }
