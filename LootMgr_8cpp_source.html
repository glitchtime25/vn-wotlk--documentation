97 // Loads a *_loot_template DB table into loot store
98 // All checks of the loaded template are called from here, no error reports at loot generation required
99 void LootStore::LoadLootTable()
100 {
101  LootTemplateMap::const_iterator tab;
102  uint32 count = 0;
103 
104  // Clearing store (for reloading case)
105  Clear();
106 
107  // 0 1 2 3 4 5 6
108  QueryResult* result = WorldDatabase.PQuery("SELECT entry, item, ChanceOrQuestChance, groupid, mincountOrRef, maxcount, condition_id FROM %s", GetName());
109 
110  if (result)
111  {
112  BarGoLink bar(result->GetRowCount());
113 
114  do
115  {
116  Field* fields = result->Fetch();
117  bar.step();
118 
119  uint32 entry = fields[0].GetUInt32();
120  uint32 item = fields[1].GetUInt32();
121  float chanceOrQuestChance = fields[2].GetFloat();
122  uint8 group = fields[3].GetUInt8();
123  int32 mincountOrRef = fields[4].GetInt32();
124  uint32 maxcount = fields[5].GetUInt32();
125  uint16 conditionId = fields[6].GetUInt16();
126 
127  if (maxcount > std::numeric_limits<uint8>::max())
128  {
129  sLog.outErrorDb("Table '%s' entry %d item %d: maxcount value (%u) to large. must be less %u - skipped", GetName(), entry, item, maxcount, std::numeric_limits<uint8>::max());
130  continue; // error already printed to log/console.
131  }
132 
133  if (conditionId)
134  {
135  const ConditionEntry* condition = sConditionStorage.LookupEntry<ConditionEntry>(conditionId);
136  if (!condition)
137  {
138  sLog.outErrorDb("Table `%s` for entry %u, item %u has condition_id %u that does not exist in `conditions`, ignoring", GetName(), entry, item, conditionId);
139  continue;
140  }
141 
142  if (mincountOrRef < 0 && !ConditionEntry::CanBeUsedWithoutPlayer(conditionId))
143  {
144  sLog.outErrorDb("Table '%s' entry %u mincountOrRef %i < 0 and has condition %u that requires a player and is not supported, skipped", GetName(), entry, mincountOrRef, conditionId);
145  continue;
146  }
147  }
148 
149  LootStoreItem storeitem = LootStoreItem(item, chanceOrQuestChance, group, conditionId, mincountOrRef, maxcount);
150 
151  if (!storeitem.IsValid(*this, entry)) // Validity checks
152  continue;
153 
154  // Looking for the template of the entry
155  // often entries are put together
156  if (m_LootTemplates.empty() || tab->first != entry)
157  {
158  // Searching the template (in case template Id changed)
159  tab = m_LootTemplates.find(entry);
160  if (tab == m_LootTemplates.end())
161  {
162  std::pair< LootTemplateMap::iterator, bool > pr = m_LootTemplates.insert(LootTemplateMap::value_type(entry, new LootTemplate));
163  tab = pr.first;
164  }
165  }
166  // else is empty - template Id and iter are the same
167  // finally iter refers to already existing or just created <entry, LootTemplate>
168 
169  // Adds current row to the template
170  tab->second->AddEntry(storeitem);
171  ++count;
172  }
173  while (result->NextRow());
174 
175  delete result;
176 
177  Verify(); // Checks validity of the loot store
178 
179  sLog.outString(">> Loaded %u loot definitions (" SIZEFMTD " templates) from table %s", count, m_LootTemplates.size(), GetName());
180  sLog.outString();
181  }
182  else
183  {
184  sLog.outString();
185  sLog.outErrorDb(">> Loaded 0 loot definitions. DB table `%s` is empty.", GetName());
186  }
187 }

287 // Checks if the entry (quest, non-quest, reference) takes it's chance (at loot generation)
288 // RATE_DROP_ITEMS is no longer used for all types of entries
289 bool LootStoreItem::Roll(bool rate) const
290 {
291  if (chance >= 100.0f)
292  return true;
293 
294  if (mincountOrRef < 0) // reference case
295  return roll_chance_f(chance * (rate ? sWorld.getConfig(CONFIG_FLOAT_RATE_DROP_ITEM_REFERENCED) : 1.0f));
296 
297  if (needs_quest)
298  return roll_chance_f(chance * (rate ? sWorld.getConfig(CONFIG_FLOAT_RATE_DROP_ITEM_QUEST) : 1.0f));
299 
300  ItemPrototype const* pProto = ObjectMgr::GetItemPrototype(itemid);
301 
302  float qualityModifier = pProto && rate ? sWorld.getConfig(qualityToRate[pProto->Quality]) : 1.0f;
303 
304  return roll_chance_f(chance * qualityModifier);
305 }

307 // Checks correctness of values
308 bool LootStoreItem::IsValid(LootStore const& store, uint32 entry) const
309 {
310  if (group >= 1 << 7) // it stored in 7 bit field
311  {
312  sLog.outErrorDb("Table '%s' entry %d item %d: group (%u) must be less %u - skipped", store.GetName(), entry, itemid, group, 1 << 7);
313  return false;
314  }
315 
316  if (mincountOrRef == 0)
317  {
318  sLog.outErrorDb("Table '%s' entry %d item %d: wrong mincountOrRef (%d) - skipped", store.GetName(), entry, itemid, mincountOrRef);
319  return false;
320  }
321 
322  if (mincountOrRef > 0) // item (quest or non-quest) entry, maybe grouped
323  {
324  ItemPrototype const* proto = ObjectMgr::GetItemPrototype(itemid);
325  if (!proto)
326  {
327  sLog.outErrorDb("Table '%s' entry %d item %d: item entry not listed in `item_template` - skipped", store.GetName(), entry, itemid);
328  return false;
329  }
330 
331  if (chance == 0 && group == 0) // Zero chance is allowed for grouped entries only
332  {
333  sLog.outErrorDb("Table '%s' entry %d item %d: equal-chanced grouped entry, but group not defined - skipped", store.GetName(), entry, itemid);
334  return false;
335  }
336 
337  if (chance != 0 && chance < 0.000001f) // loot with low chance
338  {
339  sLog.outErrorDb("Table '%s' entry %d item %d: low chance (%f) - skipped", store.GetName(), entry, itemid, chance);
340  return false;
341  }
342 
343  if (maxcount < mincountOrRef) // wrong max count
344  {
345  sLog.outErrorDb("Table '%s' entry %d item %d: max count (%u) less that min count (%i) - skipped", store.GetName(), entry, itemid, uint32(maxcount), mincountOrRef);
346  return false;
347  }
348  }
349  else // mincountOrRef < 0
350  {
351  if (needs_quest)
352  {
353  sLog.outErrorDb("Table '%s' entry %d item %d: negative chance is given for a reference, skipped", store.GetName(), entry, itemid);
354  return false;
355  }
356  if (chance == 0) // no chance for the reference
357  {
358  sLog.outErrorDb("Table '%s' entry %d item %d: zero chance is given for a reference, reference will never be used, skipped", store.GetName(), entry, itemid);
359  return false;
360  }
361  }
362  return true; // Referenced template existence is checked at whole store level
363 }

434 // Basic checks for player/item compatibility - if false no chance to see the item in the loot
435 bool LootItem::AllowedForPlayer(Player const* player, WorldObject const* lootTarget, Player const* masterLooter) const
436 {
437  if (!itemProto)
438  return false;
439 
440  // not show loot for not own team
441  if ((itemProto->Flags2 & ITEM_FLAG2_FACTION_HORDE) && player->GetTeam() != HORDE)
442  return false;
443 
444  if ((itemProto->Flags2 & ITEM_FLAG2_FACTION_ALLIANCE) && player->GetTeam() != ALLIANCE)
445  return false;
446 
447  switch (lootItemType)
448  {
449  case LOOTITEM_TYPE_NORMAL:
450  break;
451 
452  case LOOTITEM_TYPE_CONDITIONNAL:
453  // DB conditions check
454  if (!sObjectMgr.IsConditionSatisfied(conditionId, player, player->GetMap(), lootTarget, CONDITION_FROM_LOOT))
455  return false;
456  break;
457 
458  case LOOTITEM_TYPE_QUEST:
459  // Checking quests for quest-only drop (check only quests requirements in this case)
460  if (!player->HasQuestForItem(itemId))
461  return false;
462  break;
463 
464  default:
465  break;
466  }
467 
468  // If the item starts a quest and the player has that quest accepted/completed/rewarded, it can't be looted (unless the player is the master looter, or the item has ITEM_EXTRA_IGNORE_QUEST_STATUS)
469  if (itemProto->StartQuest)
470  {
471  if (!(itemProto->ExtraFlags & ITEM_EXTRA_IGNORE_QUEST_STATUS) && player != masterLooter && player->GetQuestStatus(itemProto->StartQuest) != QUEST_STATUS_NONE)
472  return false;
473  }
474 
475  return true;
476 }

478 LootSlotType LootItem::GetSlotTypeForSharedLoot(Player const* player, Loot const* loot) const
479 {
480  // Check if still have right to pick this item
481  if (!IsAllowed(player, loot))
482  return MAX_LOOT_SLOT_TYPE;
483 
484  if (freeForAll)
485  return loot->m_lootMethod == NOT_GROUP_TYPE_LOOT ? LOOT_SLOT_OWNER : LOOT_SLOT_NORMAL; // player have not yet looted a free for all item
486 
487  // quest items and conditional items cases
488  if (lootItemType == LOOTITEM_TYPE_QUEST || lootItemType == LOOTITEM_TYPE_CONDITIONNAL)
489  {
490  switch (loot->m_lootMethod)
491  {
492  case NOT_GROUP_TYPE_LOOT:
493  case FREE_FOR_ALL:
494  return LOOT_SLOT_OWNER;
495 
496  default:
497  if (!isUnderThreshold && lootItemType == LOOTITEM_TYPE_CONDITIONNAL && loot->m_lootMethod == MASTER_LOOT)
498  break;
499 
500  if (loot->m_isChest || loot->m_lootType == LOOT_FISHINGHOLE)
501  return LOOT_SLOT_OWNER;
502 
503  if (isBlocked)
504  return LOOT_SLOT_VIEW;
505 
506  // Check if its turn of that player to loot a not party loot. The loot may be released or the item may be passed by currentLooter
507  if (isReleased || currentLooterPass || loot->m_currentLooterGuid == player->GetObjectGuid())
508  return LOOT_SLOT_OWNER;
509  return MAX_LOOT_SLOT_TYPE;
510  }
511  }
512 
513  switch (loot->m_lootMethod)
514  {
515  case FREE_FOR_ALL:
516  return LOOT_SLOT_OWNER;
517  case GROUP_LOOT:
518  case NEED_BEFORE_GREED:
519  {
520  if (!isBlocked)
521  {
522  if (loot->m_isChest || loot->m_lootType == LOOT_FISHINGHOLE)
523  return LOOT_SLOT_NORMAL;
524 
525  if (isReleased || currentLooterPass || player->GetObjectGuid() == loot->m_currentLooterGuid)
526  return LOOT_SLOT_NORMAL;
527 
528  return MAX_LOOT_SLOT_TYPE;
529  }
530  return LOOT_SLOT_VIEW;
531  }
532  case MASTER_LOOT:
533  {
534  if (isUnderThreshold)
535  {
536  if (loot->m_isChest)
537  return LOOT_SLOT_OWNER;
538 
539  if (isReleased || currentLooterPass || player->GetObjectGuid() == loot->m_currentLooterGuid)
540  return LOOT_SLOT_OWNER;
541 
542  return MAX_LOOT_SLOT_TYPE;
543  }
544 
545  if (player->GetObjectGuid() == loot->m_masterOwnerGuid)
546  return LOOT_SLOT_MASTER;
547 
548  // give a chance to let others just see the content of the loot
549  if (isBlocked || sWorld.getConfig(CONFIG_BOOL_CORPSE_ALLOW_ALL_ITEMS_SHOW_IN_MASTER_LOOT))
550  return LOOT_SLOT_VIEW;
551 
552  return MAX_LOOT_SLOT_TYPE;
553  }
554  case ROUND_ROBIN:
555  {
556  if (loot->m_isChest || loot->m_lootType == LOOT_FISHINGHOLE)
557  return LOOT_SLOT_NORMAL;
558 
559  if (isReleased || currentLooterPass || player->GetObjectGuid() == loot->m_currentLooterGuid)
560  return LOOT_SLOT_OWNER;
561 
562  return MAX_LOOT_SLOT_TYPE;
563  }
564  case NOT_GROUP_TYPE_LOOT:
565  return LOOT_SLOT_OWNER;
566  default:
567  return MAX_LOOT_SLOT_TYPE;
568  }
