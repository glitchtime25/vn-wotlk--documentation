109  void MoveInLineOfSight(Unit* who) override
110  {
111  if (!m_hasTaunted && who->GetTypeId() == TYPEID_PLAYER && m_creature->IsWithinDistInMap(who, 110.0f) && m_creature->IsWithinLOSInMap(who))
112  {
113  switch (urand(0,4))

115  case 0: DoBroadcastText(SAY_GREET, m_creature); break;

117  case 2: DoBroadcastText(SAY_TAUNT2, m_creature); break;
118  case 3: DoBroadcastText(SAY_TAUNT3, m_creature); break;
