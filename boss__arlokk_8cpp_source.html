174  void ExecuteAction(uint32 action) override
175  {
176  switch (action)
177  {
178  case ARLOKK_PHASE_TRANSITION:
179  if (m_bIsPhaseTwo)
180  {
181  if (DoCastSpellIfCan(nullptr, SPELL_TRANSFORM_VISUAL) == CAST_OK)
182  {
183  m_creature->RemoveAurasDueToSpell(SPELL_PANTHER_TRANSFORM);
184  ResetCombatAction(action, 30000);
185  m_bIsPhaseTwo = false;
186  m_creature->SetSpellList(SPELL_LIST_PHASE_1);
187  }
188  }
189  else
190  {
191  if (DoCastSpellIfCan(nullptr, SPELL_PANTHER_TRANSFORM) == CAST_OK)
192  {
193  m_creature->CastSpell(nullptr, SPELL_HATE_TO_ZERO, TRIGGERED_NONE);
194  ResetCombatAction(action, 120000);
195  m_bIsPhaseTwo = true;
196  m_creature->SetSpellList(SPELL_LIST_PHASE_2);
197  }
198  }
199  break;
200  }
201  }

223  void JustRespawned() override
224  {
225  ScriptedAI::JustRespawned();
226  DoCastSpellIfCan(nullptr, SPELL_SNEAK);
227 
228  // Do only once, and only for those summoned by Arlokk
229  if (m_bMoveToAid)
230  {
231  // Add to GUID list to despawn later
232  m_lProwlerGUIDList.push_back(m_creature->GetObjectGuid());
233 
234  // Count the number of prowlers alive
235  uint32 count = 0;
236  for (GuidList::const_iterator itr = m_lProwlerGUIDList.begin(); itr != m_lProwlerGUIDList.end(); ++itr)
237  {
238  if (Unit* pProwler = m_creature->GetMap()->GetUnit(*itr))
239  if (pProwler->IsAlive())
240  count++;
241  }
242 
243  // Check if more than 40 are alive, if so, despawn
244  if (count > 40)
245  {
246  m_creature->ForcedDespawn();
247  return;
248  }
249 
250  m_creature->GetMotionMaster()->Clear();
251  m_creature->GetMotionMaster()->MovePoint(1, aArlokkWallShieldPos[0], aArlokkWallShieldPos[1], aArlokkWallShieldPos[2]);
252 
253  m_bMoveToAid = false;
254  }
255  }
