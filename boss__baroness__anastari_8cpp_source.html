82  void UpdateAI(const uint32 uiDiff) override
83  {
84  if (m_uiPossessEndTimer)
85  {
86  // Check if the possessed player has been damaged
87  if (m_uiPossessEndTimer <= uiDiff)
88  {
89  // If aura has expired, return to fight
90  if (!m_creature->HasAura(SPELL_POSSESS_INV))
91  {
92  m_uiPossessEndTimer = 0;
93  return;
94  }
95 
96  // Check for possessed player
97  Player* pPlayer = m_creature->GetMap()->GetPlayer(m_possessedPlayer);
98  if (!pPlayer || !pPlayer->IsAlive())
99  {
100  m_creature->RemoveAurasDueToSpell(SPELL_POSSESS_INV);
101  m_uiPossessEndTimer = 0;
102  return;
103  }
104 
105  // If possessed player has less than 50% health
106  if (pPlayer->GetHealth() <= pPlayer->GetMaxHealth() * .5f)
107  {
108  m_creature->RemoveAurasDueToSpell(SPELL_POSSESS_INV);
109  pPlayer->RemoveAurasDueToSpell(SPELL_POSSESSED);
110  pPlayer->RemoveAurasDueToSpell(SPELL_POSSESS);
111  m_uiPossessEndTimer = 0;
112  return;
113  }
114 
115  m_uiPossessEndTimer = 1000;
116  }
117  else
118  m_uiPossessEndTimer -= uiDiff;
119  }
120 
121  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
122  return;
123 
124  // BansheeWail
125  if (m_uiBansheeWailTimer < uiDiff)
126  {
127  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
128  {
129  if (DoCastSpellIfCan(pTarget, SPELL_BANSHEE_WAIL) == CAST_OK)
130  m_uiBansheeWailTimer = urand(2000, 3000);
131  }
132  }
133  else
134  m_uiBansheeWailTimer -= uiDiff;
135 
136  // BansheeCurse
137  if (m_uiBansheeCurseTimer < uiDiff)
138  {
139  if (DoCastSpellIfCan(m_creature, SPELL_BANSHEE_CURSE) == CAST_OK)
140  m_uiBansheeCurseTimer = 20000;
141  }
142  else
143  m_uiBansheeCurseTimer -= uiDiff;
144 
145  // Silence
146  if (m_uiSilenceTimer < uiDiff)
147  {
148  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
149  {
150  if (DoCastSpellIfCan(pTarget, SPELL_SILENCE) == CAST_OK)
151  m_uiSilenceTimer = 25000;
152  }
153  }
154  else
155  m_uiSilenceTimer -= uiDiff;
156 
157  // Possess
158  if (m_uiPossessTimer < uiDiff)
159  {
160  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_POSSESS, SELECT_FLAG_PLAYER))
161  {
162  if (DoCastSpellIfCan(pTarget, SPELL_POSSESS) == CAST_OK)
163  {
164  m_possessedPlayer = pTarget->GetObjectGuid();
165  m_uiPossessEndTimer = 1000;
166  m_uiPossessTimer = 30000;
167  }
168  }
169  }
170  else
171  m_uiPossessTimer -= uiDiff;
172 
173  DoMeleeAttackIfReady();
174  }

196  }
