199 void instance_blackrock_depths::SetData(uint32 uiType, uint32 uiData)
200 {
201  switch (uiType)
202  {
203  case TYPE_RING_OF_LAW:
204  // If finished the arena event after theldren fight
205  if (uiData == DONE && m_auiEncounter[0] == SPECIAL)
206  DoRespawnGameObject(GO_ARENA_SPOILS, HOUR);
207  else if (uiData == DONE)
208  {
209  for (auto m_sArenaCrowdNpcGuid : m_sArenaCrowdNpcGuids)
210  {
211  if (Creature* pSpectator = instance->GetCreature(m_sArenaCrowdNpcGuid))
212  pSpectator->SetFactionTemporary(FACTION_ARENA_NEUTRAL, TEMPFACTION_RESTORE_RESPAWN);
213  }
214  }
215  m_auiEncounter[0] = uiData;
216  break;
217  case TYPE_VAULT:
218  if (uiData == SPECIAL)
219  {
220  ++m_uiCofferDoorsOpened;
221 
222  if (m_uiCofferDoorsOpened == MAX_RELIC_DOORS)
223  {
224  SetData(TYPE_VAULT, IN_PROGRESS);
225 
226  Creature* pConstruct = nullptr;
227 
228  // Activate vault constructs
229  for (auto m_sVaultNpcGuid : m_sVaultNpcGuids)
230  {
231  pConstruct = instance->GetCreature(m_sVaultNpcGuid);
232  if (pConstruct)
233  pConstruct->RemoveAurasDueToSpell(SPELL_STONED);
234  }
235 
236  if (!pConstruct)
237  return;
238 
239  // Summon doomgrip
240  pConstruct->SummonCreature(NPC_WATCHER_DOOMGRIP, aVaultPositions[0], aVaultPositions[1], aVaultPositions[2], aVaultPositions[3], TEMPSPAWN_DEAD_DESPAWN, 0);
241  }
242  // No need to store in this case
243  return;
244  }
245  if (uiData == DONE)
246  {
247  DoUseDoorOrButton(GO_SECRET_DOOR);
248  DoToggleGameObjectFlags(GO_SECRET_SAFE, GO_FLAG_NO_INTERACT, false);
249  }
250  m_auiEncounter[1] = uiData;
251  break;
252  case TYPE_ROCKNOT:
253  if (uiData == SPECIAL)
254  ++m_uiBarAleCount;
255  else
256  {
257  if (uiData == DONE)
258  {
259  HandleBarPatrons(PATRON_PISSED);
260  SetBarDoorIsOpen();
261  }
262  m_auiEncounter[2] = uiData;
263  }
264  break;
265  case TYPE_TOMB_OF_SEVEN:
266  // Don't set the same data twice
267  if (uiData == m_auiEncounter[3])
268  break;
269  // Combat door
270  DoUseDoorOrButton(GO_TOMB_ENTER);
271  // Start the event
272  if (uiData == IN_PROGRESS)
273  DoCallNextDwarf();
274  if (uiData == FAIL)
275  {
276  // Reset dwarfes
277  for (unsigned int aTombDwarfe : aTombDwarfes)
278  {
279  if (Creature* pDwarf = GetSingleCreatureFromStorage(aTombDwarfe))
280  {
281  if (!pDwarf->IsAlive())
282  pDwarf->Respawn();
283  }
284  }
285 
286  m_uiDwarfRound = 0;
287  m_uiDwarfFightTimer = 0;
288  }
289  if (uiData == DONE)
290  {
291  DoRespawnGameObject(GO_CHEST_SEVEN, HOUR);
292  DoUseDoorOrButton(GO_TOMB_EXIT);
293  }
294  m_auiEncounter[3] = uiData;
295  break;
296  case TYPE_LYCEUM:
297  if (uiData == DONE)
298  {
299  DoUseDoorOrButton(GO_GOLEM_ROOM_N);
300  DoUseDoorOrButton(GO_GOLEM_ROOM_S);
301  if (Creature* magmus = GetSingleCreatureFromStorage(NPC_MAGMUS))
302  DoScriptText(YELL_MAGMUS_INTRO, magmus);
303  }
304  m_auiEncounter[4] = uiData;
305  break;
306  case TYPE_IRON_HALL:
307  switch (uiData)
308  {
309  case IN_PROGRESS:
310  DoUseDoorOrButton(GO_GOLEM_ROOM_N);
311  DoUseDoorOrButton(GO_GOLEM_ROOM_S);
312  break;
313  case FAIL:
314  DoUseDoorOrButton(GO_GOLEM_ROOM_N);
315  DoUseDoorOrButton(GO_GOLEM_ROOM_S);
316  break;
317  case DONE:
318  DoUseDoorOrButton(GO_GOLEM_ROOM_N);
319  DoUseDoorOrButton(GO_GOLEM_ROOM_S);
320  DoUseDoorOrButton(GO_THRONE_ROOM);
321  break;
322  }
323  m_auiEncounter[5] = uiData;
324  break;
325  case TYPE_QUEST_JAIL_BREAK:
326  m_auiEncounter[6] = uiData;
327  return;
328  case TYPE_FLAMELASH:
329  for (int i = 0; i < MAX_DWARF_RUNES; ++i)
330  DoUseDoorOrButton(GO_DWARFRUNE_A01 + i);
331  return;
332  case TYPE_HURLEY:
333  if (uiData == SPECIAL)
334  {
335  ++m_uiBrokenKegs;
336 
337  if (m_uiBrokenKegs == 3)
338  {
339  if (Creature* pPlugger = GetSingleCreatureFromStorage(NPC_PLUGGER_SPAZZRING))
340  {
341  // Summon Hurley Blackbreath
342  Creature* pHurley = pPlugger->SummonCreature(NPC_HURLEY_BLACKBREATH, aHurleyPositions[0], aHurleyPositions[1], aHurleyPositions[2], aHurleyPositions[3], TEMPSPAWN_DEAD_DESPAWN, 0);
343 
344  if (!pHurley)
345  return;
346 
347  // Summon cronies around Hurley
348  for (uint8 i = 0; i < MAX_CRONIES; ++i)
349  {
350  float fX, fY, fZ;
351  pPlugger->GetRandomPoint(aHurleyPositions[0], aHurleyPositions[1], aHurleyPositions[2], 2.0f, fX, fY, fZ);
352  if (Creature* pSummoned = pPlugger->SummonCreature(NPC_BLACKBREATH_CRONY, fX, fY, fZ, aHurleyPositions[3], TEMPSPAWN_DEAD_DESPAWN, 0))
353  {
354  pSummoned->SetWalk(false);
355  // The cronies should not engage anyone until their boss does so
356  // the linking is done by DB
357  pSummoned->SetImmuneToNPC(true);
358  pSummoned->SetImmuneToPlayer(true);
359  // The movement toward the kegs is handled by Hurley EscortAI
360  // and we want the cronies to follow him there
361  pSummoned->GetMotionMaster()->MoveFollow(pHurley, 1.0f, 0);
362  }
363  }
364  SetData(TYPE_HURLEY, IN_PROGRESS);
365  }
366  }
367  }
368  else
369  m_auiEncounter[8] = uiData;
370  break;
371  case TYPE_BRIDGE:
372  m_auiEncounter[9] = uiData;
373  return;
374  case TYPE_BAR:
375  if (uiData == IN_PROGRESS)
376  HandleBarPatrol(0);
377  m_auiEncounter[10] = uiData;
378  break;
379  case TYPE_PLUGGER:
380  if (uiData == SPECIAL)
381  {
382  if (GetSingleCreatureFromStorage(NPC_PLUGGER_SPAZZRING))
383  {
384  ++m_uiStolenAles;
385  if (m_uiStolenAles == 3)
386  uiData = IN_PROGRESS;
387  }
388  }
389  m_auiEncounter[11] = uiData;
390  break;
391  case TYPE_NAGMARA:
392  m_auiEncounter[12] = uiData;
393  break;
394  }
395 
396  if (uiData == DONE)
397  {
398  OUT_SAVE_INST_DATA;
399 
400  std::ostringstream saveStream;
401  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
402  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
403  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
404  << m_auiEncounter[9] << " " << m_auiEncounter[10] << " " << m_auiEncounter[11] << " "
405  << m_auiEncounter[12];
406 
407  m_strInstData = saveStream.str();
408 
409  SaveToDB();
410  OUT_SAVE_INST_DATA_COMPLETE;
411  }
412 }

514 void instance_blackrock_depths::OnCreatureDeath(Creature* pCreature)
515 {
516  switch (pCreature->GetEntry())
517  {
518  case NPC_WARBRINGER_CONST:
519  case NPC_WATCHER_DOOMGRIP:
520  if (GetData(TYPE_VAULT) == IN_PROGRESS)
521  {
522  m_sVaultNpcGuids.erase(pCreature->GetObjectGuid());
523 
524  // If all event npcs dead then set event to done
525  if (m_sVaultNpcGuids.empty())
526  SetData(TYPE_VAULT, DONE);
527  }
528  break;
529  case NPC_OGRABISI:
530  case NPC_SHILL:
531  case NPC_CREST:
532  case NPC_JAZ:
533  if (GetData(TYPE_QUEST_JAIL_BREAK) == IN_PROGRESS)
534  SetData(TYPE_QUEST_JAIL_BREAK, SPECIAL);
535  break;
536  // Handle Tomb of the Seven dwarf death event
537  case NPC_HATEREL:
538  case NPC_ANGERREL:
539  case NPC_VILEREL:
540  case NPC_GLOOMREL:
541  case NPC_SEETHREL:
542  case NPC_DOPEREL:
543  // Only handle the event when event is in progress
544  if (GetData(TYPE_TOMB_OF_SEVEN) != IN_PROGRESS)
545  return;
546  // Call the next dwarf only if it's the last one which joined the fight
547  if (pCreature->GetEntry() == aTombDwarfes[m_uiDwarfRound - 1])
548  DoCallNextDwarf();
549  break;
550  case NPC_DOOMREL:
551  SetData(TYPE_TOMB_OF_SEVEN, DONE);
552  break;
553  case NPC_MAGMUS:
554  SetData(TYPE_IRON_HALL, DONE);
555  break;
556  case NPC_HURLEY_BLACKBREATH:
557  SetData(TYPE_HURLEY, DONE);
558  break;
559  case NPC_RIBBLY_SCREWSPIGGOT:
560  // Do nothing if the patrol was already spawned or is about to:
561  // Plugger has made the bar hostile
562  {
563  // Do nothing if the patrol was already spawned or is about to:
564  // Plugger has made the bar hostile
565  if (GetData(TYPE_BAR) == IN_PROGRESS || GetData(TYPE_PLUGGER) == IN_PROGRESS || GetData(TYPE_BAR) == DONE || GetData(TYPE_PLUGGER) == DONE)
566  return;
567  SetData(TYPE_BAR, IN_PROGRESS);
568  }
569  break;
570  case NPC_SHADOWFORGE_SENATOR:
571  // Emperor Dagran Thaurissan performs a random yell upon the death
572  // of Shadowforge Senators in the Throne Room
573  if (Creature* pDagran = GetSingleCreatureFromStorage(NPC_EMPEROR))
574  {
575  if (!pDagran->IsAlive())
576  return;
577 
578  if (m_uiDagranTimer > 0)
579  return;
580 
581  switch (urand(0, 3))
582  {
583  case 0: DoScriptText(YELL_SENATOR_1, pDagran); break;
584  case 1: DoScriptText(YELL_SENATOR_2, pDagran); break;
585  case 2: DoScriptText(YELL_SENATOR_3, pDagran); break;
586  case 3: DoScriptText(YELL_SENATOR_4, pDagran); break;
587  }
588  m_uiDagranTimer = 30000; // set a timer of 30 sec to avoid Emperor Thaurissan to spam yells in case many senators are killed in a short amount of time

629 void instance_blackrock_depths::HandleBarPatrons(uint8 uiEventType)
630 {
631  switch (uiEventType)
632  {
633  // case for periodical handle of random emotes
634  case PATRON_EMOTE:
635  if (GetData(TYPE_PLUGGER) == DONE)
636  return;
637 
638  for (auto m_sBarPatronNpcGuid : m_sBarPatronNpcGuids)
639  {
640  // About 5% of patrons do emote at a given time
641  // So avoid executing follow up code for the 95% others
642  if (urand(0, 100) < 4)
643  {
644  // Only three emotes are seen in data: laugh, cheer and exclamation
645  // the last one appearing the least and the first one appearing the most
646  // emotes are stored in a table and frequency is handled there
647  if (Creature* pPatron = instance->GetCreature(m_sBarPatronNpcGuid))
648  pPatron->HandleEmote(aPatronsEmotes[urand(0, 5)]);
649  }
650  }
651  return;
652  // case for Rocknot event when breaking the barrel
653  case PATRON_PISSED:
654  // Three texts are said, one less often than the two others
655  // Only by patrons near the broken barrel react to Rocknot's rampage
656  if (GameObject* pGo = GetSingleGameObjectFromStorage(GO_BAR_KEG_SHOT))
657  {
658  for (auto m_sBarPatronNpcGuid : m_sBarPatronNpcGuids)
659  {
660  if (Creature* pPatron = instance->GetCreature(m_sBarPatronNpcGuid))
661  {
662  if (pPatron->GetPositionZ() > pGo->GetPositionZ() - 1 && pPatron->IsWithinDist2d(pGo->GetPositionX(), pGo->GetPositionY(), 18.0f))
663  {
664  switch (urand(0, 4))
665  {
666  case 0: DoScriptText(SAY_PISSED_PATRON_3, pPatron); break;
667  case 1: // case is double to give this text twice the chance of the previous one do be displayed
668  case 2: DoScriptText(SAY_PISSED_PATRON_2, pPatron); break;
669  // covers the two remaining cases
670  default: DoScriptText(SAY_PISSED_PATRON_1, pPatron); break;
671  }
672  }
673  }
674  }
675  }
676  return;
677  // case when Plugger is killed
678  case PATRON_HOSTILE:
679  for (auto m_sBarPatronNpcGuid : m_sBarPatronNpcGuids)
680  {
681  if (Creature* pPatron = instance->GetCreature(m_sBarPatronNpcGuid))
682  {

686  pPatron->GetMotionMaster()->MoveRandomAroundPoint(pPatron->GetPositionX(), pPatron->GetPositionY(), pPatron->GetPositionZ(), 2.0f);
687  }
688  }
689  // Mistress Nagmara and Private Rocknot despawn if the bar turns hostile
690  if (Creature* pRocknot = GetSingleCreatureFromStorage(NPC_PRIVATE_ROCKNOT))
691  {
692  DoScriptText(SAY_ROCKNOT_DESPAWN, pRocknot);
693  pRocknot->ForcedDespawn();
