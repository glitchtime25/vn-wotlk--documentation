82  void UpdateAI(const uint32 diff) override
83  {
84  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
85  return;
86 
87  // Hellfire
88  if (m_hellfireTimer < diff)
89  {
90  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_HELLFIRE : SPELL_HELLFIRE_H) == CAST_OK)
91  m_hellfireTimer = urand(16000, 24000);
92  }
93  else
94  m_hellfireTimer -= diff;
95 
96  // Sacrifice
97  if (m_sacrificeTimer < diff)
98  {
99  m_sacrificeTimer = 0;
100  if (!m_creature->IsNonMeleeSpellCasted(false))
101  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_SACRIFICE, SELECT_FLAG_PLAYER | SELECT_FLAG_IN_LOS))
102  if (DoCastSpellIfCan(target, SPELL_SACRIFICE) == CAST_OK)
103  m_sacrificeTimer = urand(22000, 34000);
104  }
105  else
106  m_sacrificeTimer -= diff;
107 
108  // Enrage
109  if (m_enrageTimer < diff)
110  {
111  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
112  m_enrageTimer = urand(22000, 40000);
113  }
114  else
115  m_enrageTimer -= diff;
116 
117  if (!m_below50 && m_creature->GetHealthPercent() <= 50.f)
118  {
119  DoScriptText(SAY_50_HP, m_creature);
120  m_below50 = true;
121  }
122 
123  if (!m_below20 && m_creature->GetHealthPercent() <= 20.f)
124  {
125  DoScriptText(SAY_20_HP, m_creature);
126  m_below20 = true;
127  }
128 
129  DoMeleeAttackIfReady();
130  }
