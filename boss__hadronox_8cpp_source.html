51 /* ##### Gauntlet description #####
52  * This is the timed gauntlet - waves of non-elite spiders will spawn from the 3 doors located a little above the main room
53  * They will make their way down to fight Hadronox but she will head to the main room, fighting the spiders
54  * When Hadronox enters the main room, she will web the doors, and no more spiders will spawn.

98  void MovementInform(uint32 uiMoveType, uint32 uiPointId) override
99  {
100  if (uiMoveType == WAYPOINT_MOTION_TYPE && uiPointId == 11)
101  {
102  // stop the spiders
103  DoCastSpellIfCan(m_creature, SPELL_WEB_FRONT_DOORS, CAST_TRIGGERED);
104 
105  // No more movement
106  m_creature->GetMotionMaster()->Clear(false, true);
107  m_creature->GetMotionMaster()->MoveIdle();
108 
109  // reset react state
110  SetReactState(REACT_AGGRESSIVE);
111 
112  // fail achievement and stop the spiders
113  // if boss isn't pulled before it reaches this point, the spider gauntlet will stop
114  if (m_instance)
115  {
116  m_instance->SetHadronoxDeniedAchievCriteria(false);
117  m_instance->SetData(TYPE_HADRONOX, SPECIAL);

122  void ExecuteAction(uint32 action) override
123  {
124  switch (action)
125  {
126  case HADRONOX_ACTION_PIERCE_ARMOR:
127  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_PIERCE_ARMOR) == CAST_OK)
128  ResetCombatAction(action, 20000);
129  break;
130  case HADRONOX_ACTION_ACID:
131  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
132  {
133  if (DoCastSpellIfCan(target, m_isRegularMode ? SPELL_ACID_CLOUD : SPELL_ACID_CLOUD_H) == CAST_OK)
134  ResetCombatAction(action, 15000);
135  }
136  break;
137  case HADRONOX_ACTION_LEECH:
138  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
139  {
140  if (DoCastSpellIfCan(target, m_isRegularMode ? SPELL_LEECH_POISON : SPELL_LEECH_POISON_H) == CAST_OK)
141  ResetCombatAction(action, urand(10000, 15000));
142  }
143  break;
144  case HADRONOX_ACTION_GRAB:
145  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_WEB_GRAB : SPELL_WEB_GRAB_H) == CAST_OK)
146  ResetCombatAction(action, urand(25000, 30000));
147  break;
148  case HADRONOX_ACTION_TAUNT:
149  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
150  {
151  if (DoCastSpellIfCan(target, SPELL_TAUNT) == CAST_OK)
152  ResetCombatAction(action, urand(3000, 6000));
153  }
154  break;
155  }
156  }
157 };

183 struct spell_summon_anubar_periodic_aura : public AuraScript
184 {
185  void OnPeriodicDummy(Aura* aura) const override
186  {
187  Unit* target = aura->GetTarget();
188  if (!target)
189  return;
190 
191  // random chance of 2/3 of triggering
192  if (!urand(0, 2))
193  return;
194 
195  switch (aura->GetId())
196  {
197  case 53035: // Summon Anub'ar Champion Periodic (Azjol Nerub)
198  case 53036: // Summon Anub'ar Necromancer Periodic (Azjol Nerub)
199  case 53037: // Summon Anub'ar Crypt Fiend Periodic (Azjol Nerub)
200  {
201  // gauntlet spells
202  uint32 summonSpells[3][2] =
203  {
204  {53090, 53064}, // Summon Anub'ar Champion
205  {53092, 53066}, // Summon Anub'ar Necromancer
206  {53091, 53065} // Summon Anub'ar Crypt Fiend
207  };
208 
209  // Cast different spell depending on trigger position
210  // This will summon a different npc entry on each location - each of those has individual movement patern
211  if (target->GetPositionZ() < 750.0f)
212  target->CastSpell(target, summonSpells[aura->GetId() - 53035][0], TRIGGERED_OLD_TRIGGERED);
213  else
214  target->CastSpell(target, summonSpells[aura->GetId() - 53035][1], TRIGGERED_OLD_TRIGGERED);
215  }
216  }
217  }
218 };
