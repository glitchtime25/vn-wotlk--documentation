67 void instance_temple_of_ahnqiraj::JustDidDialogueStep(int32 entry)

133 void instance_temple_of_ahnqiraj::OnCreatureCreate(Creature* creature)
134 {
135  switch (creature->GetEntry())
136  {
137  case NPC_VEKLOR:
138  case NPC_VEKNILASH:
139  if (GetData(TYPE_TWINS_INTRO) != DONE)
140  creature->SetStandState(UNIT_STAND_STATE_KNEEL);
141  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
142  break;
143  case NPC_SKERAM:
144  // Don't store the summoned images guid
145  if (GetData(TYPE_SKERAM) == IN_PROGRESS)
146  break;
147  case NPC_KRI:
148  case NPC_YAUJ:
149  case NPC_VEM:
150  case NPC_SARTURA:
151  case NPC_MASTERS_EYE:
152  case NPC_OURO_SPAWNER:
153  case NPC_CTHUN:
154  case NPC_VISCIDUS:
155  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
156  break;
157  case NPC_POISON_CLOUD:
158  m_bugTrioSpawns.push_back(creature->GetObjectGuid());
159  break;
160  case NPC_CLAW_TENTACLE:
161  case NPC_EYE_TENTACLE:
162  if (creature->AI())
163  creature->AI()->SetCombatMovement(false);
164  creature->CastSpell(creature, SPELL_SUMMON_PORTAL, TRIGGERED_OLD_TRIGGERED);
165  break;
166  case NPC_GIANT_CLAW_TENTACLE:
167  case NPC_GIANT_EYE_TENTACLE:
168  if (creature->AI())
169  creature->AI()->SetCombatMovement(false);
170  creature->CastSpell(creature, SPELL_SUMMON_GIANT_PORTAL, TRIGGERED_OLD_TRIGGERED);
171  break;
172  case NPC_EYE_OF_CTHUN:
173  // Safeguard for C'Thun encounter in case of fight abruptly ended during phase 2
174  if (GetData(TYPE_CTHUN) != DONE)
175  {
176  if (!creature->IsAlive())
177  creature->Respawn();
178  }
179  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
180  break;
181  }

226 void instance_temple_of_ahnqiraj::SetData(uint32 uiType, uint32 uiData)
227 {
228  switch (uiType)
229  {
230  case TYPE_SKERAM:
231  m_auiEncounter[uiType] = uiData;
232  if (uiData == DONE)
233  DoUseDoorOrButton(GO_SKERAM_GATE);
234  break;
235  case TYPE_BUG_TRIO:
236  if (uiData > SPECIAL)
237  {
238  ++m_uiBugTrioDeathCount;
239  Creature* deadGuy = GetSingleCreatureFromStorage(uiData);
240  // notify bugs on death to heal / remove invul
241  if (Creature* vem = GetSingleCreatureFromStorage(NPC_VEM))
242  if (vem->IsAlive())
243  vem->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, deadGuy, vem, m_uiBugTrioDeathCount);
244  if (Creature* yauj = GetSingleCreatureFromStorage(NPC_YAUJ))
245  if (yauj->IsAlive())
246  yauj->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, deadGuy, yauj, m_uiBugTrioDeathCount);
247  if (Creature* kri = GetSingleCreatureFromStorage(NPC_KRI))
248  if (kri->IsAlive())
249  kri->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, deadGuy, kri, m_uiBugTrioDeathCount);
250 
251  // don't store any special data
252  break;
253  }
254  if (uiData == FAIL)
255  {
256  // Do not reset again the encounter if already set to failed
257  if (GetData(TYPE_BUG_TRIO) == FAIL)
258  return;
259 
260  // On encounter failure, despawn Vem and Princess Yauj and make Lord Kri evade
261  // the two other bosses will respawn when Lord Kri reaches home/respawns
262  if (Creature* vem = GetSingleCreatureFromStorage(NPC_VEM))
263  vem->ForcedDespawn();
264  if (Creature* yauj = GetSingleCreatureFromStorage(NPC_YAUJ))
265  yauj->ForcedDespawn();
266  if (Creature* kri = GetSingleCreatureFromStorage(NPC_KRI))
267  {
268  if (kri->IsAlive())
269  kri->AI()->EnterEvadeMode();
270  else
271  kri->Respawn();
272  }
273  for (auto guid : m_bugTrioSpawns)
274  if (Creature* spawn = instance->GetCreature(guid))
275  spawn->ForcedDespawn();
276  m_bugTrioSpawns.clear();
277  }
278  if (uiData == IN_PROGRESS)
279  m_uiBugTrioDeathCount = 0;
280  m_auiEncounter[uiType] = uiData;
281  break;
282  case TYPE_SARTURA:
283  case TYPE_FANKRISS:
284  case TYPE_VISCIDUS:
285  m_auiEncounter[uiType] = uiData;
286  break;
287  case TYPE_HUHURAN:
288  m_auiEncounter[uiType] = uiData;
289  if (uiData == DONE)
290  DoUseDoorOrButton(GO_TWINS_ENTER_DOOR);
291  break;
292  case TYPE_TWINS:
293  // Either of the twins can set data, so return to avoid double changing
294  if (m_auiEncounter[uiType] == uiData)
295  return;
296 
297  m_auiEncounter[uiType] = uiData;
298  DoUseDoorOrButton(GO_TWINS_ENTER_DOOR);
299  if (uiData == DONE)
300  DoUseDoorOrButton(GO_TWINS_EXIT_DOOR);
301  break;
302  case TYPE_OURO:
303  switch (uiData)
304  {
305  case FAIL:
306  // Respawn the Ouro spawner on fail
307  if (Creature* pSpawner = GetSingleCreatureFromStorage(NPC_OURO_SPAWNER))
308  pSpawner->Respawn();
309  // no break;
310  case DONE:
311  // Despawn the sandworm base on Done or Fail
312  if (GameObject* pBase = GetSingleGameObjectFromStorage(GO_SANDWORM_BASE))
313  pBase->SetLootState(GO_JUST_DEACTIVATED);
314  break;
315  }
316  m_auiEncounter[uiType] = uiData;
317  break;
318  case TYPE_CTHUN:
319  if (uiData == FAIL)
320  {
321  // Respawn the Eye of C'Thun when failing in phase 2
322  if (Creature* eyeOfCthun = GetSingleCreatureFromStorage(NPC_EYE_OF_CTHUN))
323  {
324  if (!eyeOfCthun->IsAlive())
325  eyeOfCthun->Respawn();
326  // Reset combat
327  if (eyeOfCthun->AI() && eyeOfCthun->IsInCombat())
328  eyeOfCthun->AI()->EnterEvadeMode();
329  }
330  if (Creature* cthun = GetSingleCreatureFromStorage(NPC_CTHUN))
331  {
332  // Reset combat
333  if (cthun->AI() && cthun->IsInCombat())
334  cthun->AI()->EnterEvadeMode();
335  }
336  }
337  m_auiEncounter[uiType] = uiData;
338  break;
339  case TYPE_TWINS_INTRO:
340  m_auiEncounter[uiType] = uiData;
341  break;
342  }
343 
344  if (uiData == DONE)
345  {
346  OUT_SAVE_INST_DATA;
347 
348  std::ostringstream saveStream;
349  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3] << " "
350  << m_auiEncounter[4] << " " << m_auiEncounter[5] << " " << m_auiEncounter[6] << " " << m_auiEncounter[7] << " "
351  << m_auiEncounter[8] << " " << m_auiEncounter[9];
352 
353  m_strInstData = saveStream.str();
354 
355  SaveToDB();
356  OUT_SAVE_INST_DATA_COMPLETE;
357  }
358 }

392 void instance_temple_of_ahnqiraj::Update(uint32 uiDiff)
393 {
394  DialogueUpdate(uiDiff);
395 
396  if (GetData(TYPE_SKERAM) == NOT_STARTED)
397  {
398  if (m_uiSkeramProphecyTimer < uiDiff)
399  {
400  if (Player* player = GetPlayerInMap())
401  player->GetMap()->PlayDirectSoundToMap(sound_skeram_prophecy[urand(0, 4)]);
402  m_uiSkeramProphecyTimer = urand(3, 4) * MINUTE * IN_MILLISECONDS; // Timer is guesswork
403  }
404  else
405  m_uiSkeramProphecyTimer -= uiDiff;
406  }
407 
408  if (GetData(TYPE_CTHUN) == IN_PROGRESS || GetData(TYPE_CTHUN) == DONE)
409  return;
410 
411  if (m_uiCthunWhisperTimer < uiDiff)
412  {
413  if (Player* pPlayer = GetPlayerInMap())
414  {
415  if (Creature* pCthun = GetSingleCreatureFromStorage(NPC_CTHUN))
416  {
417  // ToDo: also cast the C'thun Whispering charm spell - requires additional research
418  switch (urand(0, 7))
419  {
420  case 0: DoScriptText(SAY_CTHUN_WHISPER_1, pCthun, pPlayer); break;
421  case 1: DoScriptText(SAY_CTHUN_WHISPER_2, pCthun, pPlayer); break;
422  case 2: DoScriptText(SAY_CTHUN_WHISPER_3, pCthun, pPlayer); break;
423  case 3: DoScriptText(SAY_CTHUN_WHISPER_4, pCthun, pPlayer); break;
424  case 4: DoScriptText(SAY_CTHUN_WHISPER_5, pCthun, pPlayer); break;
425  case 5: DoScriptText(SAY_CTHUN_WHISPER_6, pCthun, pPlayer); break;
426  case 6: DoScriptText(SAY_CTHUN_WHISPER_7, pCthun, pPlayer); break;

432  }
