106 uint32 SpawnGroup::GetEligibleEntry(std::map<uint32, uint32>& existingEntries, std::map<uint32, uint32>& minEntries)
107 {
108  if (m_entry.RandomEntries.empty())
109  return 0;
110 
111  if (minEntries.size() > 0)
112  {
113  auto itr = minEntries.begin();
114  std::advance(itr, urand(0, minEntries.size() - 1));
115  uint32 entry = (*itr).first;
116  --(*itr).second;
117  if ((*itr).second == 0)
118  minEntries.erase(itr);
119  return entry;
120  }
121 
122  if (m_entry.ExplicitlyChanced.size())
123  {
124  int32 roll = urand(1, 100);
125  for (auto explicitly : m_entry.ExplicitlyChanced)
126  {
127  if (existingEntries[explicitly->Entry] > 0)
128  {
129  if (roll < static_cast<int>(explicitly->Chance))
130  return explicitly->Entry;
131 
132  roll -= int32(explicitly->Chance);
133  }
134  }
135  }
136 
137  if (m_entry.EquallyChanced.empty())
138  return 0;
139 
140  auto equallyCopy = m_entry.EquallyChanced;
141  std::shuffle(equallyCopy.begin(), equallyCopy.end(), *GetRandomGenerator());
142 
143  for (auto equally : equallyCopy)
144  if (existingEntries[equally->Entry] > 0)
145  return equally->Entry;
146 
147  return 0;
148 }

150 void SpawnGroup::Spawn(bool force)
151 {
152  if (!m_enabled && !force)
153  return;
154 
155  // duplicated code for optimization - way fewer cond fails
156  if ((m_entry.Flags & SPAWN_GROUP_DESPAWN_ON_COND_FAIL) != 0) // must be before count check
157  {
158  if (!m_objects.empty() && !IsWorldstateConditionSatisfied())
159  {
160  Despawn(0, 1); // respawn delay 1 to immediately be available for spawn on next condition success
161  return;
162  }
163  }
164 
165  if (m_objects.size() >= m_entry.MaxCount)
166  return;
167 
168  if (!IsWorldstateConditionSatisfied())
169  return;
170 
171  if (m_entry.HasChancedSpawns && m_chosenSpawns.size() >= m_entry.MaxCount)
172  return;
173 
174  std::vector<SpawnGroupDbGuids const*> eligibleGuids;
175  std::map<uint32, uint32> validEntries;
176  std::map<uint32, uint32> minEntries;
177 
178  for (auto& randomEntry : m_entry.RandomEntries)
179  {
180  validEntries[randomEntry.Entry] = randomEntry.MaxCount > 0 ? randomEntry.MaxCount : std::numeric_limits<uint32>::max();
181  if (randomEntry.MinCount > 0)
182  minEntries.emplace(randomEntry.Entry, randomEntry.MinCount);
183  }
184 
185  for (auto& guid : m_entry.DbGuids)
186  eligibleGuids.push_back(&guid);
187 
188  for (auto& data : m_objects)
189  {
190  eligibleGuids.erase(std::remove_if(eligibleGuids.begin(), eligibleGuids.end(), [dbGuid = data.first](SpawnGroupDbGuids const* entry) { return entry->DbGuid == dbGuid; }), eligibleGuids.end());
191  if (validEntries.size() > 0)
192  {
193  uint32 curCount = validEntries[data.second];
194  validEntries[data.second] = curCount > 0 ? curCount - 1 : 0;
195  }
196  if (minEntries.size() > 0)
197  {
198  auto itr = minEntries.find(data.second);
199  if (itr != minEntries.end())
200  {
201  --(*itr).second;
202  if ((*itr).second == 0)
203  minEntries.erase(itr);
204  }
205  }
206  }
207 
208  time_t now = time(nullptr);
209  for (auto itr = eligibleGuids.begin(); itr != eligibleGuids.end();)
210  {
211  if (m_map.GetPersistentState()->GetObjectRespawnTime(GetObjectTypeId(), (*itr)->DbGuid) > now)
212  {
213  if (!force)
214  {
215  if (m_entry.MaxCount == 1) // rare mob case - prevent respawn until all are off CD
216  return;
217  itr = eligibleGuids.erase(itr);
218  continue;
219  }
220  else
221  m_map.GetPersistentState()->SaveObjectRespawnTime(GetObjectTypeId(), (*itr)->DbGuid, now);
222  }
223  ++itr;
224  }
225 
226  for (auto itr = eligibleGuids.begin(); itr != eligibleGuids.end();)
227  {
228  uint32 spawnMask = 0; // safeguarded on db load
229  if (GetObjectTypeId() == TYPEID_UNIT)
230  spawnMask = sObjectMgr.GetCreatureData((*itr)->DbGuid)->spawnMask;
231  else
232  spawnMask = sObjectMgr.GetGOData((*itr)->DbGuid)->spawnMask;
233  if (spawnMask && (spawnMask & (1 << m_map.GetDifficulty())) == 0)
234  {
235  itr = eligibleGuids.erase(itr);
236  continue;
237  }
238  ++itr;
239  }
240 
241  std::shuffle(eligibleGuids.begin(), eligibleGuids.end(), *GetRandomGenerator());
242 
243  for (auto itr = eligibleGuids.begin(); itr != eligibleGuids.end() && !eligibleGuids.empty() && m_objects.size() < m_entry.MaxCount; ++itr)
244  {
245  uint32 dbGuid = (*itr)->DbGuid;
246  uint32 entry = 0;
247  if (m_entry.HasChancedSpawns)
248  {
249  if ((*itr)->Chance)
250  {
251  auto spawnItr = m_chosenSpawns.find(dbGuid);
252  bool spawn = true;
253  if (spawnItr == m_chosenSpawns.end())
254  {
255  spawn = roll_chance_i((*itr)->Chance);
256  m_chosenSpawns[dbGuid] = spawn;
257  }
258  else
259  spawn = spawnItr->second;
260 
261  if (!spawn)
262  continue;
263  }
264  else // filling redundant entries when a group has chanced spawns for optimization so we can stop at start
265  m_chosenSpawns[dbGuid] = true;
266  }
267 
268  // creatures pick random entry on first spawn in dungeons - else always pick random entry
269  if (GetObjectTypeId() == TYPEID_UNIT)
270  {
271  if (m_map.IsDungeon())
272  {
273  // only held in memory - implement saving to db if it becomes a major issue
274  if (m_chosenEntries.find(dbGuid) == m_chosenEntries.end())
275  {
276  // some group members can have static entry, or selfcontained random entry
277  if ((*itr)->RandomEntry)
278  entry = sObjectMgr.GetRandomCreatureEntry(dbGuid);
279  else if ((*itr)->OwnEntry)
280  entry = (*itr)->OwnEntry;
281  else
282  entry = GetEligibleEntry(validEntries, minEntries);
283  m_chosenEntries[dbGuid] = entry;
284  }
285  else
286  entry = m_chosenEntries[dbGuid];
287  }
288  else
289  entry = GetEligibleEntry(validEntries, minEntries);
290  }
291  else // GOs always pick random entry
292  {
293  if ((*itr)->RandomEntry)
294  entry = sObjectMgr.GetRandomGameObjectEntry(dbGuid);
295  else if ((*itr)->OwnEntry)
296  entry = (*itr)->OwnEntry;
297  else
298  entry = GetEligibleEntry(validEntries, minEntries);
299  }
300 
301  float x, y;
302  if (GetObjectTypeId() == TYPEID_UNIT)
303  {
304  auto data = sObjectMgr.GetCreatureData(dbGuid);
305  x = data->posX; y = data->posY;
306  m_map.GetPersistentState()->AddCreatureToGrid(dbGuid, data);
307  }
308  else
309  {
310  auto data = sObjectMgr.GetGOData(dbGuid);
311  x = data->posX; y = data->posY;
312  m_map.GetPersistentState()->AddGameobjectToGrid(dbGuid, data);
313  }
314  AddObject(dbGuid, entry);
315  if (force || m_entry.Active || m_map.IsLoaded(x, y))
316  {
317  if (GetObjectTypeId() == TYPEID_UNIT)
318  WorldObject::SpawnCreature(dbGuid, &m_map, entry);
319  else
320  WorldObject::SpawnGameObject(dbGuid, &m_map, entry);
321  }
322  if (entry && validEntries[entry])
323  --validEntries[entry];
324  }
325 }

415 {
416  switch (event)
417  {
418  case CREATURE_GROUP_EVENT_AGGRO:
419  if ((m_entry.Flags & CREATURE_GROUP_AGGRO_TOGETHER) == 0)
420  return;
421 
422  for (auto& data : m_objects)
423  {
424  uint32 dbGuid = data.first;

426  {
427  creature->AddThreat(target);
428  target->AddThreat(creature);
429  target->SetInCombatWith(creature);
430  target->GetCombatManager().TriggerCombatTimer(creature);
431  }
432  }
433 
434  for (uint32 linkedGroup : m_entry.LinkedGroups)
435  {
436  // ensured on db load that it will be valid fetch
437  CreatureGroup* group = static_cast<CreatureGroup*>(m_map.GetSpawnManager().GetSpawnGroup(linkedGroup));
438  group->TriggerLinkingEvent(event, target);
439  }
440  break;
441  case CREATURE_GROUP_EVENT_EVADE:
442  if ((m_entry.Flags & CREATURE_GROUP_EVADE_TOGETHER) != 0)
443  {
444  for (auto& data : m_objects)
445  {
446  uint32 dbGuid = data.first;
447  if (Creature* creature = m_map.GetCreature(dbGuid))
448  if (!creature->GetCombatManager().IsEvadingHome())
449  creature->AI()->EnterEvadeMode();
450  }
451  }
452  break;
453  case CREATURE_GROUP_EVENT_HOME:
454  if (FormationData* formation = GetFormationData())
455  if (!IsEvading()) // on last evade complete
456  formation->OnHome();
457  [[fallthrough]];
458  case CREATURE_GROUP_EVENT_RESPAWN:
459  if ((m_entry.Flags & CREATURE_GROUP_RESPAWN_TOGETHER) == 0)
460  return;
461 
462  ClearRespawnTimes();
463  break;
464  case CREATURE_GROUP_EVENT_MEMBER_DIED:
465  for (auto const& data : m_objects)
466  {
467  uint32 dbGuid = data.first;
468  if (Creature* creature = m_map.GetCreature(dbGuid))

470  }
471  break;
472  }
473 }
