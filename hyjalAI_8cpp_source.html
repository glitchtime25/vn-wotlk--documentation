106 void hyjalAI::DoTalk(YellType pYellType)
107 {
108  HyjalYells const* pYell = nullptr;
109 
110  bool bGetNext = false;
111 
112  for (const auto& i : aHyjalYell)
113  {
114  if (i.uiCreatureEntry == m_creature->GetEntry() && i.m_pYellType == pYellType)
115  {
116  if (pYellType == ATTACKED)
117  {
118  if (!bGetNext && urand(0, 1))
119  {
120  bGetNext = true;
121  continue;
122  }
123  }
124  else if (pYellType == RALLY)
125  {
126  if (!bGetNext && urand(0, 2))
127  {
128  bGetNext = true;
129  continue;
130  }
131  }
132 
133  pYell = &i;
134  break;
135  }
136  }
137 
138  if (pYell)
139  {
140  if (pYell->uiCreatureEntry == NPC_THRALL && pYell->m_pYellType == DEATH) // Thrall doesn't have a text on Death, only sound
141  DoPlaySoundToSet(m_creature, 11034);
142  else
143  DoScriptText(pYell->m_iTextId, m_creature);
144  }
145 }

161 void hyjalAI::ReceiveAIEvent(AIEventType eventType, Unit* sender, Unit* /*invoker*/, uint32 miscValue)
162 {
163  switch (eventType)
164  {
165  case AI_EVENT_CUSTOM_B:
166  {
167  m_creature->SetCanEnterCombat(false);
168  SetReactState(REACT_PASSIVE);
169  DoTalk(WIN);
170  if (miscValue == 1)
171  m_creature->GetMotionMaster()->MoveWaypoint(miscValue);
172  break;
173  }
174  case AI_EVENT_CUSTOM_C: // Event failed due to exceeding max number of wave mobs
175  Retreat();
176  break;
177  case AI_EVENT_CUSTOM_D: // Boss was killed - stop yelling
178  m_started = false;
179  DisableTimer(HYJAL_AI_RALLY_TIMER);
180  break;
181  case AI_EVENT_CUSTOM_EVENTAI_B:
182  {
183  if (m_creature->GetEntry() != NPC_JAINA)
184  return;
185 
186  if (sender != m_creature) // Only retreat if AI Event is sent by Jaina herself
187  return;
188 
189  Win();
190  break;
191  }
192  default: break;
193  }
194 }
