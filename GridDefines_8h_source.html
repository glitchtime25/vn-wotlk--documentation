92  {

146  template<class RET_TYPE, int CENTER_VAL>
147  inline RET_TYPE Compute(float x, float y, float center_offset, float size)
148  {
149  // calculate and store temporary values in double format for having same result as same mySQL calculations
150  double x_offset = (double(x) - center_offset) / size;
151  double y_offset = (double(y) - center_offset) / size;
152 
153  int x_val = int(x_offset + CENTER_VAL + 0.5);
154  int y_val = int(y_offset + CENTER_VAL + 0.5);
155  return RET_TYPE(x_val, y_val);
156  }
