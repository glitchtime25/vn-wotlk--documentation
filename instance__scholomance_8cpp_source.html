53 void instance_scholomance::OnCreatureCreate(Creature* pCreature)
54 {
55  switch (pCreature->GetEntry())
56  {
57  // Store the Guids of the NPCs in the entrance room to remove them
58  // and spawn their replacement or Rattlegore's death
59  // (only store those located in the room volume)
60  case NPC_REANIMATED_CORPSE:
61  case NPC_DISEASED_GHOUL:
62  case NPC_RISEN_ABERRATION:
63  if (GetData(TYPE_RATTLEGORE) != DONE && (pCreature->GetPositionZ() > aEntranceRoom.m_fCenterZ) && (pCreature->GetPositionX() - aEntranceRoom.m_fCornerX < aEntranceRoom.m_uiLength) && (pCreature->GetPositionY() - aEntranceRoom.m_fCornerY < aEntranceRoom.m_uiWidth))
64  m_sEntranceRoomGuids.insert(pCreature->GetObjectGuid());
65  break;
66  case NPC_SCHOLOMANCE_STUDENT:
67  m_sViewingRoomGuids.insert(pCreature->GetObjectGuid());
68  break;
69  case NPC_VECTUS:
70  case NPC_MARDUK_BLACKPOOL:
71  case NPC_DARKMASTER_GANDLING:
72  m_npcEntryGuidStore[pCreature->GetEntry()] = pCreature->GetObjectGuid();
73  break;
74  case NPC_RISEN_GUARDIAN:
75  GandlingEventMap::iterator find = m_mGandlingData.find(m_uiGandlingEvent);
76  if (find != m_mGandlingData.end())
77  find->second.m_sAddGuids.insert(pCreature->GetGUIDLow());
78  break;
79  }
80 }

105 void instance_scholomance::DoRespawnEntranceRoom(Player* pSummoner)
106 {
107  // safety check to avoid the room being reset for each OnPlayerEnter() call if Rattlegore is dead
108  if (m_bIsRoomReset)
109  return;
110 
111  if (!pSummoner)
112  return;
113 
114  // Despawn the mobs already in the room with the exception of the necrofiend (not stored, so not despawned)
115  for (auto m_sEntranceRoomGuid : m_sEntranceRoomGuids)
116  {
117  if (Creature* pMob = instance->GetCreature(m_sEntranceRoomGuid))
118  pMob->ForcedDespawn();
119  }
120  // Spawn the new and less numerous groups instead
121  // Four groups, one in each corner
122  // The creatures in each point are random
123  // but follow the generic rule for each group of 4 NPCs:
124  // 2 risen aberrations, 1 diseased ghoul, 1 diseased ghoul/reanimated corpse
125  for (uint8 i = 0; i < MAX_GROUPS; ++i)
126  {
127  std::vector<uint32> uiMobList; // Vector holding the 4 creatures entries for each spawned group
128  uiMobList.push_back(NPC_RISEN_ABERRATION); // 3 static NPC entries
129  uiMobList.push_back(NPC_RISEN_ABERRATION);
130  uiMobList.push_back(NPC_DISEASED_GHOUL);
131 
132  uint32 uiMobEntry; // will hold the last random creature entry
133 
134  // Pick the fourth NPC in the group and randomize the four possible spawns
135  switch (urand(0, 1))
136  {
137  case 0: uiMobEntry = NPC_REANIMATED_CORPSE; break;
138  case 1: uiMobEntry = NPC_DISEASED_GHOUL; break;
139  }
140 
141  uiMobList.push_back(uiMobEntry);
142  std::shuffle(uiMobList.begin(), uiMobList.end(), *GetRandomGenerator());
143 
144  for (uint8 j = 0; j < MAX_NPC_PER_GROUP; ++j)
145  pSummoner->SummonCreature(uiMobList[j], aEntranceRoomSpawnLocs[4 * i + j].m_fX, aEntranceRoomSpawnLocs[4 * i + j].m_fY, aEntranceRoomSpawnLocs[4 * i + j].m_fZ, aEntranceRoomSpawnLocs[4 * i + j].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0);
146  }
147  // spawn also a patrolling necrofiend
148  // the waypoints are handled in DB creature_movement_template table (shared with the other necrofiend in the room)
149  // the two other necrofiends in the instance are using DB creature_movement table
150  if (Creature* pNecrofiend = pSummoner->SummonCreature(NPC_NECROFIEND, aEntranceRoomSpawnLocs[16].m_fX, aEntranceRoomSpawnLocs[16].m_fY, aEntranceRoomSpawnLocs[16].m_fZ, aEntranceRoomSpawnLocs[16].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0))
151  pNecrofiend->GetMotionMaster()->MoveWaypoint();
152 
153  m_bIsRoomReset = true;
154 
155  debug_log("SD2: Entrance room in Scholomance reset after Rattlegore's death");
156 }

158 void instance_scholomance::SetData(uint32 uiType, uint32 uiData)
159 {
160  switch (uiType)
161  {
162  case TYPE_KIRTONOS:
163  // This door is initially closed by DB-scripts, so only use it in case of FAIL, DONE, or on aggro after wipe
164  if (m_auiEncounter[uiType] != FAIL && uiData == IN_PROGRESS)
165  return;
166  m_auiEncounter[uiType] = uiData;
167  DoUseDoorOrButton(GO_GATE_KIRTONOS);
168  break;
169  case TYPE_RATTLEGORE:
170  m_auiEncounter[uiType] = uiData;
171  if (uiData == DONE)
172  {
173  if (Player* pPlayer = GetPlayerInMap())
174  DoRespawnEntranceRoom(pPlayer);
175  }
176  break;
177  case TYPE_RAS_FROSTWHISPER:
178  m_auiEncounter[uiType] = uiData;
179  DoUseDoorOrButton(GO_GATE_RAS);
180  break;
181  case TYPE_MALICIA: // TODO this code can be simplified, when it is known which event-ids correspond to which room
182  m_auiEncounter[uiType] = uiData;
183  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_MALICIA].m_doorGuid);
184  break;
185  case TYPE_THEOLEN:
186  m_auiEncounter[uiType] = uiData;
187  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_THEOLEN].m_doorGuid);
188  break;
189  case TYPE_POLKELT:
190  m_auiEncounter[uiType] = uiData;
191  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_POLKELT].m_doorGuid);
192  break;
193  case TYPE_RAVENIAN:
194  m_auiEncounter[uiType] = uiData;
195  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_RAVENIAN].m_doorGuid);
196  break;
197  case TYPE_ALEXEI_BAROV:
198  m_auiEncounter[uiType] = uiData;
199  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_BAROV].m_doorGuid);
200  break;
201  case TYPE_ILLUCIA_BAROV:
202  m_auiEncounter[uiType] = uiData;
203  DoUseDoorOrButton(m_mGandlingData[EVENT_ID_ILLUCIA].m_doorGuid);
204  break;
205  case TYPE_GANDLING:
206  m_auiEncounter[uiType] = uiData;
207  // Close the door to main room, because the encounter will take place only in the main hall and random around all the 6 rooms
208  DoUseDoorOrButton(GO_GATE_GANDLING);
209  break;
210  }
211 
212  // Summon Gandling
213  if (uiData == DONE)
214  DoSpawnGandlingIfCan(false);
215 
216  if (uiData == DONE)
217  {
218  OUT_SAVE_INST_DATA;
219 
220  std::ostringstream saveStream;
221  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
222  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
223  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " " << m_auiEncounter[9];
224 
225  m_strInstData = saveStream.str();
226 
227  SaveToDB();
228  OUT_SAVE_INST_DATA_COMPLETE;
229  }

232 void instance_scholomance::DoSpawnGandlingIfCan(bool bByPlayerEnter)
233 {
234  // Do not summon, if event finished
235  if (m_auiEncounter[TYPE_GANDLING] == DONE)
236  return;
237 
238  // Summon only once
239  if (GetSingleCreatureFromStorage(NPC_DARKMASTER_GANDLING))
240  return;
241 
242  Player* pPlayer = GetPlayerInMap();
243  if (!pPlayer)
244  return;
245 
246  // Check if all the six bosses are done first
247  if (m_auiEncounter[TYPE_MALICIA] == DONE && m_auiEncounter[TYPE_THEOLEN] == DONE && m_auiEncounter[TYPE_POLKELT] == DONE &&
248  m_auiEncounter[TYPE_RAVENIAN] == DONE && m_auiEncounter[TYPE_ALEXEI_BAROV] == DONE && m_auiEncounter[TYPE_ILLUCIA_BAROV] == DONE)
249  {
250  if (Creature* pGandling = pPlayer->SummonCreature(NPC_DARKMASTER_GANDLING, aGandlingSpawnLocs[0].m_fX, aGandlingSpawnLocs[0].m_fY, aGandlingSpawnLocs[0].m_fZ, aGandlingSpawnLocs[0].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0))
251  {
252  // Switch model to Christmas Gandling if Winter Veil event is active
253  if (IsHolidayActive(HOLIDAY_FEAST_OF_WINTER_VEIL))
254  pGandling->CastSpell(pGandling, SPELL_XMAS_GANDLING, TRIGGERED_NONE);
255 
256  if (!bByPlayerEnter)
257  DoScriptText(SAY_GANDLING_SPAWN, pGandling);
258  }
259  }

262 void instance_scholomance::HandlePortalEvent(uint32 uiEventId, uint32 uiData)
263 {
264  GandlingEventMap::iterator find = m_mGandlingData.find(uiEventId);
265  if (find == m_mGandlingData.end())
266  return;
267 
268  if (uiData == SPECIAL)
269  {
270  // Set current Event index
271  m_uiGandlingEvent = uiEventId;
272 
273  // Close door if needed
274  if (!find->second.m_bIsActive)
275  {
276  find->second.m_bIsActive = true;
277  DoUseDoorOrButton(find->second.m_doorGuid);
278  }
279  }
280  // Toggle door and event state in case of state-switch
281  else
282  {
283  if ((uiData == IN_PROGRESS && !find->second.m_bIsActive) ||
284  (uiData == FAIL && find->second.m_bIsActive) ||
285  (uiData == DONE && find->second.m_bIsActive))
286  {
287  find->second.m_bIsActive = !find->second.m_bIsActive;
288  DoUseDoorOrButton(find->second.m_doorGuid);
289  }
290  }
291 }

402 
403  case NPC_RISEN_GUARDIAN:
404  for (auto& itr : m_mGandlingData)
405  {
406  // if there are no minions for a room, skip it
407  if (!itr.second.m_sAddGuids.empty())
408  {
409  // search for the dead minion and erase it
410  if (itr.second.m_sAddGuids.find(pCreature->GetGUIDLow()) != itr.second.m_sAddGuids.end())
411  {
412  itr.second.m_sAddGuids.erase(pCreature->GetGUIDLow());
413 
414  // if the current list is empty; set event id as done
415  if (itr.second.m_sAddGuids.empty())
416  {
417  HandlePortalEvent(itr.first, DONE);
418  break;
419  }
420  }
421  }
422  }
423  break;
424  }

438 void instance_scholomance::HandleDawnGambitEvent()
439 {
440  if (m_uiGambitTransformTimer == 0)
441  {
442  m_uiGambitTransformTimer = 12 * IN_MILLISECONDS;
443  return;
444  }
445  m_uiGambitTransformTimer = 0;
446 
447  // Make each of the Scholomance Student cast "Viewing Room Student Transform - Effect" on himself
448  // Change faction to hostile and roam in the room
449  for (auto m_sViewingRoomGuid : m_sViewingRoomGuids)
450  {
451  if (Creature* pStudent = instance->GetCreature(m_sViewingRoomGuid))
452  {
453  pStudent->SetFactionTemporary(FACTION_SCOURGE, TEMPFACTION_RESTORE_RESPAWN);
454  pStudent->SetStandState(UNIT_STAND_STATE_STAND);
455  pStudent->GetMotionMaster()->MoveRandomAroundPoint(pStudent->GetPositionX(), pStudent->GetPositionY(), pStudent->GetPositionZ(), 2.0f);
456  pStudent->CastSpell(pStudent, SPELL_STUDENT_TRANSFORM, TRIGGERED_NONE);
457  }
458  }
459  // Change faction to hostile for Marduk and Vectus
460  if (Creature* pMarduk = GetSingleCreatureFromStorage(NPC_MARDUK_BLACKPOOL))
461  pMarduk->SetFactionTemporary(FACTION_SCOURGE, TEMPFACTION_RESTORE_RESPAWN);
462  if (Creature* pVectus = GetSingleCreatureFromStorage(NPC_VECTUS))
463  {
464  pVectus->SetFactionTemporary(FACTION_SCOURGE, TEMPFACTION_RESTORE_RESPAWN);
465  DoScriptText(YELL_VECTUS_GAMBIT, pVectus);
466  }
467 }
