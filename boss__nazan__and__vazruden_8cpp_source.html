178  void MovementInform(uint32 motionType, uint32 pointId) override
179  {
180  if (!m_instance)
181  return;
182 
183  if (motionType == POINT_MOTION_TYPE)
184  {
185  switch (pointId)
186  {
187  case POINT_ID_CENTER:
188  DoSplit();
189  break;
190  case POINT_ID_LANDING:
191  {
192  m_creature->GetMotionMaster()->MovePoint(POINT_ID_LANDED, landingPhasePos.x, landingPhasePos.y, landingPhasePos.z);
193  break;
194  }
195  case POINT_ID_LANDED:
196  {
197  m_instance->SetData(TYPE_NAZAN, IN_PROGRESS);
198 
199  // Landing
200  // undo flying
201  m_creature->RemoveByteFlag(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_FLY_ANIM);
202  m_creature->SetByteFlag(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_ALWAYS_STAND);
203  m_creature->SetLevitate(false);
204  m_creature->SetHover(false);
205  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
206  m_creature->SetCanFly(false);
207  HandleAttackDelay();
208  break;
209  }
210  case POINT_ID_FLYING:
211  HandlePhaseChange(false);
212  // [[fallthrough]]
213  case POINT_ID_FLIGHT_MOVE:
214  SetCombatScriptStatus(false);
215  m_creature->SetImmobilizedState(true);
216  m_creature->CastSpell(nullptr, SPELL_FACE_HIGHEST_THREAT, TRIGGERED_OLD_TRIGGERED);
217  break;
218  }
219  }

312  void ExecuteAction(uint32 action) override
313  {
314  switch (action)
315  {
316  case NAZAN_LAND:
317  {
318  if (!CanExecuteCombatAction() || m_creature->GetHealthPercent() > 20.0f)
319  return;
320 
321  DoMoveToCombat();
322  break;
323  }
324  case NAZAN_FLIGHT_MOVE:
325  {
326  if (!CanExecuteCombatAction())
327  return;
328 
329  m_lastFlightPoint = (m_lastFlightPoint + urand(0, 1)) % 3;
330  m_creature->SetImmobilizedState(false);
331  m_creature->GetMotionMaster()->MovePoint(POINT_ID_FLIGHT_MOVE, flyingPositions[m_lastFlightPoint].x, flyingPositions[m_lastFlightPoint].y, flyingPositions[m_lastFlightPoint].z);
332  SetCombatScriptStatus(true);
333  ResetCombatAction(action, urand(25000, 35000));
334  break;
335  }
336  case NAZAN_FIREBALL_FLIGHT:
337  {
338  if (Creature* vazruden = m_creature->GetMap()->GetCreature(m_vazrudenGuid))
339  {
340  if (Unit* target = vazruden->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
341  {
342  if (DoCastSpellIfCan(target, m_inRegularMode ? SPELL_FIREBALL_FLIGHT : SPELL_FIREBALL_FLIGHT_H) == CAST_OK)
343  ResetCombatAction(action, urand(2100, 7300));
344  }
345  }
346  break;
347  }
348  case NAZAN_FIREBALL_GROUND:
349  {
350  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
351  {
352  if (DoCastSpellIfCan(target, m_inRegularMode ? SPELL_FIREBALL_GROUND : SPELL_FIREBALL_GROUND_H) == CAST_OK)
353  ResetCombatAction(action, urand(7300, 13200));
354  }
355  break;
356  }
357  case NAZAN_CONE_OF_FIRE:
358  {
359  if (DoCastSpellIfCan(m_creature->GetVictim(), m_inRegularMode ? SPELL_CONE_OF_FIRE : SPELL_CONE_OF_FIRE_H) == CAST_OK)
360  ResetCombatAction(action, urand(7300, 13200));
361  break;
362  }
363  case NAZAN_BELLOWING_ROAR:
364  {
365  if (DoCastSpellIfCan(m_creature, SPELL_BELLOW_ROAR_H) == CAST_OK)
366  ResetCombatAction(action, urand(40000, 50000));
367  break;
368  }
369  }
370  }
