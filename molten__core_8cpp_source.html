79 void instance_molten_core::OnObjectCreate(GameObject* pGo)
80 {
81  switch (pGo->GetEntry())
82  {
83  // Runes
84  case GO_RUNE_KRESS:
85  case GO_RUNE_MOHN:
86  case GO_RUNE_BLAZ:
87  case GO_RUNE_MAZJ:
88  case GO_RUNE_ZETH:
89  case GO_RUNE_THERI:
90  case GO_RUNE_KORO:
91  // Activate the rune if it was previously doused by a player (encounter set to SPECIAL)
92  m_goEntryGuidStore[pGo->GetEntry()] = pGo->GetObjectGuid();
93  for (auto m_aMoltenCoreRune : m_aMoltenCoreRunes)
94  {
95  if (m_aMoltenCoreRune.m_uiRuneEntry == pGo->GetEntry() && GetData(m_aMoltenCoreRune.m_uiType) == DONE)
96  {
97  pGo->UseDoorOrButton();
98  break;
99  }
100  }
101  break;
102  // Runes' Flames Circles
103  case GO_CIRCLE_MAGMADAR:
104  case GO_CIRCLE_GEHENNAS:
105  case GO_CIRCLE_GARR:
106  case GO_CIRCLE_SHAZZRAH:
107  case GO_CIRCLE_BARON_GEDDON:
108  case GO_CIRCLE_SULFURON:
109  case GO_CIRCLE_GOLEMAGG:
110  // Delete the Flames Circle around the rune if the boss guarding it is killed
111  m_goEntryGuidStore[pGo->GetEntry()] = pGo->GetObjectGuid();
112  for (auto m_aMoltenCoreRune : m_aMoltenCoreRunes)
113  {
114  if (m_aMoltenCoreRune.m_uiFlamesCircleEntry == pGo->GetEntry() && GetData(m_aMoltenCoreRune.m_uiType) == DONE)
115  {
116  pGo->SetLootState(GO_JUST_DEACTIVATED);
117  break;
118  }
119  }
120  break;
121 
122  // Majordomo event chest
123  case GO_CACHE_OF_THE_FIRE_LORD:
124  // Ragnaros GOs
125  case GO_LAVA_STEAM:
126  case GO_LAVA_SPLASH:
127  m_goEntryGuidStore[pGo->GetEntry()] = pGo->GetObjectGuid();
128  break;
129  }
130 }

132 void instance_molten_core::SetData(uint32 uiType, uint32 uiData)
133 {
134  switch (uiType)
135  {
136  case TYPE_LUCIFRON:
137  m_auiEncounter[uiType] = uiData;
138  break;
139  case TYPE_MAGMADAR:
140  case TYPE_GEHENNAS:
141  case TYPE_GARR:
142  case TYPE_SHAZZRAH:
143  case TYPE_GEDDON:
144  case TYPE_GOLEMAGG:
145  case TYPE_SULFURON:
146  m_auiEncounter[uiType] = uiData;
147  if (uiData == DONE)
148  {
149  for (auto m_aMoltenCoreRune : m_aMoltenCoreRunes)
150  {
151  if (m_aMoltenCoreRune.m_uiType == uiType)
152  {
153  if (GameObject* pGo = GetSingleGameObjectFromStorage(m_aMoltenCoreRune.m_uiFlamesCircleEntry))
154  pGo->SetLootState(GO_JUST_DEACTIVATED);
155  DoUseDoorOrButton(m_aMoltenCoreRune.m_uiRuneEntry);
156  break;
157  }
158  }
159  }
160  break;
161  case TYPE_MAJORDOMO:
162  m_auiEncounter[uiType] = uiData;
163  if (uiData == DONE)
164  {
165  DoRespawnGameObject(GO_CACHE_OF_THE_FIRE_LORD, HOUR);
166  if (GameObject* chest = GetSingleGameObjectFromStorage(GO_CACHE_OF_THE_FIRE_LORD))
167  if (Player* player = GetPlayerInMap(false, false))
168  chest->GenerateLootFor(player);
169  }
170  break;
171  case TYPE_RAGNAROS:
172  m_auiEncounter[uiType] = uiData;
173  break;
174  }
175 
176  // Check if Majordomo can be summoned
177  if (uiData == DONE)
178  DoSpawnMajordomoIfCan(false);
179 
180  if (uiData == DONE)
181  {
182  OUT_SAVE_INST_DATA;
183 
184  std::ostringstream saveStream;
185  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
186  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
187  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
188  << m_auiEncounter[9];
189 
190  m_strInstData = saveStream.str();
191 
192  SaveToDB();
193  OUT_SAVE_INST_DATA_COMPLETE;
194  }
195 }

205 // Handle Majordomo summon here
206 void instance_molten_core::DoSpawnMajordomoIfCan(bool bByPlayerEnter)
207 {
208  // If both Majordomo and Ragnaros events are finished, return
209  if (m_auiEncounter[TYPE_MAJORDOMO] == DONE && m_auiEncounter[TYPE_RAGNAROS] == DONE)
210  return;
211 
212  // If already spawned return
213  if (GetSingleCreatureFromStorage(NPC_MAJORDOMO, true))
214  return;
215 
216  // Check if all rune bosses are done
217  for (uint8 i = TYPE_MAGMADAR; i < TYPE_MAJORDOMO; ++i)
218  {
219  if (m_auiEncounter[i] != DONE)
220  return;
221  }
222 
223  Player* pPlayer = GetPlayerInMap();
224  if (!pPlayer)
225  return;
226 
227  uint8 uiSummonPos = m_auiEncounter[TYPE_MAJORDOMO] == DONE ? 1 : 0;
228  SpawnMajordomo(pPlayer, !bByPlayerEnter, uiSummonPos);
229 }

231 void instance_molten_core::SpawnMajordomo(Unit* summoner, bool initialSummon, uint8 summonPos)
232 {
233  // Summon Majordomo
234  // If Majordomo encounter isn't done, summon at encounter place, else near Ragnaros
235  if (Creature* pMajordomo = summoner->SummonCreature(m_aMajordomoLocations[summonPos].m_uiEntry, m_aMajordomoLocations[summonPos].m_fX, m_aMajordomoLocations[summonPos].m_fY, m_aMajordomoLocations[summonPos].m_fZ, m_aMajordomoLocations[summonPos].m_fO, TEMPSPAWN_MANUAL_DESPAWN, 2 * HOUR * IN_MILLISECONDS))
236  {
237  if (summonPos) // Majordomo encounter already done, set faction
238  {
239  pMajordomo->SetFactionTemporary(FACTION_MAJORDOMO_FRIENDLY, TEMPFACTION_RESTORE_RESPAWN);
240  pMajordomo->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
241  pMajordomo->SetFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
242  }
243  else // Else yell and summon adds
244  {
245  if (initialSummon)
246  DoScriptText(SAY_MAJORDOMO_SPAWN, pMajordomo);
247 
248  for (auto& m_aBosspawnLoc : m_aBosspawnLocs)
249  pMajordomo->SummonCreature(m_aBosspawnLoc.m_uiEntry, m_aBosspawnLoc.m_fX, m_aBosspawnLoc.m_fY, m_aBosspawnLoc.m_fZ, m_aBosspawnLoc.m_fO, TEMPSPAWN_MANUAL_DESPAWN, DAY * IN_MILLISECONDS);
250  }
251  }
252 }

254 void instance_molten_core::Load(const char* chrIn)
255 {
256  if (!chrIn)
257  {
258  OUT_LOAD_INST_DATA_FAIL;
259  return;
260  }
261 
262  OUT_LOAD_INST_DATA(chrIn);
263 
264  std::istringstream loadStream(chrIn);
265

267  >> m_auiEncounter[4] >> m_auiEncounter[5] >> m_auiEncounter[6] >> m_auiEncounter[7]
268  >> m_auiEncounter[8] >> m_auiEncounter[9];
269 
270  for (uint32& i : m_auiEncounter)
271  {
272  if (i == IN_PROGRESS)
273  i = NOT_STARTED;
274  }
275 
276  OUT_LOAD_INST_DATA_COMPLETE;
277 }

284 void instance_molten_core::ExecuteChatCommand(ChatHandler* handler, char* args)
285 {
286  char* result = handler->ExtractLiteralArg(&args);
287  if (!result)
288  return;
289  std::string val = result;
290  if (val == "spawnmajordomo")
291  {
292  uint32 summonPos;
293  handler->ExtractUInt32(&args, summonPos);
294 
295  if (summonPos > 1)
296  {
297  handler->PSendSysMessage("Could not spawn majordomo because spawn %u was invalid. Please specify a value between 0-1", summonPos);
298  return;
299  }
300 
301  SpawnMajordomo(handler->GetSession()->GetPlayer(), true, summonPos);
