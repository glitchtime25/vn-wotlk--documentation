68 // Function to delete the data
69 void SQLStorageBase::Free()
70 {
71  if (!m_data)
72  return;
73 
74  uint32 offset = 0;
75  for (uint32 x = 0; x < m_dstFieldCount; ++x)
76  {
77  switch (m_dst_format[x])
78  {
79  case FT_LOGIC:
80  offset += sizeof(bool);
81  break;
82  case FT_STRING:
83  {
84  for (uint32 recordItr = 0; recordItr < m_recordCount; ++recordItr)
85  delete[] *(char**)((char*)(m_data + (recordItr * m_recordSize)) + offset);
86 
87  offset += sizeof(char*);
88  break;
89  }
90  case FT_NA:
91  case FT_INT:
92  offset += sizeof(uint32);
93  break;
94  case FT_BYTE:
95  case FT_NA_BYTE:
96  offset += sizeof(char);
97  break;
98  case FT_FLOAT:
99  case FT_NA_FLOAT:
100  offset += sizeof(float);
101  break;
102  case FT_NA_POINTER:
103  // TODO- possible (and small) memleak here possible
104  offset += sizeof(char*);
105  break;
106  case FT_64BITINT:
107  offset += sizeof(uint64);
108  break;
109  case FT_IND:
110  case FT_SORT:
111  assert(false && "SQL storage not have sort field types");
112  break;
113  default:
114  assert(false && "unknown format character");
115  break;
116  }
117  }
118  delete[] m_data;
119  m_data = nullptr;
120  m_recordCount = 0;
121 }
