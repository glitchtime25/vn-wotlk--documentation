162 {
163  switch (creature->GetEntry())
164  {
165  case NPC_MEDICINE_MAN:
166  case NPC_TRIBES_MAN:
167  case NPC_WARBRINGER:

169  if (m_uiBearEventPhase > 3) // prevent crashes during testing
170  return;

173  ++m_nalorakkEvent[m_uiBearEventPhase].trashKilledCount;

199  case NPC_TRIBES_MAN:

217  m_isBearPhaseInProgress = false;
218  break;
219  case NPC_TEMPEST:
220  case NPC_PROTECTOR:
221  case NPC_WIND_WALKER:

227  Creature* pTemp = instance->GetCreature(itr);

231

339 void instance_zulaman::SetData(uint32 type, uint32 data)
340 {
341  debug_log("SD2: Instance Zulaman: SetData received for type %u with data %u", type, data);
342 
343  switch (type)
344  {
345  case TYPE_EVENT_RUN:
346  if (data == SPECIAL)
347  {
348  m_auiEncounter[TYPE_EVENT_RUN] = data;
349  if (GameObject * gong = GetSingleGameObjectFromStorage(GO_STRANGE_GONG))
350  gong->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
351  if (Creature * harrisonJones = GetSingleCreatureFromStorage(NPC_HARRISON))
352  harrisonJones->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, harrisonJones, harrisonJones);
353  return;
354  }
355  if (data == IN_PROGRESS)
356  {
357  DoTimeRunSay(RUN_START);
358  DoUseDoorOrButton(GO_MASSIVE_GATE);
359  if (m_auiEncounter[TYPE_RUN_EVENT_TIME])
360  SetData(TYPE_RUN_EVENT_TIME, m_auiEncounter[TYPE_RUN_EVENT_TIME]);
361  else
362  SetData(TYPE_RUN_EVENT_TIME, 20); // 20 Minutes as default time
363 
364  static_cast<DungeonMap*>(instance)->PermBindAllPlayers();
365  }
366  if (data == FAIL)
367  {
368  DoTimeRunSay(RUN_FAIL);
369  // Kill remaining Event NPCs
370  for (auto& i : m_aEventNpcInfo)
371  {
372  // Not yet rescued, so too late
373  if (!i.uiSavePosition)
374  {
375  if (Creature* creature = instance->GetCreature(i.npGuid))
376  creature->ForcedDespawn();
377  }
378  }
379  }
380  if (data == DONE)
381  DoTimeRunSay(RUN_DONE);
382  m_auiEncounter[type] = data;
383  instance->GetVariableManager().SetVariable(WORLD_STATE_ZUL_AMAN_EVENT_RUN_IS_ACTIVE, data == IN_PROGRESS);
384  break;

388  if (data == DONE)

390  DoUseDoorOrButton(GO_ALTAR_TORCH_EAGLE_GOD);

392  {
393  m_auiEncounter[TYPE_RUN_EVENT_TIME] += 10; // Add 10 minutes

395  DoChestEvent(INDEX_AKILZON);
396  }
397  }
398  m_auiEncounter[type] = data;
399  break;
400  case TYPE_NALORAKK:

403  DoUseDoorOrButton(GO_ALTAR_TORCH_BEAR_GOD);

406  m_auiEncounter[TYPE_RUN_EVENT_TIME] += 15; // Add 15 minutes

410  }

526 void instance_zulaman::Load(const char* chrIn)
527 {
528  if (!chrIn)
529  {
530  OUT_LOAD_INST_DATA_FAIL;
531  return;
532  }
533 
534  OUT_LOAD_INST_DATA(chrIn);
535 
536  std::istringstream loadStream(chrIn);
537  loadStream >> m_auiEncounter[0] >> m_auiEncounter[1] >> m_auiEncounter[2] >> m_auiEncounter[3]
538  >> m_auiEncounter[4] >> m_auiEncounter[5] >> m_auiEncounter[6] >> m_auiEncounter[7]
539  >> m_auiEncounter[8] >> m_auiEncounter[9] >> m_auiEncounter[10] >> m_auiEncounter[11];
540 
541  // Skip m_auiEncounter[7], to start the time event properly if needed
542  for (uint8 i = 0; i < MAX_ENCOUNTER - 1; ++i)
543  {
544  if (m_auiEncounter[i] == IN_PROGRESS)
545  m_auiEncounter[i] = NOT_STARTED;
546  }
547 
548  // Restart TYPE_EVENT_RUN if was already started
549  if (m_auiEncounter[TYPE_RUN_EVENT_TIME] != 0 && m_auiEncounter[TYPE_EVENT_RUN] != DONE && m_auiEncounter[TYPE_EVENT_RUN] != FAIL)
550  SetData(TYPE_EVENT_RUN, IN_PROGRESS);
551  if (GetData(TYPE_AKILZON) == DONE)
552  SetData(TYPE_AKILZON_GAUNTLET, DONE);
553 
554  OUT_LOAD_INST_DATA_COMPLETE;
555 }
