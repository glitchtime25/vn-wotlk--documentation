107  void UpdateAI(const uint32 uiDiff) override
108  {
109  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
110  return;
111 
112  // Check if creature is below 66% or 33%; Also don't allow it to split the third time
113  if (m_creature->GetHealthPercent() < 100 - 33 * m_uiSplitPhase && m_creature->GetHealthPercent() > 5.0f)
114  {
115  DoCastSpellIfCan(m_creature, m_uiSplitPhase == 1 ? SPELL_66_ILLUSION : SPELL_33_ILLUSION, CAST_INTERRUPT_PREVIOUS);
116  DoScriptText(SAY_IMAGE, m_creature);
117  ++m_uiSplitPhase;
118  }
119 
120  if (m_uiMindRendTimer < uiDiff)
121  {
122  Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
123  if (!pTarget)
124  pTarget = m_creature->GetVictim();
125 
126  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_MIND_REND : SPELL_MIND_REND_H) == CAST_OK)
127  m_uiMindRendTimer = 8000;
128  }
129  else
130  m_uiMindRendTimer -= uiDiff;
131 
132  if (m_uiFearTimer < uiDiff)
133  {
134  Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
135  if (!pTarget)
136  pTarget = m_creature->GetVictim();
137 
138  if (DoCastSpellIfCan(pTarget, SPELL_FEAR) == CAST_OK)
139  {
140  DoScriptText(urand(0, 1) ? SAY_FEAR_1 : SAY_FEAR_2, m_creature);
141  m_uiFearTimer = 25000;
142  }
143  }
144  else
145  m_uiFearTimer -= uiDiff;
146 
147  if (m_uiDominationTimer < uiDiff)
148  {
149  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, uint32(0), SELECT_FLAG_PLAYER))
150  {
151  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_DOMINATION : SPELL_DOMINATION_H) == CAST_OK)
152  {
153  DoScriptText(urand(0, 1) ? SAY_MIND_1 : SAY_MIND_2, m_creature);
154  m_uiDominationTimer = urand(16000, 32000);
155  }
156  }
157  }
158  else
159  m_uiDominationTimer -= uiDiff;
160 
161  if (!m_bIsRegularMode)
162  {
163  if (m_uiManaBurnTimer < uiDiff)
164  {
165  Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA);
166  if (!pTarget)
167  pTarget = m_creature->GetVictim();
168 
169  if (DoCastSpellIfCan(pTarget, SPELL_MANA_BURN_H) == CAST_OK)
170  m_uiManaBurnTimer = urand(16000, 32000);
171  }
172  else
173  m_uiManaBurnTimer -= uiDiff;
174  }
175 
176  DoMeleeAttackIfReady();
177  }
