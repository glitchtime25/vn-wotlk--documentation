173  void UpdateAI(const uint32 uiDiff) override
174  {
175  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
176  return;
177 
178  if (m_uiCataclysmicBoltTimer < uiDiff)
179  {
180  // select a random unit with mana other than the main tank
181  Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, uint32(0), SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA);
182 
183  // if there aren't other units, cast on the tank
184  if (!target)
185  target = m_creature->GetVictim();
186 
187  if (target)
188  {
189  if (DoCastSpellIfCan(target, SPELL_CATACLYSMIC_BOLT) == CAST_OK)
190  m_uiCataclysmicBoltTimer = 10000;
191  }
192  }
193  else
194  m_uiCataclysmicBoltTimer -= uiDiff;
195 
196  if (!m_bBlessingOfTides && m_creature->GetHealthPercent() < 75.0f)
197  {
198  for (unsigned int aAdvisor : aAdvisors)
199  {
200  if (Creature* pAdvisor = m_instance->GetSingleCreatureFromStorage(aAdvisor))
201  {
202  // stack max three times (one for each alive)
203  if (pAdvisor->IsAlive())
204  {
205  pAdvisor->InterruptNonMeleeSpells(false);
206  pAdvisor->CastSpell(m_creature, SPELL_BLESSING_OF_THE_TIDES, TRIGGERED_OLD_TRIGGERED);
207  }
208  }
209  }
210 
211  // yell if we now have the aura
212  if (m_creature->HasAura(SPELL_BLESSING_OF_THE_TIDES))
213  DoScriptText(SAY_GAIN_BLESSING, m_creature);
214 
215  m_bBlessingOfTides = true;
216  }
217 
218  if (m_uiSearingNovaTimer < uiDiff)
219  {
220  if (DoCastSpellIfCan(m_creature, SPELL_SEAR_NOVA) == CAST_OK)
221  m_uiSearingNovaTimer = urand(20000, 30000);
222  }
223  else
224  m_uiSearingNovaTimer -= uiDiff;
225 
226  if (m_uiEnrageTimer)
227  {
228  if (m_uiEnrageTimer <= uiDiff)
229  {
230  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
231  m_uiEnrageTimer = 0;
232  }
233  else
234  m_uiEnrageTimer -= uiDiff;
235  }
236 
237  DoMeleeAttackIfReady();
238  }
