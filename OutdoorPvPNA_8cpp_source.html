105 void OutdoorPvPNA::HandleCreatureCreate(Creature* creature)

110  case NPC_QUARTERMASTER_DAVIAN:
111  case NPC_MERCHANT_ALDRAAN:
112  case NPC_VENDOR_CENDRII:
113  case NPC_AMMUNITIONER_BANRO:
114  case NPC_RESEARCHER_AMERELDINE:
115  case NPC_QUARTERMASTER_NORELIQE:
116  case NPC_MERCHANT_COREIEL:
117  case NPC_VENDOR_EMBELAR:

119  m_teamVendors.push_back(creature->GetObjectGuid());
120  break;

129  LockHalaa(creature);
130 
131  // update world state
132  SendUpdateWorldState(m_zoneMapState, WORLD_STATE_REMOVE);
133  m_zoneMapState = m_zoneOwner == ALLIANCE ? WORLD_STATE_NA_HALAA_ALLIANCE : WORLD_STATE_NA_HALAA_HORDE;
134  SendUpdateWorldState(m_zoneMapState, WORLD_STATE_ADD);
135  }
136 
137  ++m_guardsLeft;
138  SendUpdateWorldState(WORLD_STATE_NA_GUARDS_LEFT, m_guardsLeft);
139  break;
140  }

143 void OutdoorPvPNA::HandleCreatureDeath(Creature* creature)
144 {
145  if (creature->GetEntry() != NPC_HORDE_HALAANI_GUARD && creature->GetEntry() != NPC_ALLIANCE_HANAANI_GUARD)
146  return;
147 
148  // get the location of the dead guard for future respawn
149  float x, y, z, o;
150  creature->GetRespawnCoord(x, y, z, &o);
151  HalaaSoldiersSpawns location = {x, y, z, o};
152  m_deadSoldiers.push(location);
153 
154  // set the respawn timer after the last guard died - 5 min for the first time, or 1 hour if the city is under siege
155  if (!m_soldiersRespawnTimer)
156  m_soldiersRespawnTimer = m_isUnderSiege ? HOUR * IN_MILLISECONDS : 5 * MINUTE * IN_MILLISECONDS;
157 
158  // decrease the counter
159  --m_guardsLeft;
160  SendUpdateWorldState(WORLD_STATE_NA_GUARDS_LEFT, m_guardsLeft);
161 
162  if (m_guardsLeft == 0)
163  {
164  // set the zone under siege and increase the respawn timer
165  m_isUnderSiege = true;
166  m_soldiersRespawnTimer = HOUR * IN_MILLISECONDS;
167 
168  // make capturable
169  UnlockHalaa(creature);
170 
171  // update world state
172  SendUpdateWorldState(m_zoneMapState, WORLD_STATE_REMOVE);
173  m_zoneMapState = m_zoneOwner == ALLIANCE ? WORLD_STATE_NA_HALAA_NEUTRAL_A : WORLD_STATE_NA_HALAA_NEUTRAL_H;
174  SendUpdateWorldState(m_zoneMapState, WORLD_STATE_ADD);
175 
176  sWorld.SendDefenseMessage(ZONE_ID_NAGRAND, LANG_OPVP_NA_DEFENSELESS);
177  }
178 }

285 // process the capture events
286 bool OutdoorPvPNA::HandleEvent(uint32 eventId, Object* source, Object* /*target*/)
287 {
288  if (!source->IsGameObject())
289  return false;
290 
291  GameObject* go = static_cast<GameObject*>(source);
292  // If we are not using the Halaa banner return
293  if (go->GetEntry() != GO_HALAA_BANNER)
294  return false;
295 
296  bool eventHandled = true;
297 
298  switch (eventId)
299  {
300  case EVENT_HALAA_BANNER_WIN_ALLIANCE:
301  ProcessCaptureEvent(go, ALLIANCE);
302  eventHandled = false;
303  break;
304  case EVENT_HALAA_BANNER_WIN_HORDE:
305  ProcessCaptureEvent(go, HORDE);
306  eventHandled = false;
307  break;
308  case EVENT_HALAA_BANNER_PROGRESS_ALLIANCE:
309  SetBannerVisual(go, CAPTURE_ARTKIT_ALLIANCE, CAPTURE_ANIM_ALLIANCE);
310  sWorld.SendDefenseMessage(ZONE_ID_NAGRAND, LANG_OPVP_NA_PROGRESS_A);
311  break;
312  case EVENT_HALAA_BANNER_PROGRESS_HORDE:
313  SetBannerVisual(go, CAPTURE_ARTKIT_HORDE, CAPTURE_ANIM_HORDE);
314  sWorld.SendDefenseMessage(ZONE_ID_NAGRAND, LANG_OPVP_NA_PROGRESS_H);
315  break;
316  }
317 
318  // there are some events which required further DB script
319  return eventHandled;
320 }

322 void OutdoorPvPNA::ProcessCaptureEvent(GameObject* go, Team team)
323 {
324  BuffTeam(m_zoneOwner, SPELL_STRENGTH_HALAANI, true);
325 
326  // update capture point owner
327  m_zoneOwner = team;
328 
329  LockHalaa(go);
330  m_guardsLeft = MAX_NA_GUARDS;
331 
332  m_isUnderSiege = false;
333  m_soldiersRespawnTimer = 0;
334 
335  UpdateWorldState(WORLD_STATE_REMOVE);
336  DespawnVendors(go);
337  SetGraveYardLinkTeam(GRAVEYARD_ID_HALAA, GRAVEYARD_ZONE_ID_HALAA, m_zoneOwner, 530);
338 
339  if (m_zoneOwner == ALLIANCE)
340  {
341  m_zoneWorldState = WORLD_STATE_NA_GUARDS_ALLIANCE;
342  m_zoneMapState = WORLD_STATE_NA_HALAA_ALLIANCE;
343  }
344  else
345  {
346  m_zoneWorldState = WORLD_STATE_NA_GUARDS_HORDE;
347  m_zoneMapState = WORLD_STATE_NA_HALAA_HORDE;
348  }
349 
350  HandleFactionObjects(go);
351  UpdateWorldState(WORLD_STATE_ADD);
352 
353  SendUpdateWorldState(WORLD_STATE_NA_GUARDS_LEFT, m_guardsLeft);
354 
355  BuffTeam(m_zoneOwner, SPELL_STRENGTH_HALAANI);
356  sWorld.SendDefenseMessage(ZONE_ID_NAGRAND, m_zoneOwner == ALLIANCE ? LANG_OPVP_NA_CAPTURE_A : LANG_OPVP_NA_CAPTURE_H);
357 }

498 void OutdoorPvPNA::Update(uint32 diff)
499 {

503  {

507  if (m_guardsLeft == MAX_NA_GUARDS)

512  else

518 void OutdoorPvPNA::RespawnSoldier()

530  player->SummonCreature(m_zoneOwner == ALLIANCE ? NPC_ALLIANCE_HANAANI_GUARD : NPC_HORDE_HALAANI_GUARD, location.x, location.y, location.z, location.o, TEMPSPAWN_DEAD_DESPAWN, 0, true);

532  break;
