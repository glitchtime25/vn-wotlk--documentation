121  void DoSummonMushrooms(bool bIsFirstSummon)
122  {
123  // This implementation may not be 100% accurate;
124  // On aggro boss summons about 20 mushrooms; On timer it summons about 5 mushrooms per turn
125  // There is a 33% chance that the mushroom will be healthy
126  // The summon position is based on the center of the area coords
127 
128  float fX, fY, fZ;
129  uint32 uiMaxMushrooms = bIsFirstSummon ? 20 : 5;
130 
131  for (uint8 i = 0; i < uiMaxMushrooms; ++i)
132  {
133  uint32 uiMushroomEntry = roll_chance_i(33) ? NPC_HEALTHY_MUSHROOM : NPC_POISONOUS_MUSHROOM;
134  m_creature->GetRandomPoint(aMushroomPos[0], aMushroomPos[1], aMushroomPos[2], 30.0f, fX, fY, fZ);
135  m_creature->SummonCreature(uiMushroomEntry, fX, fY, fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 0);
136  }
137  }

139  void UpdateAI(const uint32 uiDiff) override
140  {
141  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
142  return;
143 
144  if (m_uiBashTimer < uiDiff)
145  {
146  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BASH) == CAST_OK)
147  m_uiBashTimer = urand(8000, 13000);
148  }
149  else
150  m_uiBashTimer -= uiDiff;
151 
152  if (m_uiVenomBoltTimer < uiDiff)
153  {
154  if (DoCastSpellIfCan(m_creature, SPELL_VENOM_BOLT_VOLLEY) == CAST_OK)
155  m_uiVenomBoltTimer = urand(15000, 20000);
156  }
157  else
158  m_uiVenomBoltTimer -= uiDiff;
159 
160  if (m_uiRootsTimer < uiDiff)
161  {
162  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
163  {
164  if (DoCastSpellIfCan(pTarget, SPELL_ENTANGLING_ROOTS) == CAST_OK)
165  m_uiRootsTimer = urand(20000, 25000);
166  }
167  }
168  else
169  m_uiRootsTimer -= uiDiff;
170 
171  if (m_uiMiniTimer < uiDiff)
172  {
173  if (DoCastSpellIfCan(m_creature, SPELL_MINI) == CAST_OK)
174  m_uiMiniTimer = 30000;
175  }
176  else
177  m_uiMiniTimer -= uiDiff;
178 
179  if (m_uiMushroomTimer < uiDiff)
180  {
181  DoSummonMushrooms(false);
182  m_uiMushroomTimer = urand(10000, 20000);
183  }
184  else
185  m_uiMushroomTimer -= uiDiff;
186 
187  // ToDo: Research if he requires out of combat area evade check
188 
189  DoMeleeAttackIfReady();
190  }
191 };
