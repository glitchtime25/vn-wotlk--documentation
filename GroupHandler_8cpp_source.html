71 void WorldSession::HandleGroupInviteOpcode(WorldPacket& recv_data)
72 {
73  std::string membername;
74  recv_data >> membername;
75  recv_data.read_skip<uint32>(); // roles mask?
76 
77  // attempt add selected player
78 
79  // cheating
80  if (!normalizePlayerName(membername))
81  {
82  SendPartyResult(PARTY_OP_INVITE, membername, ERR_BAD_PLAYER_NAME_S);
83  return;
84  }
85 
86  Player* initiator = GetPlayer();
87  Player* recipient = sObjectMgr.GetPlayer(membername.c_str());
88 
89  // no player or player trying to invite himself
90  if (!recipient || recipient == initiator)
91  {
92  SendPartyResult(PARTY_OP_INVITE, membername, ERR_BAD_PLAYER_NAME_S);
93  return;
94  }
95 
96  // can't group with
97  if (!GetPlayer()->IsGameMaster() && !sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_GROUP) && initiator->GetTeam() != recipient->GetTeam())
98  {
99  SendPartyResult(PARTY_OP_INVITE, membername, ERR_PLAYER_WRONG_FACTION);
100  return;
101  }
102 
103  if (initiator->GetInstanceId() != 0 && recipient->GetInstanceId() != 0 && initiator->GetInstanceId() != recipient->GetInstanceId() && initiator->GetMapId() == recipient->GetMapId())
104  {
105  SendPartyResult(PARTY_OP_INVITE, membername, ERR_TARGET_NOT_IN_INSTANCE_S);
106  return;
107  }
108 
109  // just ignore us
110  if (recipient->GetSocial()->HasIgnore(initiator->GetObjectGuid()))
111  {
112  SendPartyResult(PARTY_OP_INVITE, membername, ERR_IGNORING_YOU_S);
113  return;
114  }
115 
116  if (GetAnticheat()->IsSilenced())
117  {
118  SendPartyResult(PARTY_OP_INVITE, membername, ERR_PARTY_RESULT_OK);
119  return;
120  }
121 
122  Group* initiatorGroup = initiator->GetGroup();
123  if (initiatorGroup && initiatorGroup->IsBattleGroup())
124  initiatorGroup = initiator->GetOriginalGroup();
125  if (!initiatorGroup)
126  initiatorGroup = initiator->GetGroupInvite();
127 
128  if (initiatorGroup && initiatorGroup->IsRaidGroup() && !recipient->GetAllowLowLevelRaid() && (recipient->GetLevel() < sWorld.getConfig(CONFIG_UINT32_MIN_LEVEL_FOR_RAID)))
129  {
130  SendPartyResult(PARTY_OP_INVITE, "", ERR_RAID_DISALLOWED_BY_LEVEL);
131  return;
132  }
133 
134  // player already invited
135  if (recipient->GetGroupInvite())
136  {
137  SendPartyResult(PARTY_OP_INVITE, membername, ERR_ALREADY_IN_GROUP_S);
138  return;
139  }
140 
141  Group* recipientGroup = recipient->GetGroup();
142  if (recipientGroup && recipientGroup->IsBattleGroup())
143  recipientGroup = recipient->GetOriginalGroup();
144 
145  // player already in another group
146  if (recipientGroup)
147  {
148  SendPartyResult(PARTY_OP_INVITE, membername, ERR_ALREADY_IN_GROUP_S);
149 
150  // tell the player that they were invited but it failed as they were already in a group
151  SendGroupInvite(recipient, true);
152 
153  return;
154  }
155 
156  if (initiatorGroup)
157  {
158  // not have permissions for invite
159  if (!initiatorGroup->IsLeader(initiator->GetObjectGuid()) && !initiatorGroup->IsAssistant(initiator->GetObjectGuid()))
160  {
161  if (initiatorGroup->IsCreated())
162  SendPartyResult(PARTY_OP_INVITE, "", ERR_NOT_LEADER);
163  return;
164  }
165  // not have place
166  if (initiatorGroup->IsFull())
167  {
168  SendPartyResult(PARTY_OP_INVITE, "", ERR_GROUP_FULL);
169  return;
170  }
171  }
172 
173  // ok, but group not exist, start a new group
174  // but don't create and save the group to the DB until
175  // at least one person joins
176  if (!initiatorGroup)
177  {
178  initiatorGroup = new Group();
179  // new group: if can't add then delete
180  if (!initiatorGroup->AddLeaderInvite(initiator))
181  {
182  delete initiatorGroup;
183  return;
184  }
185  if (!initiatorGroup->AddInvite(recipient))
186  {
187  initiatorGroup->RemoveAllInvites();
188  delete initiatorGroup;
189  return;
190  }
191  }
192  else
193  {
194  // already existing group: if can't add then just leave
195  if (!initiatorGroup->AddInvite(recipient))
196  {
197  return;
198  }
199  }
200 
201  // Record targets for uniqueness when spamming
202  GetAnticheat()->PartyInvite(recipient->GetObjectGuid());
203 
204  SendGroupInvite(recipient);
205  SendPartyResult(PARTY_OP_INVITE, membername, ERR_PARTY_RESULT_OK);
206 }

665 void WorldSession::BuildPartyMemberStatsChangedPacket(Player* player, WorldPacket& data)
666 {
667  uint32 mask = player->GetGroupUpdateFlag();
668 
669  if (mask & GROUP_UPDATE_FLAG_POWER_TYPE) // if update power type, update current/max power also
670  mask |= (GROUP_UPDATE_FLAG_CUR_POWER | GROUP_UPDATE_FLAG_MAX_POWER);
671 
672  if (mask & GROUP_UPDATE_FLAG_PET_POWER_TYPE) // same for pets
673  mask |= (GROUP_UPDATE_FLAG_PET_CUR_POWER | GROUP_UPDATE_FLAG_PET_MAX_POWER);
674 
675  uint32 byteCount = 0;
676  for (int i = 1; i < GROUP_UPDATE_FLAGS_COUNT; ++i)
677  if (mask & (1 << i))
678  byteCount += GroupUpdateLength[i];
679 
680  data.Initialize(SMSG_PARTY_MEMBER_STATS, 8 + 4 + byteCount);
681  data << player->GetPackGUID();
682  data << uint32(mask);
683 
684  if (mask & GROUP_UPDATE_FLAG_STATUS)
685  data << uint16(GetGroupMemberStatus(player));
686 
687  if (mask & GROUP_UPDATE_FLAG_CUR_HP)
688  data << uint32(player->GetHealth());
689 
690  if (mask & GROUP_UPDATE_FLAG_MAX_HP)
691  data << uint32(player->GetMaxHealth());
692 
693  Powers powerType = player->GetPowerType();
694  if (mask & GROUP_UPDATE_FLAG_POWER_TYPE)
695  data << uint8(powerType);
696 
697  if (mask & GROUP_UPDATE_FLAG_CUR_POWER)
698  data << uint16(player->GetPower(powerType));
699 
700  if (mask & GROUP_UPDATE_FLAG_MAX_POWER)
701  data << uint16(player->GetMaxPower(powerType));
702 
703  if (mask & GROUP_UPDATE_FLAG_LEVEL)
704  data << uint16(player->GetLevel());
705 
706  if (mask & GROUP_UPDATE_FLAG_ZONE)
707  data << uint16(player->GetZoneId());
708 
709  if (mask & GROUP_UPDATE_FLAG_POSITION)
710  data << uint16(player->GetPositionX()) << uint16(player->GetPositionY());
711 
712  if (mask & GROUP_UPDATE_FLAG_AURAS)
713  {
714  const uint64& auramask = player->GetAuraUpdateMask();
715  data << uint64(auramask);
716  for (uint32 i = 0; i < MAX_AURAS; ++i)
717  {
718  if (auramask & (uint64(1) << i))
719  {
720  data << uint32(player->GetVisibleAura(i));
721  data << uint8(1);
722  }
723  }
724  }
725 
726  Unit* charm = player->GetCharm();
727  if (mask & GROUP_UPDATE_FLAG_PET_GUID)
728  data << (charm ? charm->GetObjectGuid() : ObjectGuid());
729 
730  if (mask & GROUP_UPDATE_FLAG_PET_NAME)
731  {
732  if (charm)
733  data << charm->GetName();
734  else
735  data << uint8(0);
736  }
737 
738  if (mask & GROUP_UPDATE_FLAG_PET_MODEL_ID)
739  {
740  if (charm)
741  data << uint16(charm->GetDisplayId());
742  else
743  data << uint16(0);
744  }
