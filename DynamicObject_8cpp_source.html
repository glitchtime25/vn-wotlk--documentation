60 bool DynamicObject::Create(uint32 guidlow, Unit* caster, uint32 spellId, SpellEffectIndex effIndex, float x, float y, float z, int32 duration, float radius, DynamicObjectType type, SpellTarget target, int32 damage, int32 basePoints)
61 {
62  WorldObject::_Create(guidlow, HIGHGUID_DYNAMICOBJECT, caster->GetPhaseMask());
63  SetMap(caster->GetMap());
64  Relocate(x, y, z, 0);
65 
66  if (!IsPositionValid())
67  {
68  sLog.outError("DynamicObject (spell %u eff %u) not created. Suggested coordinates isn't valid (X: %f Y: %f)", spellId, effIndex, GetPositionX(), GetPositionY());
69  return false;
70  }
71 
72  SetEntry(spellId);
73  SetObjectScale(DEFAULT_OBJECT_SCALE);
74 
75  SetCasterGuid(caster->GetObjectGuid());
76 
77  /* Bytes field, so it's really 4 bit fields. These flags are unknown, but we do know that 0x00000001 is set for most.
78  Farsight for example, does not have this flag, instead it has 0x80000002.
79  Flags are set dynamically with some conditions, so one spell may have different flags set, depending on those conditions.
80  The size of the visual may be controlled to some degree with these flags.
81 
82  uint32 bytes = 0x00000000;
83  bytes |= 0x01;
84  bytes |= 0x00 << 8;
85  bytes |= 0x00 << 16;
86  bytes |= 0x00 << 24;
87  */
88  SetByteValue(DYNAMICOBJECT_BYTES, 0, type);
89 
90  SetUInt32Value(DYNAMICOBJECT_SPELLID, spellId);
91  SetFloatValue(DYNAMICOBJECT_RADIUS, radius);
92  SetUInt32Value(DYNAMICOBJECT_CASTTIME, GetMap()->GetCurrentMSTime()); // new 2.4.0
93 
94  SpellEntry const* spellProto = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
95  if (!spellProto)
96  {
97  sLog.outError("DynamicObject (spell %u) not created. Spell not exist!", spellId);
98  return false;
99  }
100 
101  if (spellId == 44007) // Akilzon - Electrical storm - always takes 500 ms more than spell data - confirmed in sniff
102  duration += 500;
103 
104  m_aliveTime = GetMap()->GetCurrentClockTime() + std::chrono::milliseconds(duration);
105  m_radius = radius;
106  m_effIndex = effIndex;
107  m_spellId = spellId;
108  m_positive = IsPositiveEffect(spellProto, m_effIndex);
109  m_target = target;
110  m_damage = damage;
111  m_basePoints = basePoints;
112 
113  m_auraScript = SpellScriptMgr::GetAuraScript(m_spellId);
114 
115  return true;
116 }

124 void DynamicObject::Update(const uint32 /*diff*/)
125 {
126  // caster can be not in world at time dynamic object update, but dynamic object not yet deleted in Unit destructor
127  Unit* caster = GetCaster();

130  Delete();

132  }
133

145  }
146

149  OnPersistentAreaAuraEnd();

161 void DynamicObject::Delay(int32 delaytime)
162 {
163  m_aliveTime -= std::chrono::milliseconds(delaytime);
164  for (GuidSet::iterator iter = m_affected.begin(); iter != m_affected.end();)
165  {
166  Unit* target = GetMap()->GetUnit((*iter));
167  if (target)
168  {
169  SpellAuraHolder* holder = target->GetSpellAuraHolder(m_spellId, GetCasterGuid());
170  if (!holder)
171  {
172  m_affected.erase(iter++);
173  continue;
174  }
175 
176  bool foundAura = false;
177  for (int32 i = m_effIndex + 1; i < MAX_EFFECT_INDEX; ++i)
178  {
179  if ((holder->GetSpellProto()->Effect[i] == SPELL_EFFECT_PERSISTENT_AREA_AURA || holder->GetSpellProto()->Effect[i] == SPELL_EFFECT_ADD_FARSIGHT) && holder->m_auras[i])
180  {
181  foundAura = true;
182  break;
183  }
184  }
185 
186  if (!foundAura)
187  {
188  m_affected.erase(iter++);
189  continue;
190  }
191 
192  target->DelaySpellAuraHolder(m_spellId, delaytime, GetCasterGuid());
193  ++iter;
194  }
195  else
196  m_affected.erase(iter++);
197  }
198 }
