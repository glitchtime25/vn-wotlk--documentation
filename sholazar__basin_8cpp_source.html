404  void ReceiveAIEvent(AIEventType eventType, Unit* /*pSender*/, Unit* pInvoker, uint32 uiMiscValue) override
405  {
406  if (pInvoker->GetTypeId() != TYPEID_PLAYER)
407  return;
408 
409  // start event
410  if (eventType == AI_EVENT_START_EVENT)
411  {
412  m_creature->RemoveFlag(UNIT_NPC_FLAGS, UNIT_NPC_FLAG_GOSSIP);
413  DoScriptText(SAY_DISTILLATION_START, m_creature);
414  m_uiTaskTimer = 5000;
415  }
416  // check fruit tasks
417  else if (eventType == AI_EVENT_CUSTOM_A)
418  {
419  for (auto aStillAtItFruit : aStillAtItFruits)
420  {
421  if (aStillAtItFruit.uiOwnerEntry == uiMiscValue)
422  DoCheckDistillationTask(uiMiscValue);
423  }
424  }
425  // check machine tasks
426  else if (eventType == AI_EVENT_CUSTOM_B)
427  {
428  for (auto aStillAtItMachine : aStillAtItMachines)
429  {
430  if (aStillAtItMachine.uiOwnerEntry == uiMiscValue)
431  DoCheckDistillationTask(uiMiscValue);
432  }
433  }
434  }
435 
436  // wrapper to complete a distillation task
437  void DoCheckDistillationTask(uint32 uiOwnerEntry)
438  {
439  // check if the given entry matches the expected one
440  if (uiOwnerEntry == m_uiTaskOwnerEntry)
441  {
442  switch (urand(0, 3))
443  {
444  case 0: DoScriptText(SAY_ACTION_COMPLETE_1, m_creature); break;
445  case 1: DoScriptText(SAY_ACTION_COMPLETE_2, m_creature); break;
446  case 2: DoScriptText(SAY_ACTION_COMPLETE_3, m_creature); break;
447  case 3: DoScriptText(SAY_ACTION_COMPLETE_4, m_creature); break;
448  }
449 
450  m_uiTaskTimer = 6000;
451  m_uiActionTimer = 0;
452  }
453  // reset if failed
454  else
455  {
456  DoScriptText(SAY_DISTILLATION_FAIL, m_creature);
457  EnterEvadeMode();
458  }
459  }

461  void UpdateAI(const uint32 uiDiff) override
462  {
463  if (m_uiTaskTimer)
464  {
465  if (m_uiTaskTimer <= uiDiff)
466  {
467  switch (m_uiTaskIndex)
468  {
469  // fruit tasks
470  case 0:
471  case 1:
472  case 3:
473  case 6:
474  case 7:
475  {
476  uint8 uiIndex = urand(0, 2);
477  DoScriptText(aStillAtItFruits[uiIndex].iText, m_creature);
478  m_uiTaskOwnerEntry = aStillAtItFruits[uiIndex].uiOwnerEntry;
479 
480  m_uiTaskTimer = 0;
481  m_uiActionTimer = 5000;
482  break;
483  }
484  // valve or fire task
485  case 2:
486  case 4:
487  case 5:
488  case 8:
489  {
490  uint8 uiIndex = urand(0, 1);
491  DoScriptText(aStillAtItMachines[uiIndex].iText, m_creature);
492  m_uiTaskOwnerEntry = aStillAtItMachines[uiIndex].uiOwnerEntry;
493 
494  m_uiTaskTimer = 0;
495  m_uiActionTimer = 5000;
496  break;
497  }
498  // complete event
499  case 9:
500  DoScriptText(SAY_DISTILLATION_COMPLETE, m_creature);
501  if (GameObject* pPunch = GetClosestGameObjectWithEntry(m_creature, GO_THUNDERBREW_JUNGLE_PUNCH, 10.0f))
502  {
503  pPunch->SetRespawnTime(30);
504  pPunch->Refresh();
505  }
506  m_uiTaskTimer = 20000;
507  break;
508  case 10:
509  EnterEvadeMode();
510  m_uiTaskTimer = 0;
511  break;
512  }
513  ++m_uiTaskIndex;
514  }
515  else
516  m_uiTaskTimer -= uiDiff;
517  }
518 
519  // timer delay to allow player to complete the task
520  if (m_uiActionTimer)
521  {
522  if (m_uiActionTimer <= uiDiff)
523  {
524  DoScriptText(SAY_DISTILLATION_FAIL, m_creature);
525  EnterEvadeMode();
526  m_uiActionTimer = 0;
527  }
528  else
529  m_uiActionTimer -= uiDiff;
530  }
531 
532  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
533  return;
534 
535  DoMeleeAttackIfReady();
536  }
