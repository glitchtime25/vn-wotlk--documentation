72 // natural order for difficulties up-down iteration
73 // difficulties for dungeons/battleground ordered in normal way
74 // and if more high version not exist must be used lesser version
75 // for raid order different:
76 // 10 man normal version must be used instead nonexistent 10 man heroic version
77 // 25 man normal version must be used instead nonexistent 25 man heroic version
78 inline Difficulty GetPrevDifficulty(Difficulty diff, bool isRaid)
79 {
80  switch (diff)
81  {
82  default:
83  case RAID_DIFFICULTY_10MAN_NORMAL: // == DUNGEON_DIFFICULTY_NORMAL == REGULAR_DIFFICULTY
84  return REGULAR_DIFFICULTY; // return itself, caller code must properly check and not call for this case
85  case RAID_DIFFICULTY_25MAN_NORMAL: // == DUNGEON_DIFFICULTY_HEROIC
86  return RAID_DIFFICULTY_10MAN_NORMAL;
87  case RAID_DIFFICULTY_10MAN_HEROIC:
88  return isRaid ? RAID_DIFFICULTY_10MAN_NORMAL : DUNGEON_DIFFICULTY_HEROIC;
89  case RAID_DIFFICULTY_25MAN_HEROIC:
90  return isRaid ? RAID_DIFFICULTY_25MAN_NORMAL : RAID_DIFFICULTY_10MAN_HEROIC;
91  }
92 }
