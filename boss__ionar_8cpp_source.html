160  void DespawnSpark()
161  {
162  for (GuidList::const_iterator itr = m_lSparkGUIDList.begin(); itr != m_lSparkGUIDList.end(); ++itr)
163  {
164  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
165  {
166  if (pTemp->IsAlive())
167  pTemp->ForcedDespawn();
168  }
169  }
170 
171  m_lSparkGUIDList.clear();
172  }

174  // make sparks come back
175  void CallBackSparks()
176  {
177  for (GuidList::const_iterator itr = m_lSparkGUIDList.begin(); itr != m_lSparkGUIDList.end(); ++itr)
178  {
179  if (Creature* pSpark = m_creature->GetMap()->GetCreature(*itr))
180  {
181  if (pSpark->IsAlive())
182  {
183  // Required to prevent combat movement, elsewise they might switch movement on aggro-change
184  if (ScriptedAI* pSparkAI = dynamic_cast<ScriptedAI*>(pSpark->AI()))
185  pSparkAI->SetCombatMovement(false);
186 
187  pSpark->SetWalk(false);
188  pSpark->GetMotionMaster()->MovePoint(POINT_CALLBACK, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ());
189  }
190  }
191  }
192  }

212  void UpdateAI(const uint32 uiDiff) override
213  {
214  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
215  return;
216 
217  // Splitted
218  if (m_creature->GetVisibility() == VISIBILITY_OFF)
219  {
220  if (m_uiSplitTimer < uiDiff)
221  {
222  m_uiSplitTimer = 2500;
223 
224  // Return sparks to where Ionar splitted
225  if (m_bIsSplitPhase)
226  {
227  CallBackSparks();
228  m_bIsSplitPhase = false;
229  }
230  // Lightning effect and restore Ionar
231  else if (m_uiSparkAtHomeCount == MAX_SPARKS)
232  {
233  m_creature->SetVisibility(VISIBILITY_ON);
234  DoCastSpellIfCan(m_creature, SPELL_SPARK_DESPAWN);
235 
236  m_uiSparkAtHomeCount = 0;
237  m_uiSplitTimer = 25000;
238  m_bIsSplitPhase = true;
239  m_bIsDesperseCasting = false;
240 
241  if (m_creature->GetMotionMaster()->GetCurrentMovementGeneratorType() != CHASE_MOTION_TYPE)
242  {
243  if (m_creature->GetVictim())
244  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
245  }
246  }
247  }
248  else
249  m_uiSplitTimer -= uiDiff;
250 
251  return;
252  }
253 
254  if (m_uiStaticOverloadTimer < uiDiff)
255  {
256  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
257  {
258  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_STATIC_OVERLOAD_N : SPELL_STATIC_OVERLOAD_H) == CAST_OK)
259  m_uiStaticOverloadTimer = urand(5000, 6000);
260  }
261  }
262  else
263  m_uiStaticOverloadTimer -= uiDiff;
264 
265  if (m_uiBallLightningTimer < uiDiff)
266  {
267  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
268  {
269  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_BALL_LIGHTNING_N : SPELL_BALL_LIGHTNING_H) == CAST_OK)
270  m_uiBallLightningTimer = urand(10000, 11000);
271  }
272  }
273  else
274  m_uiBallLightningTimer -= uiDiff;
275 
276  // Health check
277  if (m_creature->GetHealthPercent() < float(100 - 20 * m_uiHealthAmountModifier))
278  {
279  ++m_uiHealthAmountModifier;
280 
281  if (!m_bIsDesperseCasting && DoCastSpellIfCan(m_creature, SPELL_DISPERSE, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
282  {
283  DoScriptText(urand(0, 1) ? SAY_SPLIT_1 : SAY_SPLIT_2, m_creature);
284  m_bIsDesperseCasting = true;
285  }
286  }
287 
288  DoMeleeAttackIfReady();
289  }

292 bool EffectDummyCreature_boss_ionar(Unit* /*pCaster*/, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
293 {
294  // always check spellid and effectindex
295  if (uiSpellId == SPELL_DISPERSE && uiEffIndex == EFFECT_INDEX_0)
296  {
297  if (pCreatureTarget->GetEntry() != NPC_IONAR)
298  return true;
299 
300  for (uint8 i = 0; i < MAX_SPARKS; ++i)
301  pCreatureTarget->CastSpell(pCreatureTarget, SPELL_SUMMON_SPARK, TRIGGERED_OLD_TRIGGERED);
302 
303  pCreatureTarget->AttackStop();
304  pCreatureTarget->SetVisibility(VISIBILITY_OFF);
305 
306  if (pCreatureTarget->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
307  pCreatureTarget->GetMotionMaster()->MovementExpired();
308 
309  // always return true when we are handling this spell and effect
310  return true;
311  }
312  if (uiSpellId == SPELL_SPARK_DESPAWN && uiEffIndex == EFFECT_INDEX_0)
313  {
314  if (pCreatureTarget->GetEntry() != NPC_IONAR)
315  return true;
316 
317  if (boss_ionarAI* pIonarAI = dynamic_cast<boss_ionarAI*>(pCreatureTarget->AI()))
318  pIonarAI->DespawnSpark();
319 
320  return true;
321  }
322  return false;
323 }
