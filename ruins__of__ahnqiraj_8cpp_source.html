122 void instance_ruins_of_ahnqiraj::OnCreatureDeath(Creature* pCreature)
123 {
124  switch (pCreature->GetEntry())
125  {
126  case NPC_KURINNAXX: SetData(TYPE_KURINNAXX, DONE); break;
127  case NPC_RAJAXX: SetData(TYPE_RAJAXX, DONE); break;
128  case NPC_MOAM: SetData(TYPE_MOAM, DONE); break;
129  case NPC_BURU: SetData(TYPE_BURU, DONE); break;
130  case NPC_AYAMISS: SetData(TYPE_AYAMISS, DONE); break;
131  case NPC_OSSIRIAN: SetData(TYPE_OSSIRIAN, DONE); break;
132  case NPC_COLONEL_ZERRAN:
133  case NPC_MAJOR_PAKKON:
134  case NPC_MAJOR_YEGGETH:
135  case NPC_CAPTAIN_XURREM:
136  case NPC_CAPTAIN_DRENN:
137  case NPC_CAPTAIN_TUUBID:
138  case NPC_CAPTAIN_QEEZ:
139  case NPC_QIRAJI_WARRIOR:
140  case NPC_SWARMGUARD_NEEDLER:
141  {
142  // If event isn't started by Andorov, return
143  if (GetData(TYPE_RAJAXX) != IN_PROGRESS)
144  return;
145 
146  // Check if the dead creature belongs to the current wave
147  if (m_sArmyWavesGuids[m_uiCurrentArmyWave - 1].find(pCreature->GetObjectGuid()) != m_sArmyWavesGuids[m_uiCurrentArmyWave - 1].end())
148  {
149  m_sArmyWavesGuids[m_uiCurrentArmyWave - 1].erase(pCreature->GetObjectGuid());
150 
151  // If all the soldiers from the current wave are dead, then send the next one
152  if (m_sArmyWavesGuids[m_uiCurrentArmyWave - 1].empty())
153  DoSendNextArmyWave();
154  }
155  break;
156  }
157  }

179 void instance_ruins_of_ahnqiraj::SetData(uint32 uiType, uint32 uiData)
180 {
181  switch (uiType)
182  {
183  case TYPE_KURINNAXX:
184  if (uiData == DONE)
185  {
186  DoSpawnAndorovIfCan();
187  // Yell after kurinnaxx
188  if (Creature* ossirian = GetSingleCreatureFromStorage(NPC_OSSIRIAN))
189  DoScriptText(SAY_OSSIRIAN_INTRO, ossirian); // targets kurinnaxx in sniff
190  }
191  m_auiEncounter[uiType] = uiData;
192  break;
193  case TYPE_RAJAXX:
194  m_auiEncounter[uiType] = uiData;
195  if (uiData == IN_PROGRESS)
196  DoSortArmyWaves();
197  if (uiData == DONE)
198  {
199  if (Creature* andorov = GetSingleCreatureFromStorage(NPC_GENERAL_ANDOROV))
200  if (andorov->IsAlive())
201  andorov->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, andorov, andorov);
202  }
203  break;
204  case TYPE_MOAM:
205  case TYPE_BURU:
206  case TYPE_AYAMISS:
207  case TYPE_OSSIRIAN:
208  if (uiType == FAIL)
209  {
210  GuidVector crystals;
211  GetGameObjectGuidVectorFromStorage(GO_OSSIRIAN_CRYSTAL, crystals);
212  for (uint32 i = 1; i < crystals.size(); ++i) // skip first
213  {
214  if (GameObject* go = instance->GetGameObject(crystals[i]))
215  {
216  go->SetForcedDespawn();
217  go->SetLootState(GO_JUST_DEACTIVATED);
218  }
219  }
220  }
221  m_auiEncounter[uiType] = uiData;
222  break;
223  }
224 
225  if (uiData == DONE)
226  {
227  OUT_SAVE_INST_DATA;
228 
229  std::ostringstream saveStream;
230  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2]
231  << " " << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5];
232 
233  m_strInstData = saveStream.str();
234 
235  SaveToDB();
236  OUT_SAVE_INST_DATA_COMPLETE;
237  }
238 }

302 void instance_ruins_of_ahnqiraj::DoSortArmyWaves()
303 {
304  CreatureList lCreatureList;
305 
306  // Sort the 7 army waves
307  // We need to use gridsearcher for this, because coords search is too complicated here
308  for (uint8 i = 0; i < MAX_ARMY_WAVES; ++i)
309  {
310  // Clear all the army waves
311  m_sArmyWavesGuids[i].clear();
312  lCreatureList.clear();
313 
314  if (Creature* pTemp = GetSingleCreatureFromStorage(aArmySortingParameters[i].m_uiEntry))
315  {
316  GetCreatureListWithEntryInGrid(lCreatureList, pTemp, NPC_QIRAJI_WARRIOR, aArmySortingParameters[i].m_fSearchDist);
317  GetCreatureListWithEntryInGrid(lCreatureList, pTemp, NPC_SWARMGUARD_NEEDLER, aArmySortingParameters[i].m_fSearchDist);
318 
319  for (CreatureList::const_iterator itr = lCreatureList.begin(); itr != lCreatureList.end(); ++itr)
320  {
321  if ((*itr)->IsAlive())
322  m_sArmyWavesGuids[i].insert((*itr)->GetObjectGuid());
323  }
324 
325  if (pTemp->IsAlive())
326  m_sArmyWavesGuids[i].insert(pTemp->GetObjectGuid());
327  }
328  }
329 
330  // send the first wave
331  m_uiCurrentArmyWave = 0;
332  DoSendNextArmyWave();
333 }

335 void instance_ruins_of_ahnqiraj::DoSendNextArmyWave()
336 {
337  // The next army wave is sent into battle after 2 min or after the previous wave is finished
338  if (GetData(TYPE_RAJAXX) != IN_PROGRESS)
339  return;
340 
341  // The last wave is General Rajaxx itself
342  if (m_uiCurrentArmyWave == MAX_ARMY_WAVES)
343  {
344  if (Creature* pRajaxx = GetSingleCreatureFromStorage(NPC_RAJAXX))
345  {
346  DoScriptText(SAY_INTRO, pRajaxx);
347  pRajaxx->SetInCombatWithZone();
348  pRajaxx->AI()->AttackClosestEnemy();
349  }
350  m_uiArmyDelayTimer = 0;
351  }
352  else
353  {
354  // Increase the wave id if some are already dead
355  while (m_uiCurrentArmyWave < MAX_ARMY_WAVES && m_sArmyWavesGuids[m_uiCurrentArmyWave].empty())
356  ++m_uiCurrentArmyWave;
357 
358  if (m_uiCurrentArmyWave == MAX_ARMY_WAVES)
359  {
360  script_error_log("Instance Ruins of Ahn'Qiraj: ERROR Something unexpected happened. Please report to SD2 team.");
361  return;
362  }
363 
364  float fX, fY, fZ;
365  for (auto itr : m_sArmyWavesGuids[m_uiCurrentArmyWave])
366  {
367  if (Creature* pTemp = instance->GetCreature(itr))
368  {
369  if (!pTemp->IsAlive())
370  continue;
371 
372  pTemp->SetWalk(false);
373  pTemp->GetRandomPoint(aAndorovMoveLocs[4].m_fX, aAndorovMoveLocs[4].m_fY, aAndorovMoveLocs[4].m_fZ, 10.0f, fX, fY, fZ);
374  pTemp->GetMotionMaster()->MovePoint(0, fX, fY, fZ);
375  }
376  }
377 
378  // Yell on each wave (except the first 2)
379  if (aArmySortingParameters[m_uiCurrentArmyWave].m_uiYellEntry)
380  {
381  if (Creature* pRajaxx = GetSingleCreatureFromStorage(NPC_RAJAXX))
382  DoScriptText(aArmySortingParameters[m_uiCurrentArmyWave].m_uiYellEntry, pRajaxx);
383  }
384  // on wowwiki it states that there were 3 min between the waves, but this was reduced in later patches
385  m_uiArmyDelayTimer = 2 * MINUTE * IN_MILLISECONDS;
386  }
387 
388  ++m_uiCurrentArmyWave;
389 }
