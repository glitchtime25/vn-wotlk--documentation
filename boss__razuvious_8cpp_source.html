86  void SpellHit(Unit* /*caster*/, const SpellEntry* spell) override
87  {
88  // Every time a Deathknight Understudy taunts Razuvious, he will yell his disappointment
89  if (spell->Id == SPELL_TAUNT)
90  {
91  switch (urand(0, 2))

93  case 0: DoBroadcastText(SAY_COMMAND1, m_creature); break;

96  }

112 bool NpcSpellClick_npc_obedienceCrystal(Player* player, Creature* clickedCreature, uint32 spellId)
113 {
114  if (clickedCreature->GetEntry() == NPC_OBEDIENCE_CRYSTAL)
115  {
116  if (clickedCreature->IsNonMeleeSpellCasted(true))
117  return false;
118 
119  CreatureList understudies;
120  bool castSuccess = false;
121  GetCreatureListWithEntryInGrid(understudies, clickedCreature, NPC_DEATHKNIGHT_UNDERSTUDY, 60.f);
122  for (auto understudy : understudies)
123  {
124  if (!castSuccess)
125  {
126  player->CastSpell(nullptr, spellId, TRIGGERED_OLD_TRIGGERED);
127  castSuccess = true;
128  }
129  if (understudy->GetCharmer() && understudy->GetCharmer()->GetObjectGuid() == player->GetObjectGuid())
130  clickedCreature->CastSpell(understudy, SPELL_OBEDIENCE_CHAINS, TRIGGERED_OLD_TRIGGERED);
131  }
132  clickedCreature->RemoveAllCooldowns();
133  return true;
134  }
135 
136  return true;
137 }
