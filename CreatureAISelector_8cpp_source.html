38  UnitAI* selectAI(Creature* creature)
39  {
40  // charmed creature may have some script even if its not supposed to be that way (ex: Eye of Acherus)
41  // Allow scripting AI for normal creatures and not controlled pets (guardians and mini-pets)
42  if ((!creature->IsPet() || !static_cast<Pet*>(creature)->isControlled()) || creature->HasCharmer())
43  if (UnitAI* scriptedAI = sScriptDevAIMgr.GetCreatureAI(creature))
44  return scriptedAI;
45 
46  CreatureAIRegistry& ai_registry(CreatureAIRepository::Instance());
47 
48  const CreatureAICreator* ai_factory = nullptr;
49 
50  std::string ainame = creature->GetAIName();
51 
52  // select by NPC flags
53  if (creature->IsPet())
54  {
55  if (creature->IsPlayerControlled() && static_cast<Pet*>(creature)->isControlled())
56  ai_factory = ai_registry.GetRegistryItem("PetAI");
57  else // For guardians and creature pets in general
58  ai_factory = ai_registry.GetRegistryItem("GuardianAI");
59  }
60  else if(creature->HasCharmer() && !creature->IsTemporarySummon()) // for charmed creatures but not possessed tempspawns
61  ai_factory = ai_registry.GetRegistryItem("PetAI");
62  else if (creature->IsTotem())
63  ai_factory = ai_registry.GetRegistryItem("TotemAI");
64  else if (!ainame.empty()) // select by script name
65  ai_factory = ai_registry.GetRegistryItem(ainame);
66  else if (creature->IsGuard())
67  ai_factory = ai_registry.GetRegistryItem("GuardAI");
68  else // select by permit check
69  {
70  int best_val = PERMIT_BASE_NO;
71  typedef CreatureAIRegistry::RegistryMapType RMT;
72  RMT const& l = ai_registry.GetRegisteredItems();
73  for (const auto& iter : l)
74  {
75  const CreatureAICreator* factory = iter.second;
76  const SelectableAI* p = dynamic_cast<const SelectableAI*>(factory);
77  MANGOS_ASSERT(p != nullptr);
78  int val = p->Permit(creature);
79  if (val > best_val)
80  {
81  best_val = val;
82  ai_factory = p;
83  }
84  }
85  }
86 
87  // select NullCreatureAI if not another cases
88  ainame = (ai_factory == nullptr) ? "NullCreatureAI" : ai_factory->key();
89 
90  DEBUG_FILTER_LOG(LOG_FILTER_AI_AND_MOVEGENSS, "Creature %u used AI is %s.", creature->GetGUIDLow(), ainame.c_str());
91  return (ai_factory == nullptr ? new NullCreatureAI(creature) : ai_factory->Create(creature));
92  }
