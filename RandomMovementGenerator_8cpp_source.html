25 void AbstractRandomMovementGenerator::Initialize(Unit& owner)
26 {
27  owner.addUnitState(i_stateActive);
28 
29  m_pathFinder = std::make_unique<PathFinder>(&owner);
30 
31  // Client-controlled unit should have control removed
32  if (const Player* controllingClientPlayer = owner.GetClientControlling())
33  controllingClientPlayer->UpdateClientControl(&owner, false);
34  // Non-client controlled unit with an AI should drop target
35  else if (owner.AI())
36  {
37  owner.SetTarget(nullptr);
38  owner.MeleeAttackStop(owner.GetVictim());
39  }
40 
41  // Stop any previously dispatched splines no matter the source
42  if (!owner.movespline->Finalized())
43  {
44  if (owner.IsClientControlled())
45  owner.StopMoving(true);
46  else
47  owner.InterruptMoving();
48  }
49 }

83 bool AbstractRandomMovementGenerator::Update(Unit& owner, const uint32& diff)
84 {
85  if (!owner.IsAlive())
86  return false;
87 
88  if (owner.hasUnitState(UNIT_STAT_NO_FREE_MOVE & ~i_stateActive))
89  {
90  i_nextMoveTimer.Update(diff);
91  owner.clearUnitState(i_stateMotion);
92  return true;
93  }
94 
95  if (owner.movespline->Finalized())
96  {
97  i_nextMoveTimer.Update(diff);
98 
99  if (i_nextMoveTimer.Passed())
100  {
101  if (_setLocation(owner))
102  {
103  if (i_nextMoveCount > 1)
104  --i_nextMoveCount;
105  else
106  {
107  i_nextMoveCount = urand(1, i_nextMoveCountMax);
108  i_nextMoveTimer.Reset(urand(i_nextMoveDelayMin, i_nextMoveDelayMax));
109  }
110  }
111  else
112  i_nextMoveTimer.Reset(owner.HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED) ? 100 : 500);
113  }
114  }
115 
116  return true;
117 }

119 int32 AbstractRandomMovementGenerator::_setLocation(Unit& owner)
120 {
121  // Look for a random location within certain radius of initial position
122  float x = i_x, y = i_y, z = i_z;
123 
124  if (i_pathLength != 0.0f)
125  m_pathFinder->setPathLengthLimit(i_pathLength);
126 
127  m_pathFinder->ComputePathToRandomPoint(Vector3(x, y, z), i_radius);
128 
129  if ((m_pathFinder->getPathType() & PATHFIND_NOPATH) != 0)
130  return 0;
131 
132  Movement::MoveSplineInit init(owner);
133  init.MovebyPath(m_pathFinder->getPath());
134  init.SetWalk(i_walk);
135 
136  if (owner.IsSlowedInCombat())
137  init.SetCombatSlowed(std::min(owner.GetHealthPercent(), 20.f) * 0.02 + 0.4f);
138 
139  int32 duration = init.Launch();
140 
141  if (duration)
142  owner.addUnitState(i_stateMotion);
143 
144  return duration;
145 }

216 #define MAX_QUIET_DISTANCE 43.0f

218 int32 FleeingMovementGenerator::_setLocation(Unit& owner)
219 {
220  float dist_from_source = owner.GetDistance(i_x, i_y, i_z);
221 
222  if (dist_from_source < MIN_QUIET_DISTANCE)
223  i_radius = frand(0.4f, 1.3f) * (MIN_QUIET_DISTANCE - dist_from_source);
224  else if (dist_from_source > MAX_QUIET_DISTANCE)
225  i_radius = frand(0.4f, 1.0f) * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE);
226  else // we are inside quiet range
227  i_radius = frand(0.6f, 1.2f) * (MAX_QUIET_DISTANCE - MIN_QUIET_DISTANCE);
228 
229  return AbstractRandomMovementGenerator::_setLocation(owner);
230 }
