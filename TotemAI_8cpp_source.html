58 void TotemAI::UpdateAI(const uint32 diff)
59 {
60  UpdateEventTimers(diff);
61 
62  if (getTotem().GetTotemType() != TOTEM_ACTIVE)
63  return;
64 
65  if (!m_creature->IsAlive() || m_creature->IsNonMeleeSpellCasted(false))
66  return;
67 
68  // Search spell
69  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(getTotem().GetSpell());
70  if (!spellInfo)
71  return;
72 
73  // shorthand CD check
74  if (!m_creature->IsSpellReady(*spellInfo))
75  return;
76 
77  // Get spell rangy
78  SpellRangeEntry const* srange = sSpellRangeStore.LookupEntry(spellInfo->rangeIndex);
79  float maxRange = GetSpellMaxRange(srange);
80 
81  // SPELLMOD_RANGE not applied in this place just because nonexistent range mods for attacking totems
82 
83  // pointer to appropriate target if found any
84  Unit* victim = m_creature->GetMap()->GetUnit(i_victimGuid);
85 
86  // Search victim if no, not attackable, or out of range, or friendly (possible in case duel end)
87  if (!victim ||
88  !m_creature->CanAttackOnSight(victim) || !m_creature->IsWithinDistInMap(victim, maxRange) ||
89  m_creature->CanAssist(victim) || !victim->IsVisibleForOrDetect(m_creature, m_creature, false))
90  {
91  victim = nullptr;
92 
93  if (maxRange != 0.0f)
94  {
95  MaNGOS::NearestAttackableUnitInObjectRangeCheck u_check(m_creature, m_creature->GetOwner(), maxRange);
96  MaNGOS::UnitLastSearcher<MaNGOS::NearestAttackableUnitInObjectRangeCheck> checker(victim, u_check);
97  Cell::VisitAllObjects(m_creature, checker, maxRange);
98  }
99  else
100  victim = m_creature;
101  }
102 
103  // If have target
104  if (victim)
105  {
106  // remember
107  i_victimGuid = victim->GetObjectGuid();
108 
109  // attack
110  m_creature->SetInFront(victim); // client change orientation by self
111  m_creature->CastSpell(victim, getTotem().GetSpell(), TRIGGERED_NONE);
112  }
113  else
114  i_victimGuid.Clear();
115 }
