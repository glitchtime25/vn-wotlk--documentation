86  void ExecuteAction(uint32 action) override
87  {
88  switch (action)
89  {
90  case HUHURAN_BERSERK_HP_CHECK:
91  if (m_creature->GetHealthPercent() > 30.0f)
92  return;
93  case HUHURAN_BERSERK:
94  {
95  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
96  {
97  DoScriptText(EMOTE_GENERIC_BERSERK, m_creature);
98  SetActionReadyStatus(HUHURAN_BERSERK_HP_CHECK, false);
99  DisableCombatAction(HUHURAN_BERSERK);
100  DisableCombatAction(HUHURAN_FRENZY);
101  }
102  break;
103  }
104  case HUHURAN_FRENZY:
105  {
106  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
107  {
108  DoScriptText(EMOTE_GENERIC_FRENZY_KILL, m_creature);
109  ResetCombatAction(action, urand(10, 20) * IN_MILLISECONDS);
110  }
111  break;
112  }
113  case HUHURAN_SPIT:
114  {
115  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ACID_SPIT) == CAST_OK)
116  ResetCombatAction(action, urand(5, 10) * IN_MILLISECONDS);
117  break;
118  }
119  case HUHURAN_NOXIOUS_POISON:
120  {
121  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_NOXIOUS_POISON, SELECT_FLAG_PLAYER))
122  if (DoCastSpellIfCan(target, SPELL_NOXIOUS_POISON) == CAST_OK)
123  ResetCombatAction(action, urand(12, 24) * IN_MILLISECONDS);
124  break;
125  }
126  case HUHURAN_WYVERN:
127  {
128  if (DoCastSpellIfCan(nullptr, SPELL_WYVERN_STING) == CAST_OK)
129  ResetCombatAction(action, urand(25, 35) * IN_MILLISECONDS);
130  break;
131  }
132  }
133  }
134 };
