55  const uint32 spell_list_cc[2] = { SPELL_FEAR, SPELL_KNOCKDOWN };
56  const uint32 spell_list_buff[3] = { SPELL_ENRAGE, SPELL_FRENZY, SPELL_THRASH };

58  void JustRespawned() override
59  {
60  // Set random time based CC ability on spawn (Fear or Knockdown)
61  m_uiRandomCcAbility = spell_list_cc[urand(0, 1)];
62 
63  // Set a random buff HP based ability (Frenzy or Enrage) or Thrash on spawn
64  m_uiRandomBuffAbility = spell_list_buff[urand(0, 2)];
65 
66  // Cast Thrash on spawn, since a passive ability
67  if (m_uiRandomBuffAbility == SPELL_THRASH)
68  DoCastSpellIfCan(m_creature, m_uiRandomBuffAbility, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
69  }
70 
71  void Reset() override
72  {
73  m_uiRandomCcAbilityTimer = urand(2000, 5000);
74  m_uiSoulTapTimer = urand(m_uiRandomCcAbilityTimer, 7000);
75  m_uiLightingBreathTimer = urand(m_uiSoulTapTimer, 9000);
76 
77  if (m_uiRandomBuffAbility == SPELL_THRASH)
78  DoCastSpellIfCan(m_creature, m_uiRandomBuffAbility, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
79  }

81  void UpdateAI(const uint32 uiDiff) override
82  {
83  // Check if we have a valid target, otherwise do nothing
84  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
85  return;
86 
87  // Check if we are casting / channelling, return to not interrupt process and pause CDs
88  if (m_creature->IsNonMeleeSpellCasted(false))
89  return;
90 
91  // Soul Tap
92  if (m_uiSoulTapTimer < uiDiff)
93  {
94  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SOUL_TAP, SELECT_TARGET_ANY_ENEMY))
95  if (DoCastSpellIfCan(pTarget, SPELL_SOUL_TAP) == CAST_OK)
96  m_uiSoulTapTimer = urand(10000, 15000);
97  }
98  else
99  m_uiSoulTapTimer -= uiDiff;
100 
101  // Lightning Breath
102  if (m_uiLightingBreathTimer < uiDiff)
103  {
104  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_LIGHTNING_BREATH) == CAST_OK)
105  m_uiLightingBreathTimer = urand(10000, 15000);
106  }
107  else
108  m_uiLightingBreathTimer -= uiDiff;
109 
110  // Random time based CC ability
111  // Either Fear or Knockdown
112  if (m_uiRandomCcAbilityTimer < uiDiff)
113  {
114  if (DoCastSpellIfCan(m_creature->GetVictim(), m_uiRandomCcAbility) == CAST_OK)
115  m_uiRandomCcAbilityTimer = urand(8000, 10000);
116  }
117  else
118  m_uiRandomCcAbilityTimer -= uiDiff;
119 
120  // Random HP based ability
121  // Either Frenzy or Enrage
122  if (m_uiRandomBuffAbility != SPELL_THRASH)
123  if (m_creature->GetHealthPercent() <= 30.0f)
124  if (!m_creature->HasAura(m_uiRandomBuffAbility))
125  DoCastSpellIfCan(m_creature->GetVictim(), m_uiRandomBuffAbility);
126 
127  DoMeleeAttackIfReady();
128  }
129 };
