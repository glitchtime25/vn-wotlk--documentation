92  void ExecuteAction(uint32 action) override
93  {
94  switch (action)
95  {
96  case SARTURA_ENRAGE:
97  {
98  if (m_creature->GetHealthPercent() <= 25.0f)
99  {
100  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
101  {
102  DoScriptText(EMOTE_FRENZY, m_creature);
103  SetActionReadyStatus(action, false);
104  }
105  }
106  break;
107  }
108  case SARTURA_WHIRLWIND:
109  {
110  if (DoCastSpellIfCan(nullptr, SPELL_WHIRLWIND) == CAST_OK)
111  ResetCombatAction(action, urand(20, 25) * IN_MILLISECONDS);
112  break;
113  }
114  case SARTURA_SUNDERING_CLEAVE:
115  {
116  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_SUNDERING_CLEAVE) == CAST_OK)
117  ResetCombatAction(action, urand(2, 5) * IN_MILLISECONDS);
118  break;
119  }
120  case SARTURA_BERSERK:
121  {
122  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
123  {
124  DoScriptText(EMOTE_BERSERK, m_creature);
125  DisableCombatAction(action);
126  }
127  break;
128  }
129  }
130  }

133 struct AQWhirlwind : public SpellScript
134 {
135  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
136  {
137  if (effIdx != EFFECT_INDEX_1)
138  return;
139 
140  if (!spell->GetCaster()->CanHaveThreatList() || spell->GetCaster()->getThreatManager().isThreatListEmpty())
141  return;
142 
143  if (!spell->GetTriggeredByAuraSpellInfo())
144  return;
145 
146  SpellAuraHolder* holder = spell->GetCaster()->GetSpellAuraHolder(spell->GetTriggeredByAuraSpellInfo()->Id);
147  if (holder->m_auras[EFFECT_INDEX_0]->GetAuraTicks() != holder->m_auras[EFFECT_INDEX_0]->GetAuraMaxTicks())
148  {
149  if (spell->m_spellInfo->Id == SPELL_OTHER_WHIRLWIND_TRIGGER)
150  if (urand(0, 2))
151  return;
152 
153  spell->GetCaster()->getThreatManager().modifyAllThreatPercent(-100);
154  if (Unit* target = static_cast<Creature*>(spell->GetCaster())->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
155  spell->GetCaster()->getThreatManager().addThreat(target, 100000.f);
156  }
157  else
158  spell->GetCaster()->getThreatManager().modifyAllThreatPercent(-100);
159  }
160 };
