84 void instance_onyxias_lair::SetData(uint32 uiType, uint32 uiData)
85 {
86  if (uiType != TYPE_ONYXIA)
87  return;
88 
89  m_uiEncounter = uiData;
90 
91  switch (uiData)
92  {
93  case DATA_LIFTOFF:
94  m_tPhaseTwoStart = time(nullptr);
95  break;
96  case IN_PROGRESS:
97  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_ONYXIA_ID);
98  m_uiAchievWhelpsCount = 0;
99  // Respawn dead Onyxian Warders
100  for (GuidList::const_iterator itr = m_lWarderGUIDList.begin(); itr != m_lWarderGUIDList.end(); ++itr)
101  {
102  if (Creature* pWarder = instance->GetCreature(*itr))
103  {
104  if (!pWarder->IsAlive())
105  pWarder->Respawn();
106  }
107  }
108  break;
109  case FAIL:
110  case DONE:
111  // Despawn Onyxian Warders that were respawned during fight
112  for (GuidList::const_iterator itr = m_lWarderGUIDList.begin(); itr != m_lWarderGUIDList.end(); ++itr)
113  {
114  if (Creature* pWarder = instance->GetCreature(*itr))
115  pWarder->ForcedDespawn();
116  }
117  break;
118  }
119 
120  // Currently no reason to save anything
121 }
