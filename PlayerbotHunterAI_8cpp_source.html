100 CombatManeuverReturns PlayerbotHunterAI::DoFirstCombatManeuver(Unit* pTarget)
101 {
102  m_has_ammo = m_bot.HasItemCount(m_bot.GetUInt32Value(PLAYER_AMMO_ID), 1);
103  //DEBUG_LOG("current ammo (%u)",m_bot.GetUInt32Value(PLAYER_AMMO_ID));
104  m_bot.setAttackTimer(RANGED_ATTACK, 0);
105  if (!m_has_ammo)
106  {
107  m_ai.FindAmmo();
108  //DEBUG_LOG("new ammo (%u)",m_bot.GetUInt32Value(PLAYER_AMMO_ID));
109  m_has_ammo = m_bot.HasItemCount(m_bot.GetUInt32Value(PLAYER_AMMO_ID), 1);
110  }
111  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
112  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
113  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
114  {
115  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
116  {
117  return RETURN_NO_ACTION_OK; // wait it out
118  }
119  else
120  {
121  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
122  }
123  }
124 
125  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
126  {
127  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
128  return RETURN_NO_ACTION_OK; // wait it out
129  else
130  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
131  }
132 
133  switch (m_ai.GetScenarioType())
134  {
135  case PlayerbotAI::SCENARIO_PVP_DUEL:
136  case PlayerbotAI::SCENARIO_PVP_BG:
137  case PlayerbotAI::SCENARIO_PVP_ARENA:
138  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
139  return DoFirstCombatManeuverPVP(pTarget);
140  case PlayerbotAI::SCENARIO_PVE:
141  case PlayerbotAI::SCENARIO_PVE_ELITE:
142  case PlayerbotAI::SCENARIO_PVE_RAID:
143  default:
144  return DoFirstCombatManeuverPVE(pTarget);
145  }
146 }

175 CombatManeuverReturns PlayerbotHunterAI::DoNextCombatManeuverPVE(Unit* pTarget)
176 {
177  if (!pTarget) return RETURN_NO_ACTION_ERROR;
178 
179  Unit* pVictim = pTarget->GetVictim();
180 
181  // check for pet and heal if neccessary
182  Pet* pet = m_bot.GetPet();
183  // TODO: clarify/simplify: !pet->GetDeathState() != ALIVE
184  if (pet && PET_MEND > 0 && pet->IsAlive() && pet->GetHealthPercent() < 50 && pVictim != &m_bot && !pet->HasAura(PET_MEND, EFFECT_INDEX_0) && m_ai.CastSpell(PET_MEND, m_bot) == SPELL_CAST_OK)
185  {
186  m_ai.TellMaster("healing pet.");
187  return RETURN_CONTINUE;
188  }
189  else if (pet && INTIMIDATION > 0 && pVictim == pet && !pet->HasAura(INTIMIDATION, EFFECT_INDEX_0) && m_ai.CastSpell(INTIMIDATION, m_bot) == SPELL_CAST_OK)
190  return RETURN_CONTINUE;
191 
192  // racial traits
193  if (m_bot.getRace() == RACE_ORC && !m_bot.HasAura(BLOOD_FURY, EFFECT_INDEX_0))
194  m_ai.CastSpell(BLOOD_FURY, m_bot);
195  else if (m_bot.getRace() == RACE_TROLL && !m_bot.HasAura(BERSERKING, EFFECT_INDEX_0))
196  m_ai.CastSpell(BERSERKING, m_bot);
197 
198  // check if ranged combat is possible
199  bool meleeReach = m_bot.CanReachWithMeleeAttack(pTarget);
200 
201  if (meleeReach || !m_has_ammo)
202  {
203  // switch to melee combat (target in melee range, out of ammo)
204  m_rangedCombat = false;
205  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_MELEE);
206  if (!m_bot.GetUInt32Value(PLAYER_AMMO_ID))
207  m_ai.TellMaster("Out of ammo!");
208 
209  // become monkey (increases dodge chance)...
210  if (ASPECT_OF_THE_MONKEY > 0 && !m_bot.HasAura(ASPECT_OF_THE_MONKEY, EFFECT_INDEX_0))
211  m_ai.CastSpell(ASPECT_OF_THE_MONKEY, m_bot);
212 
213  }
214  else if (!meleeReach)
215  {
216  // switch to ranged combat
217  m_rangedCombat = true;
218  m_ai.SetCombatStyle(PlayerbotAI::COMBAT_RANGED);
219 
220  // increase ranged attack power...
221  if (ASPECT_OF_THE_HAWK > 0 && !m_bot.HasAura(ASPECT_OF_THE_HAWK, EFFECT_INDEX_0))
222  m_ai.CastSpell(ASPECT_OF_THE_HAWK, m_bot);
223 
224  // m_ai.TellMaster("target dist %f",m_bot.GetCombatDistance(pTarget,true));
225  if (AUTO_SHOT > 0)
226  {
227  if (m_bot.isAttackReady(RANGED_ATTACK))
228  m_bot.CastSpell(pTarget, AUTO_SHOT, TRIGGERED_OLD_TRIGGERED);
229 
230  m_bot.setAttackTimer(RANGED_ATTACK, 500);
231 
232  const SpellEntry* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(AUTO_SHOT);
233  if (!spellInfo)
234  return RETURN_CONTINUE;
235 
236  if (!(m_ai.CheckBotCast(spellInfo) == SPELL_CAST_OK))
237  m_bot.InterruptNonMeleeSpells(true, AUTO_SHOT);
238  }
239  }
240 
241  // damage spells
242  if (m_ai.GetCombatStyle() == PlayerbotAI::COMBAT_RANGED)
243  {
244  if (HUNTERS_MARK > 0 && m_ai.In_Reach(pTarget, HUNTERS_MARK) && !pTarget->HasAura(HUNTERS_MARK, EFFECT_INDEX_0) && m_ai.CastSpell(HUNTERS_MARK, *pTarget) == SPELL_CAST_OK)
245  return RETURN_CONTINUE;
246  else if (RAPID_FIRE > 0 && m_ai.In_Reach(pTarget, RAPID_FIRE) && !m_bot.HasAura(RAPID_FIRE, EFFECT_INDEX_0) && m_ai.CastSpell(RAPID_FIRE, m_bot) == SPELL_CAST_OK)
247  return RETURN_CONTINUE;
248  else if (MULTI_SHOT > 0 && m_ai.In_Reach(pTarget, MULTI_SHOT) && m_ai.GetAttackerCount() >= 3 && m_ai.CastSpell(MULTI_SHOT, *pTarget) == SPELL_CAST_OK)
249  return RETURN_CONTINUE;
250  else if (ARCANE_SHOT > 0 && m_ai.In_Reach(pTarget, ARCANE_SHOT) && m_ai.CastSpell(ARCANE_SHOT, *pTarget) == SPELL_CAST_OK)

326 bool PlayerbotHunterAI::IsTargetEnraged(Unit* pTarget)
327 {
328  if (!pTarget) return false;
329 
330  Unit::SpellAuraHolderMap const& auras = pTarget->GetSpellAuraHolderMap();
331  for (auto itr = auras.begin(); itr != auras.end(); ++itr)
332  {
333  SpellAuraHolder* holder = itr->second;
334  // Return true is target unit has aura with DISPEL_ENRAGE dispel type
335  if ((1 << holder->GetSpellProto()->Dispel) & GetDispellMask(DISPEL_ENRAGE))
336  return true;
