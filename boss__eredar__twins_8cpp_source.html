470  // Return a random target which it's not in range of 10 yards of boss
471  Unit* GetRandomTargetAtDist(float fDist) const
472  {
473  std::vector<Unit*> m_vRangeTargets;
474 
475  ThreatList const& tList = m_creature->getThreatManager().getThreatList();
476  for (auto iter : tList)
477  {
478  if (Unit* threatTarget = m_creature->GetMap()->GetUnit(iter->getUnitGuid()))
479  {
480  if (!threatTarget->IsPlayer())
481  continue;
482 
483  if (!threatTarget->IsWithinDistInMap(m_creature, fDist))
484  m_vRangeTargets.push_back(threatTarget);
485  }
486  }
487 
488  if (!m_vRangeTargets.empty())
489  return m_vRangeTargets[urand(0, m_vRangeTargets.size() - 1)];
490  return m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
491  }
