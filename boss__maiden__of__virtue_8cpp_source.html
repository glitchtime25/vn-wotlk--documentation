83  void ExecuteAction(uint32 action) override
84  {
85  switch (action)
86  {
87  case MAIDEN_ACTION_REPENTANCE:
88  {
89  if (DoCastSpellIfCan(nullptr, SPELL_REPENTANCE) == CAST_OK)
90  {
91  DoScriptText(urand(0, 1) ? SAY_REPENTANCE1 : SAY_REPENTANCE2, m_creature);
92  ResetCombatAction(action, GetSubsequentActionTimer(action));
93  }
94  return;
95  }
96  case MAIDEN_ACTION_HOLY_FIRE:
97  {
98  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_HOLYFIRE, SELECT_FLAG_NOT_IN_MELEE_RANGE | SELECT_FLAG_PLAYER))
99  if (DoCastSpellIfCan(target, SPELL_HOLYFIRE) == CAST_OK)
100  ResetCombatAction(action, GetSubsequentActionTimer(action));
101  return;
102  }
103  case MAIDEN_ACTION_HOLY_WRATH:
104  {
105  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_HOLYWRATH, SELECT_FLAG_PLAYER))
106  if (DoCastSpellIfCan(target, SPELL_HOLYWRATH) == CAST_OK)
107  ResetCombatAction(action, GetSubsequentActionTimer(action));
108  return;
109  }
110  case MAIDEN_ACTION_HOLY_GROUND:
111  {
112  if (DoCastSpellIfCan(nullptr, SPELL_HOLYGROUND) == CAST_OK)
113  ResetCombatAction(action, GetSubsequentActionTimer(action));
114  return;
115  }
116  }
117  }
