103  void UpdateAI(const uint32 uiDiff) override
104  {
105  // Return since we have no target
106  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
107  return;
108 
109  // Sand Breath
110  if (m_uiSandBreathTimer < uiDiff)
111  {
112  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SAND_BREATH : SPELL_SAND_BREATH_H) == CAST_OK)
113  m_uiSandBreathTimer = urand(15000, 25000);
114  }
115  else
116  m_uiSandBreathTimer -= uiDiff;
117 
118  // Time Stop
119  if (m_uiTimeStopTimer < uiDiff)
120  {
121  if (DoCastSpellIfCan(m_creature, SPELL_TIME_STOP) == CAST_OK)
122  m_uiTimeStopTimer = urand(20000, 35000);
123  }
124  else
125  m_uiTimeStopTimer -= uiDiff;
126 
127  // Cleave
128  if (m_uiCleaveTimer < uiDiff)
129  {
130  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
131  m_uiCleaveTimer = urand(7000, 12000);
132  }
133  else
134  m_uiCleaveTimer -= uiDiff;
135 
136  // Frenzy
137  if (m_uiFrenzyTimer < uiDiff)
138  {
139  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
140  {
141  DoScriptText(EMOTE_GENERIC_FRENZY, m_creature);
142  m_uiFrenzyTimer = urand(20000, 35000);
143  }
144  }
145  else
146  m_uiFrenzyTimer -= uiDiff;
147 
148  DoMeleeAttackIfReady();
149  }
