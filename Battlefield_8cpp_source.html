65

68  if (itr == m_activePlayers.end())
69  InitPlayerBattlefieldData(player);
70  else
71  {
72  if (GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS && itr->second->removeTime && m_startTime > itr->second->removeTime)

74 
75  itr->second->removeTime = 0;
76  }

205 void Battlefield::StartBattle(Team defender)
206 {
207  DEBUG_LOG("Battlefield: Battle has started. Defender Team: %s.", defender == ALLIANCE ? "Alliance" : "Horde");
208 
209  m_startTime = time(nullptr);
210  m_status = BF_STATUS_IN_PROGRESS;
211  m_timer = m_battleDuration;
212  m_zoneOwner = defender;
213 
214  // refresh all players before battle
215  InitPlayersBeforeBattle();
216 
217  // disband existing raid groups
218  for (auto& m_battlefieldRaid : m_battlefieldRaids)
219  {
220  for (auto group : m_battlefieldRaid)
221  {
222  group->Disband();
223  delete group;
224  }
225  m_battlefieldRaid.clear();
226  }
227 
228  for (auto& m_zonePlayer : m_zonePlayers)
229  {
230  if (!m_zonePlayer.first)
231  continue;
232 
233  Player* player = sObjectMgr.GetPlayer(m_zonePlayer.first);
234  if (!player)
235  continue;
236 
237  uint8 idx = 0;
238  std::set<ObjectGuid>::iterator itr2 = m_queuedPlayers[idx].find(m_zonePlayer.first);
239  if (itr2 == m_queuedPlayers[idx].end())
240  {
241  ++idx;
242  itr2 = m_queuedPlayers[idx].find(m_zonePlayer.first);
243  }
244 
245  // send battle invite to players added to queue
246  if (itr2 != m_queuedPlayers[idx].end())
247  {
248  m_invitedPlayers[idx][m_zonePlayer.first] = time(nullptr) + BF_TIME_TO_ACCEPT;
249  player->GetSession()->SendBattlefieldWarInvite(m_battleFieldId, m_zoneId, BF_TIME_TO_ACCEPT);
250  m_queuedPlayers[idx].erase(itr2);
251  }
252  // kick players not in queue
253  else
254  {
255  player->GetSession()->SendBattlefieldLeaveMessage(m_battleFieldId, BATTLEFIELD_LEAVE_REASON_EXITED);
256  SendRemoveWorldStates(player);
257  if (m_activePlayers.find(m_zonePlayer.first) != m_activePlayers.end())
258  {
259  m_activePlayers[m_zonePlayer.first]->removeTime = time(nullptr);
260  m_activePlayers[m_zonePlayer.first]->removeDelay = BF_UNACCEPTED_REMOVE_DELAY;
261  }
262  }
263  }
264 
265  // non-zone queued players
266  for (uint32 i = 0; i < PVP_TEAM_COUNT; ++i)
267  {
268  for (std::set<ObjectGuid>::iterator itr = m_queuedPlayers[i].begin(); itr != m_queuedPlayers[i].end();)
269  {
270  if (Player* plr = sObjectMgr.GetPlayer(*itr, true))
271  {
272  m_invitedPlayers[i][plr->GetObjectGuid()] = time(nullptr) + BF_TIME_TO_ACCEPT;
273  plr->GetSession()->SendBattlefieldWarInvite(m_battleFieldId, m_zoneId, BF_TIME_TO_ACCEPT);
274  }
275 
276  m_queuedPlayers[i].erase(itr++);
277  }
278  }
279 }

322 void Battlefield::Update(uint32 diff)
323 {
324  // update battlefield timer to everyone
325  SendBattlefieldTimerUpdate();
326 
327  // global battlefield timer
328  if (m_timer < diff)
329  {
330  if (m_status == BF_STATUS_COOLDOWN)
331  StartBattle(m_zoneOwner);
332  else if (m_status == BF_STATUS_IN_PROGRESS)
333  EndBattle(m_zoneOwner);
334  }
335  else
336  m_timer -= diff;
337 
338  // refresh battlefield players
339  if (GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS)
340  {
341  if (m_playersUpdateTimer < diff)
342  {
343  UpdateBattlefieldPlayers();
344  m_playersUpdateTimer = 5000;
345  }
346  else
347  m_playersUpdateTimer -= diff;
348  }
349 
350  // invite all players in zone to queue
351  if (GetBattlefieldStatus() == BF_STATUS_COOLDOWN && m_timer <= m_startInviteDelay && !m_playersInvited)
352  {
353  m_playersInvited = true;
354  for (auto& m_zonePlayer : m_zonePlayers)
355  {
356  if (!m_zonePlayer.first)
357  continue;
358 
359  Player* player = sObjectMgr.GetPlayer(m_zonePlayer.first);
360  if (!player)
361  continue;
362 
363  InvitePlayerToQueue(player);
364  }
365  }
366 
367  // kick all the players marked to leave
368  if (GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS)
369  {
370  for (auto& m_invitedPlayer : m_invitedPlayers)
371  {
372  for (std::map<ObjectGuid, time_t>::iterator itr = m_invitedPlayer.begin(); itr != m_invitedPlayer.end();)
373  {
374  if (itr->second < time(nullptr))
375  {
376  if (Player* player = sObjectMgr.GetPlayer(itr->first))
377  player->GetSession()->SendBattlefieldLeaveMessage(m_battleFieldId, BATTLEFIELD_LEAVE_REASON_EXITED);
378 
379  m_invitedPlayer.erase(itr++);
380  }
381  else
382  ++itr;
383  }
384  }
385  }
386 
387  // check the queued players
388  if (m_queueUpdateTimer < diff)
389  {
390  m_queueUpdateTimer = 30000;
391 
392  for (auto& m_queuedPlayer : m_queuedPlayers)
393  {
394  for (std::set<ObjectGuid>::iterator itr = m_queuedPlayer.begin(); itr != m_queuedPlayer.end();)
395  {
396  if (!sObjectMgr.GetPlayer(*itr, true))
397  m_queuedPlayer.erase(itr++);
398  else
399  ++itr;
400  }
401  }
402  }
403  else
404  m_queueUpdateTimer -= diff;
405 }

477 bool Battlefield::CanAddPlayerToRaid(Player* player)
478 {
479  if (!player->IsInWorld())
480  return false;
481 
482  DEBUG_LOG("Battlefield: Adding player %s to raid", player->GetGuidStr().c_str());
483 
484  if (Group* group = player->GetGroup())
485  {
486  DEBUG_LOG("Battlefield: Player %s already has group %s, uninviting", player->GetGuidStr().c_str(), group->GetGuidStr().c_str());
487  group->RemoveMember(player->GetObjectGuid(), 0);
488  }
489 
490  PvpTeamIndex teamIdx = GetTeamIndexByTeamId(player->GetTeam());
491  Group* group = GetFreeRaid(teamIdx);
492  if (!group)
493  {
494  DEBUG_LOG("Battlefield: No free raid for %s!", player->GetGuidStr().c_str());
495  if (IsTeamFull(teamIdx))
496  {
497  DEBUG_LOG("Battlefield: Battlefield is full! Can't add player %s!", player->GetGuidStr().c_str());
498  return false;
499  }
500 
501  DEBUG_LOG("Battlefield: Trying to create new group for %s!", player->GetGuidStr().c_str());
502  group = new Group;
503  group->SetBattlefieldGroup(this);
504  if (group->Create(player->GetObjectGuid(), player->GetName()))
505  DEBUG_LOG("Battlefield: Successfully created new group %s", group->GetGuidStr().c_str());
506  else
507  sLog.outError("Battlefield: Failed to create group for player %s.", player->GetGuidStr().c_str());
508 
509  m_battlefieldRaids[teamIdx].insert(group);
510  }
511  else if (group->IsMember(player->GetObjectGuid()))
512  {
513  DEBUG_LOG("Battlefield: Raid already has players %s, making some shit", player->GetGuidStr().c_str());
514  uint8 subgroup = group->GetMemberGroup(player->GetObjectGuid());
515  player->SetBattleRaid(group, subgroup);
516  }
517  else
518  {
519  if (IsTeamFull(teamIdx))
520  {
521  DEBUG_LOG("Battlefield: Group %s found, but battlefield is full! Can't add player %s!", group->GetGuidStr().c_str(), player->GetGuidStr().c_str());
522  return false;
523  }
524 
525  return group->AddMember(player->GetObjectGuid(), player->GetName());
526  }
527 
528  return true;
529 }
