102  bool VMapManager2::_loadMap(unsigned int pMapId, const std::string& basePath, uint32 tileX, uint32 tileY)
103  {
104  InstanceTreeMap::iterator instanceTree = iInstanceMapTrees.find(pMapId);

107  std::string mapFileName = getMapFileName(pMapId);

118  instanceTree = iInstanceMapTrees.insert(InstanceTreeMap::value_type(pMapId, newTree)).first;
119  }
120  }
121  return instanceTree->second->LoadMapTile(tileX, tileY, this);

158  bool VMapManager2::isInLineOfSight(unsigned int pMapId, float x1, float y1, float z1, float x2, float y2, float z2, bool ignoreM2Model)
159  {
160  if (!isLineOfSightCalcEnabled()) return true;

163  if (instanceTree != iInstanceMapTrees.end())
164  {
165  Vector3 pos1 = convertPositionToInternalRep(x1, y1, z1);
166  Vector3 pos2 = convertPositionToInternalRep(x2, y2, z2);
167  if (pos1 != pos2)
168  {
169  result = instanceTree->second->isInLineOfSight(pos1, pos2, ignoreM2Model);
170  }
171  }
172  return result;
173  }

179  bool VMapManager2::getObjectHitPos(unsigned int pMapId, float x1, float y1, float z1, float x2, float y2, float z2, float& rx, float& ry, float& rz, float pModifyDist)
180  {
181  bool result = false;
182  rx = x2;
183  ry = y2;
184  rz = z2;
185  if (isLineOfSightCalcEnabled())
186  {
187  InstanceTreeMap::iterator instanceTree = iInstanceMapTrees.find(pMapId);
188  if (instanceTree != iInstanceMapTrees.end())
189  {
190  Vector3 pos1 = convertPositionToInternalRep(x1, y1, z1);
191  Vector3 pos2 = convertPositionToInternalRep(x2, y2, z2);
192  Vector3 resultPos;
193  result = instanceTree->second->getObjectHitPos(pos1, pos2, resultPos, pModifyDist);
194  resultPos = convertPositionToInternalRep(resultPos.x, resultPos.y, resultPos.z);
195  rx = resultPos.x;
196  ry = resultPos.y;
197  rz = resultPos.z;
198  }
199  }
200  return result;
201  }

266  {
267  std::lock_guard<std::mutex> lock(m_vmModelMutex);
268  ModelFileMap::iterator model = iLoadedModelFiles.find(filename);

275  delete worldmodel;
276  return nullptr;
277  }
278 
279  // insert new data
280  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "VMapManager2: loading file '%s%s'.", basepath.c_str(), filename.c_str());
281  model = iLoadedModelFiles.insert(std::pair<std::string, ManagedModel>(filename, ManagedModel())).first;
282  model->second.setModel(worldmodel);
283  }
284  model->second.incRefCount();
