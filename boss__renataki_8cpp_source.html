91 struct ThousandBladesRenataki : public SpellScript
92 {
93  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
94  {
95  if (effIdx != EFFECT_INDEX_1)
96  return;
97 
98  std::vector<Unit*> selectedTargets;
99  Unit* caster = spell->GetCaster();
100  caster->SelectAttackingTargets(selectedTargets, ATTACKING_TARGET_ALL_SUITABLE, 0, SPELL_THOUSAND_BLADES, SELECT_FLAG_PLAYER);
101  // remove current target
102  selectedTargets.erase(std::remove(selectedTargets.begin(), selectedTargets.end(), spell->m_targets.getUnitTarget()), selectedTargets.end());
103  std::shuffle(selectedTargets.begin(), selectedTargets.end(), *GetRandomGenerator());
104  selectedTargets.resize(9);
105  for (Unit* target : selectedTargets)
106  caster->CastSpell(target, SPELL_THOUSAND_BLADES, TRIGGERED_OLD_TRIGGERED);
107  }
