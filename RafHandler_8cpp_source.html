35 
36  // check cheating

39  if (!target)
40  error = ERR_REFER_A_FRIEND_NO_TARGET;
41  else if (levels == 0)
42  error = ERR_REFER_A_FRIEND_INSUFFICIENT_GRANTABLE_LEVELS;
43  else if (IsARecruiter() || GetRecruitingFriendId() != target->GetSession()->GetAccountId())
44  error = ERR_REFER_A_FRIEND_NOT_REFERRED_BY;
45  else if (target->GetTeam() != _player->GetTeam())
46  error = ERR_REFER_A_FRIEND_DIFFERENT_FACTION;
47  else if (target->GetLevel() >= _player->GetLevel())
48  error = ERR_REFER_A_FRIEND_TARGET_TOO_HIGH;
49  else if (target->GetLevel() >= sWorld.getConfig(CONFIG_UINT32_MAX_RECRUIT_A_FRIEND_BONUS_PLAYER_LEVEL))
50  error = ERR_REFER_A_FRIEND_GRANT_LEVEL_MAX_I;
51  else if (target->GetGroup() != _player->GetGroup())
52  error = ERR_REFER_A_FRIEND_NOT_IN_GROUP;
53 
54  if (error)
55  {
56  WorldPacket data(SMSG_REFER_A_FRIEND_FAILURE, 24);
57  data << uint32(error);
58  if (error == ERR_REFER_A_FRIEND_NOT_IN_GROUP)
59  data << target->GetName();
60 
61  SendPacket(data);
62  return;
63  }
64 
65  WorldPacket data2(SMSG_PROPOSE_LEVEL_GRANT, 8);
66  data2 << _player->GetPackGUID();
