105 /* An abstract die for combat rolls with premultiplied integer chances */
106 template<class Side, Side Default, uint8 Sides>
107 struct Die
108 {
109  // MSVC++13-friendly initialization, switch to {0} when we end support for it
110  explicit Die() { for (uint8 i = 0; i < Sides; ++i) chance[i] = 0; }
111  Side roll(uint32 random)
112  {
113  uint32 rolling = 0;
114  for (uint8 side = 0; side < Sides; ++side)
115  {
116  if (chance[side])
117  {
118  rolling += chance[side];
119  if (random <= rolling)
120  return Side(side);
121  }
122  }
123  return Default;
124  }
125  void set(uint8 side, float chancef)
126  {
127  if (side < Sides)
128  chance[side] = chance_u(chancef);
129  }
130  uint32 chance[Sides];
131 };

321 inline wchar_t wcharToUpper(wchar_t wchar)
322 {
323  if (wchar >= L'a' && wchar <= L'z') // LATIN SMALL LETTER A - LATIN SMALL LETTER Z
324  return wchar_t(uint16(wchar) - 0x0020);
325  if (wchar == 0x00DF) // LATIN SMALL LETTER SHARP S
326  return wchar_t(0x1E9E);
327  if (wchar >= 0x00E0 && wchar <= 0x00F6) // LATIN SMALL LETTER A WITH GRAVE - LATIN SMALL LETTER O WITH DIAERESIS
328  return wchar_t(uint16(wchar) - 0x0020);
329  if (wchar >= 0x00F8 && wchar <= 0x00FE) // LATIN SMALL LETTER O WITH STROKE - LATIN SMALL LETTER THORN
330  return wchar_t(uint16(wchar) - 0x0020);
331  if (wchar >= 0x0101 && wchar <= 0x012F) // LATIN SMALL LETTER A WITH MACRON - LATIN SMALL LETTER I WITH OGONEK (only %2=1)
332  {
333  if (wchar % 2 == 1)
334  return wchar_t(uint16(wchar) - 0x0001);
335  }
336  if (wchar >= 0x0430 && wchar <= 0x044F) // CYRILLIC SMALL LETTER A - CYRILLIC SMALL LETTER YA
337  return wchar_t(uint16(wchar) - 0x0020);
338  if (wchar == 0x0451) // CYRILLIC SMALL LETTER IO
339  return wchar_t(0x0401);
340 
341  return wchar;
342 }
