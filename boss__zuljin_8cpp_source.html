429  void ExecuteAction(uint32 action) override
430  {
431  switch (action)
432  {
433  case ZULJIN_ACTION_LYNX_RUSH_RE_CHARGE:
434  {
435  ++m_lynxRushCount;
436  if (m_lynxRushCount != MAX_LYNX_RUSH)
437  {
438  SpellCastResult result = m_creature->CastSpell(nullptr, SPELL_LYNX_RUSH, TRIGGERED_IGNORE_COOLDOWNS);
439  if (result != SPELL_CAST_OK)
440  m_creature->RemoveAurasDueToSpell(SPELL_LYNX_RUSH);
441  }
442  else
443  m_creature->RemoveAurasDueToSpell(SPELL_LYNX_RUSH);
444  SetActionReadyStatus(ZULJIN_ACTION_LYNX_RUSH_RE_CHARGE, false);
445  return;
446  }
447  case ZULJIN_ACTION_PHASE_TRANSITION:
448  {
449  if (m_creature->HasAura(SPELL_LYNX_RUSH) || m_creature->HasAura(SPELL_CLAW_RAGE_TRIGGER)) // do not allow phase transition during lynx rush/claw rage
450  return;
451 
452  if (m_creature->GetHealthPercent() < m_healthCheck)
453  ExecutePhaseTransition();
454  return;
455  }
456  case ZULJIN_ACTION_WHIRLWIND:
457  if (DoCastSpellIfCan(nullptr, SPELL_WHIRLWIND) == CAST_OK)
458  ResetCombatAction(action, urand(15000, 20000));
459  return;
460  case ZULJIN_ACTION_GRIEVOUS_THROW:
461  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_AURA))
462  if (DoCastSpellIfCan(target, SPELL_GRIEVOUS_THROW) == CAST_OK)
463  ResetCombatAction(action, 10000);
464  return;
465  case ZULJIN_ACTION_PARALYSIS:
466  if (DoCastSpellIfCan(nullptr, SPELL_CREEPING_PARALYSIS) == CAST_OK)
467  ResetCombatAction(action, 27000);
468  return;
469  case ZULJIN_ACTION_OVERPOWER:
470  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_OVERPOWER) == CAST_OK)
471  ResetCombatAction(action, urand(12000, 16000));
472  return;
473  case ZULJIN_ACTION_CLAW_RAGE:
474  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_CLAW_RAGE, SELECT_FLAG_PLAYER))
475  if (DoCastSpellIfCan(target, SPELL_CLAW_RAGE) == CAST_OK)
476  ResetCombatAction(action, urand(15000, 20000));
477  return;
478  case ZULJIN_ACTION_LYNX_RUSH:
479  if (DoCastSpellIfCan(nullptr, SPELL_LYNX_RUSH) == CAST_OK)
480  {
481  m_lynxRushCount = 0;
482  ResetCombatAction(action, urand(20000, 25000));
483  }
484  return;
485  case ZULJIN_ACTION_FLAME_WHIRL:
486  if (DoCastSpellIfCan(nullptr, SPELL_FLAME_WHIRL) == CAST_OK)
