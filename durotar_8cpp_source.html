64  void RestartWakeTimer()
65  {
66  m_uiWakeTimer = urand(100000, 120000); //Random wakeup timer 1m40s to 2m, retail times unknown, 2m is duration of sleep aura
67  m_uiSleepTimer = 0; //Ensure only one of the two timers is running at any given time

77  {

85  void Awaken(Unit* pInvoker)
86  {
87  if (pInvoker->GetTypeId() == TYPEID_PLAYER)
88  {
89  DoScriptText(SAY_PEON_AWOKEN, m_creature, pInvoker);
90  m_creature->PlayDistanceSound(urand(0, 1) ? SOUND_PEON_AWAKE_1 : SOUND_PEON_AWAKE_2);
91  m_creature->SetWalk(false); //peon will run if awoken by player
92  ((Player*)pInvoker)->KilledMonsterCredit(m_creature->GetEntry(), m_creature->GetObjectGuid());
93  }
94  else
95  m_creature->SetWalk(true); //peon will walk if awoken on its own
96 
97  m_creature->RemoveAurasDueToSpell(SPELL_PEON_SLEEP);
98  RestartSleepTimer();
99  GoLumberPile();
100  }

102  void GoLumberPile()
103  {
104  if (GameObject* pLumber = GetClosestGameObjectWithEntry(m_creature, GO_LUMBERPILE, 15.0f))
105  {
106  float fX, fY, fZ;
107  pLumber->GetContactPoint(m_creature, fX, fY, fZ, CONTACT_DISTANCE);
108  m_creature->HandleEmote(EMOTE_STATE_NONE); //cancel chopping
109  m_creature->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
110  m_uiChopTimer = 0;
111  m_uiKneelTimer = 2500; //timer duration guessed, true retail time unknown
112  }
113  else
114  script_error_log("No GameObject of entry %u was found in range or something really bad happened.", GO_LUMBERPILE);
115  }

117  void MovementInform(uint32 uiMotionType, uint32 uiPointId) override
118  {
119  if (uiMotionType == POINT_MOTION_TYPE && uiPointId)
120  {
121  switch (uiPointId)
122  {
123  case 1: //lumber pile
124  {
125  m_creature->SetWalk(true);
126  m_bIsAtPile = true;
127  break;
128  }
129  case 2: //spawn
130  {
131  m_creature->GetMotionMaster()->MoveIdle();
132  m_creature->HandleEmote(EMOTE_STATE_NONE);
133  m_creature->SetFacingTo(m_fSpawnO);
134  DoCastSpellIfCan(m_creature, SPELL_PEON_SLEEP);
135  RestartWakeTimer();
136  break;
137  }
138  }
139  }
140  else if (uiMotionType == WAYPOINT_MOTION_TYPE)
141  {
142  //tree
143  m_creature->GetMotionMaster()->MoveIdle();
144  m_creature->PlayDistanceSound(SOUND_PEON_WORK);
145  m_creature->HandleEmote(EMOTE_STATE_WORK_CHOPWOOD);
146  m_uiChopTimer = urand(25000, 30000); //timer duration guessed, true retail time unknown
147  }
148  }

150  void UpdateAI(const uint32 uiDiff) override
151  {
152  if (m_uiGetUpTimer)
153  {
154  if (m_uiGetUpTimer <= uiDiff)
155  {
156  m_uiGetUpTimer = 0;
157  m_creature->HandleEmote(EMOTE_STATE_NONE);
158  m_creature->GetMotionMaster()->MoveWaypoint();
159  m_bIsAtPile = false;
160  }
161  else
162  m_uiGetUpTimer -= uiDiff;
163  }
164 
165  if (m_uiKneelTimer)
166  {
167  if (m_uiKneelTimer <= uiDiff)
168  {
169  if (m_bIsAtPile)
170  {
171  m_uiKneelTimer = 0;
172  m_creature->HandleEmote(EMOTE_ONESHOT_KNEEL);
173  m_uiGetUpTimer = 3000; //timer duration guessed, true retail time unknown
174  }
175  else
176  m_uiKneelTimer = 1; //guarantee it's checked again on next update
177  }
178  else
179  m_uiKneelTimer -= uiDiff;
180  }
181 
182  if (m_uiChopTimer)
183  {
184  if (m_uiChopTimer <= uiDiff)
185  {
186  m_uiChopTimer = 0;
187  GoLumberPile();
188  }
189  else
190  m_uiChopTimer -= uiDiff;
191  }
192 
193  if (m_uiSleepTimer)
194  {
195  if (m_uiSleepTimer <= uiDiff)
196  {
197  float fX, fY, fZ;
198  m_creature->GetRespawnCoord(fX, fY, fZ, &m_fSpawnO);
199  m_uiSleepTimer = 0;
200  m_creature->GetMotionMaster()->MovePoint(2, fX, fY, fZ);
201  m_bIsAtPile = false;
202  m_uiGetUpTimer = 0;
203  m_uiKneelTimer = 0;
204  m_uiChopTimer = 0;
205  }
206  else
207  m_uiSleepTimer -= uiDiff;
208  }
209 
210  if (m_uiWakeTimer)
211  {
212  if (m_uiWakeTimer <= uiDiff)
213  Awaken(m_creature);
214  else
215  {
216  //reapply aura if it fell off while the peon was unloaded
217  if (!m_creature->HasAura(SPELL_PEON_SLEEP))
218  DoCastSpellIfCan(m_creature, SPELL_PEON_SLEEP);
219  //decrement timer
220  m_uiWakeTimer -= uiDiff;
221  }
222  }
223  }
