148  void ExecuteAction(uint32 action) override
149  {
150  switch (action)
151  {
152  case EREGOS_ACTION_PLANAR_SHIFT:
153  if (m_creature->GetHealthPercent() < m_fHpPercent)
154  {
155  if (DoCastSpellIfCan(m_creature, SPELL_PLANAR_SHIFT) == CAST_OK)
156  {
157  // This will summon an anomaly for each player (vehicle)
158  DoCastSpellIfCan(m_creature, SPELL_PLANAR_ANOMALIES, CAST_TRIGGERED);
159 
160  switch (urand(0, 2))
161  {
162  case 0: DoScriptText(SAY_ARCANE_SHIELD, m_creature); break;
163  case 1: DoScriptText(SAY_FIRE_SHIELD, m_creature); break;
164  case 2: DoScriptText(SAY_NATURE_SHIELD, m_creature); break;
165  }
166  DoScriptText(EMOTE_ASTRAL_PLANE, m_creature);
167 
168  // reset timers for other combat actions
169  ResetCombatAction(EREGOS_ACTION_ARCANE_BARRAGE, urand(18000, 20000));
170  ResetCombatAction(EREGOS_ACTION_ARCANE_VOLLEY, urand(25000, 30000));
171  ResetCombatAction(EREGOS_ACTION_LEY_WHELP, urand(25000, 35000));
172  ResetCombatAction(EREGOS_ACTION_ENRAGED_ASSAULT, urand(60000, 70000));
173 
174  // set next phase to 20%
175  m_fHpPercent -= 40;
176  }
177  }
178  break;
179  case EREGOS_ACTION_ARCANE_BARRAGE:
180  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
181  {
182  if (DoCastSpellIfCan(pTarget, m_isRegularMode ? SPELL_ARCANE_BARRAGE : SPELL_ARCANE_BARRAGE_H) == CAST_OK)
183  ResetCombatAction(action, urand(2000, 3000));
184  }
185  break;
186  case EREGOS_ACTION_ARCANE_VOLLEY:
187  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_ARCANE_VOLLEY : SPELL_ARCANE_VOLLEY_H) == CAST_OK)
188  ResetCombatAction(action, urand(10000, 15000));
189  break;
190  case EREGOS_ACTION_ENRAGED_ASSAULT:
191  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGED_ASSAULT) == CAST_OK)
192  {
193  DoScriptText(SAY_FRENZY, m_creature);
194  ResetCombatAction(action, urand(40000, 50000));
195  }
196  break;
197  case EREGOS_ACTION_LEY_WHELP:
198  for (uint8 i = 0; i < m_uiMaxWhelps; ++i)
199  DoCastSpellIfCan(m_creature, SPELL_SUMMON_LEY_WHELP, CAST_TRIGGERED);
200 
201  ResetCombatAction(action, 20000);
202  break;
203  }
204  }
