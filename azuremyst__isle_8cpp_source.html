115  void UpdateAI(const uint32 uiDiff) override
116  {
117  if (m_uiSayThanksTimer)
118  {
119  if (m_uiSayThanksTimer <= uiDiff)
120  {
121  m_creature->RemoveAurasDueToSpell(SPELL_IRRIDATION);
122 
123  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_casterGuid))
124  {
125  if (pPlayer->GetTypeId() != TYPEID_PLAYER)
126  return;
127 
128  switch (urand(0, 6))
129  {
130  case 0: DoScriptText(SAY_HEAL1, m_creature, pPlayer); break;
131  case 1: DoScriptText(SAY_HEAL2, m_creature, pPlayer); break;
132  case 2: DoScriptText(SAY_HEAL3, m_creature, pPlayer); break;
133  case 3: DoScriptText(SAY_HEAL4, m_creature, pPlayer); break;
134  case 4: DoScriptText(SAY_HEAL5, m_creature, pPlayer); break;
135  case 5: DoScriptText(SAY_HEAL6, m_creature, pPlayer); break;
136  case 6: break; // say nothing
137  }
138 
139  pPlayer->TalkedToCreature(m_creature->GetEntry(), m_creature->GetObjectGuid());
140  }
141 
142  m_creature->GetMotionMaster()->Clear();
143  m_creature->GetMotionMaster()->MovePoint(0, -4115.053711f, -13754.831055f, 73.508949f);
144 
145  m_uiRunAwayTimer = 10000;
146  m_uiSayThanksTimer = 0;
147  }
148  else m_uiSayThanksTimer -= uiDiff;
149 
150  return;
151  }
152 
153  if (m_uiRunAwayTimer)
154  {
155  if (m_uiRunAwayTimer <= uiDiff)
156  m_creature->ForcedDespawn();
157  else
158  m_uiRunAwayTimer -= uiDiff;
159 
160  return;
161  }
162 
163  if (m_uiSayHelpTimer < uiDiff)
164  {
165  m_bCanSayHelp = true;
166  m_uiSayHelpTimer = 20000;
167  }
168  else m_uiSayHelpTimer -= uiDiff;
169  }

228  if (Player* pPlayer = GetPlayerForEscort())

230  if (Creature* pFather = GetClosestCreatureWithEntry(m_creature, NPC_COWLEN, 30.0f))
231  {
232  pFather->SetStandState(UNIT_STAND_STATE_STAND);
233  pFather->SetFacingToObject(m_creature);
234  }
235  break;
236  case 35:
237  if (Creature* pFather = GetClosestCreatureWithEntry(m_creature, NPC_COWLEN, 30.0f))
238  DoScriptText(SAY_DAUGHTER, pFather);
239  break;
240  case 36:
241  DoScriptText(EMOTE_HUG, m_creature);
242  break;
243  case 37:
244  if (Player* pPlayer = GetPlayerForEscort())
245  DoScriptText(SAY_END2, m_creature, pPlayer);
246  break;
247  case 38:
248  if (Creature* pFather = GetClosestCreatureWithEntry(m_creature, NPC_COWLEN, 30.0f))
249  {
250  pFather->SetStandState(UNIT_STAND_STATE_SIT);
251  pFather->GetMotionMaster()->MoveTargetedHome();
252  }
253  SetEscortPaused(true);
254  m_creature->ForcedDespawn(10000);
