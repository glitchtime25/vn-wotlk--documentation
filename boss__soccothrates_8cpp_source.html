186  void SpellHitTarget(Unit* target, const SpellEntry* spell) override
187  {
188  if (spell->Id == SPELL_FELFIRE_LINE_UP)
189  {
190  // need to get even points between caster and target to reposition felfire evenly
191  float sX = m_creature->GetPositionX(), sY = m_creature->GetPositionY(); // source coords
192  float tX = m_x, tY = m_y; // target coords
193  float felfireDistX = (tX - sX) / 7, felfireDistY = (tY - sY) / 7;
194  float fX = sX + (felfireDistX * m_lineUpCounter), fY = sY + (felfireDistY * m_lineUpCounter);
195  target->NearTeleportTo(fX, fY, m_creature->GetPositionZ(), m_creature->GetOrientation());
196  m_lineUpCounter++;
197  }
198  else if (spell->Id == SPELL_CHARGE && target->GetEntry() == NPC_WRATH_SCRYER_CHARGE_TARGET)
199  SetCombatMovement(true);
200  }

217  void UpdateAI(const uint32 uiDiff) override
218  {
219  DialogueUpdate(uiDiff);
220 
221  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
222  return;
223 
224  if (m_uiFelfireShockTimer < uiDiff)
225  {
226  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_FELFIRE_SHOCK : SPELL_FELFIRE_SHOCK_H) == CAST_OK)
227  m_uiFelfireShockTimer = urand(35000, 45000);
228  }
229  else
230  m_uiFelfireShockTimer -= uiDiff;
231 
232  if (m_uiKnockAwayTimer < uiDiff)
233  {
234  if (DoCastSpellIfCan(m_creature, SPELL_KNOCK_AWAY) == CAST_OK)
235  {
236  m_uiKnockAwayTimer = urand(30000, 35000);
237  m_uiFelfireLineupTimer = 2000;
238  }
239  }
240  else
241  m_uiKnockAwayTimer -= uiDiff;
242 
243  // Prepare the boss for charging
244  if (m_uiFelfireLineupTimer)
245  {
246  if (m_uiFelfireLineupTimer <= uiDiff)
247  {
248  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
249  {
250  if (DoCastSpellIfCan(pTarget, SPELL_CHARGE_TARGETING) == CAST_OK)
251  {
252  // ToDo: the Wrath-Scryer's Felfire npcs should be summoned at this point and aligned to the chosen target!
253  m_creature->CastSpell(m_creature, SPELL_FELFIRE_LINE_UP, TRIGGERED_OLD_TRIGGERED);
254  DoScriptText(urand(0, 1) ? SAY_CHARGE_1 : SAY_CHARGE_2, m_creature, pTarget);
255 
256  m_uiChargeTimer = 2500;
257  m_uiFelfireLineupTimer = 0;
258  }
259  }
260  }
261  else
262  m_uiFelfireLineupTimer -= uiDiff;
263  }
264 
265  // Charge the target
266  if (m_uiChargeTimer)
267  {
268  if (m_uiChargeTimer <= uiDiff)
269  {
270  m_creature->RemoveAurasDueToSpell(SPELL_KNOCK_AWAY);
271  SetCombatMovement(false); // prevents interrupting charge
272  // Note: this spell will also light up the Wrath-Scryer's Felfire npcs
273  if (DoCastSpellIfCan(m_creature, SPELL_CHARGE) == CAST_OK)
274  m_uiChargeTimer = 0;
275  }
276  else
277  m_uiChargeTimer -= uiDiff;
278  }
279 
280  DoMeleeAttackIfReady();
281  }
