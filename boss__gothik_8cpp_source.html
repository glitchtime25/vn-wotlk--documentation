99  boss_gothikAI(Creature* creature) : BossAI(creature, GOTHIK_ACTIONS_MAX),

105  AddOnDeathText(SAY_DEATH);
106  AddTimerlessCombatAction(GOTHIK_STOP_TELE, true);
107  AddCustomAction(GOTHIK_SUMMON_TRAINEE, true, [&]() { StartTraineeSummons(); });
108  AddCustomAction(GOTHIK_SUMMON_DEATH_KNIGHT, true, [&]() { StartKnightSummons(); });
109  AddCustomAction(GOTHIK_SUMMON_RIDER, true, [&]() { StartRiderSummons(); });
110  AddCombatAction(GOTHIK_TELEPORT, true);
111  AddCustomAction(GOTHIK_GROUND_PHASE, true, [&]() { HandleGroundPhase(); });
112  AddCustomAction(GOTHIK_OPEN_GATES, true, [&]() { HandleOpenGates(); }); // in wotlk happens during phase 1
113  AddCustomAction(GOTHIK_SPEECH, true, [&]()
114  {
115  switch (m_speechStep)
116  {
117  case 0: DoBroadcastText(SAY_SPEECH_1, m_creature); DoPlaySoundToSet(m_creature, 8807); ResetTimer(GOTHIK_SPEECH, 4s); break;
118  case 1: DoBroadcastText(SAY_SPEECH_2, m_creature); ResetTimer(GOTHIK_SPEECH, 6s); break;
119  case 2: DoBroadcastText(SAY_SPEECH_3, m_creature); ResetTimer(GOTHIK_SPEECH, 5s); break;
120  case 3:
121  {
122  DoBroadcastText(SAY_SPEECH_4, m_creature);
123  m_uiPhase = PHASE_BALCONY;
124  break;
125  }
126  }
127  ++m_speechStep;
128  });
129  }

272  void HandleOpenGates()
273  {
274  ProcessCentralDoor();
275  Map::PlayerList const& players = m_instance->instance->GetPlayers();
276  for (const auto& playerRef : players)
277  {
278  if (Player* player = playerRef.getSource())
279  {
280  for (ObjectGuid guid : m_summonedAddGuids)
281  {
282  if (Creature* add = m_creature->GetMap()->GetCreature(guid))
283  {
284  // attack other side
285  if (add->IsAlive() && m_instance->IsInRightSideGothArea(player) != m_instance->IsInRightSideGothArea(add))
286  {
287  add->AddThreat(player);
288  }
289  }
290  }
291  }
292  }
293 
294  for (ObjectGuid guid : m_summonedAddGuids)
295  {
296  if (Creature* add = m_creature->GetMap()->GetCreature(guid))
297  {
298  if (add->IsAlive() && !add->GetVictim()) // spawn that doesnt have target - attack closest
299  add->AI()->AttackClosestEnemy();
300  }
301  }
302  }

313  void ExecuteAction(uint32 action) override
314  {
315  switch (action)
316  {
317  case GOTHIK_STOP_TELE:
318  {
319  if (m_creature->GetHealthPercent() <= 30.0f)
320  {
321  m_uiPhase = PHASE_STOP_TELEPORTING;
322  DisableCombatAction(action);
323  }
324  return;
325  }
326  case GOTHIK_TELEPORT:
327  {
328  uint32 teleportSpell = m_instance->IsInRightSideGothArea(m_creature) ? SPELL_TELEPORT_LEFT : SPELL_TELEPORT_RIGHT;
329  if (DoCastSpellIfCan(nullptr, teleportSpell) == CAST_OK)
330  break;
331  return;
332  }
333  }
334  ResetCombatAction(action, GetSubsequentActionTimer(action));
335  }
