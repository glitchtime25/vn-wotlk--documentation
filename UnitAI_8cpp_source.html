160 CanCastResult UnitAI::DoCastSpellIfCan(Unit* target, uint32 spellId, uint32 castFlags)
161 {
162  Unit* caster = m_unit;
163 
164  if (target)
165  {
166  if (castFlags & CAST_SWITCH_CASTER_TARGET)
167  std::swap(caster, target);
168 
169  if (castFlags & CAST_FORCE_TARGET_SELF)
170  caster = target;
171  }
172  else if (castFlags & (CAST_FORCE_TARGET_SELF | CAST_SWITCH_CASTER_TARGET))
173  return CAST_FAIL_OTHER;
174 
175  if (GetAIOrder() == ORDER_EVADE && !(castFlags & CAST_TRIGGERED))
176  return CAST_FAIL_EVADE;
177 
178  // Allowed to cast only if not casting (unless we interrupt ourself) or if spell is triggered
179  if (!caster->IsNonMeleeSpellCasted(false) || (castFlags & (CAST_TRIGGERED | CAST_INTERRUPT_PREVIOUS)))
180  {
181  if (const SpellEntry* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId))
182  {
183  // If cast flag CAST_AURA_NOT_PRESENT is active, check if target already has aura on them
184  if (castFlags & CAST_AURA_NOT_PRESENT)
185  {
186  if (!target)
187  {
188  if (caster->HasAura(spellId))
189  return CAST_FAIL_TARGET_AURA;
190  }
191  else if (target->HasAura(spellId))
192  return CAST_FAIL_TARGET_AURA;
193  }
194 
195  // Check if cannot cast spell
196  if (!(castFlags & (CAST_FORCE_TARGET_SELF | CAST_FORCE_CAST)))
197  {
198  CanCastResult castResult = CanCastSpell(target, spellInfo, (castFlags & CAST_TRIGGERED) != 0);
199 
200  if (castResult != CAST_OK)
201  return castResult;
202  }
203 
204  // Interrupt any previous spell
205  if (castFlags & CAST_INTERRUPT_PREVIOUS && caster->IsNonMeleeSpellCasted(false))
206  caster->InterruptNonMeleeSpells(false);
207 
208  // Creature should interrupt any current melee spell
209  caster->InterruptSpell(CURRENT_MELEE_SPELL);
210 
211  // Creature should stop wielding weapon while casting
212  // caster->SetSheath(SHEATH_STATE_UNARMED);
213 
214  uint32 flags = (castFlags & CAST_TRIGGERED ? TRIGGERED_OLD_TRIGGERED : TRIGGERED_NONE) | (castFlags & CAST_IGNORE_UNSELECTABLE_TARGET ? TRIGGERED_IGNORE_UNSELECTABLE_FLAG : TRIGGERED_NONE);
215 
216  if (flags == TRIGGERED_NONE)
217  flags |= TRIGGERED_NORMAL_COMBAT_CAST;
218 
219  SpellCastResult result;
220  if (castFlags & CAST_ONLY_XYZ)
221  result = caster->CastSpell(target->GetPositionX(), target->GetPositionY(), target->GetPositionZ(), spellInfo, flags);
222  else
223  result = caster->CastSpell(target, spellInfo, flags);
224  if (result != SPELL_CAST_OK)
225  {
226  CanCastResult canCastResult = CAST_FAIL_OTHER;
227  switch (result) // temporary adapter
228  {
229  case SPELL_FAILED_OUT_OF_RANGE:
230  canCastResult = CAST_FAIL_TOO_FAR;
231  break;
232  case SPELL_FAILED_TOO_CLOSE:
233  canCastResult = CAST_FAIL_TOO_CLOSE;
234  break;
235  case SPELL_FAILED_LINE_OF_SIGHT:
236  canCastResult = CAST_FAIL_NOT_IN_LOS;
237  break;
238  case SPELL_FAILED_PACIFIED:
239  case SPELL_FAILED_SILENCED:
240  canCastResult = CAST_FAIL_CAST_PREVENTED;
241  break;
242  case SPELL_FAILED_PREVENTED_BY_MECHANIC:
243  if (!caster->IsCrowdControlled())
244  {
245  canCastResult = CAST_FAIL_CAST_PREVENTED;
246  break;
247  }
248  [[fallthrough]];
249  case SPELL_FAILED_STUNNED:
250  case SPELL_FAILED_CONFUSED:
251  case SPELL_FAILED_FLEEING:
252  canCastResult = CAST_FAIL_STATE;
253  break;
254  case SPELL_FAILED_NOT_READY:
255  canCastResult = CAST_FAIL_COOLDOWN;
256  break;
257  case SPELL_FAILED_NO_POWER:
258  canCastResult = CAST_FAIL_POWER;
259  break;
260  case SPELL_FAILED_CASTER_AURASTATE: // valid - doesnt need logging
261  case SPELL_FAILED_TARGET_AURASTATE:
262  case SPELL_FAILED_TARGET_NO_WEAPONS:
263  case SPELL_FAILED_BAD_TARGETS:
264  case SPELL_FAILED_DONT_REPORT:
265  canCastResult = CAST_FAIL_MISCELLANEOUS;
266  break;
267  default: break;
268  }
269 
270  if (canCastResult == CAST_FAIL_OTHER)
271  sLog.outBasic("DoCastSpellIfCan by %s attempt to cast spell %u but spell failed due to unknown result %u : %s.", m_unit->GetObjectGuid().GetString().c_str(), spellId, result, GetSpellCastResultString(result));
272  return HandleSpellCastResult(canCastResult, spellInfo);
273  }
274  return HandleSpellCastResult(CAST_OK, spellInfo);
275  }
276  sLog.outErrorDb("DoCastSpellIfCan by %s attempt to cast spell %u but spell does not exist.", m_unit->GetGuidStr().c_str(), spellId);
277  return CAST_FAIL_OTHER;
278  }
279  return CAST_FAIL_IS_CASTING;

355 void UnitAI::OnSpellCastStateChange(Spell const* spell, bool state, WorldObject* target)
356 {
357  if (!state && spell != m_currentSpell)
358  return;
359 
360  SpellEntry const* spellInfo = spell->m_spellInfo;
361  if (spellInfo->HasAttribute(SPELL_ATTR_EX4_ALLOW_CAST_WHILE_CASTING) || spellInfo->HasAttribute(SPELL_ATTR_ON_NEXT_SWING_NO_DAMAGE) || spellInfo->HasAttribute(SPELL_ATTR_ON_NEXT_SWING))
362  return;
363 
364  // Creature should always stop before it will cast a non-instant spell
365  if (state)
366  if ((spell->GetCastTime() && spellInfo->InterruptFlags & SPELL_INTERRUPT_FLAG_MOVEMENT) || (IsChanneledSpell(spellInfo) && spellInfo->ChannelInterruptFlags & AURA_INTERRUPT_FLAG_MOVING))
367  m_unit->InterruptMoving();
368 
369  bool forceTarget = false;
370 
371  // Targeting seems to be directly affected by eff index 0 targets, client does the same thing (spell id 38523 exception)
372  for (uint32 i = 0; i < MAX_EFFECT_INDEX; ++i)
373  {
374  switch (spellInfo->EffectImplicitTargetA[EFFECT_INDEX_0])
375  {
376  case TARGET_ENUM_UNITS_ENEMY_IN_CONE_24: // ignores everything and keeps turning
377  return;
378  case TARGET_LOCATION_CASTER_SRC:
379  case TARGET_LOCATION_CASTER_DEST:
380  switch (spellInfo->EffectImplicitTargetB[EFFECT_INDEX_0])
381  {
382  case TARGET_ENUM_UNITS_ENEMY_AOE_AT_SRC_LOC: // ignores everything and keeps turning if instant
383  case TARGET_ENUM_UNITS_ENEMY_AOE_AT_DEST_LOC:
384  if (!spell->GetCastTime())
385  return;
386  break;
387  }
388  break;
389  case TARGET_UNIT_FRIEND:
390  case TARGET_UNIT_ENEMY: forceTarget = true; break;
391  case TARGET_UNIT_SCRIPT_NEAR_CASTER:
392  default: break;
393  }
394  if (forceTarget)
395  break;
396  }
397 
398  if (state)
399  {
400  if (!spellInfo->HasAttribute(SPELL_ATTR_EX5_AI_DOESNT_FACE_TARGET) && target && (m_unit != target || forceTarget))
401  {
402  m_unit->SetTarget(target);
403  if (m_unit != target)
404  m_unit->SetOrientation(m_unit->GetAngle(target));
405  else
406  m_unit->SetOrientation(m_unit->GetOrientation());
407  }
408  else
409  {
410  if (m_unit->HasTarget())
411  {
412  m_unit->SetFacingTo(m_unit->GetOrientation());
413  m_unit->SetTarget(nullptr);
414  }
415  }
416  }
417  else
418  {
419  std::set<uint32> spellIdsForTurning = { 31306, 33813, 38739, 44811, 46292 };
420  if (!spell->GetCastTime() && spellIdsForTurning.find(spellInfo->Id) != spellIdsForTurning.end())
421  {
422  HandleDelayedInstantAnimation(spellInfo);
423  }
424  else
425  {
426  if (m_unit->GetVictim() && !IsTargetingRestricted())
427  m_unit->SetTarget(m_unit->GetVictim());
428  else
429  m_unit->SetTarget(nullptr);
430  }
431  }
432 
433  if (state)
434  {
435  m_currentSpell = spell;
436  }
437 }
