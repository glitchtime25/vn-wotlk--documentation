69 RealmBuildInfo const* FindBuildInfo(uint16 _build)
70 {
71  // first build is low bound of always accepted range
72  if (_build >= ExpectedRealmdClientBuilds[0].build)
73  return &ExpectedRealmdClientBuilds[0];
74 
75  // continue from 1 with explicit equal check
76  for (int i = 1; ExpectedRealmdClientBuilds[i].build; ++i)
77  if (_build == ExpectedRealmdClientBuilds[i].build)
78  return &ExpectedRealmdClientBuilds[i];
79 
80  // none appropriate build
81  return nullptr;

157 void RealmList::UpdateRealms(bool init)
158 {
159  DETAIL_LOG("Updating Realm List...");
160 
162  QueryResult* result = LoginDatabase.Query("SELECT id, name, address, port, icon, realmflags, timezone, allowedSecurityLevel, population, realmbuilds FROM realmlist WHERE (realmflags & 1) = 0 ORDER BY name");
163 
165  if (result)
166  {
167  do
168  {
169  Field* fields = result->Fetch();
170 
171  uint32 Id = fields[0].GetUInt32();
172  std::string name = fields[1].GetCppString();
173  uint8 realmflags = fields[5].GetUInt8();
174  uint8 allowedSecurityLevel = fields[7].GetUInt8();
175 
176  if (realmflags & ~(REALM_FLAG_OFFLINE | REALM_FLAG_NEW_PLAYERS | REALM_FLAG_RECOMMENDED | REALM_FLAG_SPECIFYBUILD))
177  {
178  sLog.outError("Realm (id %u, name '%s') can only be flagged as OFFLINE (mask 0x02), NEWPLAYERS (mask 0x20), RECOMMENDED (mask 0x40), or SPECIFICBUILD (mask 0x04) in DB", Id, name.c_str());
179  realmflags &= (REALM_FLAG_OFFLINE | REALM_FLAG_NEW_PLAYERS | REALM_FLAG_RECOMMENDED | REALM_FLAG_SPECIFYBUILD);
180  }
181 
182  UpdateRealm(
183  Id, name, fields[2].GetCppString(), fields[3].GetUInt32(),
184  fields[4].GetUInt8(), RealmFlags(realmflags), fields[6].GetUInt8(),
185  (allowedSecurityLevel <= SEC_ADMINISTRATOR ? AccountTypes(allowedSecurityLevel) : SEC_ADMINISTRATOR),
186  fields[8].GetFloat(), fields[9].GetCppString());
187 
188  if (init)
189  sLog.outString("Added realm id %u, name '%s'", Id, name.c_str());
190  }
191  while (result->NextRow());
192  delete result;
193  }
194 }
