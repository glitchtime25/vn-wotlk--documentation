48 bool AreaTrigger_at_frostmourne_chamber(Player* pPlayer, AreaTriggerEntry const* pAt)
49 {
50  if (pAt->id == AREATRIGGER_FROSTMOURNE_ALTAR)
51  {
52  if (pPlayer->IsGameMaster() || !pPlayer->IsAlive())
53  return false;
54 
55  instance_halls_of_reflection* pInstance = (instance_halls_of_reflection*)pPlayer->GetInstanceData();
56  if (!pInstance)
57  return false;
58 
59  // reset event only after fail and if intro is complete or Maryn isn't complete
60  if (pInstance->GetData(TYPE_FROSTMOURNE_INTRO) != DONE || pInstance->GetData(TYPE_MARWYN) == DONE || pInstance->GetData(TYPE_QUEL_DELAR) == IN_PROGRESS)
61  return false;
62 
63  // reset on Falric fail
64  if (pInstance->GetData(TYPE_FALRIC) == FAIL)
65  {
66  pPlayer->SummonCreature(NPC_FALRIC, afFalricSpawnLoc[0], afFalricSpawnLoc[1], afFalricSpawnLoc[2], afFalricSpawnLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
67  pPlayer->SummonCreature(NPC_MARWYN, afMarwybSpawnLoc[0], afMarwybSpawnLoc[1], afMarwybSpawnLoc[2], afMarwybSpawnLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0);
68  }
69  // reset on Marwyn fail
70  else if (pInstance->GetData(TYPE_FALRIC) == DONE && pInstance->GetData(TYPE_MARWYN) == FAIL)
71  {
72  if (Creature* pMarwyn = pPlayer->SummonCreature(NPC_MARWYN, afMarwybSpawnLoc[0], afMarwybSpawnLoc[1], afMarwybSpawnLoc[2], afMarwybSpawnLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0))
73  DoScriptText(SAY_MARWYN_GAUNTLET_START, pMarwyn);
74  }
75  }
76 
77  return false;
78 }

121 bool AreaTrigger_at_frostworn_general(Player* pPlayer, AreaTriggerEntry const* pAt)
122 {
123  if (pAt->id == AREATRIGGER_FROSTWORN_GENERAL)
124  {
125  if (pPlayer->IsGameMaster() || !pPlayer->IsAlive())
126  return false;
127 
128  instance_halls_of_reflection* pInstance = (instance_halls_of_reflection*)pPlayer->GetInstanceData();
129  if (!pInstance)
130  return false;
131 
132  if (pInstance->GetData(TYPE_MARWYN) != DONE || pInstance->GetData(TYPE_FROSTWORN_GENERAL) == IN_PROGRESS || pInstance->GetData(TYPE_FROSTWORN_GENERAL) == DONE)
133  return false;
134 
135  // Get the dongeon traps list to get summon location
136  GuidList lDugeonTrapsGuids;
137  pInstance->GetDungeonTrapsGUIDList(lDugeonTrapsGuids);
138 
139  if (lDugeonTrapsGuids.empty())
140  {
141  script_error_log("instance_halls_of_reflection: Error: couldn't find any dungeon trap stalker.");
142  return true;
143  }
144 
145  // summon a reflection for each player in the list
146  Map::PlayerList const& allPlayers = pInstance->instance->GetPlayers();
147  if (allPlayers.isEmpty())
148  return false;
149 
150  for (const auto& allPlayer : allPlayers)
151  {
152  Player* pPlayerTarget = allPlayer.getSource();
153  if (!pPlayerTarget)
154  continue;
155 
156  // Spawn a reflection using the player as summoner and the dungeon trap for location
157  GuidList::iterator iter = lDugeonTrapsGuids.begin();
158  advance(iter, urand(0, lDugeonTrapsGuids.size() - 1));
159 
160  Creature* pSpawnCreature = pPlayerTarget->GetMap()->GetCreature(*iter);
161  if (!pSpawnCreature)
162  return false;
163 
164  // spawn a spiritual reflection
165  // ToDo: research what is the difference between the two entries
166  if (Creature* pReflection = pPlayerTarget->SummonCreature(urand(0, 1) ? NPC_SPIRITUAL_REFLECTION_1 : NPC_SPIRITUAL_REFLECTION_2, pSpawnCreature->GetPositionX(), pSpawnCreature->GetPositionY(), pSpawnCreature->GetPositionZ(), pSpawnCreature->GetOrientation(), TEMPSPAWN_DEAD_DESPAWN, 0))
167  {
168  pPlayerTarget->CastSpell(pReflection, SPELL_HALLS_OF_REFLECTION_CLONE_NAME, TRIGGERED_OLD_TRIGGERED);
169  pPlayerTarget->CastSpell(pReflection, SPELL_HALLS_OF_REFLECTION_CLONE, TRIGGERED_OLD_TRIGGERED);
170  pReflection->CastSpell(pReflection, SPELL_HALLS_OF_REFLECTION_GHOST, TRIGGERED_OLD_TRIGGERED);
171  }
172 
173  lDugeonTrapsGuids.remove(*iter);
174  }
175 
176  pInstance->SetData(TYPE_FROSTWORN_GENERAL, IN_PROGRESS);
177  }
178 
179  return false;
180 }

192 bool EffectDummyCreature_spell_summon_reflections(Unit* /*pCaster*/, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
193 {
194  if (uiSpellId == SPELL_SUMMON_REFLECTIONS && uiEffIndex == EFFECT_INDEX_0)
195  {
196  instance_halls_of_reflection* pInstance = (instance_halls_of_reflection*)pCreatureTarget->GetInstanceData();
197  if (!pInstance)
198  return true;
199 
200  GuidList lReflectionsGuids;
201  pInstance->GetReflectionsGUIDList(lReflectionsGuids);
202 
203  if (lReflectionsGuids.empty())
204  {
205  script_error_log("instance_halls_of_reflection: Error: couldn't find any spiritual reflection.");
206  return true;
207  }
208 
209  // Set all reflections to attack owner
210  for (GuidList::const_iterator itr = lReflectionsGuids.begin(); itr != lReflectionsGuids.end(); ++itr)
211  {
212  if (Creature* pCreature = pCreatureTarget->GetMap()->GetCreature(*itr))
213  {
214  if (pCreature->IsTemporarySummon())
215  {
216  if (Player* pSummoner = pCreature->GetMap()->GetPlayer(pCreature->GetSpawnerGuid()))
217  {
218  pCreature->SetLevitate(false);
219  pCreature->AI()->AttackStart(pSummoner);
220  pCreature->RemoveAurasDueToSpell(SPELL_FROZEN_POSITION);
221  pCreature->CastSpell(pSummoner, SPELL_JUMPT_TO_TARGET, TRIGGERED_OLD_TRIGGERED);
222  }
223  }
224  }
225  }
226  }
227 
228  return true;
