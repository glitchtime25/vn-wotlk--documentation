45 bool GOUse_go_mausoleum_door(Player* pPlayer, GameObject* /*pGo*/)

107  void DamageTaken(Unit* dealer, uint32& damage, DamageEffectType /*damagetype*/, SpellEntry const* /*spellInfo*/) override
108  {
109  if (damage > m_creature->GetHealth() || ((m_creature->GetHealth() - damage) * 100 / m_creature->GetMaxHealth() < 15))
110  {
111  damage = std::min(damage, m_creature->GetHealth() - 1);
112 
113  m_creature->CombatStop(true);

119  m_playerGuid = dealer->GetObjectGuid();

123  void UpdateAI(const uint32 uiDiff) override
124  {
125  if (m_uiPhase)
126  {
127  if (m_uiPhaseTimer < uiDiff)
128  m_uiPhaseTimer = 7500;
129  else
130  {
131  m_uiPhaseTimer -= uiDiff;
132  return;
133  }
134 
135  switch (m_uiPhase)
136  {
137  case 1:
138  DoScriptText(SAY_COMPLETE, m_creature);
139  ++m_uiPhase;
140  break;
141  case 2:
142  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid))
143  pPlayer->AreaExploredOrEventHappens(QUEST_590);
144 
145  m_creature->CastSpell(m_creature, SD_SPELL_DRINK, TRIGGERED_OLD_TRIGGERED);
146  ++m_uiPhase;
147  break;
148  case 3:
149  EnterEvadeMode();
150  break;
151  }
152 
153  return;
154  }
155 
156  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
157  return;
158 
159  DoMeleeAttackIfReady();
160  }
