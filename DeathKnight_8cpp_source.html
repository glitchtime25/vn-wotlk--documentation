24 struct ScourgeStrike : public SpellScript
25 {
26  void OnHit(Spell* spell, SpellMissInfo missInfo) const override
27  {
28  uint32 count = 0;
29  Unit* target = spell->GetUnitTarget();
30  Unit::SpellAuraHolderMap const& auras = target->GetSpellAuraHolderMap();
31  for (const auto& aura : auras)
32  {
33  if (aura.second->GetSpellProto()->Dispel == DISPEL_DISEASE &&

43  }

147  uint8 newRunesState = procData.spell->GetNewRuneState();

157  {
158  player->AddRuneByAuraEffect(i, RUNE_DEATH, aura);
