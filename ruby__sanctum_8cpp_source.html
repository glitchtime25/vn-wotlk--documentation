174 void instance_ruby_sanctum::SetData(uint32 uiType, uint32 uiData)
175 {
176  switch (uiType)
177  {
178  case TYPE_SAVIANA:
179  m_auiEncounter[uiType] = uiData;
180  if (uiData == DONE)
181  DoHandleZarithrianDoor();
182  break;
183  case TYPE_BALTHARUS:
184  m_auiEncounter[uiType] = uiData;
185  if (uiData == DONE)
186  {
187  DoHandleZarithrianDoor();
188  DoUseDoorOrButton(GO_FIRE_FIELD);
189 
190  // Start outro event by DB script
191  if (Creature* pXerestrasza = GetSingleCreatureFromStorage(NPC_XERESTRASZA))
192  pXerestrasza->GetMotionMaster()->MoveWaypoint();
193  }
194  break;
195  case TYPE_ZARITHRIAN:
196  m_auiEncounter[uiType] = uiData;
197  DoUseDoorOrButton(GO_FLAME_WALLS);
198  if (uiData == DONE)
199  {
200  // Start Halion summoning process
201  if (Creature* pSummoner = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
202  {
203  pSummoner->CastSpell(pSummoner, SPELL_FIRE_PILLAR, TRIGGERED_NONE);
204  m_uiHalionSummonTimer = 5000;
205  }
206  }
207  break;
208  case TYPE_HALION:
209  // Don't set the same data twice
210  if (m_auiEncounter[uiType] == uiData)
211  return;
212 
213  m_auiEncounter[uiType] = uiData;
214  DoUseDoorOrButton(GO_FLAME_RING);
215  DoUseDoorOrButton(GO_TWILIGHT_FLAME_RING);
216 
217  // encounter unit frame
218  if (uiData == DONE || uiData == FAIL)
219  {
220  // remove encounter frames
221  if (Creature* pDragon = GetSingleCreatureFromStorage(NPC_HALION_REAL))
222  SendEncounterFrame(ENCOUNTER_FRAME_DISENGAGE, pDragon->GetObjectGuid());
223  if (Creature* pDragon = GetSingleCreatureFromStorage(NPC_HALION_TWILIGHT))
224  SendEncounterFrame(ENCOUNTER_FRAME_DISENGAGE, pDragon->GetObjectGuid());
225 
226  // disable world state
227  DoUpdateWorldState(WORLD_STATE_CORPOREALITY, 0);
228  }
229 
230  // cleanup
231  switch (uiData)
232  {
233  case FAIL:
234  // Despawn the boss
235  if (Creature* pHalion = GetSingleCreatureFromStorage(NPC_HALION_REAL))
236  pHalion->ForcedDespawn();
237  if (Creature* pHalion = GetSingleCreatureFromStorage(NPC_HALION_TWILIGHT))
238  pHalion->ForcedDespawn();
239  // Note: rest of the cleanup is handled by creature_linking
240 
241  m_uiHalionResetTimer = 30000;
242  // no break;
243  case DONE:
244  // clear debuffs
245  if (Creature* pController = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
246  pController->CastSpell(pController, SPELL_CLEAR_DEBUFFS, TRIGGERED_OLD_TRIGGERED);
247 
248  // Despawn the portals
249  if (GameObject* pPortal = GetSingleGameObjectFromStorage(GO_TWILIGHT_PORTAL_ENTER_1))
250  {
251  pPortal->SetForcedDespawn();
252  pPortal->SetLootState(GO_JUST_DEACTIVATED);
253  }
254 
255  for (const auto& guid : m_lTwilightPortalsGuidList)
256  {
257  if (GameObject* pPortal = instance->GetGameObject(guid))
258  {
259  pPortal->SetForcedDespawn();
260  pPortal->SetLootState(GO_JUST_DEACTIVATED);
261  }
262  }
263 
264  // reset the controller
265  if (Creature* pController = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
266  pController->AI()->EnterEvadeMode();
267  break;
268  }
269  break;
270  }
271 
272  if (uiData == DONE)
273  {
274  OUT_SAVE_INST_DATA;
275 
276  std::ostringstream saveStream;
277  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
278 
279  strInstData = saveStream.str();
280 
281  SaveToDB();
282  OUT_SAVE_INST_DATA_COMPLETE;
283  }
284 }

294 void instance_ruby_sanctum::Update(uint32 uiDiff)
295 {
296  if (m_uiHalionSummonTimer)
297  {
298  if (m_uiHalionSummonTimer <= uiDiff)
299  {
300  switch (m_uiHalionSummonStage)
301  {
302  case 0:
303  // Burn the first line of trees
304  DoUseDoorOrButton(GO_BURNING_TREE_1);
305  DoUseDoorOrButton(GO_BURNING_TREE_2);
306  m_uiHalionSummonTimer = 5000;
307  break;
308  case 1:
309  // Burn the second line of trees
310  DoUseDoorOrButton(GO_BURNING_TREE_3);
311  DoUseDoorOrButton(GO_BURNING_TREE_4);
312  m_uiHalionSummonTimer = 4000;
313  break;
314  case 2:
315  // Cast Fiery explosion
316  if (Creature* pSummoner = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
317  pSummoner->CastSpell(pSummoner, SPELL_FIERY_EXPLOSION, TRIGGERED_OLD_TRIGGERED);
318  m_uiHalionSummonTimer = 2000;
319  case 3:
320  // Spawn Halion
321  if (Creature* pSummoner = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
322  {
323  if (Creature* pHalion = pSummoner->SummonCreature(NPC_HALION_REAL, pSummoner->GetPositionX(), pSummoner->GetPositionY(), pSummoner->GetPositionZ(), 3.159f, TEMPSPAWN_DEAD_DESPAWN, 0))
324  {
325  DoScriptText(SAY_HALION_SPAWN, pHalion);
326  pHalion->SetPhaseMask(1, true);
327  }
328 
329  if (Creature* pHalion = pSummoner->SummonCreature(NPC_HALION_TWILIGHT, pSummoner->GetPositionX(), pSummoner->GetPositionY(), pSummoner->GetPositionZ(), 3.159f, TEMPSPAWN_DEAD_DESPAWN, 0))
330  pHalion->SetPhaseMask(32, true);
331  }
332  m_uiHalionSummonTimer = 0;
333  break;
334  }
335  ++m_uiHalionSummonStage;
336  }
337  else
338  m_uiHalionSummonTimer -= uiDiff;
339  }
340 
341  // Resummon Halion if the encounter resets
342  if (m_uiHalionResetTimer)
343  {
344  if (m_uiHalionResetTimer <= uiDiff)
345  {
346  if (Creature* pSummoner = GetSingleCreatureFromStorage(NPC_HALION_CONTROLLER))
347  {
348  if (Creature* pHalion = pSummoner->SummonCreature(NPC_HALION_REAL, pSummoner->GetPositionX(), pSummoner->GetPositionY(), pSummoner->GetPositionZ(), 3.159f, TEMPSPAWN_DEAD_DESPAWN, 0))
349  pHalion->SetPhaseMask(1, true);
350 
351  if (Creature* pHalion = pSummoner->SummonCreature(NPC_HALION_TWILIGHT, pSummoner->GetPositionX(), pSummoner->GetPositionY(), pSummoner->GetPositionZ(), 3.159f, TEMPSPAWN_DEAD_DESPAWN, 0))
352  pHalion->SetPhaseMask(32, true);
353  }
354 
355  m_uiHalionResetTimer = 0;
356  }
357  else
358  m_uiHalionResetTimer -= uiDiff;
359  }
