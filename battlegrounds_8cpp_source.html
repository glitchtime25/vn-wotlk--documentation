148 struct FlagClickBg : public SpellScript
149 {
150  SpellCastResult OnCheckCast(Spell* spell, bool /*strict*/) const override
151  {
152  switch (spell->m_spellInfo->Id)
153  {
154  case 61266: // Warsong Flag
155  case 61265: // Silverwing Flag
156  return spell->GetTrueCaster()->GetMapId() == 489 && spell->GetTrueCaster()->GetMap()->IsBattleGround() ? SPELL_CAST_OK : SPELL_FAILED_REQUIRES_AREA;
157  case 34976: // Netherstorm Flag
158  return spell->GetTrueCaster()->GetMapId() == 566 && spell->GetTrueCaster()->GetMap()->IsBattleGround() ? SPELL_CAST_OK : SPELL_FAILED_REQUIRES_AREA;
159  }
160  return SPELL_CAST_OK;
161  }
162 
163  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
164  {
165  Unit* target = spell->GetUnitTarget();
166  uint32 spellId = 0;
167  uint32 eventId = 0;
168  switch (spell->m_spellInfo->Id)
169  {
170  case 61265: spellId = 23335; eventId = 8504; break; // Alliance Flag Pickup
171  case 61266: spellId = 23333; eventId = 8505; break; // Horde Flag Pickup
172  }
173 
174  // flagstand and flagdrop share spells
175  if (spell->GetTrueCaster()->IsGameObject() && static_cast<GameObject*>(spell->GetTrueCaster())->GetGoType() == GAMEOBJECT_TYPE_FLAGDROP)
176  {
177  if (WorldObject* spawner = static_cast<GameObject*>(spell->GetTrueCaster())->GetSpawner())
178  if (spawner->IsPlayer() && target->IsPlayer())
179  if (static_cast<Player*>(spawner)->GetTeam() != static_cast<Player*>(target)->GetTeam())
180  return; // return of own flag already handled
181  }
182 
183  // misusing original caster to pass along original flag GO - if in future conflicts, substitute it for something else
184  SpellCastResult result = target->CastSpell(target, spellId, TRIGGERED_IGNORE_GCD | TRIGGERED_HIDE_CAST_IN_COMBAT_LOG | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL, nullptr, nullptr, spell->GetTrueCaster()->GetObjectGuid());
185  if (result == SPELL_CAST_OK) // wotlk+ done like this
186  StartEvents_Event(spell->GetTrueCaster()->GetMap(), eventId, target, spell->GetTrueCaster(), true);
187  }
