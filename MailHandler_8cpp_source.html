44 bool WorldSession::CheckMailBox(ObjectGuid guid) const
45 {
46  // GM case
47  if (guid == GetPlayer()->GetObjectGuid())
48  {
49  // command case will return only if player have real access to command
50  if (!ChatHandler(GetPlayer()).FindCommand("mailbox"))
51  {
52  DEBUG_LOG("%s attempt open mailbox in cheating way.", guid.GetString().c_str());
53  return false;
54  }
55  }
56  // mailbox case
57  else if (guid.IsGameObject())
58  {
59  if (!GetPlayer()->GetGameObjectIfCanInteractWith(guid, GAMEOBJECT_TYPE_MAILBOX))
60  {
61  DEBUG_LOG("Mailbox %s not found or %s can't interact with him.", guid.GetString().c_str(), GetPlayer()->GetGuidStr().c_str());
62  return false;
63  }
64  }
65  // squire case
66  else if (guid.IsAnyTypeCreature())
67  {
68  Creature* creature = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_NONE);
69  if (!creature)
70  {
71  DEBUG_LOG("%s not found or %s can't interact with him.", guid.GetString().c_str(), GetPlayer()->GetGuidStr().c_str());
72  return false;
73  }
74 
75  if (!(creature->GetCreatureInfo()->CreatureTypeFlags & CREATURE_TYPEFLAGS_SQUIRE))
76  {
77  DEBUG_LOG("%s not have access to mailbox.", creature->GetGuidStr().c_str());
78  return false;
79  }
80 
81  if (creature->GetOwnerGuid() != GetPlayer()->GetObjectGuid())
82  {
83  DEBUG_LOG("%s not owned by %s for access to mailbox.", creature->GetGuidStr().c_str(), GetPlayer()->GetGuidStr().c_str());
84  return false;
85  }
86  }
87  else
88  return false;
89 
90  return true;
91 }

110 void WorldSession::HandleSendMail(WorldPacket& recv_data)

113  uint64 unk3;
114  std::string receiver, subject, body;
115  uint32 unk1, unk2, money, COD;
116  uint8 unk4;
117  recv_data >> mailboxGuid;
118  recv_data >> receiver;
119 
120  recv_data >> subject;
121 
122  recv_data >> body;
123 
124  recv_data >> unk1; // stationery?
125  recv_data >> unk2; // 0x00000000
126 
127  uint8 items_count;
128  recv_data >> items_count; // attached items count
129 
130  if (items_count > MAX_MAIL_ITEMS) // client limit
131  {
132  GetPlayer()->SendMailResult(0, MAIL_SEND, MAIL_ERR_TOO_MANY_ATTACHMENTS);
133  recv_data.rpos(recv_data.wpos()); // set to end to avoid warnings spam
134  return;
135  }
136 
137  ObjectGuid itemGuids[MAX_MAIL_ITEMS];
138 
139  for (uint8 i = 0; i < items_count; ++i)
140  {
141  recv_data.read_skip<uint8>(); // item slot in mail, not used
142  recv_data >> itemGuids[i];
143  }
144 
145  recv_data >> money >> COD; // money and cod
146  recv_data >> unk3; // const 0
147  recv_data >> unk4; // const 0
148 
149  // packet read complete, now do check
150 
151  if (!CheckMailBox(mailboxGuid))
152  return;
153 
154  if (receiver.empty())
155  return;
156 
157  Player* pl = _player;
158

172  pl->GetGuidStr().c_str(), rc.GetString().c_str(), subject.c_str(), body.c_str(), items_count, money, COD, unk1, unk2);
173 
174  if (pl->GetObjectGuid() == rc)
175  {
176  pl->SendMailResult(0, MAIL_SEND, MAIL_ERR_CANNOT_SEND_TO_SELF);
177  return;
178  }
179

184  return;
185  }

188

190 
191  if (pl->GetMoney() < reqmoney)
192  {

194  return;
195  }

198 
199  Team rc_team;
200  uint8 mails_count = 0; // do not allow to send to one player more than 100 mails
201 
202  if (receive)
203  {
204  rc_team = receive->GetTeam();

206  }
207  else
208  {
209  rc_team = sObjectMgr.GetPlayerTeamByGUID(rc);
210  if (QueryResult* result = CharacterDatabase.PQuery("SELECT COUNT(*) FROM mail WHERE receiver = '%u'", rc.GetCounter()))
211  {

215  }

227  {

229  return;

237

239  {

241  {
242  pl->SendMailResult(0, MAIL_SEND, MAIL_ERR_MAIL_ATTACHMENT_INVALID);
243  return;
244  }

253  }
254

259  }
260

283 
284  pl->SendMailResult(0, MAIL_SEND, MAIL_OK);
285 
286  if (GetAnticheat()->IsSilenced())
287  return pl->ModifyMoney(-int32(cost));
288 
289  pl->ModifyMoney(-int32(reqmoney));
290  pl->GetAchievementMgr().UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GOLD_SPENT_FOR_MAIL, cost);
291 
292  bool needItemDelay = false;
293 
294  MailDraft draft(subject, body);
295 
296  if (items_count > 0 || money > 0)
297  {
298  if (items_count > 0)
299  {
300  for (uint8 i = 0; i < items_count; ++i)
301  {
302  Item* item = items[i];
303  if (GetSecurity() > SEC_PLAYER && sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
304  {
305  sLog.outCommand(GetAccountId(), "GM %s (Account: %u) mail item: %s (Entry: %u Count: %u) to player: %s (Account: %u)",
306  GetPlayerName(), GetAccountId(), item->GetProto()->Name1, item->GetEntry(), item->GetCount(), receiver.c_str(), rc_account);
307  }
308 
309  pl->MoveItemFromInventory(items[i]->GetBagSlot(), item->GetSlot(), true);
310  CharacterDatabase.BeginTransaction();
311  item->DeleteFromInventoryDB(); // deletes item from character's inventory
312  item->SaveToDB(); // recursive and not have transaction guard into self, item not in inventory and can be save standalone
313  // owner in data will set at mail receive and item extracting
314  CharacterDatabase.PExecute("UPDATE item_instance SET owner_guid = '%u' WHERE guid='%u'", rc.GetCounter(), item->GetGUIDLow());
315  CharacterDatabase.CommitTransaction();
316 
317  draft.AddItem(item);
318  }
319 
320  // if item send to character at another account, then apply item delivery delay
321  needItemDelay = pl->GetSession()->GetAccountId() != rc_account;
322  }
323 
324  if (money > 0 && GetSecurity() > SEC_PLAYER && sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
325  {
326  sLog.outCommand(GetAccountId(), "GM %s (Account: %u) mail money: %u to player: %s (Account: %u)",
327  GetPlayerName(), GetAccountId(), money, receiver.c_str(), rc_account);
328  }
329  }
330 
331  // If theres is an item, there is a one hour delivery delay if sent to another account's character.
332  uint32 deliver_delay = needItemDelay ? sWorld.getConfig(CONFIG_UINT32_MAIL_DELIVERY_DELAY) : 0;
333 
334  // will delete item or place to receiver mail list
335  draft
336  .SetMoney(money)
337  .SetCOD(COD)
338  .SendMailTo(MailReceiver(receive, rc), pl, body.empty() ? MAIL_CHECK_MASK_COPIED : MAIL_CHECK_MASK_HAS_BODY, deliver_delay);
339 
340  CharacterDatabase.BeginTransaction();
341  pl->SaveInventoryAndGoldToDB();
342  CharacterDatabase.CommitTransaction();
343 }

468  }
469 
470  draft.SetMoney(m->money).SendReturnToSender(GetAccountId(), m->receiverGuid, ObjectGuid(HIGHGUID_PLAYER, m->sender));
471  }
472 
473  delete m; // we can deallocate old mail
474  pl->SendMailResult(mailId, MAIL_RETURNED_TO_SENDER, MAIL_OK);

615 void WorldSession::HandleGetMailList(WorldPacket& recv_data)
616 {
617  ObjectGuid mailboxGuid;
618  recv_data >> mailboxGuid;
619 
620  if (!CheckMailBox(mailboxGuid))
621  return;
622 
623  uint32 mailsCount = 0; // send to client mails amount
624  uint32 realCount = 0; // real mails amount
625 
626  WorldPacket data(SMSG_MAIL_LIST_RESULT, 200); // guess size
627  data << uint32(0); // real mail's count
628  data << uint8(0); // mail's count
629  time_t cur_time = time(nullptr);
630 
631  for (PlayerMails::iterator itr = _player->GetMailBegin(); itr != _player->GetMailEnd(); ++itr)
632  {
633  // prevent client storage overflow
634  if (mailsCount >= MAX_INBOX_CLIENT_UI_CAPACITY)
635  {
636  realCount += 1;
637  continue;
638  }
639 
640  // skip deleted or not delivered (deliver delay not expired) mails
641  if ((*itr)->state == MAIL_STATE_DELETED || cur_time < (*itr)->deliver_time)
642  continue;
643 
644  uint8 item_count = uint8((*itr)->items.size()); // max count is MAX_MAIL_ITEMS (12)
645 
646  size_t next_mail_size = 2 + 4 + 1 + ((*itr)->messageType == MAIL_NORMAL ? 8 : 4) + 4 * 8 + ((*itr)->subject.size() + 1) + ((*itr)->body.size() + 1) + 1 + item_count * (1 + 4 + 4 + 7 * 3 * 4 + 4 + 4 + 4 + 4 + 4 + 4 + 1);
647 
648  if (data.wpos() + next_mail_size > MAX_NETCLIENT_PACKET_SIZE)
649  {
650  realCount += 1;
651  continue;
652  }
653 
654  data << uint16(next_mail_size); // Message size
655  data << uint32((*itr)->messageID); // Message ID
656  data << uint8((*itr)->messageType); // Message Type
657 
658  switch ((*itr)->messageType)
659  {
660  case MAIL_NORMAL: // sender guid
661  data << ObjectGuid(HIGHGUID_PLAYER, (*itr)->sender);
662  break;
663  case MAIL_CREATURE:
664  case MAIL_GAMEOBJECT:
665  case MAIL_AUCTION:
666  data << uint32((*itr)->sender); // creature/gameobject entry, auction id
667  break;
668  case MAIL_CALENDAR:
669  data << uint32(0); // unknown
670  break;
671  }
672 
673  data << uint32((*itr)->COD); // COD
674  data << uint32(0); // unknown, probably changed in 3.3.3
675  data << uint32((*itr)->stationery); // stationery (Stationery.dbc)
676  data << uint32((*itr)->money); // copper
677  data << uint32((*itr)->checked); // flags
678  data << float(float((*itr)->expire_time - time(nullptr)) / float(DAY));// Time
679  data << uint32((*itr)->mailTemplateId); // mail template (MailTemplate.dbc)
680  data << (*itr)->subject; // Subject string - once 00, when mail type = 3, max 256
681  data << (*itr)->body; // message? max 8000
682 
683  data << uint8(item_count); // client limit is 0x10
684 
685  for (uint8 i = 0; i < item_count; ++i)
686  {
687  Item* item = _player->GetMItem((*itr)->items[i].item_guid);
688  // item index (0-6?)
689  data << uint8(i);
690  // item guid low?
691  data << uint32(item ? item->GetGUIDLow() : 0);
692  // entry
