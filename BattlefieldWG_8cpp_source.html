199  if (GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS && m_zonePlayers.empty())
200  GetBattlefieldReady(player);
201 
202  Battlefield::HandlePlayerEnterZone(player, isMainZone);
203 
204  // phase each player based on the defender (if there is one)
205  if (GetDefender() != TEAM_NONE)
206  player->CastSpell(player, wgTeamControlAuras[GetTeamIndexByTeamId(GetDefender())], TRIGGERED_OLD_TRIGGERED);
207 
208  // remove all buff auras first; will be added back shortly
209  player->RemoveAurasDueToSpell(SPELL_ESSENCE_WINTERGRASP_ZONE);
210  player->RemoveAurasDueToSpell(SPELL_TOWER_CONTROL);
211  player->RemoveAurasDueToSpell(SPELL_RECRUIT);
212  player->RemoveAurasDueToSpell(SPELL_CORPORAL);
213  player->RemoveAurasDueToSpell(SPELL_LIEUTENANT);
214 
215  // defenders are phased and get increase XP
216  if (GetBattlefieldStatus() == BF_STATUS_COOLDOWN && GetDefender() == player->GetTeam())
217  player->CastSpell(player, SPELL_ESSENCE_WINTERGRASP_ZONE, TRIGGERED_OLD_TRIGGERED);
218 
219  // flight restriction - applied based on AREA_FLAG_CANNOT_FLY in Player::UpdateArea
220 
221  // set the tower control buff
222  if (GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS)
223  {
224  uint32 maxBuff = player->GetTeam() == GetAttacker() ? MAX_WG_OFFENSE_TOWERS - m_destroyedTowers[GetTeamIndexByTeamId(GetAttacker())] : m_destroyedTowers[GetTeamIndexByTeamId(GetAttacker())];
225  for (uint8 i = 0; i < maxBuff; ++i)
226  player->CastSpell(player, SPELL_TOWER_CONTROL, TRIGGERED_OLD_TRIGGERED);

521 bool BattlefieldWG::HandleEvent(uint32 eventId, Object* source, Object* target)
522 {
523  if (!source->IsGameObject())
524  return false;
525 
526  GameObject* go = static_cast<GameObject*>(source);
527 
528  if (GetDefender() == TEAM_NONE)
529  {
530  sLog.outError("Battlefield WG: Gameobject events cannot be handled when there is no zone defender.");
531  return true;
532  }
533 
534  // no events handled during cooldown
535  if (GetBattlefieldStatus() == BF_STATUS_COOLDOWN)
536  return true;
537 
538  bool returnValue = false;
539 
540  // handle capture points
541  if (go->GetGoType() == GAMEOBJECT_TYPE_CAPTURE_POINT)
542  returnValue = HandleCapturePointEvent(eventId, go);
543  // handle destructible buildings
544  else if (go->GetGoType() == GAMEOBJECT_TYPE_DESTRUCTIBLE_BUILDING && target && target->IsUnit())
545  returnValue = HandleDestructibleBuildingEvent(eventId, go, static_cast<Unit*>(target));
546 
547  return returnValue;
548 }
