30 // Convert the geomoetry from a spline value, to an actual world XYZ
31 G3D::Vector3 TranslateLocation(G3D::Vector4 const* DBCPosition, G3D::Vector3 const* basePosition, G3D::Vector3 const* splineVector)
32 {
33  G3D::Vector3 work;
34  float x = basePosition->x + splineVector->x;
35  float y = basePosition->y + splineVector->y;
36  float z = basePosition->z + splineVector->z;
37  float const distance = sqrt((x * x) + (y * y));
38  float angle = std::atan2(x, y) - DBCPosition->w;
39 
40  if (angle < 0)
41  angle += 2 * float(M_PI);
42 
43  work.x = DBCPosition->x + (distance * sin(angle));
44  work.y = DBCPosition->y + (distance * cos(angle));
45  work.z = DBCPosition->z + z;
46  return work;
47 }

50 bool readCamera(M2Camera const* cam, uint32 buffSize, M2Header const* header, CinematicCameraEntry const* dbcentry)
51 {
52  char const* buffer = reinterpret_cast<char const*>(header);
53 
54  FlyByCameraCollection cameras;
55  FlyByCameraCollection targetcam;
56 
57  G3D::Vector4 DBCData;
58  DBCData.x = dbcentry->Origin.X;
59  DBCData.y = dbcentry->Origin.Y;
60  DBCData.z = dbcentry->Origin.Z;
61  DBCData.w = dbcentry->OriginFacing;
62 
63  // Read target locations, only so that we can calculate orientation
64  for (uint32 k = 0; k < cam->target_positions.timestamps.number; ++k)
65  {
66  // Extract Target positions
67  if (cam->target_positions.timestamps.offset_elements + sizeof(M2Array) > buffSize)
68  return false;
69  M2Array const* targTsArray = reinterpret_cast<M2Array const*>(buffer + cam->target_positions.timestamps.offset_elements);
70  if (targTsArray->offset_elements + sizeof(uint32) > buffSize || cam->target_positions.values.offset_elements + sizeof(M2Array) > buffSize)
71  return false;
72  uint32 const* targTimestamps = reinterpret_cast<uint32 const*>(buffer + targTsArray->offset_elements);
73  M2Array const* targArray = reinterpret_cast<M2Array const*>(buffer + cam->target_positions.values.offset_elements);
74 
75  if (targArray->offset_elements + sizeof(M2SplineKey<G3D::Vector3>) > buffSize)
76  return false;
77  M2SplineKey<G3D::Vector3> const* targPositions = reinterpret_cast<M2SplineKey<G3D::Vector3> const*>(buffer + targArray->offset_elements);
78 
79  // Read the data for this set
80  uint32 currPos = targArray->offset_elements;
81  for (uint32 i = 0; i < targTsArray->number; ++i)
82  {
83  if (currPos + sizeof(M2SplineKey<G3D::Vector3>) > buffSize)
84  return false;
85  // Translate co-ordinates
86  G3D::Vector3 newPos = TranslateLocation(&DBCData, &cam->target_position_base, &targPositions->p0);
87 
88  // Add to vector
89  FlyByCamera thisCam;
90  thisCam.timeStamp = targTimestamps[i];
91  thisCam.locations.x = newPos.x;
92  thisCam.locations.y = newPos.y;
93  thisCam.locations.z = newPos.z;
94  thisCam.locations.w = 0.0f;
95  targetcam.push_back(thisCam);
96  targPositions++;
97  currPos += sizeof(M2SplineKey<G3D::Vector3>);
98  }
99  }
100 
101  // Read camera positions and timestamps (translating first position of 3 only, we don't need to translate the whole spline)
102  for (uint32 k = 0; k < cam->positions.timestamps.number; ++k)
103  {
104  // Extract Camera positions for this set
105  if (cam->positions.timestamps.offset_elements + sizeof(M2Array) > buffSize)
106  return false;
107  M2Array const* posTsArray = reinterpret_cast<M2Array const*>(buffer + cam->positions.timestamps.offset_elements);
108  if (posTsArray->offset_elements + sizeof(uint32) > buffSize || cam->positions.values.offset_elements + sizeof(M2Array) > buffSize)
109  return false;
110  uint32 const* posTimestamps = reinterpret_cast<uint32 const*>(buffer + posTsArray->offset_elements);
111  M2Array const* posArray = reinterpret_cast<M2Array const*>(buffer + cam->positions.values.offset_elements);
112  if (posArray->offset_elements + sizeof(M2SplineKey<G3D::Vector3>) > buffSize)
113  return false;
114  M2SplineKey<G3D::Vector3> const* positions = reinterpret_cast<M2SplineKey<G3D::Vector3> const*>(buffer + posArray->offset_elements);
115 
116  // Read the data for this set
117  uint32 currPos = posArray->offset_elements;
118  for (uint32 i = 0; i < posTsArray->number; ++i)
119  {
120  if (currPos + sizeof(M2SplineKey<G3D::Vector3>) > buffSize)
121  return false;
122  // Translate co-ordinates
123  G3D::Vector3 newPos = TranslateLocation(&DBCData, &cam->position_base, &positions->p0);
124 
125  // Add to vector
126  FlyByCamera thisCam;
127  thisCam.timeStamp = posTimestamps[i];
128  thisCam.locations.x = newPos.x;
129  thisCam.locations.y = newPos.y;
130  thisCam.locations.z = newPos.z;
131 
132  if (targetcam.size() > 0)
133  {
134  // Find the target camera before and after this camera
135  FlyByCamera lastTarget;
136  FlyByCamera nextTarget;
137 
138  // Pre-load first item
139  lastTarget = targetcam[0];
140  nextTarget = targetcam[0];
141  for (uint32 j = 0; j < targetcam.size(); ++j)
142  {
143  nextTarget = targetcam[j];
144  if (targetcam[j].timeStamp > posTimestamps[i])
145  break;
146 
147  lastTarget = targetcam[j];
148  }
149 
150  float x = lastTarget.locations.x;
151  float y = lastTarget.locations.y;
152  float z = lastTarget.locations.z;
153 
154  // Now, the timestamps for target cam and position can be different. So, if they differ we interpolate
155  if (lastTarget.timeStamp != posTimestamps[i])
156  {
157  uint32 timeDiffTarget = nextTarget.timeStamp - lastTarget.timeStamp;
158  uint32 timeDiffThis = posTimestamps[i] - lastTarget.timeStamp;
159  float xDiff = nextTarget.locations.x - lastTarget.locations.x;
160  float yDiff = nextTarget.locations.y - lastTarget.locations.y;
161  float zDiff = nextTarget.locations.z - lastTarget.locations.z;
162  x = lastTarget.locations.x + (xDiff * (float(timeDiffThis) / float(timeDiffTarget)));
163  y = lastTarget.locations.y + (yDiff * (float(timeDiffThis) / float(timeDiffTarget)));
164  z = lastTarget.locations.z + (zDiff * (float(timeDiffThis) / float(timeDiffTarget)));
165  }
166  float xDiff = x - thisCam.locations.x;
167  float yDiff = y - thisCam.locations.y;
168  thisCam.locations.w = std::atan2(yDiff, xDiff);
169  }
170 
171  cameras.push_back(thisCam);
172  positions++;
173  currPos += sizeof(M2SplineKey<G3D::Vector3>);
174  }
175  }
176 
177  sFlyByCameraStore[dbcentry->ID] = cameras;
178  return true;
179 }

181 void LoadM2Cameras(std::string const& dataPath)
182 {
183  sFlyByCameraStore.clear();
184 
185  uint32 oldMSTime = WorldTimer::getMSTime();
186  for (uint32 i = 0; i < sCinematicCameraStore.GetNumRows(); ++i)
187  {
188  if (CinematicCameraEntry const* dbcentry = sCinematicCameraStore.LookupEntry(i))
189  {
190  std::string filename = dataPath;
191  filename.append(dbcentry->Model);
192 
193  // Replace slashes
194  std::replace(filename.begin(), filename.end(), '\\', '/');
195 
196  // Replace mdx to .m2
197  size_t loc = filename.find(".mdx");
198  if (loc != std::string::npos)
199  filename.replace(loc, 4, ".m2");
200 
201  std::ifstream m2file(filename.c_str(), std::ios::in | std::ios::binary);
202  if (!m2file.is_open())
203  continue;
204 
205  // Get file size
206  m2file.seekg(0, std::ios::end);
207  std::streamoff const fileSize = m2file.tellg();
208 
209  // Reject if not at least the size of the header
210  if (static_cast<uint32 const>(fileSize) < sizeof(M2Header))
211  {
212  sLog.outError("Camera file %s is damaged. File is smaller than header size", filename.c_str());
213  m2file.close();
214  continue;
215  }
216 
217  // Read 4 bytes (signature)
218  m2file.seekg(0, std::ios::beg);
219  char fileCheck[5];
220  m2file.read(fileCheck, 4);
221  fileCheck[4] = 0;
222 
223  // Check file has correct magic (MD20)
224  if (strcmp(fileCheck, "MD20"))
225  {
226  sLog.outError("Camera file %s is damaged. File identifier not found", filename.c_str());
227  m2file.close();
228  continue;
229  }
230 
231  // Now we have a good file, read it all into a vector of char's, then close the file.
232  std::vector<char> buffer(fileSize);
233  m2file.seekg(0, std::ios::beg);
234  if (!m2file.read(buffer.data(), fileSize))
235  {
236  m2file.close();
237  continue;
238  }
239  m2file.close();
240 
241  // Read header
242  M2Header const* header = reinterpret_cast<M2Header const*>(buffer.data());
243 
244  if (header->ofsCameras + sizeof(M2Camera) > static_cast<uint32 const>(fileSize))
245  {
246  sLog.outError("Camera file %s is damaged. Camera references position beyond file end (header)", filename.c_str());
247  continue;
248  }
249 
250  // Get camera(s) - Main header, then dump them.
251  M2Camera const* cam = reinterpret_cast<M2Camera const*>(buffer.data() + header->ofsCameras);
252  if (!readCamera(cam, fileSize, header, dbcentry))
253  sLog.outError("Camera file %s is damaged. Camera references position beyond file end (camera)", filename.c_str());
254  }
255  }
256  sLog.outString(">> Loaded %u cinematic waypoint sets in %u ms", static_cast<uint32 const>(sFlyByCameraStore.size()), WorldTimer::getMSTimeDiff(oldMSTime, WorldTimer::getMSTime()));
257  sLog.outString();
258 }
