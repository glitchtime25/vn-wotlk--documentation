167  void UpdateAI(const uint32 uiDiff) override
168  {
169  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
170  return;
171 
172  if (!m_bGuardianSummoned && m_creature->GetHealthPercent() < 50.0f)
173  {
174  // guardian is summoned at 50% of boss HP
175  if (m_pInstance)
176  {
177  if (Creature* pGuardianEgg = m_creature->GetMap()->GetCreature(m_pInstance->SelectRandomGuardianEggGuid()))
178  pGuardianEgg->CastSpell(pGuardianEgg, SPELL_SUMMON_SWARM_GUARDIAN, TRIGGERED_NONE);
179 
180  m_bGuardianSummoned = true;
181  }
182  }
183 
184  if (m_uiSummonTimer < uiDiff)
185  {
186  if (roll_chance_i(50))
187  DoScriptText(urand(0, 1) ? SAY_SUMMON_EGG_1 : SAY_SUMMON_EGG_2, m_creature);
188 
189  if (m_pInstance)
190  {
191  // There are 2 Swarmers summoned at a timer
192  if (Creature* pSwarmerEgg = m_creature->GetMap()->GetCreature(m_pInstance->SelectRandomSwarmerEggGuid()))
193  {
194  for (uint8 i = 0; i < 2; ++i)
195  pSwarmerEgg->CastSpell(pSwarmerEgg, SPELL_SUMMON_SWARMERS, TRIGGERED_NONE);
196  }
197  }
198 
199  m_uiSummonTimer = urand(5000, 10000);
200  }
201  else
202  m_uiSummonTimer -= uiDiff;
203 
204  if (m_uiBroodPlagueTimer < uiDiff)
205  {
206  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
207  DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_BROOD_PLAGUE : SPELL_BROOD_PLAGUE_H);
208 
209  m_uiBroodPlagueTimer = 20000;
210  }
211  else
212  m_uiBroodPlagueTimer -= uiDiff;
213 
214  if (!m_bIsRegularMode)
215  {
216  if (m_uiBroodRageTimer < uiDiff)
217  {
218  if (DoCastSpellIfCan(m_creature, SPELL_BROOD_RAGE) == CAST_OK)
219  m_uiBroodRageTimer = 20000;
220  }
221  else
222  m_uiBroodRageTimer -= uiDiff;
223  }
224 
225  if (!m_bBerserk && m_creature->GetPositionZ() < 24.0)
226  {
227  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
228  m_bBerserk = true;
229  }
230 
231  DoMeleeAttackIfReady();
232  }
233 };
