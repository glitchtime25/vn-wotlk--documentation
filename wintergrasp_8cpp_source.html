120  go_vehicle_teleporter(GameObject* go) : GameObjectAI(go)

130  uint32 m_uiTeleportTimer;
131 
132  // Return the teleport destination target
133  Unit* GetTeleportTarget(Creature* source)
134  {
135  // The destination trigger is between 64 and 69 away; no more than 70
136  CreatureList lTriggersInRange;
137  GetCreatureListWithEntryInGrid(lTriggersInRange, m_gameobject, NPC_WORLD_TRIGGER, 70.0f);
138 
139  if (lTriggersInRange.empty())
140  return nullptr;
141 
142  lTriggersInRange.sort(ObjectDistanceOrderReversed(source));
143  return lTriggersInRange.front();
144  }
145 
146  void UpdateAI(const uint32 uiDiff) override
147  {
148  if (m_uiTeleportTimer < uiDiff)
149  {
150  m_uiTeleportTimer = 1000;
151 
152  if (!m_gameobject)
153  return;
154 
155  if (!m_battlefield)
156  {
157  // attempt to initialize the opvp
158  OutdoorPvP* outdoorPvP = sOutdoorPvPMgr.GetScript(m_gameobject->GetZoneId());
159  if (outdoorPvP && outdoorPvP->IsBattlefield())
160  m_battlefield = (Battlefield*)outdoorPvP;
161  else
162  return;
163  }
164 
165  if (m_battlefield->GetBattlefieldStatus() == BF_STATUS_COOLDOWN)
166  return;
167 
168  for (uint8 i = 0; i < MAX_WINTERGRASP_VEHICLES; ++i)
169  {
170  // Note: the object has a very large range; might need to be corrected in DB
171  if (Creature* vehicle = GetClosestCreatureWithEntry(m_gameobject, aWintergraspVehicles[i], 0.1f))
172  {
173  if (vehicle->HasAura(SPELL_TELEPORT_VEHICLE))
174  continue;
175 
176  // validate vehicle ownership against the flag it carries
177  if ((m_battlefield->GetDefender() == ALLIANCE && vehicle->HasAura(SPELL_ALLIANCE_FLAG)) ||
178  (m_battlefield->GetDefender() == HORDE && vehicle->HasAura(SPELL_HORDE_FLAG)))
179  {
180  // teleport to the trigger which is furthest away
181  if (Unit* target = GetTeleportTarget(vehicle))
182  vehicle->CastSpell(target, SPELL_TELEPORT_VEHICLE, TRIGGERED_OLD_TRIGGERED);
183  }
184  }
185  }
186  }
187  else
188  m_uiTeleportTimer -= uiDiff;
189  }
190 };

203 {
204  if (bIsStart && pSource->GetTypeId() == TYPEID_GAMEOBJECT)

210 
211  CreatureList lCannonsInRange;
212  GetCreatureListWithEntryInGrid(lCannonsInRange, tower, NPC_WINTERGRASP_TOWER_CANNON, 50.0f);
213  for (auto cannon : lCannonsInRange)
214  cannon->ForcedDespawn();
215  }
216 
217  // always return false to allow battlefield script to process
218  return false;

225 struct spell_teleport_lake_wintergrasp : public SpellScript
226 {
227  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
228  {
229  if (effIdx != EFFECT_INDEX_0)
230  return;
231 
232  Unit* target = spell->GetUnitTarget();
233  if (!target || !target->IsPlayer())
234  return;
235 
236  OutdoorPvP* outdoorPvP = sOutdoorPvPMgr.GetScript(ZONE_ID_WINTERGRASP);
237  if (outdoorPvP && outdoorPvP->IsBattlefield())
238  {
239  Player* player = static_cast<Player*>(target);
240  Battlefield* battlefield = static_cast<Battlefield*>(outdoorPvP);
241 
242  if (!player || !battlefield)
243  return;
244 
245  // teleport to Wintergrasp, depending on location and battlefield status
246  if (battlefield->GetBattlefieldStatus() == BF_STATUS_IN_PROGRESS || battlefield->GetTimer() < battlefield->GetStartInviteDelay())
247  player->CastSpell(player, player->GetTeam() == battlefield->GetDefender() ? 60035 : 59096, TRIGGERED_OLD_TRIGGERED);
248  if (battlefield->GetBattlefieldStatus() == BF_STATUS_COOLDOWN)
249  {
250  if (player->GetTeam() == battlefield->GetDefender())
251  player->CastSpell(player, 58681, TRIGGERED_OLD_TRIGGERED);
252  else
253  player->CastSpell(player, player->GetTeam() == ALLIANCE ? 58633 : 58632, TRIGGERED_OLD_TRIGGERED);
254  }
255  }
256  }
257 };
