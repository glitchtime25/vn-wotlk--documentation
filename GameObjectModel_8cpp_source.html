42 bool GameObjectModel::initialize(const GameObject* const pGo, const GameObjectDisplayInfoEntry* const pDisplayInfo)
43 {
44  ModelList::const_iterator it = modelList.find(pDisplayInfo->Displayid);
45  if (it == modelList.end())
46  return false;
47 
48  G3D::AABox mdl_box(it->second.bound);
49  // ignore models with no bounds
50  if (mdl_box == G3D::AABox::zero())
51  {
52  sLog.outDebug("Model %s has zero bounds, loading skipped", it->second.name.c_str());
53  return false;
54  }
55 
56  iModel = ((VMAP::VMapManager2*)VMAP::VMapFactory::createOrGetVMapManager())->acquireModelInstance(sWorld.GetDataPath() + "vmaps/", it->second.name);
57 
58  if (!iModel)
59  return false;
60 
61  if (it->second.name.find(".m2") != std::string::npos)
62  iModel->setModelFlags(VMAP::MOD_M2);
63 
64  name = it->second.name;
65  iPos = Vector3(pGo->GetPositionX(), pGo->GetPositionY(), pGo->GetPositionZ());
66  phasemask = pGo->GetPhaseMask();
67  iScale = pGo->GetObjectScale();
68  iInvScale = 1.f / iScale;
69 
70  G3D::Matrix3 iRotation = G3D::Matrix3::fromEulerAnglesZYX(pGo->GetOrientation(), 0, 0);
71  iInvRot = iRotation.inverse();
72  // transform bounding box:
73  mdl_box = AABox(mdl_box.low() * iScale, mdl_box.high() * iScale);
74  AABox rotated_bounds;
75  for (int i = 0; i < 8; ++i)
76  rotated_bounds.merge(iRotation * mdl_box.corner(i));
77 
78  this->iBound = rotated_bounds + iPos;
79 
80 #ifdef SPAWN_CORNERS

112 bool GameObjectModel::intersectRay(const G3D::Ray& ray, float& MaxDist, bool StopAtFirstHit, uint32 phaseMask, bool ignoreM2Model) const
113 {
114  if (!(phasemask & phaseMask))
115  return false;
116 
117  float time = ray.intersectionTime(iBound);
118  if (time == G3D::inf())
119  return false;
120 
121  // child bounds are defined in object space:
122  Vector3 p = iInvRot * (ray.origin() - iPos) * iInvScale;
123  Ray modRay(p, iInvRot * ray.direction());
124  float distance = MaxDist * iInvScale;
125  bool hit = iModel->IntersectRay(modRay, distance, StopAtFirstHit, ignoreM2Model);
126  if (hit)
127  {
128  distance *= iScale;
129  MaxDist = distance;
130  }
131  return hit;
132 }

175 void GameObjectModel::LoadGOVmapModels()
176 {
177  modelList = GameobjectModelData::LoadGameObjectModelList(sWorld.GetDataPath() + "vmaps/" + VMAP::GAMEOBJECT_MODELS);
178  std::set<std::tuple<uint32, uint32, uint32>> usedTilesPerMap;
179  for (auto& data : BuildingMap)
180  {
181  uint32 mapId = data.first;
182  for (TileBuilding& building : data.second)
183  {
184  ModelList::const_iterator itr = modelList.find(building.displayId);
185  if (itr == modelList.end())
186  continue;
187 
188  uint32 lowX, lowY, highX, highY;
189  std::tie(lowX, lowY, highX, highY) = GameobjectModelData::CalculateBuildingTiles(building, itr->second.bound);
190  for (uint32 x = lowX; x <= highX; ++x)
191  {
192  for (uint32 y = lowY; y <= highY; ++y)
193  {
194  usedTilesPerMap.insert(std::make_tuple(mapId, x, y));
195  tilesPerGoEntryInMap[std::make_pair(mapId, building.goEntry)].push_back(std::make_pair(x, y));
196  }
197  }
198  }
199  }
200 
201  for (auto& mapTileData : usedTilesPerMap)
202  {
203  uint32 mapId, tileX, tileY;
204  std::tie(mapId, tileX, tileY) = mapTileData;
205  std::vector<TileBuilding const*> buildingsByDefault;
206  std::map<uint32, std::vector<TileBuilding const*>> buildingsInTile;
207  std::map<uint32, std::vector<TileBuilding const*>> buildingsByGroup;
208  std::map<uint32, uint32> flagToGroup;
209 
210  // yes I am reusing a function that is computationally expensive - but we do it for a trivial amount of tiles
211  std::tie(buildingsByDefault, buildingsInTile, buildingsByGroup, flagToGroup) = GameobjectModelData::GetTileBuildingData(mapId, tileX, tileY, modelList);
212  // case of tile numbers - no flag usage
213  for (auto& data : buildingsInTile)
214  for (auto& building : data.second)
215  goDisplayIdTileData[std::make_tuple(mapId, building->goEntry, building->displayId, tileX, tileY)] = std::make_pair(false, data.first);
216 
217  for (auto& data : buildingsByGroup)
218  for (auto& building : data.second)
219  goDisplayIdTileData[std::make_tuple(mapId, building->goEntry, building->displayId, tileX, tileY)] = std::make_pair(true, (1 << (data.first - 1)));
220  }
221 }
