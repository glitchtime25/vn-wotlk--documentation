208 void instance_mechanar::DoSpawnBridgeWave()
209 {
210  if (Player* pPlayer = GetPlayerInMap(true, false))
211  {
212  for (uint8 i = 0; i < MAX_BRIDGE_TRASH; ++i)
213  {
214  // Skip the blank entries
215  if (aBridgeEventLocs[m_uiBridgeEventPhase][i].m_uiSpawnEntry == 0)
216  break;
217 
218  if (Creature* pTemp = pPlayer->SummonCreature(aBridgeEventLocs[m_uiBridgeEventPhase][i].m_uiSpawnEntry, aBridgeEventLocs[m_uiBridgeEventPhase][i].m_fX, aBridgeEventLocs[m_uiBridgeEventPhase][i].m_fY, aBridgeEventLocs[m_uiBridgeEventPhase][i].m_fZ, aBridgeEventLocs[m_uiBridgeEventPhase][i].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0))
219  {
220  pTemp->CastSpell(pTemp, SPELL_ETHEREAL_TELEPORT, TRIGGERED_NONE);
221 
222  switch (m_uiBridgeEventPhase)
223  {
224  case 0:
225  SetData(TYPE_BRIDGEEVENT, IN_PROGRESS);
226  case 1:
227  case 2:
228  case 3:
229  case 4:
230  case 5:
231  pTemp->SetInCombatWithZone();
232  break;
233  case 6: // Pathaleon
234  SetData(TYPE_BRIDGEEVENT, DONE);
235  DoScriptText(SAY_PATHALEON_INTRO, pTemp);
236  m_uiPathaleonEngageTimer = 30000;
237  break;
238  }
239  }
240  }
241  }
242  ++m_uiBridgeEventPhase;
243 }

245 void instance_mechanar::Update(uint32 uiDiff)
246 {
247  if (m_uiBridgeEventTimer)
248  {
249  if (m_uiBridgeEventTimer <= uiDiff)
250  {
251  m_uiBridgeEventTimer = 2000;
252  for (const auto& data : instance->GetPlayers())
253  {
254  if (m_uiBridgeEventPhase == 0)
255  {
256  if (data.getSource()->GetPositionZ() > 26.3f && data.getSource()->GetPositionY() < -12.f && data.getSource()->GetPositionX() < 270.f)
257  {
258  DoSpawnBridgeWave();
259  }
260  }
261  else if (m_uiBridgeEventPhase == 3)
262  {
263  if (data.getSource()->GetPositionZ() > 24.5f && data.getSource()->GetPositionY() > 2.5f && data.getSource()->GetPositionX() < 155.f)
264  {
265  DoSpawnBridgeWave();
266  }
267  }
268  }
269  }
270  else
271  m_uiBridgeEventTimer -= uiDiff;
272  }
273 
274  if (m_uiPathaleonEngageTimer)
275  {
276  if (m_uiPathaleonEngageTimer <= uiDiff)
277  {
278  m_uiPathaleonEngageTimer = 0;
279  if (Creature* Pathaleon = GetSingleCreatureFromStorage(NPC_PATHALEON))
280  Pathaleon->SetInCombatWithZone();
281  }
282  else
283  m_uiPathaleonEngageTimer -= uiDiff;
284  }
