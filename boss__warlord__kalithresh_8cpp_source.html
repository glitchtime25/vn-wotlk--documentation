161  void MovementInform(uint32 uiType, uint32 uiPointId)
162  {
163  if (uiType != POINT_MOTION_TYPE || !uiPointId)
164  return;
165 
166  if (uiPointId == POINT_MOVE_DISTILLER)
167  {
168  SetCombatScriptStatus(false);
169  SetCombatMovement(true);
170  SetMeleeEnabled(true);
171  DoStartMovement(m_creature->GetVictim());
172  DoCastSpellIfCan(m_creature, SPELL_WARLORDS_RAGE);
173 
174  // Make Distiller cast on arrival
175  if (Creature* Distiller = m_creature->GetMap()->GetCreature(m_distillerGuid))
176  {
177  Distiller->CastSpell(Distiller, SPELL_WARLORDS_RAGE_NAGA, TRIGGERED_OLD_TRIGGERED);

183  void ExecuteActions()
184  {
185  if (!CanExecuteCombatAction())
186  return;
187 
188  for (uint32 i = 0; i < WARLORD_KALITHRESH_ACTION_MAX; ++i)
189  {
190  if (GetActionReadyStatus(i))
191  {
192  switch (i)
193  {
194  case WARLORD_KALITHRESH_ACTION_WARLORDS_RAGE:
195  {
196  DoScriptText(SAY_REGEN, m_creature);
197  SetCombatScriptStatus(true);
198  SetCombatMovement(false);
199  SetMeleeEnabled(false);
200  // Move to closest distiller
201  if (Creature* Distiller = GetClosestCreatureWithEntry(m_creature, NPC_NAGA_DISTILLER, 100.0f))
202  {
203  float fX, fY, fZ;
204  Distiller->GetContactPoint(m_creature, fX, fY, fZ, INTERACTION_DISTANCE);
205  m_creature->SetWalk(false, true); // Prevent him from slowing down while meleehit/casted upon while starting to move
206  m_creature->GetMotionMaster()->MovePoint(POINT_MOVE_DISTILLER, fX, fY, fZ);
207  m_distillerGuid = Distiller->GetObjectGuid();
208 
209  ResetTimer(i, GetSubsequentActionTimer(i));
210  SetActionReadyStatus(i, false);
211  return;
212  }
213  break;
214  }
215  case WARLORD_KALITHRESH_ACTION_REFLECTION:
216  {
217  if (DoCastSpellIfCan(m_creature, SPELL_SPELL_REFLECTION) == CAST_OK)
218  {
219  ResetTimer(i, GetSubsequentActionTimer(i));
220  SetActionReadyStatus(i, false);
221  return;
222  }
223  break;
224  }
225  case WARLORD_KALITHRESH_ACTION_IMPALE:
226  {
227  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_NOT_IN_MELEE_RANGE))
228  {
229  if (DoCastSpellIfCan(pTarget, SPELL_IMPALE) == CAST_OK)
230  {
231  ResetTimer(i, GetSubsequentActionTimer(i));
232  SetActionReadyStatus(i, false);
233  return;
234  }
235  }
236  break;
237  }
238  case WARLORD_KALITHRESH_ACTION_HEAD_CRACK:
239  {
240  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_HEAD_CRACK) == CAST_OK)
241  {
242  ResetTimer(i, GetSubsequentActionTimer(i));
243  SetActionReadyStatus(i, false);
244  return;
245  }
246  break;
247  }
248  }
249  }
250  }
251  }
