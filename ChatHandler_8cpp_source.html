40 bool WorldSession::CheckChatMessage(std::string& msg, bool addon/* = false*/)
41 {
42 #ifdef BUILD_PLAYERBOT
43  // bot check can be avoided
44  if (_player->GetPlayerbotAI())
45  return true;
46 #endif
47 
48  // check max length: as of 2.3.x+ no longer disconnects, silently truncates to 255 (wowwiki)
49  if (msg.length() > 255)
50  utf8limit(msg, 255);
51 
52  // skip remaining checks for addon messages or higher sec level accounts
53  if (addon || GetSecurity() > SEC_PLAYER)
54  return true;
55 
56  // strip invisible characters for non-addon messages
57  if (sWorld.getConfig(CONFIG_BOOL_CHAT_FAKE_MESSAGE_PREVENTING))
58  stripLineInvisibleChars(msg);
59 
60  // check links
61  if (sWorld.getConfig(CONFIG_UINT32_CHAT_STRICT_LINK_CHECKING_SEVERITY))
62  {
63  if (!ChatHandler::CheckEscapeSequences(msg.c_str()))
64  {
65  sLog.outError("Player %s (GUID: %u) sent a chat message with an invalid escape sequence: \"%s\"", GetPlayer()->GetName(),
66  GetPlayer()->GetGUIDLow(), msg.c_str());
67 
68  if (sWorld.getConfig(CONFIG_UINT32_CHAT_STRICT_LINK_CHECKING_KICK))
69  KickPlayer();
70 
71  return false;
72  }
73  }
74  return true;
75 }

92

95  if (!langDesc)
96  {
97  SendNotification(LANG_UNKNOWN_LANGUAGE);
98  return;
99  }
100  if (langDesc->skill_id != 0 && !_player->HasSkill(langDesc->skill_id))
101  {
102  // also check SPELL_AURA_COMPREHEND_LANGUAGE (client offers option to speak in that language)
103  Unit::AuraList const& langAuras = _player->GetAurasByType(SPELL_AURA_COMPREHEND_LANGUAGE);
104  bool foundAura = false;
105  for (auto langAura : langAuras)
106  {
107  if (langAura->GetModifier()->m_miscvalue == int32(lang))
108  {
109  foundAura = true;
110  break;
111  }
112  }
113  if (!foundAura)
114  {
115  SendNotification(LANG_NOT_LEARNED_LANGUAGE);
116  return;
117  }
118  }
119 
120  if (lang == LANG_ADDON)
121  {
122  // Disabled addon channel?
123  if (!sWorld.getConfig(CONFIG_BOOL_ADDON_CHANNEL))
124  return;
125  }
126  // LANG_ADDON should not be changed nor be affected by flood control
127  else
128  {
129  // send in universal language if player in .gmon mode (ignore spell effects)
130  if (_player->IsGameMaster())
131  lang = LANG_UNIVERSAL;
132  else
133  {
134  // send in universal language in two side iteration allowed mode
135  if (sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_INTERACTION_CHAT))
136  lang = LANG_UNIVERSAL;
137  else

139  switch (type)

142  case CHAT_MSG_RAID:
143  case CHAT_MSG_RAID_LEADER:
144  case CHAT_MSG_RAID_WARNING:

148  break;
149  case CHAT_MSG_GUILD:
150  case CHAT_MSG_OFFICER:

154  break;
155  }
156  }
157

159  Unit::AuraList const& ModLangAuras = _player->GetAurasByType(SPELL_AURA_MOD_LANGUAGE);

162  }
163 
164  if (type != CHAT_MSG_AFK && type != CHAT_MSG_DND)
165  {

175  }

584

586  }
587  else

590  _player->ToggleAFK();
591 
592  _player->ToggleDND();
593  }
594  }
595  break;
596  }
597 
598  default:
599  sLog.outError("CHAT: unknown message type %u, lang: %u", type, lang);
600  break;
601  }

652 void WorldSession::HandleTextEmoteOpcode(WorldPacket& recv_data)
653 {

656 
657  recv_data >> textEmote;
658  recv_data >> emoteNum;
659  recv_data >> guid;
660

662  return;
663 
664  if (!GetPlayer()->CanSpeak())
665  {
666  std::string timeStr = secsToTimeString(m_muteTime - time(nullptr));
667  SendNotification(GetMangosString(LANG_WAIT_BEFORE_SPEAKING), timeStr.c_str());
668  return;
669  }
670

672  if (!em)
673  return;

680  case EMOTE_STATE_SIT:

685  {

689  }
690  }
