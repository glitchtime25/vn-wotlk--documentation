299 bool AreaTrigger_at_murkdeep(Player* pPlayer, AreaTriggerEntry const* /*pAt*/)
300 {
301  // Handle Murkdeep event start
302  // The area trigger summons 3 Greymist Coastrunners; The rest of the event is handled by world map scripts
303  if (pPlayer->IsAlive() && !pPlayer->IsGameMaster() && pPlayer->GetQuestStatus(QUEST_WANTED_MURKDEEP) == QUEST_STATUS_INCOMPLETE)
304  {
305  ScriptedMap* pScriptedMap = (ScriptedMap*)pPlayer->GetInstanceData();
306  if (!pScriptedMap)
307  return false;
308 
309  // If Murkdeep is already spawned, skip the rest
310  if (pScriptedMap->GetSingleCreatureFromStorage(NPC_MURKDEEP, true))
311  return true;
312 
313  // Check if there are already coastrunners (dead or alive) around the area
314  if (GetClosestCreatureWithEntry(pPlayer, NPC_GREYMIST_COASTRUNNNER, 60.0f, false, false))
315  return true;
316 
317  float fX, fY, fZ;
318  for (uint8 i = 0; i < 3; ++i)
319  {
320  // Spawn locations are defined in World Maps Scripts.h
321  pPlayer->GetRandomPoint(aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][0], aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][1], aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][2], 5.0f, fX, fY, fZ);
322 
323  if (Creature* pTemp = pPlayer->SummonCreature(NPC_GREYMIST_COASTRUNNNER, fX, fY, fZ, aSpawnLocations[POS_IDX_MURKDEEP_SPAWN][3], TEMPSPAWN_DEAD_DESPAWN, 0))
324  {
325  pTemp->SetWalk(false);
326  pTemp->GetRandomPoint(aSpawnLocations[POS_IDX_MURKDEEP_MOVE][0], aSpawnLocations[POS_IDX_MURKDEEP_MOVE][1], aSpawnLocations[POS_IDX_MURKDEEP_MOVE][2], 5.0f, fX, fY, fZ);
327  pTemp->GetMotionMaster()->MovePoint(0, fX, fY, fZ);
328  }
329  }
330  }
331 
332  return false;
333 }
