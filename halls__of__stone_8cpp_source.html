222 void instance_halls_of_stone::SetData(uint32 uiType, uint32 uiData)
223 {
224  switch (uiType)
225  {
226  case TYPE_TRIBUNAL:
227  m_auiEncounter[uiType] = uiData;
228  switch (uiData)
229  {
230  case IN_PROGRESS:
231  SortFaces();
232  break;
233  case DONE:
234  if (GameObject* pFloor = GetSingleGameObjectFromStorage(GO_TRIBUNAL_FLOOR))
235  pFloor->ResetDoorOrButton();
236 
237  for (auto& m_aFace : m_aFaces)
238  {
239  // Shut down the faces
240  if (m_aFace.m_bIsActive)
241  if (GameObject* pFace = instance->GetGameObject(m_aFace.m_goFaceGuid))
242  pFace->ResetDoorOrButton();
243  }
244  break;
245  case FAIL:
246  for (auto& m_aFace : m_aFaces)
247  {
248  // Shut down the faces
249  if (m_aFace.m_bIsActive)
250  if (GameObject* pFace = instance->GetGameObject(m_aFace.m_goFaceGuid))
251  pFace->ResetDoorOrButton();
252 
253  m_aFace.m_bIsActive = false;
254  m_aFace.m_uiTimer = 1000;
255  m_aFace.m_uiSummonTimer = 10000;
256  }
257 
258  // despawn drawfs
259  for (const auto& guid : m_lRuneDwarfGUIDs)
260  if (Creature* pDwarf = instance->GetCreature(guid))
261  pDwarf->ForcedDespawn();
262 
263  m_lRuneDwarfGUIDs.clear();
264 
265  if (GameObject* pConsole = GetSingleGameObjectFromStorage(GO_TRIBUNAL_CONSOLE))
266  pConsole->ResetDoorOrButton();
267  break;
268  case SPECIAL:
269  for (auto& m_aFace : m_aFaces)
270  {
271  m_aFace.m_bIsActive = false;
272  m_aFace.m_uiTimer = 1000;
273 
274  // Cleanup when finished
275  if (Creature* pEye = instance->GetCreature(m_aFace.m_leftEyeGuid))
276  pEye->CastSpell(pEye, SPELL_KILL_TRIBUNAL_ADD, TRIGGERED_OLD_TRIGGERED);
277  if (Creature* pEye = instance->GetCreature(m_aFace.m_rightEyeGuid))
278  pEye->CastSpell(pEye, SPELL_KILL_TRIBUNAL_ADD, TRIGGERED_OLD_TRIGGERED);
279  }
280 
281  // mark achiev as complete
282  if (Creature* pEye = instance->GetCreature(m_aFaces[1].m_leftEyeGuid))
283  pEye->CastSpell(pEye, SPELL_ACHIEVEMENT_CHECK, TRIGGERED_OLD_TRIGGERED);
284 
285  // enable loot
286  DoToggleGameObjectFlags(instance->IsRegularDifficulty() ? GO_TRIBUNAL_CHEST : GO_TRIBUNAL_CHEST_H, GO_FLAG_NO_INTERACT, false);
287 
288  if (GameObject* pConsole = GetSingleGameObjectFromStorage(GO_TRIBUNAL_CONSOLE))
289  pConsole->ResetDoorOrButton();
290  break;
291  }
292  break;
293  case TYPE_MAIDEN:
294  m_auiEncounter[uiType] = uiData;
295  if (uiData == IN_PROGRESS)
296  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_MAIDEN_ID);
297  break;
298  case TYPE_KRYSTALLUS:
299  m_auiEncounter[uiType] = uiData;
300  break;
301  case TYPE_SJONNIR:
302  m_auiEncounter[uiType] = uiData;
303  if (uiData == DONE)
304  {
305  DoUseDoorOrButton(GO_DOOR_SJONNIR);
306 
307  if (GameObject* pConsole = GetSingleGameObjectFromStorage(GO_SJONNIR_CONSOLE))
308  pConsole->ResetDoorOrButton();
309  }
310  else if (uiData == FAIL)
311  {
312  if (GameObject* pConsole = GetSingleGameObjectFromStorage(GO_SJONNIR_CONSOLE))
313  pConsole->ResetDoorOrButton();
314  }
315  else if (uiData == IN_PROGRESS)
316  {
317  DoUseDoorOrButton(GO_DOOR_SJONNIR);
318  m_uiIronSludgeKilled = 0;
319  }
320  break;
321  }
322 
323  if (uiData == DONE)
324  {
325  OUT_SAVE_INST_DATA;
326 
327  std::ostringstream saveStream;
328  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
329 
330  m_strInstData = saveStream.str();
331 
332  SaveToDB();
333  OUT_SAVE_INST_DATA_COMPLETE;
334  }

544 void instance_halls_of_stone::Update(uint32 uiDiff)
545 {
546  // Tribunal encounter timer; handles the spell part of the event
547  if (m_auiEncounter[TYPE_TRIBUNAL] == IN_PROGRESS)
548  {
549  for (uint8 i = 0; i < MAX_FACES; ++i)
550  {
551  if (!m_aFaces[i].m_bIsActive)
552  continue;
553 
554  // spell timers
555  if (m_aFaces[i].m_uiTimer < uiDiff)
556  ProcessFaceSpell(i);
557  else
558  m_aFaces[i].m_uiTimer -= uiDiff;
559 
560  // summon timers
561  if (m_aFaces[i].m_uiSummonTimer < uiDiff)
562  ProcessFaceSummon(i);
563  else
564  m_aFaces[i].m_uiSummonTimer -= uiDiff;
565  }
566  }
567 
568  // periodic lightning channel
569  if (m_uiLightningChannelTimer < uiDiff)
570  {
571  // Note: check if they are really random or cyclical
572  uint32 spellId = 0;
573  switch (urand(0, 2))
574  {
575  case 0: spellId = SPELL_LIGHTNING_CHANNEL_1; break;
576  case 1: spellId = SPELL_LIGHTNING_CHANNEL_2; break;
577  case 2: spellId = SPELL_LIGHTNING_CHANNEL_3; break;
578  }
579 
580  if (Creature* pTribunal = instance->GetCreature(m_tribunalCasterGuid))
581  pTribunal->CastSpell(pTribunal, spellId, TRIGGERED_NONE);
582 
583  m_uiLightningChannelTimer = urand(20000, 40000);
584  }
585  else
586  m_uiLightningChannelTimer -= uiDiff;
587 }
