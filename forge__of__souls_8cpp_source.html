62 void instance_forge_of_souls::ProcessEventNpcs(Player* pPlayer)
63 {
64  if (!pPlayer)
65  return;
66 
67  if (GetData(TYPE_BRONJAHM) != DONE || GetData(TYPE_DEVOURER_OF_SOULS) != DONE)
68  {
69  // Spawn Begin Mobs
70  for (const auto& aEventBeginLocation : aEventBeginLocations)
71  {
72  if (Creature* pSummon = pPlayer->SummonCreature(m_uiTeam == HORDE ? aEventBeginLocation.uiEntryHorde : aEventBeginLocation.uiEntryAlliance,
73  aEventBeginLocation.fSpawnX, aEventBeginLocation.fSpawnY, aEventBeginLocation.fSpawnZ, aEventBeginLocation.fSpawnO, TEMPSPAWN_DEAD_DESPAWN, 24 * HOUR * IN_MILLISECONDS))
74  m_lEventMobGUIDs.push_back(pSummon->GetObjectGuid());
75  }
76  }
77  else
78  {
79  // Despawn entrance mobs
80  for (const auto& guid : m_lEventMobGUIDs)
81  if (Creature* pSummoned = instance->GetCreature(guid))
82  pSummoned->ForcedDespawn();
83 
84  // spawn exit mobs; movement and texts handled by DB script
85  for (const auto& aEventEndLocation : aEventEndLocations)
86  {
87  if (Creature* pCreature = pPlayer->SummonCreature(m_uiTeam == HORDE ? aEventEndLocation.uiEntryHorde : aEventEndLocation.uiEntryAlliance,
88  aEventEndLocation.fSpawnX, aEventEndLocation.fSpawnY, aEventEndLocation.fSpawnZ, aEventEndLocation.fOrient, TEMPSPAWN_DEAD_DESPAWN, 24 * HOUR * IN_MILLISECONDS))
89  {
90  pCreature->SetWalk(false);
91  pCreature->GetMotionMaster()->MoveWaypoint(aEventEndLocation.pathId);
92  }
93  }
94  }
95 }

110 void instance_forge_of_souls::SetData(uint32 uiType, uint32 uiData)
111 {
112  switch (uiType)
113  {
114  case TYPE_BRONJAHM:
115  m_auiEncounter[0] = uiData;
116 
117  // Despawn remaining adds and clear list
118  for (const auto& guid : m_luiSoulFragmentAliveGUIDs)
119  if (Creature* pFragment = instance->GetCreature(guid))
120  pFragment->ForcedDespawn();
121 
122  m_luiSoulFragmentAliveGUIDs.clear();
123  break;
124  case TYPE_DEVOURER_OF_SOULS:
125  m_auiEncounter[1] = uiData;
126  if (uiData == DONE)
127  ProcessEventNpcs(GetPlayerInMap());
128  break;
129  case TYPE_ACHIEV_PHANTOM_BLAST:
130  m_bCriteriaPhantomBlastFailed = (uiData == FAIL);
131  return;
132  }
133 
134  if (uiData == DONE)
135  {
136  OUT_SAVE_INST_DATA;
137 
138  std::ostringstream saveStream;
139  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1];
140 
141  m_strInstData = saveStream.str();
142 
143  SaveToDB();
144  OUT_SAVE_INST_DATA_COMPLETE;
145  }
146 }
