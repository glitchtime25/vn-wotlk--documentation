37 void BattleGroundWS::Update(uint32 diff)
38 {
39  BattleGround::Update(diff);
40 
41  if (GetStatus() != STATUS_IN_PROGRESS)
42  return;
43 
44  // compute flag respawn timers
45  for (uint8 i = 0; i < PVP_TEAM_COUNT; ++i)
46  {
47  // respawn flag after team score
48  if (m_flagOnRespawn[i])
49  {
50  if (m_flagsTimer[i] <= diff)
51  {
52  Team playerTeam = GetTeamIdByTeamIndex((PvpTeamIndex)i);
53  RespawnFlagAtBase(playerTeam, true);
54  m_flagsTimer[i] = 0;
55  }
56  else
57  m_flagsTimer[i] -= diff;
58  }
59  // respawn flag after dropped
60  else if (GetBgMap()->GetVariableManager().GetVariable(wsFlagPickedUp[i]) == BG_WS_FLAG_STATE_ON_GROUND && m_flagsDropTimer[i])
61  {
62  if (m_flagsDropTimer[i] <= diff)
63  {
64  Team playerTeam = GetTeamIdByTeamIndex((PvpTeamIndex)i);
65  m_flagsDropTimer[i] = 0;
66  RespawnDroppedFlag(playerTeam);
67  }
68  else
69  m_flagsDropTimer[i] -= diff;
70  }
71  }
72 
73  // Count the renaining time; update world state and end battleground if timer is expired
74  if (m_endTimer <= diff)
75  {
76  EndBattleGround(GetPrematureWinner());
77  }
78  else
79  {
80  uint32 minutesLeftPrev = GetRemainingTimeInMinutes();
81  m_endTimer -= diff;
82  uint32 minutesLeft = GetRemainingTimeInMinutes();
83 
84  if (minutesLeft != minutesLeftPrev)
85  GetBgMap()->GetVariableManager().SetVariable(BG_WS_STATE_TIME_REMAINING, minutesLeft);
86  }
87 
88  // as long as both flags are missing from bases, progress the FC debuff timer until 15 minutes have elapsed
89  if (GetBgMap()->GetVariableManager().GetVariable(BG_WS_STATE_FLAG_PICKED_UP_FLAG_STATE_ALLIANCE) != BG_WS_FLAG_STATE_ON_BASE
90  && GetBgMap()->GetVariableManager().GetVariable(BG_WS_STATE_FLAG_PICKED_UP_FLAG_STATE_HORDE) != BG_WS_FLAG_STATE_ON_BASE)
91  {
92  if (m_flagCarrierDebuffTimer)
93  {
94  if (m_flagCarrierDebuffTimer <= BG_WS_FOCUSED_ASSAULT_TIME && !m_focusedAssaultActive)
95  {
96  for (uint8 i = 0; i < PVP_TEAM_COUNT; ++i)
97  {
98  Team playerTeam = GetTeamIdByTeamIndex((PvpTeamIndex)i);
99  if (IsFlagPickedUp(playerTeam))
100  if (Player* player = GetBgMap()->GetPlayer(GetFlagCarrierGuid((PvpTeamIndex)i)))
101  player->CastSpell(player, BG_WS_SPELL_FOCUSED_ASSAULT, TRIGGERED_OLD_TRIGGERED);
102  }
103  m_focusedAssaultActive = true;
104  }
105  if (m_flagCarrierDebuffTimer <= diff)
106  {
107  for (uint8 i = 0; i < PVP_TEAM_COUNT; ++i)
108  {
109  Team playerTeam = GetTeamIdByTeamIndex((PvpTeamIndex)i);
110  if (IsFlagPickedUp(playerTeam))
111  {
112  if (Player* player = GetBgMap()->GetPlayer(GetFlagCarrierGuid((PvpTeamIndex)i)))
113  {
114  player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
115  player->CastSpell(player, BG_WS_SPELL_BRUTAL_ASSAULT, TRIGGERED_OLD_TRIGGERED);
116  }
117  }
118  }
119  m_focusedAssaultActive = false;
120  m_brutalAssaultActive = true;
121  m_flagCarrierDebuffTimer = 0; // stop timer here for now - no reason to resume until the current stalemate is over and another begins (capture occurs, or both flags are returned?)
122  }
123  else
124  m_flagCarrierDebuffTimer -= diff;
125  }
126  }

223 // Method that process captured flag score
224 void BattleGroundWS::ProcessPlayerFlagScoreEvent(Player* player)
225 {
226  if (GetStatus() != STATUS_IN_PROGRESS)
227  return;
228 
229  DEBUG_LOG("BattleGroundWS: Team %u has scored.", player->GetTeam());
230 
231  player->RemoveAurasWithInterruptFlags(AURA_INTERRUPT_FLAG_PVP_ACTIVE_CANCELS);
232 
233  Team team = player->GetTeam();
234  PvpTeamIndex teamIdx = GetTeamIndexByTeamId(team);
235  uint8 otherTeamIdx = GetOtherTeamIndex(teamIdx);
236 
237  if (!IsFlagPickedUp(otherTeamIdx))
238  return;
239 
240  ClearFlagCarrier(otherTeamIdx); // must be before aura remove to prevent 2 events (drop+capture) at the same time
241 
242  // Horde flag in base (but not respawned yet)
243  GetBgMap()->GetVariableManager().SetVariable(wsFlagPickedUp[otherTeamIdx], BG_WS_FLAG_STATE_ON_BASE);
244  m_flagOnRespawn[otherTeamIdx] = true;
245 
246  // Drop Horde Flag from Player
247  player->RemoveAurasDueToSpell(wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_PICKEDUP].spellId);
248  player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
249  player->RemoveAurasDueToSpell(BG_WS_SPELL_BRUTAL_ASSAULT);
250 
251  // play sound and reward reputation
252  PlaySoundToAll(wsgFlagData[teamIdx][BG_WS_FLAG_ACTION_CAPTURED].soundId);
253  RewardReputationToTeam(team == ALLIANCE ? 890 : 889, m_reputationCapture, team);
254 
255  // for flag capture is reward 2 honorable kills
256  RewardHonorToTeam(GetBonusHonorFromKill(2), team);
257 
258  // update score
259  int32 pointsWorldState = team == ALLIANCE ? BG_WS_STATE_CAPTURES_ALLIANCE : BG_WS_STATE_CAPTURES_HORDE;
260  GetBgMap()->GetVariableManager().SetVariable(pointsWorldState, GetBgMap()->GetVariableManager().GetVariable(pointsWorldState) + 1);
261 
262  GetBgMap()->GetVariableManager().SetVariable(wsFlagPickedUp[teamIdx], BG_WS_FLAG_STATE_ON_BASE);
263  GetBgMap()->GetVariableManager().SetVariable(wsFlagHUDPickedUp[teamIdx], BG_WS_FLAG_ICON_INVISIBLE);
264 
265  // despawn flags
266  SpawnEvent(WS_EVENT_FLAG_A, 0, false);
267  SpawnEvent(WS_EVENT_FLAG_H, 0, false);
268 
269  SendMessageToAll(wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_CAPTURED].messageId, wsgFlagData[teamIdx][BG_WS_FLAG_ACTION_CAPTURED].chatType, player);
270 
271  // only flag capture should be updated
272  UpdatePlayerScore(player, SCORE_FLAG_CAPTURES, 1); // +1 flag captures
273 
274  // Process match winner
275  Team winner = GetBgMap()->GetVariableManager().GetVariable(pointsWorldState) == BG_WS_MAX_TEAM_SCORE ? GetTeamIdByTeamIndex(teamIdx) : TEAM_NONE;
276 
277  // update achievements
278  player->UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_BG_OBJECTIVE_CAPTURE, 42);
279 
280  // end battleground or set respawn timer
281  if (winner)
282  {
283  GetBgMap()->GetVariableManager().SetVariable(BG_WS_STATE_FLAG_PICKED_UP_FLAG_STATE_ALLIANCE, 0);
284  GetBgMap()->GetVariableManager().SetVariable(BG_WS_STATE_FLAG_PICKED_UP_FLAG_STATE_HORDE, 0);
285  GetBgMap()->GetVariableManager().SetVariable(BG_WS_STATE_FLAG_ALLIANCE, BG_WS_FLAG_ICON_INVISIBLE);
286  GetBgMap()->GetVariableManager().SetVariable(BG_WS_STATE_FLAG_HORDE, BG_WS_FLAG_ICON_INVISIBLE);
287 
288  EndBattleGround(winner);
289  }
290  else
291  m_flagsTimer[otherTeamIdx] = BG_WS_FLAG_RESPAWN_TIME;
292 }

294 // Method that handles the flag drop
295 void BattleGroundWS::HandlePlayerDroppedFlag(Player* player)
296 {
297  DEBUG_LOG("BattleGroundWS: Team %u has dropped the flag.", player->GetTeam());
298 
299  Team team = player->GetTeam();
300  PvpTeamIndex teamIdx = GetTeamIndexByTeamId(team);
301  PvpTeamIndex otherTeamIdx = GetOtherTeamIndex(teamIdx);
302 
303  player->RemoveAurasDueToSpell(BG_WS_SPELL_FOCUSED_ASSAULT);
304  player->RemoveAurasDueToSpell(BG_WS_SPELL_BRUTAL_ASSAULT);
305 
306  if (GetStatus() != STATUS_IN_PROGRESS)
307  {
308  if (IsFlagPickedUp(teamIdx) && GetFlagCarrierGuid(teamIdx) == player->GetObjectGuid())
309  {
310  ClearFlagCarrier(teamIdx);
311  player->RemoveAurasDueToSpell(wsgFlagData[teamIdx][BG_WS_FLAG_ACTION_PICKEDUP].spellId);
312  }
313  }
314  // Check if the other team has picked the flag and if the flag carrier stored is the player which triggers this function
315  else if (IsFlagPickedUp(otherTeamIdx) && GetFlagCarrierGuid(otherTeamIdx) == player->GetObjectGuid())
316  {
317  ClearFlagCarrier(otherTeamIdx);
318  player->RemoveAurasDueToSpell(wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_PICKEDUP].spellId);
319 
320  GetBgMap()->GetVariableManager().SetVariable(wsFlagPickedUp[otherTeamIdx], BG_WS_FLAG_STATE_ON_GROUND);
321  player->CastSpell(player, wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_DROPPED].spellId, TRIGGERED_OLD_TRIGGERED);
322 
323  player->CastSpell(player, SPELL_RECENTLY_DROPPED_FLAG, TRIGGERED_OLD_TRIGGERED);
324  GetBgMap()->GetVariableManager().SetVariable(wsFlagHUDPickedUp[otherTeamIdx], BG_WS_FLAG_ICON_INVISIBLE);
325 
326  SendMessageToAll(wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_DROPPED].messageId, wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_DROPPED].chatType, player);
327 
328  m_flagsDropTimer[otherTeamIdx] = BG_WS_FLAG_DROP_TIME;
329  }

344

355 
356  // not meant to be a triggered cast - clears a lot during cast like stealth

358  if (m_brutalAssaultActive)

363  PlaySoundToAll(wsgFlagData[otherTeamIdx][BG_WS_FLAG_ACTION_PICKEDUP].soundId);

366 
367  // start timed achiev

431 }

449 // Handle player exit

477  }
478  else
479  HandlePlayerDroppedFlag(player);
480  }
