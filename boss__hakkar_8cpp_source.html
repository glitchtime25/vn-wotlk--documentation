77  // For each of the High Priests that is alive, update Hakkar's Power Stacks (updating Hakkar's HP)
78  void InitiateHakkarPowerStacks()
79  {
80  m_creature->RemoveAurasDueToSpell(SPELL_HAKKAR_POWER);
81  for (uint8 i = 0; i < MAX_PRIESTS; i++)
82  {
83  if (m_instance->GetData(i) != DONE)
84  m_creature->CastSpell(nullptr, SPELL_HAKKAR_POWER, TRIGGERED_NONE);
85  }
86  }
