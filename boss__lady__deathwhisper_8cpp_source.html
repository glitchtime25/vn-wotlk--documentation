222  // Wrapper to help sort the summoning stalkers
223  void DoSortSummoningStalkers(GuidList& lDeathwhisperStalkers)
224  {
225  CreatureList lRightStalkers;
226  CreatureList lLeftStalkers;
227 
228  if (!lDeathwhisperStalkers.empty())
229  {
230  for (const auto& guid : lDeathwhisperStalkers)
231  {
232  if (Creature* pStalker = m_creature->GetMap()->GetCreature(guid))
233  {
234  if (pStalker->GetPositionZ() > 60.0f)
235  m_middleStalkerGuid = pStalker->GetObjectGuid();
236  else if (pStalker->GetPositionY() < 2215.0f)
237  lLeftStalkers.push_back(pStalker);
238  else
239  lRightStalkers.push_back(pStalker);
240  }
241  }
242 
243  lLeftStalkers.sort(sortFromNorthToSouth);
244  lRightStalkers.sort(sortFromNorthToSouth);
245 
246  // Store the sorted stalkers in a vector for each side
247  for (CreatureList::const_iterator itr = lLeftStalkers.begin(); itr != lLeftStalkers.end(); ++itr)
248  m_vLeftStalkersGuidVector.push_back((*itr)->GetObjectGuid());
249  for (CreatureList::const_iterator itr = lRightStalkers.begin(); itr != lRightStalkers.end(); ++itr)
250  m_vRightStalkersGuidVector.push_back((*itr)->GetObjectGuid());
251  }
252  else
253  script_error_log("instance_icecrown_citadel: Error: cannot find creature %u in instance", NPC_DEATHWHISPER_SPAWN_STALKER);
254  }
255 
256  static bool sortFromNorthToSouth(Creature* pFirst, Creature* pSecond)
257  {
258  return pFirst && pSecond && pFirst->GetPositionX() < pSecond->GetPositionX();
259  }

263  {
264  std::vector<Creature*> vCultists;
265  vCultists.reserve(m_lCultistSpawnedGuidList.size());
266

268  {

270  {
271  // Allow to be sorted them by entry
272  if (!uiEntry)

285  // Wrapper to handle the adds summmoning
286  void DoSummonCultistWave()
287  {
288  if (!m_instance)
289  return;
290 
291  // failsafe check
292  if (m_vLeftStalkersGuidVector.size() < 3 || m_vRightStalkersGuidVector.size() < 3)
293  {
294  script_error_log("instance_icecrown_citadel: Error: Cannot find all left or right spawn stalkers.");
295  return;
296  }
297 
298  // On 25 man mode we need to summon on all points
299  if (m_instance->Is25ManDifficulty())
300  {
301  for (uint8 i = 0; i < 3; ++i)
302  {
303  if (Creature* pStalker = m_creature->GetMap()->GetCreature(m_vLeftStalkersGuidVector[i]))
304  m_creature->SummonCreature(aLeftSummonedCultists[i], pStalker->GetPositionX(), pStalker->GetPositionY(), pStalker->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
305  if (Creature* pStalker = m_creature->GetMap()->GetCreature(m_vRightStalkersGuidVector[i]))
306  m_creature->SummonCreature(aRightSummonedCultists[i], pStalker->GetPositionX(), pStalker->GetPositionY(), pStalker->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
307  }
308 
309  if (Creature* pStalker = m_creature->GetMap()->GetCreature(m_middleStalkerGuid))
310  m_creature->SummonCreature(roll_chance_i(50) ? NPC_CULT_FANATIC : NPC_CULT_ADHERENT, pStalker->GetPositionX(), pStalker->GetPositionY(), pStalker->GetPositionZ(), 0.0f, TEMPSPAWN_CORPSE_DESPAWN, 0);
311  }
312  // On 10 man mode we summon on the left or on the right
313  else
314  {
315  // Summon just 1 add in phase 2 heroic
316  if (m_bIsHeroicMode && !m_bIsPhaseOne)
317  {
318  if (Creature* pStalker = m_creature->GetMap()->GetCreature(m_middleStalkerGuid))
319  m_creature->SummonCreature(roll_chance_i(50) ? NPC_CULT_FANATIC : NPC_CULT_ADHERENT, pStalker->GetPositionX(), pStalker->GetPositionY(), pStalker->GetPositionZ(), 0.0f, TEMPSPAWN_CORPSE_DESPAWN, 0);
320  }
321  else
322  {
323  GuidVector vTempVector = m_bIsLeftSideSummon ? m_vLeftStalkersGuidVector : m_vRightStalkersGuidVector;
324  for (uint8 i = 0; i < 3; ++i)
325  {
326  if (Creature* pStalker = m_creature->GetMap()->GetCreature(vTempVector[i]))
327  m_creature->SummonCreature(m_bIsLeftSideSummon ? aLeftSummonedCultists[i] : aRightSummonedCultists[i], pStalker->GetPositionX(), pStalker->GetPositionY(), pStalker->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
328  }
329 
330  // change sides for next summoning
331  m_bIsLeftSideSummon = !m_bIsLeftSideSummon;
332  }
333  }
334  }

361  void ExecuteAction(uint32 action) override
362  {
363  switch (action)
364  {
365  case DEATHWHISPER_BERSRK:
366  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
367  {
368  DoScriptText(SAY_BERSERK, m_creature);
369  DisableCombatAction(action);
370  }
371  break;
372  case DEATHWHISPER_DEATH_AND_DECAY:
373  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
374  {
375  if (DoCastSpellIfCan(target, SPELL_DEATH_AND_DECAY) == CAST_OK)
376  ResetCombatAction(action, 20000);
377  }
378  break;
379  case DEATHWHISPER_DOMINATE_MIND:
380  for (uint8 i = 0; i < m_uiMindControlCount; ++i)
381  {
382  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_DOMINATE_MIND, SELECT_FLAG_PLAYER))
383  DoCastSpellIfCan(target, SPELL_DOMINATE_MIND, CAST_TRIGGERED);
384  }
385 
386  DoScriptText(SAY_DOMINATE_MIND, m_creature);
387  ResetCombatAction(action, 45000);
388  break;
389  case DEATHWHISPER_SHADOW_BOLT:
390  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
391  {
392  if (DoCastSpellIfCan(target, SPELL_SHADOW_BOLT) == CAST_OK)
393  ResetCombatAction(action, urand(2000, 3000));
394  }
395  break;
396  case DEATHWHISPER_CULTIST_SUMMON:
397  DoSummonCultistWave();
398  ResetCombatAction(DEATHWHISPER_CULTIST_BUFF, 10000);
399  ResetCombatAction(DEATHWHISPER_CULTIST_MARTYRDOM, 40000);
400  ResetCombatAction(action, m_bIsHeroicMode ? 45000 : 60000);
401  break;
402  case DEATHWHISPER_CULTIST_BUFF:
403  {
404  // Choose a random of Fanatic or Adherent
405  bool bIsFanatic = roll_chance_i(50);
406  uint32 uiNpcEntry = bIsFanatic ? NPC_CULT_FANATIC : NPC_CULT_ADHERENT;
407  uint32 uiSpellEntry = bIsFanatic ? SPELL_DARK_TRANSFORMATION : SPELL_DARK_EMPOWERMENT;
408  int32 iTextEntry = bIsFanatic ? SAY_DARK_TRANSFORMATION : SAY_DARK_EMPOWERMENT;
409 
410  // Try to select a random Cultist of entry
411  if (Creature* target = DoSelectRandomCultist(uiNpcEntry))
412  {
413  if (DoCastSpellIfCan(target, uiSpellEntry) == CAST_OK)
414  {
415  // Remove the selected cultist from the list because we don't want it selected twice
416  m_lCultistSpawnedGuidList.remove(target->GetObjectGuid());
417  DoScriptText(iTextEntry, m_creature);
418  DisableCombatAction(action);
419  }
420  }
421 
422  break;
423  }
424  case DEATHWHISPER_CULTIST_MARTYRDOM:
425  // Try to get a target on which to cast Martyrdom
426  if (Creature* target = DoSelectRandomCultist())
427  {
428  if (DoCastSpellIfCan(target, SPELL_DARK_MARTYRDOM) == CAST_OK)
429  {
430  DoScriptText(SAY_ANIMATE_DEAD, m_creature);
431  DisableCombatAction(action);
432  }
433  }
434  break;
435  case DEATHWHISPER_INSIGNIFICANCE:
436  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_INSIGNIFICANCE) == CAST_OK)
437  ResetCombatAction(action, 7000);
438  break;
439  case DEATHWHISPER_FROSTBOLT:
440  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
441  {
442  if (DoCastSpellIfCan(target, SPELL_FROSTBOLT) == CAST_OK)
443  ResetCombatAction(action, 12000);
444  }
445  break;
446  case DEATHWHISPER_FROSTBOLT_VOLLEY:
447  if (DoCastSpellIfCan(m_creature, SPELL_FROSTBOLT_VOLLEY) == CAST_OK)
448  ResetCombatAction(action, 20000);
449  break;
450  case DEATHWHISPER_SUMMON_SPIRIT:
451  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
452  {
453  if (DoCastSpellIfCan(target, SPELL_SUMMON_SPIRIT) == CAST_OK)
454  ResetCombatAction(action, 10000);
455  }
456  break;
457  }
458  }
