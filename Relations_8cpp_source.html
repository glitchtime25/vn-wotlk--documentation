109 {

112 
113  // Original logic begins
114

119  {

133  {
134  if (i == otherTemplate->faction)
135  return REP_FRIENDLY;
136  }
137  }
138 
139  if (thisTemplate->factionGroupMask & otherTemplate->friendGroupMask)
140  return REP_FRIENDLY;

143  {
144  for (unsigned int i : otherTemplate->friendFaction)
145  {
146  if (i == thisTemplate->faction)
147  return REP_FRIENDLY;
148  }

208 ReputationRank Unit::GetReactionTo(Unit const* unit) const
209 {
210  MANGOS_ASSERT(unit)
211 
212  // Original logic begins
213 
214  if (this == unit)
215  return REP_FRIENDLY;
216 
217  if (HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
218  {
219  const Player* thisPlayer = GetControllingPlayer();
220 
221  if (unit->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
222  {
223  const Player* unitPlayer = unit->GetControllingPlayer();
224 
225  // WotLK+ player checks: does not default to neutral when either player controlled unit misses its player
226  if (thisPlayer && unitPlayer)
227  {
228  if (thisPlayer == unitPlayer)
229  return REP_FRIENDLY;
230 
231  if (thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) && thisPlayer->GetGuidValue(PLAYER_DUEL_ARBITER) == unitPlayer->GetGuidValue(PLAYER_DUEL_ARBITER))
232  return thisPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) != unitPlayer->GetUInt32Value(PLAYER_DUEL_TEAM) ? REP_HOSTILE : REP_FRIENDLY;
233 
234  // WotLK+ group check: faction to unit
235  if (thisPlayer->IsInGroup(unitPlayer))
236  return GetFactionReaction(GetFactionTemplateEntry(), unit);
237  }
238 
239  // WotLK+ FFA check
240  if (IsPvPFreeForAll() && unit->IsPvPFreeForAll())
241  return REP_HOSTILE;
242  }
243 
244  if (thisPlayer)
245  {
246  if (const FactionTemplateEntry* unitFactionTemplate = unit->GetFactionTemplateEntry())
247  {
248  if (const ReputationRank* rank = thisPlayer->GetReputationMgr().GetForcedRankIfAny(unitFactionTemplate))
249  return (*rank);
250 
251  const FactionEntry* unitFactionEntry = sFactionStore.LookupEntry(unitFactionTemplate->faction);
252 
253  // If the faction has reputation ranks available, "at war" and contested PVP flags decide outcome
254  if (!this->HasFlag(UNIT_FIELD_FLAGS_2, UNIT_FLAG2_IGNORE_REPUTATION) && unitFactionEntry && unitFactionEntry->HasReputation())
255  {
256  if (thisPlayer->HasFlag(PLAYER_FLAGS, PLAYER_FLAGS_CONTESTED_PVP) && unitFactionTemplate->IsContestedGuardFaction())
257  return REP_HOSTILE;
258 
259  return thisPlayer->GetReputationMgr().IsAtWar(unitFactionEntry) ? REP_HOSTILE : REP_FRIENDLY;
260  }
261  }
262  }
263  }
264  // Default fallback if player-specific checks didn't catch anything: facton to unit
265  return GetFactionReaction(GetFactionTemplateEntry(), unit);
266 }
