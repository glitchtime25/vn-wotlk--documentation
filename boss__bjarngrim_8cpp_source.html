174  void UpdateAI(const uint32 uiDiff) override
175  {
176  // Return since we have no target
177  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
178  return;
179 
180  // Change stance
181  if (m_uiChangeStanceTimer < uiDiff)
182  {
183  // wait for current spell to finish before change stance
184  if (m_creature->IsNonMeleeSpellCasted(false))
185  return;
186 
187  int uiTempStance = irand() % (3 - 1);
188 
189  if (uiTempStance >= m_uiStance)
190  ++uiTempStance;
191 
192  m_uiStance = uiTempStance;
193 
194  switch (m_uiStance)
195  {
196  case STANCE_DEFENSIVE:
197  DoScriptText(SAY_DEFENSIVE_STANCE, m_creature);
198  DoScriptText(EMOTE_DEFENSIVE_STANCE, m_creature);
199  DoCastSpellIfCan(m_creature, SPELL_DEFENSIVE_STANCE);
200  break;
201  case STANCE_BERSERKER:
202  DoScriptText(SAY_BERSEKER_STANCE, m_creature);
203  DoScriptText(EMOTE_BERSEKER_STANCE, m_creature);
204  DoCastSpellIfCan(m_creature, SPELL_BERSEKER_STANCE);
205  break;
206  case STANCE_BATTLE:
207  DoScriptText(SAY_BATTLE_STANCE, m_creature);
208  DoScriptText(EMOTE_BATTLE_STANCE, m_creature);
209  DoCastSpellIfCan(m_creature, SPELL_BATTLE_STANCE);
210  break;
211  }
212 
213  m_uiChangeStanceTimer = urand(20000, 25000);
214  return;
215  }
216  m_uiChangeStanceTimer -= uiDiff;
217 
218  switch (m_uiStance)
219  {
220  case STANCE_DEFENSIVE:
221  {
222  if (m_uiReflectionTimer < uiDiff)
223  {
224  if (DoCastSpellIfCan(m_creature, SPELL_SPELL_REFLECTION) == CAST_OK)
225  m_uiReflectionTimer = urand(8000, 9000);
226  }
227  else
228  m_uiReflectionTimer -= uiDiff;
229 
230  if (m_uiKnockAwayTimer < uiDiff)
231  {
232  if (DoCastSpellIfCan(m_creature, SPELL_KNOCK_AWAY) == CAST_OK)
233  m_uiKnockAwayTimer = urand(20000, 21000);
234  }
235  else
236  m_uiKnockAwayTimer -= uiDiff;
237 
238  if (m_uiPummelTimer < uiDiff)
239  {
240  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_PUMMEL) == CAST_OK)
241  m_uiPummelTimer = urand(10000, 11000);
242  }
243  else
244  m_uiPummelTimer -= uiDiff;
245 
246  if (m_uiIronformTimer < uiDiff)
247  {
248  if (DoCastSpellIfCan(m_creature, SPELL_IRONFORM) == CAST_OK)
249  m_uiIronformTimer = urand(25000, 26000);
250  }
251  else
252  m_uiIronformTimer -= uiDiff;
253 
254  break;
255  }
256  case STANCE_BERSERKER:
257  {
258  if (m_uiInterceptTimer < uiDiff)
259  {
260  // not much point is this, better random target and more often?
261  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_INTERCEPT) == CAST_OK)
262  m_uiInterceptTimer = urand(45000, 46000);

264  else
