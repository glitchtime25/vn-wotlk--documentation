396 template<typename Do>
397 inline void MapPersistentStateManager::DoForAllStatesWithMapId(uint32 mapId, Do& _do)
398 {
399  MapEntry const* mapEntry = sMapStore.LookupEntry(mapId);
400  if (!mapEntry)
401  return;
402 
403  if (mapEntry->Instanceable())
404  {
405  for (PersistentStateMap::iterator itr = m_instanceSaveByInstanceId.begin(); itr != m_instanceSaveByInstanceId.end();)
406  {
407  if (itr->second->GetMapId() == mapId)
408  _do((itr++)->second);
409  else
410  ++itr;
411  }
412  }
413  else
414  {
415  if (MapPersistentState* state = GetPersistentState(mapId, 0))
416  _do(state);
417  }
418 }
