183  void JustSummoned(Creature* summoned) override
184  {
185  switch (summoned->GetEntry())
186  {
187  case NPC_AMANI_HATCHER_1:
188  m_hatcherOneGuid = summoned->GetObjectGuid();
189  // If all the eggs from one side are hatched, move to the other side
190  if (m_uiEggsHatchedRight == MAX_EGGS_ON_SIDE)
191  summoned->GetMotionMaster()->MoveWaypoint(PATH_ID_LEFT);
192  else
193  summoned->GetMotionMaster()->MoveWaypoint(PATH_ID_RIGHT);
194  break;
195  case NPC_AMANI_HATCHER_2:
196  m_hatcherTwoGuid = summoned->GetObjectGuid();
197  // If all the eggs from one side are hatched, move to the other side
198  if (m_uiEggsHatchedLeft == MAX_EGGS_ON_SIDE)
199  summoned->GetMotionMaster()->MoveWaypoint(PATH_ID_RIGHT);
200  else
201  summoned->GetMotionMaster()->MoveWaypoint(PATH_ID_LEFT);
202  break;
203  case NPC_FIRE_BOMB:
204  summoned->AI()->SetCombatMovement(false);
205  m_creature->CastSpell(summoned, SPELL_FIRE_BOMB_THROW, TRIGGERED_OLD_TRIGGERED);
206  m_summons.push_back(summoned->GetObjectGuid());
207  break;
208  case NPC_HATCHLING: // forwarded from hatcher
209  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
210  summoned->AI()->AttackStart(target);
211  summoned->SetInCombatWithZone();
212  // Count the Hatched eggs
213  summoned->GetPositionY() > 1100.0f ? ++m_uiEggsHatchedLeft : ++m_uiEggsHatchedRight;
214  // Notify the script when all the eggs were hatched
215  if (m_uiEggsHatchedRight == MAX_EGGS_ON_SIDE && m_uiEggsHatchedLeft == MAX_EGGS_ON_SIDE)
216  {
217  DisableCombatAction(JANALAI_ACTION_SUMMON_HATCHER);
218  SetActionReadyStatus(JANALAI_ACTION_HATCH_ALL_EGGS, false);
219  if (Creature* hatcher = m_creature->GetMap()->GetCreature(m_hatcherOneGuid))
220  hatcher->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, hatcher);
221  if (Creature* hatcher = m_creature->GetMap()->GetCreature(m_hatcherTwoGuid))
222  hatcher->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, hatcher);
223  }
224  // Change the side of the hatcher if necessary
225  if (m_uiEggsHatchedRight == MAX_EGGS_ON_SIDE && m_uiEggsHatchedLeft < MAX_EGGS_ON_SIDE)
226  {
227  if (Creature* hatcher = m_creature->GetMap()->GetCreature(m_hatcherOneGuid))
228  hatcher->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, m_creature, hatcher, PATH_ID_LEFT_SHORT);
229  }
230  if (m_uiEggsHatchedLeft == MAX_EGGS_ON_SIDE && m_uiEggsHatchedRight < MAX_EGGS_ON_SIDE)
231  {
232  if (Creature* hatcher = m_creature->GetMap()->GetCreature(m_hatcherTwoGuid))
233  hatcher->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, m_creature, hatcher, PATH_ID_RIGHT_SHORT);
234  }
235  break;
236  }
237  }

248  void ExecuteAction(uint32 action) override
249  {
250  switch (action)
251  {
252  case JANALAI_ACTION_BERSERK:
253  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
254  {
255  DoScriptText(SAY_BERSERK, m_creature);
256  DisableCombatAction(action);
257  }
258  return;
259  case JANALAI_ACTION_ENRAGE_HP:
260  if (m_creature->GetHealthPercent() > 20.0f)
261  return;
262  case JANALAI_ACTION_ENRAGE_TIMER:
263  if (DoCastSpellIfCan(nullptr, SPELL_ENRAGE) == CAST_OK)
264  {
265  SetActionReadyStatus(JANALAI_ACTION_ENRAGE_HP, false);
266  DisableCombatAction(JANALAI_ACTION_ENRAGE_TIMER);
267  }
268  return;
269  case JANALAI_ACTION_HATCH_ALL_EGGS:
270  if (m_creature->GetHealthPercent() > 35.0f)
271  return;
272  if (DoCastSpellIfCan(nullptr, SPELL_HATCH_ALL_EGGS) == CAST_OK)
273  {
274  DoScriptText(SAY_ALL_EGGS, m_creature);
275  SetActionReadyStatus(JANALAI_ACTION_HATCH_ALL_EGGS, false);
276  DisableCombatAction(JANALAI_ACTION_SUMMON_HATCHER);
277  }
278  return;
279  case JANALAI_ACTION_SUMMON_HATCHER:
280  {
281  DoScriptText(SAY_SUMMON_HATCHER, m_creature);
282 
283  Creature* hatcher1 = m_creature->GetMap()->GetCreature(m_hatcherOneGuid);
284  Creature* hatcher2 = m_creature->GetMap()->GetCreature(m_hatcherTwoGuid);
285 
286  if (!hatcher1 || !hatcher1->IsAlive())
287  DoCastSpellIfCan(nullptr, SPELL_SUMMON_HATCHER_1, CAST_TRIGGERED);
288 
289  if (!hatcher2 || !hatcher2->IsAlive())
290  DoCastSpellIfCan(nullptr, SPELL_SUMMON_HATCHER_2, CAST_TRIGGERED);
291 
292  ResetCombatAction(action, 90000);
293  return;
294  }
295  case JANALAI_ACTION_FIRE_BOMBS:
296  if (DoCastSpellIfCan(nullptr, SPELL_FIRE_BOMB_CHANNEL) == CAST_OK)
297  {
298  DoCastSpellIfCan(nullptr, SPELL_TELEPORT_TO_CENTER, CAST_TRIGGERED);
299  DoCastSpellIfCan(nullptr, SPELL_SUMMON_ALL_PLAYERS, CAST_TRIGGERED);
300  DoScriptText(SAY_FIRE_BOMBS, m_creature);
301  DoCreateFireWall();
302 
303  ResetCombatAction(action, 30000);
304  }
305  return;
306  case JANALAI_ACTION_FIRE_BREATH:
307  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
308  if (DoCastSpellIfCan(pTarget, SPELL_FLAME_BREATH) == CAST_OK)
309  ResetCombatAction(action, 8000);
310  return;
311  }
312  }
