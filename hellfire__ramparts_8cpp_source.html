71 void instance_ramparts::SetData(uint32 uiType, uint32 uiData)
72 {
73  debug_log("SD2: Instance Ramparts: SetData received for type %u with data %u", uiType, uiData);
74 
75  switch (uiType)
76  {
77  case TYPE_VAZRUDEN:
78  if (m_auiEncounter[0] == uiData)
79  return;
80  if (uiData == DONE && m_auiEncounter[1] == DONE)
81  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_FEL_IRON_CHEST : GO_FEL_IRON_CHEST_H, HOUR);
82  if (uiData == FAIL && m_auiEncounter[0] != FAIL)
83  DoFailVazruden();
84  m_auiEncounter[0] = uiData;
85  break;
86  case TYPE_NAZAN:
87  if (m_auiEncounter[1] == uiData)
88  return;
89  if (uiData == SPECIAL) // SPECIAL set via ACID
90  {
91  ++m_uiSentryCounter;
92 
93  if (m_uiSentryCounter == 2)
94  {
95  m_auiEncounter[1] = uiData;
96  if (Creature* pVazruden = GetSingleCreatureFromStorage(NPC_VAZRUDEN_HERALD))
97  {
98  if (ScriptedAI* pVazrudenAI = dynamic_cast<ScriptedAI*>(pVazruden->AI()))
99  pVazrudenAI->SendAIEvent(AI_EVENT_CUSTOM_A, pVazruden, pVazruden);
100  }
101  }
102 
103  return;
104  }
105  if (uiData == DONE && m_auiEncounter[0] == DONE)
106  {
107  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_FEL_IRON_CHEST : GO_FEL_IRON_CHEST_H, HOUR);
108  DoToggleGameObjectFlags(instance->IsRegularDifficulty() ? GO_FEL_IRON_CHEST : GO_FEL_IRON_CHEST_H, GO_FLAG_NO_INTERACT, false);
109  if (GameObject* chest = GetSingleGameObjectFromStorage(instance->IsRegularDifficulty() ? GO_FEL_IRON_CHEST : GO_FEL_IRON_CHEST_H))
110  if (Player* player = GetPlayerInMap(false, false))
111  chest->GenerateLootFor(player);
112  }
113  if (uiData == FAIL && m_auiEncounter[1] != FAIL)
114  DoFailVazruden();
115 
116  m_auiEncounter[1] = uiData;
117  break;
118  }
119 }

132 void instance_ramparts::DoFailVazruden()
133 {
134  // Store FAIL for both types
135  m_auiEncounter[0] = FAIL;
136  m_auiEncounter[1] = FAIL;
137 
138  // Restore Sentries (counter and respawn them)
139  m_uiSentryCounter = 0;
140  // respawns sentry group
141  m_sentryGroup->Spawn(true);
142 
143  // Respawn or Reset Vazruden the herald
144  if (Creature* pVazruden = GetSingleCreatureFromStorage(NPC_VAZRUDEN_HERALD))
145  {
146  if (!pVazruden->IsAlive())
147  pVazruden->Respawn();
148  else
149  {
150  if (ScriptedAI* pVazrudenAI = dynamic_cast<ScriptedAI*>(pVazruden->AI()))
151  pVazrudenAI->EnterEvadeMode();
152  }
153  }
154 
155  // Despawn Vazruden
156  if (Creature* pVazruden = GetSingleCreatureFromStorage(NPC_VAZRUDEN))
157  pVazruden->ForcedDespawn();
158 }
