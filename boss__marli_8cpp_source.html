113  void ExecuteAction(uint32 action) override
114  {
115  switch (action)
116  {
117  case MARLI_PHASE_CHANGE:
118  if (!m_phaseTwo)
119  {
120  if (DoCastSpellIfCan(nullptr, SPELL_SPIDER_FORM) == CAST_OK)
121  {
122  DoScriptText(SAY_TRANSFORM, m_creature);
123  DoResetThreat();
124  ResetCombatAction(action, 60000);
125  m_creature->SetSpellList(SPELL_LIST_PHASE_2);
126  m_phaseTwo = true;
127  }
128  }
129  else
130  {
131  if (DoCastSpellIfCan(nullptr, SPELL_TRANSFORM_BACK) == CAST_OK)
132  {
133  DoScriptText(SAY_TRANSFORM_BACK, m_creature);
134  m_creature->RemoveAurasDueToSpell(SPELL_SPIDER_FORM);
135  ResetCombatAction(action, 60000);
136  m_creature->SetSpellList(SPELL_LIST_PHASE_1);
137  m_phaseTwo = false;
138  }
139  }
140  break;
141  case MARLI_CHARGE:
142  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_CHARGE, SELECT_FLAG_PLAYER))
143  if (DoCastSpellIfCan(target, SPELL_CHARGE) == CAST_OK)
144  DisableCombatAction(action);
145  break;
146  }
147  }
148 };
