187  template<class T> inline void initLengths(T& cacher)
188  {
189  index_type i = index_lo;
190  lengths.resize(index_hi + 1);
191  length_type prev_length = 0, new_length = 0;
192  while (i < index_hi)
193  {
194  new_length = cacher(*this, i);
195 
196  if (new_length < 0) // length overflowed, assign to max positive value (stop case only?)
197  new_length = std::numeric_limits<length_type>::max();
198 
199  lengths[++i] = new_length;
200 
201  MANGOS_ASSERT(prev_length <= new_length);
202  prev_length = new_length;
203  }
204  }
