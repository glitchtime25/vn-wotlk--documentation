92  void ExecuteAction(uint32 action) override
93  {
94  switch (action)
95  {
96  case ZARITHRIAN_CLEAVE_ARMOR:
97  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE_ARMOR) == CAST_OK)
98  ResetCombatAction(action, 15000);
99  break;
100  case ZARITHRIAN_INTIMIDATING_ROAR:
101  if (DoCastSpellIfCan(m_creature, SPELL_INTIMIDATING_ROAR) == CAST_OK)
102  ResetCombatAction(action, 32000);
103  break;
104  case ZARITHRIAN_SUMMON_FLAMECALLER:
105  if (m_instance)
106  {
107  GuidList m_lStalkersGuidList;
108  m_instance->GetSpawnStalkersGuidList(m_lStalkersGuidList);
109 
110  // each stalker summons a flamekaller; there are 2 on 10 player mode and 4 on 25 player mode
111  for (const auto& guid : m_lStalkersGuidList)
112  if (Creature* pStalker = m_creature->GetMap()->GetCreature(guid))
113  pStalker->CastSpell(pStalker, SPELL_SUMMON_FLAMECALLER, TRIGGERED_OLD_TRIGGERED);
114 
115  DoScriptText(SAY_SUMMON, m_creature);
116  ResetCombatAction(action, 45000);
117  }
118  break;
119  }
120  }
121 };
