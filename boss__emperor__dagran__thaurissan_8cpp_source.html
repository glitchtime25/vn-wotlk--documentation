58  void Aggro(Unit* /*pWho*/) override
59  {
60  switch (urand(0, 2))
61  {
62  case 0: DoScriptText(YELL_AGGRO_1, m_creature); break;
63  case 1: DoScriptText(YELL_AGGRO_2, m_creature); break;
64  case 2: DoScriptText(YELL_AGGRO_3, m_creature); break;

69  void JustDied(Unit* /*pVictim*/) override
70  {
71  if (!m_pInstance)
72  return;
73 
74  if (Creature* pPrincess = m_pInstance->GetSingleCreatureFromStorage(NPC_PRINCESS))
75  {
76  // check if we didn't update the entry
77  if (pPrincess->GetEntry() != NPC_PRINCESS)
78  return;
79 
80  if (pPrincess->IsAlive())
81  {
82  pPrincess->SetFactionTemporary(FACTION_NEUTRAL, TEMPFACTION_NONE);
83  pPrincess->AI()->EnterEvadeMode();
84  }
85  }
86  }
87 
88  void KilledUnit(Unit* /*pVictim*/) override
89  {
90  DoScriptText(YELL_SLAY, m_creature);
91  }
92 
93  void UpdateAI(const uint32 uiDiff) override
94  {
95  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
96  return;
97 
98  if (m_uiHandOfThaurissanTimer < uiDiff)
99  {
100  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
101  {
102  if (DoCastSpellIfCan(pTarget, SPELL_HANDOFTHAURISSAN) == CAST_OK)
103  m_uiHandOfThaurissanTimer = urand(5, 10) * 1000;
104  }
105  }
106  else
107  m_uiHandOfThaurissanTimer -= uiDiff;
108 
109  // AvatarOfFlame_Timer
110  if (m_uiAvatarOfFlameTimer < uiDiff)
111  {
112  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_AVATAROFFLAME) == CAST_OK)
113  m_uiAvatarOfFlameTimer = 18000;
114  }
115  else
116  m_uiAvatarOfFlameTimer -= uiDiff;
117 
118  DoMeleeAttackIfReady();
119  }
120 };
