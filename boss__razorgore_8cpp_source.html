174  void UpdateAI(const uint32 uiDiff) override
175  {
176  // Set visual only on OOC timer
177  if (m_uiIntroVisualTimer)
178  {
179  if (m_uiIntroVisualTimer <= uiDiff)
180  {
181  if (!m_instance)
182  {
183  script_error_log("Instance Blackwing Lair: ERROR Failed to load instance data for this instace.");
184  return;
185  }
186 
187  // If Razorgore is not respawned yet: wait
188  if (Creature* pRazorgore = m_instance->GetSingleCreatureFromStorage(NPC_RAZORGORE))
189  {
190  if (!(pRazorgore->IsAlive()))
191  {
192  m_uiIntroVisualTimer = 2000;
193  return;
194  }
195  }
196 
197  // If Grethok the Controller is here and spawned, start the visual, else wait for him
198  if (Creature* grethok = GetClosestCreatureWithEntry(m_creature, NPC_GRETHOK_CONTROLLER, 2.0f))
199  {
200  if (grethok->IsAlive())
201  {
202  m_creature->CastSpell(m_creature, SPELL_POSSESS_VISUAL, TRIGGERED_OLD_TRIGGERED);
203  grethok->CastSpell(grethok, SPELL_CONTROL_ORB, TRIGGERED_OLD_TRIGGERED);
204  m_uiIntroVisualTimer = 0;
205  }
206  }
207  else
208  m_uiIntroVisualTimer = 2000;
209  }
210  else
211  m_uiIntroVisualTimer -= uiDiff;
212  }
213  }

228 struct DestroyEgg : public SpellScript

230  void OnEffectExecute(Spell* spell, SpellEffectIndex effIdx) const override
231  {
232  GameObject* target = spell->GetGOTarget();
233  Unit* razorgore = spell->GetCaster();
234  if (!target || !razorgore)
235  return;
236 
237  if (ScriptedInstance* pInstance = static_cast<ScriptedInstance*>(target->GetInstanceData()))

239  if (urand(0, 1))
240  {
241  switch (urand(0, 2))
242  {
243  case 0: DoScriptText(SAY_EGGS_BROKEN_1, razorgore); break;
244  case 1: DoScriptText(SAY_EGGS_BROKEN_2, razorgore); break;
245  case 2: DoScriptText(SAY_EGGS_BROKEN_3, razorgore); break;
246  }
247  }
248 
249  // Store the eggs which are destroyed, in order to count them for the second phase
250  pInstance->SetData64(DATA_DRAGON_EGG, target->GetObjectGuid());
