151 void instance_culling_of_stratholme::OnPlayerEnter(Player* pPlayer)
152 {
153  // spawn Chromie
154  DoSpawnChromieIfNeeded(pPlayer);
155 
156  // spawn Arthas if intro is completed / passed
157  if (GetData(TYPE_ARTHAS_INTRO_EVENT) == DONE)
158  {
159  DoSpawnArthasIfNeeded(pPlayer);
160 
161  // resume scourge waves if required - this can only happen in case of server reload
162  if (GetData(TYPE_SALRAMM_EVENT) != DONE)
163  {
164  // this will restart the wave event with a delayed timer
165  if (GetData(TYPE_MEATHOOK_EVENT) == DONE)
166  {
167  if (GetData(TYPE_SALRAMM_EVENT) != IN_PROGRESS)
168  {
169  SetData(TYPE_SALRAMM_EVENT, IN_PROGRESS);
170  m_uiScourgeWaveTimer = 30000;
171  m_uiScourgeWaveCount = 5;
172  }
173  }
174  else
175  {
176  if (GetData(TYPE_MEATHOOK_EVENT) != IN_PROGRESS)
177  {
178  SetData(TYPE_MEATHOOK_EVENT, IN_PROGRESS);
179  m_uiScourgeWaveTimer = 30000;
180  }
181  }
182  }
183  }
184 
185  // Show World States if needed
186  // Grain event world states
187  if (GetData(TYPE_GRAIN_EVENT) == IN_PROGRESS || GetData(TYPE_GRAIN_EVENT) == SPECIAL)
188  pPlayer->SendUpdateWorldState(WORLD_STATE_CRATES, 1); // Show Crates Counter
189  else
190  pPlayer->SendUpdateWorldState(WORLD_STATE_CRATES, 0); // Remove Crates Counter
191 
192  // Scourge waves
193  if (GetData(TYPE_MEATHOOK_EVENT) == IN_PROGRESS || GetData(TYPE_SALRAMM_EVENT) == IN_PROGRESS)
194  pPlayer->SendUpdateWorldState(WORLD_STATE_WAVE, m_uiScourgeWaveCount); // Add WaveCounter
195  else
196  pPlayer->SendUpdateWorldState(WORLD_STATE_WAVE, 0); // Remove WaveCounter
197 
198  // Infinite corruptor
199  if (GetData(TYPE_INFINITE_CORRUPTER_TIME))
200  {
201  DoSpawnCorruptorIfNeeded(pPlayer);
202 
203  pPlayer->SendUpdateWorldState(WORLD_STATE_TIME, 1); // Show Timer
204  pPlayer->SendUpdateWorldState(WORLD_STATE_TIME_COUNTER, GetData(TYPE_INFINITE_CORRUPTER_TIME) / (MINUTE * IN_MILLISECONDS));
205  }
206  else
207  pPlayer->SendUpdateWorldState(WORLD_STATE_TIME, 0); // Remove Timer
208 }

289 void instance_culling_of_stratholme::SetData(uint32 uiType, uint32 uiData)
290 {
291  switch (uiType)
292  {
293  case TYPE_GRAIN_EVENT:
294  m_auiEncounter[uiType] = uiData;
295  if (uiData == SPECIAL)
296  DoUpdateWorldState(WORLD_STATE_CRATES, 1);
297  else if (uiData == IN_PROGRESS)
298  {
299  // safety check
300  if (m_uiGrainCrateCount >= MAX_GRAIN_CRATES)
301  return;
302 
303  ++m_uiGrainCrateCount;
304  DoUpdateWorldState(WORLD_STATE_CRATES_COUNT, m_uiGrainCrateCount);
305 
306  if (m_uiGrainCrateCount == MAX_GRAIN_CRATES)
307  {
308  m_uiRemoveCrateStateTimer = 20000;
309  SetData(TYPE_GRAIN_EVENT, DONE);
310  }
311  }
312  break;
313  case TYPE_ARTHAS_INTRO_EVENT:
314  m_auiEncounter[uiType] = uiData;
315  if (uiData == DONE)
316  {
317  m_uiScourgeWaveCount = 0;
318  m_uiScourgeWaveTimer = 1000;
319  DoUpdateZombieResidents();
320 
321  SetData(TYPE_MEATHOOK_EVENT, IN_PROGRESS);
322  }
323  break;
324  case TYPE_MEATHOOK_EVENT:
325  m_auiEncounter[uiType] = uiData;
326  if (uiData == DONE)
327  {
328  m_uiScourgeWaveTimer = 20000;
329  SetData(TYPE_SALRAMM_EVENT, IN_PROGRESS);
330  }
331  break;
332  case TYPE_SALRAMM_EVENT:
333  m_auiEncounter[uiType] = uiData;
334  if (uiData == DONE)
335  m_uiScourgeWaveTimer = 5000;
336  break;
337  case TYPE_ARTHAS_TOWNHALL_EVENT:
338  m_auiEncounter[uiType] = uiData;
339  if (uiData == DONE)
340  {
341  // despawn arthas and spawn him in the next point
342  if (Creature* pArthas = GetSingleCreatureFromStorage(NPC_ARTHAS))
343  pArthas->ForcedDespawn();
344 
345  if (Player* pPlayer = GetPlayerInMap())
346  DoSpawnArthasIfNeeded(pPlayer);
347  }
348  break;
349  case TYPE_EPOCH_EVENT:
350  m_auiEncounter[uiType] = uiData;
351  break;
352  case TYPE_ARTHAS_ESCORT_EVENT:
353  // use fail in order to respawn Arthas
354  if (uiData == FAIL)
355  {
356  m_uiArthasRespawnTimer = 10000;
357 
358  // despawn the bosses if Arthas dies in order to avoid exploits
359  if (Creature* pEpoch = GetSingleCreatureFromStorage(NPC_LORD_EPOCH, true))
360  pEpoch->ForcedDespawn();
361  if (Creature* pMalganis = GetSingleCreatureFromStorage(NPC_MALGANIS, true))
362  pMalganis->ForcedDespawn();
363  }
364  else
365  m_auiEncounter[uiType] = uiData;
366  break;
367  case TYPE_MALGANIS_EVENT:
368  m_auiEncounter[uiType] = uiData;
369  if (uiData == DONE)
370  {
371  DoUseDoorOrButton(GO_CITY_ENTRANCE_GATE);
372  DoToggleGameObjectFlags(instance->IsRegularDifficulty() ? GO_DARK_RUNED_CHEST : GO_DARK_RUNED_CHEST_H, GO_FLAG_NO_INTERACT, false);
373  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_DARK_RUNED_CHEST : GO_DARK_RUNED_CHEST_H, 30 * MINUTE);
374 
375  if (Player* pPlayer = GetPlayerInMap())
376  DoSpawnChromieIfNeeded(pPlayer);
377  }
378  break;
379  case TYPE_INFINITE_CORRUPTER_TIME:
380  m_auiEncounter[uiType] = uiData;
381  if (!uiData)
382  {
383  DoUpdateWorldState(WORLD_STATE_TIME, 0); // Remove Timer
384  DoUpdateWorldState(WORLD_STATE_TIME_COUNTER, 0);
385  }
386  else
387  DoUpdateWorldState(WORLD_STATE_TIME_COUNTER, uiData / (MINUTE * IN_MILLISECONDS));
388  break;
389  case TYPE_INFINITE_CORRUPTER:
390  m_auiEncounter[uiType] = uiData;
391  switch (uiData)
392  {
393  case IN_PROGRESS:
394  if (!GetData(TYPE_INFINITE_CORRUPTER_TIME))
395  {
396  SetData(TYPE_INFINITE_CORRUPTER_TIME, MINUTE * 25 * IN_MILLISECONDS);
397  DoUpdateWorldState(WORLD_STATE_TIME, 1);
398  DoChromieWhisper(WHISPER_CHROMIE_GUARDIAN);
399 
400  // spawn the corruptor for the first time
401  if (Creature* pArthas = GetSingleCreatureFromStorage(NPC_ARTHAS))
402  DoSpawnCorruptorIfNeeded(pArthas);
403  }
404  break;
405  case DONE:
406  // event completed - epilog handled by dbscript
407  SetData(TYPE_INFINITE_CORRUPTER_TIME, 0);
408  break;
409  case SPECIAL:
410  DoChromieWhisper(WHISPER_CHROMIE_HURRY);
411  break;
412  case FAIL:
413  // event failed - despawn the corruptor
414  SetData(TYPE_INFINITE_CORRUPTER_TIME, 0);
415  if (Creature* pCorrupter = GetSingleCreatureFromStorage(NPC_INFINITE_CORRUPTER))
416  {
417  DoOrSimulateScriptTextForThisInstance(SAY_CORRUPTOR_DESPAWN, NPC_INFINITE_CORRUPTER);
418 
419  if (pCorrupter->IsAlive())
420  pCorrupter->ForcedDespawn();
421  }
422  break;
423  }
424  break;
425  }
426 
427  if (uiData == DONE || uiType == TYPE_INFINITE_CORRUPTER_TIME)
428  {
429  OUT_SAVE_INST_DATA;
430 
431  std::ostringstream saveStream;
432  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
433  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
434  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
435  << m_auiEncounter[9];
436 
437  m_strInstData = saveStream.str();
438 
439  SaveToDB();
440  OUT_SAVE_INST_DATA_COMPLETE;
441  }
442 }

502  m_auiEncounter[i] = NOT_STARTED;
503  }
504  }
505 
506  // If already started counting down time, the event is "in progress"
507  if (m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME])
508  m_auiEncounter[TYPE_INFINITE_CORRUPTER] = IN_PROGRESS;
509 
510  OUT_LOAD_INST_DATA_COMPLETE;

861 void instance_culling_of_stratholme::Update(uint32 uiDiff)
862 {
863  // 25min Run - decrease time, update worldstate every ~20s
864  // as the time is always saved by m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME], there is no need for an extra timer
865  if (m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME])
866  {
867  if (m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME] <= uiDiff)
868  SetData(TYPE_INFINITE_CORRUPTER, FAIL);
869  else
870  {
871  m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME] -= uiDiff;
872  if (m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME] / IN_MILLISECONDS % 20 == 0)
873  SetData(TYPE_INFINITE_CORRUPTER_TIME, m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME]);
874  }
875 
876  // This part is needed for a small "hurry up guys" note, TODO, verify 20min
877  if (m_auiEncounter[TYPE_INFINITE_CORRUPTER] == IN_PROGRESS && m_auiEncounter[TYPE_INFINITE_CORRUPTER_TIME] <= 24 * MINUTE * IN_MILLISECONDS)
878  SetData(TYPE_INFINITE_CORRUPTER, SPECIAL);
879  }
880 
881  // Small Timer, to remove Grain-Crate WorldState and Spawn Second Chromie
882  if (m_uiRemoveCrateStateTimer)
883  {
884  if (m_uiRemoveCrateStateTimer <= uiDiff)
885  {
886  if (Player* pPlayer = GetPlayerInMap())
887  DoSpawnChromieIfNeeded(pPlayer);
888 
889  DoUpdateWorldState(WORLD_STATE_CRATES, 0);
890  DoChromieWhisper(WHISPER_CHROMIE_CRATES);
891  m_uiRemoveCrateStateTimer = 0;
892  }
893  else
894  m_uiRemoveCrateStateTimer -= uiDiff;
895  }
896 
897  // Respawn Arthas after some time
898  if (m_uiArthasRespawnTimer)
899  {
900  if (m_uiArthasRespawnTimer <= uiDiff)
901  {
902  if (Player* pPlayer = GetPlayerInMap())
903  DoSpawnArthasIfNeeded(pPlayer);
904 
905  m_uiArthasRespawnTimer = 0;
906  }
907  else
908  m_uiArthasRespawnTimer -= uiDiff;
909  }
910 
911  // Handle undead waves
912  if (m_uiScourgeWaveTimer)
913  {
914  if (m_uiScourgeWaveTimer <= uiDiff)
915  {
916  if (GetData(TYPE_SALRAMM_EVENT) == DONE)
917  {
918  DoOrSimulateScriptTextForThisInstance(SAY_MEET_TOWN_HALL, NPC_ARTHAS);
919  DoUpdateWorldState(WORLD_STATE_WAVE, 0); // Remove WaveCounter

923  pArthas->ForcedDespawn();
924  if (Player* pPlayer = GetPlayerInMap())
925  DoSpawnArthasIfNeeded(pPlayer);
926  }
