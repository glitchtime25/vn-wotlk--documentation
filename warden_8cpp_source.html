111 void Warden::HandleChallengeResponse(ByteBuffer &buff)
112 {
113  buff.rpos(buff.wpos());
114 
115  // shouldn't happen under normal conditions
116  if (!_crk)
117  {
118  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Challenge response without request being sent");
119  _session->KickPlayer();
120  return;
121  }
122 
123  // Verify key
124  if (buff.wpos() != 1 + sizeof(_crk->reply) || !!memcmp(buff.contents() + 1, _crk->reply, sizeof(_crk->reply)))
125  {
126  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Challenge response failed");
127  _session->KickPlayer();
128  return;
129  }
130 
131  StopTimeoutClock();
132 
133  _inputCrypto.Init(_crk->clientKey);
134  _outputCrypto.Init(_crk->serverKey);
135 
136  _xor = _crk->clientKey[0];
137 
138  sLog.outDebug("WARDEN: Challenge response validated. Warden packet encryption initialized.");
139 
140  _crk = nullptr;

182 void Warden::RequestScans(std::vector<std::shared_ptr<const Scan>> &&scans)
183 {
184  // start by appending the requested scans to the queue, which will usually be empty when this is called
185  _enqueuedScans.insert(_enqueuedScans.end(), scans.cbegin(), scans.cend());
186 
187  // if for whatever reason there is nothing to do, stop
188  if (_enqueuedScans.empty())
189  return;
190 
191  // if there are already some scans pending, stop
192  if (!_pendingScans.empty())
193  return;
194 
195  // begin building the scan request, taking as many scans from the queue as we can
196 
197  std::vector<std::string> strings;
198  ByteBuffer scan;
199 
200  size_t request = 0, reply = 0;
201 
202  bool queueUpdated = false;
203 
204  for (auto i = _enqueuedScans.begin(); i != _enqueuedScans.end(); ++i)
205  {
206  // too many or too big? keep the current scan, as well as all following it, in the queue, and halt
207  if (request + (*i)->requestSize > MaxRequest || reply + (*i)->replySize > MaxReply ||
208  _pendingScans.size() >= sAnticheatConfig.GetWardenScanCount())
209  {
210  _enqueuedScans = std::vector<std::shared_ptr<const Scan>>(i, _enqueuedScans.end());
211  queueUpdated = true;
212  break;
213  }
214 
215  // if we can fit this one, do it
216 
217  auto const startSize = scan.wpos();
218  auto const startStringSize = strings.size();
219 
220  (*i)->Build(this, strings, scan);
221 
222  // if the scan did not change the buffer size or the string size, consider
223  // it a NOP and don't bother marking it as pending
224  if (scan.wpos() != startSize || strings.size() != startStringSize)
225  _pendingScans.push_back(*i);
226  }
227 
228  // if there are still no pending scans, it means that there is a single scan which is too big.
229  // this should never happen, so if it does, just crash
230  MANGOS_ASSERT(!_pendingScans.empty());
231 
232  // if the scan queue has not been updated, its because we were able to fit the entire queue into one request.
233  // therefore, the queue can be emptied
234  if (!queueUpdated)
235  _enqueuedScans.clear();
236 
237  // this buffer is where we will combine the different portions of the packet
238  ByteBuffer buff;
239 
240  // warden opcode
241  buff << static_cast<uint8>(WARDEN_SMSG_CHEAT_CHECKS_REQUEST);
242 
243  if (_session->GetOS() == CLIENT_OS_WIN)
244  {
245  // string table for this request
246  for (auto const& s : strings)
247  {
248  buff << static_cast<uint8>(s.length());
249  buff.append(s.c_str(), s.length());
250  }
251 
252  // end of string table
253  buff << static_cast<uint8>(0);
254  }
255 
256  // all scan requests
257  buff.append(scan);
258 
259  if (_session->GetOS() == CLIENT_OS_WIN)
260  {
261  // indicates to the client that there are no further requests in this packet
262  buff << _xor;
263  }
264 
265  BeginTimeoutClock();
266  SendPacket(buff);
267 }

269 void Warden::ReadScanResults(ByteBuffer &buff)
270 {
271  // move the scans to the stack because the check handlers we call may enqueue additional scans
272  auto const scans = std::move(_pendingScans);
273 
274  auto const inWorld = _session->GetPlayer() && _session->GetPlayer()->IsInWorld();
275 
276  for (auto const &s : scans)
277  {
278  // checks return true when they have discovered a hack
279  if (s->Check(this, buff))
280  {
281  // if this scan requires being in the world and they are not in the world (meaning they left
282  // the world between the request and response), ignore the results
283  if (!!(s->flags & InWorld) && !inWorld)
284  continue;
285 
286  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Check \"%s\" failed", s->comment.c_str());
287  }
288  }
289 
290  // if there are enqueued scans at this point, the check handlers must have requested them.
291  // when this happens, request them to be sent immediately.
292  if (!_enqueuedScans.empty())
293  RequestScans({});
294 }

343 uint32 Warden::BuildChecksum(const uint8* data, size_t size)
344 {
345  uint8 hash[Sha1Hash::GetLength()];
346  SHA1(data, size, hash);
347 
348  uint32 checkSum = 0;
349 
350  for (auto i = 0u; i < sizeof(hash) / sizeof(uint32); ++i)
351  checkSum ^= *reinterpret_cast<uint32 *>(&hash[i * 4]);
352 
353  return checkSum;
354 }

356 void Warden::HandlePacket(WorldPacket& recvData)
357 {
358  // initialize decrypt packet
359  DecryptData(const_cast<uint8*>(recvData.contents()), recvData.size());
360 
361  uint8 opcode;
362  recvData >> opcode;
363 
364  // when there is a challenge/response pending, the only packet we expect is the hash result
365  if (!!_crk && opcode != WARDEN_CMSG_HASH_RESULT)
366  {
367  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Unexpected opcode (%u) received while awaiting hash challenge response", opcode);
368  recvData.rpos(recvData.wpos());
369  _session->KickPlayer();
370  return;
371  }
372 
373  switch (opcode)
374  {
375  case WARDEN_CMSG_MODULE_MISSING:
376  {
377  if (!_module)
378  {
379  sLog.outBasic("WARDEN: Account %u IP %s requested module when none was offered.", _session->GetAccountId(), _session->GetRemoteAddress().c_str());
380  _session->KickPlayer();
381  break;
382  }
383 
384  if (_moduleSendPending)
385  {
386  sLog.outBasic("WARDEN: Account %u IP %s failed to load module. Kicking.", _session->GetAccountId(), _session->GetRemoteAddress().c_str());
387  _session->KickPlayer();
388  _moduleSendPending = false;
389  break;
390  }
391 
392  sLog.outBasic("WARDEN: Account - %u get opcode 00 - Load module failed or module is missing...", _session->GetAccountId());
393  SendModuleToClient();
394  break;
395  }
396 
397  case WARDEN_CMSG_MODULE_OK:
398  {
399  if (!_module)
400  {
401  sLog.outBasic("WARDEN: Account %u IP %s loaded module without server request.", _session->GetAccountId(), _session->GetRemoteAddress().c_str());
402  _session->KickPlayer();
403  break;
404  }
405 
406  _moduleSendPending = false;
407  sLog.outBasic("WARDEN: Account - %u get opcode 01 - Module has loaded. Recv answer....", _session->GetAccountId());
408  RequestChallenge();
409  break;
410  }
411 
412  case WARDEN_CMSG_CHEAT_CHECKS_RESULT:
413  {
414  if (_session->GetOS() == CLIENT_OS_WIN)
415  {
416  // verify checksum integrity
417  uint16 length;
418  uint32 checksum;
419  recvData >> length >> checksum;
420 
421  if (BuildChecksum(recvData.contents() + recvData.rpos(), length) != checksum)
422  {
423  recvData.rpos(recvData.wpos());
424  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Packet checksum fail");
425  _session->KickPlayer();
426  return;
427  }
428  }
429 
430  // this function will also act on the results
431  ReadScanResults(recvData);
432 
433  StopTimeoutClock();
434  BeginScanClock();
435 
436  break;
437  }
438 
439  // FIXME: Find when/why/how this actually happens and how to handle it
440  case WARDEN_CMSG_MEM_CHECKS_RESULT:
441  {
442  sLog.outDebug("WARDEN: Account - %u received opcode 03", _session->GetAccountId());
443  break;
444  }
445 
446  case WARDEN_CMSG_HASH_RESULT:
447  {
448  sLog.outDebug("WARDEN: Account - %u received opcode 04", _session->GetAccountId());
449 
450  HandleChallengeResponse(recvData);
451 
452  // if the challenge failed, this will still be set. by now, the session will be closed. do nothing further.
453  if (!!_crk)
454  return;
455 
456  // at this point the client has our module loaded. send whatever packets are necessary to initialize Warden
457  InitializeClient();
458 
459  // send any initial hack scans that the scan manager may have for us
460  RequestScans(SelectScans(ScanFlags::InitialLogin));
461 
462  // begin the scan clock (note that even if the clock expires before any initial scans are answered, no new
463  // checks will be requested until the reply is received).
464  BeginScanClock();
465 
466  break;
467  }
468 
469  case WARDEN_CMSG_MODULE_FAILED:
470  {
471  sLog.outBasic("WARDEN: Account - %u received opcode 05 - Module load failed!", _session->GetAccountId());
472  _session->KickPlayer();
473  break;
474  }
475 
476  default:
477  {
478  _anticheat->RecordCheatInternal(CheatType::CHEAT_TYPE_WARDEN, "Received unknown opcode %02X of size %u.", opcode, recvData.size() - 1);
479  _session->KickPlayer();
480  break;
481  }
482  }
483 }

485 void Warden::Update(uint32 /*diff*/)
486 {
487  if (!!_timeoutClock && WorldTimer::getMSTime() > _timeoutClock)
488  {
489  sLog.outBasic("WARDEN: Account %u ip %s timeout", _session->GetAccountId(), _session->GetRemoteAddress().c_str());
490  _session->KickPlayer();
491  return;
492  }
493 
494  if (_pendingScans.empty())
495  {
496  // if there are enqueued scans which may now be requested, do so immediately (with no additional scans)
497  if (!_enqueuedScans.empty())
498  RequestScans({});
499  // otherwise, if the scan clock is running and has expired, request randomly selected scans
500  else if (!!_scanClock && WorldTimer::getMSTime() > _scanClock)
501  {
502  auto const inWorld = _session->GetPlayer() ? _session->GetPlayer()->IsInWorld() : false;
503 
504  RequestScans(SelectScans(inWorld ? InWorld : None));
505 
506  // if no scans were found, just begin the clock again, in case things change
507  if (_pendingScans.empty())
508  BeginScanClock();
509  else
510  {
511  // let us keep things simple and not request checks while others are pending, but only if scans were found and requested
512  StopScanClock();
513  BeginTimeoutClock();
514  }
515  }
516  }
