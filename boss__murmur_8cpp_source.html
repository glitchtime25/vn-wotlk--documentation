134  void UpdateAI(const uint32 uiDiff) override
135  {
136  if (!m_creature->IsInCombat())
137  {
138  if (m_uiAttackTimer)
139  {
140  if (m_uiAttackTimer <= uiDiff)
141  {
142  // kill one that's moving
143  if (urand(0, 1))
144  {
145  GuidVector moversVector;
146  for (ObjectGuid& guid : spellbindersVector)
147  {
148  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
149  {
150  if (creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
151  {
152  moversVector.push_back(guid);
153  }
154  }
155  }
156  for (ObjectGuid& guid : summonersVector)
157  {
158  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
159  {
160  if (creature->GetMotionMaster()->GetCurrentMovementGeneratorType() == WAYPOINT_MOTION_TYPE)
161  {
162  moversVector.push_back(guid);
163  }
164  }
165  }
166  if (moversVector.size() > 0)
167  {
168  if (ObjectGuid& guid = moversVector[urand(0, moversVector.size() - 1)])
169  {
170  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
171  {
172  DoCastSpellIfCan(creature, SPELL_MURMURS_WRATH);
173  }
174  }
175  }
176  }
177  // stun 5 targets
178  else
179  DoCastSpellIfCan(m_creature, SPELL_SUPPRESSION_BLAST);
180 
181  m_uiAttackTimer = 3000;
182  }
183  else
184  m_uiAttackTimer -= uiDiff;
185  }
186 
187  if (m_uiCastersAttackMurmurTimer)
188  {
189  if (m_uiCastersAttackMurmurTimer <= uiDiff)
190  {
191  for (ObjectGuid& guid : spellbindersVector)
192  {
193  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
194  {
195  m_creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_EVENTAI_A, m_creature, creature);
196  }
197  }
198  for (ObjectGuid& guid : summonersVector)
199  {
200  if (Creature* creature = m_creature->GetMap()->GetCreature(guid))
201  {
202  m_creature->AI()->SendAIEvent(AI_EVENT_CUSTOM_EVENTAI_A, m_creature, creature);
203  }
204  }
205  m_uiCastersAttackMurmurTimer = urand(3000, 8000);
206  }
207  else
208  m_uiCastersAttackMurmurTimer -= uiDiff;
209  }
210  }
211 
212  // Return since we have no target
213  if (!m_creature->SelectHostileTarget())
214  return;
215 
216  // SonicBoom_Timer
217  if (m_uiSonicBoomTimer < uiDiff)
218  {
219  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_SONIC_BOOM : SPELL_SONIC_BOOM_H) == CAST_OK)
220  {
221  DoScriptText(EMOTE_SONIC_BOOM, m_creature);
222  m_uiSonicBoomTimer = urand(SPELL_SONIC_BOOM_MIN_TIMER, SPELL_SONIC_BOOM_MAX_TIMER);
223  }
224  }
225  else
226  m_uiSonicBoomTimer -= uiDiff;
227 
228  // MurmursTouch_Timer
229  if (m_uiMurmursTouchTimer < uiDiff)
230  {
231  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_MURMURS_TOUCH : SPELL_MURMURS_TOUCH_H) == CAST_OK)
232  m_uiMurmursTouchTimer = m_bIsRegularMode ? SPELL_MURMURS_TOUCH_TIMER_N : urand(SPELL_MURMURS_TOUCH_MIN_TIMER_H, SPELL_MURMURS_TOUCH_MAX_TIMER_H);
233  }
234  else
235  m_uiMurmursTouchTimer -= uiDiff;
236 
237  if (m_bIsRegularMode)
238  {
239  // Magnetic Pull
240  if (m_uiMagneticPullTimer < uiDiff)
241  {
242  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MAGNETIC_PULL, SELECT_FLAG_PLAYER))
243  {
244  if (DoCastSpellIfCan(pTarget, SPELL_MAGNETIC_PULL) == CAST_OK)
245  m_uiMagneticPullTimer = urand(SPELL_MAGNETIC_PULL_MIN_TIMER, SPELL_MAGNETIC_PULL_MAX_TIMER);
246  }
247  }
248  else
249  m_uiMagneticPullTimer -= uiDiff;
250  }
251  else
252  {
253  //Sonic Shock
254  if (m_uiSonicShockTimer < uiDiff)
255  {
256  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SONIC_SHOCK, SELECT_FLAG_PLAYER | SELECT_FLAG_IN_MELEE_RANGE))
257  {
258  if (DoCastSpellIfCan(pTarget, SPELL_SONIC_SHOCK) == CAST_OK)
259  m_uiSonicShockTimer = urand(SPELL_SONIC_SHOCK_MIN_TIMER, SPELL_SONIC_SHOCK_MAX_TIMER);
260  }
261  }
262  else
263  m_uiSonicShockTimer -= uiDiff;
264 
265  // Thundering Storm
266  if (m_uiThunderingStormTimer < uiDiff)
267  {
268  m_uiThunderingStormTimer = 0;
269  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_FARTHEST_AWAY, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_RANGE_AOE_RANGE, m_thunderingParams))
270  {
271  if (DoCastSpellIfCan(target, SPELL_THUNDERING_STORM) == CAST_OK)
272  m_uiThunderingStormTimer = urand(SPELL_THUNDERING_STORM_MIN_TIMER, SPELL_THUNDERING_STORM_MAX_TIMER);
273  }
274  }
275  else
276  m_uiThunderingStormTimer -= uiDiff;
277  }
278 
279  // Resonance_Timer - cast if no target is in range
280  if (m_creature->GetVictim() && m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
281  DoMeleeAttackIfReady();
282  else
283  {
284  if (m_uiResonanceTimer < uiDiff)
285  {
286  if (DoCastSpellIfCan(m_creature, SPELL_RESONANCE) == CAST_OK)
287  m_uiResonanceTimer = urand(SPELL_RESONANCE_MIN_TIMER, SPELL_RESONANCE_MAX_TIMER);
288  }
289  else
290  m_uiResonanceTimer -= uiDiff;
291  }
292  }
