147  void JustDidDialogueStep(int32 iEntry) override
148  {
149  switch (iEntry)
150  {
151  case TYPE_WARDEN_2:
152  if (m_pInstance)
153  m_pInstance->SetData(TYPE_WARDEN_2, DONE);
154  break;
155  case SAY_WATER:
156  DoCastSpellIfCan(m_creature, SPELL_CONJURE_WATER);
157  break;
158  case SAY_BUFFS:
159  DoCastSpellIfCan(m_creature, SPELL_ARCANE_INTELLECT);
160  break;
161  case SPELL_ICE_ARMOR:
162  DoCastSpellIfCan(m_creature, SPELL_ICE_ARMOR);
163  break;
164  case SAY_DRINK:
165  DoCastSpellIfCan(m_creature, SD_SPELL_DRINK);
166  break;
167  case POINT_ID_CENTER:
168  m_creature->SetWalk(false);
169  m_creature->GetMotionMaster()->MovePoint(1, fRoomCenterCoords[0], fRoomCenterCoords[1], fRoomCenterCoords[2]);
170  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
171  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC);
172  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
173  break;
174  }
175  }

177  void UpdateAI(const uint32 uiDiff) override
178  {
179  DialogueUpdate(uiDiff);
180 
181  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
182  return;
183 
184  if (!m_bHasLowHp && m_creature->GetHealthPercent() < 20.0f)
185  {
186  DoScriptText(SAY_LOWHP, m_creature);
187  m_bHasLowHp = true;
188  }
189 
190  if (m_uiPyroblastTimer < uiDiff)
191  {
192  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_PYROBLAST) == CAST_OK)
193  {
194  m_uiPyroblastTimer = 40000;
195  DoScriptText(SAY_PYRO, m_creature);
196  }
197  }
198  else
199  m_uiPyroblastTimer -= uiDiff;
200 
201  if (m_uiFireballTimer < uiDiff)
202  {
203  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FIREBALL) == CAST_OK)
204  m_uiFireballTimer = 4000;
205  }
206  else
207  m_uiFireballTimer -= uiDiff;
208 
209  if (m_uiFrostBoltTimer < uiDiff)
210  {
211  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FROSTBOLT) == CAST_OK)
212  m_uiFrostBoltTimer = urand(4000, 6000);
213  }
214  else
215  m_uiFrostBoltTimer -= uiDiff;
216 
217  if (m_uiConeColtTimer < uiDiff)
218  {
219  if (DoCastSpellIfCan(m_creature, SPELL_CONE_OF_COLD) == CAST_OK)
220  m_uiConeColtTimer = urand(7000, 12000);
221  }
222  else
223  m_uiConeColtTimer -= uiDiff;
224 
225  if (m_uiFireBlastTimer < uiDiff)
226  {
227  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FIRE_BLAST) == CAST_OK)
228  m_uiFireBlastTimer = urand(5000, 16000);
229  }
230  else
231  m_uiFireBlastTimer -= uiDiff;
232

234  {

236  m_uiArcaneMissileTimer = urand(5000, 8000);
237  }
238  else
239  m_uiArcaneMissileTimer -= uiDiff;
240 
241  DoMeleeAttackIfReady();
