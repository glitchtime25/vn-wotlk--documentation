257  // function to handle toss head phase
258  void DoTossHead()
259  {
260  // in the first transition; spawn the head
261  if (m_creature->HasAura(SPELL_BODY_STAGE_1))
262  m_creature->SummonCreature(NPC_HEAD_OF_HORSEMAN, m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ(), 0, TEMPSPAWN_DEAD_DESPAWN, 0);
263 
264  // make head available
265  if (Creature* head = m_creature->GetMap()->GetCreature(m_headGuid))
266  {
267  head->CastSpell(m_creature, SPELL_HEAD_REPOSITIONS, TRIGGERED_OLD_TRIGGERED); // also should use SPELL_COMMAND_HEAD_REPOSITIONS
268  DoCastSpellIfCan(head, SPELL_SEND_HEAD, CAST_TRIGGERED);
269  }
270 
271  // only from second transition we start whirlwind
272  if (m_creature->HasAura(SPELL_BODY_STAGE_2) || m_creature->HasAura(SPELL_BODY_STAGE_3))
273  DoCastSpellIfCan(m_creature, SPELL_WHIRLWIND, CAST_TRIGGERED);
274 
275  // remove head visual and set transition phase auras
276  m_creature->RemoveAurasDueToSpell(SPELL_HEAD_VISUAL);
277  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN_PROC, CAST_TRIGGERED);
278  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN, CAST_TRIGGERED);
279  DoCastSpellIfCan(m_creature, SPELL_BODY_REGEN_CONFUSE, CAST_TRIGGERED);
280 
281  m_fightPhase = PHASE_HEAD_TOSS;
282  m_bHeadRequested = false;
283  }
284 
285  // function to handle the head rejoin
286  void DoRejoinHead()
287  {
288  // remove transition auras and set the head visual
289  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN_CONFUSE);
290  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN);
291  m_creature->RemoveAurasDueToSpell(SPELL_BODY_REGEN_PROC);
292  m_creature->RemoveAurasDueToSpell(SPELL_WHIRLWIND);
293 
294  DoScriptText(SAY_REJOINED, m_creature);
295  DoCastSpellIfCan(m_creature, SPELL_BODY_HEAD_VISUAL, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
296 
297  // switch from phase 1 to phase 2
298  if (m_creature->HasAura(SPELL_BODY_STAGE_1))
299  {
300  m_fightPhase = PHASE_CONFLAGRATION;
301  m_creature->RemoveAurasDueToSpell(SPELL_BODY_STAGE_1);
302  DoCastSpellIfCan(m_creature, SPELL_BODY_STAGE_2, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
303  }
304  // switch from phase 2 to phase 3 or repeat phase 3
305  else if (m_creature->HasAura(SPELL_BODY_STAGE_2) || m_creature->HasAura(SPELL_BODY_STAGE_3))
306  {
307  m_fightPhase = PHASE_PUMPKINS;
308  m_creature->RemoveAurasDueToSpell(SPELL_BODY_STAGE_2);
309  DoCastSpellIfCan(m_creature, SPELL_BODY_STAGE_3, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
310  DoCastSpellIfCan(m_creature, SPELL_SUMMON_PUMPKIN, CAST_TRIGGERED);
311  }
312  }

314  void UpdateAI(const uint32 uiDiff) override
315  {
316  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
317  return;
318 
319  switch (m_fightPhase)
320  {
321  case PHASE_PUMPKINS:
322 
323  if (m_uiPumpkinTimer < uiDiff)
324  {
325  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_PUMPKIN) == CAST_OK)
326  m_uiPumpkinTimer = urand(35000, 40000);
327  }
328  else
329  m_uiPumpkinTimer -= uiDiff;
330 
331  // no break;
332  case PHASE_CONFLAGRATION:
333 
334  // conflagration not happening during pumpkin phase
335  if (m_fightPhase != PHASE_PUMPKINS)
336  {
337  if (m_uiConflagrationTimer < uiDiff)
338  {
339  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
340  {
341  if (DoCastSpellIfCan(pTarget, SPELL_CONFLAGRATION) == CAST_OK)
342  {
343  DoScriptText(SAY_CONFLAGRATION, m_creature);
344  m_uiConflagrationTimer = urand(15000, 20000);
345  }
346  }
347  }
348  else
349  m_uiConflagrationTimer -= uiDiff;
350  }
351 
352  // no break;
353  case PHASE_HORSEMAN:
354 
355  // cleave - all phases
356  if (m_uiCleaveTimer < uiDiff)
357  {
358  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_HORSEMAN_CLEAVE) == CAST_OK)
359  m_uiCleaveTimer = 5000;
360  }
361  else
362  m_uiCleaveTimer -= uiDiff;
363 
364  DoMeleeAttackIfReady();
365  break;
366  case PHASE_HEAD_TOSS:
367  // rejoin head by force at 100% hp
368  if (!m_bHeadRequested && m_creature->GetHealthPercent() == 100.0f)
369  {
370  if (Creature* pHead = m_creature->GetMap()->GetCreature(m_headGuid))
371  SendAIEvent(AI_EVENT_CUSTOM_B, m_creature, pHead);
372 
373  m_bHeadRequested = true;
374  }
375  break;
376  }
377  }
