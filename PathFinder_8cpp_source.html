136 dtPolyRef PathFinder::getPathPolyByPosition(const dtPolyRef* polyPath, uint32 polyPathSize, const float* point, float* distance) const
137 {
138  if (!polyPath || !polyPathSize)
139  return INVALID_POLYREF;
140 
141  dtPolyRef nearestPoly = INVALID_POLYREF;
142  float minDist3d = std::numeric_limits<float>::max();
143 
144  for (uint32 i = 0; i < polyPathSize; ++i)
145  {
146  float closestPoint[VERTEX_SIZE];
147  if (dtStatusFailed(m_navMeshQuery->closestPointOnPoly(polyPath[i], point, closestPoint, nullptr)))
148  continue;
149 
150  float d = dtVdistSqr(point, closestPoint);
151  if (d < minDist3d)
152  {
153  minDist3d = d;
154  nearestPoly = polyPath[i];
155  }
156 
157  if (minDist3d < 1.0f) // shortcut out - close enough for us
158  break;
159  }
160 
161  if (distance)
162  *distance = dtMathSqrtf(minDist3d);
163 
164  return (minDist3d < 3.0f) ? nearestPoly : INVALID_POLYREF;
165 }

167 dtPolyRef PathFinder::getPolyByLocation(const float* point, float* distance)
168 {
169  // first we check the current path
170  // if the current path doesn't contain the current poly,
171  // we need to use the expensive navMesh.findNearestPoly
172  dtPolyRef polyRef = getPathPolyByPosition(m_pathPolyRefs.data(), m_polyLength, point, distance);
173  if (polyRef != INVALID_POLYREF)
174  return polyRef;
175 
176  // we don't have it in our old path
177  // try to get it by findNearestPoly()
178  // first try with NearPolySearchBound
179  float closestPoint[VERTEX_SIZE] = {0.0f, 0.0f, 0.0f};
180  dtStatus dtResult = m_navMeshQuery->findNearestPoly(point, NearPolySearchBound, &m_filter, &polyRef, closestPoint);
181  if (dtStatusSucceed(dtResult) && polyRef != INVALID_POLYREF)
182  {
183  *distance = dtVdist(closestPoint, point);
184  m_pathPolyRefs.push_back(polyRef);
185  return polyRef;
186  }
187 
188  // still nothing ..
189  // try with bigger search box
190  // From dtNavMeshQuery::findNearestPoly: "If the search extents overlaps more than
191  // 128 polygons it may return an invalid result".We just use FarPolySearchBound.
192 
193  dtResult = m_navMeshQuery->findNearestPoly(point, FarPolySearchBound, &m_filter, &polyRef, closestPoint);
194  if (dtStatusSucceed(dtResult) && polyRef != INVALID_POLYREF)
195  {
196  *distance = dtVdist(closestPoint, point);
197  m_pathPolyRefs.push_back(polyRef);
198  return polyRef;
199  }
200 
201  return INVALID_POLYREF;
202 }

210  if (distance > 300.f)

263  buildShotrcut = true;
264  }
265  else

550 void PathFinder::BuildPointPath(const float* startPoint, const float* endPoint)
551 {
552  if (m_pointPathLimit * VERTEX_SIZE > m_cachedPoints.size())
553  m_cachedPoints.resize(m_pointPathLimit * VERTEX_SIZE);
554  uint32 pointCount = 0;
555  dtStatus dtResult = DT_FAILURE;
556 
557  if (m_straightLine)
558  {
559  Vector3 startVec = Vector3(startPoint[2], startPoint[0], startPoint[1]);
560  Vector3 endVec = Vector3(endPoint[2], endPoint[0], endPoint[1]);
561  Vector3 pathDir = (endVec - startVec);
562  float pathLength = pathDir.magnitude();
563 
564  // not sure about the exact limit value below but it should be ok
565  if (pathLength > 0.1f)
566  {
567  float stepCountFloat = pathLength / SMOOTH_PATH_STEP_SIZE;
568  uint32 stepCount = static_cast<uint32>(stepCountFloat);
569  float stepPart = stepCountFloat - stepCount;
570 
571  // merge last point with previous point if it is 30% less than SMOOTH_PATH_STEP_SIZE
572  if (stepCount > 1.0f && stepPart < 0.3f)
573  --stepCount;
574 
575  Vector3 prevVec = startVec;
576  Vector3 diffVec = pathDir * (SMOOTH_PATH_STEP_SIZE / pathLength);
577 
578  // little optimization
579  m_pathPoints.clear();
580  m_pathPoints.reserve(stepCount + 2);
581 
582  // first point
583  m_pathPoints.push_back(startVec);
584 
585  float distanceToPoly;

646  }

650

653  {

670  }

676  {
677  if (tempPathPoints[i] != emptyVec)

681  }
682  }
683

685  }

688  m_pathPoints.resize(pointCount);
689  for (uint32 i = 0; i < pointCount; ++i)
690  {

692  m_pathPoints[i] = { pathPoints[pointPos + 2], pathPoints[pointPos], pathPoints[pointPos + 1] };
693  }
694  }
695  }
696

700  // single point paths can be generated here
701  // TODO : check the exact cases
702  DEBUG_FILTER_LOG(LOG_FILTER_PATHFINDING, "++ PathFinder::BuildPointPath FAILED! path sized %d returned\n", pointCount);
703  BuildShortcut();
704  m_type = PATHFIND_NOPATH;
705  return;
706  }
707 
708  if (pointCount == m_pointPathLimit)
709  {
710  DEBUG_FILTER_LOG(LOG_FILTER_PATHFINDING, "BuildPointPath FAILED! path sized %d returned, lower than limit set to %d\n", pointCount, m_pointPathLimit);
711  BuildShortcut();
712  m_type = PATHFIND_SHORT;
713  return;
714  }
715 
716  // first point is always our current location - we need the next one
717  setActualEndPosition(m_pathPoints[pointCount - 1]);
718 
719  // force the given destination, if needed
720  if (m_forceDestination &&
721  (!(m_type & PATHFIND_NORMAL) || !inRange(getEndPosition(), getActualEndPosition(), 1.0f, 1.0f)))
722  {
723  // we may want to keep partial subpath
724  if (dist3DSqr(getActualEndPosition(), getEndPosition()) < 0.3f * dist3DSqr(getStartPosition(), getEndPosition()))
725  {
726  setActualEndPosition(getEndPosition());
727  m_pathPoints[m_pathPoints.size() - 1] = getEndPosition();
728  }
729  else

732  BuildShortcut();
733  }
734

736  }

856 uint32 PathFinder::fixupCorridor(dtPolyRef* path, uint32 npath, uint32 maxPath, dtPolyRef const* visited, uint32 nvisited)
857 {
858  int32 furthestPath = -1;
859  int32 furthestVisited = -1;
860 
861  // Find furthest common polygon.
862  for (int32 i = npath - 1; i >= 0; --i)
863  {
864  bool found = false;
865  for (int32 j = nvisited - 1; j >= 0; --j)
866  {
867  if (path[i] == visited[j])
868  {
869  furthestPath = i;
870  furthestVisited = j;
871  found = true;
872  }
873  }
874  if (found)
875  break;
876  }
877 
878  // If no intersection found just return current path.
879  if (furthestPath == -1 || furthestVisited == -1)
880  return npath;
881 
882  // Concatenate paths.
883 
884  // Adjust beginning of the buffer to include the visited.
885  uint32 req = nvisited - furthestVisited;
886  uint32 orig = uint32(furthestPath + 1) < npath ? furthestPath + 1 : npath;
887  uint32 size = npath > orig ? npath - orig : 0;
888  if (req + size > maxPath)
889  size = maxPath - req;
890 
891  if (req >= maxPath)
892  return npath;
893 
894  if (size)
895  memmove(path + req, path + orig, size * sizeof(dtPolyRef));
896 
897  // Store visited
898  for (uint32 i = 0; i < req; ++i)
899  path[i] = visited[(nvisited - 1) - i];
900 
901  return req + size;
902 }

904 bool PathFinder::getSteerTarget(const float* startPos, const float* endPos,
905  float minTargetDist, const dtPolyRef* path, uint32 pathSize,
906  float* steerPos, unsigned char& steerPosFlag, dtPolyRef& steerPosRef) const
907 {
908  // Find steer target.
909  static const uint32 MAX_STEER_POINTS = 3;
910  float steerPath[MAX_STEER_POINTS * VERTEX_SIZE];
911  unsigned char steerPathFlags[MAX_STEER_POINTS];
912  dtPolyRef steerPathPolys[MAX_STEER_POINTS];
913  uint32 nsteerPath = 0;
914  dtStatus dtResult = m_navMeshQuery->findStraightPath(startPos, endPos, path, pathSize,
915  steerPath, steerPathFlags, steerPathPolys, (int*)&nsteerPath, MAX_STEER_POINTS);
916  if (!nsteerPath || dtStatusFailed(dtResult))
917  return false;
918 
919  // Find vertex far enough to steer to.
920  uint32 ns = 0;
921  while (ns < nsteerPath)
922  {
923  // Stop at Off-Mesh link or when point is further than slop away.
924  if ((steerPathFlags[ns] & DT_STRAIGHTPATH_OFFMESH_CONNECTION) ||
925  !inRangeYZX(&steerPath[ns * VERTEX_SIZE], startPos, minTargetDist, 1000.0f))
926  break;
927  ++ns;
928  }
929  // Failed to find good point to steer to.
930  if (ns >= nsteerPath)
931  return false;
932 
933  dtVcopy(steerPos, &steerPath[ns * VERTEX_SIZE]);
934  steerPos[1] = startPos[1]; // keep Z value
935  steerPosFlag = steerPathFlags[ns];
936  steerPosRef = steerPathPolys[ns];
937 
938  return true;
939 }

941 dtStatus PathFinder::findSmoothPath(const float* startPos, const float* endPos,
942  const dtPolyRef* polyPath, uint32 polyPathSize,
943  float* smoothPath, int* smoothPathSize, uint32 maxSmoothPathSize)
944 {
945  *smoothPathSize = 0;
946  uint32 nsmoothPath = 0;
947 
948  if (m_pointPathLimit > m_smoothPathPolyRefs.size())
949  m_smoothPathPolyRefs.resize(m_pointPathLimit);
950  memcpy(m_smoothPathPolyRefs.data(), polyPath, polyPathSize * sizeof(dtPolyRef));
951  uint32 npolys = polyPathSize;
952 
953  float iterPos[VERTEX_SIZE];
954  if (dtStatusFailed(m_navMeshQuery->closestPointOnPolyBoundary(m_smoothPathPolyRefs[0], startPos, iterPos)))
955  return DT_FAILURE;
956 
957  float targetPos[VERTEX_SIZE];
958  if (dtStatusFailed(m_navMeshQuery->closestPointOnPolyBoundary(m_smoothPathPolyRefs[npolys - 1], endPos, targetPos)))
959  return DT_FAILURE;
960 
961  dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
962  ++nsmoothPath;
963 
964  // Move towards target a small advancement at a time until target reached or
965  // when ran out of memory to store the path.
966  while (npolys && nsmoothPath < maxSmoothPathSize)
967  {
968  // Find location to steer towards.
969  float steerPos[VERTEX_SIZE];
970  unsigned char steerPosFlag;
971  dtPolyRef steerPosRef = INVALID_POLYREF;
972 
973  if (!getSteerTarget(iterPos, targetPos, SMOOTH_PATH_SLOP, m_smoothPathPolyRefs.data(), npolys, steerPos, steerPosFlag, steerPosRef))
974  break;
975 
976  const bool endOfPath = (steerPosFlag & DT_STRAIGHTPATH_END) != 0;
977  const bool offMeshConnection = (steerPosFlag & DT_STRAIGHTPATH_OFFMESH_CONNECTION) != 0;
978 
979  // Find movement delta.
980  float delta[VERTEX_SIZE];
981  dtVsub(delta, steerPos, iterPos);
982  float len = dtMathSqrtf(dtVdot(delta, delta));
983  // If the steer target is end of path or off-mesh link, do not move past the location.
984  if ((endOfPath || offMeshConnection) && len < SMOOTH_PATH_STEP_SIZE)
985  len = 1.0f;
986  else
987  len = SMOOTH_PATH_STEP_SIZE / len;
988 
989  float moveTgt[VERTEX_SIZE];
990  dtVmad(moveTgt, iterPos, delta, len);
991 
992  // Move
993  float result[VERTEX_SIZE];
994  const static uint32 MAX_VISIT_POLY = 16;
995  dtPolyRef visited[MAX_VISIT_POLY];
996 
997  uint32 nvisited = 0;
998  m_navMeshQuery->moveAlongSurface(m_smoothPathPolyRefs[0], iterPos, moveTgt, &m_filter, result, visited, (int*)&nvisited, MAX_VISIT_POLY);
999  npolys = fixupCorridor(m_smoothPathPolyRefs.data(), npolys, m_pointPathLimit, visited, nvisited);
1000 
1001  m_navMeshQuery->getPolyHeight(m_smoothPathPolyRefs[0], result, &result[1]);
1002  result[1] += 0.5f;
1003  dtVcopy(iterPos, result);
1004 
1005  // Handle end of path and off-mesh links when close enough.
1006  if (endOfPath && inRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
1007  {
1008  // Reached end of path.
1009  dtVcopy(iterPos, targetPos);
1010  if (nsmoothPath < maxSmoothPathSize)
1011  {
1012  dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
1013  ++nsmoothPath;
1014  }
1015  break;
1016  }
1017  if (offMeshConnection && inRangeYZX(iterPos, steerPos, SMOOTH_PATH_SLOP, 1.0f))
1018  {
1019  // Advance the path up to and over the off-mesh connection.
1020  dtPolyRef prevRef = INVALID_POLYREF;
1021  dtPolyRef polyRef = m_smoothPathPolyRefs[0];
1022  uint32 npos = 0;
1023  while (npos < npolys && polyRef != steerPosRef)
1024  {
1025  prevRef = polyRef;
1026  polyRef = m_smoothPathPolyRefs[npos];
1027  ++npos;
1028  }
1029 
1030  for (uint32 i = npos; i < npolys; ++i)
1031  m_smoothPathPolyRefs[i - npos] = m_smoothPathPolyRefs[i];
1032 
1033  npolys -= npos;
1034 
1035  // Handle the connection.
1036  float newStartPos[VERTEX_SIZE], newEndPos[VERTEX_SIZE];
1037  if (dtStatusSucceed(m_navMesh->getOffMeshConnectionPolyEndPoints(prevRef, polyRef, newStartPos, newEndPos)))
1038  {
1039  if (nsmoothPath < maxSmoothPathSize)
1040  {
1041  dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], startPos);
1042  ++nsmoothPath;
1043  }
1044  // Move position at the other side of the off-mesh link.
1045  dtVcopy(iterPos, endPos);
1046 
1047  m_navMeshQuery->getPolyHeight(m_smoothPathPolyRefs[0], iterPos, &iterPos[1]);
1048  iterPos[1] += 0.5f;
1049  }
1050  }
1051 
1052  // Store results.
1053  if (nsmoothPath < maxSmoothPathSize)
1054  {
1055  dtVcopy(&smoothPath[nsmoothPath * VERTEX_SIZE], iterPos);
1056  ++nsmoothPath;
1057  }
1058  }
1059 
1060  *smoothPathSize = nsmoothPath;
1061 
1062  // this is most likely a loop
1063  return nsmoothPath < m_pointPathLimit ? DT_SUCCESS : DT_FAILURE;
1064 }
