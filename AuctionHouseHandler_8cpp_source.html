65 void WorldSession::SendAuctionCommandResult(AuctionEntry* auc, AuctionAction Action, AuctionError ErrorCode, InventoryResult invError) const
66 {
67  WorldPacket data(SMSG_AUCTION_COMMAND_RESULT, 16);
68  data << uint32(auc ? auc->Id : 0);
69  data << uint32(Action);
70  data << uint32(ErrorCode);
71 
72  switch (ErrorCode)
73  {
74  case AUCTION_OK:
75  if (Action == AUCTION_BID_PLACED)
76  data << uint32(auc->GetAuctionOutBid()); // new AuctionOutBid?
77  break;
78  case AUCTION_ERR_INVENTORY:
79  data << uint32(invError);
80  break;
81  case AUCTION_ERR_HIGHER_BID:
82  data << ObjectGuid(HIGHGUID_PLAYER, auc->bidder); // new bidder guid
83  data << uint32(auc->bid); // new bid
84  data << uint32(auc->GetAuctionOutBid()); // new AuctionOutBid?
85  break;
86  default:
87  break;
88  }
89 
90  SendPacket(data);
91 }

196 AuctionHouseEntry const* WorldSession::GetCheckedAuctionHouseForAuctioneer(ObjectGuid guid) const
197 {
198  Unit* auctioneer;
199 
200  // GM case
201  if (guid == GetPlayer()->GetObjectGuid())
202  {
203  // command case will return only if player have real access to command
204  // using special access modes (1,-1) done at mode set in command, so not need recheck
205  if (GetPlayer()->GetAuctionAccessMode() == 0 && !ChatHandler(GetPlayer()).FindCommand("auction"))
206  {
207  DEBUG_LOG("%s attempt open auction in cheating way.", guid.GetString().c_str());
208  return nullptr;
209  }
210 
211  auctioneer = GetPlayer();
212  }
213  // auctioneer case
214  else
215  {
216  auctioneer = GetPlayer()->GetNPCIfCanInteractWith(guid, UNIT_NPC_FLAG_AUCTIONEER);
217  if (!auctioneer)
218  {
219  DEBUG_LOG("Auctioneer %s accessed in cheating way.", guid.GetString().c_str());
220  return nullptr;
221  }
222  }
223 
224  // always return pointer
225  return AuctionHouseMgr::GetAuctionHouseEntry(auctioneer);
226 }

228 // this void creates new auction and adds auction to some auctionhouse
229 void WorldSession::HandleAuctionSellItem(WorldPacket& recv_data)
230 {
231  DEBUG_LOG("WORLD: HandleAuctionSellItem");
232 
233  ObjectGuid auctioneerGuid;
234  uint32 etime, bid, buyout, itemCount;
235  GuidVector guids;
236  std::vector<uint32> stackSizes;
237 
238  recv_data >> auctioneerGuid;
239  recv_data >> itemCount;
240 
241  if (itemCount > MAX_BAG_SIZE * 5)
242  {
243  recv_data.rpos(recv_data.wpos()); // should not happen
244  return;
245  }
246 
247  guids.resize(itemCount);
248  stackSizes.resize(itemCount);
249 
250  for (uint32 i = 0; i < itemCount; ++i)
251  {
252  recv_data >> guids[i]; // item guid
253  recv_data >> stackSizes[i]; // stack size
254  }
255 
256  recv_data >> bid;
257  recv_data >> buyout;
258  recv_data >> etime;
259 
260  if (!bid || !etime)
261  return; // check for cheaters
262 
263  Player* pl = GetPlayer();
264 
265  AuctionHouseEntry const* auctionHouseEntry = GetCheckedAuctionHouseForAuctioneer(auctioneerGuid);
266  if (!auctionHouseEntry)
267  return;
268 
269  // always return pointer
270  AuctionHouseObject* auctionHouse = sAuctionMgr.GetAuctionsMap(auctionHouseEntry);
271 
272  // client send time in minutes, convert to common used sec time
273  etime *= MINUTE;
274 
275  // client understand only 3 auction time
276  switch (etime)
277  {
278  case 1*MIN_AUCTION_TIME:
279  case 2*MIN_AUCTION_TIME:
280  case 4*MIN_AUCTION_TIME:
281  break;
282  default:
283  return;
284  }
285 
286  // client allows to send too high money amount
287  if (bid > MAX_MONEY_AMOUNT || buyout > MAX_MONEY_AMOUNT)
288  {
289  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_DATABASE);
290  return;
291  }
292 
293  for (uint32 i = 0; i < itemCount; ++i)
294  {
295  ObjectGuid itemGuid = guids[i];
296 
297  if (!itemGuid)
298  continue;
299 
300  uint32 stackSize = stackSizes[i];
301 
302  Item* it = pl->GetItemByGuid(itemGuid);
303 
304  // do not allow to sell already auctioned items
305  if (sAuctionMgr.GetAItem(itemGuid.GetCounter()))
306  {
307  sLog.outError("AuctionError, %s is sending %s, but item is already in another auction", pl->GetGuidStr().c_str(), itemGuid.GetString().c_str());
308  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_INVENTORY, EQUIP_ERR_ITEM_NOT_FOUND);
309  continue;
310  }
311 
312  // prevent sending bag with items (cheat: can be placed in bag after adding equipped empty bag to auction)
313  if (!it)
314  {
315  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_INVENTORY, EQUIP_ERR_ITEM_NOT_FOUND);
316  continue;
317  }
318 
319  if (!it->CanBeTraded())
320  {
321  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_INVENTORY, EQUIP_ERR_CANNOT_TRADE_THAT);
322  continue;
323  }
324 
325  if ((it->GetProto()->Flags & ITEM_FLAG_CONJURED) || it->GetUInt32Value(ITEM_FIELD_DURATION))
326  {
327  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_INVENTORY, EQUIP_ERR_CANNOT_TRADE_THAT);
328  continue;
329  }
330 
331  // check money for deposit
332  uint32 deposit = AuctionHouseMgr::GetAuctionDeposit(auctionHouseEntry, etime, it);
333  if (pl->GetMoney() < deposit)
334  {
335  SendAuctionCommandResult(nullptr, AUCTION_STARTED, AUCTION_ERR_NOT_ENOUGH_MONEY);
336  continue;
337  }
338 
339  if (GetSecurity() > SEC_PLAYER && sWorld.getConfig(CONFIG_BOOL_GM_LOG_TRADE))
340  {
341  sLog.outCommand(GetAccountId(), "GM %s (Account: %u) create auction: %s (Entry: %u Count: %u)",
342  GetPlayerName(), GetAccountId(), it->GetProto()->Name1, it->GetEntry(), it->GetCount());
343  }
344 
345  if (stackSize == 0)
346  stackSize = 1;
347 
348  if (stackSize > it->GetMaxStackCount()) // too big stack size
349  stackSize = it->GetMaxStackCount();
350 
351  if (!pl->HasItemCount(it->GetEntry(), stackSize)) // not enough items
352  continue;
353 
354  Item* newItem = it->CloneItem(stackSize);
355 
356  pl->DestroyItemCount(it, stackSize, true);
357 
358  pl->ModifyMoney(-int32(deposit));
359 
360  AuctionEntry* AH = auctionHouse->AddAuction(auctionHouseEntry, newItem, etime, bid, buyout, deposit, pl);
361 
362  DETAIL_LOG("selling %s to auctioneer %s with initial bid %u with buyout %u and with time %u (in sec) in auctionhouse %u",
363  itemGuid.GetString().c_str(), auctioneerGuid.GetString().c_str(), bid, buyout, etime, auctionHouseEntry->houseId);
364 
365  SendAuctionCommandResult(AH, AUCTION_STARTED, AUCTION_OK);
366 
367  GetPlayer()->GetAchievementMgr().UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_CREATE_AUCTION, 1);
368  }
369 }

371 // this function is called when client bids or buys out auction
372 void WorldSession::HandleAuctionPlaceBid(WorldPacket& recv_data)
373 {
374  DEBUG_LOG("WORLD: HandleAuctionPlaceBid");
375 
376  ObjectGuid auctioneerGuid;
377  uint32 auctionId;
378  uint32 price;
379  recv_data >> auctioneerGuid;
380  recv_data >> auctionId >> price;
381 
382  if (!auctionId || !price)
383  return; // check for cheaters
384 
385  AuctionHouseEntry const* auctionHouseEntry = GetCheckedAuctionHouseForAuctioneer(auctioneerGuid);
386  if (!auctionHouseEntry)
387  return;
388 
389  // always return pointer
390  AuctionHouseObject* auctionHouse = sAuctionMgr.GetAuctionsMap(auctionHouseEntry);
391 
392  AuctionEntry* auction = auctionHouse->GetAuction(auctionId);
393  Player* pl = GetPlayer();
394 
395  if (!auction || auction->owner == pl->GetGUIDLow())
396  {
397  // you cannot bid your own auction:
398  SendAuctionCommandResult(nullptr, AUCTION_BID_PLACED, AUCTION_ERR_BID_OWN);
399  return;
400  }
401 
402  ObjectGuid ownerGuid = ObjectGuid(HIGHGUID_PLAYER, auction->owner);
403 
404  // impossible have online own another character (use this for speedup check in case online owner)
405  Player* auction_owner = sObjectMgr.GetPlayer(ownerGuid);
406  if (!auction_owner && sObjectMgr.GetPlayerAccountIdByGUID(ownerGuid) == pl->GetSession()->GetAccountId())
407  {
408  // you cannot bid your another character auction:
409  SendAuctionCommandResult(nullptr, AUCTION_BID_PLACED, AUCTION_ERR_BID_OWN);
410  return;
411  }
412 
413  // cheating or client lags
414  if (price <= auction->bid)
415  {
416  // client test but possible in result lags
417  SendAuctionCommandResult(auction, AUCTION_BID_PLACED, AUCTION_ERR_HIGHER_BID);
418  return;
419  }
420 
421  // price too low for next bid if not buyout
422  if ((price < auction->buyout || auction->buyout == 0) &&
423  price < auction->bid + auction->GetAuctionOutBid())
424  {
425  // client test but possible in result lags
426  SendAuctionCommandResult(auction, AUCTION_BID_PLACED, AUCTION_ERR_BID_INCREMENT);
427  return;
428  }
429 
430  if (price > pl->GetMoney())
431  {
432  // you don't have enough money!, client tests!
433  // SendAuctionCommandResult(auction->auctionId, AUCTION_ERR_INVENTORY, EQUIP_ERR_NOT_ENOUGH_MONEY);
434  return;
435  }
436 
437  // cheating
438  if (price < auction->startbid)
439  return;
440 
441  SendAuctionCommandResult(auction, AUCTION_BID_PLACED, AUCTION_OK);
442 
443  if (auction->UpdateBid(price, pl))
444  pl->GetAchievementMgr().UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, price);
445  else
446  pl->GetAchievementMgr().UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_HIGHEST_AUCTION_BID, auction->buyout);
447 }

451 {
452  DEBUG_LOG("WORLD: HandleAuctionRemoveItem");
453 
454  ObjectGuid auctioneerGuid;
455  uint32 auctionId;
456  recv_data >> auctioneerGuid;

458  // DEBUG_LOG("Cancel AUCTION AuctionID: %u", auctionId);

499

506  SendAuctionCommandResult(auction, AUCTION_REMOVED, AUCTION_OK);
507  // Now remove the auction
508  CharacterDatabase.BeginTransaction();
509  auction->DeleteFromDB();
510  pl->SaveInventoryAndGoldToDB();
511  CharacterDatabase.CommitTransaction();
512  sAuctionMgr.RemoveAItem(auction->itemGuidLow);
513  auctionHouse->RemoveAuction(auction->Id);
514  delete auction;
515 }
