109  // LightningCloud_Timer

113  {

118  m_uiLightningCloudTimer = m_bIsRegularMode ? 30000 : 10000;
119  }
120  }
121  }
122  else
123  m_uiLightningCloudTimer -= uiDiff;
124 
125  // LungBurst_Timer
126  if (m_uiLungBurstTimer < uiDiff)
127  {
128  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
129  {
130  if (DoCastSpellIfCan(pTarget, SPELL_LUNG_BURST) == CAST_OK)
131  m_uiLungBurstTimer = urand(7000, 12000);
132  }
133  }
134  else
135  m_uiLungBurstTimer -= uiDiff;
136 
137  // EnvelopingWinds_Timer
138  if (m_uiEnvelopingWindsTimer < uiDiff)
139  {
140  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
141  {
142  if (DoCastSpellIfCan(pTarget, SPELL_ENVELOPING_WINDS) == CAST_OK)
143  m_uiEnvelopingWindsTimer = m_bIsRegularMode ? 10000 : 15000;
144  }
145  }
146  else
147  m_uiEnvelopingWindsTimer -= uiDiff;
148 
149  DoMeleeAttackIfReady();
150  }
