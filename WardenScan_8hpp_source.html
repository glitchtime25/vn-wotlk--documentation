54 class Scan
55 {
56  public:
57  using BuildT = std::function<void(const Warden *, std::vector<std::string> &, ByteBuffer &)>;
58  using CheckT = std::function<bool(const Warden *, ByteBuffer &)>;
59 
60  private:
61  BuildT _builder;
62  CheckT _checker;
63 
64  protected: // should not be called by the user
65  Scan(BuildT builder, CheckT checker, uint32 f, size_t req, size_t rep, const std::string &c)
66  : _builder(builder), _checker(checker), flags(f), comment(c), requestSize(req), replySize(rep)
67  { MANGOS_ASSERT(!((flags & WinAllBuild) && (flags & MacAllBuild))); }
68 
69  public:
70  uint32 flags;
71  std::string comment;
72  size_t requestSize; // maximum size of request
73  size_t replySize; // maximum size of reply
74 
75  Scan() = delete;
76  virtual ~Scan() = default;
77 
78  void Build(const Warden *warden, std::vector<std::string> &strings, ByteBuffer &scan) const { _builder(warden, strings, scan); }
79 
80  // return true when the scan revealed a hack
81  bool Check(const Warden *warden, ByteBuffer &buff) const { return _checker(warden, buff); }
82 };
