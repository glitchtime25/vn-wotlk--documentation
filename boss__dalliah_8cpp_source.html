130  void UpdateAI(const uint32 uiDiff) override
131  {
132  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
133  return;
134 
135  if (m_uiGiftDoomsayerTimer < uiDiff)
136  {
137  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_GIFT_DOOMSAYER : SPELL_GIFT_DOOMSAYER_H) == CAST_OK)
138  m_uiGiftDoomsayerTimer = urand(14000, 19000);
139  }
140  else
141  m_uiGiftDoomsayerTimer -= uiDiff;
142 
143  if (m_uiWhirlwindTimer < uiDiff)
144  {
145  if (DoCastSpellIfCan(m_creature, SPELL_WHIRLWIND) == CAST_OK)
146  {
147  DoScriptText(urand(0, 1) ? SAY_WHIRLWIND_1 : SAY_WHIRLWIND_2, m_creature);
148  m_uiWhirlwindTimer = urand(25000, 30000);
149  m_uiHealTimer = 6000;
150  }
151  }
152  else
153  m_uiWhirlwindTimer -= uiDiff;
154 
155  if (m_uiHealTimer)
156  {
157  if (m_uiHealTimer <= uiDiff)
158  {
159  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_HEAL : SPELL_HEAL_H) == CAST_OK)
160  {
161  DoScriptText(urand(0, 1) ? SAY_HEAL_1 : SAY_HEAL_2, m_creature);
162  m_uiHealTimer = 0;
163  }
164  }
165  else
166  m_uiHealTimer -= uiDiff;
167  }
168 
169  if (!m_bIsRegularMode)
170  {
171  if (m_uiShadowWaveTimer < uiDiff)
172  {
173  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_FARTHEST_AWAY, 0, SPELL_SHADOW_WAVE, SELECT_FLAG_PLAYER))
174  {
175  if (DoCastSpellIfCan(pTarget, SPELL_SHADOW_WAVE) == CAST_OK)
176  m_uiShadowWaveTimer = urand(13000, 17000);
177  }
178  }
179  else
180  m_uiShadowWaveTimer -= uiDiff;
181  }
182 
183  if (!m_bHasTaunted && m_creature->GetHealthPercent() < 25.0f)
184  {
185  // Taunt if Soccothares isn't dead yet
186  if (m_pInstance && m_pInstance->GetData(TYPE_SOCCOTHRATES) != DONE)
187  {
188  if (Creature* pSoccothares = m_pInstance->GetSingleCreatureFromStorage(NPC_SOCCOTHRATES))
189  {
190  switch (urand(0, 2))
191  {
192  case 0: DoScriptText(SAY_SOCCOTHRATES_TAUNT_1, pSoccothares); break;
193  case 1: DoScriptText(SAY_SOCCOTHRATES_TAUNT_2, pSoccothares); break;
194  case 2: DoScriptText(SAY_SOCCOTHRATES_TAUNT_3, pSoccothares); break;
195  }
196  }
197  }
198 
199  m_bHasTaunted = true;
200  }
201 
202  DoMeleeAttackIfReady();
203  }
