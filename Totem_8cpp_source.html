34 bool Totem::Create(uint32 guidlow, CreatureCreatePos& cPos, CreatureInfo const* cinfo, Unit* owner)
35 {
36  SetMap(cPos.GetMap());
37  SetPhaseMask(cPos.GetPhaseMask(), false);
38 
39  if (!CreateFromProto(guidlow, guidlow, cinfo))
40  return false;
41 
42  // special model selection case for totems
43  if (owner->GetTypeId() == TYPEID_PLAYER && owner->getClass() != CLASS_DEATH_KNIGHT)
44  if (uint32 modelid_race = sObjectMgr.GetModelForRace(GetNativeDisplayId(), owner->getRaceMask()))
45  SetDisplayId(modelid_race);
46 
47  cPos.SelectFinalPoint(this, false);
48 
49  // totem must be at same Z in case swimming caster and etc.
50  if (fabs(cPos.m_pos.z - owner->GetPositionZ()) > 5.0f)
51  cPos.m_pos.z = owner->GetPositionZ();
52 
53  if (!cPos.Relocate(this))
54  return false;
55 
56  // Notify the map's instance data.
57  // Only works if you create the object in it, not if it is moves to that map.
58  // Normally non-players do not teleport to other maps.
59  if (InstanceData* iData = GetMap()->GetInstanceData())
60  iData->OnCreatureCreate(this);
61 
62  LoadCreatureAddon(false);
63 
64  SetCanDodge(false);
65  SetCanParry(false);
66  SetCanBlock(false);
67 
68  if (GetCreatureInfo()->SpellList)
69  SetSpellList(GetCreatureInfo()->SpellList);
70  else // legacy compatibility
71  SetSpellList(cinfo->Entry * 100 + 0);
72 
73  SetAOEImmune(true);
74  SetChainImmune(true);
75 
76  return true;

99 void Totem::Summon(Unit* owner)
100 {
101  owner->GetMap()->Add((Creature*)this);
102  AIM_Initialize();
103 
104  if (owner->AI())
105  owner->AI()->JustSummoned((Creature*)this);
106 
107  // there are some totems, which exist just for their visual appeareance
108  for (auto& data : m_spellList.Spells)
109  {
110  uint32 spellId = data.second.SpellId;
111  if (!spellId)
112  break;
113  switch (m_type)
114  {
115  case TOTEM_PASSIVE:
116  CastSpell(nullptr, spellId, TRIGGERED_OLD_TRIGGERED);
117  break;
118  case TOTEM_STATUE:
119  CastSpell(GetOwner(), spellId, TRIGGERED_OLD_TRIGGERED);
120  break;
121  default: break;
122  }
123  }

126 void Totem::UnSummon()
127 {
128  CombatStop(true);
129  RemoveAurasDueToSpell(GetSpell());
130 
131  AI()->OnUnsummon();
132 
133  if (Unit* owner = GetOwner())
134  {
135  owner->_RemoveTotem(this);
136  owner->RemoveAurasDueToSpell(GetSpell());
137 
138  // remove aura all party members too
139  if (owner->GetTypeId() == TYPEID_PLAYER)
140  {
141  ((Player*)owner)->SendAutoRepeatCancel(this);
142 
143  // Not only the player can summon the totem (scripted AI)
144  if (Group* pGroup = ((Player*)owner)->GetGroup())
145  {
146  for (GroupReference* itr = pGroup->GetFirstMember(); itr != nullptr; itr = itr->next())
147  {
148  Player* Target = itr->getSource();
149  if (Target && pGroup->SameSubGroup((Player*)owner, Target))
150  Target->RemoveAurasDueToSpell(GetSpell());
151  }
152  }
153  }
154 
155  if (owner->AI())
156  owner->AI()->SummonedCreatureDespawn((Creature*)this);
157  }
158 
159  // any totem unsummon look like as totem kill, req. for proper animation
160  if (IsAlive())
161  SetDeathState(DEAD);
162 
163  AddObjectToRemoveList();
164 }

256 bool Totem::IsImmuneToSpellEffect(SpellEntry const* spellInfo, SpellEffectIndex index, bool castOnSelf) const
257 {
258  // Totem may affected by some specific spells
259  // Mana Spring, Healing stream, Mana tide
260  // Flags : 0x00000002000 | 0x00000004000 | 0x00004000000 -> 0x00004006000
261  if (spellInfo->SpellFamilyName == SPELLFAMILY_SHAMAN && spellInfo->IsFitToFamilyMask(uint64(0x00004006000)))
262  return false;
263 
264  switch (spellInfo->Effect[index])
265  {
266  case SPELL_EFFECT_ATTACK_ME:
267  // immune to any type of regeneration effects hp/mana etc.
268  case SPELL_EFFECT_HEAL:
269  case SPELL_EFFECT_HEAL_MAX_HEALTH:
270  case SPELL_EFFECT_HEAL_MECHANICAL:
271  case SPELL_EFFECT_HEAL_PCT:
272  case SPELL_EFFECT_ENERGIZE:
273  case SPELL_EFFECT_ENERGIZE_PCT:
274  return true;
275  default:
276  break;
277  }
278 
279  if (!IsPositiveSpell(spellInfo))
280  {
281  // immune to all negative auras
282  if (IsAuraApplyEffect(spellInfo, index))
283  return true;
284  }
285  else
286  {
287  // immune to any type of regeneration auras hp/mana etc.
288  if (IsPeriodicRegenerateEffect(spellInfo, index))
289  return true;
290  }
291 
292  return Creature::IsImmuneToSpellEffect(spellInfo, index, castOnSelf);
293 }
