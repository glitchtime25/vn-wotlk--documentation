563  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
564  {
565  Unit* unitTarget = spell->GetUnitTarget();
566  if (!unitTarget)
567  return;
568 
569  spell->GetCaster()->getThreatManager().modifyThreatPercent(unitTarget, -100);
570  // Cast the spectral realm effect spell, visual spell and spectral blast rift summoning
