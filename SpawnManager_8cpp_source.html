90 void SpawnManager::RespawnCreature(uint32 dbguid, uint32 respawnDelay)
91 {
92  bool found = false;
93  auto itr = m_spawns.begin();
94  for (; itr != m_spawns.end(); )
95  {
96  auto& spawnInfo = *itr;
97  if (!spawnInfo.IsUsed() && spawnInfo.GetDbGuid() == dbguid && spawnInfo.GetHighGuid() == HIGHGUID_UNIT)
98  {
99  found = true;
100  m_map.GetPersistentState()->SaveCreatureRespawnTime(dbguid, time(nullptr) + respawnDelay);
101  if (respawnDelay > 0)
102  spawnInfo.SetRespawnTime(m_map.GetCurrentClockTime() + std::chrono::seconds(respawnDelay));
103  break;
104  }
105  ++itr;
106  }
107  if (!found)
108  AddCreature(respawnDelay, dbguid);
109  else if (respawnDelay == 0)
110  (*itr).ConstructForMap(m_map);
111  if (respawnDelay > 0)

153 void SpawnManager::Update()
154 {
155  auto now = m_map.GetCurrentClockTime();
156  for (auto itr = m_spawns.begin(); itr != m_spawns.end();)
157  {
158  auto& spawnInfo = *itr;
159  if (spawnInfo.IsUsed() ||
160  (spawnInfo.GetRespawnTime() <= now && spawnInfo.ConstructForMap(m_map)))
161  itr = m_spawns.erase(itr);
162  else
163  ++itr;
164  }
165 
166  for (auto& group : m_spawnGroups)
167  group.second->Update();
168 }

170 std::string SpawnManager::GetRespawnList()
171 {
172  std::string output = "";

175  output += "DBGuid: " + std::to_string(data.GetDbGuid()) + "HighGuid: " + (data.GetHighGuid() == HIGHGUID_UNIT ? "Creature" : "GameObject") + "Respawn Time ";
176  auto diff = (data.GetRespawnTime() - m_map.GetCurrentClockTime()).count();
177  if (auto hours = diff / (HOUR * IN_MILLISECONDS))
178  {
179  output += std::to_string(hours) + "h ";
180  diff -= hours * HOUR * IN_MILLISECONDS;
181  }
182  if (auto minutes = diff / (MINUTE * IN_MILLISECONDS))
183  {
184  output += std::to_string(minutes) + "m ";
185  diff -= minutes * MINUTE * IN_MILLISECONDS;
186  }
187  if (auto seconds = diff / (IN_MILLISECONDS))
188  {
189  output += std::to_string(seconds) + "s ";
190  diff -= seconds * IN_MILLISECONDS;
191  }
192  }
