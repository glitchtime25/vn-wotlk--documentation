119  // Wrapper to attack a new target and remove the speed gathering buff
120  void DoAttackNewTarget()
121  {
122  if (m_uiPhase == PHASE_TRANSFORM)
123  return;
124 
125  SetCombatMovement(true);
126  DoResetThreat();
127 
128  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
129  {
130  m_creature->AddThreat(target, 1000000.f);
131  AttackStart(target);
132  DoScriptText(EMOTE_TARGET, m_creature, target);
133  }
134 
135  ResetCombatAction(BURU_FULL_SPEED, 60000);
136  ResetCombatAction(BURU_GATHERING_SPEED, 9000);
137  }
138 
139  void HandlePhaseTwo()
140  {
141  switch (m_transitionStep)
142  {
143  case 0:
144  {
145  DoCastSpellIfCan(nullptr, SPELL_BURU_TRANSFORM);
146  m_creature->RemoveAurasDueToSpell(SPELL_THORNS);
147  ResetTimer(BURU_PHASE_2_TRANSITION_STEP, 5000);
148  break;
149  }
150  case 1:
151  {
152  DoCastSpellIfCan(nullptr, SPELL_FULL_SPEED, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
153  SetCombatScriptStatus(false);
154  SetMeleeEnabled(true);
155  SetCombatMovement(true, true);
156  break;
157  }
158  }
159  ++m_transitionStep;
160  }
161 
162  void ExecuteAction(uint32 action) override
163  {
164  switch (action)
165  {
166  case BURU_PHASE_2_TRANSITION:
167  {
168  if (m_creature->GetHealthPercent() < 20.0f)
169  {
170  SetActionReadyStatus(action, false);
171  m_uiPhase = PHASE_TRANSFORM;
172  DisableCombatAction(BURU_DISMEMBER);
173  SetActionReadyStatus(BURU_NEW_TARGET, false);
174  DisableCombatAction(BURU_GATHERING_SPEED);
175  DisableCombatAction(BURU_FULL_SPEED);
176  ResetCombatAction(BURU_CREEPING_PLAGUE, 0);
177  DoResetThreat();
178  ResetTimer(BURU_PHASE_2_TRANSITION_STEP, 2000);
179  SetCombatScriptStatus(true);
180  SetMeleeEnabled(false);
181  SetCombatMovement(false, true);
182  m_creature->SetTarget(nullptr);
183  }
184  break;
185  }
186  case BURU_NEW_TARGET:
187  {
188  m_creature->RemoveAurasDueToSpell(SPELL_FULL_SPEED);
189  m_creature->RemoveAurasDueToSpell(SPELL_GATHERING_SPEED);
190  if (DoCastSpellIfCan(nullptr, SPELL_CREATURE_SPECIAL) == CAST_OK)
191  {
192  SetCombatMovement(false, true);
193  SetActionReadyStatus(action, false);
194  }
195  break;
196  }
197  case BURU_DISMEMBER:
198  {
199  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_DISMEMBER) == CAST_OK)
200  ResetCombatAction(action, 5000);
201  break;
202  }
203  case BURU_CREEPING_PLAGUE:
204  {
205  if (DoCastSpellIfCan(nullptr, SPELL_CREEPING_PLAGUE) == CAST_OK)
206  ResetCombatAction(action, 6000);
207  break;
208  }
209  case BURU_GATHERING_SPEED:
210  {
211  if (DoCastSpellIfCan(nullptr, SPELL_GATHERING_SPEED) == CAST_OK)
212  ResetCombatAction(action, 9000);
213  break;
214  }
215  case BURU_FULL_SPEED:
216  {
217  if (DoCastSpellIfCan(nullptr, SPELL_FULL_SPEED) == CAST_OK)
218  {
219  m_creature->RemoveAurasDueToSpell(SPELL_GATHERING_SPEED);
220  DisableCombatAction(action);
221  DisableCombatAction(BURU_GATHERING_SPEED);
222  }
223  break;
224  }
225  }
226  }
