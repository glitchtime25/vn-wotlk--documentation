96 void instance_eye_of_eternity::SetData(uint32 uiType, uint32 uiData)
97 {
98  if (uiType != TYPE_MALYGOS)
99  return;
100 
101  m_uiEncounter = uiData;
102 
103  if (uiData == IN_PROGRESS)
104  {
105  // Portal and iris despawn handled in DB
106  DoStartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_KILL_CREATURE, ACHIEV_START_MALYGOS_ID);
107  }
108  else if (uiData == FAIL)
109  {
110  // respawn iris and portal
111  DoRespawnGameObject(GO_EXIT_PORTAL, 7 * DAY);
112  DoRespawnGameObject(instance->GetDifficulty() == RAID_DIFFICULTY_10MAN_NORMAL ? GO_FOCUSING_IRIS : GO_FOCUSING_IRIS_H, 7 * DAY);
113 
114  // rebuild platform
115  if (Creature* pMalygos = GetSingleCreatureFromStorage(NPC_MALYGOS))
116  {
117  if (GameObject* pPlatform = GetSingleGameObjectFromStorage(GO_PLATFORM))
118  {
119  pPlatform->RebuildGameObject(pMalygos);
120  pPlatform->SetGoState(GO_STATE_READY);
121  instance->ChangeGOPathfinding(194232, 8546, true);
122  }
123 
124  // despawn and respawn boss
125  pMalygos->ForcedDespawn(5000);
126  m_uiMalygosResetTimer = 30000;
127  }
128  }
129  else if (uiData == DONE)
130  {
131  // Note: ending event handled by DB
132 
133  // Spawn the Heart of Malygos immediately after death
134  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_HEART_OF_MAGIC : GO_HEART_OF_MAGIC_H, 24 * HOUR);
135 
136  // respawn the other loot on timer
137  m_uiMalygosCompleteTimer = 18000;
138  }
139 
140  // Currently no reason to save anything
141 }

143 void instance_eye_of_eternity::OnCreatureDeath(Creature* pCreature)
144 {
145  switch (pCreature->GetEntry())
146  {
147  case NPC_NEXUS_LORD:
148  case NPC_SCION_OF_ETERNITY:
149  // disk vehicles fall down once passenger is killed
150  if (pCreature->GetTransportInfo() && pCreature->GetTransportInfo()->IsOnVehicle())
151  {
152  if (Unit* pVehicle = static_cast<Unit*>(pCreature->GetTransportInfo()->GetTransport()))
153  {
154  pVehicle->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE, pCreature->GetObjectGuid());
155  pVehicle->GetMotionMaster()->Clear(false, true);
156  pVehicle->SetLevitate(false);
157  pVehicle->GetMotionMaster()->MoveFall();
158  }
159  }
160 
161  // remove passenger from riders list
162  m_lDiskRidersGuids.remove(pCreature->GetObjectGuid());
163 
164  // start phase 3 if all adds are dead
165  if (m_lDiskRidersGuids.empty())
166  {
167  if (Creature* pMalygos = GetSingleCreatureFromStorage(NPC_MALYGOS))
168  pMalygos->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, pMalygos, pMalygos);
169 
170  // despawn all disks and arcane bombs before phase 3
171  for (const auto& guid : m_lSecondPhaseCreaturesGuids)
172  if (Creature* pCreature = instance->GetCreature(guid))
173  pCreature->ForcedDespawn();
174  }
175  break;
176  }
177 }

179 void instance_eye_of_eternity::Update(uint32 uiDiff)
180 {
181  // Respawn Malygos
182  if (m_uiMalygosResetTimer)
183  {

185  {
186  if (Creature* pMalygos = GetSingleCreatureFromStorage(NPC_MALYGOS))
187  pMalygos->Respawn();
188 
189  m_uiMalygosResetTimer = 0;
190  }
191  else
192  m_uiMalygosResetTimer -= uiDiff;
193  }
194 
195  // respawn loot and portal
196  if (m_uiMalygosCompleteTimer)
197  {
198  if (m_uiMalygosCompleteTimer <= uiDiff)
199  {
200  DoRespawnGameObject(GO_EXIT_PORTAL, 24 * HOUR);
201  DoRespawnGameObject(instance->IsRegularDifficulty() ? GO_ALEXSTRASZAS_GIFT : GO_ALEXSTRASZAS_GIFT_H, 24 * HOUR);
202 
203  m_uiMalygosCompleteTimer = 0;
204  }
205  else
206  m_uiMalygosCompleteTimer -= uiDiff;
207  }
