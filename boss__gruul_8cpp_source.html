115  void ExecuteAction(uint32 action) override
116  {
117  switch (action)
118  {
119  case GRUUL_ACTION_SHATTER:
120  {
121  if (DoCastSpellIfCan(nullptr, SPELL_SHATTER) == CAST_OK)
122  {
123  DoScriptText(EMOTE_SLAM, m_creature);
124  DoScriptText(urand(0, 1) ? SAY_SHATTER1 : SAY_SHATTER2, m_creature);
125  SetActionReadyStatus(action, false);
126  return;
127  }
128  return;
129  }
130  case GRUUL_ACTION_GROW:
131  {
132  if (DoCastSpellIfCan(nullptr, SPELL_GROWTH) == CAST_OK)
133  {
134  DoScriptText(EMOTE_GROW, m_creature);
135  ResetCombatAction(action, 30000);
136  return;
137  }
138  return;
139  }
140  case GRUUL_ACTION_GROUND_SLAM:
141  {
142  if (DoCastSpellIfCan(nullptr, SPELL_GROUND_SLAM_DUMMY) == CAST_OK)
143  {
144  m_creature->CastSpell(nullptr, SPELL_GROUND_SLAM, TRIGGERED_NONE);
145  DoScriptText(urand(0, 1) ? SAY_SLAM1 : SAY_SLAM2, m_creature);
146  m_creature->SetTarget(nullptr);
147  ResetCombatAction(action, urand(70000, 80000));
148  SetActionReadyStatus(GRUUL_ACTION_SHATTER, true); // top priority, blocked by stun
149  m_lookAround = true;
150  return;
151  }
152  return;
153  }
154  case GRUUL_ACTION_REVERBERATION:
155  {
156  if (DoCastSpellIfCan(nullptr, SPELL_REVERBERATION) == CAST_OK)
157  ResetCombatAction(action, urand(35000, 45000));
158  return;
159  }
160  case GRUUL_ACTION_CAVE_IN:
161  {
162  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
163  if (DoCastSpellIfCan(target, SPELL_CAVE_IN) == CAST_OK)
164  ResetCombatAction(action, 8500);
165  return;
166  }
167  case GRUUL_ACTION_HURTFUL_STRIKE:
168  {
169  if (DoCastSpellIfCan(nullptr, SPELL_HURTFUL_STRIKE) == CAST_OK)
170  ResetCombatAction(action, 8000);
171  break;
172  }
173  }
174  }

205  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
206  {
207  Unit* target = spell->GetUnitTarget();
208  Unit* caster = spell->GetCaster();
209  auto& targetInfo = spell->GetTargetList();
210  if (!target || targetInfo.rbegin()->targetGUID != target->GetObjectGuid())
211  return;
212 
213  for (auto& targetInfo : targetInfo)
214  {
215  if (caster->GetMap()->GetPlayer(targetInfo.targetGUID) == caster->GetVictim())
216  continue;
217 
218  if (caster->getThreatManager().getThreat(caster->GetMap()->GetPlayer(targetInfo.targetGUID)) > caster->getThreatManager().getThreat(target) || target == caster->GetVictim())
219  target = caster->GetMap()->GetPlayer(targetInfo.targetGUID);
220  }
221 
222  caster->CastSpell(target, 33813, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_NORMAL_COMBAT_CAST);
223  }
