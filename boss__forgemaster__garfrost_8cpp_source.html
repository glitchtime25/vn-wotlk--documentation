173  void MovementInform(uint32 motionType, uint32 pointId) override
174  {
175  if (motionType != EFFECT_MOTION_TYPE)
176  return;
177 
178  if (pointId != PHASE_BLADE_ENCHANTMENT && pointId != PHASE_MACE_ENCHANTMENT)
179  return;
180 
181  // Cast and say expected spell
182  if (DoCastSpellIfCan(m_creature, pointId == PHASE_BLADE_ENCHANTMENT ? SPELL_FORGE_FROZEN_BLADE : SPELL_FORGE_FROSTBORN_MACE) == CAST_OK)
183  {
184  DoScriptText(pointId == PHASE_BLADE_ENCHANTMENT ? SAY_FORGE_1 : SAY_FORGE_2, m_creature);
185  ResetCombatAction(GARFROST_RESUME_MOVEMENT, 8000u);
186  ResetCombatAction(GARFROST_THROW_SARONITE, 16000u);
187  ResetCombatAction(pointId == PHASE_BLADE_ENCHANTMENT ? GARFROST_CHILLING_WAVE : GARFROST_DEEP_FREEZE, 14000u);
188  m_creature->SetVirtualItem(VIRTUAL_ITEM_SLOT_0, pointId == PHASE_BLADE_ENCHANTMENT ? EQUIP_SLOT_ID_SWORD : EQUIP_SLOT_ID_MACE);
189  }

192  void ExecuteAction(uint32 action) override
193  {
194  switch (action)
195  {
196  case GARFROST_THROW_SARONITE:
197  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
198  {
199  if (DoCastSpellIfCan(pTarget, SPELL_THROW_SARONITE) == CAST_OK)
200  {
201  DoScriptText(EMOTE_THROW_SARONITE, m_creature, pTarget);
202  ResetCombatAction(action, 16000);
203  }
204  }
205  break;
206  case GARFROST_RESUME_MOVEMENT:
207  SetCombatMovement(true);
208  SetMeleeEnabled(true);
209  if (m_creature->GetVictim())
210  {
211  m_creature->SetTarget(m_creature->GetVictim());
212  m_creature->GetMotionMaster()->MoveChase(m_creature->GetVictim());
213  }
214  DisableCombatAction(action);
215  break;
216  case GARFROST_FROZEN_BLADE:
217  if (m_creature->GetHealthPercent() < 66.0f)
218  {
219  if (DoCastSpellIfCan(m_creature, SPELL_THUNDERING_STOMP) == CAST_OK)
220  {
221  // disable movement and melee before jumping; otherwise the jump won't finalize properly
222  SetCombatMovement(false);
223  SetMeleeEnabled(false);
224 
225  m_creature->GetMotionMaster()->MoveJump(aGarfrostMoveLocs[0][0], aGarfrostMoveLocs[0][1], aGarfrostMoveLocs[0][2], 3 * m_creature->GetSpeed(MOVE_RUN), 10.0f, PHASE_BLADE_ENCHANTMENT);
226  m_creature->SetTarget(nullptr);
227 
228  SetActionReadyStatus(action, false);
229  DisableCombatAction(GARFROST_THROW_SARONITE);
230  }
231  }
232  break;
233  case GARFROST_FROSTBORN_MACE:
234  if (m_creature->GetHealthPercent() < 33.0f)
235  {
236  if (DoCastSpellIfCan(m_creature, SPELL_THUNDERING_STOMP) == CAST_OK)
237  {
238  m_creature->RemoveAurasDueToSpell(SPELL_FORGE_FROZEN_BLADE);
239 
240  // disable movement and melee before jumping; otherwise the jump won't finalize properly
241  SetCombatMovement(false);
242  SetMeleeEnabled(false);
243 
244  m_creature->GetMotionMaster()->MoveJump(aGarfrostMoveLocs[1][0], aGarfrostMoveLocs[1][1], aGarfrostMoveLocs[1][2], 3 * m_creature->GetSpeed(MOVE_RUN), 10.0f, PHASE_MACE_ENCHANTMENT);
245  m_creature->SetTarget(nullptr);
246 
247  SetActionReadyStatus(action, false);
248  DisableCombatAction(GARFROST_CHILLING_WAVE);
249  DisableCombatAction(GARFROST_THROW_SARONITE);
250  }
251  }
252  break;
253  case GARFROST_CHILLING_WAVE:
254  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CHILLING_WAVE) == CAST_OK)
255  ResetCombatAction(action, 14000);
256  break;
257  case GARFROST_DEEP_FREEZE:
258  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
259  {
260  if (DoCastSpellIfCan(pTarget, SPELL_DEEP_FREEZE) == CAST_OK)
261  {
262  DoScriptText(EMOTE_DEEP_FREEZE, m_creature, pTarget);
263  ResetCombatAction(action, 20000);
264  }
265  }
266  break;
267  }
268  }
