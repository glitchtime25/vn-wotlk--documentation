76 void instance_gnomeregan::SetData(uint32 uiType, uint32 uiData)
77 {
78  switch (uiType)
79  {
80  case TYPE_GRUBBIS:
81  m_auiEncounter[0] = uiData;
82  if (uiData == IN_PROGRESS)
83  {
84  // Sort the explosive charges if needed
85  if (!m_luiExplosiveChargeGUIDs.empty())
86  {
87  GameObjectList lExplosiveCharges;
88  for (GuidList::const_iterator itr = m_luiExplosiveChargeGUIDs.begin(); itr != m_luiExplosiveChargeGUIDs.end(); ++itr)
89  {
90  if (GameObject* pCharge = instance->GetGameObject(*itr))
91  lExplosiveCharges.push_back(pCharge);
92  }
93  m_luiExplosiveChargeGUIDs.clear();
94 
95  // Sort from east to west
96  lExplosiveCharges.sort(sortFromEastToWest);
97 
98  // Sort to south and north
99  uint8 uiCounterSouth = 0;
100  uint8 uiCounterNorth = 0;
101  GameObject* pCaveInSouth = GetSingleGameObjectFromStorage(GO_CAVE_IN_SOUTH);
102  GameObject* pCaveInNorth = GetSingleGameObjectFromStorage(GO_CAVE_IN_NORTH);
103  if (pCaveInSouth && pCaveInNorth)
104  {
105  for (auto& lExplosiveCharge : lExplosiveCharges)
106  {
107  if (lExplosiveCharge->GetDistanceOrder(pCaveInSouth, pCaveInNorth) && uiCounterSouth < MAX_EXPLOSIVES_PER_SIDE)
108  {
109  m_aExplosiveSortedGuids[0][uiCounterSouth] = lExplosiveCharge->GetObjectGuid();
110  ++uiCounterSouth;
111  }
112  else if (uiCounterNorth < MAX_EXPLOSIVES_PER_SIDE)
113  {
114  m_aExplosiveSortedGuids[1][uiCounterNorth] = lExplosiveCharge->GetObjectGuid();
115  ++uiCounterNorth;
116  }
117  }
118  }
119  }
120  }
121  if (uiData == FAIL)
122  {
123  // Despawn possible spawned explosive charges
124  SetData(TYPE_EXPLOSIVE_CHARGE, DATA_EXPLOSIVE_CHARGE_USE);
125  }
126  if (uiData == DONE)
127  {
128  for (GuidList::const_iterator itr = m_lRedRocketGUIDs.begin(); itr != m_lRedRocketGUIDs.end(); ++itr)
129  DoRespawnGameObject(*itr, HOUR);
130  }
131  break;
132  case TYPE_EXPLOSIVE_CHARGE:
133  switch (uiData)
134  {
135  case DATA_EXPLOSIVE_CHARGE_1:
136  DoRespawnGameObject(m_aExplosiveSortedGuids[0][0], HOUR);
137  m_luiSpawnedExplosiveChargeGUIDs.push_back(m_aExplosiveSortedGuids[0][0]);
138  break;
139  case DATA_EXPLOSIVE_CHARGE_2:
140  DoRespawnGameObject(m_aExplosiveSortedGuids[0][1], HOUR);
141  m_luiSpawnedExplosiveChargeGUIDs.push_back(m_aExplosiveSortedGuids[0][1]);
142  break;
143  case DATA_EXPLOSIVE_CHARGE_3:
144  DoRespawnGameObject(m_aExplosiveSortedGuids[1][0], HOUR);
145  m_luiSpawnedExplosiveChargeGUIDs.push_back(m_aExplosiveSortedGuids[1][0]);
146  break;
147  case DATA_EXPLOSIVE_CHARGE_4:
148  DoRespawnGameObject(m_aExplosiveSortedGuids[1][1], HOUR);
149  m_luiSpawnedExplosiveChargeGUIDs.push_back(m_aExplosiveSortedGuids[1][1]);
150  break;
151  case DATA_EXPLOSIVE_CHARGE_USE:
152  Creature* pBlastmaster = GetSingleCreatureFromStorage(NPC_BLASTMASTER_SHORTFUSE);
153  if (!pBlastmaster)
154  break;
155  for (GuidList::const_iterator itr = m_luiSpawnedExplosiveChargeGUIDs.begin(); itr != m_luiSpawnedExplosiveChargeGUIDs.end(); ++itr)
156  {
157  if (GameObject* pExplosive = instance->GetGameObject(*itr))
158  pExplosive->Use(pBlastmaster);
159  }
160  m_luiSpawnedExplosiveChargeGUIDs.clear();
161  break;
162  }
163  return;
164  case TYPE_THERMAPLUGG:
165  m_auiEncounter[1] = uiData;
166  if (uiData == IN_PROGRESS)
167  {
168  // Make Door locked
169  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_THE_FINAL_CHAMBER))
170  {
171  pDoor->SetFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
172  if (pDoor->GetLootState() == GO_ACTIVATED)
173  pDoor->ResetDoorOrButton();
174  }
175 
176  // Always directly activates this bomb-face
177  DoActivateBombFace(2);
178  }
179  else if (uiData == DONE || uiData == FAIL)
180  {
181  // Make Door unlocked again
182  if (GameObject* pDoor = GetSingleGameObjectFromStorage(GO_THE_FINAL_CHAMBER))
183  {
184  pDoor->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_LOCKED);
185  if (pDoor->GetLootState() == GO_READY)
186  pDoor->UseDoorOrButton();
187  }
188 
189  // Deactivate all remaining BombFaces
190  for (uint8 i = 0; i < MAX_GNOME_FACES; ++i)
191  DoDeactivateBombFace(i);
192  }
193  break;
194  }
195 
196  if (uiData == DONE)
197  {
198  OUT_SAVE_INST_DATA;
199 
200  std::ostringstream saveStream;
201  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1];
202 
203  m_strInstData = saveStream.str();
204 
205  SaveToDB();
206  OUT_SAVE_INST_DATA_COMPLETE;
207  }
208 }
