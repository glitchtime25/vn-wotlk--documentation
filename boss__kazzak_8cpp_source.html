99  uint32 GetInitialActionTimer(const uint32 action) const
100  {
101  switch (action)
102  {
103  case KAZZAK_ACTION_SUPREME: return 3 * MINUTE * IN_MILLISECONDS;
104  case KAZZAK_ACTION_VOIDBOLT: return 30000;
105  case KAZZAK_ACTION_TWISTED_REFLECTION: return 33000;
106  case KAZZAK_ACTION_MARK_OF_KAZZAK: return 25000;
107  case KAZZAK_ACTION_SHADOW_BOLT_VOLLEY: return urand(3000, 12000);
108  case KAZZAK_ACTION_THUNDERCLAP: return urand(16000, 20000);
109  case KAZZAK_ACTION_CLEAVE: return 7000;
110  default: return 0; // never occurs but for compiler
111  }
112  }

158  void ExecuteActions()
159  {
160  if (!CanExecuteCombatAction())
161  return;
162 
163  for (uint32 i = 0; i < KAZZAK_ACTION_MAX; ++i)
164  {
165  if (GetActionReadyStatus(i))
166  {
167  switch (i)
168  {
169  case KAZZAK_ACTION_SUPREME:
170  {
171  // Enrage - cast shadowbolt volley every second
172  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
173  {
174  DoScriptText(urand(0, 1) ? SAY_SURPREME1 : SAY_SURPREME2, m_creature);
175  DisableCombatAction(KAZZAK_ACTION_SUPREME);
176  return;
177  }
178  break;
179  }
180  case KAZZAK_ACTION_VOIDBOLT:
181  {
182  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_VOIDBOLT) == CAST_OK)
183  {
184  ResetTimer(i, GetSubsequentActionTimer(i));
185  SetActionReadyStatus(i, false);
186  return;
187  }
188  break;
189  }
190  case KAZZAK_ACTION_TWISTED_REFLECTION:
191  {
192  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_TWISTED_REFLECTION, SELECT_FLAG_PLAYER))
193  {
194  if (DoCastSpellIfCan(pTarget, SPELL_TWISTED_REFLECTION) == CAST_OK)
195  {
196  ResetTimer(i, GetSubsequentActionTimer(i));
197  SetActionReadyStatus(i, false);
198  return;
199  }
200  }
201  break;
202  }
203  case KAZZAK_ACTION_MARK_OF_KAZZAK:
204  {
205  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MARK_OF_KAZZAK, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
206  {
207  if (DoCastSpellIfCan(pTarget, SPELL_MARK_OF_KAZZAK) == CAST_OK)
208  {
209  ResetTimer(i, GetSubsequentActionTimer(i));
210  SetActionReadyStatus(i, false);
211  return;
212  }
213  break;
214  }
215  }
216  case KAZZAK_ACTION_SHADOW_BOLT_VOLLEY:
217  {
218  // Cast shadowbolt volley on timer before Berserk
219  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_VOLLEY, SELECT_FLAG_USE_EFFECT_RADIUS) == CAST_OK)
220  {
221  ResetTimer(i, GetSubsequentActionTimer(i));
222  SetActionReadyStatus(i, false);
223  return;
224  }
225  break;
226  }
227  case KAZZAK_ACTION_THUNDERCLAP:
228  {
229  if (DoCastSpellIfCan(nullptr, SPELL_THUNDERCLAP, SELECT_FLAG_USE_EFFECT_RADIUS) == CAST_OK)
230  {
231  ResetTimer(i, GetSubsequentActionTimer(i));
232  SetActionReadyStatus(i, false);
233  return;
234  }
235  break;
236  }
237  case KAZZAK_ACTION_CLEAVE:
238  {
239  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
240  {
241  ResetTimer(i, GetSubsequentActionTimer(i));
242  SetActionReadyStatus(i, false);
243  return;
244  }
245  break;
246  }
247  }
248  }
249  }
