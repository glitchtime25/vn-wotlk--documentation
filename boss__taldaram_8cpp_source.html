180  // Wrapper which sends each sphere in a different direction
181  void DoSetSpheresInMotion()
182  {
183  float fX, fY;
184  uint8 uiIndex = m_bIsRegularMode ? urand(0, 2) : 0;
185  for (GuidList::const_iterator itr = m_lFlameOrbsGuidList.begin(); itr != m_lFlameOrbsGuidList.end(); ++itr)
186  {
187  if (Creature* pOrb = m_creature->GetMap()->GetCreature(*itr))
188  {
189  pOrb->CastSpell(pOrb, m_bIsRegularMode ? SPELL_FLAME_SPHERE_PERIODIC : SPELL_FLAME_SPHERE_PERIODIC_H, TRIGGERED_OLD_TRIGGERED);
190 
191  pOrb->GetNearPoint2d(fX, fY, 70.0f, (2 * M_PI_F / 3)*uiIndex);
192  pOrb->GetMotionMaster()->MovePoint(0, fX, fY, pOrb->GetPositionZ());
193  }
194  ++uiIndex;
195  }
196  }

198  void UpdateAI(const uint32 uiDiff) override
199  {
200  if (m_uiVisualTimer)
201  {
202  if (m_uiVisualTimer <= uiDiff)
203  {
204  GuidList lControllersList;
205  if (m_pInstance)
206  m_pInstance->GetJedogaControllersList(lControllersList);
207 
208  for (GuidList::const_iterator itr = lControllersList.begin(); itr != lControllersList.end(); ++itr)
209  {
210  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
211  pTemp->CastSpell(m_creature, SPELL_BEAM_VISUAL, TRIGGERED_NONE);
212  }
213  m_uiVisualTimer = 0;
214  }
215  else
216  m_uiVisualTimer -= uiDiff;
217  }
218 
219  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
220  return;
221 
222  // Cast Embrace of the Vampyr after Vanish expires - note: because of the invisibility effect, the timers won't decrease during vanish
223  if (m_uiEmbraceTimer)
224  {
225  if (m_uiEmbraceTimer <= uiDiff)
226  {
227  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
228  {
229  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_EMBRACE_OF_THE_VAMPYR : SPELL_EMBRACE_OF_THE_VAMPYR_H) == CAST_OK)
230  {
231  DoScriptText(urand(0, 1) ? SAY_FEED_1 : SAY_FEED_2, m_creature);
232  m_uiEmbraceTimer = 0;
233  }
234  }
235  }
236  else
237  m_uiEmbraceTimer -= uiDiff;
238 
239  // do not use other abilities during vanish
240  return;
241  }
242 
243  if (m_uiVanishTimer)
244  {
245  if (m_uiVanishTimer <= uiDiff)
246  {
247  if (DoCastSpellIfCan(m_creature, SPELL_VANISH) == CAST_OK)
248  {
249  DoScriptText(urand(0, 1) ? SAY_VANISH_1 : SAY_VANISH_2, m_creature);
250  m_uiVanishTimer = 0;
251  m_uiEmbraceTimer = 2000;
252  }
253  }
254  else
255  m_uiVanishTimer -= uiDiff;
256  }
257 
258  if (m_uiBloodthirstTimer < uiDiff)
259  {
260  if (DoCastSpellIfCan(m_creature, SPELL_BLOODTHIRST) == CAST_OK)
261  m_uiBloodthirstTimer = urand(20000, 25000);
262  }
263  else
264  m_uiBloodthirstTimer -= uiDiff;
265 
266  if (m_uiFlameOrbTimer < uiDiff)
267  {
268  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CONJURE_FLAME_SPHERE) == CAST_OK)
269  {
270  m_lFlameOrbsGuidList.clear();
271 
272  // Flame speres are summoned above the boss
273  m_creature->CastSpell(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 5.0f, SPELL_FLAME_SPHERE_SUMMON_1, TRIGGERED_OLD_TRIGGERED);
274 
275  // 2 more spheres on heroic
276  if (!m_bIsRegularMode)
277  {
278  m_creature->CastSpell(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 5.0f, SPELL_FLAME_SPHERE_SUMMON_2, TRIGGERED_OLD_TRIGGERED);
279  m_creature->CastSpell(m_creature->GetPositionX(), m_creature->GetPositionY(), m_creature->GetPositionZ() + 5.0f, SPELL_FLAME_SPHERE_SUMMON_3, TRIGGERED_OLD_TRIGGERED);
280  }
281 
282  m_uiFlameOrbTimer = urand(50000, 60000);
283  m_uiVanishTimer = 12000;
284  }
285  }
286  else
287  m_uiFlameOrbTimer -= uiDiff;
288 
289  DoMeleeAttackIfReady();
290  }
