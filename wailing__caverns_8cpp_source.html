86  }
87 
88  // Set to special in order to start the escort event; only if all four bosses are done
89  if (m_auiEncounter[0] == DONE && m_auiEncounter[1] == DONE && m_auiEncounter[2] == DONE && m_auiEncounter[3] == DONE && (m_auiEncounter[4] == NOT_STARTED || m_auiEncounter[4] == FAIL))
90  {
91  // Yell intro text; only the first time
92  if (m_auiEncounter[4] == NOT_STARTED)
93  {
94  if (Creature* pDisciple = GetSingleCreatureFromStorage(NPC_DISCIPLE))
95  DoBroadcastText(SAY_INTRO, pDisciple);
96  }
97 
98  m_auiEncounter[4] = SPECIAL;
99  }
100 
101  if (uiData == DONE)
102  {
103  OUT_SAVE_INST_DATA;
104 
105  std::ostringstream saveStream;
106 
107  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
108  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5];
109 
110  m_strInstData = saveStream.str();
111  SaveToDB();
112  OUT_SAVE_INST_DATA_COMPLETE;
113  }
114 }
