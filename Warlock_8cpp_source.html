239  // if damage is more than need

247 
248  // Cast finish spell (triggeredByAura already not exist!)
249  if (Unit* caster = procData.triggeredByAura->GetCaster())
250  caster->CastSpell(procData.victim, GetSeedDamageSpell(aura->GetId()), TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_IGNORE_CASTER_AURA_STATE | TRIGGERED_HIDE_CAST_IN_COMBAT_LOG);
251  return SPELL_AURA_PROC_OK; // no hidden cooldown
252  }
253 
254  // Damage counting
