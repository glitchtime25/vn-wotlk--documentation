317 void BattleGround::Update(uint32 diff)
318 {
319  if (!GetPlayersSize())
320  {
321  // BG is empty
322  // if there are no players invited, delete BG
323  // this will delete arena or bg object, where any player entered
324  // [[ but if you use battleground object again (more battles possible to be played on 1 instance)
325  // then this condition should be removed and code:
326  // if (!GetInvitedCount(HORDE) && !GetInvitedCount(ALLIANCE))
327  // this->AddToFreeBGObjectsQueue(); // not yet implemented
328  // should be used instead of current
329  // ]]
330  // BattleGround Template instance cannot be updated, because it would be deleted
331  if (!GetInvitedCount(HORDE) && !GetInvitedCount(ALLIANCE))
332  delete this;
333 
334  return;
335  }
336 
337  // remove offline players from bg after 5 minutes
338  if (!m_offlineQueue.empty())
339  {
340  BattleGroundPlayerMap::iterator itr = m_players.find(*(m_offlineQueue.begin()));
341  if (itr != m_players.end())
342  {
343  if (itr->second.offlineRemoveTime <= sWorld.GetGameTime())
344  {
345  // add deserter at next login
346  CharacterDatabase.PExecute("UPDATE characters SET at_login = at_login | '%u' WHERE guid = '%u'", uint32(AT_LOGIN_ADD_BG_DESERTER), itr->first.GetCounter());
347 
348  RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
349  m_offlineQueue.pop_front(); // remove from offline queue
350  // do not use itr for anything, because it is erased in RemovePlayerAtLeave()
351  }
352  }
353  }
354 
355  /*********************************************************/
356  /*** BATTLEGROUND BALLANCE SYSTEM ***/
357  /*********************************************************/
358 
359  // if less then minimum players are in on one side, then start premature finish timer
360  if (GetStatus() == STATUS_IN_PROGRESS && !IsArena() && sBattleGroundMgr.GetPrematureFinishTime() && (GetPlayersCountByTeam(ALLIANCE) < GetMinPlayersPerTeam() || GetPlayersCountByTeam(HORDE) < GetMinPlayersPerTeam()))
361  {
362  if (!m_prematureCountDown)
363  {
364  m_prematureCountDown = true;
365  m_prematureCountDownTimer = sBattleGroundMgr.GetPrematureFinishTime();
366  }
367  else if (m_prematureCountDownTimer < diff)
368  {
369  EndBattleGround(GetPrematureWinner());
370  m_prematureCountDown = false;
371  }
372  else if (!sBattleGroundMgr.IsTesting())
373  {
374  uint32 newtime = m_prematureCountDownTimer - diff;
375  // announce every minute
376  if (newtime > (MINUTE * IN_MILLISECONDS))
377  {
378  if (newtime / (MINUTE * IN_MILLISECONDS) != m_prematureCountDownTimer / (MINUTE * IN_MILLISECONDS))
379  PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_prematureCountDownTimer / (MINUTE * IN_MILLISECONDS)));
380  }
381  else
382  {
383  // announce every 15 seconds
384  if (newtime / (15 * IN_MILLISECONDS) != m_prematureCountDownTimer / (15 * IN_MILLISECONDS))
385  PSendMessageToAll(LANG_BATTLEGROUND_PREMATURE_FINISH_WARNING_SECS, CHAT_MSG_SYSTEM, nullptr, (uint32)(m_prematureCountDownTimer / IN_MILLISECONDS));
386  }
387  m_prematureCountDownTimer = newtime;
388  }
389  }
390  else if (m_prematureCountDown)
391  m_prematureCountDown = false;
392 
393  /*********************************************************/
394  /*** ARENA BUFF OBJECT SPAWNING ***/
395  /*********************************************************/
396  if (IsArena() && !m_arenaBuffSpawned)
397  {
398  // 60 seconds after start the buffobjects in arena should get spawned
399  if (m_startTime > uint32(m_startDelayTimes[BG_STARTING_EVENT_FIRST] + ARENA_SPAWN_BUFF_OBJECTS))
400  {
401  SpawnEvent(ARENA_BUFF_EVENT, 0, true);
402  m_arenaBuffSpawned = true;
403  }
404  }
405 
406  /*********************************************************/
407  /*** BATTLEGROUND STARTING SYSTEM ***/
408  /*********************************************************/
409 
410  if (GetStatus() == STATUS_WAIT_JOIN && GetPlayersSize())
411  {
412  float maxDist = GetStartMaxDist();
413  if (maxDist > 0.0f)
414  {
415  if (m_validStartPositionTimer < diff)
416  {
417  for (const auto& itr : GetPlayers())
418  {
419  if (Player* player = sObjectMgr.GetPlayer(itr.first))
420  {
421  float x, y, z, o;
422  GetTeamStartLoc(player->GetTeam(), x, y, z, o);
423  if (!player->IsWithinDist3d(x, y, z, maxDist))
424  {
425  player->TeleportTo(GetMapId(), x, y, z, o);
426  }
427  }
428  }
429  m_validStartPositionTimer = CHECK_PLAYER_POSITION_INVERVAL;
430  }
431  else
432  m_validStartPositionTimer -= diff;
433  }
434 
435  ModifyStartDelayTime(diff);
436 
437  if (!(m_events & BG_STARTING_EVENT_1))
438  {
439  m_events |= BG_STARTING_EVENT_1;
440 
441  StartingEventCloseDoors();
442  SetStartDelayTime(m_startDelayTimes[BG_STARTING_EVENT_FIRST]);
443  // first start warning - 2 or 1 minute, only if defined
444  if (m_startMessageIds[BG_STARTING_EVENT_FIRST])
445  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_FIRST], CHAT_MSG_BG_SYSTEM_NEUTRAL);
446 
447  // Announce BG Started
448  if (sWorld.getConfig(CONFIG_BOOL_BATTLEGROUND_QUEUE_ANNOUNCER_START))
449  {
450  BattleGroundBracketId bracketId = GetBracketId();
451  BattleGroundTypeId BgTypeId = GetTypeId();
452 
453  if (!IsArena())
454  sWorld.SendWorldText(LANG_BG_STARTED_ANNOUNCE_WORLD, GetName(), GetMinLevel(), GetMaxLevel());
455  else
456  sWorld.SendWorldText(LANG_ARENA_STARTED_ANNOUNCE_WORLD, GetName(), GetArenaType(), GetArenaType());
457  }
458  }
459  // After 1 minute or 30 seconds, warning is signalled
460  else if (GetStartDelayTime() <= m_startDelayTimes[BG_STARTING_EVENT_SECOND] && !(m_events & BG_STARTING_EVENT_2))
461  {
462  m_events |= BG_STARTING_EVENT_2;
463  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_SECOND], CHAT_MSG_BG_SYSTEM_NEUTRAL);
464  }
465  // After 30 or 15 seconds, warning is signalled
466  else if (GetStartDelayTime() <= m_startDelayTimes[BG_STARTING_EVENT_THIRD] && !(m_events & BG_STARTING_EVENT_3))
467  {
468  m_events |= BG_STARTING_EVENT_3;
469  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_THIRD], CHAT_MSG_BG_SYSTEM_NEUTRAL);
470  }
471  // delay expired (atfer 2 or 1 minute)
472  else if (GetStartDelayTime() <= 0 && !(m_events & BG_STARTING_EVENT_4))
473  {
474  m_events |= BG_STARTING_EVENT_4;
475 
476  StartingEventOpenDoors();
477 
478  if (m_startMessageIds[BG_STARTING_EVENT_FOURTH])
479  SendMessageToAll(m_startMessageIds[BG_STARTING_EVENT_FOURTH], CHAT_MSG_BG_SYSTEM_NEUTRAL);
480 
481  SetStatus(STATUS_IN_PROGRESS);
482  SetStartDelayTime(m_startDelayTimes[BG_STARTING_EVENT_FOURTH]);
483 
484  // remove preparation
485  if (IsArena())
486  {
487  for (BattleGroundPlayerMap::const_iterator itr = m_players.begin(); itr != m_players.end(); ++itr)
488  {
489  if (Player* player = sObjectMgr.GetPlayer(itr->first))
490  {
491  player->RemoveAurasDueToSpell(SPELL_ARENA_PREPARATION);
492  player->RemoveAurasWithAttribute(SPELL_ATTR_EX5_REMOVE_ENTERING_ARENA);
493  }
494  }
495 
496  CheckArenaWinConditions();
497  }
498  else
499  {
500  PlaySoundToAll(SOUND_BG_START);
501 
502  for (BattleGroundPlayerMap::const_iterator itr = m_players.begin(); itr != m_players.end(); ++itr)
503  if (Player* plr = sObjectMgr.GetPlayer(itr->first))
504  plr->RemoveAurasDueToSpell(SPELL_PREPARATION);
505  }
506  }
507  }
508 
509  /*********************************************************/
510  /*** BATTLEGROUND ENDING SYSTEM ***/
511  /*********************************************************/
512  if (GetStatus() == STATUS_IN_PROGRESS && IsArena())
513  {
514  // after 45 minutes without one team losing, the arena closes with no winner and -16 rating change for both
515  if (m_startTime > uint32(m_startDelayTimes[BG_STARTING_EVENT_FIRST] + ARENA_FORCED_DRAW))
516  {
517  EndBattleGround(TEAM_NONE);
518  }
519  }
520 
521  if (GetStatus() == STATUS_WAIT_LEAVE)
522  {
523  // remove all players from battleground after 2 minutes
524  m_endTime -= diff;
525  if (m_endTime <= 0)
526  {
527  m_endTime = 0;
528  BattleGroundPlayerMap::iterator next;
529  for (BattleGroundPlayerMap::iterator itr = m_players.begin(); itr != m_players.end(); itr = next)
530  {
531  next = itr;
532  ++next;
533  // itr is erased here!
534  RemovePlayerAtLeave(itr->first, true, true);// remove player from BG
535  // do not change any battleground's private variables
536  }
537  }
538  }
539 
540  // update start time
541  m_startTime += diff;
542 }
