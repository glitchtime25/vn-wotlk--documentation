77 // Method that will reset the event in case of a fail
78 void instance_violet_hold::ResetEvent()
79 {
80  // reset everything
81  ResetVariables();
82  UpdateWorldState(false);
83  ResetGuards();
84  ResetIntroPortals();
85  UpdateCrystals(true);
86 
87  // respawn sinclari and the controller
88  if (Creature* pSinclari = GetSingleCreatureFromStorage(NPC_SINCLARI))
89  {
90  if (!pSinclari->IsAlive())
91  pSinclari->Respawn();
92  }
93  if (Creature* pController = GetSingleCreatureFromStorage(NPC_EVENT_CONTROLLER))
94  {
95  if (!pController->IsAlive())
96  pController->Respawn();
97  }
98 
99  for (auto spawn : m_vRandomBosses)
100  {
101  const BossInformation* pData = GetBossInformation(spawn->uiEntry);
102  if (!pData)
103  return;
104 
105  if (m_auiEncounter[pData->uiType] == DONE)
106  {
107  // Despawn ghost boss
108  if (Creature* pGhostBoss = GetSingleCreatureFromStorage(pData->uiGhostEntry, true))
109  pGhostBoss->ForcedDespawn();
110 
111  // Spawn new boss replacement
112  if (Creature* pSummoner = GetSingleCreatureFromStorage(NPC_SINCLARI_ALT))
113  pSummoner->SummonCreature(pData->uiGhostEntry, spawn->fX, spawn->fY, spawn->fZ, spawn->fO, TEMPSPAWN_DEAD_DESPAWN, 0);
114 
115  // Replace Erekem guards
116  if (pData->uiType == TYPE_EREKEM)
117  {
118  // Despawn ghost guards
119  for (const auto& guid : m_lArakkoaGuardList)
120  if (Creature* pGhostGuard = instance->GetCreature(guid))
121  pGhostGuard->ForcedDespawn();
122 
123  m_lArakkoaGuardList.clear();
124 
125  // Spawn new guards replacement
126  float fX, fY, fZ, fO;
127  for (const auto& guid : m_lErekemGuardList)
128  {
129  if (Creature* pGuard = instance->GetCreature(guid))
130  {
131  // Don't allow alive original guards while the boss is dead
132  if (!pGuard->IsDead())
133  pGuard->ForcedDespawn();
134 
135  // Spawn a ghost guard for each original guard
136  pGuard->GetRespawnCoord(fX, fY, fZ, &fO);
137  pGuard->SummonCreature(NPC_ARAKKOA_GUARD, fX, fY, fZ, fO, TEMPSPAWN_DEAD_DESPAWN, 0);
138  }
139  }
140  }
141  }
142  // failed encounters make boss to respawn
143  else if (m_auiEncounter[pData->uiType] == FAIL)
144  {
145  if (Creature* pBoss = GetSingleCreatureFromStorage(pData->uiEntry))
146  {
147  pBoss->Respawn();
148 
149  // make sure it has proper flags just in case; normally handled directly in core
150  pBoss->SetImmuneToNPC(true);
151  pBoss->SetImmuneToPlayer(true);
152  }
153 
154  if (pData->uiType == TYPE_EREKEM)
155  {
156  // respawn guards
157  for (const auto& guid : m_lErekemGuardList)
158  {
159  if (Creature* pGuard = instance->GetCreature(guid))
160  {
161  pGuard->Respawn();
162 
163  // make sure it has proper flags just in case
164  pGuard->SetImmuneToNPC(true);
165  pGuard->SetImmuneToPlayer(true);
166  }
167  }
168  }
169  }
170 
171  // Close Door if still open
172  if (pData && (m_auiEncounter[pData->uiType] == DONE || m_auiEncounter[pData->uiType] == FAIL))
173  UpdateCellForBoss(pData->uiEntry, true);
174  }
175 
176  // reset data
177  SetData(TYPE_MAIN, NOT_STARTED);
178 }

345 void instance_violet_hold::SetData(uint32 uiType, uint32 uiData)
346 {
347  debug_log("SD2: instance_violet_hold: SetData got type %u, data %u.", uiType, uiData);
348 
349  switch (uiType)
350  {
351  case TYPE_MAIN:
352  {
353  if (uiData == m_auiEncounter[uiType])
354  return;
355  if (m_auiEncounter[uiType] == DONE)
356  return;
357 
358  switch (uiData)
359  {
360  case IN_PROGRESS:
361  StartEvent();
362  break;
363  case FAIL:
364  // Despawn sinclari and the controller
365  // creature despawn (creature cleanup is handled in creature_linking)
366  if (Creature* pController = GetSingleCreatureFromStorage(NPC_EVENT_CONTROLLER))
367  pController->ForcedDespawn();
368 
369  // open instance door
370  DoUseDoorOrButton(GO_PRISON_SEAL_DOOR);
371  m_uiEventResetTimer = 20000; // Timer may not be correct - 20 sec is default reset timer for blizz
372  break;
373  case DONE:
374  DoUseDoorOrButton(GO_PRISON_SEAL_DOOR);
375  UpdateWorldState(false);
376  break;
377  case SPECIAL:
378  m_uiIntroSummonTimer = 0;
379  m_uiEventStartTimer = 20000;
380  break;
381  }
382  m_auiEncounter[uiType] = uiData;
383  break;
384  }
385  case TYPE_SEAL:
386  m_auiEncounter[uiType] = uiData;
387  if (uiData == SPECIAL)
388  {
389  --m_uiWorldStateSealCount;
390  DoUpdateWorldState(WORLD_STATE_SEAL, m_uiWorldStateSealCount);
391 
392  // Yell at 75%, 50% and 25% shield
393  if (m_uiWorldStateSealCount < 100 - 25 * m_uiSealYellCount)
394  {
395  if (Creature* pSinclari = GetSingleCreatureFromStorage(NPC_SINCLARI_ALT))
396  {
397  // ToDo: I'm not sure if the last yell should be at 25% or at 5%. Needs research
398  ++m_uiSealYellCount;
399  DoScriptText(aSealWeakYell[m_uiSealYellCount - 1], pSinclari);
400  }
401  }
402 
403  // set achiev to failed
404  if (m_bIsDefenseless)
405  m_bIsDefenseless = false;
406 
407  if (!m_uiWorldStateSealCount)
408  {
409  SetData(TYPE_MAIN, FAIL);
410  SetData(TYPE_SEAL, NOT_STARTED);
411  }
412  }
413  break;
414  case TYPE_PORTAL:
415  {
416  switch (uiData)
417  {
418  case SPECIAL: // timer to next
419  m_uiPortalTimer = 90000;
420  break;
421  case DONE: // portal done, set timer to 5 secs
422  m_uiPortalTimer = 3000;
423  break;
424  }
425  m_auiEncounter[uiType] = uiData;
426  break;
427  }
428  case TYPE_LAVANTHOR:
429  case TYPE_MORAGG:
430  case TYPE_EREKEM:
431  case TYPE_ICHORON:
432  case TYPE_XEVOZZ:
433  case TYPE_ZURAMAT:
434  if (uiData == DONE)
435  m_uiPortalTimer = 35000;
436  if (m_auiEncounter[uiType] != DONE) // Keep the DONE-information stored
437  m_auiEncounter[uiType] = uiData;
438  // Handle achievements if necessary
439  if (uiData == IN_PROGRESS)
440  {
441  if (uiType == TYPE_ZURAMAT)
442  m_bIsVoidDance = true;
443  else if (uiType == TYPE_ICHORON)
444  m_bIsDehydratation = true;
445  }
446  if (uiData == SPECIAL && uiType == TYPE_ICHORON)
447  m_bIsDehydratation = false;
448  if (uiData == FAIL)
449  SetData(TYPE_MAIN, FAIL);
450  break;
451  case TYPE_CYANIGOSA:
452  if (uiData == DONE)
453  SetData(TYPE_MAIN, DONE);
454  if (uiData == FAIL)
455  SetData(TYPE_MAIN, FAIL);
456  m_auiEncounter[uiType] = uiData;
457  break;
458  default:
459  return;
460  }
461 
462  if (uiData == DONE)
463  {
464  OUT_SAVE_INST_DATA;
465 
466  std::ostringstream saveStream;
467  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
468  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
469  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8] << " "
470  << m_auiEncounter[9];
471 
472  m_strInstData = saveStream.str();
473 
474  SaveToDB();
475  OUT_SAVE_INST_DATA_COMPLETE;
476  }
477 }

601 // Method that will release the boss from the cage
602 void instance_violet_hold::DoReleaseBoss(uint32 entry)
603 {
604  const BossInformation* pData = GetBossInformation(entry);
605  if (!pData)
606  {
607  script_error_log("instance_violet_hold: cannot find boss information");
608  return;
609  }
610 
611  Creature* pBoss = GetSingleCreatureFromStorage(GetData(pData->uiType) != DONE ? pData->uiEntry : pData->uiGhostEntry);
612  if (!pBoss)
613  {
614  script_error_log("instance_violet_hold: cannot find boss creature %u", pBoss->GetEntry());
615  return;
616  }
617 
618  UpdateCellForBoss(pData->uiEntry);
619 
620  if (pData->iSayEntry)
621  DoScriptText(pData->iSayEntry, pBoss);
622 
623  // move boss outside of the cell; flags are removed by DB script
624  pBoss->GetMotionMaster()->MoveWaypoint();
625 
626  // Handle Erekem guards

836 void instance_violet_hold::Update(uint32 uiDiff)

867  // Note: timer and frequency of intro mobs requires more research
868  DoSpawnIntroMob();
869  m_uiIntroSummonTimer = urand(5000, 10000);
870  }
871  else
872  m_uiIntroSummonTimer -= uiDiff;
873  }
874 
875  if (m_uiCyanigosaMoveTimer)
876  {
877  if (m_uiCyanigosaMoveTimer <= uiDiff)

879  if (Creature* pCyanigosa = GetSingleCreatureFromStorage(NPC_CYANIGOSA))
880  pCyanigosa->GetMotionMaster()->MoveJump(afPortalLocation[8].fX, afPortalLocation[8].fY, afPortalLocation[8].fZ, pCyanigosa->GetSpeed(MOVE_RUN) * 2, 10.0f);
881 
882  m_uiCyanigosaMoveTimer = 0;
883  }
884  else
885  m_uiCyanigosaMoveTimer -= uiDiff;
886  }
887 
888  // no update when event isn't in progress
889  if (m_auiEncounter[TYPE_MAIN] != IN_PROGRESS)
890  return;
891 
892  // portal timer
893  if (m_uiPortalTimer)
894  {
895  if (m_uiPortalTimer <= uiDiff)

897  DoUpdateWorldState(WORLD_STATE_PORTALS, ++m_uiWorldStatePortalCount);

902  m_uiPortalTimer = 0;
