72 void instance_black_temple::OnCreatureCreate(Creature* creature)
73 {
74  switch (creature->GetEntry())
75  {
76  case NPC_SPIRIT_OF_OLUM:
77  case NPC_SPIRIT_OF_UDALO:
78  // Use only the summoned versions
79  if (!creature->IsTemporarySummon())
80  break;
81  case NPC_AKAMA:
82  case NPC_ILLIDAN_STORMRAGE:
83  case NPC_MAIEV_SHADOWSONG:
84  case NPC_AKAMA_SHADE:
85  case NPC_SHADE_OF_AKAMA:
86  case NPC_RELIQUARY_OF_SOULS:
87  case NPC_GATHIOS:
88  case NPC_ZEREVOR:
89  case NPC_LADY_MALANDE:
90  case NPC_VERAS:
91  case NPC_ILLIDARI_COUNCIL:
92  case NPC_COUNCIL_VOICE:
93  case NPC_ILLIDAN_DOOR_TRIGGER:
94  case NPC_BLACK_TEMPLE_TRIGGER:
95  case NPC_TERON_GOREFIEND:
96  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
97  break;
98  case NPC_SHADOWY_CONSTRUCT:
99  m_shadowyConstructs.push_back(creature->GetObjectGuid());
100  break;
101  case NPC_ASH_CHANNELER:
102  m_channelersGuidVector.push_back(creature->GetObjectGuid());
103  break;
104  case NPC_CREATURE_GENERATOR:
105  m_creatureGeneratorGuidVector.push_back(creature->GetObjectGuid());
106  break;
107  case NPC_GLAIVE_TARGET:
108  m_vGlaiveTargetGuidVector.push_back(creature->GetObjectGuid());
109  break;
110  case NPC_ASHTONGUE_BATTLELORD:
111  case NPC_ASHTONGUE_MYSTIC:
112  case NPC_ASHTONGUE_STORMCALLER:
113  case NPC_ASHTONGUE_PRIMALIST:
114  case NPC_STORM_FURY:
115  case NPC_ASHTONGUE_FERAL_SPIRIT:
116  case NPC_ASHTONGUE_STALKER:
117  m_ashtongue.push_back(creature->GetObjectGuid());
118  if (m_auiEncounter[TYPE_SHADE] == DONE)
119  creature->setFaction(FACTION_ASHTONGUE_FRIENDLY);
120  break;
121  case NPC_VENGEFUL_SPIRIT:
122  if (Creature* teron = GetSingleCreatureFromStorage(NPC_TERON_GOREFIEND))
123  teron->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, creature, teron);
124  break;
125  case NPC_SKY_STALKER:
126  case NPC_WIND_REAVER:
127  if (creature->GetPositionZ() > 160.0f)
128  creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
129  break;
130  case NPC_RELIQUARY_COMBAT_TRIGGER:
131  creature->GetCombatManager().SetLeashingDisable(true);
132  m_npcEntryGuidStore[creature->GetEntry()] = creature->GetObjectGuid();
133  break;
134  case NPC_ANGERED_SOUL_FRAGMENT:
135  case NPC_SUFFERING_SOUL_FRAGMENT:
136  case NPC_HUNGERING_SOUL_FRAGMENT:
137  m_soulFragments.push_back(creature->GetObjectGuid());
138  break;
139  case NPC_ENSLAVED_SOUL:
140  m_enslavedSouls.push_back(creature->GetObjectGuid());
141  break;
142  case NPC_WORLD_TRIGGER:
143  {
144  float z = creature->GetPositionZ();
145  if (z > 355.f && z < 365.f)
146  m_illidanTriggersLower.push_back(creature->GetObjectGuid());
147  else if (z > 365.f)
148  m_illidanTriggersUpper.push_back(creature->GetObjectGuid());
149  break;
150  }
151  case NPC_BLADE_OF_AZZINOTH:
152  GuidVector& blades = m_npcEntryGuidCollection[creature->GetEntry()];
153  if (blades.size() == 2) // new attempt
154  blades.clear();
155  blades.push_back(creature->GetObjectGuid());
156  break;
157  }
158 
159  switch (creature->GetEntry())
160  {
161  case NPC_FLAME_CRASH:
162  case NPC_BLADE_OF_AZZINOTH:
163  case NPC_ILLIDAN_TARGET:
164  case NPC_SHADOW_DEMON:
165  case NPC_ILLIDARI_ELITE:
166  case NPC_PARASITIC_SHADOWFIEND:
167  case NPC_FLAME_OF_AZZINOTH:
168  case NPC_DEMON_FIRE:
169  case NPC_BLAZE:
170  case NPC_CAGE_TRAP_DISTURB_TRIGGER:
171  m_illidanSpawns.push_back(creature->GetObjectGuid());
172  break;
173  }
174 }

202 void instance_black_temple::OnObjectCreate(GameObject* go)
203 {
204  switch (go->GetEntry())
205  {
206  case GO_NAJENTUS_GATE: // Gate past Naj'entus (at the entrance to Supermoose's courtyards)
207  if (m_auiEncounter[TYPE_NAJENTUS] == DONE)
208  go->SetGoState(GO_STATE_ACTIVE);
209  break;
210  case GO_SUPREMUS_DOORS: // Main Temple Doors - right past Supermoose (Supremus)
211  if (m_auiEncounter[TYPE_SUPREMUS] == DONE)
212  go->SetGoState(GO_STATE_ACTIVE);
213  break;
214  case GO_SHADE_OF_AKAMA: // Door close during encounter
215  case GO_GOREFIEND_DOOR: // Door close during encounter
216  break;
217  case GO_GURTOGG_DOOR: // Door opens after encounter
218  if (m_auiEncounter[TYPE_BLOODBOIL] == DONE)
219  go->SetGoState(GO_STATE_ACTIVE);
220  break;
221  case GO_PRE_SHAHRAZ_DOOR: // Door leading to Mother Shahraz
222  if (m_auiEncounter[TYPE_SHADE] == DONE && m_auiEncounter[TYPE_GOREFIEND] == DONE && m_auiEncounter[TYPE_BLOODBOIL] == DONE && m_auiEncounter[TYPE_RELIQUIARY] == DONE)
223  go->SetGoState(GO_STATE_ACTIVE);
224  break;
225  case GO_POST_SHAHRAZ_DOOR: // Door after shahraz
226  if (m_auiEncounter[TYPE_SHAHRAZ] == DONE)
227  go->SetGoState(GO_STATE_ACTIVE);
228  break;
229  case GO_ILLIDAN_GATE: // Gate leading to Temple Summit
230  if (m_auiEncounter[TYPE_ILLIDAN] == DONE)
231  go->SetGoState(GO_STATE_ACTIVE);
232  break;
233  case GO_PRE_COUNCIL_DOOR: // Door leading to the Council (grand promenade)
234  case GO_COUNCIL_DOOR: // Door leading to the Council (inside)
235  case GO_ILLIDAN_DOOR_R: // Right door at Temple Summit
236  case GO_ILLIDAN_DOOR_L: // Left door at Temple Summit
237  break;
238  case GO_IMPALING_SPINE:
239  m_impalingSpines.push_back(go->GetObjectGuid());
240  break;
241 
242  default:
243  return;
244  }
245  m_goEntryGuidStore[go->GetEntry()] = go->GetObjectGuid();
246 }

278 void instance_black_temple::SetData(uint32 type, uint32 data)
279 {
280  switch (type)
281  {
282  case TYPE_NAJENTUS:
283  m_auiEncounter[type] = data;
284  if (data == DONE)
285  {
286  DoUseDoorOrButton(GO_NAJENTUS_GATE);
287  if (Creature* trigger = GetSingleCreatureFromStorage(NPC_BLACK_TEMPLE_TRIGGER))
288  DoScriptText(EMOTE_OPEN_NAJENTUS_DOOR, trigger);
289  }
290  break;
291  case TYPE_SUPREMUS:
292  m_auiEncounter[type] = data;
293  DoUseDoorOrButton(GO_NAJENTUS_GATE);
294  if (data == DONE)
295  DoUseDoorOrButton(GO_SUPREMUS_DOORS);
296  break;
297  case TYPE_SHADE:
298  m_auiEncounter[type] = data;
299  // combat door
300  DoUseDoorOrButton(GO_SHADE_OF_AKAMA);
301  if (data == FAIL)
302  // Despawn channelers on fail
303  for (ObjectGuid guid : m_channelersGuidVector)
304  if (Creature* channeler = instance->GetCreature(guid))
305  channeler->ForcedDespawn();
306  if (data != IN_PROGRESS)
307  for (ObjectGuid guid : m_creatureGeneratorGuidVector) // notify generators to stop spawning
308  if (Creature* generator = instance->GetCreature(guid))
309  generator->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, generator, generator);
310  if (data == DONE)
311  {
312  DoOpenPreMotherDoor();
313  for (ObjectGuid guid : m_ashtongue)
314  if (Creature* ashtongue = instance->GetCreature(guid))
315  ashtongue->setFaction(FACTION_ASHTONGUE_FRIENDLY);
316  }
317  break;
318  case TYPE_GOREFIEND:
319  m_auiEncounter[type] = data;
320  DoUseDoorOrButton(GO_GOREFIEND_DOOR);
321  if (data == DONE)
322  DoOpenPreMotherDoor();
323  break;
324  case TYPE_BLOODBOIL:
325  m_auiEncounter[type] = data;
326  if (data == DONE)
327  {
328  DoOpenPreMotherDoor();
329  DoUseDoorOrButton(GO_GURTOGG_DOOR);
330  }
331  break;
332  case TYPE_RELIQUIARY:
333  m_auiEncounter[type] = data;
334  if (data == IN_PROGRESS)
335  {
336  if (Creature* trigger = GetSingleCreatureFromStorage(NPC_RELIQUARY_COMBAT_TRIGGER))
337  {
338  trigger->SetActiveObjectState(true);
339  trigger->SetInCombatWithZone(false);
340  if (!trigger->IsInCombat())
341  {
342  SetData(TYPE_RELIQUIARY, FAIL);
343  return;
344  }
345  }
346  }
347  if (data == DONE || data == FAIL)
348  {
349  if (Creature* trigger = GetSingleCreatureFromStorage(NPC_RELIQUARY_COMBAT_TRIGGER))
350  {
351  trigger->CombatStop();
352  trigger->SetActiveObjectState(false);
353  }
354  }
355  if (data == DONE)
356  {
357  for (ObjectGuid guid : m_soulFragments)
358  {
359  if (Creature* soul = instance->GetCreature(guid))
360  {
361  soul->SetRespawnDelay(time(nullptr) + 7 * DAY);
362  soul->ForcedDespawn();
363  }
364  }
365  DoOpenPreMotherDoor();
366  }
367  if (data == FAIL)
368  {
369  for (ObjectGuid guid : m_soulFragments)
370  {
371  if (Creature* soul = instance->GetCreature(guid))
372  {
373  soul->SetRespawnDelay(15);
374  soul->Respawn();
375  }
376  }
377  }
378  break;
379  case TYPE_SHAHRAZ:
380  if (data == DONE)
381  DoUseDoorOrButton(GO_POST_SHAHRAZ_DOOR);
382  m_auiEncounter[type] = data;
383  break;
384  case TYPE_COUNCIL:
385  // Don't set the same data twice
386  if (m_auiEncounter[type] == data)
387  return;
388  m_auiEncounter[type] = data;
389  DoUseDoorOrButton(GO_COUNCIL_DOOR);
390  if (data == DONE)
391  DoTeleportAkamaIfCan();
392  break;
393  case TYPE_ILLIDAN:
394  if (data == IN_PROGRESS && GetData(TYPE_COUNCIL) != DONE)
395  BanPlayersIfNoGm("Player engaged Illidan without killing council and Gamemaster being present in instance.");
396  DoUseDoorOrButton(GO_ILLIDAN_DOOR_R);
397  DoUseDoorOrButton(GO_ILLIDAN_DOOR_L);
398  m_auiEncounter[type] = data;
399  if (data == FAIL)
400  {
401  // Cleanup encounter
402  if (Creature* illidan = GetSingleCreatureFromStorage(NPC_ILLIDAN_STORMRAGE))
403  illidan->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, illidan, illidan);
404  }
405  break;
406  default:
407  script_error_log("Instance Black Temple: ERROR SetData = %u for type %u does not exist/not implemented.", type, data);
408  return;
409  }
410 
411  if (data == DONE)
412  {
413  OUT_SAVE_INST_DATA;
414 
415  std::ostringstream saveStream;
416  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
417  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " "
418  << m_auiEncounter[6] << " " << m_auiEncounter[7] << " " << m_auiEncounter[8];
419 
420  m_strInstData = saveStream.str();
421 
422  SaveToDB();
423  OUT_SAVE_INST_DATA_COMPLETE;
424  }
425 }

474 void instance_black_temple::Load(const char* chrIn)
