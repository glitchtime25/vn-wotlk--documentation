94  // Workaround for the shared health pool
95  void DamageTaken(Unit* /*dealer*/, uint32& damage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
96  {
97  if (!m_instance || (spellInfo && spellInfo->Id == SPELL_TWIN_EMPATHY))
98  return;
99 
100  if (Creature* twin = m_instance->GetSingleCreatureFromStorage(m_creature->GetEntry() == NPC_VEKLOR ? NPC_VEKNILASH : NPC_VEKLOR))
101  {
102  float fDamPercent = ((float)damage) / ((float)m_creature->GetMaxHealth());
103  int32 twinDamage = (int32)(fDamPercent * ((float)twin->GetMaxHealth()));
104  if (twin->GetHealth() <= uint32(twinDamage))
105  twin->CastSpell(nullptr, SPELL_TWIN_SUICIDE, TRIGGERED_OLD_TRIGGERED);
106  else
107  twin->CastCustomSpell(nullptr, SPELL_TWIN_EMPATHY, &twinDamage, nullptr, nullptr, TRIGGERED_OLD_TRIGGERED);
108  }
109  }

111  // Workaround for the shared health pool
112  void HealedBy(Unit* /*healer*/, uint32& uiHealedAmount) override
113  {
114  if (!m_instance)
115  return;
116 
117  if (Creature* twin = m_instance->GetSingleCreatureFromStorage(m_creature->GetEntry() == NPC_VEKLOR ? NPC_VEKNILASH : NPC_VEKLOR))
118  {
119  float fHealPercent = ((float)uiHealedAmount) / ((float)m_creature->GetMaxHealth());
120  uint32 uiTwinHeal = (uint32)(fHealPercent * ((float)twin->GetMaxHealth()));
121  uint32 uiTwinHealth = twin->GetHealth() + uiTwinHeal;
122  twin->SetHealth(uiTwinHealth < twin->GetMaxHealth() ? uiTwinHealth : twin->GetMaxHealth());
