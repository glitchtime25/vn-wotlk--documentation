100 CombatManeuverReturns PlayerbotWarriorAI::DoFirstCombatManeuver(Unit* pTarget)
101 {
102  // There are NPCs in BGs and Open World PvP, so don't filter this on PvP scenarios (of course if PvP targets anyone but tank, all bets are off anyway)
103  // Wait until the tank says so, until any non-tank gains aggro or X seconds - whichever is shortest
104  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO)
105  {
106  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.GroupTankHoldsAggro())
107  {
108  if (PlayerbotAI::ORDERS_TANK & m_ai.GetCombatOrder())
109  {
110  if (m_bot.GetDistance(pTarget, true, DIST_CALC_COMBAT_REACH_WITH_MELEE) <= ATTACK_DISTANCE)
111  {
112  // Set everyone's UpdateAI() waiting to 2 seconds
113  m_ai.SetGroupIgnoreUpdateTime(2);
114  // Clear their TEMP_WAIT_TANKAGGRO flag
115  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
116  // Start attacking, force target on current target
117  m_ai.Attack(m_ai.GetCurrentTarget());
118 
119  // While everyone else is waiting 2 second, we need to build up aggro, so don't return
120  }
121  else
122  {
123  // TODO: add check if target is ranged
124  return RETURN_NO_ACTION_OK; // wait for target to get nearer
125  }
126  }
127  else
128  return RETURN_NO_ACTION_OK; // wait it out
129  }
130  else
131  {
132  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_TANKAGGRO);
133  }
134  }
135 
136  if (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TEMP_WAIT_OOC)
137  {
138  if (m_WaitUntil > m_ai.CurrentTime() && m_ai.IsGroupReady())
139  return RETURN_NO_ACTION_OK; // wait it out
140  else
141  m_ai.ClearGroupCombatOrder(PlayerbotAI::ORDERS_TEMP_WAIT_OOC);
142  }
143 
144  switch (m_ai.GetScenarioType())
145  {
146  case PlayerbotAI::SCENARIO_PVP_DUEL:
147  case PlayerbotAI::SCENARIO_PVP_BG:
148  case PlayerbotAI::SCENARIO_PVP_ARENA:
149  case PlayerbotAI::SCENARIO_PVP_OPENWORLD:
150  return DoFirstCombatManeuverPVP(pTarget);
151  case PlayerbotAI::SCENARIO_PVE:
152  case PlayerbotAI::SCENARIO_PVE_ELITE:
153  case PlayerbotAI::SCENARIO_PVE_RAID:
154  default:
155  return DoFirstCombatManeuverPVE(pTarget);
156  }

159 CombatManeuverReturns PlayerbotWarriorAI::DoFirstCombatManeuverPVE(Unit* pTarget)
160 {
161  float fTargetDist = m_bot.GetDistance(pTarget, true, DIST_CALC_COMBAT_REACH_WITH_MELEE);
162 
163  if (DEFENSIVE_STANCE && (m_ai.GetCombatOrder() & PlayerbotAI::ORDERS_TANK))
164  {
165  if (!m_bot.HasAura(DEFENSIVE_STANCE, EFFECT_INDEX_0) && m_ai.CastSpell(DEFENSIVE_STANCE) == SPELL_CAST_OK)
166  return RETURN_CONTINUE;
167  else if (TAUNT > 0 && m_bot.HasAura(DEFENSIVE_STANCE, EFFECT_INDEX_0) && m_ai.CastSpell(TAUNT, *pTarget) == SPELL_CAST_OK)
168  return RETURN_FINISHED_FIRST_MOVES;
169  }
170 
171  if (BERSERKER_STANCE)
172  {
173  if (!m_bot.HasAura(BERSERKER_STANCE, EFFECT_INDEX_0) && m_ai.CastSpell(BERSERKER_STANCE) == SPELL_CAST_OK)
174  return RETURN_CONTINUE;
175  if (BLOODRAGE > 0 && m_bot.HasAura(BERSERKER_STANCE, EFFECT_INDEX_0) && m_ai.GetRageAmount() <= 10)
176  return m_ai.CastSpell(BLOODRAGE) == SPELL_CAST_OK ? RETURN_FINISHED_FIRST_MOVES : RETURN_NO_ACTION_ERROR;
177  if (INTERCEPT > 0 && m_bot.HasAura(BERSERKER_STANCE, EFFECT_INDEX_0))
178  {
179  if (fTargetDist < 8.0f)
180  return RETURN_NO_ACTION_OK;
181  else if (fTargetDist > 25.0f)
182  return RETURN_CONTINUE; // wait to come into range
183  else if (INTERCEPT > 0 && m_ai.CastSpell(INTERCEPT, *pTarget) == SPELL_CAST_OK)
184  {
185  float x, y, z;
186  pTarget->GetContactPoint(&m_bot, x, y, z, 3.666666f);
187  m_bot.Relocate(x, y, z);
188  return RETURN_FINISHED_FIRST_MOVES;
189  }
190  }
191  }
192 
193  if (BATTLE_STANCE)
194  {
195  if (!m_bot.HasAura(BATTLE_STANCE, EFFECT_INDEX_0) && m_ai.CastSpell(BATTLE_STANCE) == SPELL_CAST_OK)
196  return RETURN_CONTINUE;
197  if (CHARGE > 0 && m_bot.HasAura(BATTLE_STANCE, EFFECT_INDEX_0))
198  {
199  if (fTargetDist < 8.0f)
200  return RETURN_NO_ACTION_OK;
201  if (fTargetDist > 25.0f)
202  return RETURN_CONTINUE; // wait to come into range
203  else if (CHARGE > 0 && m_ai.CastSpell(CHARGE, *pTarget) == SPELL_CAST_OK)
204  {
205  float x, y, z;
206  pTarget->GetContactPoint(&m_bot, x, y, z, 3.666666f);
207  m_bot.Relocate(x, y, z);
208  return RETURN_FINISHED_FIRST_MOVES;
209  }
210  }
211  }
212 
213  return RETURN_NO_ACTION_OK;
214 }
