100  boss_nothAI(Creature* creature) : BossAI(creature, NOTH_ACTIONS_MAX),
101  m_instance(static_cast<instance_naxxramas*>(creature->GetInstanceData())),
102  m_isRegularMode(creature->GetMap()->IsRegularDifficulty())
103  {
104  SetDataType(TYPE_NOTH);
105  AddOnKillText(SAY_SLAY1, SAY_SLAY2);
106  AddOnDeathText(SAY_DEATH);
107  AddOnAggroText(SAY_AGGRO1, SAY_AGGRO2, SAY_AGGRO3);
108  AddCombatAction(NOTH_PHASE_BALCONY, 90s);
109  AddCombatAction(NOTH_PHASE_GROUND, true);
110  Reset();
111  }

119  void Reset() override
120  {
121  m_uiPhase = PHASE_GROUND;
122  m_uiPhaseSub = PHASE_GROUND;

172  std::chrono::milliseconds GetSubsequentActionTimer(uint32 action)
173  {
174  switch (action)
175  {
176  case NOTH_PHASE_GROUND:
177  {
178  return 110s;
179  }
180  case NOTH_PHASE_BALCONY:
181  {
182  return 70s;
183  }
184  default: return 0s;
185  }
186  }
187 
188  void ExecuteAction(uint32 action) override
189  {
190  switch (action)
191  {
192  case NOTH_PHASE_BALCONY:
193  {
194  if (DoCastSpellIfCan(nullptr, SPELL_TELEPORT) == CAST_OK)
195  {
196  static const std::vector<uint32> GROUND_PHASE_ACTIONS = {NOTH_PHASE_BALCONY};
197  for (uint32 action : GROUND_PHASE_ACTIONS)
198  DisableCombatAction(action);
199 
200  DoBroadcastText(EMOTE_TELEPORT, m_creature);
201  SetAIImmobilizedState(true);
202  m_creature->ApplySpellImmune(nullptr, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_ALL, true);
203  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_IMMUNE_TO_PLAYER);
204  SetMeleeEnabled(false);
205  StopTargeting(true);
206  m_creature->SetTarget(nullptr);
207  ++m_uiPhaseSub;
208  m_uiPhase = PHASE_BALCONY;
209 
210  ResetCombatAction(NOTH_PHASE_GROUND, GetSubsequentActionTimer(NOTH_PHASE_GROUND));
211  }
212  return;
213  }
214  case NOTH_PHASE_GROUND:
215  {
216  if (DoCastSpellIfCan(nullptr, SPELL_TELEPORT_RETURN) == CAST_OK)
217  {
218  DoBroadcastText(EMOTE_TELEPORT_RETURN, m_creature);
219  SetAIImmobilizedState(false);
220  m_creature->ApplySpellImmune(nullptr, IMMUNITY_DAMAGE, SPELL_SCHOOL_MASK_ALL, false);
221  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_IMMUNE_TO_PLAYER);
222  SetMeleeEnabled(true);
223  HandleTargetRestoration();
224  m_uiPhase = PHASE_GROUND;
225  ResetCombatAction(NOTH_PHASE_BALCONY, GetSubsequentActionTimer(NOTH_PHASE_BALCONY));
226  DisableCombatAction(action);
227  AddInitialCooldowns();
228  }
229  return;
230  }
231  }
232  ResetCombatAction(action, GetSubsequentActionTimer(action));
233  }
