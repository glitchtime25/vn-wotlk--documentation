110 void OutdoorPvP::HandlePlayerKill(Player* killer, Unit* victim)
111 {
112  if (Group* group = killer->GetGroup())
113  {
114  for (GroupReference* itr = group->GetFirstMember(); itr != nullptr; itr = itr->next())
115  {
116  Player* groupMember = itr->getSource();
117 
118  if (!groupMember)
119  continue;
120 
121  // skip if too far away
122  if (!groupMember->IsAtGroupRewardDistance(victim))
123  continue;
124 
125  if (victim->IsTrivialForTarget(groupMember))
126  continue;
127 
128  // creature kills must be notified, even if not inside objective / not outdoor pvp active
129  // player kills only count if active and inside objective
130  if (groupMember->CanUseCapturePoint())
131  HandlePlayerKillInsideArea(groupMember, victim);
132  }
133  }
134  else
135  {
136  // creature kills must be notified, even if not inside objective / not outdoor pvp active
137  if (killer && killer->CanUseCapturePoint() && !victim->IsTrivialForTarget(killer))
138  HandlePlayerKillInsideArea(killer, victim);
139  }
140 }
