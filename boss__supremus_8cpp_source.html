209  void ExecuteAction(uint32 action) override
210  {
211  switch (action)
212  {
213  case SUPREMUS_ACTION_PHASE_SWITCH:
214  {
215  if (m_bTankPhase)
216  {
217  m_creature->CastSpell(nullptr, SPELL_SLOW_SELF, TRIGGERED_OLD_TRIGGERED);
218  DoScriptText(EMOTE_GROUND_CRACK, m_creature);
219  m_bTankPhase = false;
220  DoResetThreat();
221  DisableTimer(SUPREMUS_ACTION_HATEFUL_STRIKE);
222  SetActionReadyStatus(SUPREMUS_ACTION_HATEFUL_STRIKE, false);
223  ResetTimer(SUPREMUS_ACTION_VOLCANIC_ERUPTION, GetInitialActionTimer(SUPREMUS_ACTION_VOLCANIC_ERUPTION));
224  SetCombatScriptStatus(true);
225  SetCombatMovement(false, true);
226  SetMeleeEnabled(false);
227  m_creature->SetTarget(nullptr);
228  ResetTimer(SUPREMUS_ACTION_DELAY, 1500);
229  }
230  else
231  {
232  if (m_creature->HasAura(SPELL_SLOW_SELF))
233  m_creature->RemoveAurasDueToSpell(SPELL_SLOW_SELF);
234 
235  m_creature->FixateTarget(nullptr); // fixate aura runs a tad longer
236  m_bTankPhase = true;
237  DoResetThreat();
238  DoScriptText(EMOTE_PUNCH_GROUND, m_creature);
239  DisableCombatAction(SUPREMUS_ACTION_SWITCH_TARGET);
240  DisableCombatAction(SUPREMUS_ACTION_VOLCANIC_ERUPTION);
241  ResetTimer(SUPREMUS_ACTION_HATEFUL_STRIKE, GetInitialActionTimer(SUPREMUS_ACTION_HATEFUL_STRIKE));
242  SetCombatScriptStatus(true);
243  SetCombatMovement(false, true);
244  SetMeleeEnabled(false);
245  m_creature->SetTarget(nullptr);
246  ResetTimer(SUPREMUS_ACTION_DELAY, 1500);
247  }
248 
249  ResetCombatAction(action, GetSubsequentActionTimer(SupremusActions(action)));
250  return;
251  }
252  case SUPREMUS_ACTION_BERSERK:
253  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
254  DisableCombatAction(action);
255  return;
256  case SUPREMUS_ACTION_MOLTEN_PUNCH:
257  if (DoCastSpellIfCan(nullptr, SPELL_MOLTEN_PUNCH) == CAST_OK)
258  ResetCombatAction(action, GetSubsequentActionTimer(SupremusActions(action)));
259  return;
260  case SUPREMUS_ACTION_VOLCANIC_ERUPTION:
261  {
262  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK))
263  if (DoCastSpellIfCan(target, SPELL_VOLCANIC_ERUPTION) == CAST_OK)
264  ResetCombatAction(action, GetSubsequentActionTimer(SupremusActions(action)));
265  return;
266  }
267  case SUPREMUS_ACTION_SWITCH_TARGET:
268  if (DoCastSpellIfCan(nullptr, SPELL_RANDOM_TARGET) == CAST_OK)
269  {
270  DoScriptText(EMOTE_NEW_TARGET, m_creature);
271  DisableCombatAction(action);
272  }
273  return;
274  case SUPREMUS_ACTION_HATEFUL_STRIKE:
275  if (DoCastSpellIfCan(nullptr, SPELL_HATEFUL_STRIKE_PRIMER) == CAST_OK)
276  ResetCombatAction(action, GetSubsequentActionTimer(SupremusActions(action)));
277  return;
278  }
279  }

300 struct SupremusHatefulStrikePrimer : public SpellScript
301 {
302  void OnEffectExecute(Spell* spell, SpellEffectIndex /*effIdx*/) const override
303  {
304  Unit* target = spell->GetUnitTarget();
305  Unit* caster = spell->GetCaster();
306  auto& targetInfo = spell->GetTargetList();
307  if (!target || targetInfo.rbegin()->targetGUID != target->GetObjectGuid())
308  return;
309 
310  uint32 uiHealth = 0;
311  target = nullptr;
312  Unit* victim = caster->GetVictim();
313 
314  for (auto& targetInfo : targetInfo)
315  {
316  Unit* unit = caster->GetMap()->GetUnit(targetInfo.targetGUID);
317 
318  if (unit && unit != victim && caster->CanReachWithMeleeAttack(unit))
319  {
320  if (unit->GetHealth() > uiHealth)
321  {
322  uiHealth = unit->GetHealth();
323  target = unit;
324  }
325  }
326  }
327  if (!target && victim && caster->CanReachWithMeleeAttack(victim))
328  target = victim;
329 
330  caster->CastSpell(target, SPELL_HATEFUL_STRIKE, TRIGGERED_IGNORE_GCD | TRIGGERED_IGNORE_CURRENT_CASTED_SPELL | TRIGGERED_NORMAL_COMBAT_CAST);
331  }
