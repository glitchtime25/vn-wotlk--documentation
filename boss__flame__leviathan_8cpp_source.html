326  // check for all active towers
327  void FetchTowers()
328  {
329  if (!m_pInstance)
330  return;
331 
332  // the orbital support applies the tower auras
333  Creature* pOrbital = m_pInstance->GetSingleCreatureFromStorage(NPC_ORBITAL_SUPPORT);
334  if (!pOrbital)
335  return;
336 
337  uint8 uiActiveTowers = 0;
338 
339  // check the states twice: at reset and at aggro to make sure that some towers were not destroyed in the meanwhile
340  if (m_pInstance->GetData(TYPE_TOWER_HODIR) == DONE)
341  {
342  pOrbital->CastSpell(pOrbital, SPELL_TOWER_OF_FROST, TRIGGERED_OLD_TRIGGERED);
343  ++uiActiveTowers;
344  m_bUlduarTower[TOWER_ID_HODIR] = true;
345  }
346  else
347  pOrbital->RemoveAurasDueToSpell(SPELL_TOWER_OF_FROST);
348  if (m_pInstance->GetData(TYPE_TOWER_FREYA) == DONE)
349  {
350  pOrbital->CastSpell(pOrbital, SPELL_TOWER_OF_LIFE, TRIGGERED_OLD_TRIGGERED);
351  ++uiActiveTowers;
352  m_bUlduarTower[TOWER_ID_FREYA] = true;
353  }
354  else
355  pOrbital->RemoveAurasDueToSpell(SPELL_TOWER_OF_LIFE);
356  if (m_pInstance->GetData(TYPE_TOWER_MIMIRON) == DONE)
357  {
358  pOrbital->CastSpell(pOrbital, SPELL_TOWER_OF_FLAMES, TRIGGERED_OLD_TRIGGERED);
359  ++uiActiveTowers;
360  m_bUlduarTower[TOWER_ID_MIMIRON] = true;
361  }
362  else
363  pOrbital->RemoveAurasDueToSpell(SPELL_TOWER_OF_FLAMES);

410  void UpdateAI(const uint32 uiDiff) override
411  {
412  // fetch all tower buffs on the first update
413  if (!m_bInitTowers)
414  {
415  FetchTowers();
416  m_bInitTowers = true;
417  }
418 
419  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
420  return;
421 
422  if (m_uiPursueTimer < uiDiff)
423  {
424  if (DoCastSpellIfCan(m_creature, SPELL_PURSUED) == CAST_OK)
425  {
426  // don't yell from the beginning
427  if (!m_uiHardModeTimer)
428  {
429  switch (urand(0, 2))
430  {
431  case 0: DoScriptText(SAY_CHANGE_1, m_creature); break;
432  case 1: DoScriptText(SAY_CHANGE_2, m_creature); break;
433  case 2: DoScriptText(SAY_CHANGE_3, m_creature); break;
434  }
435  }
436  m_uiPursueTimer = 30000;
437  }
438  }
439  else
440  m_uiPursueTimer -= uiDiff;
441 
442  if (m_uiFlameVentsTimer < uiDiff)
443  {
444  if (DoCastSpellIfCan(m_creature, SPELL_FLAME_VENTS) == CAST_OK)
445  m_uiFlameVentsTimer = urand(15000, 25000);
446  }
447  else
448  m_uiFlameVentsTimer -= uiDiff;
449 
450  if (m_uiBatteringRamTimer < uiDiff)
451  {
452  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BATTERING_RAM) == CAST_OK)
453  m_uiBatteringRamTimer = urand(10000, 15000);
454  }
455  else
456  m_uiBatteringRamTimer -= uiDiff;
457 
458  if (m_uiMissileBarrageTimer < uiDiff)
459  {
460  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
461  {
462  if (DoCastSpellIfCan(pTarget, SPELL_MISSILE_BARRAGE_TARGET) == CAST_OK)
463  m_uiMissileBarrageTimer = urand(1000, 2000);
464  }
465  }
466  else
467  m_uiMissileBarrageTimer -= uiDiff;
468 
469  if (m_uiGatheringSpeedTimer < uiDiff)
470  {
471  if (DoCastSpellIfCan(m_creature, SPELL_GATHERING_SPEED) == CAST_OK)
472  m_uiGatheringSpeedTimer = 10000;
473  }
474  else
475  m_uiGatheringSpeedTimer -= uiDiff;
476 
477  // start hard mode
478  if (m_uiHardModeTimer)
479  {
480  if (m_uiHardModeTimer <= uiDiff)
481  {
482  // if all towers are deactivated then skip the rest
483  if (!m_bUlduarTower[TOWER_ID_HODIR] && !m_bUlduarTower[TOWER_ID_FREYA] && !m_bUlduarTower[TOWER_ID_MIMIRON] && !m_bUlduarTower[TOWER_ID_THORIM])
484  {
485  DoScriptText(SAY_TOWER_DOWN, m_creature);
486  m_uiHardModeTimer = 0;
487  }
488  else
489  {
490  // yell hard mode start and start activating each tower one by one
491  switch (m_uiHardModeStep)
492  {
493  case 0:
494  DoScriptText(SAY_HARD_MODE, m_creature);
495  m_uiHardModeTimer = 10000;
496  ++m_uiHardModeStep;
497  break;
498  default:
499  // iterate through all towers to check which is active; skip the ones which are deactivated without triggering the timer
500  for (uint8 i = m_uiHardModeStep - 1; i < KEEPER_ENCOUNTER; ++i)
501  {
502  if (m_bUlduarTower[i])
503  {
504  // yell tower active
505  DoScriptText(aLeviathanTowerYell[i], m_creature);
506  DoScriptText(aLeviathanTowerEmote[i], m_creature);
507 
508  // activate the timer for each tower ability
509  switch (i)
510  {
511  case TOWER_ID_HODIR: DoSpawnHodirFury(); break;
512  case TOWER_ID_FREYA: DoSpawnFreyaWard(); break;
513  case TOWER_ID_MIMIRON: DoSpawnMimironInferno(); break;
514  case TOWER_ID_THORIM: m_uiThorimHammerTimer = 1000; break;
515  }
516 
517  // reset timer and wait for another turn
518  m_uiHardModeTimer = 10000;
519  ++m_uiHardModeStep;
520  break;
521  }
522  ++m_uiHardModeStep;
523 
524  // stop the timer after the final element
525  if (i == KEEPER_ENCOUNTER - 1)
526  m_uiHardModeTimer = 0;
527  }
528  break;
529  }
530  }
531  }
532  else
533  m_uiHardModeTimer -= uiDiff;
534  }
535 
536  // Tower of Storm abilities
537  if (m_uiThorimHammerTimer)
538  {
539  if (m_uiThorimHammerTimer <= uiDiff)
540  {
541  DoSpawnThorimHammer();
542  ++m_uiThorimHammerCount;
543 
544  if (m_uiThorimHammerCount == MAX_THORIM_HAMMER)
545  m_uiThorimHammerTimer = 0;
546  else
547  m_uiThorimHammerTimer = 1000;
548  }
549  else
550  m_uiThorimHammerTimer -= uiDiff;
551  }
552 
553  DoMeleeAttackIfReady();
554  }
555 };
