27 WorldSafeLocsEntry const* GraveyardManager::GetClosestGraveyardHelper(GraveYardMapBounds bounds, float x, float y, float z, uint32 mapId, Team team) const
28 {
29  // Simulate std. algorithm:
30  // found some graveyard associated to (ghost_zone,ghost_map)
31  //
32  // if mapId == graveyard.mapId (ghost in plain zone or city or battleground) and search graveyard at same map
33  // then check faction
34  // if mapId != graveyard.mapId (ghost in instance) and search any graveyard associated
35  // then check faction
36 
37  // at corpse map
38  bool foundNear = false;
39  float distNear = std::numeric_limits<float>::max();
40  WorldSafeLocsEntry const* entryNear = nullptr;
41 
42  // at entrance map for corpse map
43  bool foundEntr = false;
44  float distEntr = std::numeric_limits<float>::max();
45  WorldSafeLocsEntry const* entryEntr = nullptr;
46 
47  // some where other
48  WorldSafeLocsEntry const* entryFar = nullptr;
49 
50  MapEntry const* mapEntry = sMapStore.LookupEntry(mapId);
51 
52  for (GraveYardMap::const_iterator itr = bounds.first; itr != bounds.second; ++itr)
53  {
54  GraveYardData const& data = itr->second;
55 
56  // Checked on load
57  WorldSafeLocsEntry const* entry = sWorldSafeLocsStore.LookupEntry<WorldSafeLocsEntry>(data.safeLocId);
58 
59  // skip enemy faction graveyard
60  // team == TEAM_BOTH_ALLOWED case can be at call from .neargrave
61  // TEAM_INVALID != team for all teams
62  if (data.team != TEAM_BOTH_ALLOWED && data.team != team && team != TEAM_BOTH_ALLOWED)
63  continue;
64 
65  // find now nearest graveyard at other (continent) map
66  if (mapId != entry->map_id)
67  {
68  // if find graveyard at different map from where entrance placed (or no entrance data), use any first
69  if (!mapEntry ||
70  mapEntry->ghost_entrance_map < 0 ||
71  uint32(mapEntry->ghost_entrance_map) != entry->map_id ||
72  (mapEntry->ghost_entrance_x == 0 && mapEntry->ghost_entrance_y == 0))
73  {
74  // not have any coordinates for check distance anyway
75  entryFar = entry;
76  continue;
77  }
78 
79  // at entrance map calculate distance (2D);
80  float dist2 = (entry->x - mapEntry->ghost_entrance_x) * (entry->x - mapEntry->ghost_entrance_x)
81  + (entry->y - mapEntry->ghost_entrance_y) * (entry->y - mapEntry->ghost_entrance_y);
82  if (foundEntr)
83  {
84  if (dist2 < distEntr)
85  {
86  distEntr = dist2;
87  entryEntr = entry;
88  }
89  }
90  else
91  {
92  foundEntr = true;
93  distEntr = dist2;
94  entryEntr = entry;
95  }
96  }
97  // find now nearest graveyard at same map
98  else
99  {
100  float dist2 = (entry->x - x) * (entry->x - x) + (entry->y - y) * (entry->y - y) + (entry->z - z) * (entry->z - z);
101  if (foundNear)
102  {
103  if (dist2 < distNear)
104  {
105  distNear = dist2;
106  entryNear = entry;
107  }
108  }
109  else
110  {
111  foundNear = true;
112  distNear = dist2;
113  entryNear = entry;
114  }
115  }
116  }
117 
118  if (entryNear)
119  return entryNear;
120 
121  if (entryEntr)
122  return entryEntr;
123 
124  return entryFar;
125 }

134 WorldSafeLocsEntry const* GraveyardManager::GetClosestGraveYard(float x, float y, float z, uint32 mapId, Team team) const
135 {
136  // Search for the closest linked graveyard by decreasing priority:
137  // - First try linked to the current area id (if we have one)
138  // - Then try linked to the current zone id (if we have one)
139  // - Then try linked to the current map id
140  const uint32 zoneId = sTerrainMgr.GetZoneId(mapId, x, y, z);
141  const uint32 areaId = sTerrainMgr.GetAreaId(mapId, x, y, z);
142 
143  WorldSafeLocsEntry const* graveyard = nullptr;
144  if (areaId != 0)
145  {
146  auto bounds = m_graveyardMap.equal_range(GraveyardLinkKey(areaId, GRAVEYARD_AREALINK));
147  graveyard = GetClosestGraveyardHelper(bounds, x, y, z, mapId, team);
148  }
149 
150  if (zoneId != 0 && graveyard == nullptr)
151  {
152  auto bounds = m_graveyardMap.equal_range(GraveyardLinkKey(zoneId, GRAVEYARD_AREALINK));
153  graveyard = GetClosestGraveyardHelper(bounds, x, y, z, mapId, team);
154  }
155 
156  if (graveyard == nullptr)
157  {
158  auto bounds = m_graveyardMap.equal_range(GraveyardLinkKey(mapId, GRAVEYARD_MAPLINK));
159  graveyard = GetClosestGraveyardHelper(bounds, x, y, z, mapId, team);
160  }
161 
162  if (graveyard == nullptr)
163  sLog.outErrorDb("Table `game_graveyard_zone` incomplete: Map %u Zone "
164  "%u Area %u Team %u does not have a linked graveyard.",
165  mapId, zoneId, areaId, uint32(team));
166  return graveyard;
167 }
