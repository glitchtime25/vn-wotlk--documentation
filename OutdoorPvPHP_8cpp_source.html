163 // process the capture events
164 bool OutdoorPvPHP::HandleEvent(uint32 eventId, Object* source, Object* /*target*/)
165 {
166  if (!source->IsGameObject())
167  return false;
168 
169  GameObject* go = static_cast<GameObject*>(source);
170  for (uint8 i = 0; i < MAX_HP_TOWERS; ++i)
171  {
172  if (hellfireBanners[i] == go->GetEntry())
173  {
174  for (uint8 j = 0; j < 4; ++j)
175  {
176  if (hellfireTowerEvents[i][j].eventEntry == eventId)
177  {
178  // prevent processing if the owner did not change (happens if progress event is called after contest event)
179  if (hellfireTowerEvents[i][j].team != m_towerOwner[i])
180  {
181  if (hellfireTowerEvents[i][j].defenseMessage)
182  sWorld.SendDefenseMessage(ZONE_ID_HELLFIRE_PENINSULA, hellfireTowerEvents[i][j].defenseMessage);
183 
184  return ProcessCaptureEvent(go, i, hellfireTowerEvents[i][j].team, hellfireTowerEvents[i][j].worldState, hellfireTowerEvents[i][j].towerArtKit, hellfireTowerEvents[i][j].towerAnim);
185  }
186  // no need to iterate other events or towers
187  return false;
188  }
189  }
190  // no need to iterate other towers
191  return false;
192  }
193  }
194 
195  return false;
196 }

198 bool OutdoorPvPHP::ProcessCaptureEvent(GameObject* go, uint32 towerId, Team team, uint32 newWorldState, uint32 towerArtKit, uint32 towerAnim)
199 {
200  // set artkits and process buffs
201  if (team == ALLIANCE)
202  {
203  SetBannerVisual(go, CAPTURE_ARTKIT_ALLIANCE, CAPTURE_ANIM_ALLIANCE);
204 
205  // update counter state
206  ++m_towersAlliance;
207  SendUpdateWorldState(WORLD_STATE_HP_TOWER_COUNT_ALLIANCE, m_towersAlliance);
208 
209  if (m_towersAlliance == MAX_HP_TOWERS)
210  BuffTeam(ALLIANCE, SPELL_HELLFIRE_SUPERIORITY_ALLIANCE);
211  }
212  else if (team == HORDE)
213  {
214  SetBannerVisual(go, CAPTURE_ARTKIT_HORDE, CAPTURE_ANIM_HORDE);
215 
216  // update counter
217  ++m_towersHorde;
218  SendUpdateWorldState(WORLD_STATE_HP_TOWER_COUNT_HORDE, m_towersHorde);
219 
220  if (m_towersHorde == MAX_HP_TOWERS)
221  BuffTeam(HORDE, SPELL_HELLFIRE_SUPERIORITY_HORDE);
222  }
223  else
224  {
225  SetBannerVisual(go, CAPTURE_ARTKIT_NEUTRAL, CAPTURE_ANIM_NEUTRAL);
226 
227  if (m_towerOwner[towerId] == ALLIANCE)
228  {
229  if (m_towersAlliance == MAX_HP_TOWERS)
230  BuffTeam(ALLIANCE, SPELL_HELLFIRE_SUPERIORITY_ALLIANCE, true);
231 
232  // update counter
233  --m_towersAlliance;
234  SendUpdateWorldState(WORLD_STATE_HP_TOWER_COUNT_ALLIANCE, m_towersAlliance);
235  }
236  else
237  {
238  if (m_towersHorde == MAX_HP_TOWERS)
239  BuffTeam(HORDE, SPELL_HELLFIRE_SUPERIORITY_HORDE, true);
240 
241  // update counter
242  --m_towersHorde;
243  SendUpdateWorldState(WORLD_STATE_HP_TOWER_COUNT_HORDE, m_towersHorde);
244  }
245  }
246 
247  SetBannerVisual(go, m_banners[towerId], towerArtKit, towerAnim);
248 
249  // update tower state
250  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_REMOVE);
251  m_towerWorldState[towerId] = newWorldState;
252  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_ADD);
253 
254  // update capture point owner
255  m_towerOwner[towerId] = team;
256 
257  // the are no DB exceptions in this case
258  return true;
259 }
