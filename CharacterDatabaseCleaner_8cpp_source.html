27 void CharacterDatabaseCleaner::CleanDatabase()
28 {
29  // config to disable
30  if (!sWorld.getConfig(CONFIG_BOOL_CLEAN_CHARACTER_DB))
31  return;
32 
33  sLog.outString("Cleaning character database...");
34 
35  // check flags which clean ups are necessary
36  QueryResult* result = CharacterDatabase.PQuery("SELECT cleaning_flags FROM saved_variables");
37  if (!result)
38  return;
39  uint32 flags = (*result)[0].GetUInt32();
40  delete result;
41 
42  // clean up
43  if (flags & CLEANING_FLAG_ACHIEVEMENT_PROGRESS)
44  CleanCharacterAchievementProgress();
45  if (flags & CLEANING_FLAG_SKILLS)
46  CleanCharacterSkills();
47  if (flags & CLEANING_FLAG_SPELLS)
48  CleanCharacterSpell();
49  if (flags & CLEANING_FLAG_TALENTS)
50  CleanCharacterTalent();
51  CharacterDatabase.Execute("UPDATE saved_variables SET cleaning_flags = 0");
52 }

54 void CharacterDatabaseCleaner::CheckUnique(const char* column, const char* table, bool (*check)(uint32))
55 {
56  QueryResult* result = CharacterDatabase.PQuery("SELECT DISTINCT %s FROM %s", column, table);
57  if (!result)
58  {
59  sLog.outString("Table %s is empty.", table);
60  return;
61  }
62 
63  bool found = false;
64  std::ostringstream ss;
65  BarGoLink bar(result->GetRowCount());
66  do
67  {
68  bar.step();
69 
70  Field* fields = result->Fetch();
71 
72  uint32 id = fields[0].GetUInt32();
73 
74  if (!check(id))
75  {
76  if (!found)
77  {
78  ss << "DELETE FROM " << table << " WHERE " << column << " IN (";
79  found = true;
80  }
81  else
82  ss << ",";
83  ss << id;
84  }
85  }
86  while (result->NextRow());
87  delete result;
88 
89  if (found)
90  {
91  ss << ")";
92  CharacterDatabase.Execute(ss.str().c_str());
93  }
94 }
