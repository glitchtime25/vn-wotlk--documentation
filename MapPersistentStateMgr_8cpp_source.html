380 void DungeonResetScheduler::LoadResetTimes()
381 {
382  time_t now = time(nullptr);
383  time_t today = (now / DAY) * DAY;
384  time_t nextWeek = today + (7 * DAY);
385 
386  // NOTE: Use DirectPExecute for tables that will be queried later
387 
388  // get the current reset times for normal instances (these may need to be updated)
389  // these are only kept in memory for InstanceSaves that are loaded later
390  // resettime = 0 in the DB for raid/heroic instances so those are skipped
391  typedef std::pair < uint32 /*PAIR32(map,difficulty)*/, time_t > ResetTimeMapDiffType;
392  typedef std::map<uint32, ResetTimeMapDiffType> InstResetTimeMapDiffType;
393  InstResetTimeMapDiffType instResetTime;
394 
395  QueryResult* result = CharacterDatabase.Query("SELECT id, map, difficulty, resettime FROM instance WHERE resettime > 0");
396  if (result)
397  {
398  do
399  {
400  if (time_t resettime = time_t((*result)[3].GetUInt64()))
401  {
402  uint32 id = (*result)[0].GetUInt32();
403  uint32 mapid = (*result)[1].GetUInt32();
404  uint32 difficulty = (*result)[2].GetUInt32();
405 
406  MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
407 
408  if (!mapEntry || !mapEntry->IsDungeon() || !GetMapDifficultyData(mapid, Difficulty(difficulty)))
409  {
410  sMapPersistentStateMgr.DeleteInstanceFromDB(id);
411  continue;
412  }
413 
414  instResetTime[id] = ResetTimeMapDiffType(MAKE_PAIR32(mapid, difficulty), resettime);
415  }
416  }
417  while (result->NextRow());
418  delete result;
419 
420  // update reset time for normal instances with the max creature respawn time + X hours
421  result = CharacterDatabase.Query("SELECT MAX(respawntime), instance FROM creature_respawn WHERE instance > 0 GROUP BY instance");
422  if (result)
423  {
424  do
425  {
426  Field* fields = result->Fetch();
427 
428  time_t resettime = time_t(fields[0].GetUInt64() + 2 * HOUR);
429  uint32 instance = fields[1].GetUInt32();
430 
431  InstResetTimeMapDiffType::iterator itr = instResetTime.find(instance);
432  if (itr != instResetTime.end() && itr->second.second != resettime)
433  {
434  CharacterDatabase.DirectPExecute("UPDATE instance SET resettime = '" UI64FMTD "' WHERE id = '%u'", uint64(resettime), instance);
435  itr->second.second = resettime;
436  }
437  }
438  while (result->NextRow());
439  delete result;
440  }
441 
442  // schedule the reset times
443  for (InstResetTimeMapDiffType::iterator itr = instResetTime.begin(); itr != instResetTime.end(); ++itr)
444  if (itr->second.second > now)
445  ScheduleReset(true, itr->second.second, DungeonResetEvent(RESET_EVENT_NORMAL_DUNGEON, PAIR32_LOPART(itr->second.first), Difficulty(PAIR32_HIPART(itr->second.first)), itr->first));
446  }
447 
448  // load the global respawn times for raid/heroic instances
449  uint32 diff = sWorld.getConfig(CONFIG_UINT32_INSTANCE_RESET_TIME_HOUR) * HOUR;
450  result = CharacterDatabase.Query("SELECT mapid, difficulty, resettime FROM instance_reset");
451  if (result)
452  {
453  do
454  {
455  Field* fields = result->Fetch();
456 
457  uint32 mapid = fields[0].GetUInt32();
458  Difficulty difficulty = Difficulty(fields[1].GetUInt32());
459  uint64 oldresettime = fields[2].GetUInt64();
460 
461  MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
462 
463  if (!mapEntry || !mapEntry->IsDungeon() || !GetMapDifficultyData(mapid, difficulty))
464  {
465  sLog.outError("MapPersistentStateManager::LoadResetTimes: invalid mapid(%u)/difficulty(%u) pair in instance_reset!", mapid, difficulty);
466  CharacterDatabase.DirectPExecute("DELETE FROM instance_reset WHERE mapid = '%u' AND difficulty = '%u'", mapid, difficulty);
467  continue;
468  }
469 
470  // update the reset time if the hour in the configs changes
471  uint64 newresettime = (oldresettime / DAY) * DAY + diff;
472  if (oldresettime != newresettime)
473  CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty = '%u'", newresettime, mapid, difficulty);
474 
475  SetResetTimeFor(mapid, difficulty, newresettime);
476  }
477  while (result->NextRow());
478  delete result;
479  }
480 
481  // clean expired instances, references to them will be deleted in CleanupInstances
482  // must be done before calculating new reset times
483  m_InstanceSaves._CleanupExpiredInstancesAtTime(now);
484 
485  // calculate new global reset times for expired instances and those that have never been reset yet
486  // add the global reset times to the priority queue
487  for (MapDifficultyMap::const_iterator itr = sMapDifficultyMap.begin(); itr != sMapDifficultyMap.end(); ++itr)
488  {
489  uint32 map_diff_pair = itr->first;
490  uint32 mapid = PAIR32_LOPART(map_diff_pair);
491  Difficulty difficulty = Difficulty(PAIR32_HIPART(map_diff_pair));
492  MapDifficultyEntry const* mapDiff = itr->second;
493 
494  // skip mapDiff without global reset time
495  if (!mapDiff->resetTime)
496  continue;
497 
498  MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
499  if (!mapEntry || !mapEntry->IsDungeon())
500  continue;
501 
502  uint32 period = GetMaxResetTimeFor(mapDiff);
503  time_t t = GetResetTimeFor(mapid, difficulty);
504  if (!t)
505  {
506  // initialize the reset time
507  // generate time by config on first server launch
508  tm localTm = *localtime(&now);
509  localTm.tm_hour = sWorld.getConfig(CONFIG_UINT32_QUEST_DAILY_RESET_HOUR);
510  localTm.tm_min = 0;
511  localTm.tm_sec = 0;
512  if (period > DAY) // resets bigger than 1 day start on config day
513  localTm.tm_mday += ((7 - localTm.tm_wday + sWorld.getConfig(CONFIG_UINT32_ARENA_FIRST_RESET_DAY)) % 7);
514  else // resets day and less start on next day
515  localTm.tm_mday += 1;
516  localTm.tm_isdst = -1;
517  t = mktime(&localTm);
518  CharacterDatabase.DirectPExecute("INSERT INTO instance_reset VALUES ('%u','%u','" UI64FMTD "')", mapid, difficulty, (uint64)t);
519  }
520 
521  if (t < now || t > nextWeek)
522  {
523  // assume that expired instances have already been cleaned
524  // calculate the next reset time
525  t = (t / DAY) * DAY;
526  t += ((today - t) / period + 1) * period + diff;
527  CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty= '%u'", (uint64)t, mapid, difficulty);
528  }
529 
530  SetResetTimeFor(mapid, difficulty, t);
531 
532  // schedule the global reset/warning
533  ResetEventType type = RESET_EVENT_INFORM_1;
534  for (; type < RESET_EVENT_INFORM_LAST; type = ResetEventType(type + 1))
535  if (t > time_t(now + resetEventTypeDelay[type]))
536  break;
537 
538  ScheduleReset(true, t - resetEventTypeDelay[type], DungeonResetEvent(type, mapid, difficulty, 0));
539  }
540 }

542 void DungeonResetScheduler::ScheduleReset(bool add, time_t time, DungeonResetEvent event)
543 {
544  if (add)
545  m_resetTimeQueue.insert(std::pair<time_t, DungeonResetEvent>(time, event));
546  else
547  {
548  // find the event in the queue and remove it
549  ResetTimeQueue::iterator itr;
550  std::pair<ResetTimeQueue::iterator, ResetTimeQueue::iterator> range = m_resetTimeQueue.equal_range(time);
551  for (itr = range.first; itr != range.second; ++itr)
552  {
553  if (itr->second == event)
554  {
555  m_resetTimeQueue.erase(itr);
556  return;
557  }
558  }
559  // in case the reset time changed (should happen very rarely), we search the whole queue
560  if (itr == range.second)
561  {
562  for (itr = m_resetTimeQueue.begin(); itr != m_resetTimeQueue.end(); ++itr)
563  {
564  if (itr->second == event)
565  {
566  m_resetTimeQueue.erase(itr);
567  return;
568  }
569  }
570 
571  if (itr == m_resetTimeQueue.end())
572  sLog.outError("DungeonResetScheduler::ScheduleReset: cannot cancel the reset, the event(%d,%d,%d) was not found!", event.type, event.mapid, event.instanceId);
573  }
574  }

577 void DungeonResetScheduler::Update()
578 {
579  time_t now = time(nullptr);
580  while (!m_resetTimeQueue.empty() && m_resetTimeQueue.begin()->first < now)
581  {
582  DungeonResetEvent& event = m_resetTimeQueue.begin()->second;
583  if (event.type == RESET_EVENT_NORMAL_DUNGEON)
584  {
585  // for individual normal instances, max creature respawn + X hours
586  m_InstanceSaves._ResetInstance(event.mapid, event.instanceId);
587  }
588  else
589  {
590  // global reset/warning for a certain map
591  time_t resetTime = GetResetTimeFor(event.mapid, event.difficulty);
592  uint32 timeLeft = uint32(std::max(int32(resetTime - now), 0));
593  bool warn = event.type != RESET_EVENT_INFORM_LAST && event.type != RESET_EVENT_FORCED_INFORM_LAST;
594  m_InstanceSaves._ResetOrWarnAll(event.mapid, event.difficulty, warn, timeLeft);
595  if (event.type != RESET_EVENT_INFORM_LAST && event.type != RESET_EVENT_FORCED_INFORM_LAST)
596  {
597  // schedule the next warning/reset
598  event.type = ResetEventType(event.type + 1);
599  ScheduleReset(true, resetTime - resetEventTypeDelay[event.type], event);
600  }
601  else
602  {
603  // re-schedule the next/new global reset/warning
604  // calculate the next reset time
605  MapDifficultyEntry const* mapDiff = GetMapDifficultyData(event.mapid, event.difficulty);
606  MANGOS_ASSERT(mapDiff);
607 
608  time_t next_reset = DungeonResetScheduler::CalculateNextResetTime(mapDiff, resetTime);
609 
610  CharacterDatabase.DirectPExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty = '%u'", uint64(next_reset), uint32(event.mapid), uint32(event.difficulty));
611 
612  SetResetTimeFor(event.mapid, event.difficulty, next_reset);
613 
614  ResetEventType type = RESET_EVENT_INFORM_1;
615  for (; type < RESET_EVENT_INFORM_LAST; type = ResetEventType(type + 1))
616  if (next_reset > time_t(now + resetEventTypeDelay[type]))
617  break;
618 
619  // add new scheduler event to the queue
620  event.type = type;
621  ScheduleReset(true, next_reset - resetEventTypeDelay[event.type], event);
622  }
623  }
624  m_resetTimeQueue.erase(m_resetTimeQueue.begin());
625  }

889 {
890  // unbind all players bound to the instance
891  // do not allow UnbindInstance to automatically unload the InstanceSaves
892  lock_instLists = true;
893 
894  bool shouldDelete = true;
895  DungeonPersistentState* state = dynamic_cast<DungeonPersistentState*>(itr->second);
896  if (state)
897  {

899  std::vector<Player*> temp; // list of expired binds that should be unbound
900  for (Player* player : pList)

903  {
904  MANGOS_ASSERT(bind->state == itr->second);
905  if (bind->perm && bind->extendState) // permanent and not already expired

911  }
912  }

914  }
915  for (Player* player : temp)
916  {
917  player->UnbindInstance(itr->second->GetMapId(), itr->second->GetDifficulty(), true);

925  }
926  }
927

929  {
930  delete itr->second;
931  itr = holder.erase(itr);
932  }
933  else
934  ++itr;
935

984 void MapPersistentStateManager::_ResetOrWarnAll(uint32 mapid, Difficulty difficulty, bool warn, uint32 timeLeft)
985 {
986  // global reset for all instances of the given map
987  MapEntry const* mapEntry = sMapStore.LookupEntry(mapid);
988  if (!mapEntry->IsDungeon())
989  return;
990 
991  time_t now = time(nullptr);
992 
993  if (!warn)
994  {
995  MapDifficultyEntry const* mapDiff = GetMapDifficultyData(mapid, difficulty);
996  if (!mapDiff || !mapDiff->resetTime)
997  {
998  sLog.outError("MapPersistentStateManager::ResetOrWarnAll: not valid difficulty or no reset delay for map %d", mapid);
999  return;
1000  }
1001 
1002  // remove all binds for online player
1003  std::list<DungeonPersistentState*> unbindList;
1004 
1005  // note that we must build a list of states to unbind and then unbind them in two steps. this is because the unbinding may
1006  // trigger the modification of the collection, which would invalidate the iterator and cause a crash.
1007  for (auto& itr : m_instanceSaveByInstanceId)
1008  if (itr.second->GetMapId() == mapid && itr.second->GetDifficulty() == difficulty)
1009  unbindList.push_back((DungeonPersistentState*)itr.second);
1010 
1011  for (auto i : unbindList)
1012  i->UnbindThisState();
1013 
1014  // reset maps, teleport player automaticaly to their homebinds and unload maps
1015  MapPersistantStateResetWorker worker;
1016  sMapMgr.DoForAllMapsWithMapId(mapid, worker);
1017 
1018  // delete them from the DB, even if not loaded
1019  // TODO: add difficulty to queries if it becomes a problem
1020  CharacterDatabase.BeginTransaction();
1021  CharacterDatabase.PExecute("DELETE FROM character_instance USING character_instance LEFT JOIN instance ON character_instance.instance = id WHERE (ExtendState = 0 or permanent = 0) and map = '%u'", mapid);
1022  CharacterDatabase.PExecute("DELETE FROM group_instance USING group_instance LEFT JOIN instance ON group_instance.instance = id WHERE map = '%u'", mapid);
1023  CharacterDatabase.PExecute("DELETE FROM instance WHERE map = '%u' and (SELECT guid FROM character_instance WHERE ExtendState != 0 AND instance = id LIMIT 1) IS NULL", mapid);
1024  CharacterDatabase.CommitTransaction();
1025 
1026  // calculate the next reset time
1027  time_t next_reset = DungeonResetScheduler::CalculateNextResetTime(mapDiff, now + timeLeft);
1028  // update it in the DB
1029  CharacterDatabase.PExecute("UPDATE instance_reset SET resettime = '" UI64FMTD "' WHERE mapid = '%u' AND difficulty = '%u'", (uint64)next_reset, mapid, difficulty);
1030  // decrement extension state
1031  CharacterDatabase.PExecute("UPDATE character_instance LEFT JOIN instance ON character_instance.instance = id SET ExtendState=ExtendState-1 WHERE map = '%u'", mapid);
1032  return;
1033  }
1034

1036  MapPersistantStateWarnWorker worker(timeLeft);
1037  sMapMgr.DoForAllMapsWithMapId(mapid, worker);
