72 struct example_creatureAI : public ScriptedAI
73 {
74  // *** HANDLED FUNCTION ***
75  // This is the constructor, called only once when the creature is first created
76  example_creatureAI(Creature* pCreature) : ScriptedAI(pCreature) { Reset(); }
77 
78  // *** CUSTOM VARIABLES ****
79  // These variables are for use only by this individual script.
80  // Nothing else will ever call them but us.
81 
82  uint32 m_uiSayTimer; // Timer for random chat
83  uint32 m_uiRebuffTimer; // Timer for rebuffing
84  uint32 m_uiSpellOneTimer; // Timer for spell 1 when in combat
85  uint32 m_uiSpellTwoTimer; // Timer for spell 1 when in combat
86  uint32 m_uiSpellThreeTimer; // Timer for spell 1 when in combat
87  uint32 m_uiBeserkTimer; // Timer until we go into Beserk (enraged) mode
88  uint32 m_uiPhase; // The current battle phase we are in
89  uint32 m_uiPhaseTimer; // Timer until phase transition
90 
91  // *** HANDLED FUNCTION ***
92  // This is called whenever the core decides we need to evade
93  void Reset() override
94  {
95  m_uiPhase = 1; // Start in phase 1
96  m_uiPhaseTimer = 60000; // 60 seconds
97  m_uiSpellOneTimer = 5000; // 5 seconds
98  m_uiSpellTwoTimer = 37000; // 37 seconds
99  m_uiSpellThreeTimer = 19000; // 19 seconds
100  m_uiBeserkTimer = 120000; // 2 minutes
101  }
102 
103  // *** HANDLED FUNCTION ***
104  // Aggro is called when we enter combat, against an enemy, and haven't been in combat before
105  void Aggro(Unit* pWho) override
106  {
107  // Say some stuff
108  DoScriptText(SAY_AGGRO, m_creature, pWho);
109  }
110 
111  // *** HANDLED FUNCTION ***
112  // Our Recive emote function
113  void ReceiveEmote(Player* /*pPlayer*/, uint32 uiTextEmote) override
114  {
115  m_creature->HandleEmote(uiTextEmote);
116 
117  switch (uiTextEmote)
118  {
119  case TEXTEMOTE_DANCE:
120  DoScriptText(SAY_DANCE, m_creature);
121  break;
122  case TEXTEMOTE_SALUTE:
123  DoScriptText(SAY_SALUTE, m_creature);
124  break;
125  }
126  }
127 
128  // *** HANDLED FUNCTION ***
129  // Update AI is called Every single map update (roughly once every 100ms if a player is within the grid)
130  void UpdateAI(const uint32 uiDiff) override
131  {
132  // Out of combat timers
133  if (!m_creature->GetVictim())
134  {
135  // Random Say timer
136  if (m_uiSayTimer < uiDiff)
137  {
138  // Random switch between 5 outcomes
139  switch (urand(0, 4))
140  {
141  case 0: DoScriptText(SAY_RANDOM_0, m_creature); break;
142  case 1: DoScriptText(SAY_RANDOM_1, m_creature); break;
143  case 2: DoScriptText(SAY_RANDOM_2, m_creature); break;
144  case 3: DoScriptText(SAY_RANDOM_3, m_creature); break;
145  case 4: DoScriptText(SAY_RANDOM_4, m_creature); break;
146  }
147 
148  m_uiSayTimer = 45 * IN_MILLISECONDS; // Say something agian in 45 seconds
149  }
150  else
151  m_uiSayTimer -= uiDiff;
152 
153  // Rebuff timer
154  if (m_uiRebuffTimer < uiDiff)
155  {
156  DoCastSpellIfCan(m_creature, SPELL_BUFF);
157  // Rebuff agian in 15 minutes
158  m_uiRebuffTimer = 15 * MINUTE * IN_MILLISECONDS;
159  }
160  else
161  m_uiRebuffTimer -= uiDiff;
162  }
163 
164  // Return since we have no target
165  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
166  return;
167 
168  // Abilities of all phases
169  // Spell One timer
170  if (m_uiSpellOneTimer < uiDiff)
171  {
172  // Cast spell one on our current target.
173  if (rand() % 50 > 10)
174  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ONE_ALT);
175  else if (m_creature->IsWithinDist(m_creature->GetVictim(), 25.0f))
176  DoCastSpellIfCan(m_creature->GetVictim(), SPELL_ONE);
177 
178  m_uiSpellOneTimer = 5000;
179  }
180  else
181  m_uiSpellOneTimer -= uiDiff;
182 
183  // Spell Two timer
184  if (m_uiSpellTwoTimer < uiDiff)
185  {
186  // Cast spell two on self (AoE spell with only self-target) if we can
187  if (DoCastSpellIfCan(m_creature, SPELL_TWO) == CAST_OK)
188  m_uiSpellTwoTimer = 37 * IN_MILLISECONDS; // Only Update Timer, if we could start casting
189  }
190  else
191  m_uiSpellTwoTimer -= uiDiff;
192 
193  // End of abliities of all phases
194 
195  // Phase 1 abilities
196  if (m_uiPhase == 1)
197  {
198  // Phase timer
199  if (m_uiPhaseTimer < uiDiff)
200  {
201  // Only switch phase and display phase-switich text, if out cast was started sucessfull
202  if (DoCastSpellIfCan(m_creature, SPELL_ENRAGE) == CAST_OK)
203  {
204  // Go to next phase
205  ++m_uiPhase;
206  DoScriptText(SAY_PHASE, m_creature);
207  }
208  }
209  else
210  m_uiPhaseTimer -= uiDiff;
211  }
212  // Phase 2 abilities
213  else if (m_uiPhase > 1)
214  {
215  // Spell Three timer
216  if (m_uiSpellThreeTimer < uiDiff)
217  {
218  // Cast spell three on self (AoE spell with only self-target)
219  if (DoCastSpellIfCan(m_creature, SPELL_THREE) == CAST_OK)
220  m_uiSpellThreeTimer = 19000;
221  }
222  else
223  m_uiSpellThreeTimer -= uiDiff;
224 
225  // Beserk timer
226  if (m_uiBeserkTimer < uiDiff)
227  {
228  // Cast uber death spell if possible
229  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_BESERK) == CAST_OK)
230  {
231  // Say our line if we cast
232  DoScriptText(SAY_BESERK, m_creature, m_creature->GetVictim());
233 
234  // Cast our beserk spell agian in 12 seconds (if we didn't kill everyone)
235  m_uiBeserkTimer = 12000;
236  }
237  }
238  else
239  m_uiBeserkTimer -= uiDiff;
240  }
241 
242  // Normal behaviour: if possible mobs do attack with melee
243  DoMeleeAttackIfReady();
244  }
245 };

264 // This function is called when the player clicks an option on the gossip menu
265 // In this case here the faction change could be handled by world-DB gossip, hence it should be handled there!
266 bool GossipSelect_example_creature(Player* pPlayer, Creature* pCreature, uint32 /*uiSender*/, uint32 uiAction)
267 {
268  if (uiAction == GOSSIP_ACTION_INFO_DEF + 1)
269  {
270  pPlayer->CLOSE_GOSSIP_MENU();
271  // Set our faction to hostile towards all
272  pCreature->SetFactionTemporary(FACTION_WORGEN, TEMPFACTION_RESTORE_RESPAWN);
273  pCreature->AI()->AttackStart(pPlayer);
274  }
275 
276  return true;
