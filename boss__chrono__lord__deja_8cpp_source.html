98  void UpdateAI(const uint32 uiDiff) override
99  {
100  // Return since we have no target
101  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
102  return;
103 
104  // Arcane Blast
105  if (m_uiArcaneBlastTimer < uiDiff)
106  {
107  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_ARCANE_BLAST : SPELL_ARCANE_BLAST_H) == CAST_OK)
108  m_uiArcaneBlastTimer = urand(15000, 25000);
109  }
110  else
111  m_uiArcaneBlastTimer -= uiDiff;
112 
113  // Arcane Discharge
114  if (m_uiArcaneDischargeTimer < uiDiff)
115  {
116  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_ARCANE_DISCHARGE : SPELL_ARCANE_DISCHARGE_H) == CAST_OK)
117  m_uiArcaneDischargeTimer = urand(20000, 30000);
118  }
119  else
120  m_uiArcaneDischargeTimer -= uiDiff;
121 
122  // Time Lapse
123  if (m_uiTimeLapseTimer < uiDiff)
124  {
125  if (DoCastSpellIfCan(m_creature, SPELL_TIME_LAPSE) == CAST_OK)
126  m_uiTimeLapseTimer = urand(15000, 25000);
127  }
128  else
129  m_uiTimeLapseTimer -= uiDiff;
130 
131  // Attraction
132  if (!m_bIsRegularMode)
133  {
134  if (m_uiAttractionTimer < uiDiff)
135  {
136  if (DoCastSpellIfCan(m_creature, SPELL_ATTRACTION) == CAST_OK)
137  m_uiAttractionTimer = urand(25000, 35000);
138  }
139  else
140  m_uiAttractionTimer -= uiDiff;
141  }
142 
143  DoMeleeAttackIfReady();
144  }
