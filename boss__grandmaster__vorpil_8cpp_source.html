171  // Wrapper to teleport all players to the platform - Workaround for missing spell
172  void DoTeleportToPlatform()
173  {
174  m_creature->NearTeleportTo(aVorpilTeleportLoc[0], aVorpilTeleportLoc[1], aVorpilTeleportLoc[2], 0.0f);
175 
176  float fX, fY, fZ;
177 
178  GuidVector vGuids;
179  m_creature->FillGuidsListFromThreatList(vGuids);
180  for (GuidVector::const_iterator itr = vGuids.begin(); itr != vGuids.end(); ++itr)
181  {
182  Unit* pTarget = m_creature->GetMap()->GetUnit(*itr);
183 
184  if (pTarget && pTarget->GetTypeId() == TYPEID_PLAYER)
185  {
186  pTarget->GetRandomPoint(aVorpilTeleportLoc[0], aVorpilTeleportLoc[1], aVorpilTeleportLoc[2], 4.0f, fX, fY, fZ);
187  DoTeleportPlayer(pTarget, fX, fY, fZ, m_creature->GetAngle(fX, fY));
188  }

192  void UpdateAI(const uint32 uiDiff) override
193  {
194  // Return since we have no target
195  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
196  return;
197 
198  if (m_uiRainOfFireTimer)
199  {
200  if (m_uiRainOfFireTimer <= uiDiff)
201  {
202  SetCombatMovement(false, true);
203  DoTeleportToPlatform();
204 
205  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_RAIN_OF_FIRE : SPELL_RAIN_OF_FIRE_H, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
206  m_uiRainOfFireTimer = 0;
207 
208  SetCombatMovement(true);
209 
210  return; // Nothing more todo after the players had been teleported
211  }
212  m_uiRainOfFireTimer -= uiDiff;
213  }
214 
215  if (m_uiShadowBoltVolleyTimer < uiDiff)
216  {
217  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_BOLT_VOLLEY) == CAST_OK)
218  m_uiShadowBoltVolleyTimer = urand(10000, 26000);
219  }
220  else
221  m_uiShadowBoltVolleyTimer -= uiDiff;
222 
223  if (m_uiDrawShadowsTimer < uiDiff)
224  {
225  if (DoCastSpellIfCan(m_creature, SPELL_DRAW_SHADOWS) == CAST_OK)
226  {
227  m_uiDrawShadowsTimer = urand(36000, 44000);
228  m_uiRainOfFireTimer = 1000;
229  }
230  }
231  else
232  m_uiDrawShadowsTimer -= uiDiff;
233 
234  if (m_uiVoidTravelerTimer < uiDiff)
235  {
236  if (DoCastSpellIfCan(m_creature, aTravelerSummonSpells[urand(0, 4)]) == CAST_OK)
237  {
238  DoScriptText(SAY_HELP, m_creature);
239  m_uiVoidTravelerTimer = urand(10000, 15000);
240  }
241  }
242  else
243  m_uiVoidTravelerTimer -= uiDiff;
244 
245  if (!m_bIsRegularMode)
246  {
247  if (m_uiBanishTimer < uiDiff)
248  {
249  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
250  {
251  if (DoCastSpellIfCan(pTarget, SPELL_BANISH_H) == CAST_OK)
252  m_uiBanishTimer = urand(17000, 23000);
253  }
254  }
255  else
256  m_uiBanishTimer -= uiDiff;
257  }
258 
259  DoMeleeAttackIfReady();
260  }
