63 std::string AntispamMgr::NormalizeStringInternal(const std::string &string, uint32 mask) const
64 {
65  auto newMsg = string;
66 
67  if (mask & NF_CUT_COLOR)
68  {
69  static const std::regex regex1("(\\|c\\w{8})");
70  static const std::regex regex2("(\\|H[\\w|\\W]{1,}\\|h)");
71  newMsg = std::regex_replace(newMsg, regex1, "");
72  ReplaceAll(newMsg, "|h|r", "");
73  newMsg = std::regex_replace(newMsg, regex2, "");
74  }
75 
76  if (mask & NF_REPLACE_WORDS)
77  {
78  for (auto const& e : _asciiReplace)
79  ReplaceAll(newMsg, e.first, e.second);
80  }
81 
82  if (mask & NF_CUT_CTRL)
83  {
84  static const std::regex regex4("([[:cntrl:]]+)");
85  newMsg = std::regex_replace(newMsg, regex4, "");
86  }
87 
88  if (mask & NF_CUT_PUNCT)
89  {
90  static const std::regex regex5("([[:punct:]]+)");
91  newMsg = std::regex_replace(newMsg, regex5, "");
92  }
93 
94  if (mask & NF_CUT_SPACE)
95  {
96  static const std::regex regex6("(\\s+|_)");
97  newMsg = std::regex_replace(newMsg, regex6, "");
98  }
99 
100  if (mask & NF_CUT_NUMBERS)
101  {
102  static std::regex regex3("(\\d+)");
103  newMsg = std::regex_replace(newMsg, regex3, "");
104  }
105 
106  if (mask & NF_REPLACE_UNICODE)
107  {
108  std::wstring w_tempMsg, w_tempMsg2;
109  Utf8toWStr(newMsg, w_tempMsg);
110  wstrToUpper(w_tempMsg);
111 
112  if (!isBasicLatinString(w_tempMsg, true))
113  {
114  for (auto const& s : _unicodeReplace)
115  ReplaceAllW(w_tempMsg, s.first, s.second);
116 
117  if (mask & NF_REMOVE_NON_LATIN)
118  {
119  for (size_t i = 0; i < w_tempMsg.size(); ++i)
120  if (isBasicLatinCharacter(w_tempMsg[i]) || isNumeric(w_tempMsg[i]))
121  w_tempMsg2.push_back(w_tempMsg[i]);
122  }
123  else
124  w_tempMsg2 = w_tempMsg;
125  }
126  else
127  w_tempMsg2 = w_tempMsg;
128 
129  newMsg = std::string(w_tempMsg2.begin(), w_tempMsg2.end());
130  }
131  else
132  std::transform(newMsg.begin(), newMsg.end(), newMsg.begin(), ::toupper);
133 
134  if (mask & NF_REMOVE_REPEATS)
135  newMsg.erase(std::unique(newMsg.begin(), newMsg.end()), newMsg.end());
136 
137  return newMsg;
138 }

148 void AntispamMgr::WorkerLoop()
149 {
150  while (!_shutdownRequested)
151  {
152  auto const duration = sAnticheatConfig.GetAntispamAnalysisTimer() * IN_MILLISECONDS;
153 
154  auto const start = std::chrono::high_resolution_clock::now();
155  auto const startMS = WorldTimer::getMSTime();
156  auto const expireMS = sAnticheatConfig.GetAntispamCacheExpiration() * IN_MILLISECONDS;
157 
158  if (sAnticheatConfig.EnableAntispam())
159  {
160  std::unordered_set<std::shared_ptr<Antispam> > workQueue;
161 
162  // lock the mutex only long enough to move the work queue to a local container and expire old blacklist history
163  {
164  std::lock_guard<std::mutex> guard(_mutex);
165  workQueue = std::move(_workQueue);
166 
167  for (auto i = _temporaryCache.begin(); i != _temporaryCache.end(); )
168  {
169  if (i->second.first + expireMS <= startMS)
170  i = _temporaryCache.erase(i);
171  else
172  ++i;
173  }
174  }
175 
176  for (auto const &s : workQueue)
177  s->Analyze();
178  }
179  else
180  {
181  std::lock_guard<std::mutex> guard(_mutex);
182  _workQueue.clear();
183  _temporaryCache.clear();
184  }
185 
186  auto stop = std::chrono::high_resolution_clock::now();
187  auto runTime = std::chrono::duration_cast<std::chrono::milliseconds>(stop - start).count();
188 
189  // if we completed sooner than expected, sleep for the remaining time
190  while (duration > runTime && !_shutdownRequested)
191  {
192  stop = std::chrono::high_resolution_clock::now();
193  runTime = std::chrono::duration_cast<std::chrono::milliseconds>(stop - start).count();
194  std::this_thread::sleep_for(std::chrono::milliseconds(500));
195  }
196  }
197 }

199 void AntispamMgr::LoadFromDB()
200 {
201  std::lock_guard<std::mutex> guard(_mutex);
202 
203  auto const normMask = sAnticheatConfig.GetSpamNormalizationMask();
204 
205  _blacklist.clear();
206 
207  std::unique_ptr<QueryResult> result(LoginDatabase.Query("SELECT `string` FROM antispam_blacklist"));
208 
209  if (result)
210  do
211  {
212  auto fields = result->Fetch();
213  auto entry = fields[0].GetCppString();
214 
215  if (entry.empty())
216  {
217  sLog.outError("Refusing to load empty antispam blacklist entry");
218  continue;
219  }
220 
221  std::transform(entry.begin(), entry.end(), entry.begin(), ::toupper);
222 
223  for (auto const &b : _blacklist)
224  if (b.first == entry)
225  {
226  sLog.outError("Duplicate entry \"%s\" in antispam blacklist", entry.c_str());
227  continue;
228  }
229 
230  auto const normEntry = NormalizeStringInternal(entry, normMask);
231 
232  _blacklist.emplace_back(entry, normEntry);
233  } while (result->NextRow());
234 
235  sLog.outString(">> %lu blacklist entries loaded and normalized", uint64(_blacklist.size()));
236 
237  result.reset(LoginDatabase.Query("SELECT `from`, `to` FROM antispam_replacement"));
238 
239  _asciiReplace.clear();
240 
241  if (result)
242  do
243  {
244  auto fields = result->Fetch();
245  _asciiReplace.emplace_back(fields[0].GetCppString(), fields[1].GetCppString());
246  } while (result->NextRow());
247 
248  sLog.outString(">> %lu ASCII string replacements loaded", uint64(_asciiReplace.size()));
249 
250  result.reset(LoginDatabase.Query("SELECT `from`, `to` FROM antispam_unicode_replacement"));
251 
252  _unicodeReplace.clear();
253 
254  if (result)
255  do
256  {
257  std::wostringstream wss;
258 
259  auto fields = result->Fetch();
260 
261  wss.str(std::wstring());
262  wss << wchar_t(fields[0].GetUInt32());
263  std::wstring key = wss.str();
264 
265  wss.str(std::wstring());
266  wss << wchar_t(fields[1].GetUInt32());
267  std::wstring value = wss.str();
268 
269  _unicodeReplace.emplace_back(key, value);
270  } while (result->NextRow());
271 
272  sLog.outString(">> %lu unicode character replacements loaded", uint64(_unicodeReplace.size()));
273 }

275 void AntispamMgr::BlacklistAdd(const std::string &string_)
276 {
277  std::lock_guard<std::mutex> guard(_mutex);
278 
279  // cannot be empty!
280  if (string_.empty())
281  return;
282 
283  // convert to upper case, since all comparisons are done upper case
284  std::string entry;
285  std::transform(string_.begin(), string_.end(), std::back_inserter(entry), ::toupper);
286 
287  auto const normEntry = NormalizeStringInternal(entry, sAnticheatConfig.GetSpamNormalizationMask());
288 
289  // if already in the blacklist, do not add again
290  for (auto const &b : _blacklist)
291  if (b.first == entry)
292  return;
293 
294  static SqlStatementID insertBlacklist;
295 
296  LoginDatabase.BeginTransaction();
297 
298  auto ins = LoginDatabase.CreateStatement(insertBlacklist, "INSERT INTO antispam_blacklist (string) VALUES(?)");
299 
300  ins.addString(entry);
301  ins.Execute();
302 
303  LoginDatabase.CommitTransaction();
304 
305  _blacklist.emplace_back(entry, normEntry);
306 }

308 uint32 AntispamMgr::CheckBlacklist(const std::string &string, std::string &log) const
309 {
310  std::lock_guard<std::mutex> guard(_mutex);
311 
312  auto const normalizationMask = sAnticheatConfig.GetSpamNormalizationMask();
313  auto const msg = NormalizeStringInternal(string, normalizationMask);
314 
315  std::stringstream logstr;
316  logstr << "Original message:\n" << string << "\nNormalized message:\n" << msg << "\nBlacklist violations:";
317 
318  uint32 result = 0;
319 
320  // for each entry, search the given string for multiple occurrences of both the original and normalized form of the entry
321  for (auto const& entry : _blacklist)
322  {
323  // search the original string for the original blacklist entry
324  if (!entry.first.empty())
325  {
326  for (auto pos = string.find(entry.first); pos != std::string::npos; pos = string.find(entry.first, pos + entry.first.length()))
327  {
328  logstr << "\nOriginal: \"" << entry.first << "\"";
329  ++result;
330  }
331  }
332 
333  // search for the normalized blacklist entry
334  if (!entry.second.empty())
335  {
336  for (auto pos = msg.find(entry.second); pos != std::string::npos; pos = msg.find(entry.second, pos + entry.second.length()))
337  {
338  logstr << "\nNormalized: \"" << entry.second << "\"";
339  ++result;
340  }
341  }
342  }
343 
344  logstr << "\n";
345 
346  // if there were results found, save the log
347  if (!!result)
348  log = logstr.str();
349 
350  return result;
351 }
