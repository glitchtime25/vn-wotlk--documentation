33 bool GOUse_go_main_chambers_access_panel(Player* /*pPlayer*/, GameObject* pGo)
34 {
35  ScriptedInstance* pInstance = (ScriptedInstance*)pGo->GetInstanceData();
36 
37  if (!pInstance)
38  return true;
39 
40  if (pGo->GetEntry() == GO_ACCESS_PANEL_HYDRO && pInstance->GetData(TYPE_HYDROMANCER_THESPIA) != SPECIAL)
41  pInstance->SetData(TYPE_HYDROMANCER_THESPIA, SPECIAL);
42  else if (pGo->GetEntry() == GO_ACCESS_PANEL_MEK && pInstance->GetData(TYPE_MEKGINEER_STEAMRIGGER) != SPECIAL)
43  pInstance->SetData(TYPE_MEKGINEER_STEAMRIGGER, SPECIAL);
44 
45  return false;
46 }

72 void instance_steam_vault::OnObjectCreate(GameObject* pGo)
73 {
74  switch (pGo->GetEntry())
75  {
76  case GO_MAIN_CHAMBERS_DOOR:
77  if (m_auiEncounter[TYPE_HYDROMANCER_THESPIA] == SPECIAL && m_auiEncounter[TYPE_MEKGINEER_STEAMRIGGER] == SPECIAL)
78  pGo->SetGoState(GO_STATE_ACTIVE);
79  break;
80  case GO_ACCESS_PANEL_HYDRO:
81  if (m_auiEncounter[TYPE_HYDROMANCER_THESPIA] == DONE)
82  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
83  break;
84  case GO_ACCESS_PANEL_MEK:
85  if (m_auiEncounter[TYPE_MEKGINEER_STEAMRIGGER] == DONE)
86  pGo->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
87  break;
88  default:
89  return;
90  }
91  m_goEntryGuidStore[pGo->GetEntry()] = pGo->GetObjectGuid();
92 }

104 void instance_steam_vault::SetData(uint32 uiType, uint32 uiData)
105 {
106  switch (uiType)
107  {
108  case TYPE_HYDROMANCER_THESPIA:
109  if (uiData == DONE)
110  DoToggleGameObjectFlags(GO_ACCESS_PANEL_HYDRO, GO_FLAG_NO_INTERACT, false);
111  if (uiData == SPECIAL)
112  {
113  if (GetData(TYPE_MEKGINEER_STEAMRIGGER) == SPECIAL)
114  DoUseDoorOrButton(GO_MAIN_CHAMBERS_DOOR);
115  }
116  m_auiEncounter[uiType] = uiData;
117  break;
118  case TYPE_MEKGINEER_STEAMRIGGER:
119  if (uiData == DONE)
120  DoToggleGameObjectFlags(GO_ACCESS_PANEL_MEK, GO_FLAG_NO_INTERACT, false);
121  if (uiData == SPECIAL)
122  {
123  if (GetData(TYPE_HYDROMANCER_THESPIA) == SPECIAL)
124  DoUseDoorOrButton(GO_MAIN_CHAMBERS_DOOR);
125  }
126  m_auiEncounter[uiType] = uiData;
127  break;
128  case TYPE_WARLORD_KALITHRESH:
129  DoUseDoorOrButton(GO_MAIN_CHAMBERS_DOOR);
130  if (uiData == FAIL)
131  {
132  // Reset Distiller flags - respawn is handled by DB
133  for (GuidList::const_iterator itr = m_lNagaDistillerGuidList.begin(); itr != m_lNagaDistillerGuidList.end(); ++itr)
134  {
135  if (Creature* pDistiller = instance->GetCreature(*itr))
136  {
137  if (!pDistiller->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE))
138  pDistiller->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
139  }
140  }
141  }
142  m_auiEncounter[uiType] = uiData;
143  break;
144  }
145 
146  if (uiData == DONE || uiData == SPECIAL)
147  {
148  OUT_SAVE_INST_DATA;
149 
150  std::ostringstream saveStream;
151  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2];
152 
153  m_strInstData = saveStream.str();
154 
155  SaveToDB();
156  OUT_SAVE_INST_DATA_COMPLETE;
157  }
158 }
