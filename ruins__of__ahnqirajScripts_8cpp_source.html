119  void UpdateAI(const uint32 uiDiff) override
120  {
121  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
122  return;
123 
124  // Meteor or Plague
125  if (m_uiSpell1Timer < uiDiff)
126  {
127  DoCastSpellIfCan(m_creature->GetVictim(), m_uiSpell1);
128  m_uiSpell1Timer = 15000;
129  }
130  else
131  m_uiSpell1Timer -= uiDiff;
132 
133  // Shadow Storm or Thunder Clap
134  if (m_uiSpell2Timer < uiDiff)
135  {
136  DoCastSpellIfCan(m_creature->GetVictim(), m_uiSpell2);
137  m_uiSpell2Timer = 15000;
138  }
139  else
140  m_uiSpell2Timer -= uiDiff;
141 
142  // summon Anubisath Swarmguard or Anubisath Warrior
143  if (m_uiSpell5Timer < uiDiff)
144  {
145  // change for summon spell
146  if (m_uiSummonCount < 4)
147  DoCastSpellIfCan(m_creature->GetVictim(), m_uiSpell5);
148 
149  m_uiSpell5Timer = 15000;
150  }
151  else
152  m_uiSpell5Timer -= uiDiff;
153 
154  DoMeleeAttackIfReady();
155  }
