33  Creature Evade Timer & Leash range
34 
35  1) Leash timer 15 sec if creature can reach you
36  - Timer will be reset upon if player deals damage or if creature does damage to player
37  - Only direct damage will reset damage
38 
39  2) There is no timer for creature that cannot reach you

47 void CombatManager::Update(const uint32 diff)
48 {
49  if (m_owner->IsInCombat())
50  {
51  if (m_evadeTimer)
52  {
53  if (m_evadeTimer <= diff)
54  {
55  m_evadeTimer = 0;
56  m_owner->EvadeTimerExpired();
57  }
58  else
59  m_evadeTimer -= diff;
60  }
61 
62  if (!m_leashingDisabled)
63  {
64  if (m_combatTimer)
65  {
66  if (m_owner->IsPlayerControlled() || !m_owner->IsCrowdControlled())
67  {
68  if (m_combatTimer <= diff)
69  m_combatTimer = 0;
70  else
71  m_combatTimer -= diff;
72  }
73  }
74  }
75 
76  m_combatTick += diff;
77  if (m_combatTick >= COMBAT_MANAGER_TICK)
78  {
79  m_combatTick -= COMBAT_MANAGER_TICK;
80  if (!m_leashingDisabled)
81  {
82  // disabled in instances except for players in BGs
83  if (!m_owner->GetMap()->IsDungeon() || m_owner->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
84  {
85  if (!m_owner->GetMap()->IsDungeon() && m_owner->IsImmobilizedState())
86  m_owner->getThreatManager().DeleteOutOfRangeReferences();
87 
88  if (!m_combatTimer)
89  {
90  bool check = !m_owner->HasMaster();
91  if (!check)
92  {
93  Unit* master = m_owner->GetMaster();
94  if (!master || !master->IsAlive()) // if charmer alive, he will evade this charm
95  check = true;
96  }
97  if (check)
98  {
99  if (m_owner->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
100  {
101  if (m_owner->getHostileRefManager().getSize() == 0)
102  m_owner->HandleExitCombat(m_owner->IsPlayer());
103  }
104  // if timer ran out and we are far away from last refresh pos, evade
105  else if (m_owner->GetVictim() && m_owner->GetVictim()->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
106  {
107  // immobilized passive mobs ignore last refresh pos
108  if (m_owner->IsImmobilizedState() && m_owner->AI()->GetReactState() == REACT_PASSIVE)
109  m_owner->HandleExitCombat(false);
110  else if (m_owner->GetVictim()->GetDistance2d(m_lastRefreshPos.GetPositionX(), m_lastRefreshPos.GetPositionY()) > sWorld.getConfig(CONFIG_FLOAT_LEASH_RADIUS))
111  m_owner->HandleExitCombat(false);
112  }
113  }
114  }
115  }
116  }
117 
118  if (m_owner->IsCreature() && !m_owner->HasCharmer()) // charmer should have leashing check or leash set
119  {
120  Creature* creatureOwner = static_cast<Creature*>(m_owner);
121  // If creature is within 30yd from combat start do not exit combat
122  if (!creatureOwner->HasFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_PLAYER_CONTROLLED))
123  {
124  if (m_leashingCheck)
125  {
126  float x, y, z;
127  creatureOwner->GetPosition(x, y, z);
128  if (m_leashingCheck(creatureOwner, x, y, z))
129  creatureOwner->HandleExitCombat();
130  }
131  else if (creatureOwner->GetCreatureInfo()->Leash) // If creature has set maximum leashing distance
132  {
133  Position pos;
134  creatureOwner->GetCombatStartPosition(pos);
135  if (creatureOwner->GetDistance2d(pos.GetPositionX(), pos.GetPositionY()) > creatureOwner->GetCreatureInfo()->Leash)
136  creatureOwner->HandleExitCombat();
137  }
138  }
139  }
140  }
141  }
142 }

161 void CombatManager::SetEvadeState(EvadeState state)
162 {
163  if (m_evadeState == state)
164  return;
165 
166  if (state == EVADE_NONE)
167  m_evadeTimer = 0;
168  if (state == EVADE_NONE && m_evadeState == EVADE_HOME)
169  m_owner->AI()->SetAIOrder(ORDER_NONE);
170  else if (state == EVADE_HOME)
171  m_owner->AI()->SetAIOrder(ORDER_EVADE);
172  m_evadeState = state;
173 
174  // Do not propagate during charm
175  if (!m_owner->HasCharmer())
176  m_owner->CallForAllControlledUnits(SetEvadeHelper(state), CONTROLLED_PET | CONTROLLED_TOTEMS | CONTROLLED_GUARDIANS | CONTROLLED_CHARM);
177 }
