98  inline void C_Evaluate(const Vector3* vertice, float t, const Matrix4& matr, Vector3& result)
99  {
100  Vector4 tvec(t * t * t, t * t, t, 1.f);
101  Vector4 weights(tvec * matr);
102 
103  result = vertice[0] * weights[0] + vertice[1] * weights[1]
104  + vertice[2] * weights[2] + vertice[3] * weights[3];
105  }

220  void SplineBase::InitLinear(const Vector3* controls, index_type count, bool isCyclic, index_type cyclic_point)
221  {
222  MANGOS_ASSERT(count >= 2);
223  const int real_size = count + 1;
224 
225  points.resize(real_size);
226 
227  memcpy(&points[0], controls, sizeof(Vector3) * count);
228 
229  // first and last two indexes are space for special 'virtual points'
230  // these points are required for proper C_Evaluate and C_Evaluate_Derivative methtod work
231  if (isCyclic)
232  points[count] = controls[cyclic_point];
233  else
234  points[count] = controls[count - 1];
235 
236  index_lo = 0;
237  index_hi = isCyclic ? count : (count - 1);
238  }

240  void SplineBase::InitCatmullRom(const Vector3* controls, index_type count, bool isCyclic, index_type cyclic_point)
241  {
242  const int real_size = count + (isCyclic ? (1 + 2) : (1 + 1));
243 
244  points.resize(real_size);
245 
246  int lo_index = 1;
247  int high_index = lo_index + count - 1;
248 
249  memcpy(&points[lo_index], controls, sizeof(Vector3) * count);
250 
251  // first and last two indexes are space for special 'virtual points'
252  // these points are required for proper C_Evaluate and C_Evaluate_Derivative methtod work
253  if (isCyclic)
254  {
255  if (cyclic_point == 0)
256  points[0] = controls[count - 1];
257  else
258  points[0] = controls[0].lerp(controls[1], -1);
259 
260  points[high_index + 1] = controls[cyclic_point];
261  points[high_index + 2] = controls[cyclic_point + 1];
262  }
263  else
264  {
265  points[0] = controls[0].lerp(controls[1], -1);
266  points[high_index + 1] = controls[count - 1];
267  }
268 
269  index_lo = lo_index;
270  index_hi = high_index + (isCyclic ? 1 : 0);
271  }
