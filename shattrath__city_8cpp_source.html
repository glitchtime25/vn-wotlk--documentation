112  uint32 NextStep(uint32 uiStep)
113  {
114  Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid);
115 
116  if (!pPlayer)
117  {
118  if (Creature* pCreepjack = m_creature->GetMap()->GetCreature(m_creepjackGuid))
119  {
120  if (!pCreepjack->GetCombatManager().IsInEvadeMode() && pCreepjack->IsAlive())
121  pCreepjack->AI()->EnterEvadeMode();
122 
123  pCreepjack->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
124  }
125 
126  if (Creature* pMalone = m_creature->GetMap()->GetCreature(m_maloneGuid))
127  {
128  if (!pMalone->GetCombatManager().IsInEvadeMode() && pMalone->IsAlive())
129  pMalone->AI()->EnterEvadeMode();
130 
131  pMalone->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
132  }
133 
134  EnterEvadeMode();
135  return 0;
136  }
137 
138  switch (uiStep)
139  {
140  case 1:
141  if (Creature* pCreepjack = m_creature->GetMap()->GetCreature(m_creepjackGuid))
142  pCreepjack->SetStandState(UNIT_STAND_STATE_STAND);
143 
144  if (Creature* pMalone = m_creature->GetMap()->GetCreature(m_maloneGuid))
145  pMalone->SetStandState(UNIT_STAND_STATE_STAND);
146 
147  DoScriptText(SAY_START, m_creature, pPlayer);
148  return 2000;
149  case 2:
150  if (Creature* pMalone = m_creature->GetMap()->GetCreature(m_maloneGuid))
151  {
152  pMalone->SetFacingToObject(pPlayer);
153  DoScriptText(EMOTE_KNUCKLES, pMalone, pPlayer);
154  }
155  if (Creature* pCreepjack = m_creature->GetMap()->GetCreature(m_creepjackGuid))
156  {
157  pCreepjack->SetFacingToObject(pPlayer);
158  DoScriptText(EMOTE_KNUCKLES, pCreepjack, pPlayer);
159  }
160  return 2000;
161  case 3: DoScriptText(SAY_COUNT, m_creature, pPlayer); return 5000;
162  case 4: DoScriptText(SAY_COUNT_1, m_creature, pPlayer); return 2000;
163  case 5: DoScriptText(SAY_COUNT_2, m_creature, pPlayer); return 2000;
164  case 6: DoScriptText(SAY_ATTACK, m_creature, pPlayer); return 2000;
165  case 7:
166  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
167  SetReactState(REACT_AGGRESSIVE);
168 
169  if (pPlayer->IsAlive())
170  {
171  if (!m_creature->IsInCombat())
172  AttackStart(pPlayer);
173 
174  if (Creature* pMalone = m_creature->GetMap()->GetCreature(m_maloneGuid))
175  {
176  pMalone->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
177  if (!pMalone->IsInCombat())
178  pMalone->AI()->AttackStart(pPlayer);
179  }
180 
181  if (Creature* pCreepjack = m_creature->GetMap()->GetCreature(m_creepjackGuid))
182  {
183  pCreepjack->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_PLAYER);
184  if (!pCreepjack->IsInCombat())
185  pCreepjack->AI()->AttackStart(pPlayer);
186  }
187  }
188 
189  bActiveAttack = true;
190  return 500;
191  case 8:
192  DoScriptText(SAY_GIVEUP, m_creature, pPlayer);
193  return 4000;
194  case 9:
195  m_creature->GetMotionMaster()->MoveTargetedHome();
196  EnterEvadeMode();
197  return 0;
198  default:
199  return 0;
200  }
201  }
