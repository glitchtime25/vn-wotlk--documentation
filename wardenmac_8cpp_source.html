24 {
25  sWardenScanMgr.AddMacScan(std::make_shared<MacScan>(
26  // builder
27  [](const Warden *warden, std::vector<std::string> &, ByteBuffer &scan)
28  {
29  auto const macWarden = reinterpret_cast<const WardenMac *>(warden);
30 
31  MANGOS_ASSERT(macWarden->_hashString.size() <= 0xFF);
32 
33  scan << static_cast<uint8>(macWarden->_hashString.size());
34 
35  // skip null terminator this way
36  scan.append(macWarden->_hashString.c_str(), macWarden->_hashString.size());
37  },
38  // checker

40  {
41  auto const macWarden = reinterpret_cast<const WardenMac *>(warden);
42 
43  uint8 sha[Sha1Hash::GetLength()];
44  uint8 md5[MD5Hash::GetLength()];
45 
46  buff.read(sha, sizeof(sha));
47  buff.read(md5, sizeof(md5));
48 
49  return !!memcmp(sha, macWarden->_hashSHA, sizeof(sha)) || !!memcmp(md5, macWarden->_hashMD5, sizeof(md5));
50  }, 128, sizeof(uint8) + Sha1Hash::GetLength() + MD5Hash::GetLength(), "Mac string hash check"));

53 WardenMac::WardenMac(WorldSession *session, const BigNumber &K, SessionAnticheatInterface *anticheat)
54  : Warden(session, session->GetPlatform() == CLIENT_PLATFORM_X86 ? sWardenModuleMgr.GetMacModule() : nullptr, K, anticheat), _fingerprintSaved(false)
55 {
56  std::stringstream hash;
57 
58  // the only known capability of the Mac Warden module is hashing a string and sending back the hashed value
59  // so at least we can make the string that we ask it to hash change by account, I guess...
60  hash << "namreeb was here. please be good. your username is " << _session->GetAccountName();
61 
62  _hashString = hash.str();
63 
64  static constexpr uint32 magic = 0xFEEDFACE;
65 
66  Sha1Hash sha1;
67  sha1.UpdateData(_hashString);
68  if (_module) // this constant is only used if there is a module
69  sha1.UpdateData(reinterpret_cast<const uint8*>(&magic), sizeof(magic));
70  sha1.Finalize();
71 
72  memcpy(_hashSHA, sha1.GetDigest(), sizeof(_hashSHA));
73 
74  MD5Hash md5;
75  md5.UpdateData(_hashString);
76  md5.Finalize();
77  std::memcpy(_hashMD5, md5.GetDigest(), md5.GetLength());
78 
79  // PPC no module, begin string hashing requests directly
80  if (!_module)
81  {
82  // at this point the client has our module loaded. send whatever packets are necessary to initialize Warden
83  InitializeClient();
84 
85  // send any initial hack scans that the scan manager may have for us
86  RequestScans(SelectScans(ScanFlags::InitialLogin));
87 
88  // begin the scan clock (note that even if the clock expires before any initial scans are answered, no new
89  // checks will be requested until the reply is received).
90  BeginScanClock();
91  }
92 }
