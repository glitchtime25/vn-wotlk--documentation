297 void BattleGroundAV::Update(uint32 diff)
298 {
299  BattleGround::Update(diff);
300 
301  if (m_mineYellTimer)
302  {
303  if (m_mineYellTimer <= diff)
304  {
305  if (Creature* morloch = GetSingleCreatureFromStorage(BG_AV_NPC_MORLOCH))
306  SendBcdToTeam(YELL_MORLOCH_MINE, CHAT_MSG_MONSTER_YELL, morloch, ALLIANCE);

309  m_mineYellTimer = 0;
310  }
311  else
312  m_mineYellTimer -= diff;
313  }
314 
315  if (GetStatus() != STATUS_IN_PROGRESS)
316  return;
317 
318  // add points from mine owning, and look if the neutral team can reclaim the mine
319  for (uint8 mine = 0; mine < BG_AV_MAX_MINES; ++mine)
320  {
321  if (m_mineOwner[mine] != TEAM_INDEX_NEUTRAL)
322  {
323  if (m_mineTimer[mine] < diff)
324  {
325  UpdateScore(PvpTeamIndex(m_mineOwner[mine]), 1);
326  m_mineTimer[mine] = BG_AV_MINE_TICK_TIMER;
327  }
328  else
329  m_mineTimer[mine] -= diff;
330 
331  if (m_mineReclaimTimer[mine] > diff)
332  m_mineReclaimTimer[mine] -= diff;
333  else
334  ChangeMineOwner(AVMineIds(mine), TEAM_INDEX_NEUTRAL);
335  }
336  }
337 
338  // looks for all timers of the nodes and destroy the building (for graveyards the building wont get destroyed, it goes just to the other team
339  for (uint8 i = 0; i < BG_AV_MAX_NODES; ++i)
340  {
341  if (m_nodes[i].state == POINT_ASSAULTED)
342  {
343  if (m_nodes[i].timer > diff)
344  m_nodes[i].timer -= diff;
345  else
346  ProcessPlayerDestroyedPoint(AVNodeIds(i));
347  }
348  }

390 void BattleGroundAV::EndBattleGround(Team winner)
391 {
392  // calculate bonuskills for both teams:
393  uint32 gravesOwned[PVP_TEAM_COUNT] = {0, 0};
394  uint32 minesOwned[PVP_TEAM_COUNT] = {0, 0};
395 
396  // graves all controlled
397  for (uint8 i = 0; i < BG_AV_MAX_NODES; ++i)
398  if (m_nodes[i].state == POINT_CONTROLLED && m_nodes[i].owner != TEAM_INDEX_NEUTRAL)
399  ++gravesOwned[m_nodes[i].owner];
400 
401  for (auto& i : m_mineOwner)
402  if (i != TEAM_INDEX_NEUTRAL)
403  ++minesOwned[i];
404 
405  for (uint8 i = 0; i < PVP_TEAM_COUNT; ++i)
406  {
407  Team team = GetTeamIdByTeamIndex(PvpTeamIndex(i));
408 
409  uint32 bonusHonor = GetBonusHonorFromKill(m_homeTowersControlled[i] * BG_AV_KILL_SURVIVING_TOWER);
410  uint32 bonusReputation = m_homeTowersControlled[i] * m_repSurviveTower;
411 
412  // bonus tower reputation and honor
413  if (bonusReputation)
414  RewardReputationToTeam(avTeamFactions[i], m_homeTowersControlled[i] * m_repSurviveTower, team);
415  if (bonusHonor)
416  RewardHonorToTeam(bonusHonor, team);
417 
418  DEBUG_LOG("BattleGroundAV: Team %u wins the battleground. Controlled towers: %u bonus towers honor: %u bonus towers reputation: %u", i, m_homeTowersControlled[i], bonusHonor, bonusReputation);
419 
420  // bonus graveyards and mines reputation
421  if (gravesOwned[i])
422  RewardReputationToTeam(avTeamFactions[i], gravesOwned[i] * m_repOwnedGrave, team);
423  if (minesOwned[i])
424  RewardReputationToTeam(avTeamFactions[i], minesOwned[i] * m_repOwnedMine, team);
425 
426  // captain survived; bonus reputation and honor

439  RewardHonorToTeam(m_honorMapComplete, HORDE);
440  }
