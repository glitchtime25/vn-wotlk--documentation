253 static uint32 ReadDBCBuild(const std::string& dbc_path, LocaleNameStr const* localeNameStr = nullptr)
254 {
255  std::string text;
256 
257  if (!localeNameStr)
258  {
259  for (LocaleNameStr const* itr = &fullLocaleNameList[0]; itr->name; ++itr)
260  {
261  if (ReadDBCBuildFileText(dbc_path, itr->name, text))
262  {
263  break;
264  }
265  }
266  }
267  else
268  ReadDBCBuildFileText(dbc_path, localeNameStr->name, text);
269 
270  if (text.empty())
271  return 0;
272 
273  size_t pos = text.find("version=\"");
274  size_t pos1 = pos + strlen("version=\"");
275  size_t pos2 = text.find('\"', pos1);
276  if (pos == std::string::npos || pos2 == std::string::npos || pos1 >= pos2)
277  return 0;
278 
279  std::string build_str = text.substr(pos1, pos2 - pos1);
280 
281  int build = atoi(build_str.c_str());
282  if (build <= 0)
283  return 0;
284 
285  return build;
286 }

308 template<class T>
309 inline void LoadDBC(LocalData& localeData, BarGoLink& bar, StoreProblemList& errlist, DBCStorage<T>& storage, const std::string& dbc_path, const std::string& filename)
310 {
311  // compatibility format and C++ structure sizes
312  MANGOS_ASSERT(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()) == sizeof(T) || LoadDBC_assert_print(DBCFileLoader::GetFormatRecordSize(storage.GetFormat()), sizeof(T), filename));
313 
314  std::string dbc_filename = dbc_path + filename;
315  if (storage.Load(dbc_filename.c_str()))
316  {
317  bar.step();
318  for (uint8 i = 0; fullLocaleNameList[i].name; ++i)
319  {
320  if (!(localeData.availableDbcLocales & (1 << i)))
321  continue;
322 
323  LocaleNameStr const* localStr = &fullLocaleNameList[i];
324 
325  std::string dbc_dir_loc = dbc_path + localStr->name + "/";
326 
327  if (!(localeData.checkedDbcLocaleBuilds & (1 << i)))
328  {
329  localeData.checkedDbcLocaleBuilds |= (1 << i); // mark as checked for speedup next checks
330 
331  uint32 build_loc = ReadDBCBuild(dbc_dir_loc, localStr);
332  if (localeData.main_build != build_loc)
333  {
334  localeData.availableDbcLocales &= ~(1 << i); // mark as not available for speedup next checks
335 
336  // exist but wrong build
337  if (build_loc)
338  {
339  std::string dbc_filename_loc = dbc_path + localStr->name + "/" + filename;
340  char buf[200];
341  snprintf(buf, 200, " (exist, but DBC locale subdir %s have DBCs for build %u instead expected build %u, it and other DBC from subdir skipped)", localStr->name, build_loc, localeData.main_build);
342  errlist.push_back(dbc_filename_loc + buf);
343  }
344 
345  continue;
346  }
347  }
348 
349  std::string dbc_filename_loc = dbc_path + localStr->name + "/" + filename;
350  if (!storage.LoadStringsFrom(dbc_filename_loc.c_str()))
351  localeData.availableDbcLocales &= ~(1 << i);// mark as not available for speedup next checks
352  }
353  }
354  else
355  {
356  // sort problematic dbc to (1) non compatible and (2) nonexistent
357  FILE* f = fopen(dbc_filename.c_str(), "rb");
358  if (f)
359  {
360  char buf[100];
361  snprintf(buf, 100, " (exist, but have %u fields instead " SIZEFMTD ") Wrong client version DBC file?", storage.GetFieldCount(), strlen(storage.GetFormat()));
362  errlist.push_back(dbc_filename + buf);
363  fclose(f);
364  }
365  else
366  errlist.push_back(dbc_filename);
367  }
368 }
