28 void PointMovementGenerator::Initialize(Unit& unit)
29 {
30  if (unit.hasUnitState(UNIT_STAT_NO_FREE_MOVE | UNIT_STAT_NOT_MOVE))
31  return;
32 
33  // Stop any previously dispatched splines no matter the source
34  if (!unit.movespline->Finalized() && !m_speedChanged)
35  {
36  if (unit.IsClientControlled())
37  unit.StopMoving(true);
38  else
39  unit.InterruptMoving();
40  }
41 
42  unit.addUnitState(UNIT_STAT_ROAMING | UNIT_STAT_ROAMING_MOVE);
43 
44  Move(unit);
45 
46  m_speedChanged = false;
47 }

91 void PointMovementGenerator::Move(Unit& unit)
92 {
93  Movement::MoveSplineInit init(unit);
94  init.MoveTo(m_x, m_y, m_z, m_generatePath);
95  if (m_forcedMovement == FORCED_MOVEMENT_WALK)
96  init.SetWalk(true);
97  else if (m_forcedMovement == FORCED_MOVEMENT_RUN)
98  init.SetWalk(false);
99  else
100  init.SetWalk(!unit.hasUnitState(UNIT_STAT_RUNNING));
101  if (m_forcedMovement == FORCED_MOVEMENT_FLIGHT)
102  init.SetFly();
103  if (m_o != 0.f)
104  init.SetFacing(m_o);
105  init.SetVelocity(m_speed);
106  init.Launch();
107 }

109 void PointMovementGenerator::MovementInform(Unit& unit)
110 {
111  MovementGeneratorType const type = GetMovementGeneratorType();
112 
113  if (UnitAI* ai = unit.AI())
114  ai->MovementInform(type, m_id);
115 
116  if (unit.GetTypeId() == TYPEID_UNIT && static_cast<Creature&>(unit).IsTemporarySummon())
117  {
118  if (unit.GetSpawnerGuid().IsAnyTypeCreature())
119  if (Creature* pSummoner = unit.GetMap()->GetCreature(unit.GetSpawnerGuid()))
120  {
121  if (UnitAI* ai = pSummoner->AI())
122  ai->SummonedMovementInform(static_cast<Creature*>(&unit), type, m_id);
123  }
124  }
125 
126  if (m_relayId)
127  unit.GetMap()->ScriptsStart(SCRIPT_TYPE_RELAY, m_relayId, &unit, m_guid ? unit.GetMap()->GetWorldObject(m_guid) : nullptr);
128 }
