161 {
162  int length = (minSize >= GetNumBytes()) ? minSize : GetNumBytes();
163 
164  std::vector<uint8> byteArray(length);
165 
166  // If we need more bytes than length of BigNumber set the rest to 0
167  if (length > GetNumBytes())
168  memset((void*)byteArray.data(), 0, length);
169 
170  // Padding should add leading zeroes, not trailing
171  auto const paddingOffset = length - GetNumBytes();
172 
173  BN_bn2bin(_bn, (unsigned char*)byteArray.data() + paddingOffset);
174
