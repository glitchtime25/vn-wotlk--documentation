234 uint32 TimeStringToSecs(const std::string& timestring)
235 {
236  uint32 secs = 0;
237  uint32 buffer = 0;
238  uint32 multiplier = 0;
239 
240  for (char itr : timestring)
241  {
242  if (isdigit(itr))
243  {
244  buffer *= 10;
245  buffer += itr - '0';
246  }
247  else
248  {
249  switch (itr)
250  {
251  case 'd': multiplier = DAY; break;
252  case 'h': multiplier = HOUR; break;
253  case 'm': multiplier = MINUTE; break;
254  case 's': multiplier = 1; break;
255  default : return 0; // bad format
256  }
257  buffer *= multiplier;
258  secs += buffer;
259  buffer = 0;
260  }
261  }
262 
263  return secs;
264 }

329 bool Utf8toWStr(const std::string& utf8str, std::wstring& wstr, size_t max_len)
330 {
331  if (utf8str.empty())
332  {
333  wstr = std::wstring();
334  return true;
335  }
336 
337  try
338  {
339  // A UTF8 string can have a maximum of 4 octets per character
340  // A 4 octet char can take up to two UTF16 characters (4*8 = 32 / 16 = 2)
341  // The UTF8 string may also actually be ASCII, in which case no truncation
342  // takes place! The final string length is therefore unknown. Reserve
343  // as long as the OG string, and back-insert
344  wstr.resize(utf8str.size());
345 
346  auto end = utf8::utf8to16(utf8str.cbegin(), utf8str.cend(), wstr.begin());
347 
348  if (end != wstr.end())
349  wstr.erase(end, wstr.end());
350 
351  // truncate to max len
352  if (!!max_len && wstr.size() > max_len)
353  {
354  wstr.resize(max_len);
355  }
356  }
357  catch (const std::exception&)
358  {
359  wstr = L"";
360  return false;
361  }
362 
363  return true;
364 }
