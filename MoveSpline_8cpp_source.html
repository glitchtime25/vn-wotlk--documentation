30  Location MoveSpline::ComputePosition() const
31  {
32  MANGOS_ASSERT(Initialized());
33 
34  float u = 1.f;
35  int32 seg_time = spline.length(point_Idx, point_Idx + 1);
36  if (seg_time > 0)
37  u = (time_passed - spline.length(point_Idx)) / (float)seg_time;
38  Location c;
39  c.orientation = initialOrientation;
40  spline.evaluate_percent(point_Idx, u, c);
41 
42  if (splineflags.animation)
43  ;// MoveSplineFlag::Animation disables falling or parabolic movement
44  else if (splineflags.parabolic)
45  computeParabolicElevation(c.z);
46  else if (splineflags.falling)
47  computeFallElevation(c.z);
48 
49  if (splineflags.done && splineflags.isFacing())
50  {
51  if (splineflags.final_angle)
52  c.orientation = facing.angle;
53  else if (splineflags.final_point)
54  c.orientation = atan2(facing.f.y - c.y, facing.f.x - c.x);
55  // nothing to do for MoveSplineFlag::Final_Target flag
56  }
57  else
58  {
59  if (!splineflags.hasFlag(MoveSplineFlag::OrientationFixed | MoveSplineFlag::Falling))
60  {
61  Vector3 hermite;
62  spline.evaluate_derivative(point_Idx, u, hermite);
63  c.orientation = atan2(hermite.y, hermite.x);
64  }
65 
66  if (splineflags.orientationInversed)
67  c.orientation = -c.orientation;
68  }
69  return c;
70  }

72  void MoveSpline::computeParabolicElevation(float& el) const
73  {
74  if (time_passed > static_cast<uint32>(parabolic_start_time))
75  {
76  float t_passedf = MSToSec(time_passed - parabolic_start_time);
77  float t_durationf = MSToSec(Duration() - parabolic_start_time); // client use not modified duration here
78 
79  // -a*x*x + bx + c:
80  //(dur * v3->z_acceleration * dt)/2 - (v3->z_acceleration * dt * dt)/2 + Z;
81  el += (t_durationf - t_passedf) * 0.5f * vertical_acceleration * t_passedf;
82  }
83  }

127  void MoveSpline::init_spline(const MoveSplineInitArgs& args)
128  {
129  const SplineBase::EvaluationMode modes[2] = {SplineBase::ModeLinear, SplineBase::ModeCatmullrom};
130  if (args.flags.cyclic)
131  {
132  uint32 cyclic_point = 0;
133  if (splineflags.enter_cycle)
134  cyclic_point = 1; // shouldn't be modified, came from client
135  spline.init_cyclic_spline(&args.path[0], args.path.size(), modes[args.flags.isSmooth()], cyclic_point);
136  }
137  else
138  {
139  spline.init_spline(&args.path[0], args.path.size(), modes[args.flags.isSmooth()]);
140  }
141 
142  // init spline timestamps
143  if (splineflags.falling)
144  {
145  FallInitializer init(spline.getPoint(spline.first()).z);
146  spline.initLengths(init);
147  }
148  else
149  {
150  CommonInitializer init(args.velocity);
151  spline.initLengths(init);
152  }
153 
154  // TODO: what to do in such cases? problem is in input data (all points are at same coords)
155  if (spline.length() < minimal_duration)
156  {
157  sLog.outError("MoveSpline::init_spline: zero length spline, wrong input data?");
158  spline.set_length(spline.last(), spline.isCyclic() ? 1000 : 1);
159  }
160  point_Idx = spline.first();
161  speed = args.velocity;
162  }

164  void MoveSpline::Initialize(const MoveSplineInitArgs& args)
165  {
166  splineflags = args.flags;
167  facing = args.facing;
168  m_Id = args.splineId;
169  point_Idx_offset = args.path_Idx_offset;
170  initialOrientation = args.initialOrientation;
171

173  vertical_acceleration = 0.f;
174  parabolic_start_time = 0;
175  animation_start_time = 0;
176 
177  // detect Stop command
178  if (splineflags.done)
179  {
180  spline.clear();
181  return;
182  }
183 
184  init_spline(args);
185 
186  // init parabolic / animation
187  // spline initialized, duration known and i able to compute parabolic acceleration
188  if (args.flags & (MoveSplineFlag::Parabolic))
189  {
190  parabolic_start_time = DurationAtPointIdx(args.parabolic_start_Idx);
191  if (args.flags.parabolic && static_cast<uint32>(parabolic_start_time) < Duration())
192  {
193  float f_duration = MSToSec(Duration() - parabolic_start_time);
194  vertical_acceleration = args.parabolic_amplitude * 8.f / (f_duration * f_duration);
195  }
196  }
197  if (args.flags & (MoveSplineFlag::Animation))
198  animation_start_time = DurationAtPointIdx(args.animation_start_Idx);
199  }

226 // MONSTER_MOVE packet format limitation for not CatmullRom movement:
227 // each vertex offset packed into 11 bytes
228  bool MoveSplineInitArgs::_checkPathBounds() const
229  {
230  if (!(flags & MoveSplineFlag::Mask_CatmullRom) && path.size() > 2)
231  {
232  enum
233  {
234  MAX_OFFSET = (1 << 11) / 2,
235  };
236  Vector3 middle = (path.front() + path.back()) / 2;
237  for (uint32 i = 1; i < path.size() - 1; ++i)
238  {
239  Vector3 offset = path[i] - middle;
240  if (fabs(offset.x / 0.25f) >= MAX_OFFSET || fabs(offset.y / 0.25f) >= MAX_OFFSET || fabs(offset.z / 0.25f) >= MAX_OFFSET)
241  {
242  sLog.outError("MoveSplineInitArgs::_checkPathBounds check failed");
243  return false;
244  }
245  }
246  }
247  return true;

252  MoveSpline::UpdateResult MoveSpline::_updateState(uint32& ms_time_diff)
253  {
254  if (Finalized())
255  {
256  ms_time_diff = 0;
257  return Result_Arrived;
258  }
259 
260  UpdateResult result = Result_None;
261 
262  uint32 minimal_diff = std::min(ms_time_diff, segment_time_elapsed());
263  time_passed += minimal_diff;
264  ms_time_diff -= minimal_diff;
265 
266  if (time_passed >= next_timestamp())
267  {
268  ++point_Idx;
269  if (point_Idx < spline.last())
270  {
271  result = Result_NextSegment;
272  }
273  else
274  {
275  if (spline.isCyclic())
276  {
277  point_Idx = spline.first();
278  time_passed = time_passed % Duration();
279  result = Result_NextSegment;
280 
281  // Remove first point from the path after one full cycle.
282  // That point was the position of the unit prior to entering the cycle and it shouldn't be repeated with continuous cycles.
283  if (splineflags.enter_cycle)
284  {
285  splineflags.enter_cycle = false;
286 
287  MoveSplineInitArgs args{ (size_t)spline.getPointCount() };
288  args.path.assign(spline.getPoints().begin() + spline.first() + 1, spline.getPoints().begin() + spline.last());
289  args.facing = facing;
290  args.flags = splineflags;
291  args.path_Idx_offset = point_Idx_offset;
292  // MoveSplineFlag::Parabolic | MoveSplineFlag::Animation not supported currently
293  //args.parabolic_amplitude = ?;
294  //args.time_perc = ?;
295  args.splineId = m_Id;
296  args.velocity = 1.0f; // Calculated below
297  if (args.Validate(nullptr))
298  {
299  // New cycle should preserve previous cycle's duration for some weird reason, even though
300  // the path is really different now. Developers are weird. Or this was just a simple oversight.
301  // Since our splines precalculate length with zspeed in mind, if we want to find the desired
302  // zspeed, we have to make a fake spline, calculate its duration and then compare it to the
303  // desired duration, thus finding out how much the zspeed has to be increased for them to match.
304  MoveSpline tempSpline;
305  tempSpline.Initialize(args);
306  args.velocity = (float)tempSpline.Duration() / Duration();
307 
308  if (args.Validate(nullptr))
309  init_spline(args);
310  }
311  }
312  }
313  else
314  {
315  _Finalize();
316  ms_time_diff = 0;
317  result = Result_Arrived;
318  }
319  }
320  }
321 
322  return result;
323  }
