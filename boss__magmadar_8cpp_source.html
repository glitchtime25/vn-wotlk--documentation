87  void ExecuteAction(uint32 action) override
88  {
89  switch (action)
90  {
91  case MAGMADAR_FRENZY:
92  {
93  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
94  {
95  DoScriptText(EMOTE_GENERIC_FRENZY_KILL, m_creature);
96  ResetCombatAction(action, urand(15000, 20000));
97  }
98  break;
99  }
100  case MAGMADAR_PANIC:
101  {
102  if (DoCastSpellIfCan(nullptr, SPELL_PANIC) == CAST_OK)
103  ResetCombatAction(action, urand(16000, 21000));
104  break;
105  }
106  case MAGMADAR_LAVABOMB:
107  {
108  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_LAVABOMB, SELECT_FLAG_POWER_NOT_MANA))
109  if (DoCastSpellIfCan(target, SPELL_LAVABOMB) == CAST_OK)
110  ResetCombatAction(action, urand(12000, 15000));
111  break;
112  }
113  case MAGMADAR_LAVABOMB_MANA:
114  {
115  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_LAVABOMB_MANA, SELECT_FLAG_POWER_MANA))
116  if (DoCastSpellIfCan(target, SPELL_LAVABOMB_MANA) == CAST_OK)
117  ResetCombatAction(action, urand(12000, 15000));
118  break;
119  }
120  }
121  }
122 };
