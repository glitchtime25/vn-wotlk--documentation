348  void MovementInform(uint32 uiMoveType, uint32 uiPointId) override
349  {
350  // remove flight anim and start moving
351  if (uiMoveType == POINT_MOTION_TYPE)
352  {
353  switch (uiPointId)
354  {
355  case POINT_ID_COMBAT:
356  m_creature->SetLevitate(false);
357  SetCombatMovement(true);
358  DoStartMovement(m_creature->GetVictim());
359  m_creature->RemoveByteFlag(UNIT_FIELD_BYTES_1, 3, UNIT_BYTE1_FLAG_FLY_ANIM);
360 
361  // after vortex phase inform the Power sparks and set phase
362  if (m_uiPhase != PHASE_FLOOR)
363  {
364  SendAIEventAround(AI_EVENT_CUSTOM_D, m_creature, 0, 50.0f);
365  m_uiPowerSparkTimer = 10000;
366  m_uiPhase = PHASE_FLOOR;
367  }
368  break;
369  case POINT_ID_VORTEX:
370  if (m_pInstance)
371  {
372  if (Creature* pTrigger = m_pInstance->GetSingleCreatureFromStorage(NPC_LARGE_TRIGGER))
373  pTrigger->CastSpell(pTrigger, SPELL_VORTEX_VISUAL, TRIGGERED_NONE);
374  }
375  if (DoCastSpellIfCan(m_creature, SPELL_VORTEX) == CAST_OK)
376  {
377  DoCastSpellIfCan(m_creature, SPELL_VORTEX_STUN, CAST_TRIGGERED);
378  m_creature->GetMotionMaster()->MoveIdle();
379  m_uiPhase = PHASE_VORTEX;
380  m_uiVortexEndTimer = 10000;
381  }
382  break;
383  case POINT_ID_TRANSITION_1:
384  case POINT_ID_TRANSITION_2:
385  // no action here
386  break;
387  case POINT_ID_FLIGHT:
388  // start wp movement during phase 2; use path id 1
389  m_creature->GetMotionMaster()->MoveWaypoint(1);
390  break;
391  case POINT_ID_SURGE:
392  if (DoCastSpellIfCan(m_creature, SPELL_SURGE_OF_POWER_PULSE) == CAST_OK)
393  {
394  DoScriptText(SAY_DEEP_BREATH, m_creature);
395  DoScriptText(SAY_EMOTE_BREATH, m_creature);
396  m_uiArcanePulseEndTimer = 8000;
397  m_uiPhase = PHASE_SURGE;
398  }
399  break;
400  }
401  }
402  // intro yells; handled once per boss circle; ToDo: maybe this is more dyanmic than this
403  else if (uiMoveType == WAYPOINT_MOTION_TYPE && uiPointId == 1 && m_uiIntroStage < 5 && !m_creature->IsInCombat())

415  }
416  }
