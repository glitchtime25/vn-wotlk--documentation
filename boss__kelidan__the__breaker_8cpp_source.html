137  void DoSetupAdds()
138  {
139  DoCastSpellIfCan(nullptr, SPELL_EVOCATION);
140  if (!m_instance)
141  return;
142 
143  GuidList lAddGuids;
144  m_instance->GetKelidanAddList(lAddGuids);
145 
146  // Sort Adds to vector if not already done
147  if (!lAddGuids.empty())
148  {
149  m_vAddGuids.reserve(lAddGuids.size());
150  CreatureList lAdds;
151  for (GuidList::const_iterator itr = lAddGuids.begin(); itr != lAddGuids.end(); ++itr)
152  {
153  if (Creature* pAdd = m_instance->instance->GetCreature(*itr))
154  lAdds.push_back(pAdd);
155  }
156  // Sort them by angle
157  lAdds.sort([=](Creature* left, Creature* right) -> bool {return m_creature->GetAngle(left) < m_creature->GetAngle(right); });
158  for (std::list<Creature*>::const_iterator itr = lAdds.begin(); itr != lAdds.end(); ++itr)
159  m_vAddGuids.push_back((*itr)->GetObjectGuid());
160  }
161 
162  // Respawn killed adds and reset counter
163  m_uiKilledAdds = 0;
164  for (GuidVector::const_iterator itr = m_vAddGuids.begin(); itr != m_vAddGuids.end(); ++itr)
165  {
166  Creature* pAdd = m_instance->instance->GetCreature(*itr);
167  if (pAdd)
168  {
169  if (pAdd->GetCombatManager().IsEvadingHome())
170  {
171  ResetTimer(KELIDAN_SETUP_ADDS, 3000);
172  return;
173  }
174  if (!pAdd->IsAlive())
175  pAdd->Respawn();
176  }
177  }
178 
179  // Cast pentagram
180  uint8 s = m_vAddGuids.size();
181  for (uint8 i = 0; i < s; ++i)
182  {
183  Creature* caster = m_instance->instance->GetCreature(m_vAddGuids[i]);
184  Creature* target = m_instance->instance->GetCreature(m_vAddGuids[(i + 2) % s]);
185  if (caster && target)
186  caster->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, target, caster);
187  }
188 
189  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
190  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_IMMUNE_TO_NPC);
191  }
192
