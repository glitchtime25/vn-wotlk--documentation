67 
68  if (iter != tokens.end())

71  port_or_socket_dir = *iter++;
72  if (iter != tokens.end())
73  user = *iter++;
74  if (iter != tokens.end())
75  password = *iter++;
76  if (iter != tokens.end())
77  database = *iter++;
78 
79  if (host == ".")
80  mPGconn = PQsetdbLogin(nullptr, port_or_socket_dir == "." ? nullptr : port_or_socket_dir.c_str(), nullptr, nullptr, database.c_str(), user.c_str(), password.c_str());
81  else
82  mPGconn = PQsetdbLogin(host.c_str(), port_or_socket_dir.c_str(), nullptr, nullptr, database.c_str(), user.c_str(), password.c_str());
83 
84  /* check to see that the backend connection was successfully made */
85  if (PQstatus(mPGconn) != CONNECTION_OK)
86  {
87  sLog.outError("Could not connect to Postgre database at %s: %s",
88  host.c_str(), PQerrorMessage(mPGconn));
89  PQfinish(mPGconn);
90  mPGconn = nullptr;
91  return false;
92  }
93 
94  DETAIL_LOG("Connected to Postgre database %s@%s:%s/%s", user.c_str(), host.c_str(), port_or_socket_dir.c_str(), database.c_str());
95  sLog.outString("PostgreSQL server ver: %d", PQserverVersion(mPGconn));
96  return true;
97 }

99 bool PostgreSQLConnection::_Query(const char* sql, PGresult** pResult, uint64* pRowCount, uint32* pFieldCount)
100 {
101  if (!mPGconn)
102  return false;
103 
104  uint32 _s = WorldTimer::getMSTime();
105  // Send the query
106  *pResult = PQexec(mPGconn, sql);
107  if (!*pResult)
108  return false;
109 
110  if (PQresultStatus(*pResult) != PGRES_TUPLES_OK)
111  {
112  sLog.outErrorDb("SQL : %s", sql);
113  sLog.outErrorDb("SQL %s", PQerrorMessage(mPGconn));
114  PQclear(*pResult);
115  return false;
116  }
117  else
118  {
119  DEBUG_FILTER_LOG(LOG_FILTER_SQL_TEXT, "[%u ms] SQL: %s", WorldTimer::getMSTimeDiff(_s, WorldTimer::getMSTime()), sql);
120  }
121 
122  *pRowCount = PQntuples(*pResult);
123  *pFieldCount = PQnfields(*pResult);
124  // end guarded block
125 
126  if (!*pRowCount)
127  {
128  PQclear(*pResult);
129  return false;
130  }
131 
132  return true;
133 }
