37 void BIH::subdivide(int left, int right, std::vector<uint32>& tempTree, buildData& dat, AABound& gridBox, AABound& nodeBox, int nodeIndex, int depth, BuildStats& stats)
38 {
39  if ((right - left + 1) <= dat.maxPrims || depth >= MAX_STACK_SIZE)
40  {
41  // write leaf node
42  stats.updateLeaf(depth, right - left + 1);
43  createNode(tempTree, nodeIndex, left, right);
44  return;
45  }
46  // calculate extents
47  int axis = -1, rightOrig;
48  float clipL;
49  float clipR;
50  float prevClip = G3D::fnan();
51  float split = G3D::fnan();
52  bool wasLeft = true;
53  while (true)
54  {
55  int prevAxis = axis;
56  float prevSplit = split;
57  // perform quick consistency checks
58  Vector3 d(gridBox.hi - gridBox.lo);
59  if (d.x < 0 || d.y < 0 || d.z < 0)
60  throw std::logic_error("negative node extents");
61  for (int i = 0; i < 3; ++i)
62  {
63  if (nodeBox.hi[i] < gridBox.lo[i] || nodeBox.lo[i] > gridBox.hi[i])
64  {
65  // UI.printError(Module.ACCEL, "Reached tree area in error - discarding node with: %d objects", right - left + 1);
66  throw std::logic_error("invalid node overlap");
67  }
68  }
69  // find longest axis
70  axis = d.primaryAxis();
71  split = 0.5f * (gridBox.lo[axis] + gridBox.hi[axis]);
72  // partition L/R subsets
73  clipL = -G3D::inf();
74  clipR = G3D::inf();
75  rightOrig = right; // save this for later
76  float nodeL = G3D::inf();
77  float nodeR = -G3D::inf();
78  for (int i = left; i <= right;)
79  {
80  int obj = dat.indices[i];
81  float minb = dat.primBound[obj].low()[axis];
82  float maxb = dat.primBound[obj].high()[axis];
83  float center = (minb + maxb) * 0.5f;
84  if (center <= split)
85  {
86  // stay left
87  ++i;
88  if (clipL < maxb)
89  clipL = maxb;
90  }
91  else
92  {
93  // move to the right most
94  int t = dat.indices[i];
95  dat.indices[i] = dat.indices[right];
96  dat.indices[right] = t;
97  --right;
98  if (clipR > minb)
99  clipR = minb;
100  }
101  nodeL = std::min(nodeL, minb);
102  nodeR = std::max(nodeR, maxb);
103  }
104  // check for empty space
105  if (nodeL > nodeBox.lo[axis] && nodeR < nodeBox.hi[axis])
106  {
107  float nodeBoxW = nodeBox.hi[axis] - nodeBox.lo[axis];
108  float nodeNewW = nodeR - nodeL;
109  // node box is too big compare to space occupied by primitives?
110  if (1.3f * nodeNewW < nodeBoxW)
111  {
112  stats.updateBVH2();
113  int nextIndex = tempTree.size();
114  // allocate child
115  tempTree.push_back(0);
116  tempTree.push_back(0);
117  tempTree.push_back(0);
118  // write bvh2 clip node
119  stats.updateInner();
120  tempTree[nodeIndex + 0] = (axis << 30) | (1 << 29) | nextIndex;
121  tempTree[nodeIndex + 1] = floatToRawIntBits(nodeL);
122  tempTree[nodeIndex + 2] = floatToRawIntBits(nodeR);
123  // update nodebox and recurse
124  nodeBox.lo[axis] = nodeL;
125  nodeBox.hi[axis] = nodeR;
126  subdivide(left, rightOrig, tempTree, dat, gridBox, nodeBox, nextIndex, depth + 1, stats);
127  return;
128  }
129  }
130  // ensure we are making progress in the subdivision
131  if (right == rightOrig)
132  {
133  // all left
134  if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split))
135  {
136  // we are stuck here - create a leaf
137  stats.updateLeaf(depth, right - left + 1);
138  createNode(tempTree, nodeIndex, left, right);
139  return;
140  }
141  if (clipL <= split)
142  {
143  // keep looping on left half
144  gridBox.hi[axis] = split;
145  prevClip = clipL;
146  wasLeft = true;
147  continue;
148  }
149  gridBox.hi[axis] = split;
150  prevClip = G3D::fnan();
151  }
152  else if (left > right)
153  {
154  // all right
155  right = rightOrig;
156  if (prevAxis == axis && G3D::fuzzyEq(prevSplit, split))
157  {
158  // we are stuck here - create a leaf
159  stats.updateLeaf(depth, right - left + 1);
160  createNode(tempTree, nodeIndex, left, right);
161  return;
162  }
163  if (clipR >= split)
164  {
165  // keep looping on right half
166  gridBox.lo[axis] = split;
167  prevClip = clipR;
168  wasLeft = false;
169  continue;
170  }
171  gridBox.lo[axis] = split;
172  prevClip = G3D::fnan();
173  }
174  else
175  {
176  // we are actually splitting stuff
177  if (prevAxis != -1 && !std::isnan(prevClip))
178  {
179  // second time through - lets create the previous split
180  // since it produced empty space
181  int nextIndex = tempTree.size();
182  // allocate child node
183  tempTree.push_back(0);
184  tempTree.push_back(0);
185  tempTree.push_back(0);
186  if (wasLeft)
187  {
188  // create a node with a left child
189  // write leaf node
190  stats.updateInner();
191  tempTree[nodeIndex + 0] = (prevAxis << 30) | nextIndex;
192  tempTree[nodeIndex + 1] = floatToRawIntBits(prevClip);
193  tempTree[nodeIndex + 2] = floatToRawIntBits(G3D::inf());
194  }
195  else
196  {
197  // create a node with a right child
198  // write leaf node
199  stats.updateInner();
200  tempTree[nodeIndex + 0] = (prevAxis << 30) | (nextIndex - 3);
201  tempTree[nodeIndex + 1] = floatToRawIntBits(-G3D::inf());
202  tempTree[nodeIndex + 2] = floatToRawIntBits(prevClip);
203  }
204  // count stats for the unused leaf
205  ++depth;
206  stats.updateLeaf(depth, 0);
207  // now we keep going as we are, with a new nodeIndex:
208  nodeIndex = nextIndex;
209  }
210  break;
211  }
212  }
213  // compute index of child nodes
214  int nextIndex = tempTree.size();
215  // allocate left node
216  int nl = right - left + 1;
217  int nr = rightOrig - (right + 1) + 1;
218  if (nl > 0)
219  {
220  tempTree.push_back(0);
221  tempTree.push_back(0);
222  tempTree.push_back(0);
223  }
224  else
225  nextIndex -= 3;
226  // allocate right node
227  if (nr > 0)
228  {
229  tempTree.push_back(0);
230  tempTree.push_back(0);
231  tempTree.push_back(0);
232  }
233  // write leaf node
234  stats.updateInner();
235  tempTree[nodeIndex + 0] = (axis << 30) | nextIndex;
236  tempTree[nodeIndex + 1] = floatToRawIntBits(clipL);
237  tempTree[nodeIndex + 2] = floatToRawIntBits(clipR);
238  // prepare L/R child boxes
239  AABound gridBoxL(gridBox), gridBoxR(gridBox);
240  AABound nodeBoxL(nodeBox), nodeBoxR(nodeBox);
241  gridBoxL.hi[axis] = gridBoxR.lo[axis] = split;
242  nodeBoxL.hi[axis] = clipL;
243  nodeBoxR.lo[axis] = clipR;
244  // recurse
245  if (nl > 0)
246  subdivide(left, right, tempTree, dat, gridBoxL, nodeBoxL, nextIndex, depth + 1, stats);
247  else
248  stats.updateLeaf(depth + 1, 0);
249  if (nr > 0)
250  subdivide(right + 1, rightOrig, tempTree, dat, gridBoxR, nodeBoxR, nextIndex + 3, depth + 1, stats);
251  else
252  stats.updateLeaf(depth + 1, 0);
253 }
