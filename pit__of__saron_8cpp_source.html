170 void instance_pit_of_saron::SetData(uint32 uiType, uint32 uiData)
171 {
172  switch (uiType)
173  {
174  case TYPE_GARFROST:
175  if (uiData == DONE && m_auiEncounter[TYPE_KRICK] == DONE)
176  DoUseDoorOrButton(GO_ICEWALL);
177  if (uiData == DONE || uiData == FAIL)
178  {
179  for (const auto& guid : m_lSaroniteRockGuidList)
180  if (GameObject* pRock = instance->GetGameObject(guid))
181  pRock->SetLootState(GO_JUST_DEACTIVATED);
182 
183  m_lSaroniteRockGuidList.clear();
184  }
185  if (uiData == IN_PROGRESS)
186  SetSpecialAchievementCriteria(TYPE_ACHIEV_DOESNT_GO_ELEVEN, true);
187  m_auiEncounter[uiType] = uiData;
188  break;
189  case TYPE_KRICK:
190  if (uiData == DONE && m_auiEncounter[TYPE_GARFROST] == DONE)
191  DoUseDoorOrButton(GO_ICEWALL);
192  m_auiEncounter[uiType] = uiData;
193  break;
194  case TYPE_TYRANNUS:
195  if (uiData == DONE)
196  {
197  // Force resume WP movement after combat
198  for (const auto guid : m_lEndingCreaturesGuidList)
199  {
200  if (Creature* creature = instance->GetCreature(guid))
201  {
202  creature->AI()->EnterEvadeMode();
203  creature->GetMotionMaster()->MoveWaypoint(creature->GetMotionMaster()->GetPathId(), 0, 1000);
204  creature->GetMotionMaster()->SetNextWaypoint(4);
205  }
206  }
207 
208  m_uiSummonUndeadTimer = 0;
209  }
210  else if (uiData == SPECIAL)
211  DoStartTyrannusEvent();
212  else if (uiData == IN_PROGRESS)
213  m_uiSummonUndeadTimer = 1000;
214  m_auiEncounter[uiType] = uiData;
215  break;
216  case TYPE_AMBUSH:
217  if (uiData == DONE)
218  {
219  // Complete tunnel achievement
220  if (Creature* pTyrannus = GetSingleCreatureFromStorage(NPC_TYRANNUS))
221  pTyrannus->CastSpell(pTyrannus, SPELL_ACHIEVEMENT_CHECK, TRIGGERED_OLD_TRIGGERED);
222 
223  // Despawn the Tyrannus at the entrance
224  if (Creature* creature = GetSingleCreatureFromStorage(NPC_TYRANNUS_INTRO))
225  creature->ForcedDespawn();
226 
227  // Despawn Jaina / Sylvanas
228  if (Creature* creature = GetSingleCreatureFromStorage(GetPlayerTeam() == HORDE ? NPC_SYLVANAS_PART1 : NPC_JAINA_PART1))
229  creature->ForcedDespawn();
230 
231  m_uiIciclesTimer = 0;
232  m_uiEyeLichKingTimer = 0;
233  }
234  else if (uiData == IN_PROGRESS)
235  DoStartAmbushEvent();
236  m_auiEncounter[uiType] = uiData;
237  break;
238  default:
239  return;
240  }
241 
242  if (uiData == DONE)
243  {
244  OUT_SAVE_INST_DATA;
245 
246  std::ostringstream saveStream;
247  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
248 
249  m_strInstData = saveStream.str();
250 
251  SaveToDB();
252  OUT_SAVE_INST_DATA_COMPLETE;
253  }

522  }
523

536  }
537  m_uiIciclesTimer = urand(3000, 5000);
538  }
539  else
540  m_uiIciclesTimer -= uiDiff;
541  }
542 
543  if (m_uiEyeLichKingTimer)
544  {
545  if (m_uiEyeLichKingTimer <= uiDiff)
546  {
547  // the eye will randomly face the players
548  if (Creature* pEye = GetSingleCreatureFromStorage(NPC_EYE_LICH_KING))

550  Player* pPlayer = GetPlayerInMap();
551  if (!pPlayer)

554  m_uiEyeLichKingTimer = 0;
555  }
556 
557  pEye->SetFacingToObject(pPlayer);
558 
559  m_uiEyeLichKingTimer = urand(10000, 30000);
560  }
561  }
562  else
563  m_uiEyeLichKingTimer -= uiDiff;
564  }
565 
566  if (m_uiSummonUndeadTimer)
567  {
568  if (m_uiSummonUndeadTimer <= uiDiff)
569  {
570  DoSpawnTyrannusUndead();
571  m_uiSummonUndeadTimer = urand(15000, 20000);
572  }
573  else
574  m_uiSummonUndeadTimer -= uiDiff;
575  }
