278 void ReputationMgr::SetReputation(FactionEntry const* factionEntry, int32 standing, bool incremental)
279 {
280  if (!factionEntry)
281  return;
282 
283  bool anyRankIncreased = false;
284  // if spillover definition exists in DB, override DBC
285  if (const RepSpilloverTemplate* repTemplate = sObjectMgr.GetRepSpilloverTemplate(factionEntry->ID))
286  {
287  for (uint32 i = 0; i < MAX_SPILLOVER_FACTIONS; ++i)
288  {
289  if (repTemplate->faction[i])
290  {
291  if (m_player->GetReputationRank(repTemplate->faction[i]) <= ReputationRank(repTemplate->faction_rank[i]))
292  {
293  // bonuses are already given, so just modify standing by rate
294  int32 spilloverRep = standing * repTemplate->faction_rate[i];
295  if (SetOneFactionReputation(sFactionStore.LookupEntry(repTemplate->faction[i]), spilloverRep, incremental))
296  anyRankIncreased = true;
297  }
298  }
299  }
300  }
301  else
302  {
303  float spillOverRepOut = standing;
304  // check for sub-factions that receive spillover
305  SimpleFactionsList const* flist = GetFactionTeamList(factionEntry->ID);
306  // if has no sub-factions, check for factions with same parent
307  if (!flist && factionEntry->team && factionEntry->spilloverRateOut != 0.0f)
308  {
309  spillOverRepOut *= factionEntry->spilloverRateOut;
310  if (FactionEntry const* parent = sFactionStore.LookupEntry(factionEntry->team))
311  {
312  FactionStateList::iterator parentState = m_factions.find(parent->reputationListID);
313  // some team factions have own reputation standing, in this case do not spill to other sub-factions
314  if (parentState != m_factions.end() && (parentState->second.Flags & FACTION_FLAG_TEAM_REPUTATION))
315  {
316  if (SetOneFactionReputation(parent, int32(spillOverRepOut), incremental))
317  anyRankIncreased = true;
318  }
319  else // spill to "sister" factions
320  {
321  flist = GetFactionTeamList(factionEntry->team);
322  }
323  }
324  }
325  if (flist)
326  {
327  // Spillover to affiliated factions
328  for (uint32 itr : *flist)
329  {
330  if (FactionEntry const* factionEntryCalc = sFactionStore.LookupEntry(itr))
331  {
332  if (factionEntryCalc == factionEntry || GetRank(factionEntryCalc) > ReputationRank(factionEntryCalc->spilloverMaxRankIn))
333  continue;
334 
335  int32 spilloverRep = int32(spillOverRepOut * factionEntryCalc->spilloverRateIn);
336  if (spilloverRep != 0 || !incremental)
337  if (SetOneFactionReputation(factionEntryCalc, spilloverRep, incremental))
338  anyRankIncreased = true;
339  }
340  }
341  }
342  }
343  // spillover done, update faction itself
344  FactionStateList::iterator faction = m_factions.find(RepListID(factionEntry->reputationListID));
345  if (faction != m_factions.end())
346  {
347  if (SetOneFactionReputation(factionEntry, standing, incremental))
348  anyRankIncreased = true;
349 
350  // only this faction gets reported to client, even if it has no own visible standing
351  SendState(&faction->second, anyRankIncreased);
352  }
353 }

355 bool ReputationMgr::SetOneFactionReputation(FactionEntry const* factionEntry, int32 standing, bool incremental)
356 {
357  if (!factionEntry)
358  return false;
359 
360  FactionStateList::iterator itr = m_factions.find(RepListID(factionEntry->reputationListID));
361  if (itr != m_factions.end())
362  {
363  FactionState& faction = itr->second;
364  int32 BaseRep = GetBaseReputation(factionEntry);
365 
366  if (incremental)
367  standing += faction.Standing + BaseRep;
368 
369  if (standing > Reputation_Cap)
370  standing = Reputation_Cap;
371  else if (standing < Reputation_Bottom)
372  standing = Reputation_Bottom;
373 
374  ReputationRank rankOld = ReputationToRank(faction.Standing + BaseRep);
375  ReputationRank rankNew = ReputationToRank(standing);
376 
377  faction.Standing = standing - BaseRep;
378  faction.needSend = true;
379  faction.needSave = true;
380 
381  SetVisible(&faction);
382 
383  if (rankNew != rankOld)
384  {
385  // Server alters "At war" flag on two occasions:
386  // * When reputation dips to "Hostile": forced tick and now locked for manual changes
387  if (rankNew < REP_UNFRIENDLY && rankNew < rankOld && rankOld > REP_HOSTILE)
388  SetAtWar(&itr->second, true);
389  // * When reputation improves to "Neutral": untick by id, can be manually overriden for eligible factions
390  else if (rankNew > REP_UNFRIENDLY && rankNew > rankOld && rankOld < REP_NEUTRAL)
391  SetAtWar(RepListID(factionEntry->reputationListID), false);
392  }
393 
394  UpdateRankCounters(rankOld, rankNew);
395 
396  m_player->ReputationChanged(factionEntry);
397 
398  AchievementMgr& achievementManager = m_player->GetAchievementMgr();
399  achievementManager.UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_KNOWN_FACTIONS, factionEntry->ID);
400  achievementManager.UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_REPUTATION, factionEntry->ID);
401  achievementManager.UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_EXALTED_REPUTATION, factionEntry->ID);
402  achievementManager.UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_REVERED_REPUTATION, factionEntry->ID);
403  achievementManager.UpdateAchievementCriteria(ACHIEVEMENT_CRITERIA_TYPE_GAIN_HONORED_REPUTATION, factionEntry->ID);
404 
405  if (rankNew > rankOld)
406  return true;
407  }
408  return false;
409 }

548

551

554 
555  if (dbFactionFlags & FACTION_FLAG_AT_WAR) // DB at war
556  SetAtWar(faction, true); // have internal checks for FACTION_FLAG_PEACE_FORCED
557  else // DB not at war
558  {
559  // allow remove if visible (and then not FACTION_FLAG_INVISIBLE_FORCED or FACTION_FLAG_HIDDEN)
560  if (faction->Flags & FACTION_FLAG_VISIBLE)
561  SetAtWar(faction, false); // have internal checks for FACTION_FLAG_PEACE_FORCED
562  }
563 
564  // set atWar for hostile
565  ForcedReactions::const_iterator forceItr = m_forcedReactions.find(factionEntry->ID);
566  if (forceItr != m_forcedReactions.end())
567  {
568  if (forceItr->second <= REP_HOSTILE)
569  SetAtWar(faction, true);
570  }
571  else if (GetRank(factionEntry) <= REP_HOSTILE)
572  SetAtWar(faction, true);
573 
574  // reset changed flag if values similar to saved in DB
575  if (faction->Flags == dbFactionFlags)
576  {
577  faction->needSend = false;
578  faction->needSave = false;
579  }
580  }
581  }
582  while (result->NextRow());
583 
584  delete result;
585  }
