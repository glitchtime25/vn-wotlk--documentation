114 bool ChatHandler::GetDeletedCharacterInfoList(DeletedInfoList& foundList, std::string searchString)
115 {
116  QueryResult* resultChar;
117  if (!searchString.empty())
118  {
119  // search by GUID
120  if (isNumeric(searchString))
121  resultChar = CharacterDatabase.PQuery("SELECT guid, deleteInfos_Name, deleteInfos_Account, deleteDate FROM characters WHERE deleteDate IS NOT NULL AND guid = %u", uint32(atoi(searchString.c_str())));
122  // search by name
123  else
124  {
125  if (!normalizePlayerName(searchString))
126  return false;
127 
128  resultChar = CharacterDatabase.PQuery("SELECT guid, deleteInfos_Name, deleteInfos_Account, deleteDate FROM characters WHERE deleteDate IS NOT NULL AND deleteInfos_Name " _LIKE_ " " _CONCAT3_("'%%'", "'%s'", "'%%'"), searchString.c_str());
129  }
130  }
131  else
132  resultChar = CharacterDatabase.Query("SELECT guid, deleteInfos_Name, deleteInfos_Account, deleteDate FROM characters WHERE deleteDate IS NOT NULL");
133 
134  if (resultChar)
135  {
136  if (resultChar->GetRowCount() > 100)
137  {
138  PSendSysMessage("Too many results %u. Narrow it down.", (uint32)resultChar->GetRowCount());
139  SetSentErrorMessage(true);
140  delete resultChar;
141  return false;
142  }
143 
144  do
145  {
146  Field* fields = resultChar->Fetch();
147 
148  DeletedInfo info;
149 
150  info.lowguid = fields[0].GetUInt32();
151  info.name = fields[1].GetCppString();
152  info.accountId = fields[2].GetUInt32();
153 
154  // account name will be empty for nonexistent account
155  sAccountMgr.GetName(info.accountId, info.accountName);
156 
157  info.deleteDate = time_t(fields[3].GetUInt64());
158 
159  foundList.push_back(info);
160  }
161  while (resultChar->NextRow());
162 
163  delete resultChar;
164  }
165

323  DeletedInfoList foundList;

334  HandleCharacterDeletedListHelper(foundList);
335 
336  if (newCharName.empty())
337  {
338  // Drop nonexistent account cases
339  for (auto& itr : foundList)
340  HandleCharacterDeletedRestoreHelper(itr);

342  else if (foundList.size() == 1 && normalizePlayerName(newCharName))

346  // update name

348 
349  // if new account provided update deleted info

352  delInfo.accountId = newAccount;
353  sAccountMgr.GetName(newAccount, delInfo.accountName);
354  }
355 
356  HandleCharacterDeletedRestoreHelper(delInfo);

609 void CliRunnable::run()
610 {
612  WorldDatabase.ThreadStart(); // let thread do safe mySQL requests
613 
614  char commandbuf[256];
615 
617  sLog.outString();
618 
619  if (sConfig.GetBoolDefault("BeepAtStart", true))
620  printf("\a"); // \a = Alert
621 
622  // print this here the first time
623  // later it will be printed after command queue updates
624  printf("mangos>");
625 
626 #ifdef __unix__
627  //Set stdin IO to nonblocking - prevent Server from hanging in shutdown process till enter is pressed
628  int fd = fileno(stdin);
629  int flags = fcntl(fd, F_GETFL, 0);
630  flags |= O_NONBLOCK;
631  fcntl(fd, F_SETFL, flags);
632 #endif
633 
635  while (!World::IsStopped())
636  {
637  fflush(stdout);
638 #ifdef __unix__
639  while (!kb_hit_return() && !World::IsStopped())
640  {
641  // With this, we limit CLI to 10commands/second
642  std::this_thread::sleep_for(std::chrono::nanoseconds(100000));
643  // Check for world stoppage after each sleep interval
644  if (World::IsStopped())
645  break;
646  }
647 #endif
648  char* command_str = fgets(commandbuf, sizeof(commandbuf), stdin);
649  if (command_str != nullptr)
650  {
651  for (int x = 0; command_str[x]; ++x)
652  if (command_str[x] == '\r' || command_str[x] == '\n')
653  {
654  command_str[x] = 0;
655  break;
656  }
657 
658  if (!*command_str)
659  {
660  printf("mangos>");
661  continue;
662  }
663 
664  std::string command;
665  if (!consoleToUtf8(command_str, command)) // convert from console encoding to utf8
666  {
667  printf("mangos>");
668  continue;
669  }
670 
671  sWorld.QueueCliCommand(new CliCommandHolder(0, SEC_CONSOLE, command.c_str(), &utf8print, &commandFinished));
672  }
673  else if (feof(stdin))
674  {
675  World::StopNow(SHUTDOWN_EXIT_CODE);
676  }
677  }
678 
680  WorldDatabase.ThreadEnd(); // free mySQL thread resources
681 }
