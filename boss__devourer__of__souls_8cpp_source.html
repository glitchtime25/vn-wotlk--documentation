190  void UpdateAI(const uint32 uiDiff) override
191  {
192  if (!m_creature->IsInCombat())
193  return;
194 
195  // Ending a phase
196  if (m_uiEndPhaseTimer)
197  {
198  if (m_uiEndPhaseTimer <= uiDiff)
199  {
200  DoCastSpellIfCan(m_creature, SPELL_DRUID_MORPH, CAST_INTERRUPT_PREVIOUS);
201  // Sumberge visual might be cast as effect of the above spell.
202  DoCastSpellIfCan(m_creature, SPELL_SUBMERGE_VISUAL, CAST_TRIGGERED);
203  m_uiEndPhaseTimer = 0;
204 
205  m_uiFace = FACE_NORMAL;
206  }
207  else
208  m_uiEndPhaseTimer -= uiDiff;
209  }
210 
211  // No additional spells, no target selection for wailing souls
212  if (m_uiFace == FACE_WAILING)
213  {
214  // Some special handling in case of starting phase of wailings
215  if (Unit* pTarget = m_creature->GetTarget())
216  m_creature->SetFacingTo(m_creature->GetAngle(pTarget));
217 
218  if (m_creature->getThreatManager().isThreatListEmpty() || !m_creature->getThreatManager().getHostileTarget())
219  m_creature->SelectHostileTarget(); // Most likely must evade, use additional checks in case evading would be prevented
220  return;
221  }
222 
223  // Update Target and do Combat Spells
224  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
225  return;
226 
227  // No additional abilities while unleashing
228  if (m_uiFace == FACE_UNLEASHING)
229  return;
230 
231  // Phantom Blast
232  if (m_uiPhantomBlastTimer < uiDiff)
233  {
234  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
235  {
236  if (DoCastSpellIfCan(pTarget, m_bIsRegularMode ? SPELL_PHANTOM_BLAST : SPELL_PHANTOM_BLAST_H) == CAST_OK)
237  m_uiPhantomBlastTimer = urand(5000, 10000); // TODO
238  }
239  }
240  else
241  m_uiPhantomBlastTimer -= uiDiff;
242 
243  // Jump towards random enemy
244  if (m_uiWellTimer < uiDiff)
245  {
246  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
247  {
248  if (DoCastSpellIfCan(pTarget, SPELL_WELL_OF_SOULS) == CAST_OK)
249  m_uiWellTimer = urand(15000, 25000); // TODO
250  }
251  }
252  else
253  m_uiWellTimer -= uiDiff;
254 
255  // DMG reflection
256  if (m_uiMirrorTimer < uiDiff)
257  {
258  if (!m_creature->IsNonMeleeSpellCasted(true))
259  {
260  DoCastSpellIfCan(m_creature, SPELL_MIRRORED_SOUL, CAST_TRIGGERED);
261  m_uiMirrorTimer = urand(25000, 35000); // TODO
262  DoScriptText(EMOTE_MIRRORED_SOUL, m_creature);
263  }
264  }
265  else
266  m_uiMirrorTimer -= uiDiff;
267 
268  // Spawning of Adds
269  if (m_uiUnleashTimer < uiDiff)
270  {
271  if (DoCastSpellIfCan(m_creature, SPELL_UNLEASHED_SOULS) == CAST_OK && DoCastSpellIfCan(m_creature, SPELL_DRUID_MORPH_0_5, CAST_TRIGGERED) == CAST_OK)
272  {
273  DoScriptText(EMOTE_UNLEASH_SOULS, m_creature);
274  m_uiUnleashTimer = urand(30000, 60000); // TODO
275 
276  m_uiFace = FACE_UNLEASHING;
277  DoScriptText(aTexts[4][m_uiFace], m_creature);
278  m_uiEndPhaseTimer = 4500; // 5000 (Duration of Unleasing) + 850(Cast time for unleashing) - 1000(duration of whirl) - a bit air
279  }
280  }
281  else
282  m_uiUnleashTimer -= uiDiff;
283 
284  if (m_uiWailingTimer < uiDiff)
285  {
286  if (DoCastSpellIfCan(m_creature, SPELL_WALIING_SOULS_TARGETS) == CAST_OK)
287  {
288  DoCastSpellIfCan(m_creature, SPELL_DRUID_MORPH_1_5, CAST_TRIGGERED);
289  DoScriptText(EMOTE_WAILING_SOULS, m_creature);
290  m_uiFace = FACE_WAILING;
291  m_uiEndPhaseTimer = 12500; // 100000 (Duration of Wailing) + 3000(casting time) - 1000 (duration of whirl) + 500 (some add. time)
292  m_uiWailingTimer = urand(25000, 35000); // TODO
293  }
294  }
295  else
296  m_uiWailingTimer -= uiDiff;
297 
298  DoMeleeAttackIfReady();
299  }
