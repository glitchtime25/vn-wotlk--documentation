94 void CreatureAI::AttackPassengersIfCan(Unit* who)
95 {
96  if (!who->IsVehicle())
97  return;
98 
99  // set the victim's passengers in combat (if possible)
100  for (uint8 i = 0; i < MAX_VEHICLE_SEAT; ++i)
101  {
102  uint32 seatId = who->GetVehicleInfo()->GetVehicleEntry()->m_seatID[i];
103  if (!seatId)
104  continue;
105 
106  Unit* passenger = who->GetVehicleInfo()->GetPassenger(i);
107  if (!passenger)
108  continue;
109 
110  if (VehicleSeatEntry const* seatEntry = sVehicleSeatStore.LookupEntry(seatId))
111  {
112  // validate seat flags
113  if (seatEntry->m_flags & SEAT_FLAG_NOT_SELECTABLE || seatEntry->m_flags & SEAT_FLAG_HIDE_PASSENGER)
114  continue;
115 
116  m_creature->AddThreat(passenger);
117  m_creature->SetInCombatWith(passenger);
118  passenger->SetInCombatWith(m_creature);
119  }
120  }
121 }

181 bool CreatureAI::DoRetreat()
182 {
183  Unit* victim = m_creature->GetVictim();
184  if (!victim)
185  return false;
186 
187  float radius = sWorld.getConfig(CONFIG_FLOAT_CREATURE_FAMILY_FLEE_ASSISTANCE_RADIUS);
188  if (radius <= 0)
189  return false;
190 
191  Creature* ally = nullptr;
192 
193  MaNGOS::NearestAssistCreatureInCreatureRangeCheck check(m_creature, victim, radius);
194  MaNGOS::CreatureLastSearcher<MaNGOS::NearestAssistCreatureInCreatureRangeCheck> searcher(ally, check);
195  Cell::VisitGridObjects(m_creature, searcher, radius);
196 
197  // Check if an ally to call for was found
198  if (!ally)
199  return false;
200 
201  uint32 delay = sWorld.getConfig(CONFIG_UINT32_CREATURE_FAMILY_ASSISTANCE_DELAY);
202 
203  Position pos;
204  ally->GetFirstCollisionPosition(pos, ally->GetCombatReach(), ally->GetAngle(m_creature));
205  m_creature->GetMotionMaster()->MoveRetreat(pos.x, pos.y, pos.z, ally->GetAngle(victim), delay);
206 
207  SetAIOrder(ORDER_RETREATING);
208  SetCombatScriptStatus(true);
209  return true;
210 }
