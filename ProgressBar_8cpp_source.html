71 void BarGoLink::step()
72 {
73  if (!m_showOutput)
74  return;
75 
76  size_t i;
77 
78  if (num_rec == 0) return;
79  ++rec_no;
80  size_t n = rec_no * indic_len / num_rec;
81  if (n != rec_pos)
82  {
83 #ifdef _WIN32
84  printf("\r\x3D");
85 #else
86  printf("\r[");
87 #endif
88  for (i = 0; i < n; ++i) printf(full);
89  for (; i < indic_len; ++i) printf(empty);
90  int percent = int((double(n) / double(indic_len)) * 100);
91 #ifdef _WIN32
92  printf("\x3D %i%% \r\x3D", percent);
93 #else
94  printf("] %i%% \r[", percent);
95 #endif
96  fflush(stdout);
97 
98  rec_pos = n;
99  }
100 }
