83 void BattleGroundSA::Update(uint32 diff)
84 {
85  BattleGround::Update(diff);
86 
87  if (m_boatStartTimer)
88  {
89  if (m_boatStartTimer <= diff)
90  {
91  // Set the boats in motion
92  for (const auto& guid : m_transportShipGuids[GetAttacker()])
93  {
94  if (GameObject* pShip = GetBgMap()->GetGameObject(guid))
95  pShip->SetGoState(GO_STATE_ACTIVE);
96  }
97 
98  // ToDo: stop the boats when they reach the destination. Should be handled by event id 22095 and 18829
99  m_boatStartTimer = 0;
100  }
101  else
102  m_boatStartTimer -= diff;
103  }
104 
105  if (GetStatus() != STATUS_IN_PROGRESS)
106  return;
107 
108  // banner change timers; swap the flags between teams
109  for (uint8 i = 0; i < BG_SA_MAX_GRAVEYARDS; ++i)
110  {
111  if (m_strandGraveyard[i].changeTimer)
112  {
113  if (m_strandGraveyard[i].changeTimer <= diff)
114  {
115  ChangeBannerState(i);
116  m_strandGraveyard[i].changeTimer = 0;
117  }
118  else
119  m_strandGraveyard[i].changeTimer -= diff;
120  }
121  }
122 
123  // battle timer
124  if (m_battleRoundTimer)
125  {
126  if (m_battleRoundTimer < diff)
127  {
128  switch (m_battleStage)
129  {
130  case BG_SA_STAGE_ROUND_1:
131  // eject all vehicle passengers and despawn vehicles immediately
132  for (uint32 dbGuid : m_demolishersGuids)
133  {
134  if (Creature* demolisher = GetBgMap()->GetCreature(dbGuid))
135  {
136  demolisher->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
137  demolisher->ForcedDespawn();
138  }
139  }
140  for (ObjectGuid guid : m_tempDemolishersGuids)
141  {
142  if (Creature* demolisher = GetBgMap()->GetCreature(guid))
143  {
144  demolisher->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
145  demolisher->ForcedDespawn();
146  }
147  }
148  for (uint32 dbGuid : m_cannonsGuids)
149  {
150  if (Creature* cannon = GetBgMap()->GetCreature(dbGuid))
151  {
152  cannon->RemoveSpellsCausingAura(SPELL_AURA_CONTROL_VEHICLE);
153  cannon->ForcedDespawn();
154  }
155  }
156  // send warning
157  SendBattlegroundWarning(LANG_BG_SA_ROUND_FINISHED);
158  GetBgMap()->GetVariableManager().SetVariable(BG_SA_STATE_ENABLE_TIMER, WORLD_STATE_REMOVE);
159 
160  // cast end of round spell; spell will also force resurrect dead players
161  CastSpellOnTeam(BG_SA_SPELL_END_OF_ROUND, ALLIANCE);
162  CastSpellOnTeam(BG_SA_SPELL_END_OF_ROUND, HORDE);
163 
164  // award bonus honor
165  AwardBonusHonor();
166 
167  m_battleRoundTimer = 5000;
168  break;
169  case BG_SA_STAGE_RESET:
170  {
171  // invert the defender
172  SetDefender(m_defendingTeamIdx == TEAM_INDEX_ALLIANCE ? TEAM_INDEX_HORDE : TEAM_INDEX_ALLIANCE);
173 
174  // reset
175  SetupBattleground(false);
176 
177  // setup player spells
178  for (auto& m_player : m_players)
179  {
180  if (!m_player.first)
181  continue;
182 
183  Player* player = sObjectMgr.GetPlayer(m_player.first);
184  if (player)
185  {
186  if (GetAttacker() == TEAM_INDEX_ALLIANCE)
187  player->CastSpell(nullptr, BG_SA_SPELL_HORDE_CONTROL_PHASE_SHIFT, TRIGGERED_OLD_TRIGGERED);
188  else
189  player->CastSpell(nullptr, BG_SA_SPELL_ALLIANCE_CONTROL_PHASE_SHIFT, TRIGGERED_OLD_TRIGGERED);
190 
191  // teleport to the right spot and set phase
192  TeleportPlayerToStartArea(player);
193  }
194  }
195 
196  m_battleRoundTimer = 2000;
197  m_boatStartTimer = 2000;
198  break;
199  }
200  case BG_SA_STAGE_SECOND_ROUND_1:
201  SendMessageToAll(LANG_BG_SA_ROUND_START_ONE_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
202 
203  // cast preparation again before the 2nd round
204  CastSpellOnTeam(SPELL_PREPARATION, ALLIANCE);
205  CastSpellOnTeam(SPELL_PREPARATION, HORDE);
206 
207  // make sure that the cannons have the right faction set; for some reason this isn't always correctly set
208  for (uint32 dbGuid : m_cannonsGuids)
209  {
210  if (Creature* cannon = GetBgMap()->GetCreature(dbGuid))
211  cannon->SetFactionTemporary(sotaTeamFactions[m_defendingTeamIdx], TEMPFACTION_NONE);
212  }
213 
214  m_battleRoundTimer = 30000;
215  break;
216  case BG_SA_STAGE_SECOND_ROUND_2:
217  SendMessageToAll(LANG_BG_SA_ROUND_START_HALF_MINUTE, CHAT_MSG_BG_SYSTEM_NEUTRAL);
218  m_battleRoundTimer = 30000;
219  break;
220  case BG_SA_STAGE_SECOND_ROUND_3:
221 
222  // remove preparation aura
223  for (auto& m_player : m_players)
224  if (Player* player = sObjectMgr.GetPlayer(m_player.first))
225  player->RemoveAurasDueToSpell(SPELL_PREPARATION);
226 
227  SendBattlegroundWarning(LANG_BG_SA_BEGIN);
228  GetBgMap()->GetVariableManager().SetVariable(BG_SA_STATE_ENABLE_TIMER, WORLD_STATE_ADD);
229  EnableDemolishers();
230  StartTimedAchievement(ACHIEVEMENT_CRITERIA_TYPE_BE_SPELL_TARGET, GetAttacker() == TEAM_INDEX_ALLIANCE ? BG_SA_ACHIEV_START_ID_STORM_BEACH_ALLY : BG_SA_ACHIEV_START_ID_STORM_BEACH_HORDE);
231  m_battleRoundTimer = BG_SA_TIMER_ROUND_LENGTH;
232  break;
233  case BG_SA_STAGE_ROUND_2:
234  ProcessBattlegroundWinner();
235  m_battleRoundTimer = 0;
236  break;
237  }
238 
239  ++m_battleStage;
240  }
241  else
242  {
243  // update timer - if BattlegroundMgr update timer interval needs to be lowered replace this line with the commented-out ones below
244  if (m_battleStage == BG_SA_STAGE_ROUND_1 || m_battleStage == BG_SA_STAGE_ROUND_2)
245  UpdateTimerWorldState();
246 
247  /*if (m_zoneUpdateTimer < diff)
248  {
249  // update timer
250  UpdateTimerWorldState();
251  m_zoneUpdateTimer = BG_SA_TIMER_UPDATE_TIME;
252  }
253  else
254  m_zoneUpdateTimer -= diff;*/
255 
256  m_battleRoundTimer -= diff;
257  }
258  }

459 // process the gate and relic events
460 bool BattleGroundSA::HandleEvent(uint32 eventId, Object* source, Object* target)
461 {
462  if (!source->IsGameObject())
463  return false;
464 
465  GameObject* go = static_cast<GameObject*>(source);
466 
467  // handle ships stop event
468  if (eventId == BG_SA_EVENT_ID_SHIP_PAUSE_1 || eventId == BG_SA_EVENT_ID_SHIP_PAUSE_2)
469  {
470  DEBUG_LOG("BattleGroundSA: Ship with entry %u has reached the docks.", go->GetEntry());
471 
472  // ToDo: implement ship stop logic once transports can handle this properly
473  return false;
474  }
475 
476  // handle round end by relic click
477  if (eventId == BG_SA_EVENT_ID_RELIC)
478  {
479  DEBUG_LOG("BattleGroundSA: Relic with entry %u has been activated", go->GetEntry());
480 
481  // update score count and keep track of the time
482  if (go->GetEntry() == BG_SA_GO_TITAN_RELIC_ALLIANCE)
483  {
484  ++m_scoreCount[TEAM_INDEX_ALLIANCE];
485  m_winTime[TEAM_INDEX_ALLIANCE] = m_battleRoundTimer;
486  CastSpellOnTeam(BG_SA_SPELL_ACHIEV_STORM_BEACH, ALLIANCE);
487  }
488  else if (go->GetEntry() == BG_SA_GO_TITAN_RELIC_HORDE)
489  {
490  ++m_scoreCount[TEAM_INDEX_HORDE];
491  m_winTime[TEAM_INDEX_HORDE] = m_battleRoundTimer;
492  CastSpellOnTeam(BG_SA_SPELL_ACHIEV_STORM_BEACH, HORDE);
493  }
494 
495  // process winner
496  if (m_battleStage == BG_SA_STAGE_ROUND_2)
497  ProcessBattlegroundWinner();
498  // prepare second round
499  else if (m_battleStage == BG_SA_STAGE_ROUND_1)
500  m_battleRoundTimer = 500;
501 
502  m_defenseLineCaptured[3] = true;
503  return false;
504  }
505 
506  // special event for the ancient shrine door
507  if (go->GetEntry() == BG_SA_GO_GATE_ANCIENT_SHRINE && eventId == BG_SA_EVENT_SHRINE_DOOR_DESTROY)
508  {
509  DEBUG_LOG("BattleGroundSA: Ancient Shrine id %u has been activated by event %u", go->GetEntry(), eventId);
510 
511  // Remove the no interact flag when door is destroyed
512  if (GameObject* relic = go->GetMap()->GetGameObject(m_relicGuid[GetAttacker()]))
513  relic->RemoveFlag(GAMEOBJECT_FLAGS, GO_FLAG_NO_INTERACT);
514  }
515 
516  // handle world state updates and warnings
517  for (const auto& i : sotaObjectData)
518  {
519  if (go->GetEntry() == i.goEntry)
520  {
521  if (eventId == i.eventDamaged)
522  {
523  DEBUG_LOG("BattleGroundSA: Gate with entry %u has been damaged", go->GetEntry());
524 
525  m_gateStateValue[i.index] = BG_SA_STATE_VALUE_GATE_DAMAGED;

690  winner = HORDE;
