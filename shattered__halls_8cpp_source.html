123 void instance_shattered_halls::SetData(uint32 uiType, uint32 uiData)
124 {
125  switch (uiType)
126  {
127  case TYPE_NETHEKURSE:
128  m_auiEncounter[uiType] = uiData;
129  if (uiData == DONE)
130  {
131  DoUseDoorOrButton(GO_NETHEKURSE_DOOR);
132  DoUseOpenableObject(GO_NETHEKURSE_ENTER_DOOR, true);
133  }
134  break;
135  case TYPE_OMROGG:
136  m_auiEncounter[uiType] = uiData;
137  break;
138  case TYPE_BLADEFIST:
139  m_auiEncounter[uiType] = uiData;
140  if (uiData == DONE)
141  {
142  // Make executioner attackable only after the final boss is dead
143  if (Creature* pExecutioner = GetSingleCreatureFromStorage(NPC_EXECUTIONER, true))
144  pExecutioner->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_SPAWNING);
145  }
146  break;
147  case TYPE_EXECUTION:
148  m_auiEncounter[uiType] = uiData;
149  if (uiData == IN_PROGRESS && !GetSingleCreatureFromStorage(NPC_EXECUTIONER, true))
150  {
151  if (Player* pPlayer = GetPlayerInMap())
152  {
153  // summon the 3 npcs for execution
154  for (uint8 i = 2; i < 5; ++i)
155  pPlayer->SummonCreature(m_uiTeam == ALLIANCE ? aSoldiersLocs[i].m_uiAllianceEntry : aSoldiersLocs[i].m_uiHordeEntry, aSoldiersLocs[i].m_fX, aSoldiersLocs[i].m_fY, aSoldiersLocs[i].m_fZ, aSoldiersLocs[i].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0);
156 
157  // Summon the executioner; Note: according to wowhead he shouldn't be targetable until Kargath encounter is finished
158  if (Creature* pExecutioner = pPlayer->SummonCreature(NPC_EXECUTIONER, afExecutionerLoc[0], afExecutionerLoc[1], afExecutionerLoc[2], afExecutionerLoc[3], TEMPSPAWN_DEAD_DESPAWN, 0, true))
159  pExecutioner->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE | UNIT_FLAG_SPAWNING);
160 
161  // cast the execution spell
162  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_1);
163  }
164  }
165  if (uiData == DONE)
166  {
167  // If the officer is already killed, then skip the quest completion
168  if (m_uiExecutionStage)
169  break;
170 
171  // Complete quest 9524 or 9525
172  if (Creature* pOfficer = GetSingleCreatureFromStorage(m_uiTeam == ALLIANCE ? NPC_OFFICER_ALLIANCE : NPC_OFFICER_HORDE))
173  {
174  Map::PlayerList const& lPlayers = instance->GetPlayers();
175 
176  for (const auto& lPlayer : lPlayers)
177  {
178  if (Player* pPlayer = lPlayer.getSource())
179  pPlayer->KilledMonsterCredit(pOfficer->GetEntry(), pOfficer->GetObjectGuid());
180  }
181  }
182  }
183  break;
184  case TYPE_GAUNTLET:
185  if (m_auiEncounter[uiType] == DONE) // do not allow any exploits
186  return;
187 
188  switch (uiData)
189  {
190  case FAIL: // Called on wipe/players left/Boss Evade
191  FailGauntlet();
192  break;
193  case DONE: // Called on boss kill
194  EndGauntlet();
195  break;
196  case SPECIAL: // Called on Boss Aggro
197  StopGauntlet();
198  break;
199  case IN_PROGRESS:
200  DoInitialGets();
201  break;
202  default:
203  break;
204  }
205 
206  m_auiEncounter[uiType] = uiData;
207  break;
208  }
209 
210  if (uiData == DONE)
211  {
212  OUT_SAVE_INST_DATA;
213 
214  std::ostringstream saveStream;
215 
216  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3];
217  m_strInstData = saveStream.str();
218 
219  SaveToDB();
220  OUT_SAVE_INST_DATA_COMPLETE;
221  }

415 void instance_shattered_halls::Update(uint32 uiDiff)
416 {
417  if (m_auiEncounter[TYPE_GAUNTLET] == IN_PROGRESS)
418  {
419  if (!GetPlayerInMap(true))
420  {
421  SetData(TYPE_GAUNTLET, FAIL);
422  }
423  else
424  {
425  for (auto& blaze : m_vBlazeTimers)
426  {
427  if (blaze.second <= uiDiff)
428  {
429  blaze.second = 2000;
430  if (GameObject* blazeGo = instance->GetGameObject(blaze.first))
431  blazeGo->CastSpell(nullptr, nullptr, SPELL_FLAMES, TRIGGERED_NONE, nullptr, nullptr, blazeGo->GetObjectGuid());
432  }
433  else
434  blaze.second -= uiDiff;
435  }
436  }
437 
438  if (m_gauntletStopped)
439  return;
440 
441  if (m_bInitialWavesSpawned)
442  {
443  if (m_bPorungDoneYelling)
444  {
445  if (m_uiWaveTimer < uiDiff) // Periodic waves
446  {
447  if (m_bZealotOneOrTwo) // second zealot, long delay
448  {
449  DoSummonSHZealot();
450  m_uiWaveTimer = WAVE_TIMER;
451  m_bZealotOneOrTwo = false;
452  }
453  else // first zealot, short delay
454  {
455  DoSummonSHZealot();
456  m_uiWaveTimer = DELAY_350_MILLI;
457  m_bZealotOneOrTwo = true;
458  }
459  }
460  else
461  m_uiWaveTimer -= uiDiff;
462 
463  if (m_uiShootFlamingArrowTimer_1 < uiDiff) // Left Archer
464  {
465  DoBeginArcherAttack(true);
466  m_uiShootFlamingArrowTimer_1 = ARCHER_SHOOT_DELAY + urand(0, 2000);
467  }
468  else
469  m_uiShootFlamingArrowTimer_1 -= uiDiff;
470 
471  if (m_uiShootFlamingArrowTimer_2 < uiDiff) // Right Archer
472  {
473  DoBeginArcherAttack(false);
474  m_uiShootFlamingArrowTimer_2 = ARCHER_SHOOT_DELAY + urand(0, 2000);
475  }
476  else
477  m_uiShootFlamingArrowTimer_2 -= uiDiff;
478  }
479  else // Not done yelling
480  {
481  if (m_uiPorungYellDelay < uiDiff)
482  {
483  switch (m_uiPorungYellNumber)
484  {
485  case 0:
486  DoScriptText(PORUNG_FORM_RANKS_YELL, m_porung);
487  m_uiPorungYellDelay = PORUNG_YELL_DELAY_1;
488  m_uiPorungYellNumber += 1;
489  break;
490  case 1:
491  DoScriptText(PORUNG_READY_YELL, m_porung);
492  DoBeginArcherAttack(true);
493  DoBeginArcherAttack(false);
494  m_uiPorungYellDelay = PORUNG_YELL_DELAY_2;
495  m_uiPorungYellNumber += 1;
496  break;
497  case 2:
498  DoScriptText(PORUNG_AIM_YELL, m_porung);
499  m_uiPorungYellDelay = PORUNG_YELL_DELAY_2;
500  m_uiPorungYellNumber += 1;
501  break;
502  case 3:
503  DoScriptText(PORUNG_FIRE_YELL, m_porung);
504  m_bPorungDoneYelling = true;
505  break;
506  }
507  }
508  else
509  m_uiPorungYellDelay -= uiDiff;
510  }
511  }
512  else // not done spawning first waves
513  {
514  if (m_uiInitialWaves_Delay < uiDiff)
515  {
516  switch (m_uiNumInitialWaves)
517  {
518  case 0:
519  case 1:
520  case 2:
521  DoSummonInitialWave();
522  m_uiNumInitialWaves++;
523  m_uiInitialWaves_Delay = DELAY_350_MILLI;
524  break;
525  case 3:
526  DoSummonInitialWave();
527  m_bInitialWavesSpawned = true;
528  break;
529  }
530  }
531  else
532  m_uiInitialWaves_Delay -= uiDiff;
533  }
534  }
535 
536  if (m_auiEncounter[TYPE_EXECUTION] != IN_PROGRESS)
537  return;
538 
539  if (m_uiExecutionTimer < uiDiff)
540  {
541  switch (m_uiExecutionStage)
542  {
543  case 0:
544  // Kill the officer
545  if (Creature* pSoldier = GetSingleCreatureFromStorage(m_uiTeam == ALLIANCE ? NPC_OFFICER_ALLIANCE : NPC_OFFICER_HORDE))
546  pSoldier->Suicide();
547 
548  // Make Kargath yell
549  DoOrSimulateScriptTextForThisInstance(m_uiTeam == ALLIANCE ? SAY_KARGATH_EXECUTE_ALLY : SAY_KARGATH_EXECUTE_HORDE, NPC_KARGATH_BLADEFIST);
550 
551  // Set timer for the next execution
552  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_2);
553  m_uiExecutionTimer = 10 * MINUTE * IN_MILLISECONDS;
554  break;
555  case 1:
556  if (Creature* pSoldier = GetSingleCreatureFromStorage(m_uiTeam == ALLIANCE ? NPC_SOLDIER_ALLIANCE_2 : NPC_SOLDIER_HORDE_2))
557  pSoldier->Suicide();
558 
559  DoCastGroupDebuff(SPELL_KARGATH_EXECUTIONER_3);
560  m_uiExecutionTimer = 15 * MINUTE * IN_MILLISECONDS;
561  break;
562  case 2:
563  if (Creature* pSoldier = GetSingleCreatureFromStorage(m_uiTeam == ALLIANCE ? NPC_SOLDIER_ALLIANCE_3 : NPC_SOLDIER_HORDE_3))
564  pSoldier->Suicide();
565

567  m_uiExecutionTimer = 0;
568  break;
569  }

572  }
