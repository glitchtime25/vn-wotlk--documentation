46 void OutdoorPvPTF::FillInitialWorldStates(WorldPacket& data, uint32& count)
47 {
48  FillInitialWorldState(data, count, m_zoneWorldState, WORLD_STATE_ADD);
49  if (m_zoneWorldState == WORLD_STATE_TF_TOWERS_CONTROLLED)
50  {
51  FillInitialWorldState(data, count, WORLD_STATE_TF_TOWER_COUNT_H, m_towersHorde);
52  FillInitialWorldState(data, count, WORLD_STATE_TF_TOWER_COUNT_A, m_towersAlliance);

143 // process the capture events
144 bool OutdoorPvPTF::HandleEvent(uint32 eventId, Object* source, Object* /*target*/)
145 {
146  if (!source->IsGameObject())
147  return false;
148 
149  GameObject* go = static_cast<GameObject*>(source);
150  for (uint8 i = 0; i < MAX_TF_TOWERS; ++i)
151  {
152  if (terokkarTowers[i] == go->GetEntry())
153  {
154  for (uint8 j = 0; j < 4; ++j)
155  {
156  if (terokkarTowerEvents[i][j].eventEntry == eventId)
157  {
158  // prevent processing if the owner did not change (happens if progress event is called after contest event)
159  if (terokkarTowerEvents[i][j].team != m_towerOwner[i])
160  {
161  sWorld.SendDefenseMessage(ZONE_ID_TEROKKAR_FOREST, terokkarTowerEvents[i][j].defenseMessage);
162 
163  return ProcessCaptureEvent(go, i, terokkarTowerEvents[i][j].team, terokkarTowerEvents[i][j].worldState);
164  }
165  // no need to iterate other events or towers
166  return false;
167  }
168  }
169  // no need to iterate other towers
170  return false;
171  }
172  }
173 
174  return false;
175 }

177 bool OutdoorPvPTF::ProcessCaptureEvent(GameObject* go, uint32 towerId, Team team, uint32 newWorldState)
178 {
179  if (team == ALLIANCE)
180  {
181  // update banner
182  SetBannerVisual(go, CAPTURE_ARTKIT_ALLIANCE, CAPTURE_ANIM_ALLIANCE);
183 
184  // update tower count
185  ++m_towersAlliance;
186 
187  // if all towers are captured then process event
188  if (m_towersAlliance == MAX_TF_TOWERS)
189  {
190  LockZone(go, towerId, team, newWorldState);
191  return true;
192  }
193 
194  // update tower count world state
195  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_A, m_towersAlliance);
196  }
197  else if (team == HORDE)
198  {
199  // update banner
200  SetBannerVisual(go, CAPTURE_ARTKIT_HORDE, CAPTURE_ANIM_HORDE);
201 
202  // update tower count
203  ++m_towersHorde;
204 
205  // if all towers are captured then process event
206  if (m_towersHorde == MAX_TF_TOWERS)
207  {
208  LockZone(go, towerId, team, newWorldState);
209  return true;
210  }
211 
212  // update tower count world state
213  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_H, m_towersHorde);
214  }
215  else
216  {
217  // update banner
218  SetBannerVisual(go, CAPTURE_ARTKIT_NEUTRAL, CAPTURE_ANIM_NEUTRAL);
219 
220  // update tower count
221  if (m_towerOwner[towerId] == ALLIANCE)
222  {
223  --m_towersAlliance;
224  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_A, m_towersAlliance);
225  }
226  else
227  {
228  --m_towersHorde;
229  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_H, m_towersHorde);
230  }
231  }
232 
233  // update tower state
234  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_REMOVE);
235  m_towerWorldState[towerId] = newWorldState;
236  SendUpdateWorldState(m_towerWorldState[towerId], WORLD_STATE_ADD);
237 
238  // update capture point owner
239  m_towerOwner[towerId] = team;
240 
241  // the are no DB exceptions in this case
242  return true;
243 }

245 // Handle the zone lock when the timer is activated
246 void OutdoorPvPTF::LockZone(GameObject* go, uint32 towerId, Team team, uint32 newWorldState)
247 {
248  SendUpdateWorldState(m_zoneWorldState, WORLD_STATE_REMOVE);
249  m_zoneWorldState = team == ALLIANCE ? WORLD_STATE_TF_LOCKED_ALLIANCE : WORLD_STATE_TF_LOCKED_HORDE;
250  SendUpdateWorldState(m_zoneWorldState, WORLD_STATE_ADD);
251 
252  m_zoneLockTimer = TIMER_TF_LOCK_TIME;
253  UpdateTimerWorldState();
254 
255  m_zoneOwner = team;
256  BuffTeam(team, SPELL_AUCHINDOUN_BLESSING);
257 
258  // lock the towers
259  LockTowers(go);
260 
261  sWorld.SendDefenseMessage(ZONE_ID_TEROKKAR_FOREST, team == ALLIANCE ? LANG_OPVP_TF_CAPTURE_ALL_TOWERS_A : LANG_OPVP_TF_CAPTURE_ALL_TOWERS_H);
262 
263  // remove tower states when zone has been captured and locked
264  for (unsigned int i : m_towerWorldState)
265  SendUpdateWorldState(i, WORLD_STATE_REMOVE);
266 
267  m_towerWorldState[towerId] = newWorldState;
268 }

270 // Handle the zone reset when the timer expires
271 void OutdoorPvPTF::UnlockZone()
272 {
273  // remove buffs
274  BuffTeam(m_zoneOwner, SPELL_AUCHINDOUN_BLESSING, true);
275 
276  m_zoneOwner = TEAM_NONE;
277 
278  // reset world states and towers

280  m_zoneWorldState = WORLD_STATE_TF_TOWERS_CONTROLLED;
281  SendUpdateWorldState(m_zoneWorldState, WORLD_STATE_ADD);
282 
283  // reset tower states
284  m_towerWorldState[0] = WORLD_STATE_TF_WEST_TOWER_NEUTRAL;
285  m_towerWorldState[1] = WORLD_STATE_TF_NORTH_TOWER_NEUTRAL;
286  m_towerWorldState[2] = WORLD_STATE_TF_EAST_TOWER_NEUTRAL;
287  m_towerWorldState[3] = WORLD_STATE_TF_SOUTH_EAST_TOWER_NEUTRAL;
288  m_towerWorldState[4] = WORLD_STATE_TF_SOUTH_TOWER_NEUTRAL;
289  for (unsigned int i : m_towerWorldState)
290  SendUpdateWorldState(i, WORLD_STATE_ADD);
291 
292  // update tower count
293  m_towersAlliance = 0;
294  m_towersHorde = 0;
295  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_A, m_towersAlliance);
296  SendUpdateWorldState(WORLD_STATE_TF_TOWER_COUNT_H, m_towersHorde);
297 
298  for (GuidZoneMap::const_iterator itr = m_zonePlayers.begin(); itr != m_zonePlayers.end(); ++itr)
299  {
300  // Find player who is in main zone (Terokkar Forest) to get correct map reference
301  if (!itr->second)
302  continue;
303 
304  if (Player* player = sObjectMgr.GetPlayer(itr->first))
305  {
306  ResetTowers(player);
307  break;
308  }
309  }

351 void OutdoorPvPTF::CalculateTimerWorldStateValues(uint32& firstDigit, uint32& secondDigit, uint32& hoursLeft)
352 {
