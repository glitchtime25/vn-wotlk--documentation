120 void PacketLog::LogPacket(WorldPacket const& packet, Direction direction, boost::asio::ip::address const& addr, uint16 port)
121 {
122  std::lock_guard<std::mutex> lock(_logPacketLock);
123 
124  PacketHeader header;
125  header.Direction = direction == CLIENT_TO_SERVER ? 0x47534d43 : 0x47534d53;
126  header.ConnectionId = 0;
127  header.ArrivalTicks = WorldTimer::getMSTime();
128 
129  header.OptionalDataSize = sizeof(header.OptionalData);
130  memset(header.OptionalData.SocketIPBytes, 0, sizeof(header.OptionalData.SocketIPBytes));
131  if (addr.is_v4())
132  {
133  auto bytes = addr.to_v4().to_bytes();
134  memcpy(header.OptionalData.SocketIPBytes, bytes.data(), bytes.size());
135  }
136  else if (addr.is_v6())
137  {
138  auto bytes = addr.to_v6().to_bytes();
139  memcpy(header.OptionalData.SocketIPBytes, bytes.data(), bytes.size());
140  }
141 
142  header.OptionalData.SocketPort = port;
143  header.Length = packet.size() + sizeof(header.Opcode);
144  header.Opcode = packet.GetOpcode();
145 
146  fwrite(&header, sizeof(header), 1, _file);
147  if (!packet.empty())
148  fwrite(packet.contents(), 1, packet.size(), _file);
149 
150  fflush(_file);
