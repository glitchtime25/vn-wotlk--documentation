61  bool MMapFactory::IsPathfindingEnabled(uint32 mapId, const Unit* unit = nullptr)
62  {
63  if (!sWorld.getConfig(CONFIG_BOOL_MMAP_ENABLED))
64  return false;
65 
66  if (unit)
67  {
68  // always use mmaps for players
69  if (unit->GetTypeId() == TYPEID_PLAYER)
70  return true;
71 
72  if (IsPathfindingForceDisabled(unit))
73  return false;
74 
75  if (IsPathfindingForceEnabled(unit))
76  return true;
77 
78  // always use mmaps for pets of players (can still be disabled by extra-flag for pet creature)
79  if (unit->GetTypeId() == TYPEID_UNIT && ((Creature*)unit)->IsPet() && unit->GetOwner() &&
80  unit->GetOwner()->GetTypeId() == TYPEID_PLAYER)
81  return true;
82  }
83 
84  return g_mmapDisabledIds->find(mapId) == g_mmapDisabledIds->end();
85  }

131  bool MMapManager::loadMapData(uint32 mapId, uint32 instanceId)
132  {
133  // we already have this map loaded?
134  if (m_loadedMMaps.find(packInstanceId(mapId, instanceId)) != m_loadedMMaps.end())
135  return true;
136 
137  // load and init dtNavMesh - read parameters from file
138  uint32 pathLen = sWorld.GetDataPath().length() + strlen("mmaps/%03i.mmap") + 1;
139  char* fileName = new char[pathLen];
140  snprintf(fileName, pathLen, (sWorld.GetDataPath() + "mmaps/%03i.mmap").c_str(), mapId);
141 
142  FILE* file = fopen(fileName, "rb");
143  if (!file)
144  {
145  if (MMapFactory::IsPathfindingEnabled(mapId))
146  sLog.outError("MMAP:loadMapData: Error: Could not open mmap file '%s'", fileName);
147  delete[] fileName;
148  return false;
149  }
150 
151  dtNavMeshParams params;
152  fread(&params, sizeof(dtNavMeshParams), 1, file);
153  fclose(file);
154 
155  dtNavMesh* mesh = dtAllocNavMesh();
156  MANGOS_ASSERT(mesh);
157  dtStatus dtResult = mesh->init(&params);
158  if (dtStatusFailed(dtResult))
159  {
160  dtFreeNavMesh(mesh);
161  sLog.outError("MMAP:loadMapData: Failed to initialize dtNavMesh for mmap %03u from file %s", mapId, fileName);
162  delete[] fileName;
163  return false;
164  }
165 
166  delete[] fileName;
167 
168  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:loadMapData: Loaded %03i.mmap", mapId);
169 
170  // store inside our map list
171  m_loadedMMaps.emplace(packInstanceId(mapId, instanceId), std::make_unique<MMapData>(mesh));
172  return true;
173  }

202  bool MMapManager::loadMap(uint32 mapId, uint32 instanceId, int32 x, int32 y, uint32 number)
203  {
204  // make sure the mmap is loaded and ready to load tiles
205  if (!loadMapData(mapId, instanceId))
206  return false;
207 
208  // get this mmap data
209  const auto& mmapData = m_loadedMMaps[packInstanceId(mapId, instanceId)];
210  MANGOS_ASSERT(mmapData->navMesh);
211 
212  char fileName[100];
213  if (number == 0)
214  sprintf(fileName, "%03u%02i%02i.mmtile", mapId, x, y);
215  else
216  sprintf(fileName, "%03u%02i%02i_%02i.mmtile", mapId, x, y, number);
217 
218  // check if we already have this tile loaded
219  uint32 packedGridPos = packTileID(x, y);
220  if (mmapData->mmapLoadedTiles.find(packedGridPos) != mmapData->mmapLoadedTiles.end())
221  {
222  sLog.outError("MMAP:loadMap: Asked to load already loaded navmesh tile. ");
223  return false;
224  }
225 
226  std::string filePath = sWorld.GetDataPath() + std::string("mmaps/") + fileName;
227  // load this tile
228  FILE* file = fopen(filePath.c_str(), "rb");
229  if (!file)
230  {
231  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "ERROR: MMAP:loadMap: Could not open mmtile file '%s'", fileName);
232  return false;
233  }
234 
235  // read header
236  MmapTileHeader fileHeader;
237  fread(&fileHeader, sizeof(MmapTileHeader), 1, file);
238 
239  if (fileHeader.mmapMagic != MMAP_MAGIC)
240  {
241  sLog.outError("MMAP:loadMap: Bad header in mmap %s", fileName);
242  fclose(file);
243  return false;
244  }
245 
246  if (fileHeader.mmapVersion != MMAP_VERSION)
247  {
248  sLog.outError("MMAP:loadMap: %s was built with generator v%i, expected v%i",
249  fileName, fileHeader.mmapVersion, MMAP_VERSION);
250  fclose(file);
251  return false;
252  }
253 
254  unsigned char* data = (unsigned char*)dtAlloc(fileHeader.size, DT_ALLOC_PERM);
255  MANGOS_ASSERT(data);
256 
257  size_t result = fread(data, fileHeader.size, 1, file);
258  if (!result)
259  {
260  sLog.outError("MMAP:loadMap: Bad header or data in mmap %s", fileName);
261  fclose(file);
262  return false;
263  }
264 
265  fclose(file);
266 
267  dtMeshHeader* header = (dtMeshHeader*)data;
268  dtTileRef tileRef = 0;
269 
270  // memory allocated for data is now managed by detour, and will be deallocated when the tile is removed
271  dtStatus dtResult = mmapData->navMesh->addTile(data, fileHeader.size, DT_TILE_FREE_DATA, 0, &tileRef);
272  if (dtStatusFailed(dtResult))
273  {
274  sLog.outError("MMAP:loadMap: Could not load %s into navmesh", fileName);
275  dtFree(data);
276  return false;
277  }
278 
279  mmapData->mmapLoadedTiles.insert(std::pair<uint32, dtTileRef>(packedGridPos, tileRef));
280  ++m_loadedTiles;
281  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:loadMap: Loaded into %03i[%02i,%02i]", fileName, mapId, header->x, header->y);
282  return true;
283  }

357  bool MMapManager::unloadMap(uint32 mapId, uint32 instanceId, int32 x, int32 y)
358  {
359  // check if we have this map loaded
360  auto itr = m_loadedMMaps.find(packInstanceId(mapId, instanceId));
361  if (itr == m_loadedMMaps.end())
362  {
363  // file may not exist, therefore not loaded
364  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Asked to unload not loaded navmesh map. %03u%02i%02i.mmtile", mapId, x, y);
365  return false;
366  }
367 
368  const auto& mmapData = (*itr).second;
369 
370  // check if we have this tile loaded
371  uint32 packedGridPos = packTileID(x, y);
372  if (mmapData->mmapLoadedTiles.find(packedGridPos) == mmapData->mmapLoadedTiles.end())
373  {
374  // file may not exist, therefore not loaded
375  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Asked to unload not loaded navmesh tile. %03u%02i%02i.mmtile", mapId, x, y);
376  return false;
377  }
378 
379  dtTileRef tileRef = mmapData->mmapLoadedTiles[packedGridPos];
380 
381  // unload, and mark as non loaded
382  dtStatus dtResult = mmapData->navMesh->removeTile(tileRef, nullptr, nullptr);
383  if (dtStatusFailed(dtResult))
384  {
385  // this is technically a memory leak
386  // if the grid is later reloaded, dtNavMesh::addTile will return error but no extra memory is used
387  // we cannot recover from this error - assert out
388  sLog.outError("MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
389  MANGOS_ASSERT(false);
390  }
391  else
392  {
393  mmapData->mmapLoadedTiles.erase(packedGridPos);
394  --m_loadedTiles;
395  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Unloaded mmtile %03i[%02i,%02i] from %03i", mapId, x, y, mapId);
396  return true;
397  }
398 
399  return false;
400  }

402  bool MMapManager::unloadMap(uint32 mapId)
403  {
404  bool success = false;
405  // unload all maps with given mapId
406  for (auto itr = m_loadedMMaps.begin(); itr != m_loadedMMaps.end();)
407  {
408  if (itr->first != uint64(mapId) << 32)
409  {
410  ++itr;
411  continue;
412  }
413 
414  // unload all tiles from given map
415  const auto& mmapData = (*itr).second;
416  for (MMapTileSet::iterator i = mmapData->mmapLoadedTiles.begin(); i != mmapData->mmapLoadedTiles.end(); ++i)
417  {
418  uint32 x = (i->first >> 16);
419  uint32 y = (i->first & 0x0000FFFF);
420  dtStatus dtResult = mmapData->navMesh->removeTile(i->second, nullptr, nullptr);
421  if (dtStatusFailed(dtResult))
422  sLog.outError("MMAP:unloadMap: Could not unload %03u%02i%02i.mmtile from navmesh", mapId, x, y);
423  else
424  {
425  --m_loadedTiles;
426  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Unloaded mmtile %03i[%02i,%02i] from %03i", mapId, x, y, mapId);
427  }
428  }
429 
430  itr = m_loadedMMaps.erase(itr);
431  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Unloaded %03i.mmap", mapId);
432  success = true;
433  }
434 
435  if (!success)
436  // file may not exist, therefore not loaded
437  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "MMAP:unloadMap: Asked to unload not loaded navmesh map %03u", mapId);
438 
439  return success;
440  }
