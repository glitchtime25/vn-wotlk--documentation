81  void ExecuteAction(uint32 action) override
82  {
83  switch (action)
84  {
85  case SHIRRAK_CARNIVOROUS_BITE:
86  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_CARNIVOROUS_BITE : SPELL_CARNIVOROUS_BITE_H) == CAST_OK)
87  ResetCombatAction(action, urand(4000, 10000));
88  break;
89  case SHIRRAK_FOCUS_FIRE:
90  {
91  ++m_uiFocusFireCount;
92  Unit* target = nullptr;
93 
94  switch (m_uiFocusFireCount)
95  {
96  case 1:
97  {
98  // engage the target
99  target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
100 
101  if (!target)
102  target = m_creature->GetVictim();
103 
104  DoScriptText(EMOTE_FOCUS, m_creature, target);
105  m_focusTargetGuid = target->GetObjectGuid();
106  // no break;
107  }
108  case 2:
109  // we have a delay of 1 sec between the summons
110  ResetCombatAction(action, 1000);
111  break;
112  case 3:
113  // reset the timers and the summon count
114  m_uiFocusFireCount = 0;
115  ResetCombatAction(action, 15000);
116  break;
117  }
118 
119  if (!target)
120  target = m_creature->GetMap()->GetUnit(m_focusTargetGuid);
121 
122  // Summon focus fire at target location
123  if (target)
124  target->CastSpell(nullptr, SPELL_FOCUS_FIRE_SUMMON, TRIGGERED_OLD_TRIGGERED);
125  break;
126  }
127  case SHIRRAK_ATTRACT_MAGIC:
128  if (DoCastSpellIfCan(m_creature, SPELL_ATTRACT_MAGIC) == CAST_OK)
129  ResetCombatAction(action, urand(25000, 38000));
130  break;
131  }
132  }
133 };
