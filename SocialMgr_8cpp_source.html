51 bool PlayerSocial::AddToSocialList(ObjectGuid friend_guid, bool ignore)
52 {
53  // check client limits
54  if (ignore)
55  {
56  if (GetNumberOfSocialsWithFlag(SOCIAL_FLAG_IGNORED) >= SOCIALMGR_IGNORE_LIMIT)
57  return false;
58  }
59  else
60  {
61  if (GetNumberOfSocialsWithFlag(SOCIAL_FLAG_FRIEND) >= SOCIALMGR_FRIEND_LIMIT)
62  return false;

64 
65  uint32 flag = SOCIAL_FLAG_FRIEND;
66  if (ignore)
67  flag = SOCIAL_FLAG_IGNORED;
68 
69  PlayerSocialMap::const_iterator itr = m_playerSocialMap.find(friend_guid.GetCounter());
70  if (itr != m_playerSocialMap.end())
71  {
72  CharacterDatabase.PExecute("UPDATE character_social SET flags = (flags | %u) WHERE guid = '%u' AND friend = '%u'", flag, m_playerLowGuid, friend_guid.GetCounter());
73  m_playerSocialMap[friend_guid.GetCounter()].Flags |= flag;
74  }
75  else
76  {
77  CharacterDatabase.PExecute("INSERT INTO character_social (guid, friend, flags) VALUES ('%u', '%u', '%u')", m_playerLowGuid, friend_guid.GetCounter(), flag);

79  fi.Flags |= flag;
80  m_playerSocialMap[friend_guid.GetCounter()] = fi;

139  data << uint32(friendInfo.Flags); // player flag (0x1-friend?, 0x2-ignored?, 0x4-muted?)

141  if (friendInfo.Flags & SOCIAL_FLAG_FRIEND) // if IsFriend()
142  {
143  data << uint8(friendInfo.Status); // online/offline/etc?
144  if (friendInfo.Status) // if online
145  {
146  data << uint32(friendInfo.Area); // player area
147  data << uint32(friendInfo.Level); // player level
148  data << uint32(friendInfo.Class); // player class
149  }
150  }
151  }
152 
153  plr->GetSession()->SendPacket(data);
154  DEBUG_LOG("WORLD: Sent SMSG_CONTACT_LIST");

179 void SocialMgr::GetFriendInfo(Player* player, uint32 friend_lowguid, FriendInfo& friendInfo) const
180 {
181  if (!player)
182  return;
183 
184  Player* pFriend = ObjectAccessor::FindPlayer(ObjectGuid(HIGHGUID_PLAYER, friend_lowguid));
185 
186  Team team = player->GetTeam();
187  AccountTypes security = player->GetSession()->GetSecurity();
188  bool allowTwoSideWhoList = sWorld.getConfig(CONFIG_BOOL_ALLOW_TWO_SIDE_WHO_LIST);
189  AccountTypes gmLevelInWhoList = AccountTypes(sWorld.getConfig(CONFIG_UINT32_GM_LEVEL_IN_WHO_LIST));
190 
191  PlayerSocialMap::iterator itr = player->GetSocial()->m_playerSocialMap.find(friend_lowguid);
192  if (itr != player->GetSocial()->m_playerSocialMap.end())
193  friendInfo.Note = itr->second.Note;
194 
195  // PLAYER see his team only and PLAYER can't see MODERATOR, GAME MASTER, ADMINISTRATOR characters
196  // MODERATOR, GAME MASTER, ADMINISTRATOR can see all
197  if (pFriend && pFriend->GetName() &&
198  (security > SEC_PLAYER ||
199  ((pFriend->GetTeam() == team || allowTwoSideWhoList) && (pFriend->GetSession()->GetSecurity() <= gmLevelInWhoList))) &&
200  pFriend->IsVisibleGloballyFor(player))
201  {
202  friendInfo.Status = FRIEND_STATUS_ONLINE;
203  if (pFriend->isAFK())
204  friendInfo.Status = FRIEND_STATUS_AFK;
205  if (pFriend->isDND())
206  friendInfo.Status = FRIEND_STATUS_DND;
207  friendInfo.Area = pFriend->GetZoneId();
208  friendInfo.Level = pFriend->GetLevel();
209  friendInfo.Class = pFriend->getClass();
210  }
211  else
212  {
213  friendInfo.Status = FRIEND_STATUS_OFFLINE;
214  friendInfo.Area = 0;
215  friendInfo.Level = 0;
216  friendInfo.Class = 0;
217  }

296 PlayerSocial* SocialMgr::LoadFromDB(QueryResult* result, ObjectGuid guid)
297 {
298  PlayerSocial* social = &m_socialMap[guid.GetCounter()];
299  social->SetPlayerGuid(guid);
300 
301  if (!result)
302  return social;
303 
304  // used to speed up check below. Using GetNumberOfSocialsWithFlag will cause unneeded iteration
305  uint32 friendCounter = 0, ignoreCounter = 0;
306 
307  do
308  {
309  Field* fields = result->Fetch();
310 
311  uint32 friend_guid = fields[0].GetUInt32();
312  uint32 flags = fields[1].GetUInt32();
313  std::string note = fields[2].GetCppString();
314 
315  if ((flags & SOCIAL_FLAG_IGNORED) && ignoreCounter >= SOCIALMGR_IGNORE_LIMIT)
316  continue;
317  if ((flags & SOCIAL_FLAG_FRIEND) && friendCounter >= SOCIALMGR_FRIEND_LIMIT)
318  continue;
319 
320  social->m_playerSocialMap[friend_guid] = FriendInfo(flags, note);
321 
322  if (flags & SOCIAL_FLAG_IGNORED)
323  ++ignoreCounter;
324  else
325  ++friendCounter;
326  }
327  while (result->NextRow());
328  delete result;
329  return social;
