40 void CinematicMgr::BeginCinematic()
41 {
42  // Sanity check for active camera set
43  if (m_activeCinematicCameraId == 0)
44  return;
45 
46  auto itr = sFlyByCameraStore.find(m_activeCinematicCameraId);
47  if (itr != sFlyByCameraStore.end())
48  {
49  // Initialize diff, and set camera
50  m_cinematicDiff = 0;
51  m_cinematicCamera = &itr->second;
52 
53  auto camitr = m_cinematicCamera->begin();
54  if (camitr != m_cinematicCamera->end())
55  {
56  Position pos { camitr->locations.x, camitr->locations.y, camitr->locations.z, camitr->locations.w };
57  if (!MaNGOS::IsValidMapCoord(pos.x, pos.y, pos.z, pos.o))
58  return;
59 
60  player->GetMap()->ForceLoadGrid(camitr->locations.x, camitr->locations.y);
61  m_CinematicObject = player->SummonCreature(VISUAL_WAYPOINT, pos.x, pos.y, pos.z, 0.0f, TEMPSPAWN_TIMED_DESPAWN, 5 * MINUTE * IN_MILLISECONDS);
62  if (m_CinematicObject)
63  {
64  m_CinematicObject->SetActiveObjectState(true);
65  player->GetCamera().SetView(m_CinematicObject, true);
66  }
67 
68  // Get cinematic length
69  FlyByCameraCollection::const_reverse_iterator camrevitr = m_cinematicCamera->rbegin();
70  if (camrevitr != m_cinematicCamera->rend())
71  m_cinematicLength = camrevitr->timeStamp;
72  else
73  m_cinematicLength = 0;
74  }
75  }
76 }

96 bool CinematicMgr::UpdateCinematicLocation(uint32 /*diff*/)
97 {
98  // should not occur
99  if (m_activeCinematicCameraId == 0)
100  return false;
101 
102  // If we never received an end packet 10 seconds after the final timestamp then force an end
103  if (m_cinematicDiff > m_cinematicLength + 10 * IN_MILLISECONDS)
104  {
105  EndCinematic();
106  return false;
107  }
108 
109  // just created and waiting BeginCinematic call
110  if (!m_cinematicCamera || m_cinematicCamera->size() == 0)
111  return true;
112 
113  Position lastPosition;
114  uint32 lastTimestamp = 0;
115  Position nextPosition;
116  uint32 nextTimestamp = 0;
117 
118  // Obtain direction of travel
119  for (FlyByCamera cam : *m_cinematicCamera)
120  {
121  if (cam.timeStamp > m_cinematicDiff)
122  {
123  nextPosition = { cam.locations.x, cam.locations.y, cam.locations.z, cam.locations.w };
124  nextTimestamp = cam.timeStamp;
125  break;
126  }
127  lastPosition = { cam.locations.x, cam.locations.y, cam.locations.z, cam.locations.w };
128  lastTimestamp = cam.timeStamp;
129  }
130 
131  float dx = nextPosition.x - lastPosition.x;
132  float dy = nextPosition.y - lastPosition.y;
133 
134  float angle = std::atan2(dy, dx);
135  angle = (angle >= 0) ? angle : 2 * float(M_PI) + angle;
136 
137  angle -= lastPosition.o;
138  if (angle < 0)
139  angle += 2 * float(M_PI);
140 
141  // Look for position around 2 second ahead of us.
142  int32 workDiff = m_cinematicDiff;
143 
144  // Modify result based on camera direction (Humans for example, have the camera point behind)
145  workDiff += static_cast<int32>(float(CINEMATIC_LOOKAHEAD) * cos(angle));
146 
147  // Get an iterator to the last entry in the cameras, to make sure we don't go beyond the end
148  FlyByCameraCollection::const_reverse_iterator endItr = m_cinematicCamera->rbegin();
149  if (endItr != m_cinematicCamera->rend() && workDiff > static_cast<int32>(endItr->timeStamp))
150  workDiff = endItr->timeStamp;
151 
152  // Never try to go back in time before the start of cinematic!
153  if (workDiff < 0)
154  workDiff = m_cinematicDiff;
155 
156  // Obtain the previous and next waypoint based on timestamp
157  for (FlyByCamera cam : *m_cinematicCamera)
158  {
159  if (static_cast<int32>(cam.timeStamp) >= workDiff)
160  {
161  nextPosition = { cam.locations.x, cam.locations.y, cam.locations.z, cam.locations.w };
162  nextTimestamp = cam.timeStamp;
163  break;
164  }
165  lastPosition = { cam.locations.x, cam.locations.y, cam.locations.z, cam.locations.w };
166  lastTimestamp = cam.timeStamp;
167  }
168 
169  // Never try to go beyond the end of the cinematic
170  if (workDiff > static_cast<int32>(nextTimestamp))
171  workDiff = static_cast<int32>(nextTimestamp);
172 
173  // Interpolate the position for this moment in time (or the adjusted moment in time)
174  uint32 timeDiff = nextTimestamp - lastTimestamp;
175  uint32 interDiff = workDiff - lastTimestamp;
176  float xDiff = nextPosition.x - lastPosition.x;
177  float yDiff = nextPosition.y - lastPosition.y;
178  float zDiff = nextPosition.z - lastPosition.z;
179  Position interPosition = { lastPosition.x + (xDiff * (float(interDiff) / float(timeDiff))), lastPosition.y +
180  (yDiff * (float(interDiff) / float(timeDiff))), lastPosition.z + (zDiff * (float(interDiff) / float(timeDiff))), 0.0f };
181 
182  // Advance (at speed) to this position. The remote sight object is used
183  // to send update information to player in cinematic
184  if (m_CinematicObject && MaNGOS::IsValidMapCoord(interPosition.x, interPosition.y, interPosition.z, interPosition.o))
185  m_CinematicObject->MonsterMoveWithSpeed(interPosition.x, interPosition.y, interPosition.z, 500.0f, false, true);
186 
187  return true;
188 }
