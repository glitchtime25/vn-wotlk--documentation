124  m_acolyteGUIDs.push_back(creature->GetObjectGuid());

128  case NPC_CRIMSON_GUARDSMAN:

137  // Do not store - again - GUIDs for Timmy spawn point, they were done previously with a different range
138  if (i == TIMMY)
139  continue;
140  // Store the GUID of the nearby NPCs for each defense point
141  if (creature->IsWithinDist2d(defensePoints[i].m_fX, defensePoints[i].m_fY, 8.0f))
142  {
143  m_crimsonDefendersLowGuids[i].push_back(creature->GetObjectGuid());
144  break;
145  }
146  }
147  break;
148  }

231  debug_log("SD2: Instance Stratholme: Baron run in progress.");
232  break;
233  case FAIL:
234  // may add code to remove aura from players, but in theory the time should be up already and removed.
235  break;
236  case DONE:
237  m_baronRunTimer = 0;
238  break;
239  default:
240  break;
241  }
242  m_auiEncounter[type] = data;
243  break;
244  case TYPE_BARONESS:
245  case TYPE_NERUB:
246  case TYPE_PALLID:
247  m_auiEncounter[type] = data;
248  if (data == DONE)
249  {
250  DoSortZiggurats();
251  DoUseDoorOrButton(m_zigguratStorage[type - TYPE_BARONESS].m_doorGuid);
252  }
253  if (data == SPECIAL)
254  StartSlaughterSquare();
255  break;
256  case TYPE_RAMSTEIN:
257  if (data == SPECIAL)
258  {
259  if (m_auiEncounter[type] != SPECIAL && m_auiEncounter[type] != DONE)
260  {
261  m_slaughterSquareTimer = 20 * IN_MILLISECONDS; // TODO - unknown, also possible that this is not the very correct place..
262  DoUseDoorOrButton(GO_PORT_GAUNTLET);
263  }
264 
265  uint32 abomCount = m_abominationGUIDs.size();
266  for (auto abomGuid : m_abominationGUIDs)
267  {
268  if (Creature* abom = instance->GetCreature(abomGuid))
269  {
270  if (!abom->IsAlive())
271  --abomCount;
272  }
273  else
274  {
275  // Remove obsolete guid from set and decrement count
276  m_abominationGUIDs.erase(abomGuid);
277  --abomCount;
278  }
279  }
280 
281  if (!abomCount)
282  {
283  // Open the Slaughterhouse door and set a timer to close it after 10 sec to let some time to Ramstein to move out
284  DoOpenSlaughterhouseDoor(true);
285  m_slaughterDoorTimer = 10 * IN_MILLISECONDS;
286 
287  // No more handlng of Abominations
288  m_slaughterSquareTimer = 0;
289 
290  if (Creature* baron = GetSingleCreatureFromStorage(NPC_BARON))
291  {
292  DoScriptText(SAY_ANNOUNCE_RAMSTEIN, baron);
293  if (Creature* ramstein = baron->SummonCreature(NPC_RAMSTEIN, stratholmeLocation[2].m_fX, stratholmeLocation[2].m_fY, stratholmeLocation[2].m_fZ, stratholmeLocation[2].m_fO, TEMPSPAWN_DEAD_DESPAWN, 0))

295 
296  debug_log("SD2: Instance Stratholme - Slaughter event: Ramstein spawned.");
297  }
298  }
299  else
300  debug_log("SD2: Instance Stratholme - Slaughter event: %u Abomination left to kill.", abomCount);
301  }
302  // After fail aggroing Ramstein means wipe on Ramstein, so close door again
303  if (data == IN_PROGRESS && m_auiEncounter[type] == FAIL)
304  DoUseDoorOrButton(GO_PORT_GAUNTLET);
305  if (data == DONE)
306  {
307  // Open side gate and start summoning skeletons
308  DoUseDoorOrButton(GO_PORT_SLAUGHTER_GATE);

312  m_mindlessSummonTimer = 500;
313  m_mindlessCount = 0;
314  m_undeadGUIDs.clear();

317  if (Creature* baron = GetSingleCreatureFromStorage(NPC_BARON))
318  {
319  for (uint8 i = 0; i < 5; ++i)
320  {

326 
327  debug_log("SD2: Instance Stratholme - Slaughter event: Summoned 5 guards.");
328  }
329  }

331  if (data == FAIL && m_auiEncounter[type] != FAIL)
332  {
333  DoUseDoorOrButton(GO_PORT_GAUNTLET);
334  m_slaughterSquareTimer = 0;
335 
336  // Let already moving Abominations stop
337  for (auto abomGuid : m_abominationGUIDs)
338  {
339  Creature* abom = instance->GetCreature(abomGuid);
340  if (abom && abom->GetMotionMaster()->GetCurrentMovementGeneratorType() == POINT_MOTION_TYPE)
341  abom->GetMotionMaster()->MovementExpired();
342  }
343  }
344 
345  m_auiEncounter[type] = data;
346  break;
347  case TYPE_BARON:
348  if (data == IN_PROGRESS)
349  {
350  DoUseOpenableObject(GO_ZIGGURAT_DOOR_5, false);

387  {
388  DoScriptText(SAY_EPILOGUE, ysida);

494  m_auiEncounter[type] = data;

499  }
500

504

508  << m_auiEncounter[6] << " " << m_auiEncounter[7];
509

511

514  }
