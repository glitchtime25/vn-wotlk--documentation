297 void instance_trial_of_the_crusader::SetData(uint32 uiType, uint32 uiData)
298 {
299  switch (uiType)
300  {
301  case TYPE_WIPE_COUNT:
302  // Update data before updating worldstate
303  m_auiEncounter[uiType] = uiData;
304  if (IsHeroicDifficulty())
305  DoUpdateWorldState(WORLD_STATE_WIPES_COUNT, MAX_WIPES_ALLOWED >= GetData(TYPE_WIPE_COUNT) ? MAX_WIPES_ALLOWED - GetData(TYPE_WIPE_COUNT) : 0);
306  break;
307  case TYPE_NORTHREND_BEASTS:
308  if (uiData == SPECIAL)
309  {
310  if (Creature* pTirion = GetSingleCreatureFromStorage(NPC_TIRION_A))
311  DoScriptText(m_auiEncounter[uiType] != FAIL ? SAY_TIRION_RAID_INTRO_LONG : SAY_RAID_TRIALS_INTRO, pTirion);
312  StartNextDialogueText(TYPE_NORTHREND_BEASTS);
313  }
314  else if (uiData == FAIL)
315  {
316  SetData(TYPE_WIPE_COUNT, m_auiEncounter[TYPE_WIPE_COUNT] + 1);
317  StartNextDialogueText(SAY_TIRION_BEAST_WIPE);
318  m_lSummonedGuidsList.clear();
319  }
320  else if (uiData == DONE)
321  StartNextDialogueText(SAY_TIRION_BEAST_SLAY);
322  // combat doors
323  if (uiData != SPECIAL)
324  {
325  DoUseDoorOrButton(GO_WEST_GATE);
326  DoUseDoorOrButton(GO_NORTH_GATE);
327  }
328  m_auiEncounter[uiType] = uiData;
329  break;
330  case TYPE_JARAXXUS:
331  if (uiData == SPECIAL)
332  // TODO - Confirm if we are not missing something
333  StartNextDialogueText(m_auiEncounter[uiType] != FAIL ? TYPE_JARAXXUS : EVENT_JARAXXUS_RESET_DELAY);
334  else if (uiData == FAIL)
335  {
336  SetData(TYPE_WIPE_COUNT, m_auiEncounter[TYPE_WIPE_COUNT] + 1);
337  StartNextDialogueText(NPC_RAMSEY_2);
338  }
339  else if (uiData == DONE)
340  StartNextDialogueText(SAY_JARAXXUS_DEATH);
341  else if (uiData == IN_PROGRESS)
342  m_lSummonedGuidsList.clear();
343  // combat doors
344  if (uiData != SPECIAL)
345  {
346  DoUseDoorOrButton(GO_WEST_GATE);
347  DoUseDoorOrButton(GO_NORTH_GATE);
348  }
349  m_auiEncounter[uiType] = uiData;
350  break;
351  case TYPE_FACTION_CHAMPIONS:
352  if (uiData == SPECIAL)
353  StartNextDialogueText(m_auiEncounter[uiType] != FAIL ? int32(SAY_TIRION_PVP_INTRO_1) : int32(TYPE_FACTION_CHAMPIONS));
354  else if (uiData == FAIL)
355  {
356  SetData(TYPE_WIPE_COUNT, m_auiEncounter[TYPE_WIPE_COUNT] + 1);
357  StartNextDialogueText(NPC_RAMSEY_3);
358 
359  // cleanup and reset crusaders
360  DoCleanupCrusaders();
361  m_uiKilledCrusaders = 0;
362  m_uiCrusadersAchievTimer = 0;
363  m_bCrusadersSummoned = false;
364  m_bCrusadersAchievCheck = false;
365  }
366  else if (uiData == DONE)
367  {
368  DoRespawnGameObject(GO_CRUSADERS_CACHE, 60 * MINUTE);
369  StartNextDialogueText(SAY_VARIAN_PVP_A_WIN);
370  }
371  // combat doors
372  if (uiData != SPECIAL)
373  {
374  DoUseDoorOrButton(GO_WEST_GATE);
375  DoUseDoorOrButton(GO_NORTH_GATE);
376  }
377  m_auiEncounter[uiType] = uiData;
378  break;
379  case TYPE_TWIN_VALKYR:
380  if (uiData == SPECIAL)
381  {
382  if (Creature* pTirion = GetSingleCreatureFromStorage(NPC_TIRION_A))
383  DoScriptText(m_auiEncounter[uiType] != FAIL ? SAY_TIRION_TWINS_INTRO : SAY_RAID_INTRO_SHORT, pTirion);

388  SetData(TYPE_WIPE_COUNT, m_auiEncounter[TYPE_WIPE_COUNT] + 1);
389  StartNextDialogueText(NPC_RAMSEY_4);

406  else if (uiData == FAIL)
407  SetData(TYPE_WIPE_COUNT, m_auiEncounter[TYPE_WIPE_COUNT] + 1);

653 // Function that will select the faction champions entries
654 void instance_trial_of_the_crusader::DoSelectCrusaders()
655 {
656  std::vector<uint32> vCrusaderHealers;
657  std::vector<uint32> vCrusaderOthers;
658 
659  // add all the healers and dps crusaders to vector
660  for (uint8 i = 0; i < MAX_CRUSADERS_HEALERS; ++i)
661  vCrusaderHealers.push_back(m_uiTeam == ALLIANCE ? aHordeHealerCrusaders[i] : aAllyHealerCrusaders[i]);
662  for (uint8 i = 0; i < MAX_CRUSADERS_OTHER; ++i)
663  vCrusaderOthers.push_back(m_uiTeam == ALLIANCE ? aHordeOtherCrusaders[i] : aAllyOtherCrusaders[i]);
664 
665  // replace random healers with corresponding dps
666  uint8 uiIndex = urand(0, vCrusaderHealers.size() - 1);
667  vCrusaderOthers.push_back(m_uiTeam == ALLIANCE ? aHordeReplacementCrusaders[uiIndex] : aAllyReplacementCrusaders[uiIndex]);
668  vCrusaderHealers.erase(vCrusaderHealers.begin() + uiIndex);
669 
670  if (!Is25ManDifficulty())
671  {
672  // on 10 man we replace a second healer
673  uiIndex = urand(0, vCrusaderHealers.size() - 1);
674 
675  uint32 uiCrusaderEntry = vCrusaderHealers[uiIndex];
676  for (uint8 i = 0; i < MAX_CRUSADERS_HEALERS; ++i)
677  {
678  if (uiCrusaderEntry == (m_uiTeam == ALLIANCE ? aHordeHealerCrusaders[i] : aAllyHealerCrusaders[i]))
679  {
680  vCrusaderOthers.push_back(m_uiTeam == ALLIANCE ? aHordeReplacementCrusaders[i] : aAllyReplacementCrusaders[i]);
681  break;
682  }
683  }
684  vCrusaderHealers.erase(vCrusaderHealers.begin() + uiIndex);
685 
686  // remove 4 random dps crusaders
687  for (uint8 i = 0; i < MAX_CRUSADERS_HEALERS; ++i)
688  vCrusaderOthers.erase(vCrusaderOthers.begin() + urand(0, vCrusaderOthers.size() - 1));
689  }
690 
691  // set the final list of crusaders
692  for (unsigned int vCrusaderHealer : vCrusaderHealers)
693  m_vCrusadersEntries.push_back(vCrusaderHealer);
694  for (unsigned int vCrusaderOther : vCrusaderOthers)
695  m_vCrusadersEntries.push_back(vCrusaderOther);
