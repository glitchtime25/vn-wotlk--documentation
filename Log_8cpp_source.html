346 std::string Log::GetTimestampStr()
347 {
348  time_t t = time(nullptr);
349  tm* aTm = localtime(&t);
350  // YYYY year
351  // MM month (2 digits 01-12)
352  // DD day (2 digits 01-31)
353  // HH hour (2 digits 00-23)
354  // MM minutes (2 digits 00-59)
355  // SS seconds (2 digits 00-59)
356  char buf[20];
357  int snRes = snprintf(buf, 20, "%04d-%02d-%02d_%02d-%02d-%02d", aTm->tm_year + 1900, aTm->tm_mon + 1, aTm->tm_mday, aTm->tm_hour, aTm->tm_min, aTm->tm_sec);
358  if (snRes < 0 || snRes >= sizeof(buf))
359  return "";
360  return std::string(buf);
361 }
