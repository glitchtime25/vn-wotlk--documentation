96  // Return true, if succeeded
97  virtual bool DoSpecialDragonAbility() = 0;
98 
99  // Return true to handle shared timers and MeleeAttack
100  virtual bool UpdateDragonAI(const uint32 /*uiDiff*/) { return true; }
101 
102  void UpdateAI(const uint32 uiDiff) override
103  {
104  // Return since we have no target
105  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
106  return;
107 
108  // Trigger special ability function at 75, 50 and 25% health
109  if (m_creature->GetHealthPercent() < 100.0f - m_uiEventCounter * 25.0f && DoSpecialDragonAbility())
110  ++m_uiEventCounter;
111 
112  // Call dragon specific virtual function
113  if (!UpdateDragonAI(uiDiff))
114  return;
115 
116  if (m_uiSeepingFogTimer < uiDiff)
117  {
118  DoCastSpellIfCan(m_creature, SPELL_SEEPING_FOG_R, CAST_TRIGGERED);
119  DoCastSpellIfCan(m_creature, SPELL_SEEPING_FOG_L, CAST_TRIGGERED);
120  m_uiSeepingFogTimer = 60000;
121  }
122  else
123  m_uiSeepingFogTimer -= uiDiff;
124 
125  if (m_uiNoxiousBreathTimer < uiDiff)
126  {
127  if (DoCastSpellIfCan(m_creature, SPELL_NOXIOUS_BREATH) == CAST_OK)
128  m_uiNoxiousBreathTimer = urand(9000, 12000);
129  }
130  else
131  m_uiNoxiousBreathTimer -= uiDiff;
132 
133  if (m_uiTailsweepTimer < uiDiff)
134  {
135  if (DoCastSpellIfCan(m_creature, SPELL_TAILSWEEP) == CAST_OK)
136  m_uiTailsweepTimer = urand(10000, 30000);
137  }
138  else
139  m_uiTailsweepTimer -= uiDiff;
140 
141  DoMeleeAttackIfReady();
142  }
143 };
