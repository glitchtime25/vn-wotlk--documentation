101  {

109  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
110  m_creature->ClearAllReactives();
111  m_creature->GetMotionMaster()->Clear();
112  m_creature->GetMotionMaster()->MoveIdle();
113  m_creature->SetStandState(UNIT_STAND_STATE_DEAD);
114  m_creature->CastSpell(nullptr, SPELL_PACIFY_SELF, TRIGGERED_OLD_TRIGGERED);
115 
116  SetCombatMovement(false);
117  SetCombatScriptStatus(true);
118 
119  m_uiPhase = PHASE_FAKE_DEATH;
120 
121  OnFakeingDeath();
122  }

124  void Revive(bool bOnlyFlags = false)
125  {
126  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
127  m_creature->SetStandState(UNIT_STAND_STATE_STAND);
128 
129  m_creature->RemoveAurasDueToSpell(SPELL_PACIFY_SELF);
130 
131  if (bOnlyFlags)
132  return;
133 
134  m_creature->SetHealth(m_creature->GetMaxHealth());
135  m_uiPhase = PHASE_NORMAL;
136 
137  DoResetThreat();
138  Reset();
139 
140  // Assume Attack
141  if (!OnRevive())
142  {
143  SetCombatMovement(true, true);
144 
145  SetDeathPrevention(true);
146 
147  SetCombatScriptStatus(false);
148  }
149  }

154  m_creature->InterruptNonMeleeSpells(true);
155  }

165 struct boss_thekalAI : public boss_thekalBaseAI
166 {
167  boss_thekalAI(Creature* creature) : boss_thekalBaseAI(creature, THEKAL_ACTION_MAX)
168  {
169  AddTimerlessCombatAction(THEKAL_TIGER_ENRAGE, false);
170  AddCustomAction(ACTION_RESSURECTION, true, [&]()
171  {
172  // resurrect him in any case
173  DoCastSpellIfCan(nullptr, SPELL_RESURRECT);
174 
175  m_uiPhase = PHASE_WAITING;
176  if (m_instance)
177  {
178  CanPreventAddsResurrect();
179  m_instance->SetData(TYPE_THEKAL, IN_PROGRESS);
180  }
181  });
182  AddCustomAction(THEKAL_RESS_PHASE_2_DELAY, true, [&]()
183  {
184  DoCastSpellIfCan(nullptr, SPELL_TIGER_FORM);
185  m_uiPhase = PHASE_TIGER;
186  SetDeathPrevention(false);
187  SetCombatScriptStatus(false);
188  SetMeleeEnabled(true);
189  SetCombatMovement(true, true);
190  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
191  m_creature->SetSpellList(SPELL_LIST_PHASE_2);
192  SetActionReadyStatus(THEKAL_TIGER_ENRAGE, true);
193  });
194  }

243  // Only call in context where m_instance is valid
244  bool CanPreventAddsResurrect() const
245  {
246  // If any add is alive, return false
247  if (m_instance->GetData(TYPE_ZATH) != SPECIAL || m_instance->GetData(TYPE_LORKHAN) != SPECIAL)
248  return false;
249 
250  // Else Prevent them Resurrecting
251  if (Creature* lorkhan = m_instance->GetSingleCreatureFromStorage(NPC_LORKHAN))
252  {
253  if (boss_thekalBaseAI* pFakerAI = dynamic_cast<boss_thekalBaseAI*>(lorkhan->AI()))
254  pFakerAI->PreventRevive();
255  }
256  if (Creature* zath = m_instance->GetSingleCreatureFromStorage(NPC_ZATH))
257  {
258  if (boss_thekalBaseAI* pFakerAI = dynamic_cast<boss_thekalBaseAI*>(zath->AI()))
259  pFakerAI->PreventRevive();
260  }
261 
262  return true;
263  }

265  void OnFakeingDeath() override
266  {
267  ResetTimer(ACTION_RESSURECTION, 10000);
268 
269  if (m_instance)
270  {
271  m_instance->SetData(TYPE_THEKAL, SPECIAL);
272 
273  // If both Adds are already dead, don't wait 10 seconds
274  if (CanPreventAddsResurrect())
275  ResetTimer(ACTION_RESSURECTION, 1000);
276  }
277  }
278 
279  bool OnRevive() override
280  {
281  if (!m_instance)
282  return false;
283 
284  // Both Adds are 'dead' enter tiger phase
285  if (CanPreventAddsResurrect())
286  {
287  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
288  SetMeleeEnabled(false);
289  DoCastSpellIfCan(nullptr, SPELL_RESSURECTION_IMPACT_VISUAL);
290  m_creature->SetTarget(nullptr);
291  ResetTimer(THEKAL_RESS_PHASE_2_DELAY, 5000);
292  return true;
293  }
294  return false;
295  }
