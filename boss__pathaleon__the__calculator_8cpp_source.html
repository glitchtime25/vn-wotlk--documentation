125  void UpdateAI(const uint32 uiDiff) override
126  {
127  // Return since we have no target
128  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
129  return;
130 
131  if (m_uiManaTapTimer < uiDiff)
132  {
133  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MANA_TAP, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
134  {
135  if (DoCastSpellIfCan(pTarget, SPELL_MANA_TAP) == CAST_OK)
136  m_uiManaTapTimer = urand(16000, 34000);
137  }
138  }
139  else
140  m_uiManaTapTimer -= uiDiff;
141 
142  if (m_uiArcaneTorrentTimer < uiDiff)
143  {
144  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_TORRENT) == CAST_OK)
145  m_uiArcaneTorrentTimer = urand(40000, 52000);
146  }
147  else
148  m_uiArcaneTorrentTimer -= uiDiff;
149 
150  if (m_uiDominationTimer < uiDiff)
151  {
152  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER))
153  {
154  if (DoCastSpellIfCan(pTarget, SPELL_DOMINATION) == CAST_OK)
155  {
156  DoScriptText(urand(0, 1) ? SAY_DOMINATION_1 : SAY_DOMINATION_2, m_creature);
157  m_uiDominationTimer = urand(25000, 30000);
158  }
159  }
160  }
161  else
162  m_uiDominationTimer -= uiDiff;
163 
164  // Only casting if Heroic Mode is used
165  if (!m_bIsRegularMode)
166  {
167  if (m_uiArcaneExplosionTimer < uiDiff)
168  {
169  if (DoCastSpellIfCan(m_creature, SPELL_ARCANE_EXPLOSION_H) == CAST_OK)
170  m_uiArcaneExplosionTimer = urand(13000, 25000);
171  }
172  else
173  m_uiArcaneExplosionTimer -= uiDiff;
174  }
175 
176  if (!m_bIsEnraged && m_creature->GetHealthPercent() < 21.0f)
177  {
178  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
179  {
180  DoCastSpellIfCan(m_creature, SPELL_SUICIDE, CAST_TRIGGERED);
181  DoScriptText(SAY_ENRAGE, m_creature);
182  m_bIsEnraged = true;
183  }
184  }
185 
186  // Summon and empower Nether Wraiths only when not enraged
187  if (!m_bIsEnraged)
188  {
189  if (m_uiSummonTimer < uiDiff)
190  {
191  uint8 uiMaxWraith = urand(3, 4);
192  for (uint8 i = 0; i < uiMaxWraith; ++i)
193  DoCastSpellIfCan(m_creature, aWraithSummonSpells[i], CAST_TRIGGERED);
194 
195  DoScriptText(SAY_SUMMON, m_creature);
196  m_uiSummonTimer = urand(45000, 50000);
197  }
198  else
199  m_uiSummonTimer -= uiDiff;
200 
201  if (m_uiAngerTimer < uiDiff)
202  {
203  if (DoCastSpellIfCan(m_creature, SPELL_DISGRUNTLED_ANGER) == CAST_OK)
204  m_uiAngerTimer = urand(55000, 84000);
205  }
206  else
207  m_uiAngerTimer -= uiDiff;
208  }
209 
210  DoMeleeAttackIfReady();
211  }
