183  void ExecuteAction(uint32 action) override
184  {
185  switch (action)
186  {
187  case SJONNIR_ACTION_CHAIN_LIGHTNING:
188  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
189  {
190  if (DoCastSpellIfCan(pTarget, m_isRegularMode ? SPELL_CHAIN_LIGHTNING : SPELL_CHAIN_LIGHTNING_H) == CAST_OK)
191  ResetCombatAction(action, 15000);
192  }
193  break;
194  case SJONNIR_ACTION_LIGHTNING_SHIELD:
195  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_LIGHTNING_SHIELD : SPELL_LIGHTNING_SHIELD_H) == CAST_OK)
196  ResetCombatAction(action, 15000);
197  break;
198  case SJONNIR_ACTION_STATIC_CHARGE:
199  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_STATIC_CHARGE : SPELL_STATIC_CHARGE_H) == CAST_OK)
200  ResetCombatAction(action, 25000);
201  break;
202  case SJONNIR_ACTION_LIGHTNING_RING:
203  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_LIGHTNING_RING : SPELL_LIGHTNING_RING_H) == CAST_OK)
204  ResetCombatAction(action, 25000);
205  break;
206  case SJONNIR_ACTION_SUMMON:
207  if (m_creature->GetHealthPercent() <= (float)m_uiHpCheck)
208  {
209  if (!m_instance)
210  return;
211 
212  switch (m_uiHpCheck)
213  {
214  case 75:
215  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_SUMMON_IRON_TROGG : SPELL_SUMMON_IRON_TROGG_H, CAST_TRIGGERED) == CAST_OK)
216  {
217  m_creature->RemoveAurasDueToSpell(m_isRegularMode ? SPELL_SUMMON_IRON_DWARF : SPELL_SUMMON_IRON_DWARF_H);
218 
219  if (Creature* pBrann = m_instance->instance->GetCreature(m_instance->GetBrannSummonedGuid()))
220  DoScriptText(SAY_SPAWN_TROGG, pBrann);
221 
222  m_uiHpCheck = 50;
223  }
224  break;
225  case 50:
226  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_SUMMON_MALFORMED_OOZE : SPELL_SUMMON_MALFORMED_OOZE_H, CAST_TRIGGERED) == CAST_OK)
227  {
228  m_creature->RemoveAurasDueToSpell(m_isRegularMode ? SPELL_SUMMON_IRON_TROGG : SPELL_SUMMON_IRON_TROGG_H);
229 
230  if (Creature* pBrann = m_instance->instance->GetCreature(m_instance->GetBrannSummonedGuid()))
231  DoScriptText(SAY_SPAWN_OOZE, pBrann);
232 
233  m_uiHpCheck = 15;
234  }
235  break;
236  case 15:
237  if (DoCastSpellIfCan(m_creature, SPELL_FRENZY) == CAST_OK)
238  {
239  m_creature->RemoveAurasDueToSpell(m_isRegularMode ? SPELL_SUMMON_MALFORMED_OOZE : SPELL_SUMMON_MALFORMED_OOZE_H);
240 
241  DoCastSpellIfCan(m_creature, SPELL_SUMMON_EARTHEN_DWARF, CAST_TRIGGERED);
242 
243  if (Creature* pBrann = m_instance->instance->GetCreature(m_instance->GetBrannSummonedGuid()))
244  DoScriptText(SAY_SPAWN_EARTHEN, pBrann);
245 
246  m_uiHpCheck = 0;
247  }
248  break;
249  }
250  }
251  break;
252  }
