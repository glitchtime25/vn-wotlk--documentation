124  void JustSummoned(Creature* summoned) override
125  {
126  if (summoned->GetEntry() != NPC_RHINO_SPIRIT)
127  return;
128  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, m_isRegularMode ? SPELL_STAMPEDE_RHINO : SPELL_STAMPEDE_RHINO_H, SELECT_FLAG_PLAYER))
129  {
130  summoned->CastSpell(m_creature, SPELL_STAMPEDE_EFFECT, TRIGGERED_OLD_TRIGGERED);
131  summoned->CastSpell(nullptr, SPELL_STAMPEDE_PROC, TRIGGERED_OLD_TRIGGERED);
132  summoned->CastSpell(target, m_isRegularMode ? SPELL_STAMPEDE_RHINO : SPELL_STAMPEDE_RHINO_H, TRIGGERED_NONE, nullptr, nullptr, m_creature->GetObjectGuid());
133  }
134  }

136  void DoPhaseSwitch()
137  {
138  if (!m_isTrollPhase)
139  m_creature->RemoveAurasDueToSpell(SPELL_RHINO_TRANSFORM);
140 
141  m_isTrollPhase = !m_isTrollPhase;
142 
143  if (m_isTrollPhase)
144  {
145  DoCastSpellIfCan(nullptr, SPELL_TROLL_TRANSFORM);
146  m_creature->SetSpellList(m_isRegularMode ? SPELLSET_TROLL_NHC : SPELLSET_TROLL_HC);
147  }
148  else
149  {
150  DoBroadcastText(urand(0, 1) ? SAY_TRANSFORM_1 : SAY_TRANSFORM_2, m_creature);
151  DoCastSpellIfCan(nullptr, SPELL_RHINO_TRANSFORM);
152  m_creature->SetSpellList(m_isRegularMode ? SPELLSET_RHINO_NHC : SPELLSET_RHINO_HC);
153  }
154  m_abilityCount = 0;
155  }
156 
157  void ExecuteAction(uint32 action) override
158  {
