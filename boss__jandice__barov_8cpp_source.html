56  {

59  return;
60 
61  // CurseOfBlood_Timer

63  {
64  if (DoCastSpellIfCan(m_creature, SPELL_CURSE_OF_BLOOD) == CAST_OK)
65  m_uiCurseOfBloodTimer = urand(30000, 35000);
66  }
67  else
68  m_uiCurseOfBloodTimer -= uiDiff;
69 
70  // Banish
71  if (m_uiBanishTimer < uiDiff)
72  {
73  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
74  {
75  if (DoCastSpellIfCan(pTarget, SPELL_BANISH) == CAST_OK)
76  m_uiBanishTimer = urand(17000, 21000);
77  }
78  }
79  else
80  m_uiBanishTimer -= uiDiff;
81 
82  // Illusion_Timer
83  if (m_uiIllusionTimer < uiDiff)
84  {
85  // Summon illusions and becomes invisible for 8 seconds while hiding into the illusions
86  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_ILLUSIONS) == CAST_OK)
87  {
88  m_uiIllusionTimer = 25000;
89  DoCastSpellIfCan(m_creature, SPELL_SPREAD_JANDICE);
90  }
91  }
92  else
93  m_uiIllusionTimer -= uiDiff;
94
