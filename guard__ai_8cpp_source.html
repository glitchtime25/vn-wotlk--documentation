63 void guardAI::UpdateAI(const uint32 diff)
64 {
65  // Always decrease our global cooldown first
66  if (m_globalCooldown > diff)
67  m_globalCooldown -= diff;
68  else
69  m_globalCooldown = 0;
70 
71  // Buff timer (only buff when we are alive and not in combat
72  if (m_creature->IsAlive() && !m_creature->IsInCombat())
73  {
74  if (m_buffTimer < diff)
75  {
76  // Find a spell that targets friendly and applies an aura (these are generally buffs)
77  const SpellEntry* spellInfo = SelectSpell(m_creature, -1, -1, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_AURA);
78 
79  if (spellInfo && !m_globalCooldown)
80  {
81  // Cast the buff spell
82  m_creature->CastSpell(m_creature, spellInfo, TRIGGERED_NONE);
83 
84  // Set our global cooldown
85  m_globalCooldown = GENERIC_CREATURE_COOLDOWN;
86 
87  // Set our timer to 10 minutes before rebuff
88  m_buffTimer = 600000;
89  } // Try again in 30 seconds
90  else
91  m_buffTimer = 30000;
92  }
93  else
94  m_buffTimer -= diff;
95  }
96 
97  // Return since we have no target
98  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
99  return;
100 
101  // Make sure our attack is ready and we arn't currently casting
102  if (!m_creature->IsNonMeleeSpellCasted(false))
103  {
104  // If we are within range melee the target
105  if (m_creature->CanReachWithMeleeAttack(m_creature->GetVictim()))
106  {
107  bool bHealing = false;
108  const SpellEntry* spellInfo = nullptr;
109 
110  // Select a healing spell if less than 30% hp
111  if (m_creature->GetHealthPercent() < 30.0f && !urand(0, 2))
112  spellInfo = SelectSpell(m_creature, -1, -1, SELECT_TARGET_ANY_FRIEND, 0, 0, 0, 0, SELECT_EFFECT_HEALING);
113 
114  // No healing spell available, select a hostile spell
115  if (spellInfo)
116  bHealing = true;
117  else
118  spellInfo = SelectSpell(m_creature->GetVictim(), -1, -1, SELECT_TARGET_ANY_ENEMY, 0, 0, 0, 0, SELECT_EFFECT_DONTCARE);
119 
120  // 20% chance to replace our white hit with a spell
121  if (spellInfo && !urand(0, 4) && !m_globalCooldown)
122  {
123  // Cast spell
124  DoCastSpellIfCan(bHealing ? m_creature : m_creature->GetVictim(), spellInfo->Id);
125 
126  // Set our global cooldown
127  m_globalCooldown = GENERIC_CREATURE_COOLDOWN;
128  }
129  }
130  }
131 
132  DoMeleeAttackIfReady();
133 }
