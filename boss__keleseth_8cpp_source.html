104  void Revive()
105  {
106  DoCastSpellIfCan(m_creature, SPELL_SHADOW_FISSURE, CAST_TRIGGERED);
107  DoCastSpellIfCan(m_creature, SPELL_FULL_HEAL, CAST_TRIGGERED);
108  DoScriptText(EMOTE_RISE_SKELETON, m_creature);
109

117  void DamageTaken(Unit* pDealer, uint32& uiDamage, DamageEffectType /*damagetype*/, SpellEntry const* spellInfo) override
118  {
119  // ignore damage done by self
120  if (pDealer == m_creature)
121  return;
122 
123  if (m_uiReviveTimer)
124  {
125  uiDamage = 0;
126  return;
127  }
128 
129  if (m_creature->GetHealth() < uiDamage)
130  {
131  // start faking death
132  uiDamage = 0;
133  m_uiReviveTimer = 6000;
134  m_creature->SetTarget(nullptr);
135  m_creature->SetHealth(0);
136  m_creature->RemoveAllAurasOnDeath();
137  m_creature->GetMotionMaster()->Clear();
138  m_creature->SetFlag(UNIT_DYNAMIC_FLAGS, UNIT_DYNFLAG_DEAD);
139  m_creature->SetStandState(UNIT_STAND_STATE_DEAD);
140  }
141  }
142 
143  void UpdateAI(const uint32 uiDiff) override
144  {
145  if (m_uiAttackTimer)
146  {
147  if (m_uiAttackTimer <= uiDiff)
148  {
149  if (m_pInstance)
150  {
151  if (Creature* pKeleseth = m_pInstance->GetSingleCreatureFromStorage(NPC_KELESETH))
152  AttackStart(pKeleseth->GetVictim());
153  }
154 
155  m_uiCastTimer = urand(5000, 10000);
156  m_uiAttackTimer = 0;
157  }
158  else
159  m_uiAttackTimer -= uiDiff;
160 
161  return;
162  }
163 
164  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
165  return;
166 
167  if (m_uiReviveTimer)
168  {
169  if (m_uiReviveTimer <= uiDiff)
170  {
171  Revive();
172  m_uiReviveTimer = 0;
173  }
174  else
175  m_uiReviveTimer -= uiDiff;
176 
177  return;
178  }
179 
180  if (m_uiCastTimer < uiDiff)
181  {
182  if (DoCastSpellIfCan(m_creature->GetVictim(), m_bIsRegularMode ? SPELL_DECREPIFY : SPELL_DECREPIFY_H) == CAST_OK)
183  {
184  if (!m_bIsRegularMode && urand(0, 1))
185  DoCastSpellIfCan(nullptr, SPELL_BONE_ARMOR, CAST_TRIGGERED);
186 
187  m_uiCastTimer = urand(5000, 15000);
188  }
189  }
190  else
191  m_uiCastTimer -= uiDiff;
192 
193  DoMeleeAttackIfReady();
194  }
195 };

242  m_creature->SummonCreature(NPC_VRYKUL_SKELETON, fAddPosition[0] + urand() % 7, fAddPosition[1] + urand() % 7, fAddPosition[2], fAddPosition[3], TEMPSPAWN_DEAD_DESPAWN, 0);
243  }

267 
268  void ExecuteAction(uint32 action) override
269  {
270  switch (action)
271  {
272  case KELESETH_ACTION_SUMMON_ADDS:
273  SummonAdds();
274  DisableCombatAction(action);
275  break;
276  case KELESETH_ACTION_SHADOW_BOLT:
277  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_SHADOWBOLT : SPELL_SHADOWBOLT_H) == CAST_OK)
278  ResetCombatAction(action, 3000);
279  break;
280  case KELESETH_ACTION_FROST_TOMB:
281  if (Unit* pTombTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_SUMMON_FROST_TOMB, SELECT_FLAG_PLAYER))
282  {
283  if (DoCastSpellIfCan(pTombTarget, SPELL_FROST_TOMB_TARGET) == CAST_OK)
284  {
285  DoScriptText(SAY_FROSTTOMB, m_creature);
286  DoScriptText(EMOTE_FROST_TOMB, m_creature, pTombTarget);
287  ResetCombatAction(action, 25000);
288  }
289  }
290  break;
291  }
292  }
