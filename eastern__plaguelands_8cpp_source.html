111  void JustSummoned(Creature* pSummoned) override
112  {
113  switch (pSummoned->GetEntry())
114  {
115  case NPC_INJURED_PEASANT:
116  case NPC_PLAGUED_PEASANT:
117  float fX, fY, fZ;
118  pSummoned->GetRandomPoint(aPeasantMoveLoc[0], aPeasantMoveLoc[1], aPeasantMoveLoc[2], 10.0f, fX, fY, fZ);
119  pSummoned->GetMotionMaster()->MovePoint(1, fX, fY, fZ);
120  m_uiTotalCounter++;
121  break;
122  case NPC_SCOURGE_FOOTSOLDIER:
123  case NPC_THE_CLEANER:
124  if (Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid))
125  pSummoned->AI()->AttackStart(pPlayer);
126  break;
127  case NPC_SCOURGE_ARCHER:
128  break;
129  }

131  m_lSummonedGuidList.push_back(pSummoned->GetObjectGuid());
132  }

134  void SummonedMovementInform(Creature* pSummoned, uint32 uiMotionType, uint32 uiPointId) override
135  {
136  if (uiMotionType != POINT_MOTION_TYPE || !uiPointId)
137  return;
138 
139  if (uiPointId)
140  {
141  ++m_uiSaveCounter;
142  pSummoned->GetMotionMaster()->Clear();
143 
144  pSummoned->RemoveAllAuras();
145  pSummoned->CastSpell(pSummoned, SPELL_ENTER_THE_LIGHT_DND, TRIGGERED_NONE);
146  pSummoned->ForcedDespawn(10000);
147 
148  // Event ended
149  if (m_uiSaveCounter >= MAX_SAVED_PEASANT)
150  DoBalanceEventEnd();
151  // Phase ended
152  else if (m_uiSaveCounter + m_uiKillCounter == m_uiTotalCounter)
153  DoHandlePhaseEnd();
154  }

157  void SummonedCreatureJustDied(Creature* pSummoned) override
158  {

160  {
161  ++m_uiKillCounter;

173  DoHandlePhaseEnd();

177  void DoSummonWave(uint32 uiSummonId = 0)
178  {
179  float fX, fY, fZ;
180 
181  if (!uiSummonId)
182  {
183  for (uint8 i = 0; i < BASE_PEASANTS_PER_WAVE + m_uiCurrentWave; ++i)
184  {
185  uint32 uiSummonEntry = roll_chance_i(70) ? NPC_INJURED_PEASANT : NPC_PLAGUED_PEASANT;
186  m_creature->GetRandomPoint(aPeasantSpawnLoc[0], aPeasantSpawnLoc[1], aPeasantSpawnLoc[2], 10.0f, fX, fY, fZ);
187  if (Creature* pTemp = m_creature->SummonCreature(uiSummonEntry, fX, fY, fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0))
188  {
189  // Only the first mob needs to yell
190  if (!i)
191  DoScriptText(aPeasantSpawnYells[urand(0, 2)], pTemp);
192  }
193  }
194 
195  ++m_uiCurrentWave;
196  }
197  else if (uiSummonId == NPC_SCOURGE_FOOTSOLDIER)
198  {
199  uint8 uiRand = urand(2, 3);
200  for (uint8 i = 0; i < uiRand; ++i)
201  {
202  m_creature->GetRandomPoint(aPeasantSpawnLoc[0], aPeasantSpawnLoc[1], aPeasantSpawnLoc[2], 15.0f, fX, fY, fZ);
203  m_creature->SummonCreature(NPC_SCOURGE_FOOTSOLDIER, fX, fY, fZ, 0, TEMPSPAWN_DEAD_DESPAWN, 0);
204  }
205  }
206  else if (uiSummonId == NPC_SCOURGE_ARCHER)
207  {
208  for (auto i : aArcherSpawn)
209  m_creature->SummonCreature(NPC_SCOURGE_ARCHER, i[0], i[1], i[2], i[3], TEMPSPAWN_DEAD_DESPAWN, 0);
210  }
211  }

244  void DoDespawnSummons(bool bIsEventEnd = false)
245  {
246  for (GuidList::const_iterator itr = m_lSummonedGuidList.begin(); itr != m_lSummonedGuidList.end(); ++itr)
247  {
248  if (Creature* pTemp = m_creature->GetMap()->GetCreature(*itr))
249  {
250  if (bIsEventEnd && (pTemp->GetEntry() == NPC_INJURED_PEASANT || pTemp->GetEntry() == NPC_PLAGUED_PEASANT))
251  continue;
252 
253  pTemp->ForcedDespawn();
254  }
255  }
256  }

258  // Loop over all Scourge Archer NPCs to pick a target and attack it
259  void DoAttackArchersTarget()
260  {
261  for (GuidList::const_iterator itr = m_lSummonedGuidList.begin(); itr != m_lSummonedGuidList.end(); ++itr)
262  {
263  Creature* pTemp = m_creature->GetMap()->GetCreature(*itr);
264  if (pTemp && pTemp->GetEntry() == NPC_SCOURGE_ARCHER)
265  {
266  // Archer has a victim and is in range, keep on attacking it
267  if (pTemp->GetVictim() && pTemp->IsWithinDistInMap(pTemp->GetVictim(), 30.0f))
268  continue;
269  // Else, find a new target in range
270  // First look for an Injured Peasant in range (arbitrary choice), if none look for a Plagued Peasant
271  Creature* pTarget = GetClosestCreatureWithEntry(pTemp, NPC_INJURED_PEASANT, 30.0f);
272  if (!pTarget)
273  pTarget = GetClosestCreatureWithEntry(pTemp, NPC_PLAGUED_PEASANT, 30.0f);
274 
275  if (pTarget)
276  pTemp->AI()->AttackStart(pTarget);
277  }
278  }

281  void UpdateAI(const uint32 uiDiff) override
282  {
283  if (m_uiEventTimer)
284  {
285  if (m_uiEventTimer <= uiDiff)
286  {
287  switch (m_uiPhase)
288  {
289  case 0:
290  DoSummonWave(NPC_SCOURGE_ARCHER);
291  m_uiEventTimer = 5000;
292  break;
293  case 1:
294  DoSummonWave();
295  m_uiEventTimer = urand(60000, 80000);
296  break;
297  default:
298  // The summoning timer of the soldiers isn't very clear
299  DoSummonWave(NPC_SCOURGE_FOOTSOLDIER);
300  m_uiEventTimer = urand(5000, 30000);
301  break;
302  }
303  ++m_uiPhase;
304  }
305  else
306  m_uiEventTimer -= uiDiff;
307  }
308 
309  // Handle event end in case of fail
310  if (m_uiSadEndTimer)
311  {
312  if (m_uiSadEndTimer <= uiDiff)
313  {
314  DoScriptText(SAY_EVENT_FAIL_2, m_creature);
315  m_creature->ForcedDespawn(5000);
316  DoDespawnSummons();
317  m_uiSadEndTimer = 0;
318  }
319  else
320  m_uiSadEndTimer -= uiDiff;
321  }
322 
323  // Timer to force Scourge Archer NPCs to refresh their target
324  if (m_uiArcherCheckTargetTimer)
325  {
326  if (m_uiArcherCheckTargetTimer <= uiDiff)
327  {
328  // Make Scourge Archer search for a target in range and attack it
329  DoAttackArchersTarget();
330  m_uiArcherCheckTargetTimer = 2000;
331  }
332  else
333  m_uiArcherCheckTargetTimer -= uiDiff;
334  }
335  }
