75 bool EffectDummyCreature_npc_restless_frostborn(Unit* pCaster, uint32 uiSpellId, SpellEffectIndex uiEffIndex, Creature* pCreatureTarget, ObjectGuid /*originalCasterGuid*/)
76 {
77  if (uiSpellId == SPELL_BLOW_HODIRS_HORN && uiEffIndex == EFFECT_INDEX_0 && !pCreatureTarget->IsAlive() && pCaster->GetTypeId() == TYPEID_PLAYER)
78  {
79  uint32 uiCredit = 0;
80  uint32 uiSpawnSpell = 0;
81  switch (pCreatureTarget->GetEntry())
82  {
83  case NPC_NIFFELEM_FOREFATHER:
84  uiCredit = NPC_FROST_GIANT_GHOST_KC;
85  uiSpawnSpell = SPELL_SUMMON_FROST_GIANG_SPIRIT;
86  break;
87  case NPC_FROSTBORN_WARRIOR:
88  uiCredit = NPC_FROST_DWARF_GHOST_KC;
89  uiSpawnSpell = SPELL_SUMMON_FROST_WARRIOR_SPIRIT;
90  break;
91  case NPC_FROSTBORN_GHOST:
92  uiCredit = NPC_FROST_DWARF_GHOST_KC;
93  uiSpawnSpell = SPELL_SUMMON_FROST_GHOST_SPIRIT;
94  break;
95  }
96 
97  // spawn the spirit and give the credit; spirit animation is handled by the script above
98  pCaster->CastSpell(pCaster, uiSpawnSpell, TRIGGERED_OLD_TRIGGERED);
99  ((Player*)pCaster)->KilledMonsterCredit(uiCredit);
100  return true;
101  }
102 
103  return false;
104 }
