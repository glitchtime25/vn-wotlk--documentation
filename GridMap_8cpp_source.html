80 bool GridMap::loadData(char const* filename)
81 {
82  // Unload old data if exist
83  unloadData();
84 
85  GridMapFileHeader header;
86  // Not return error if file not found
87  FILE* in = fopen(filename, "rb");
88  if (!in)
89  {
90  DEBUG_FILTER_LOG(LOG_FILTER_MAP_LOADING, "Failled to found %s", filename);
91  // its a valid error only in case of no vmap files are available too
92  return true;
93  }
94 
95  if (fread(&header, sizeof(header), 1, in) != 1)
96  {
97  sLog.outError("Error loading GridMapFileHeader\n");
98  fclose(in);
99  return false;
100  }
101 
102  if (header.mapMagic == *((uint32 const*)(MAP_MAGIC)) &&
103  header.versionMagic == *((uint32 const*)(MAP_VERSION_MAGIC)) &&
104  IsAcceptableClientBuild(header.buildMagic))
105  {
106  // loadup area data
107  if (header.areaMapOffset && !loadAreaData(in, header.areaMapOffset, header.areaMapSize))
108  {
109  sLog.outError("Error loading map area data\n");
110  fclose(in);
111  return false;
112  }
113 
114  // loadup height data
115  if (header.heightMapOffset && !loadHeightData(in, header.heightMapOffset, header.heightMapSize))
116  {
117  sLog.outError("Error loading map height data\n");
118  fclose(in);
119  return false;
120  }
121 
122  // loadup liquid data
123  if (header.liquidMapOffset && !loadGridMapLiquidData(in, header.liquidMapOffset, header.liquidMapSize))
124  {
125  sLog.outError("Error loading map liquids data\n");
126  fclose(in);
127  return false;
128  }
129 
130  // loadup holes data (if any. check header.holesOffset)
131  if (header.holesOffset && !loadHolesData(in, header.holesOffset, header.holesSize))
132  {
133  sLog.outError("Error loading map holes data\n");
134  fclose(in);
135  return false;
136  }
137 
138  fclose(in);
139  return true;
140  }
141 
142  sLog.outError("Map file '%s' has the wrong version. Please extract the mapfiles again with the latest extractors.", filename);
143  fclose(in);
144  return false;
145 }

309 float GridMap::getHeightFromFloat(float x, float y) const
310 {
311  if (!m_V8 || !m_V9)
312  return INVALID_HEIGHT_VALUE;
313 
314  x = MAP_RESOLUTION * (32 - x / SIZE_OF_GRIDS);
315  y = MAP_RESOLUTION * (32 - y / SIZE_OF_GRIDS);
316 
317  int x_int = (int)x;
318  int y_int = (int)y;
319  x -= x_int;
320  y -= y_int;
321  x_int &= (MAP_RESOLUTION - 1);
322  y_int &= (MAP_RESOLUTION - 1);
323 
324  if (m_holes && isHole(x_int, y_int))
325  return INVALID_HEIGHT_VALUE;
326 
327  // Height stored as: h5 - its v8 grid, h1-h4 - its v9 grid
328  // +--------------> X
329  // | h1-------h2 Coordinates is:
330  // | | \ 1 / | h1 0,0
331  // | | \ / | h2 0,1
332  // | | 2 h5 3 | h3 1,0
333  // | | / \ | h4 1,1
334  // | | / 4 \ | h5 1/2,1/2
335  // | h3-------h4
336  // V Y
337  // For find height need
338  // 1 - detect triangle
339  // 2 - solve linear equation from triangle points
340  // Calculate coefficients for solve h = a*x + b*y + c
341 
342  float a, b, c;
343  // Select triangle:
344  if (x + y < 1)
345  {
346  if (x > y)
347  {
348  // 1 triangle (h1, h2, h5 points)
349  float h1 = m_V9[(x_int) * 129 + y_int];
350  float h2 = m_V9[(x_int + 1) * 129 + y_int];
351  float h5 = 2 * m_V8[x_int * 128 + y_int];
352  a = h2 - h1;
353  b = h5 - h1 - h2;
354  c = h1;
355  }
356  else
357  {
358  // 2 triangle (h1, h3, h5 points)
359  float h1 = m_V9[x_int * 129 + y_int ];
360  float h3 = m_V9[x_int * 129 + y_int + 1];
361  float h5 = 2 * m_V8[x_int * 128 + y_int];
362  a = h5 - h1 - h3;
363  b = h3 - h1;
364  c = h1;
365  }
366  }
367  else
368  {
369  if (x > y)
370  {
371  // 3 triangle (h2, h4, h5 points)
372  float h2 = m_V9[(x_int + 1) * 129 + y_int ];
373  float h4 = m_V9[(x_int + 1) * 129 + y_int + 1];
374  float h5 = 2 * m_V8[x_int * 128 + y_int];
375  a = h2 + h4 - h5;
376  b = h4 - h2;
377  c = h5 - h4;
378  }
379  else
380  {
381  // 4 triangle (h3, h4, h5 points)
382  float h3 = m_V9[(x_int) * 129 + y_int + 1];
383  float h4 = m_V9[(x_int + 1) * 129 + y_int + 1];
384  float h5 = 2 * m_V8[x_int * 128 + y_int];
385  a = h4 - h3;
386  b = h3 + h4 - h5;
387  c = h5 - h4;
388  }
389  }
390  // Calculate height
391  return a * x + b * y + c;
392 }

848 float TerrainInfo::GetHeightStatic(float x, float y, float z, bool useVmaps/*=true*/, float maxSearchDist/*=DEFAULT_HEIGHT_SEARCH*/) const
849 {
850  float mapHeight = VMAP_INVALID_HEIGHT_VALUE; // Store Height obtained by maps
851  float vmapHeight = VMAP_INVALID_HEIGHT_VALUE; // Store Height obtained by vmaps (in "corridor" of z (or slightly above z)
852 
853  // find raw .map surface under Z coordinates (or well-defined above)
854  if (GridMap* gmap = const_cast<TerrainInfo*>(this)->GetGrid(x, y))
855  mapHeight = gmap->getHeight(x, y);
856 
857  if (useVmaps)
858  {
859  if (m_vmgr->isHeightCalcEnabled())
860  {
861  float z2 = z + 2.0f;
862 
863  // if mapHeight has been found search vmap height at least until mapHeight point
864  // this prevent case when original Z "too high above ground and vmap height search fail"
865  // this will not affect most normal cases (no map in instance, or stay at ground at continent)
866  if (mapHeight > INVALID_HEIGHT && z2 - mapHeight > maxSearchDist)
867  maxSearchDist = z2 - mapHeight + 1.0f; // 1.0 make sure that we not fail for case when map height near but above for vamp height
868 
869  // look from a bit higher pos to find the floor
870  vmapHeight = m_vmgr->getHeight(GetMapId(), x, y, z2, maxSearchDist);
871 
872  // if not found in expected range, look for infinity range (case of far above floor, but below terrain-height)
873  if (vmapHeight <= INVALID_HEIGHT)
874  vmapHeight = m_vmgr->getHeight(GetMapId(), x, y, z2, 10000.0f);
875 
876  // look upwards
877  if (vmapHeight <= INVALID_HEIGHT && mapHeight > z2 && std::abs(z2 - mapHeight) > 30.f)
878  vmapHeight = m_vmgr->getHeight(GetMapId(), x, y, z2, -maxSearchDist);
879 
880  // still not found, look near terrain height
881  if (vmapHeight <= INVALID_HEIGHT && mapHeight > INVALID_HEIGHT && z2 < mapHeight)
882  vmapHeight = m_vmgr->getHeight(GetMapId(), x, y, mapHeight + 2.0f, DEFAULT_HEIGHT_SEARCH);
883  }
884  }
885 
886  // mapHeight set for any above raw ground Z or <= INVALID_HEIGHT
887  // vmapheight set for any under Z value or <= INVALID_HEIGHT
888  if (vmapHeight > INVALID_HEIGHT)
889  {
890  if (mapHeight > INVALID_HEIGHT)
891  {
892  // we have mapheight and vmapheight and must select more appropriate
893 
894  // vmap height above map height
895  // or if the distance of the vmap height is less the land height distance
896  if (vmapHeight > mapHeight || fabs(mapHeight - z) > fabs(vmapHeight - z))
897  return vmapHeight;
898  return mapHeight;
899  // better use .map surface height
900  }
901  return vmapHeight; // we have only vmapHeight (if have)
902  }
903 
904  return mapHeight;
