159  void ReceiveAIEvent(AIEventType eventType, Unit* /*sender*/, Unit* /*invoker*/, uint32 miscValue) override
160  {
161  switch (eventType)
162  {
163  case AI_EVENT_CUSTOM_A:
164  m_creature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
165  m_creature->SetVirtualItem(VIRTUAL_ITEM_SLOT_0, 0);
166  break;
167  case AI_EVENT_CUSTOM_B:
168  switch (miscValue)
169  {
170  // eventAi must make sure clones cast spells when each of them die
171  case SPELL_FIRE_DIES:
172  case SPELL_ARCANE_DIES:
173  case SPELL_FROST_DIES:
174  ++m_uiCloneDeadCount;
175 
176  // After the first clone from each split phase is dead start the achiev timer
177  if (m_uiCloneDeadCount == 1 || m_uiCloneDeadCount == 4)
178  {
179  m_bCanCheckAchiev = true;
180  m_uiPersonalityTimer = 0;
181  }
182 
183  // All clones dead
184  if (m_uiCloneDeadCount == 3 || m_uiCloneDeadCount == 6)
185  {
186  ResetTimer(TELESTRA_SPAWN_BACK_IN, 6000);
187 
188  // Check if it took longer than 5 sec
189  if (m_uiPersonalityTimer > 5000)
190  {
191  if (m_instance)
192  m_instance->SetSpecialAchievementCriteria(TYPE_ACHIEV_SPLIT_PERSONALITY, false);
193  }
194  m_bCanCheckAchiev = false;
195  }
196  break;
197 
198  default:
199  break;
200  }
201  break;
202 
203  default:
204  break;
205  }
206  }

222  void HandlePersonalityMerge()
223  {
224  if (DoCastSpellIfCan(m_creature, SPELL_SPAWN_BACK_IN, CAST_TRIGGERED) == CAST_OK)
225  {
226  m_creature->RemoveAurasDueToSpell(SPELL_SUMMON_CLONES);
227  m_creature->RemoveAurasDueToSpell(SPELL_FIRE_DIES);

229  m_creature->RemoveAurasDueToSpell(SPELL_FROST_DIES);
230 
231  m_creature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
232  m_creature->LoadEquipment(m_creature->GetCreatureInfo()->EquipmentTemplateId, true);
233 
234  DoScriptText(SAY_MERGE, m_creature);
235 
236  ResetCombatAction(TELESTRA_ACTION_GRAVITY_WELL, 10000);
237  ResetCombatAction(TELESTRA_ACTION_ICE_NOVA, 15000);
238  ResetCombatAction(TELESTRA_ACTION_FIRE_BOMB, urand(1000, 2000));

240  SetCombatMovement(true);

244  void ExecuteAction(uint32 action) override
245  {
246  switch (action)
247  {
248  case TELESTRA_ACTION_FIRE_BOMB:
249  if (DoCastSpellIfCan(m_creature->GetVictim(), m_isRegularMode ? SPELL_FIREBOMB : SPELL_FIREBOMB_H) == CAST_OK)
250  ResetCombatAction(action, 2000);
251  break;
252  case TELESTRA_ACTION_ICE_NOVA:
253  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_ICE_NOVA : SPELL_ICE_NOVA_H) == CAST_OK)
254  ResetCombatAction(action, urand(10000, 15000));
255  break;
256  case TELESTRA_ACTION_GRAVITY_WELL:
257  if (DoCastSpellIfCan(m_creature, SPELL_GRAVITY_WELL) == CAST_OK)
258  ResetCombatAction(action, urand(10000, 15000));
259  break;
260  case TELESTRA_ACTION_CLONES:
261  if (m_creature->GetHealthPercent() < 50.0f)
262  {
263  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_CLONES, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
264  {
265  DoScriptText(urand(0, 1) ? SAY_SPLIT_1 : SAY_SPLIT_2, m_creature);
266 
267  SetActionReadyStatus(action, false);
268  ResetCombatAction(TELESTRA_ACTION_TELEPORT, 10000);
269  DisableCombatAction(TELESTRA_ACTION_GRAVITY_WELL);
270  DisableCombatAction(TELESTRA_ACTION_ICE_NOVA);
271  DisableCombatAction(TELESTRA_ACTION_FIRE_BOMB);
272 
273  SetCombatMovement(false);
274  }
275  }
276  break;
277  case TELESTRA_ACTION_CLONES_H:
278  if (m_creature->GetHealthPercent() < 15.0f)
279  {
280  if (DoCastSpellIfCan(m_creature, SPELL_SUMMON_CLONES, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
281  {
282  DoScriptText(urand(0, 1) ? SAY_SPLIT_1 : SAY_SPLIT_2, m_creature);
283 
284  SetActionReadyStatus(action, false);
285  ResetCombatAction(TELESTRA_ACTION_TELEPORT, 10000);
286  DisableCombatAction(TELESTRA_ACTION_GRAVITY_WELL);
287  DisableCombatAction(TELESTRA_ACTION_ICE_NOVA);
288  DisableCombatAction(TELESTRA_ACTION_FIRE_BOMB);
289 
290  SetCombatMovement(false);
291  }
292  }
293  break;
294  case TELESTRA_ACTION_TELEPORT:
295  m_creature->NearTeleportTo(aRoomCenterCoords[0], aRoomCenterCoords[1], aRoomCenterCoords[2], m_creature->GetOrientation());
296  DisableCombatAction(action);
297  break;
298  }
299  }
