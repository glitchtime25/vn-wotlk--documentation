50 struct ServerPktHeader
51 {
55  ServerPktHeader(uint32 size, uint16 cmd) : size(size)
56  {
57  uint8 headerIndex = 0;
58  if (isLargePacket())
59  {
60  DEBUG_LOG("initializing large server to client packet. Size: %u, cmd: %u", size, cmd);
61  header[headerIndex++] = 0x80 | (0xFF & (size >> 16));
62  }
63  header[headerIndex++] = 0xFF & (size >> 8);
64  header[headerIndex++] = 0xFF & size;
65 
66  header[headerIndex++] = 0xFF & cmd;
67  header[headerIndex] = 0xFF & (cmd >> 8);
68  }
69 
70  uint8 getHeaderLength() const
71  {
72  // cmd = 2 bytes, size= 2||3bytes
73  return 2 + (isLargePacket() ? 3 : 2);
74  }
75 
76  bool isLargePacket() const
77  {
78  return size > 0x7FFF;
79  }
80 
81  const uint32 size;
82  uint8 header[5];
83 };

121 {
122  if (IsClosed())
123  return;
124 
125  if (sPacketLog->CanLogPacket() && IsLoggingPackets())

127 
128  // Dump outgoing packet.

133 
134  ServerPktHeader header(pct.size() + 2, pct.GetOpcode());

136 
137  if (!pct.empty())

139  else
140  Write(reinterpret_cast<const char*>(&header.header), header.getHeaderLength());
141 
142  if (immediate)
143  ForceFlushOut();
144 
145  m_opcodeHistoryOut.push_front(uint32(pct.GetOpcode()));
146  if (m_opcodeHistoryOut.size() > 50)
147  m_opcodeHistoryOut.resize(30);

173 bool WorldSocket::ProcessIncomingData()
174 {
175  ClientPktHeader header;
176 
177  if (m_useExistingHeader)
178  {
179  m_useExistingHeader = false;
180  header = m_existingHeader;
181 
182  ReadSkip(sizeof(ClientPktHeader));
183  }
184  else
185  {
186  if (!Read((char*)&header, sizeof(ClientPktHeader)))
187  {
188  errno = EBADMSG;
189  return false;
190  }
191 
192  // thread safe due to always being called from service context
193  m_crypt.DecryptRecv((uint8*)&header, sizeof(ClientPktHeader));
194 
195  EndianConvertReverse(header.size);
196  EndianConvert(header.cmd);
197  }
198 
199  // there must always be at least four bytes for the opcode,
200  // and 0x2800 is the largest supported buffer in the client
201  if ((header.size < 4) || (header.size > 0x2800) || (header.cmd >= NUM_MSG_TYPES))
202  {
203  sLog.outError("WorldSocket::ProcessIncomingData: client sent malformed packet size = %u , cmd = %u", header.size, header.cmd);
204 
205  errno = EINVAL;
206  return false;
207  }
208 
209  // the minus four is because we've already read the four byte opcode value
210  const uint16 validBytesRemaining = header.size - 4;
211 
212  // check if the client has told us that there is more data than there is
213  if (validBytesRemaining > ReadLengthRemaining())
214  {
215  // we must preserve the decrypted header so as not to corrupt the crypto state, and to prevent duplicating work
216  m_useExistingHeader = true;
217  m_existingHeader = header;
218 
219  // we move the read pointer backward because it will be skipped again later. this is a slight kludge, but to solve
220  // it more elegantly would require introducing protocol awareness into the socket library, which we want to avoid
221  ReadSkip(-static_cast<int>(sizeof(ClientPktHeader)));
222 
223  errno = EBADMSG;
224  return false;
225  }
226 
227  const Opcodes opcode = static_cast<Opcodes>(header.cmd);
228 
229  if (IsClosed())
230  return false;
231 
232  std::unique_ptr<WorldPacket> pct(new WorldPacket(opcode, validBytesRemaining));
233 
234  if (validBytesRemaining)
235  {
236  pct->append(InPeak(), validBytesRemaining);
237  ReadSkip(validBytesRemaining);
238  }
239 
240  if (sPacketLog->CanLogPacket() && IsLoggingPackets())
241  sPacketLog->LogPacket(*pct, CLIENT_TO_SERVER, GetRemoteIpAddress(), GetRemotePort());
242 
243  sLog.outWorldPacketDump(GetRemoteEndpoint().c_str(), pct->GetOpcode(), pct->GetOpcodeName(), *pct, true);
244 
245  if (WorldSocket::m_packetCooldowns[opcode])
246  {
247  auto now = std::chrono::time_point_cast<std::chrono::milliseconds>(Clock::now());
248  if (now < m_lastPacket[opcode]) // packet on cooldown
249  return true;
250  else // start cooldown and allow execution
251  m_lastPacket[opcode] = now + std::chrono::milliseconds(WorldSocket::m_packetCooldowns[opcode]);
252  }
253 
254  try
255  {
256  switch (opcode)
257  {
258  case CMSG_AUTH_SESSION:
259  if (m_session)
260  {
261  sLog.outError("WorldSocket::ProcessIncomingData: Player send CMSG_AUTH_SESSION again");
262  return false;
263  }
264 
265  return HandleAuthSession(*pct);
266 
267  case CMSG_PING:
268  return HandlePing(*pct);
269 
270  case CMSG_KEEP_ALIVE:
271  DEBUG_LOG("CMSG_KEEP_ALIVE ,size: " SIZEFMTD " ", pct->size());
272 
273  return true;
274 
275  case CMSG_TIME_SYNC_RESP:
276  pct->SetReceivedTime(std::chrono::steady_clock::now());
277  default:
278  {
279  m_opcodeHistoryInc.push_front(uint32(pct->GetOpcode()));
280  if (m_opcodeHistoryInc.size() > 50)
281  m_opcodeHistoryInc.resize(30);
282 
283  if (!m_session)
284  {
285  sLog.outError("WorldSocket::ProcessIncomingData: Client not authed opcode = %u", uint32(opcode));
286  return false;
287  }
288 
289  m_session->QueuePacket(std::move(pct));
290 
291  return true;
292  }
293  }
294  }
295  catch (ByteBufferException&)
296  {
297  sLog.outError("WorldSocket::ProcessIncomingData ByteBufferException occured while parsing an instant handled packet (opcode: %u) from client %s, accountid=%i.",
298  opcode, GetRemoteAddress().c_str(), m_session ? m_session->GetAccountId() : -1);
299 
300  if (sLog.HasLogLevelOrHigher(LOG_LVL_DEBUG))
301  {
302  DEBUG_LOG("Dumping error-causing packet:");
303  pct->hexlike();
304  }
305 
306  if (sWorld.getConfig(CONFIG_BOOL_KICK_PLAYER_ON_BAD_PACKET))
307  {
308  DETAIL_LOG("Disconnecting session [account id %i / address %s] for badly formatted packet.",
309  m_session ? m_session->GetAccountId() : -1, GetRemoteAddress().c_str());
310  return false;
311  }
312  }
313 
314  return true;
315 }

317 bool WorldSocket::HandleAuthSession(WorldPacket& recvPacket)
318 {
319  // NOTE: ATM the socket is singlethread, have this in mind ...
320  uint8 digest[20];
321  uint32 clientSeed;
322  uint32 ClientBuild;
323  uint32 unk2;
324  LocaleConstant locale;
325  std::string account, os;
326  Sha1Hash sha1;
327  BigNumber v, s, g, N, K;
328  WorldPacket packet;
329 
330  // Read the content of the packet
331  recvPacket >> ClientBuild;
332  recvPacket >> unk2;
333  recvPacket >> account;
334  recvPacket.read_skip<uint32>();
335  recvPacket >> clientSeed;
336  recvPacket.read_skip<uint32>();
337  recvPacket.read_skip<uint32>();
338  recvPacket.read_skip<uint32>();
339  recvPacket.read_skip<uint64>();

341 
342  DEBUG_LOG("WorldSocket::HandleAuthSession: client build %u, account %s, clientseed %X",
343  ClientBuild,
344  account.c_str(),
345  clientSeed);
346 
347  // Check the version of client trying to connect
348  if (!IsAcceptableClientBuild(ClientBuild))
349  {
350  packet.Initialize(SMSG_AUTH_RESPONSE, 1);
351  packet << uint8(AUTH_VERSION_MISMATCH);
352 
353  SendPacket(packet);
354 
355  sLog.outError("WorldSocket::HandleAuthSession: Sent Auth Response (version mismatch).");
356  return false;
357  }
358 
359  // Get the account information from the realmd database
360  std::string safe_account = account; // Duplicate, else will screw the SHA hash verification below
361  LoginDatabase.escape_string(safe_account);
362  // No SQL injection, username escaped.
363 
364  QueryResult* result =
365  LoginDatabase.PQuery("SELECT "
366  "a.id, " //0
367  "gmlevel, " //1
368  "sessionkey, " //2
369  "lockedIp, " //3
370  "locked, " //4
371  "v, " //5
372  "s, " //6
373  "expansion, " //7
374  "mutetime, " //8
375  "locale, " //9
376  "os, " //10
377  "flags, " //11
378  "platform " //12
379  "FROM account a "
380  "WHERE username = '%s'",
381  safe_account.c_str());
382 
383  // Stop if the account is not found
384  if (!result)
385  {
386  packet.Initialize(SMSG_AUTH_RESPONSE, 1);
387  packet << uint8(AUTH_UNKNOWN_ACCOUNT);
388 
389  SendPacket(packet);
390 
391  sLog.outError("WorldSocket::HandleAuthSession: Sent Auth Response (unknown account).");
392  return false;
393  }
394 
395  Field* fields = result->Fetch();
396 
397  N.SetHexStr("894B645E89E1535BBDAD5B8B290650530801B18EBFBF5E8FAB3C82872A3E9BB7");
398  g.SetDword(7);
399 
400  v.SetHexStr(fields[5].GetString());
401  s.SetHexStr(fields[6].GetString());
402  m_s = s;
403 
404  const char* sStr = s.AsHexStr(); // Must be freed by OPENSSL_free()
405  const char* vStr = v.AsHexStr(); // Must be freed by OPENSSL_free()
406 
407  DEBUG_LOG("WorldSocket::HandleAuthSession: (s,v) check s: %s v: %s",
408  sStr,
409  vStr);
410 
411  OPENSSL_free((void*) sStr);
412  OPENSSL_free((void*) vStr);
413 
415  if (fields[4].GetUInt8() == 1) // if ip is locked
416  {
417  if (strcmp(fields[3].GetString(), GetRemoteAddress().c_str()))
418  {
419  packet.Initialize(SMSG_AUTH_RESPONSE, 1);
420  packet << uint8(AUTH_FAILED);
421  SendPacket(packet);
422 
423  delete result;
424  BASIC_LOG("WorldSocket::HandleAuthSession: Sent Auth Response (Account IP differs).");
425  return false;
426  }
427  }
428 
429  uint32 id = fields[0].GetUInt32();
430  uint32 security = fields[1].GetUInt16();
431  if (security > SEC_ADMINISTRATOR) // prevent invalid security settings in DB
432  security = SEC_ADMINISTRATOR;
433 
434  uint8 maxServerExpansion = sWorld.getConfig(CONFIG_UINT32_EXPANSION);
435  uint8 currentServerExpansion = sWorldState.GetExpansion();
436  uint8 playerAddonLevel = fields[7].GetUInt8();
437  uint8 expansion;
438  if (security >= SEC_GAMEMASTER)
439  expansion = std::min(playerAddonLevel, maxServerExpansion);
440  else
441  expansion = std::min(playerAddonLevel, currentServerExpansion);
442 
443  K.SetHexStr(fields[2].GetString());
444 
445  time_t mutetime = time_t (fields[8].GetUInt64());
446 
447  locale = GetLocaleByName(fields[9].GetString());
448 
449  os = fields[10].GetString();
450 
451  uint32 accountFlags = fields[11].GetUInt32();
452  std::string platform = fields[12].GetString();
453 
454  delete result;
455 
456  // Re-check account ban (same check as in realmd)
457  QueryResult* banresult =
458  LoginDatabase.PQuery("SELECT 1 FROM account_banned WHERE account_id = %u AND active = 1 AND (expires_at > UNIX_TIMESTAMP() OR expires_at = banned_at)"
459  "UNION "
460  "SELECT 1 FROM ip_banned WHERE (expires_at = banned_at OR expires_at > UNIX_TIMESTAMP()) AND ip = '%s'",
461  id, GetRemoteAddress().c_str());
462 
463  if (banresult) // if account banned
464  {
465  packet.Initialize(SMSG_AUTH_RESPONSE, 1);
466  packet << uint8(AUTH_BANNED);
467  SendPacket(packet);
468 
469  delete banresult;
470 
471  sLog.outError("WorldSocket::HandleAuthSession: Sent Auth Response (Account banned).");
472  return false;
473  }
474 
475  // Check locked state for server
476  AccountTypes allowedAccountType = sWorld.GetPlayerSecurityLimit();
477 
478  if (allowedAccountType > SEC_PLAYER && AccountTypes(security) < allowedAccountType)
479  {
480  WorldPacket Packet(SMSG_AUTH_RESPONSE, 1);
481  Packet << uint8(AUTH_UNAVAILABLE);
482 
483  SendPacket(packet);
484 
485  BASIC_LOG("WorldSocket::HandleAuthSession: User tries to login but his security level is not enough");
486  return false;
487  }
488 
489  // Check that Key and account name are the same on client and server
490  Sha1Hash sha;
491 
492  uint32 t = 0;
493  uint32 seed = m_seed;
494 
495  sha.UpdateData(account);
496  sha.UpdateData((uint8*) & t, 4);
497  sha.UpdateData((uint8*) & clientSeed, 4);
498  sha.UpdateData((uint8*) & seed, 4);
499  sha.UpdateBigNumbers(&K, nullptr);

501 
502  if (memcmp(sha.GetDigest(), digest, 20))
503  {
504  packet.Initialize(SMSG_AUTH_RESPONSE, 1);
505  packet << uint8(AUTH_FAILED);
506 
507  SendPacket(packet);
508 
509  sLog.outError("WorldSocket::HandleAuthSession: Sent Auth Response (authentification failed), account ID: %u.", id);
510  return false;
511  }
512 
513  const std::string& address = GetRemoteAddress();
514 
515  DEBUG_LOG("WorldSocket::HandleAuthSession: Client '%s' authenticated successfully from %s.",
516  account.c_str(),
517  address.c_str());
518 
519  ClientOSType clientOS;
520  if (os == "Win")
521  clientOS = CLIENT_OS_WIN;
522  else if (os == "OSX")
523  clientOS = CLIENT_OS_MAC;
524  else

538

540 
541  ClientPlatformType clientPlatform;
542  if (platform == "x86")
543  clientPlatform = CLIENT_PLATFORM_X86;
544  else if (platform == "PPC" && clientOS == CLIENT_OS_MAC)
545  clientPlatform = CLIENT_PLATFORM_PPC;
546  else
547  {
548  sLog.outError("WorldSocket::HandleAuthSession: Unrecognized platform '%s' for account '%s' from %s", platform.c_str(), account.c_str(), address.c_str());
549  return false;
550  }
551 
552  if (m_session)
553  {
554  // Session exist so player is reconnecting
555  // check if we can request a new socket
556  if (!m_session->RequestNewSocket(this))
557  return false;
558 
559  uint32 counter = 0;
560 
561  // wait session going to be ready
562  while (m_session->GetState() != WORLD_SESSION_STATE_CHAR_SELECTION)

566  std::this_thread::sleep_for(std::chrono::milliseconds(1));
567

569  return false;
570  }
571 
572  m_session->SetGameBuild(ClientBuild);
573  m_session->SetOS(clientOS);
574  m_session->SetPlatform(clientPlatform);
575 
576  m_session->SendAuthOk();
577 
578  std::unique_ptr<SessionAnticheatInterface> anticheat = sAnticheatLib->NewSession(m_session, K);
579 
580  // when false, the client sent invalid addon data. kick!
581  WorldPacket addonPacket; // yes its copypasted atm cos of reconnect
582  if (!anticheat->ReadAddonInfo(&recvPacket, addonPacket))
583  {
584  sLog.outBasic("WorldSocket::HandleAuthSession: Account %s (id %u) IP %s sent bad addon info. Kicking.",
585  account.c_str(), id, address.c_str());
586  return false;
587  }
588  else
589  SendPacket(addonPacket);
590

592  m_session->GetMessager().AddMessage([](WorldSession* session)

594  session->AssignAnticheat();
595  });
596  }
597  else
598  {
599  uint32 otherRaf = 0;
600  bool isRecruiter = false;
601  {
602  std::unique_ptr<QueryResult> result(LoginDatabase.PQuery("SELECT referrer, referred FROM account_raf WHERE referrer=%u OR referred=%u", id, id));

606  uint32 recruiter = fields[0].GetUInt32();

608  {

611  }
612  else
613  otherRaf = recruiter;
614  }
615  }
616 
617  // new session
618  if (!(m_session = new WorldSession(id, this, AccountTypes(security), expansion, mutetime, locale, account, accountFlags, otherRaf, isRecruiter)))
619  return false;
620 
621  m_session->LoadGlobalAccountData();
622  m_session->LoadTutorialsData();
623  m_session->SetGameBuild(ClientBuild);
624  m_session->SetOS(clientOS);
625  m_session->SetPlatform(clientPlatform);
626  m_session->InitializeAnticheat(K);
627 
628  m_session->SendAuthOk();
629 
630  // when false, the client sent invalid addon data. kick!
631  WorldPacket addonPacket;
632  if (!m_session->GetAnticheat()->ReadAddonInfo(&recvPacket, addonPacket))
633  {
634  sLog.outBasic("WorldSocket::HandleAuthSession: Account %s (id %u) IP %s sent bad addon info. Kicking.",
635  account.c_str(), id, address.c_str());
636  return false;
637  }
638  else
639  SendPacket(addonPacket);
640 
641  sWorld.AddSession(m_session);
642  }
643 
644  return true;
645 }

647 bool WorldSocket::HandlePing(WorldPacket& recvPacket)
648 {
649  uint32 ping;
650  uint32 latency;
651 
652  // Get the ping packet content
653  recvPacket >> ping;
654  recvPacket >> latency;
655 
656  if (m_lastPingTime == std::chrono::system_clock::time_point::min())
657  m_lastPingTime = std::chrono::system_clock::now(); // for 1st ping
658  else
659  {
660  auto now = std::chrono::system_clock::now();
661  std::chrono::seconds seconds = std::chrono::duration_cast<std::chrono::seconds>(now - m_lastPingTime);
662  m_lastPingTime = now;
663 
664  if (seconds.count() < 27)
665  {
666  ++m_overSpeedPings;
667 
668  const uint32 max_count = sWorld.getConfig(CONFIG_UINT32_MAX_OVERSPEED_PINGS);
669 
670  if (max_count && m_overSpeedPings > max_count)
671  {
672  if (m_session && m_session->GetSecurity() == SEC_PLAYER)
673  {
674  sLog.outError("WorldSocket::HandlePing: Player kicked for "
675  "overspeeded pings address = %s",
676  GetRemoteAddress().c_str());
677  return false;
678  }
679  }
680  }
681  else
682  m_overSpeedPings = 0;
683  }
684 
685  // critical section
686  {
687  if (m_session)
688  {
689  m_session->SetLatency(latency);
690  m_session->ResetClientTimeDelay();
691  }
692  else
693  {
694  sLog.outError("WorldSocket::HandlePing: peer sent CMSG_PING, "
