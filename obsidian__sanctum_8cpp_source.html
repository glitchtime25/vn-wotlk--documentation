45 void instance_obsidian_sanctum::OnCreatureCreate(Creature* pCreature)
46 {
47  switch (pCreature->GetEntry())
48  {
49  // The three dragons below set to active state once created.
50  // We must expect bigger raid to encounter main boss, and then three dragons must be active due to grid differences
51  case NPC_TENEBRON:
52  case NPC_SHADRON:
53  case NPC_VESPERON:
54  pCreature->SetActiveObjectState(true);
55  case NPC_SARTHARION:
56  case NPC_VESPERON_CONTROLLER:
57  case NPC_VESPERON_CONTROLLER_DEBUFF_CLEAR:
58  case NPC_TENEBRON_EGG_CONTROLLER:
59  m_npcEntryGuidStore[pCreature->GetEntry()] = pCreature->GetObjectGuid();
60  break;
61  case NPC_FIRE_CYCLONE:
62  m_lFireCycloneGuidList.push_back(pCreature->GetObjectGuid());
63  break;
64  }
65 }

67 void instance_obsidian_sanctum::SetData(uint32 uiType, uint32 uiData)
68 {
69  if (uiType == TYPE_SARTHARION_EVENT)
70  {
71  m_auiEncounter[0] = uiData;
72  if (uiData == IN_PROGRESS)
73  m_sVolcanoBlowFailPlayers.clear();
74  }
75  else if (uiType == TYPE_ALIVE_DRAGONS)
76  m_uiAliveDragons = uiData;
77  else if (uiType == TYPE_VOLCANO_BLOW_FAILED)
78  {
79  // Insert the players who fail the achiev and haven't been already inserted in the set
80  if (m_sVolcanoBlowFailPlayers.find(uiData) == m_sVolcanoBlowFailPlayers.end())
81  m_sVolcanoBlowFailPlayers.insert(uiData);
82  }
83 
84  // No need to save anything here
85 }

106 bool instance_obsidian_sanctum::CheckAchievementCriteriaMeet(uint32 uiCriteriaId, Player const* pSource, Unit const* /*pTarget*/, uint32 /*uiMiscValue1 = 0*/) const

120  case ACHIEV_CRIT_VOLCANO_BLOW_H:
