355  return m_creature->SelectHostileTarget() && m_creature->GetVictim();
356 
357  // Special handling for PHASE_AIR
358 
359  // Not started combat or evading prevented
360  if (!m_creature->IsInCombat() || m_creature->HasAuraType(SPELL_AURA_MOD_TAUNT))
361  return false;
362 
363  // Check if there are still enemies (players)
364  ThreatList const& threatList = m_creature->getThreatManager().getThreatList();
365  for (auto itr : threatList)
366  {
367  if (itr->getUnitGuid().IsPlayer())
368  return true;

370 
371  // Evade in air-phase
372  EnterEvadeMode();
373  return false;
