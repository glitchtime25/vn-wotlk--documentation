91 /*
92 Chases after target
93 Has these custom features:
94 Backpedaling - when player is standing inside model, mob backs away with 2 second delay, likely for smoothness
95 Fanning - likewise after 2 seconds, all mobs spread out at a given target - has to be coordinated with all targets which attack same victim
96 TODO: Implement variable chase speed for pet chase
97 */
98 enum ChaseMovementMode
99 {
100  CHASE_MODE_NORMAL, // chasing target
101  CHASE_MODE_BACKPEDAL, // collision movement
102  CHASE_MODE_DISTANCING, // running away from melee
103  CHASE_MODE_FANNING, // mob collision movement
104 };
