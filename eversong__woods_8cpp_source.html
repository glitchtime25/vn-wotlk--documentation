133  void DoSpawnChallengers()
134  {
135  for (uint8 i = 0; i < MAX_CHALLENGER; ++i)
136  {
137  if (Creature* pCreature = m_creature->SummonCreature(uiChallengerId[i],
138  fChallengerLoc[i][0], fChallengerLoc[i][1],
139  fChallengerLoc[i][2], fChallengerLoc[i][3],
140  TEMPSPAWN_TIMED_OOC_OR_DEAD_DESPAWN, 600000))
141  {
142  m_aChallengerGuids[i] = pCreature->GetObjectGuid();
143  pCreature->SetFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SPAWNING);
144  }

148  void UpdateAI(const uint32 uiDiff) override
149  {
150  if (m_bIsEventInProgress)
151  {
152  if (m_uiTimeOutTimer)
153  {
154  if (m_uiTimeOutTimer <= uiDiff)
155  {
156  if (!m_playerGuid) // player are expected to use GO within a minute, if not, event will fail.
157  {
158  Reset();
159  return;
160  }
161  m_uiTimeOutTimer = 0;
162  }
163  else
164  m_uiTimeOutTimer -= uiDiff;
165  }
166 
167  if (m_uiCheckAliveStateTimer < uiDiff)
168  {
169  Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid);
170  if (!pPlayer || !pPlayer->IsAlive() || pPlayer->GetDistance(m_creature) > 100.f)
171  {
172  Reset();
173  return;
174  }
175 
176  Creature* pChallenger = m_creature->GetMap()->GetCreature(m_aChallengerGuids[m_uiChallengerCount]);
177  if (pChallenger && !pChallenger->IsAlive())
178  {
179  ++m_uiChallengerCount;
180 
181  // count starts at 0
182  if (m_uiChallengerCount == MAX_CHALLENGER)
183  {
184  pPlayer->RewardPlayerAndGroupAtEventExplored(QUEST_SECOND_TRIAL, m_creature);
185  Reset();
186  return;
187  }
188  m_uiEngageTimer = 15000;
189  }
190  m_uiCheckAliveStateTimer = 2500;
191  }
192  else
193  m_uiCheckAliveStateTimer -= uiDiff;
194 
195  if (m_uiEngageTimer)
196  {
197  if (m_uiEngageTimer <= uiDiff)
198  {
199  Player* pPlayer = m_creature->GetMap()->GetPlayer(m_playerGuid);
200  if (!pPlayer || !pPlayer->IsAlive())
201  {
202  Reset();
203  return;
204  }
205 
206  if (Creature* pCreature = m_creature->GetMap()->GetCreature(m_aChallengerGuids[m_uiChallengerCount]))
207  {
208  DoScriptText(uiSayId[m_uiChallengerCount], m_creature, pPlayer);
209  pCreature->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_SPAWNING);
210  pCreature->AI()->AttackStart(pPlayer);
211  }
212 
213  m_uiEngageTimer = 0;
214  }
215  else
216  m_uiEngageTimer -= uiDiff;
217  }
218  }
219  }
