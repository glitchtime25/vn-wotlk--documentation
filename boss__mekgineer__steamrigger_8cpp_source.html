165  void UpdateAI(const uint32 uiDiff) override
166  {
167  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
168  return;
169 
170  if (m_uiShrinkTimer < uiDiff)
171  {
172  if (DoCastSpellIfCan(m_creature, SPELL_SUPER_SHRINK_RAY) == CAST_OK)
173  m_uiShrinkTimer = 20000;
174  }
175  else
176  m_uiShrinkTimer -= uiDiff;
177 
178  if (m_uiSawBladeTimer < uiDiff)
179  {
180  Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, nullptr, SELECT_FLAG_PLAYER);
181  if (!pTarget)
182  pTarget = m_creature->GetVictim();
183 
184  if (pTarget)
185  {
186  if (DoCastSpellIfCan(pTarget, SPELL_SAW_BLADE) == CAST_OK)
187  m_uiSawBladeTimer = 15000;
188  }
189  }
190  else
191  m_uiSawBladeTimer -= uiDiff;
192 
193  if (m_uiElectrifiedNetTimer < uiDiff)
194  {
195  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
196  {
197  if (DoCastSpellIfCan(pTarget, SPELL_ELECTRIFIED_NET) == CAST_OK)
198  m_uiElectrifiedNetTimer = 10000;
199  }
200  }
201  else
202  m_uiElectrifiedNetTimer -= uiDiff;
203 
204  // On Heroic mode summon a mechanic at each 20 secs
205  if (!m_bIsRegularMode)
206  {
207  if (m_uiBerserkTimer <= uiDiff)
208  {
209  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK, CAST_INTERRUPT_PREVIOUS) == CAST_OK)
210  {
211  m_uiBerserkTimer = 0;
212  }
213  }
214  else
215  m_uiBerserkTimer -= uiDiff;
216 
217  if (m_uiMechanicTimer < uiDiff)
218  {
219  m_creature->CastSpell(nullptr, gnomeSpells[urand(0, 2)], TRIGGERED_OLD_TRIGGERED);
220  m_uiMechanicTimer = 20000;
221  }
222  else
223  m_uiMechanicTimer -= uiDiff;
224  }
225 
226  if (m_creature->GetHealthPercent() < (100 - 25 * m_uiMechanicPhaseCount))
227  {
228  SummonMechanichs();
229  ++m_uiMechanicPhaseCount;
230  }
231 
232  DoMeleeAttackIfReady();
233  }

266  void MoveInLineOfSight(Unit* pWho) override
267  {
268  // Return if already in combat
269  if (m_bCanStartAttack)
270  return;
271 
272  // Don't attack players unless attacked
273  if (pWho->GetEntry() == NPC_STEAMRIGGER)
274  {
275  if (m_pInstance->GetData(TYPE_MEKGINEER_STEAMRIGGER) == IN_PROGRESS)
276  {
277  // Channel the repair spell on Steamrigger
278  // This will also stop creature movement and will allow them to continue to follow the boss after channeling is finished or the boss is out of range
279  if (m_creature->IsWithinDistInMap(pWho, 2 * INTERACTION_DISTANCE))
280  DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_REPAIR : SPELL_REPAIR_H);
281  }
282  }
283  }

286  {
