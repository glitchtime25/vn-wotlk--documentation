78  void ExecuteAction(uint32 action)
79  {
80  switch (action)
81  {
82  case GEDDON_ARMAGEDDON:
83  {
84  if (m_creature->GetHealthPercent() <= 2.0f)
85  {
86  if (DoCastSpellIfCan(nullptr, SPELL_ARMAGEDDON) == CAST_OK)
87  {
88  DoScriptText(EMOTE_SERVICE, m_creature);
89  SetActionReadyStatus(action, false);
90  return;
91  }
92  }
93  break;
94  }
95  case GEDDON_INFERNO:
96  {
97  if (DoCastSpellIfCan(nullptr, SPELL_INFERNO) == CAST_OK)
98  ResetCombatAction(action, 45000);
99  break;
100  }
101  case GEDDON_IGNITE_MANA:
102  {
103  if (DoCastSpellIfCan(nullptr, SPELL_IGNITE_MANA) == CAST_OK)
104  ResetCombatAction(action, 30000);
105  break;
106  }
107  case GEDDON_LIVING_BOMB:
108  {
109  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_LIVING_BOMB, SELECT_FLAG_PLAYER))
110  if (DoCastSpellIfCan(target, SPELL_LIVING_BOMB) == CAST_OK)
111  ResetCombatAction(action, 35000);
112  break;
113  }
114  }
115  }
116 };
