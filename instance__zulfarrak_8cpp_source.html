67 void instance_zulfarrak::SetData(uint32 uiType, uint32 uiData)
68 {
69  switch (uiType)
70  {
71  case TYPE_VELRATHA:
72  case TYPE_GAHZRILLA:
73  case TYPE_ANTUSUL:
74  case TYPE_THEKA:
75  case TYPE_ZUMRAH:
76  case TYPE_CHIEF_SANDSCALP:
77  m_auiEncounter[uiType] = uiData;
78  break;
79  case TYPE_NEKRUM:
80  m_auiEncounter[uiType] = uiData;
81  if (uiData == DONE && GetData(TYPE_SEZZZIZ) == DONE)
82  SetData(TYPE_PYRAMID_EVENT, DONE);
83  break;
84  case TYPE_SEZZZIZ:
85  m_auiEncounter[uiType] = uiData;
86  if (uiData == DONE && GetData(TYPE_NEKRUM) == DONE)
87  SetData(TYPE_PYRAMID_EVENT, DONE);
88  break;
89  case TYPE_PYRAMID_EVENT:
90  m_auiEncounter[uiType] = uiData;
91  if (uiData == IN_PROGRESS)
92  m_uiPyramidEventTimer = 20000;
93  else if (uiData == DONE)
94  m_uiPyramidEventTimer = 0;
95  break;
96  default:
97  return;
98  }
99 
100  if (uiData == DONE)
101  {
102  OUT_SAVE_INST_DATA;
103 
104  std::ostringstream saveStream;
105  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " " << m_auiEncounter[3]
106  << " " << m_auiEncounter[4] << " " << m_auiEncounter[5] << " " << m_auiEncounter[6] << " " << m_auiEncounter[7]
107  << " " << m_auiEncounter[8];
108 
109  m_strInstData = saveStream.str();
110 
111  SaveToDB();
112  OUT_SAVE_INST_DATA_COMPLETE;
113  }
114 }

193 void instance_zulfarrak::Update(uint32 uiDiff)
194 {
195  if (m_uiPyramidEventTimer)
196  {
197  if (m_uiPyramidEventTimer <= uiDiff)
198  {
199  if (m_lPyramidTrollsGuidList.empty())
200  {
201  m_uiPyramidEventTimer = urand(3000, 10000);
202  return;
203  }
204 
205  GuidList::iterator iter = m_lPyramidTrollsGuidList.begin();
206  advance(iter, urand(0, m_lPyramidTrollsGuidList.size() - 1));
207 
208  // Remove the selected troll
209  ObjectGuid selectedGuid = *iter;
210  m_lPyramidTrollsGuidList.erase(iter);
211 
212  // Move the selected troll to the top of the pyramid. Note: the algorythm may be more complicated than this, but for the moment this will do.
213  if (Creature* pTroll = instance->GetCreature(selectedGuid))
214  {
215  // Pick another one if already in combat or already killed
216  if (pTroll->GetVictim() || !pTroll->IsAlive())
217  {
218  m_uiPyramidEventTimer = urand(0, 2) ? urand(3000, 10000) : 1000;
219  return;
220  }
221 
222  float fX, fY, fZ;
223  if (Creature* pBly = GetSingleCreatureFromStorage(NPC_SERGEANT_BLY))
224  {
225  // ToDo: research if there is anything special if these guys die
226  if (!pBly->IsAlive())
227  {
228  m_uiPyramidEventTimer = 0;
229  return;
230  }
231 
232  pBly->GetRandomPoint(pBly->GetPositionX(), pBly->GetPositionY(), pBly->GetPositionZ(), 4.0f, fX, fY, fZ);
233  pTroll->SetWalk(false);
234  pTroll->GetMotionMaster()->MovePoint(0, fX, fY, fZ);
235  }
236  }
237  m_uiPyramidEventTimer = urand(0, 2) ? urand(3000, 10000) : 1000;
238  }
239  else
240  m_uiPyramidEventTimer -= uiDiff;
241  }
242 }
