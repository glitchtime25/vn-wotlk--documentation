91  void ExecuteAction(uint32 action) override
92  {
93  switch (action)
94  {
95  case VENOXIS_PHASE_2:
96  if (m_creature->GetHealthPercent() > 50.f)
97  break;
98 
99  if (DoCastSpellIfCan(nullptr, SPELL_SNAKE_FORM) == CAST_OK)
100  {
101  DoScriptText(SAY_TRANSFORM, m_creature);
102  DoCastSpellIfCan(nullptr, SPELL_POISON_CLOUD); // an instant cloud on change
103  DoCastSpellIfCan(nullptr, SPELL_VIRULENT_POISON_PROC, CAST_TRIGGERED | CAST_AURA_NOT_PRESENT);
104  DoResetThreat();
105  m_creature->SetSpellList(SPELL_LIST_PHASE_2);
106  DisableCombatAction(action);
107  }
108  break;
109  case VENOXIS_PHASE_3:
110  if (m_creature->GetHealthPercent() < 25.f)
111  {
112  m_creature->SetSpellList(SPELL_LIST_PHASE_3);
113  DisableCombatAction(action);
114  }
115  break;
116  case VENOXIS_FRENZY:
117  if (m_creature->GetHealthPercent() < 20.f) // maybe even 25
118  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
119  DisableCombatAction(action);
120  break;
121  }
122  }
123 };
