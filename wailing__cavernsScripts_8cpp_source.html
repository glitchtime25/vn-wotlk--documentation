261  void UpdateEscortAI(const uint32 uiDiff) override
262  {
263  if (m_uiEventTimer)
264  {
265  if (m_uiEventTimer <= uiDiff)
266  {
267  switch (m_uiPoint)
268  {
269  // Corner stop -> raptors
270  case 12:
271  switch (m_uiSubeventPhase)
272  {
273  case 0:
274  // Summon raptors at first stop
275  for (auto& i : raptorLocations)
276  m_creature->SummonCreature(i.uiEntry, i.fX, i.fY, i.fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 20000, true, true, i.uiPathId);
277  m_uiEventTimer = 0;
278  ++m_uiSubeventPhase;
279  break;
280  case 1:
281  // After the summoned mobs are killed continue
282  SetEscortPaused(false);
283  m_uiEventTimer = 0;
284  break;
285  }
286  break;
287  // Circle stop -> vipers
288  case 30:
289  switch (m_uiSubeventPhase)
290  {
291  case 0:
292  DoBroadcastText(SAY_CIRCLE_BANISH, m_creature);
293  DoCastSpellIfCan(m_creature, SPELL_CLEANSING);
294  m_uiEventTimer = 20000;
295  ++m_uiSubeventPhase;
296  break;
297  case 1:
298  // Summon vipers at the first circle
299  for (uint8 i = 0; i < 3; ++i)
300  m_creature->SummonCreature(urand(0, 1) ? NPC_DEVIATE_ADDER : NPC_DEVIATE_VIPER, circleLocations[i][0], circleLocations[i][1], circleLocations[i][2], circleLocations[i][3], TEMPSPAWN_TIMED_OOC_DESPAWN, 20000, true);
301  m_uiEventTimer = 0;
302  ++m_uiSubeventPhase;
303  break;
304  case 2:
305  // Wait for casting to be complete - TODO, if spawned creatures are dead and cast finished we continue without any waiting
306  ++m_uiSubeventPhase;
307  m_uiEventTimer = 10000;
308  break;
309  case 3:
310  m_creature->HandleEmote(EMOTE_ONESHOT_TALK);
311  DoBroadcastText(SAY_PURIFIED, m_creature);
312  m_uiEventTimer = 0;
313  ++m_uiPoint; // Increment this in order to avoid the special case evade
314  SetEscortPaused(false);
315  break;
316  }
317  break;
318  // Chamber stop -> ritual and final boss
319  case 70:
320  switch (m_uiSubeventPhase)
321  {
322  case 0:
323  DoBroadcastText(SAY_BEGIN_RITUAL, m_creature);
324  m_creature->HandleEmote(EMOTE_ONESHOT_BEG);
325  SetCombatScriptStatus(true);
326  SetCombatMovement(false);
327  m_uiEventTimer = 5000;
328  ++m_uiSubeventPhase;
329  break;
330  case 1:
331  DoCastSpellIfCan(m_creature, SPELL_AWAKENING);
332  DoBroadcastText(EMOTE_RITUAL_BEGIN, m_creature);
333  m_uiEventTimer = 3000;
334  ++m_uiSubeventPhase;
335  break;
336  case 2:
337  // First set of mobs
338  for (auto& i : moccasinLocations)
339  m_creature->SummonCreature(i.uiEntry, i.fX, i.fY, i.fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 20000, false, false, i.uiPathId);
340  m_uiEventTimer = 5000;
341  ++m_uiSubeventPhase;
342  break;
343  case 3:
344  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
345  DoBroadcastText(EMOTE_NARALEX_AWAKE, pNaralex);
346  m_uiEventTimer = 0;
347  ++m_uiSubeventPhase;
348  break;
349  case 4:
350  // Second set of mobs
351  for (auto& i : ectoplasmLocations)
352  m_creature->SummonCreature(i.uiEntry, i.fX, i.fY, i.fZ, 0.0f, TEMPSPAWN_DEAD_DESPAWN, 20000, true, false, i.uiPathId);
353  m_uiEventTimer = 20000;
354  ++m_uiSubeventPhase;
355  break;
356  case 5:
357  // Advance only when all mobs are dead
358  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
359  DoBroadcastText(EMOTE_BREAK_THROUGH, pNaralex);
360  ++m_uiSubeventPhase;
361  m_uiEventTimer = 0;
362  break;
363  case 6:
364  // Mutanus
365  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
366  DoBroadcastText(EMOTE_VISION, pNaralex);
367 
368  m_creature->SummonCreature(NPC_MUTANUS, mutanosLocations[0], mutanosLocations[1], mutanosLocations[2], 0.0f, TEMPSPAWN_DEAD_DESPAWN, 20000, true, false, mutanosLocations[3]);
369  m_uiEventTimer = 0;
370  ++m_uiSubeventPhase;
371  break;
372  case 7:
373  // Awaken Naralex after mutanus is defeated
374  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
375  {
376  pNaralex->SetStandState(UNIT_STAND_STATE_STAND);
377  DoBroadcastText(SAY_NARALEX_AWAKE, pNaralex);
378  }
379  m_instance->SetData(TYPE_DISCIPLE, DONE);
380  ++m_uiSubeventPhase;
381  m_uiEventTimer = 5000;
382  break;
383  case 8:
384  m_creature->InterruptNonMeleeSpells(false, SPELL_AWAKENING);
385  m_creature->RemoveAurasDueToSpell(SPELL_AWAKENING);
386  m_creature->HandleEmote(EMOTE_ONESHOT_POINT);
387  DoBroadcastText(SAY_AWAKE, m_creature);
388  m_uiEventTimer = 1000;
389  ++m_uiSubeventPhase;
390  break;
391  case 9:
392  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
393  {
394  pNaralex->HandleEmote(EMOTE_ONESHOT_BOW);
395  DoBroadcastText(SAY_NARALEX_THANKYOU, pNaralex);
396  }
397  m_uiEventTimer = 7000;
398  ++m_uiSubeventPhase;
399  break;
400  case 10:
401  // Say Farewell
402  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
403  {
404  pNaralex->HandleEmote(EMOTE_ONESHOT_TALK);
405  DoBroadcastText(SAY_FAREWELL, pNaralex);
406  }
407  m_uiEventTimer = 3000;
408  ++m_uiSubeventPhase;
409  break;
410  case 11:
411  // Shapeshift into a bird
412  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
413  pNaralex->CastSpell(pNaralex, SPELL_SHAPESHIFT, TRIGGERED_NONE);
414  DoCastSpellIfCan(m_creature, SPELL_SHAPESHIFT);
415  m_uiEventTimer = 8000;
416  ++m_uiSubeventPhase;
417  break;
418  case 12:
419  SetEscortPaused(false);
420  m_creature->SetLevitate(true);
421  m_creature->SetHover(true);
422  SetRun();
423  // Send them flying somewhere outside of the room
424  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
425  {
426  // Set to flying
427  pNaralex->SetLevitate(true);
428  pNaralex->SetHover(true);
429  pNaralex->SetWalk(false);
430 
431  // Set following
432  pNaralex->GetMotionMaster()->MoveFollow(m_creature, 5.0f, 0);
433  }
434  m_uiEventTimer = 30000;
435  break;
436  case 13:
437  if (Creature* pNaralex = m_instance->GetSingleCreatureFromStorage(NPC_NARALEX))
438  pNaralex->ForcedDespawn(1000); // Despawn after some time
439  m_creature->ForcedDespawn(1000);
440  m_instance->DespawnAll(); // whole instance despawns
441  break;
442  }
443  break;
444  }
445  }
446  else
447  m_uiEventTimer -= uiDiff;
448  }
449 
450  if (m_uiPotionTimer < uiDiff)
451  {
452  // if health lower than 80%, cast heal
453  if (m_creature->GetHealthPercent() < 80.0f)
454  {
455  if (DoCastSpellIfCan(m_creature, SPELL_POTION) == CAST_OK)
456  m_uiPotionTimer = 45000;
457  }
458  else
459  m_uiPotionTimer = 5000;
460  }
461  else
462  m_uiPotionTimer -= uiDiff;
463 
464  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
465  return;
466 
467  if (m_uiSleepTimer < uiDiff)
468  {
469  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1))
470  {
471  if (DoCastSpellIfCan(pTarget, SPELL_SLEEP) == CAST_OK)
472  m_uiSleepTimer = 30000;
473  }
474  }
475  else
476  m_uiSleepTimer -= uiDiff;
477 
478  DoMeleeAttackIfReady();
479  }
