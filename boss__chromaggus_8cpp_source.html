95  void ExecuteAction(uint32 action) override
96  {
97  switch (action)
98  {
99  case CHROMAGGUS_ENRAGE:
100  {
101  if (m_creature->GetHealthPercent() < 20.0f)
102  {
103  DoCastSpellIfCan(nullptr, SPELL_ENRAGE);
104  DoScriptText(EMOTE_GENERIC_FRENZY, m_creature);
105  SetActionReadyStatus(action, false);
106  }
107  break;
108  }
109  case CHROMAGGUS_ELEMENTAL_SHIELD:
110  {
111  if (DoCastSpellIfCan(nullptr, SPELL_ELEMENTAL_SHIELD_BWL) == CAST_OK)
112  {
113  DoScriptText(EMOTE_SHIMMER, m_creature);
114  ResetCombatAction(action, 45000);
115  }
116  break;
117  }
118  case CHROMAGGUS_BREATH_LEFT:
119  {
120  if (DoCastSpellIfCan(nullptr, m_breathLeftSpell) == CAST_OK)
121  ResetCombatAction(action, 60000);
122  break;
123  }
124  case CHROMAGGUS_BREATH_RIGHT:
125  {
126  if (DoCastSpellIfCan(nullptr, m_breathRightSpell) == CAST_OK)
127  ResetCombatAction(action, 60000);
128  break;
129  }
130  case CHROMAGGUS_BROOD_AFFLICTION:
131  {
132  if (DoCastSpellIfCan(nullptr, SPELL_BROOD_AFFLICTION) == CAST_OK)
133  ResetCombatAction(action, 7 * IN_MILLISECONDS);
134  break;
135  }
136  case CHROMAGGUS_FRENZY:
137  {
138  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
139  {
140  DoScriptText(EMOTE_GENERIC_FRENZY_KILL, m_creature);
141  ResetCombatAction(action, 15 * IN_MILLISECONDS);
142  }
143  break;
144  }
145  }
146  }
