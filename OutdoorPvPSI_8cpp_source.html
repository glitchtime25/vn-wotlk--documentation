66 // Handle case when player returns a silithyst
67 bool OutdoorPvPSI::HandleAreaTrigger(Player* player, uint32 triggerId)
68 {
69  if (player->IsGameMaster() || player->IsDead())
70  return false;
71 
72  switch (triggerId)
73  {
74  case AREATRIGGER_SILITHUS_ALLIANCE:
75  if (player->GetTeam() != ALLIANCE || !player->HasAura(SPELL_SILITHYST))
76  return false;
77 
78  // update counter
79  ++ m_resourcesAlliance;
80  SendUpdateWorldState(WORLD_STATE_SI_GATHERED_A, m_resourcesAlliance);
81 
82  // handle the case when the faction has reached maximum resources allowed
83  if (m_resourcesAlliance == MAX_SILITHYST)
84  {
85  // NOTE: On retail it would not reset until server restart but we do not support weekly restart :)
86  m_zoneOwner = ALLIANCE;
87  m_resourcesAlliance = 0;
88  m_resourcesHorde = 0;
89 
90  // also update the horde counter if resources were reset
91  SendUpdateWorldState(WORLD_STATE_SI_GATHERED_H, m_resourcesHorde);
92 
93  // apply buff to owner team
94  BuffTeam(ALLIANCE, SPELL_CENARION_FAVOR);
95 
96  // Send defense message
97  sWorld.SendDefenseMessage(ZONE_ID_SILITHUS, LANG_OPVP_SI_CAPTURE_A);
98  }
99 
100  // give quest credit if necessary
101  if (player->GetQuestStatus(QUEST_SCOURING_DESERT_ALLIANCE) == QUEST_STATUS_INCOMPLETE)
102  player->KilledMonsterCredit(NPC_SILITHUS_DUST_QUEST_ALLIANCE);
103  break;
104  case AREATRIGGER_SILITHUS_HORDE:
105  if (player->GetTeam() != HORDE || !player->HasAura(SPELL_SILITHYST))
106  return false;
107 
108  // update counter
109  ++ m_resourcesHorde;
110  SendUpdateWorldState(WORLD_STATE_SI_GATHERED_H, m_resourcesHorde);
111 
112  // handle the case when the faction has reached maximum resources allowed
113  if (m_resourcesHorde == MAX_SILITHYST)
114  {
115  // NOTE: On retail it would not reset until server restart but we do not support weekly restart :)
116  m_zoneOwner = HORDE;
117  m_resourcesAlliance = 0;
118  m_resourcesHorde = 0;
119 
120  // also update the alliance counter if resources were reset
121  SendUpdateWorldState(WORLD_STATE_SI_GATHERED_A, m_resourcesAlliance);
122 
123  // apply buff to owner team
124  BuffTeam(HORDE, SPELL_CENARION_FAVOR);
125 
126  // Send defense message
127  sWorld.SendDefenseMessage(ZONE_ID_SILITHUS, LANG_OPVP_SI_CAPTURE_H);
128  }
129 
130  // give quest credit if necessary
131  if (player->GetQuestStatus(QUEST_SCOURING_DESERT_HORDE) == QUEST_STATUS_INCOMPLETE)
132  player->KilledMonsterCredit(NPC_SILITHUS_DUST_QUEST_HORDE);
133  break;
134  default:
135  return false;
136  }
137 
138  // remove silithyst aura
139  player->RemoveAurasDueToSpell(SPELL_SILITHYST);
140 
141  // reward the player
142  player->CastSpell(player, SPELL_TRACES_OF_SILITHYST, TRIGGERED_OLD_TRIGGERED);
143  player->RewardHonor(nullptr, 1, HONOR_REWARD_SILITHYST);
144  player->GetReputationMgr().ModifyReputation(sFactionStore.LookupEntry(FACTION_CENARION_CIRCLE), REPUTATION_REWARD_SILITHYST);
145 
146  return true;
147 }

162 bool OutdoorPvPSI::HandleDropFlag(Player* player, uint32 spellId)
163 {
164  if (spellId != SPELL_SILITHYST)
165  return false;
166 
167  // don't drop flag at area trigger
168  // we are checking distance from the AT hard-coded coordinates because it's much faster than checking the area trigger store
169  switch (player->GetTeam())
170  {
171  case ALLIANCE:
172  if (player->IsWithinDist3d(silithusFlagDropLocations[0].x, silithusFlagDropLocations[0].y, silithusFlagDropLocations[0].z, 5.0f))
173  return false;
174  break;
175  case HORDE:
176  if (player->IsWithinDist3d(silithusFlagDropLocations[1].x, silithusFlagDropLocations[1].y, silithusFlagDropLocations[1].z, 5.0f))
177  return false;
178  break;
179  default:
180  break;
181  }
182 
183  // drop the flag in other case
184  player->CastSpell(player, SPELL_SILITHYST_FLAG_DROP, TRIGGERED_OLD_TRIGGERED);
185  return true;
186 }
