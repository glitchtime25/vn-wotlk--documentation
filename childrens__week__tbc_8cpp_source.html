229  void MoveInLineOfSight(Unit* who) override
230  {
231  if (triggers.find(who->GetEntry()) != triggers.end() && m_creature->IsWithinDist(who, 20.f))
232  {

239  case NPC_TRIGGER_AERIS_LANDING: questId = QUEST_JHEEL_IS_AT_AERIS_LANDING; break;
240  case NPC_TRIGGER_THRONE_OF_ELEMENTS: questId = QUEST_VISIT_THE_THRONE_OF_THE_ELEMENTS; break;
241  case NPC_TRIGGER_EXODAR_01: questId = QUEST_THE_SEAT_OF_THE_NAARU; break;
242  case NPC_TRIGGER_SILVERMOON_01: questId = QUEST_WHEN_I_GROW_UP; break; // location for old script for future
243  case NPC_TRIGGER_SILVERMOON_02: questId = QUEST_WHEN_I_GROW_UP; break; // location for old script for future
244  case NPC_TRIGGER_CAVERNS_OF_TIME: questId = IsHorde() ? QUEST_TIME_TO_VISIT_THE_CAVERNS_H : QUEST_TIME_TO_VISIT_THE_CAVERNS_A; break;
245  case NPC_TRIGGER_EXODAR_02: questId = QUEST_CALL_ON_THE_FARSEER; break;
246  case NPC_TRIGGER_SILVERMOON_LVL70ETC: questId = QUEST_NOW_WHEN_I_GROW_UP; break;
247  }
248  if (questId)
249  {
250  Player* player = const_cast<Player*>(m_creature->GetControllingPlayer());
251  if (player->GetQuestStatus(questId) == QUEST_STATUS_INCOMPLETE && !player->IsQuestExplored(questId))
252  {
253  player->AreaExploredOrEventHappens(questId);
254  m_eventStage = 0;
255  StartQuestSequence(questId);
256  }
257  }
258  }

335  break;

340  Creature* soldier = nullptr;
341  for (auto creature : soldiers)
342  {
343  if (creature->GetSpawnerGuid() == m_creature->GetSpawnerGuid())

360  soldier->SetFacingToObject(m_creature->GetSpawner());
361  timer = 1500;
