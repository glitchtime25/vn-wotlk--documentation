107  GameObject* SelectNearbyShallowGrave()
108  {
109  if (!m_pInstance)
110  return nullptr;
111 
112  // Get the list of usable graves (not used already by players)
113  GuidList lTempList;
114  GameObjectList lGravesInRange;
115 
116  m_pInstance->GetShallowGravesGuidList(lTempList);
117  for (GuidList::const_iterator itr = lTempList.begin(); itr != lTempList.end(); ++itr)
118  {
119  GameObject* pGo = m_creature->GetMap()->GetGameObject(*itr);
120  // Go spawned and no looting in process
121  if (pGo && pGo->IsSpawned() && pGo->GetLootState() == GO_READY)
122  lGravesInRange.push_back(pGo);
123  }
124 
125  if (lGravesInRange.empty())
126  return nullptr;
127 
128  // Sort the graves
129  lGravesInRange.sort(ObjectDistanceOrder(m_creature));
130 
131  return *lGravesInRange.begin();
132  }

134  void UpdateAI(const uint32 uiDiff) override
135  {
136  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
137  return;
138 
139  if (m_uiSpawnZombieTimer)
140  {
141  if (m_uiSpawnZombieTimer <= uiDiff)
142  {
143  // Use a nearby grave to spawn zombies
144  if (GameObject* pGrave = SelectNearbyShallowGrave())
145  {
146  m_creature->CastSpell(pGrave->GetPositionX(), pGrave->GetPositionY(), pGrave->GetPositionZ(), SPELL_SUMMON_ZOMBIES, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, pGrave->GetObjectGuid());
147  pGrave->SetLootState(GO_JUST_DEACTIVATED);
148 
149  if (roll_chance_i(30))
150  DoScriptText(SAY_SUMMON, m_creature);
151 
152  m_uiSpawnZombieTimer = 20000;
153  }
154  else // No Grave usable any more
155  m_uiSpawnZombieTimer = 0;
156  }
157  else
158  m_uiSpawnZombieTimer -= uiDiff;
159  }
160 
161  if (m_uiShadowBoltTimer < uiDiff)
162  {
163  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
164  {
165  if (DoCastSpellIfCan(pTarget, SPELL_SHADOW_BOLT) == CAST_OK)
166  m_uiShadowBoltTimer = urand(3500, 5000);
167  }
168  }
169  else
170  m_uiShadowBoltTimer -= uiDiff;
171 
172  if (m_uiShadowBoltVolleyTimer < uiDiff)
173  {
174  if (DoCastSpellIfCan(m_creature, SPELL_SHADOW_BOLT_VOLLEY) == CAST_OK)
175  m_uiShadowBoltVolleyTimer = urand(10000, 18000);
176  }
177  else
178  m_uiShadowBoltVolleyTimer -= uiDiff;
179 
180  if (m_uiWardOfZumrahTimer < uiDiff)
181  {
182  if (DoCastSpellIfCan(m_creature, SPELL_WARD_OF_ZUMRAH) == CAST_OK)
183  m_uiWardOfZumrahTimer = urand(15000, 32000);
184  }
185  else
186  m_uiWardOfZumrahTimer -= uiDiff;
187 
188  if (m_uHealingWaveTimer < uiDiff)
189  {
190  if (Unit* pTarget = DoSelectLowestHpFriendly(40.0f))
191  {
192  if (DoCastSpellIfCan(pTarget, SPELL_HEALING_WAVE) == CAST_OK)
193  m_uHealingWaveTimer = urand(15000, 23000);
194  }
195  }
196  else
197  m_uHealingWaveTimer -= uiDiff;
198 
199  DoMeleeAttackIfReady();
200  }
