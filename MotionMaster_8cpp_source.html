59 void MotionMaster::Initialize()
60 {
61 #ifdef BUILD_METRICS
62  metric::duration<std::chrono::microseconds> meas("motionmaster.initialize", {
63  { "entry", std::to_string(m_owner->GetEntry()) },
64  { "guid", std::to_string(m_owner->GetGUIDLow()) },
65  { "unit_type", std::to_string(m_owner->GetGUIDHigh()) },
66  { "map_id", std::to_string(m_owner->GetMapId()) },
67  { "instance_id", std::to_string(m_owner->GetInstanceId()) }
68  }, 1000);
69 #endif
70  // stop current move
71  m_owner->StopMoving();
72 
73  // clear ALL movement generators (including default)
74  Clear(false, true);
75 
76  // set new default movement generator
77  if (m_owner->IsCreature() && !m_owner->hasUnitState(UNIT_STAT_POSSESSED))
78  {
79  auto creature = static_cast<Creature*>(m_owner);
80  m_currentPathId = m_defaultPathId;
81  MovementGenerator* movement = FactorySelector::selectMovementGenerator(creature);
82  push(movement == nullptr ? &si_idleMovement : movement);
83  top()->Initialize(*m_owner);
84 
85  if (top()->GetMovementGeneratorType() == WAYPOINT_MOTION_TYPE || top()->GetMovementGeneratorType() == LINEAR_WP_MOTION_TYPE)
86  {
87  // check if creature is part of formation and use waypoint path as path origin
88  WaypointPathOrigin pathOrigin = WaypointPathOrigin::PATH_NO_PATH;
89  auto creatureGroup = creature->GetCreatureGroup();
90  if (creatureGroup && creatureGroup->GetFormationEntry() && creatureGroup->GetGroupEntry().GetFormationSlotId(m_owner->GetDbGuid()) == 0)
91  {
92  m_currentPathId = creatureGroup->GetFormationEntry()->MovementIdOrWander;
93  pathOrigin = WaypointPathOrigin::PATH_FROM_WAYPOINT_PATH;
94  }
95 
96  (static_cast<WaypointMovementGenerator<Creature>*>(top()))->InitializeWaypointPath(*creature, m_currentPathId, pathOrigin, 0);
97  }
98  }
99  else
100  push(&si_idleMovement);
101 }

115 void MotionMaster::UpdateMotion(uint32 diff)
116 {
117  if (m_owner->hasUnitState(UNIT_STAT_CAN_NOT_MOVE))
118  return;
119 #ifdef BUILD_METRICS
120  metric::duration<std::chrono::microseconds> meas("motionmaster.updatemotion", {
121  { "entry", std::to_string(m_owner->GetEntry()) },
122  { "guid", std::to_string(m_owner->GetGUIDLow()) },
123  { "unit_type", std::to_string(m_owner->GetGUIDHigh()) },
124  { "map_id", std::to_string(m_owner->GetMapId()) },
125  { "instance_id", std::to_string(m_owner->GetInstanceId()) }
126  }, 1000);
127 #endif
128 
129  MANGOS_ASSERT(!empty());
130  m_cleanFlag |= MMCF_UPDATE;
131 
132  if (!top()->Update(*m_owner, diff))
133  {
134  m_cleanFlag &= ~MMCF_UPDATE;
135  MovementExpired();
136  }
137  else
138  m_cleanFlag &= ~MMCF_UPDATE;
139 
140  if (m_expList)
141  {
142  for (auto mg : *m_expList)
143  {
144  if (!isStatic(mg))
145  delete mg;
146  }
147 
148  delete m_expList;
149  m_expList = nullptr;
150 
151  if (empty())
152  Initialize();
153 
154  if (m_cleanFlag & MMCF_RESET)
155  {
156  top()->Reset(*m_owner);
157  m_cleanFlag &= ~MMCF_RESET;
158  }
159  }
160 }

182 {

205 void MotionMaster::DirectExpire(bool reset)
206 {
207  if (empty() || size() == 1)
208  return;
209 
210  MovementGenerator* curr = top();
211  bool onlyRemoveOne = false; // this is an ugly hack and is indicative of a need for a rework
212  if (curr->GetMovementGeneratorType() == EFFECT_MOTION_TYPE)
213  onlyRemoveOne = true;
214  pop();
215 
216  // also drop stored under top() targeted motions
217  if (!onlyRemoveOne)
218  {
219  while (!empty() && (top()->IsRemovedOnExpire()))
220  {
221  MovementGenerator* temp = top();
222  pop();
223  temp->Finalize(*m_owner);
224  delete temp;
225  }
226  }
227 
228  // Store current top MMGen, as Finalize might push a new MMGen
229  MovementGenerator* nowTop = empty() ? nullptr : top();
230  // it can add another motions instead
231  curr->Finalize(*m_owner);
232 
233  if (!isStatic(curr))
234  delete curr;
235 
236  if (empty())
237  Initialize();
238 
239  // Prevent reseting possible new pushed MMGen
240  if (reset && top() == nowTop)
241  top()->Reset(*m_owner);
242 }

577 bool MotionMaster::MoveFall(ObjectGuid guid/* = ObjectGuid()*/, uint32 relayId/* = 0*/)
578 {
579  const float x = m_owner->GetPositionX(), y = m_owner->GetPositionY(), z = m_owner->GetPositionZ();
580 
581  // use larger distance for vmap height search than in most other cases
582  float tz = m_owner->GetMap()->GetHeight(m_owner->GetPhaseMask(), x, y, z);
583 
584  if (tz <= INVALID_HEIGHT)
585  {
586  DEBUG_LOG("MotionMaster::MoveFall: unable retrive a proper height at map %u (x: %f, y: %f, z: %f).", m_owner->GetMap()->GetId(), x, y, z);
587  return false;
588  }
589 
590  // Abort too if the ground is very near
591  if (fabs(z - tz) < g_moveFallMinFallDistance)
592  return false;
593 
594  Movement::MoveSplineInit init(*m_owner);
595  init.MoveTo(x, y, tz);
596  init.SetFall();
597  Mutate(new FallMovementGenerator(init, EVENT_FALL, false, guid, relayId));
598  return true;
599 }
