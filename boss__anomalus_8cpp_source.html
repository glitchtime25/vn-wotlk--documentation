153  void ExecuteAction(uint32 action) override
154  {
155  switch (action)
156  {
157  case ANOMALUS_ACTION_CHAOTIC_RIFT:
158  if (m_creature->GetHealthPercent() < 50.0f)
159  {
160  // create a rift then set shield up and finally charge rift
161  if (DoCastSpellIfCan(m_creature, SPELL_CREATE_RIFT, CAST_TRIGGERED) == CAST_OK)
162  {
163  // emotes are in this order
164  DoScriptText(SAY_SHIELD, m_creature);
165  DoScriptText(EMOTE_SHIELD, m_creature);
166  DoScriptText(EMOTE_OPEN_RIFT, m_creature);
167 
168  DoCastSpellIfCan(m_creature, SPELL_RIFT_SHIELD, CAST_TRIGGERED);
169  DoCastSpellIfCan(m_creature, SPELL_CHARGE_RIFT, CAST_TRIGGERED);
170 
171  SetActionReadyStatus(action, false);
172  ResetCombatAction(ANOMALUS_ACTION_RESUME_COMBAT, 45000);
173  DisableCombatAction(ANOMALUS_ACTION_SPARK);
174  DisableCombatAction(ANOMALUS_ACTION_CREATE_RIFT);
175  }
176  }
177  break;
178  case ANOMALUS_ACTION_SPARK:
179  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0))
180  {
181  if (DoCastSpellIfCan(pTarget, m_isRegularMode ? SPELL_SPARK : SPELL_SPARK_H) == CAST_OK)
182  ResetCombatAction(action, urand(8000, 10000));
183  }
184  break;
185  case ANOMALUS_ACTION_CREATE_RIFT:
186  if (DoCastSpellIfCan(m_creature, SPELL_CREATE_RIFT) == CAST_OK)
187  {
188  DoScriptText(SAY_RIFT, m_creature);
189  DoScriptText(EMOTE_OPEN_RIFT, m_creature);
190  ResetCombatAction(action, 25000);
191  }
192  break;
193  case ANOMALUS_ACTION_RESUME_COMBAT:
194  DoResumeCombat();
195  DisableCombatAction(action);
196  break;
197  }
198  }
