262  // Wrapper to handle movement to the closest trigger
263  void DoMoveToClosestTrigger(bool ground)
264  {
265  if (!m_instance)
266  return;
267 
268  Unit* chosenTrigger = nullptr;
269  GuidList lTriggersList;
270  float fX, fY, fZ;
271 
272  // get the list of wanted triggers
273  m_instance->GetNightbaneTriggers(lTriggersList, ground);
274 
275  // calculate the closest trigger from the list
276  for (GuidList::const_iterator itr = lTriggersList.begin(); itr != lTriggersList.end(); ++itr)
277  {
278  if (Creature* pTrigger = m_creature->GetMap()->GetCreature(*itr))
279  {
280  if (!chosenTrigger || (ground ? !m_creature->GetDistanceOrder(pTrigger, chosenTrigger, false, DIST_CALC_NONE) : m_creature->GetDistanceOrder(pTrigger, chosenTrigger, false, DIST_CALC_NONE)))
281  chosenTrigger = pTrigger;
282  }
283  }
284 
285  // Move to trigger position
286  if (chosenTrigger)
287  {
288  chosenTrigger->GetPosition(fX, fY, fZ);
289  m_creature->GetMotionMaster()->MovePointTOL(ground ? POINT_ID_GROUND : POINT_ID_AIR, fX, fY, fZ, !ground, FORCED_MOVEMENT_WALK);
290  }
291  }

327  void ExecuteAction(uint32 action) override
328  {
329  switch (action)
330  {
331  case NIGHTBANE_PHASE_RESET:
332  {
333  DoScriptText(urand(0, 1) ? SAY_LAND_PHASE_1 : SAY_LAND_PHASE_2, m_creature);
334  m_creature->GetMotionMaster()->MovePath(1, PATH_FROM_ENTRY, FORCED_MOVEMENT_WALK);
335  m_phase = PHASE_TRANSITION;
336  SetCombatScriptStatus(true);
337  DisableCombatAction(action);
338  break;
339  }
340  case NIGHTBANE_PHASE_2:
341  {
342  if (m_creature->GetHealthPercent() < 100 - 25 * m_uiFlightPhase)
343  {
344  // Start air phase movement (handled by creature_movement_template)
345  SetCombatMovement(false);
346  m_creature->HandleEmote(EMOTE_ONESHOT_LIFTOFF);
347  m_creature->SetByteValue(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_FLY_ANIM);
348  m_creature->RemoveByteFlag(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_ALWAYS_STAND);
349  m_creature->SetCanFly(true);
350  m_creature->SetHover(true);
351  DoMoveToClosestTrigger(false);
352 
353  DoScriptText(SAY_AIR_PHASE, m_creature);
354  m_phase = PHASE_TRANSITION;
355  SetDeathPrevention(true);
356  SetMeleeEnabled(false);
357  SetCombatScriptStatus(true);
358  DoResetThreat();
359  m_creature->SetSupportThreatOnly(true);
360  ++m_uiFlightPhase;
361  SetActionReadyStatus(action, true);
362  }
363  break;
364  }
365  case NIGHTBANE_BELLOWING_ROAR:
366  {
367  if (DoCastSpellIfCan(nullptr, SPELL_BELLOWING_ROAR) == CAST_OK)
368 #ifdef PRENERF_2_0_3
369  ResetCombatAction(action, urand(30000, 45000));
370 #else
371  ResetCombatAction(action, urand(38000, 48000));
372 #endif
373  break;
374  }
375  case NIGHTBANE_CHARRED_EARTH:
376  {
377  if (DoCastSpellIfCan(nullptr, SPELL_CHARRED_EARTH_TARGETING) == CAST_OK) // shouldnt be sent to client
378  ResetCombatAction(action, urand(25000, 35000));
379  break;
380  }
381  case NIGHTBANE_SMOLDERING_BREATH:
382  {
383  if (DoCastSpellIfCan(nullptr, SPELL_SMOLDERING_BREATH) == CAST_OK)
384  ResetCombatAction(action, urand(14000, 20000));
385  break;
386  }
387  case NIGHTBANE_TAIL_SWEEP:
388  {
389  if (DoCastSpellIfCan(nullptr, SPELL_TAIL_SWEEP) == CAST_OK)
390  ResetCombatAction(action, urand(14000, 20000));
391  break;
392  }
393  case NIGHTBANE_CLEAVE:
394  {
395  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
396  ResetCombatAction(action, urand(6000, 12000));
397  break;
398  }
399  case NIGHTBANE_DISTRACTING_ASH:
400  {
401  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
402  if (DoCastSpellIfCan(target, m_phase == PHASE_GROUND ? SPELL_DISTRACTING_ASH_GROUND : SPELL_DISTRACTING_ASH) == CAST_OK) // should be used in both phases
403  ResetCombatAction(action, urand(7000, 13000));
404  break;
405  }
406  case NIGHTBANE_RAIN_OF_BONES:
407  {
408  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
409  {
410  if (m_creature->CastSpell(target, SPELL_RAIN_OF_BONES, TRIGGERED_NONE) == SPELL_CAST_OK)
411  {
412  DoScriptText(EMOTE_DEEP_BREATH, m_creature);
413  DisableCombatAction(action);
414  }
415  }
416  break;
417  }
418  case NIGHTBANE_SMOKING_BLAST:
419  {
420  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
421  if (DoCastSpellIfCan(target, SPELL_SMOKING_BLAST) == CAST_OK) // should hit highest aggro target
422  ResetCombatAction(action, urand(1000, 3000));
423  break;
424  }
425  case NIGHTBANE_FIREBALL_BARRAGE:
426  {
427  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_FARTHEST_AWAY, 0, SPELL_FIREBALL_BARRAGE, SELECT_FLAG_PLAYER))
428  if (m_creature->GetDistance(target, false, DIST_CALC_COMBAT_REACH) < 60.f || (m_creature->CastSpell(m_creature->GetVictim(), SPELL_FIREBALL_BARRAGE, TRIGGERED_NONE) == SPELL_CAST_OK))
429  ResetCombatAction(action, urand(3000, 6000)); // if farthest target is 40+ yd away
430  break;
431  }
432  }
433  }

436 bool ProcessEventId_event_spell_summon_nightbane(uint32 /*eventId*/, Object* source, Object* /*target*/, bool bIsStart)
437 {
438  if (bIsStart && source->GetTypeId() == TYPEID_PLAYER)
439  {
440  ScriptedInstance* instance = static_cast<ScriptedInstance*>(static_cast<Player*>(source)->GetInstanceData());
441  if (!instance)
442  return false;
443 
444  if (instance->GetData(TYPE_NIGHTBANE) == NOT_STARTED || instance->GetData(TYPE_NIGHTBANE) == FAIL)
445  {
446  Creature* nightbane = instance->GetSingleCreatureFromStorage(NPC_NIGHTBANE);
447  if (nightbane && nightbane->IsAlive())
448  {
449  DoScriptText(EMOTE_AWAKEN, ((Player*)source));
450  instance->SetData(TYPE_NIGHTBANE, IN_PROGRESS);
451 
452  // Sort of a hack, it is unclear how this really work but the values appear to be valid (see Onyxia, too)
453  nightbane->RemoveFlag(UNIT_FIELD_FLAGS, UNIT_FLAG_UNINTERACTIBLE);
454  nightbane->SetStandState(UNIT_STAND_STATE_STAND);
455  nightbane->SetByteValue(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_FLY_ANIM);
456  nightbane->RemoveByteFlag(UNIT_FIELD_BYTES_1, UNIT_BYTES_1_OFFSET_MISC_FLAGS, UNIT_BYTE1_FLAG_ALWAYS_STAND);
457  nightbane->SetLevitate(true);
458  nightbane->SetHover(true);
459 
460  // Switch to waypoint movement
461  if (boss_nightbaneAI* nightbaneAI = dynamic_cast<boss_nightbaneAI*>(nightbane->AI()))
462  nightbaneAI->StartIntro();
463  }
464  }
465  }
466 
467  return true;
