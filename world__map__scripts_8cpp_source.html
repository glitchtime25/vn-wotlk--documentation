78 bool ProcessEventTransports(uint32 uiEventId, Object* pSource, Object* /*pTarget*/, bool /*bIsStart*/)
79 {
80  sWorldState.HandleConditionStateChange(pSource->GetEntry(), uiEventId);
81 
82  WorldObject* transport = (WorldObject*)pSource;
83  uint32 entry = 0;
84  int32 text_entry = 0;
85  uint32 relay_entry = 0;
86  switch (uiEventId)
87  {
88  case EVENT_UC_FROM_GROMGOL_ARRIVAL: // UC arrival from gromgol
89  entry = NPC_HINDENBURG;
90  text_entry = SAY_ST_FROM_GROMGOL_ARRIVAL;
91  relay_entry = RELAY_UC_FROM_GROMGOL_ARRIVAL;
92  break;
93  case EVENT_GROMGOL_FROM_UC_ARRIVAL: // gromgol arrival from UC
94  entry = NPC_SQUIBBY_OVERSPECK;
95  text_entry = SAY_TIRISFAL_FROM_UC_ARRIVAL;
96  relay_entry = RELAY_GROMGOL_FROM_UC_ARRIVAL;
97  break;
98  case EVENT_OG_FROM_UC_ARRIVAL: // OG arrival from UC
99  entry = NPC_FREZZA;
100  text_entry = SAY_TIRISFAL_FROM_UC_ARRIVAL;
101  relay_entry = RELAY_OG_FROM_UC_ARRIVAL;
102  break;
103  case EVENT_UC_FROM_OG_ARRIVAL: // UC arrival from OG
104  entry = NPC_ZAPETTA;
105  text_entry = SAY_DUROTAR_FROM_OG_ARRIVAL;
106  relay_entry = RELAY_UC_FROM_OG_ARRIVAL;
107  break;
108  case EVENT_OG_FROM_GROMGOL_ARRIVAL: // OG arrival from gromgol
109  entry = NPC_SNURK_BUCKSQUICK;
110  text_entry = SAY_ST_FROM_GROMGOL_ARRIVAL;
111  relay_entry = RELAY_OG_FROM_GROMGOL_ARRIVAL;
112  break;
113  case EVENT_GROMGOL_FROM_OG_ARRIVAL: // gromgol arrival from OG
114  entry = NPC_NEZRAZ;
115  text_entry = SAY_DUROTAR_FROM_OG_ARRIVAL;
116  relay_entry = RELAY_GROMGOL_FROM_OG_ARRIVAL;
117  break;
118  case EVENT_WK_ARRIVAL: // WestGuard Keep arrival
119  entry = NPC_HARROWMEISER;
120  text_entry = SAY_WK_ARRIVAL;
121  break;
122  case EVENT_WK_DEPARTURE: // WestGuard Keep departure
123  entry = NPC_HARROWMEISER;
124  text_entry = SAY_WK_DEPARTURE;
125  break;
126  case EVENT_VL_FROM_UC_ARRIVAL: // Vengance Landing arrival from UC
127  entry = NPC_DRENK_SPANNERSPARK;
128  text_entry = SAY_TIRISFAL_FROM_UC_ARRIVAL;
129  relay_entry = RELAY_VL_FROM_UC_ARRIVAL;
130  break;
131  case EVENT_UC_FROM_VL_ARRIVAL: // UC arrival from Vengance Landing
132  entry = NPC_MEEFI_FARTHROTTLE;
133  text_entry = SAY_UC_FROM_VL_ARRIVAL;
134  relay_entry = RELAY_UC_FROM_VL_ARRIVAL;
135  break;
136  case EVENT_OG_FROM_BT_ARRIVAL: // OG arrival from BT
137  entry = NPC_GREEB_RAMROCKET;
138  text_entry = SAY_OG_FROM_BT_ARRIVAL;
139  relay_entry = RELAY_OG_FROM_BT_ARRIVAL;
140  break;
141  case EVENT_BT_FROM_OG_ARRIVAL: // BT arrival from OG
142  entry = NPC_NARGO_SCREWBORE;
143  text_entry = SAY_DUROTAR_FROM_OG_ARRIVAL;
144  relay_entry = RELAY_BT_FROM_OG_ARRIVAL;
145  break;
146  case EVENT_OG_FROM_TB_ARRIVAL: // OG arrival from TB
147  entry = NPC_ZELLI_HOTNOZZLE;
148  text_entry = SAY_OG_FROM_TB_ARRIVAL;
149  relay_entry = RELAY_OG_FROM_TB_ARRIVAL;
150  break;
151  case EVENT_TB_FROM_OG_ARRIVAL: // TB arrival from OG
152  entry = NPC_KRENDLE_BIGPOCKETS;
153  text_entry = SAY_DUROTAR_FROM_OG_ARRIVAL;
154  relay_entry = RELAY_TB_FROM_OG_ARRIVAL;
155  break;
156  }
157  if (entry)
158  if (Creature* zeppelinMaster = static_cast<ScriptedInstance*>(transport->GetMap()->GetInstanceData())->GetSingleCreatureFromStorage(entry, true))
159  {
160  zeppelinMaster->PlayDistanceSound(SOUND_ZEPPELIN_HORN);
161  DoScriptText(text_entry, zeppelinMaster);
162  zeppelinMaster->GetMap()->ScriptsStart(SCRIPT_TYPE_RELAY, relay_entry, zeppelinMaster, zeppelinMaster);
163  }
164  return true;
165 }

182 bool ShadeOfTheHorsemanData::IsConditionFulfilled(uint32 conditionId, uint32 areaId) const
183 {
184  uint32 checkAreaId = 0;
185  switch (conditionId)
186  {
187  case INSTANCE_CONDITION_ID_FIRE_BRIGADE_PRACTICE_GOLDSHIRE:
188  checkAreaId = AREAID_GOLDSHIRE;
189  break;
190  case INSTANCE_CONDITION_ID_FIRE_BRIGADE_PRACTICE_KHARANOS:
191  checkAreaId = AREAID_KHARANOS;
192  break;
193  case INSTANCE_CONDITION_ID_FIRE_BRIGADE_PRACTICE_AZURE_WATCH:
194  checkAreaId = AREAID_AZURE_WATCH;
195  break;
196  case INSTANCE_CONDITION_ID_FIRE_TRAINING_BRILL:
197  checkAreaId = AREAID_BRILL;
198  break;
199  case INSTANCE_CONDITION_ID_FIRE_TRAINING_RAZOR_HILL:
200  checkAreaId = AREAID_RAZOR_HILL;
201  break;
202  case INSTANCE_CONDITION_ID_FIRE_TRAINING_FALCONWING:
203  checkAreaId = AREAID_FALCONWING_SQUARE;
204  break;
205  }
206 
207  if (checkAreaId)
208  return m_shadeOfTheHorsemanAttackPhases[conditionId - INSTANCE_CONDITION_ID_FIRE_BRIGADE_PRACTICE_GOLDSHIRE] != SHADE_PHASE_SPAWNED && areaId == checkAreaId;
209 
210  switch (conditionId)
211  {
212  case INSTANCE_CONDITION_ID_STOP_THE_FIRES_ALLIANCE:
213  return (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_SPAWNED && areaId == AREAID_GOLDSHIRE) ||
214  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_KHARANOS] == SHADE_PHASE_SPAWNED && areaId == AREAID_KHARANOS) ||
215  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_AZURE_WATCH] == SHADE_PHASE_SPAWNED && areaId == AREAID_AZURE_WATCH);
216  case INSTANCE_CONDITION_ID_STOP_THE_FIRES_HORDE:
217  return (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_BRILL] == SHADE_PHASE_SPAWNED && areaId == AREAID_BRILL) ||
218  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_RAZOR_HILL] == SHADE_PHASE_SPAWNED && areaId == AREAID_RAZOR_HILL) ||
219  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_FALCONWING_SQUARE] == SHADE_PHASE_SPAWNED && areaId == AREAID_FALCONWING_SQUARE);
220  case INSTANCE_CONDITION_ID_THE_HEADLESS_HORSEMAN_ALLIANCE:
221  return (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_VICTORY && areaId == AREAID_GOLDSHIRE) ||
222  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_KHARANOS] == SHADE_PHASE_VICTORY && areaId == AREAID_KHARANOS) ||
223  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_AZURE_WATCH] == SHADE_PHASE_VICTORY && areaId == AREAID_AZURE_WATCH);
224  case INSTANCE_CONDITION_ID_THE_HEADLESS_HORSEMAN_HORDE:
225  return (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_BRILL] == SHADE_PHASE_VICTORY && areaId == AREAID_BRILL) ||
226  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_RAZOR_HILL] == SHADE_PHASE_VICTORY && areaId == AREAID_RAZOR_HILL) ||
227  (m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_FALCONWING_SQUARE] == SHADE_PHASE_VICTORY && areaId == AREAID_FALCONWING_SQUARE);
228  case INSTANCE_CONDITION_ID_LET_THE_FIRES_COME_ALLIANCE:
229  return ((m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_ALL_CLEAR || m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_VICTORY) && areaId == AREAID_GOLDSHIRE) ||
230  ((m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_KHARANOS] == SHADE_PHASE_ALL_CLEAR || m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_VICTORY) && areaId == AREAID_KHARANOS) ||
231  ((m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_AZURE_WATCH] == SHADE_PHASE_ALL_CLEAR || m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_GOLDSHIRE] == SHADE_PHASE_VICTORY) && areaId == AREAID_AZURE_WATCH);
232  case INSTANCE_CONDITION_ID_LET_THE_FIRES_COME_HORDE:
233  return ((m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_BRILL] == SHADE_PHASE_ALL_CLEAR || m_shadeOfTheHorsemanAttackPhases[SHADE_VILLAGE_BRILL] == SHADE_PHASE_VICTORY) && areaId == AREAID_BRILL) ||
