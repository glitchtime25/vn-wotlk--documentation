166  }

169  {
170  m_councilDialogue.DialogueUpdate(diff);
171 
172  if (m_uiEnrageTimer)
173  {
174  if (m_uiEnrageTimer <= diff)
175  {
176  // Cast berserk on all members
177  for (unsigned int i : aCouncilMember)
178  {
179  if (Creature* pMember = m_instance->GetSingleCreatureFromStorage(i))
180  pMember->CastSpell(nullptr, SPELL_BERSERK, TRIGGERED_OLD_TRIGGERED);
181  }
182  // Start yells
183  m_councilDialogue.StartNextDialogueText(SAY_GATH_BERSERK);
184  m_uiEnrageTimer = 0;
185  }
186  else
187  m_uiEnrageTimer -= diff;
188  }

216  {

235  void DoStartEvent()
236  {
237  if (!m_instance || m_eventBegun)
238  return;
239 
240  // Prevent further handling for next council uiMember aggroing
241  m_eventBegun = true;
242  ResetTimer(GENERIC_ACTION_BALANCE, urand(2000, 3000));
243 
244  // Start the event for the Voice Trigger
245  if (Creature* voiceTrigger = m_instance->GetSingleCreatureFromStorage(NPC_COUNCIL_VOICE))
246  {
247  if (mob_blood_elf_council_voice_triggerAI* voiceAI = dynamic_cast<mob_blood_elf_council_voice_triggerAI*>(voiceTrigger->AI()))
248  voiceAI->StartVoiceEvent();
249  }
250 
251  m_creature->SetInCombatWithZone();
252  for (uint32 i : aCouncilMember)
253  {
254  Creature* member = m_instance->GetSingleCreatureFromStorage(i);
255  if (member)
256  member->SetInCombatWithZone();
257  }
258  DoCastSpellIfCan(nullptr, SPELL_EMPYREAL_BALANCE);
259  }
260 
261  void DoEndEvent()
262  {
263  if (!m_instance || m_eventEnd)
264  return;
265 
266  // Prevent further handling for next council uiMember death
267  m_eventEnd = true;
268  DisableTimer(GENERIC_ACTION_BALANCE);
269 
270  // Kill all the other council members
271  for (uint32 i : aCouncilMember)
272  {
273  Creature* member = m_instance->GetSingleCreatureFromStorage(i);
274  if (member && member->IsAlive())
275  member->CastSpell(nullptr, SPELL_QUIET_SUICIDE, TRIGGERED_OLD_TRIGGERED);
276  }
277 
278  if (Creature* voiceTrigger = m_instance->GetSingleCreatureFromStorage(NPC_COUNCIL_VOICE))
279  voiceTrigger->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, m_creature, voiceTrigger);
280  }
281 };
