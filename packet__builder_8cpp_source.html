35  void PacketBuilder::WriteCommonMonsterMovePart(const MoveSpline& move_spline, WorldPacket& data)
36  {
37  MoveSplineFlag splineflags = move_spline.splineflags;
38 
39  data << uint8(0);
40  data << move_spline.spline.getPoint(move_spline.spline.first());
41  data << move_spline.GetId();
42 
43  switch (splineflags & MoveSplineFlag::Mask_Final_Facing)
44  {
45  default:
46  data << uint8(MonsterMoveNormal);
47  break;
48  case MoveSplineFlag::Final_Target:
49  data << uint8(MonsterMoveFacingTarget);
50  data << move_spline.facing.target;
51  break;
52  case MoveSplineFlag::Final_Angle:
53  data << uint8(MonsterMoveFacingAngle);
54  data << move_spline.facing.angle;
55  break;
56  case MoveSplineFlag::Final_Point:
57  data << uint8(MonsterMoveFacingSpot);
58  data << move_spline.facing.f.x << move_spline.facing.f.y << move_spline.facing.f.z;
59  break;
60  }
61 
62  data << uint32(splineflags & static_cast<uint32>(~MoveSplineFlag::Mask_No_Monster_Move));
63 
64  if (splineflags.animation)
65  {
66  data << splineflags.getAnimationId();
67  data << move_spline.parabolic_start_time;
68  }
69 
70  data << move_spline.Duration();
71 
72  if (splineflags.parabolic)
73  {
74  data << move_spline.vertical_acceleration;
75  data << move_spline.animation_start_time;
76  }
77  }

79  void WriteLinearPath(const Spline<int32>& spline, ByteBuffer& data)
80  {
81  uint32 last_idx = spline.getPointCount() - 3;
82  const Vector3* real_path = &spline.getPoint(1);
83 
84  data << last_idx;
85  data << real_path[last_idx]; // destination
86  if (last_idx > 1)
87  {
88  Vector3 middle = (real_path[0] + real_path[last_idx]) / 2.f;
89  // first and last points already appended
90  for (uint32 i = 1; i < last_idx; ++i)
91  {
92  Vector3 offset = middle - real_path[i];
93  data.appendPackXYZ(offset.x, offset.y, offset.z);
94  }
95  }
96  }
