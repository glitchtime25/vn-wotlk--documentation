111

113 
114  // create DB connections
115 
116  // setup connection pool size
117  if (nConns < MIN_CONNECTION_POOL_SIZE)
118  m_nQueryConnPoolSize = MIN_CONNECTION_POOL_SIZE;
119  else if (nConns > MAX_CONNECTION_POOL_SIZE)
120  m_nQueryConnPoolSize = MAX_CONNECTION_POOL_SIZE;
121  else
122  m_nQueryConnPoolSize = nConns;
123 
124  // create connection pool for sync requests
125  for (int i = 0; i < m_nQueryConnPoolSize; ++i)
126  {

128  if (!pConn->Initialize(infoString))
129  {
130  delete pConn;

133

135  }
136 
137  // create and initialize connection for async requests
138  m_pAsyncConn = CreateConnection();
139  if (!m_pAsyncConn->Initialize(infoString))
140  return false;
141 
142  m_pResultQueue = new SqlResultQueue;
143 
144  InitDelayThread();

274  fclose(log_file);
275  }
276  else
277  {
278  // The file could not be opened
279  sLog.outError("SQL-Logging is disabled - Log file for the SQL commands could not be openend: %s", fName);
