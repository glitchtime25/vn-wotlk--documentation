36 void PlayerAI::AddPlayerSpellAction(uint32 spellId, std::function<Unit*()> selector)
37 {
38  if (!selector)
39  {
40  SpellEntry const* spellInfo = sSpellTemplate.LookupEntry<SpellEntry>(spellId);
41  if (spellInfo)
42  {
43  if (IsNextMeleeSwingSpell(spellInfo))
44  selector = [&]()->Unit* { return m_player->GetVictim(); };
45  else if (HasSpellTarget(spellInfo, TARGET_UNIT_ENEMY) || (spellInfo->Targets & TARGET_FLAG_DEST_LOCATION)) // always random
46  selector = [&, spellId = spellInfo->Id]()->Unit* { return m_player->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, spellId, SELECT_FLAG_PLAYER); };
47  if (HasSpellTarget(spellInfo, TARGET_UNIT_FRIEND) || HasSpellTarget(spellInfo, TARGET_UNIT_FRIEND_CHAIN_HEAL)) // heals only target self
48  selector = [&]()->Unit* { return m_player; };
49  }
50  if (!selector) // fallback
51  selector = [&]()->Unit* { return nullptr; };
52  }
53  m_playerSpellActions.emplace_back(spellId, selector);
54 }

56 void PlayerAI::ExecuteSpells()
57 {
58  if (m_executeTargetChange && CanExecuteCombatAction())
59  {
60  if (Unit* target = m_player->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
61  AttackStart(target);
62 
63  ResetTimer(GENERIC_THREAT_CHANGE, urand(10000, 20000));
64  m_executeTargetChange = false;
65  }
66 
67  if (m_spellsDisabled)
68  return;
69 
70  std::shuffle(m_playerSpellActions.begin(), m_playerSpellActions.end(), *GetRandomGenerator());
71 
72  bool success = false;
73  for (auto& data : m_playerSpellActions)
74  if (DoCastSpellIfCan(data.targetFinder(), data.spellId) == CAST_OK)
75  success = true;
76 
77  if (success)
78  {
79  uint32 randomInterval = urand(0, 5);
80  if (randomInterval != 0)
81  {
82  m_spellsDisabled = true;
83  ResetTimer(GENERIC_ACTION_RESET, urand(6000, 12000));
84  }
85  }
86 }
