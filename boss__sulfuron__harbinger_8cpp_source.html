82  void ExecuteAction(uint32 action) override
83  {
84  switch (action)
85  {
86  case SULFURON_DEMORALIZING_SHOUT:
87  {
88  if (DoCastSpellIfCan(nullptr, SPELL_DEMORALIZING_SHOUT) == CAST_OK)
89  ResetCombatAction(action, urand(15000, 20000));
90  break;
91  }
92  case SULFURON_INSPIRE:
93  {
94  Creature* target = nullptr;
95  CreatureList pList = DoFindFriendlyMissingBuff(45.0f, SPELL_INSPIRE, false);
96  if (!pList.empty())
97  {
98  CreatureList::iterator i = pList.begin();
99  advance(i, (rand() % pList.size()));
100  target = (*i);
101  }
102 
103  if (!target)
104  target = m_creature;
105 
106  if (DoCastSpellIfCan(target, SPELL_INSPIRE) == CAST_OK)
107  ResetCombatAction(action, 10000);
108  break;
109  }
110  case SULFURON_HAND_OF_RAGNAROS:
111  {
112  if (DoCastSpellIfCan(nullptr, SPELL_HAND_OF_RAGNAROS) == CAST_OK)
113  ResetCombatAction(action, urand(12000, 15000));
114  break;
115  }
116  case SULFURON_FLAME_SPEAR:
117  {
118  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_SHADOWWORD_PAIN, SELECT_FLAG_PLAYER))
119  if (DoCastSpellIfCan(target, SPELL_FLAMESPEAR) == CAST_OK)
120  ResetCombatAction(action, urand(12000, 16000));
121  break;
122  }
123  }
124  }
