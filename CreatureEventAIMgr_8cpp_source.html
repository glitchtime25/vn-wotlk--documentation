90 void CreatureEventAIMgr::CheckUnusedAISummons()
91 {
92  std::set<int32> idx_set;
93  // check not used strings this is negative range
94  for (auto itr = m_CreatureEventAI_Summon_Map.begin(); itr != m_CreatureEventAI_Summon_Map.end(); ++itr)
95  idx_set.insert(itr->first);
96 
97  auto process = [&](CreatureEventAI_Event_Map const& map)
98  {
99  for (auto itr = map.begin(); itr != map.end(); ++itr)
100  {
101  for (const auto& event : itr->second)
102  {
103  for (auto action : event.action)
104  {
105  switch (action.type)
106  {
107  case ACTION_T_SUMMON_ID:
108  {
109  if (action.summon_id.spawnId)
110  idx_set.erase(action.summon_id.spawnId);
111  break;
112  }
113  default: break;
114  }
115  }
116  }
117  }
118  };
119 
120  process(*m_CreatureEventAIEventEntryMap);
121  process(*m_CreatureEventAIEventGuidMap);
122 
123  for (int32 itr : idx_set)
124  sLog.outErrorEventAI("Entry %i in table `creature_ai_summons` but not used in EventAI scripts.", itr);
125 }

128 bool IsValidTargetType(EventAI_Type eventType, EventAI_ActionType actionType, uint32 targetType, uint32 eventId, uint8 action)
129 {
130  switch (targetType)
131  {
132  case TARGET_T_SELF:
133  if (actionType == ACTION_T_QUEST_EVENT || actionType == ACTION_T_CAST_EVENT || actionType == ACTION_T_QUEST_EVENT_ALL || actionType == ACTION_T_KILLED_MONSTER)
134  {
135  sLog.outErrorEventAI("Event %u Action%u uses incorrect Target type %u for event-type %u (must target player)", eventId, action, targetType, eventType);
136  return false;
137  }
138  return true; // Can always be used
139  case TARGET_T_HOSTILE_RANDOM:
140  case TARGET_T_HOSTILE_RANDOM_NOT_TOP:
141  if (actionType == ACTION_T_QUEST_EVENT || actionType == ACTION_T_CAST_EVENT || actionType == ACTION_T_QUEST_EVENT_ALL || actionType == ACTION_T_KILLED_MONSTER)
142  sLog.outErrorEventAI("Event %u Action%u uses LIKELY bad Target type %u for event-type %u (must target player)", eventId, action, targetType, eventType);
143  // no break, check if valid at all
144  case TARGET_T_HOSTILE:
145  case TARGET_T_HOSTILE_SECOND_AGGRO:
146  case TARGET_T_HOSTILE_LAST_AGGRO:
147  case TARGET_T_HOSTILE_RANDOM_PLAYER:
148  case TARGET_T_HOSTILE_RANDOM_NOT_TOP_PLAYER:
149  switch (eventType)
150  {
151  case EVENT_T_TIMER_OOC:
152  case EVENT_T_OOC_LOS:
153  case EVENT_T_REACHED_HOME:
154  sLog.outErrorEventAI("Event %u Action%u uses incorrect Target type %u for event-type %u (cannot be used OOC)", eventId, action, targetType, eventType);
155  return false;
156  case EVENT_T_TIMER_GENERIC:
157  sLog.outErrorEventAI("Event %u Action%u uses LIKELY incorrect Target type %u for event-type %u (cannot be used OOC)", eventId, action, targetType, eventType);
158  return true; // Does not need to be an error
159  default:
160  return true;
161  }
162  case TARGET_T_ACTION_INVOKER: // Unit who caused this Event to occur (only works for EVENT_T_AGGRO, EVENT_T_KILL, EVENT_T_DEATH, EVENT_T_SPELLHIT, EVENT_T_OOC_LOS, EVENT_T_FRIENDLY_HP, EVENT_T_FRIENDLY_IS_CC, EVENT_T_FRIENDLY_MISSING_BUFF, EVENT_T_RECEIVE_EMOTE, EVENT_T_RECEIVE_AI_EVENT)

164  switch (eventType)
165  {
166  case EVENT_T_AGGRO:
167  case EVENT_T_KILL:
168  case EVENT_T_DEATH:
169  case EVENT_T_SPELLHIT:
170  case EVENT_T_OOC_LOS:
171  case EVENT_T_FRIENDLY_HP:
172  case EVENT_T_FRIENDLY_IS_CC:
173  case EVENT_T_FRIENDLY_MISSING_BUFF:
174  case EVENT_T_RECEIVE_EMOTE:
175  case EVENT_T_RECEIVE_AI_EVENT:
176  case EVENT_T_SPELLHIT_TARGET:
177  return true;
178  default:
179  sLog.outErrorEventAI("Event %u Action%u uses incorrect Target type %u for event-type %u", eventId, action, targetType, eventType);
180  return false;
181  }
182  case TARGET_T_EVENT_SENDER: // Unit who sent an AIEvent that was received with EVENT_T_RECEIVE_AI_EVENT
183  if (eventType != EVENT_T_RECEIVE_AI_EVENT && eventType != EVENT_T_SPELLHIT && eventType != EVENT_T_SPELLHIT_TARGET)
184  {
185  sLog.outErrorEventAI("Event %u Action%u uses incorrect Target type %u for event-type %u", eventId, action, targetType, eventType);
186  return false;
187  }
188  return true;
189  case TARGET_T_SPAWNER:
190  case TARGET_T_EVENT_SPECIFIC:
191  case TARGET_T_PLAYER_INVOKER:
192  case TARGET_T_PLAYER_TAPPED:
193  case TARGET_T_NONE:
194  case TARGET_T_HOSTILE_RANDOM_MANA:
195  case TARGET_T_NEAREST_AOE_TARGET:
196  case TARGET_T_HOSTILE_FARTHEST_AWAY:
197  return true;
198  default:
199  sLog.outErrorEventAI("Event %u Action%u uses incorrect Target type", eventId, action);
200  return false;
201  }
202 }

205 void CreatureEventAIMgr::LoadCreatureEventAI_Scripts()
206 {
207  // Drop Existing EventAI List
208  std::shared_ptr<CreatureEventAI_Event_Map> newEntryContainer = std::make_shared<CreatureEventAI_Event_Map>();
209  std::shared_ptr<CreatureEventAI_Event_Map> newGuidContainer = std::make_shared<CreatureEventAI_Event_Map>();
210  std::shared_ptr<CreatureEventAI_EventComputedData_Map> newComputedContainer = std::make_shared<CreatureEventAI_EventComputedData_Map>();
211 
212  // Gather event data
213  std::unique_ptr<QueryResult> result(WorldDatabase.Query("SELECT id, creature_id, event_type, event_inverse_phase_mask, event_chance, event_flags, "
214  "event_param1, event_param2, event_param3, event_param4, event_param5, event_param6, "
215  "action1_type, action1_param1, action1_param2, action1_param3, "
216  "action2_type, action2_param1, action2_param2, action2_param3, "
217  "action3_type, action3_param1, action3_param2, action3_param3 "
218  "FROM creature_ai_scripts ORDER BY id"));
219  if (result)
220  {
221  BarGoLink bar(result->GetRowCount());
222  uint32 Count = 0;
223 
224  do
225  {
226  bar.step();
227  Field* fields = result->Fetch();
228 
229  CreatureEventAI_Event temp;
230  temp.event_id = EventAI_Type(fields[0].GetUInt32());
231  uint32 eventId = temp.event_id;
232 
233  int32 keyField = fields[1].GetInt32();
234 
235  if (keyField > 0)
236  {
237  temp.creature_id = keyField;
238  temp.creature_guid = 0;
239 
240  // Creature does not exist in database
241  if (!sCreatureStorage.LookupEntry<CreatureInfo>(temp.creature_id))
242  {
243  sLog.outErrorEventAI("Event %u has script for non-existing creature entry (%u), skipping.", eventId, temp.creature_id);
244  continue;
245  }
246  }
247  else
248  {
249  temp.creature_id = 0;
250  temp.creature_guid = -keyField;
251  if (!sObjectMgr.GetCreatureData(temp.creature_guid)) // guid does not exist in database
252  {
253  sLog.outErrorEventAI("Event %u have missing dbguid (%u), skipping.", eventId, temp.creature_guid);
254  continue;
255  }
256  }
257 
258  uint32 e_type = fields[2].GetUInt32();
259  // Report any errors in event
260  if (e_type >= EVENT_T_END)
261  {
262  sLog.outErrorEventAI("Event %u have wrong type (%u), skipping.", eventId, e_type);
263  continue;
264  }
265  temp.event_type = EventAI_Type(e_type);
266 
267  temp.event_inverse_phase_mask = fields[3].GetUInt32();
268  temp.event_chance = fields[4].GetUInt8();
269  temp.event_flags = fields[5].GetUInt32();
270  for (uint32 i = 0; i < 6; ++i)
271  temp.raw.params[i] = fields[6 + i].GetUInt32();
272 
273  // No chance of this event occuring
274  if (temp.event_chance == 0)
275  sLog.outErrorEventAI("Event %u has 0 percent chance. Event will never trigger!", eventId);
276  // Chance above 100, force it to be 100
277  else if (temp.event_chance > 100)
278  {
279  sLog.outErrorEventAI("Creature %d are using event %u with more than 100 percent chance. Adjusting to 100 percent.", keyField, eventId);
280  temp.event_chance = 100;
281  }
282 
283  // Individual event checks
284  switch (temp.event_type)
285  {
286  case EVENT_T_TIMER_IN_COMBAT:
287  case EVENT_T_TIMER_OOC:
288  case EVENT_T_TIMER_GENERIC:
289  if (temp.timer.initialMax < temp.timer.initialMin)
290  sLog.outErrorEventAI("Creature %d are using timed event(%u) with param2 < param1 (InitialMax < InitialMin). Event will never repeat.", keyField, eventId);
291  if (temp.timer.repeatMax < temp.timer.repeatMin)
292  sLog.outErrorEventAI("Creature %d are using repeatable event(%u) with param4 < param3 (RepeatMax < RepeatMin). Event will never repeat.", keyField, eventId);
293  break;
294  case EVENT_T_HP:
295  case EVENT_T_MANA:
296  case EVENT_T_TARGET_HP:
297  case EVENT_T_TARGET_MANA:
298  case EVENT_T_ENERGY:
299  if (temp.percent_range.percentMax > 100)
300  sLog.outErrorEventAI("Creature %d are using percentage event(%u) with param2 (MinPercent) > 100. Event will never trigger! ", keyField, eventId);
301 
302  if (temp.percent_range.percentMax <= temp.percent_range.percentMin)
303  sLog.outErrorEventAI("Creature %d are using percentage event(%u) with param1 <= param2 (MaxPercent <= MinPercent). Event will never trigger! ", keyField, eventId);
304 
305  if (temp.event_flags & EFLAG_REPEATABLE && !temp.percent_range.repeatMin && !temp.percent_range.repeatMax)
306  {
307  sLog.outErrorEventAI("Creature %d has param3 and param4=0 (RepeatMin/RepeatMax) but cannot be repeatable without timers. Removing EFLAG_REPEATABLE for event %u.", keyField, eventId);
308  temp.event_flags &= ~EFLAG_REPEATABLE;
309  }
310  break;
311  case EVENT_T_SPELLHIT:
312  if (temp.spell_hit.spellId)
313  {
314  SpellEntry const* pSpell = sSpellTemplate.LookupEntry<SpellEntry>(temp.spell_hit.spellId);

319  }
320 
321  if ((temp.spell_hit.schoolMask & pSpell->SchoolMask) != pSpell->SchoolMask)
322  sLog.outErrorEventAI("Creature %d has param1(spellId %u) but param2 is not -1 and not equal to spell's school mask. Event %u can never trigger.", keyField, temp.spell_hit.schoolMask, eventId);
323  }
324 
325  if (!temp.spell_hit.schoolMask)
326  sLog.outErrorEventAI("Creature %d is using invalid SpellSchoolMask(%u) defined in event %u.", keyField, temp.spell_hit.schoolMask, eventId);
327 
328  if (temp.spell_hit.repeatMax < temp.spell_hit.repeatMin)
329  sLog.outErrorEventAI("Creature %d are using repeatable event(%u) with param4 < param3 (RepeatMax < RepeatMin). Event will never repeat.", keyField, eventId);
330  break;
331  case EVENT_T_RANGE:
332  if (temp.range.maxDist < temp.range.minDist)
333  sLog.outErrorEventAI("Creature %d are using event(%u) with param2 < param1 (MaxDist < MinDist). Event will never repeat.", keyField, eventId);
334  if (temp.range.repeatMax < temp.range.repeatMin)
335  sLog.outErrorEventAI("Creature %d are using repeatable event(%u) with param4 < param3 (RepeatMax < RepeatMin). Event will never repeat.", keyField, eventId);
336  break;
337  case EVENT_T_OOC_LOS:
338  if (temp.ooc_los.conditionId && !sConditionStorage.LookupEntry<ConditionEntry>(temp.ooc_los.conditionId))
339  {
340  sLog.outErrorDb("Creature %d has `ConditionId` = %u but does not exist. Setting ConditionId to 0 for event %u.", keyField, temp.ooc_los.conditionId, eventId);
341  temp.ooc_los.conditionId = 0;
342  }

351  break;

359  break;
