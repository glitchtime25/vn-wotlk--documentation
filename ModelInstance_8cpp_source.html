36  {
37  if (!iModel)
38  {
39 #ifdef VMAP_DEBUG
40  DEBUG_LOG("<object not loaded>");

42  return false;
43  }
44  float time = pRay.intersectionTime(iBound);
45  if (time == G3D::inf())
46  {
47 #ifdef VMAP_DEBUG
48  DEBUG_LOG("Ray does not hit '%s'", name.c_str());
49 #endif
50  return false;
51  }
52  // child bounds are defined in object space:
53  Vector3 p = iInvRot * (pRay.origin() - iPos) * iInvScale;
54  Ray modRay(p, iInvRot * pRay.direction());
55  float distance = pMaxDist * iInvScale;
56  bool hit = iModel->IntersectRay(modRay, distance, pStopAtFirstHit, ignoreM2Model);
57  if (hit)
58  {

60  pMaxDist = distance;
61  }

65  void ModelInstance::intersectPoint(const G3D::Vector3& p, AreaInfo& info) const
66  {
67  if (!iModel)
68  {
69 #ifdef VMAP_DEBUG
70  DEBUG_LOG("<object not loaded>");
71 #endif
72  return;
73  }
74 
75  // M2 files don't contain area info, only WMO files
76  if (flags & MOD_M2)
77  return;
78  if (!iBound.contains(p))
79  return;
80  // child bounds are defined in object space:
81  Vector3 pModel = iInvRot * (p - iPos) * iInvScale;
82  Vector3 zDirModel = iInvRot * Vector3(0.f, 0.f, -1.f);
83  float zDist;
84  if (iModel->IntersectPoint(pModel, zDirModel, zDist, info))
85  {
86  Vector3 modelGround = pModel + zDist * zDirModel;
87  // Transform back to world space. Note that:
88  // Mat * vec == vec * Mat.transpose()
89  // and for rotation matrices: Mat.inverse() == Mat.transpose()
90  float world_Z = ((modelGround * iInvRot) * iScale + iPos).z;
91  if (info.ground_Z < world_Z)
92  {
93  info.ground_Z = world_Z;
94  info.adtId = adtId;
95  }
