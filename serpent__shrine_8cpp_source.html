58 void instance_serpentshrine_cavern::OnCreatureCreate(Creature* pCreature)
59 {

61  {
62  case NPC_KARATHRESS:
63  case NPC_LADYVASHJ:
64  case NPC_SHARKKIS:
65  case NPC_TIDALVESS:
66  case NPC_CARIBDIS:
67  case NPC_LEOTHERAS:
68  case NPC_WORLD_TRIGGER:
69  m_npcEntryGuidStore[pCreature->GetEntry()] = pCreature->GetObjectGuid();
70  break;

77  case NPC_HYDROSS_BEAM_HELPER:
78  m_lBeamHelpersGUIDList.push_back(pCreature->GetObjectGuid());
79  break;
80  case NPC_TINY_TRIGGER:
81  m_tinyTriggerGUIDList.push_back(pCreature->GetObjectGuid());
82  break;
83  case NPC_COILFANG_PRIESTESS:
84  case NPC_COILFANG_SHATTERER:
85  case NPC_VASHJIR_HONOR_GUARD:
86  case NPC_GREYHEART_TECHNICIAN:
87  // Filter only the mobs spawned on the platforms
88  if (pCreature->GetPositionZ() > 0)
89  m_sPlatformMobsGUIDSet.insert(pCreature->GetObjectGuid());
90  break;
91  case NPC_COILFANG_FRENZY_CORPSE:
92  pCreature->ForcedDespawn();

94  break;

136 void instance_serpentshrine_cavern::SetData(uint32 uiType, uint32 uiData)
137 {
138  switch (uiType)
139  {
140  case TYPE_THELURKER_EVENT:
141  switch (uiData)
142  {
143  case IN_PROGRESS:
144  if (Player* player = GetPlayerInMap())
145  player->CastSpell(player, SPELL_LURKER_SPAWN_TRIGGER, TRIGGERED_OLD_TRIGGERED); // TODO: wotlk spell should be casted by special someone
146  break;
147  case DONE:
148  if (Creature* pWorldTrigger = GetSingleCreatureFromStorage(NPC_WORLD_TRIGGER))
149  pWorldTrigger->ForcedDespawn();
150  break;
151  case FAIL:
152  if (GameObject* go = GetSingleGameObjectFromStorage(GO_STRANGE_POOL))
153  {
154  go->SetRespawnTime(2);
155  go->Refresh();
156  }
157  break;
158  }
159  case TYPE_LEOTHERAS_EVENT:
160  if (uiData == FAIL)
161  {
162  for (GuidList::const_iterator itr = m_lSpellBindersGUIDList.begin(); itr != m_lSpellBindersGUIDList.end(); ++itr)
163  {
164  if (Creature* pSpellBinder = instance->GetCreature(*itr))
165  pSpellBinder->Respawn();
166  }
167 
168  m_uiSpellBinderCount = 0;
169  }
170  // no break;
171  case TYPE_HYDROSS_EVENT:
172  case TYPE_KARATHRESS_EVENT:
173  case TYPE_MOROGRIM_EVENT:
174  m_auiEncounter[uiType] = uiData;
175  if (uiData == DONE)
176  EngageBossConsole(uiType, 0);
177  if (uiData == SPECIAL)
178  EngageBridgeConsole();
179  break;
180  case TYPE_LADYVASHJ_EVENT:
181  m_auiEncounter[uiType] = uiData;
182  if (uiData == FAIL)
183  {
184  // interrupt the shield
185  for (GuidList::const_iterator itr = m_tinyTriggerGUIDList.begin(); itr != m_tinyTriggerGUIDList.end(); ++itr)
186  {
187  if (Creature* pGenerator = instance->GetCreature(*itr))
188  pGenerator->InterruptNonMeleeSpells(false);
189  }
190 
191  // reset generators
192  DoToggleGameObjectFlags(GO_SHIELD_GENERATOR_1, GO_FLAG_NO_INTERACT, false);
193  DoToggleGameObjectFlags(GO_SHIELD_GENERATOR_2, GO_FLAG_NO_INTERACT, false);
194  DoToggleGameObjectFlags(GO_SHIELD_GENERATOR_3, GO_FLAG_NO_INTERACT, false);
195  DoToggleGameObjectFlags(GO_SHIELD_GENERATOR_4, GO_FLAG_NO_INTERACT, false);
196 
197  if (Creature* vashj = GetSingleCreatureFromStorage(NPC_LADYVASHJ))
198  {
199  vashj->SetRespawnDelay(30, true);
200  vashj->ForcedDespawn();
201  }
202  }
203  break;
204  case TYPE_LEOTHERAS_EVENT_DEMONS:
205  m_auiEncounter[uiType] = uiData;
206  break;
207  }
208 
209  if (uiData == DONE || uiData == SPECIAL)
210  {
211  OUT_SAVE_INST_DATA;
212 
213  std::ostringstream saveStream;
214  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1] << " " << m_auiEncounter[2] << " "
215  << m_auiEncounter[3] << " " << m_auiEncounter[4] << " " << m_auiEncounter[5];
216 
217  m_strInstData = saveStream.str();
218 
219  SaveToDB();
220  OUT_SAVE_INST_DATA_COMPLETE;
221  }
222 }

224 void instance_serpentshrine_cavern::Load(const char* chrIn)
225 {
226  if (!chrIn)
227  {
228  OUT_LOAD_INST_DATA_FAIL;
229  return;
230  }
231 
232  OUT_LOAD_INST_DATA(chrIn);
233 
234  std::istringstream loadStream(chrIn);
235  loadStream >> m_auiEncounter[TYPE_HYDROSS_EVENT] >> m_auiEncounter[TYPE_KARATHRESS_EVENT] >> m_auiEncounter[TYPE_LADYVASHJ_EVENT] >> m_auiEncounter[TYPE_LEOTHERAS_EVENT]
236  >> m_auiEncounter[TYPE_MOROGRIM_EVENT] >> m_auiEncounter[TYPE_THELURKER_EVENT];
237 
238  for (uint32& i : m_auiEncounter)
239  {
240  if (i == IN_PROGRESS)
241  i = NOT_STARTED;
242  }
243 
244  if (m_auiEncounter[TYPE_THELURKER_EVENT] >= DONE)
245  SpawnFishCorpses();
246 
247  OUT_LOAD_INST_DATA_COMPLETE;
248 }
