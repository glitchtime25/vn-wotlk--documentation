217  void HandlePhaseTransition()
218  {
219  uint32 timer = 0;
220  switch (m_uiTransitionCount)
221  {
222  case 0:
223  m_creature->HandleEmote(0);
224  // Shake the room
225  DoCastSpellIfCan(m_creature, SPELL_CAMERA_SHAKE);
226  if (m_instance)
227  m_instance->SetData(TYPE_MAGTHERIDON_EVENT, SPECIAL);
228 
229  timer = 8000;
230  break;
231  case 1:
232  DoCastSpellIfCan(m_creature, SPELL_DEBRIS_KNOCKDOWN);
233  SetCombatScriptStatus(false);
234  SetMeleeEnabled(true);
235  SetCombatMovement(true, true);
236  break;
237  }
238 
239  ++m_uiTransitionCount;
240  if (timer)
241  ResetTimer(MAGTHERIDON_TRANSITION_TIMER, timer);
242  }

409 bool GOUse_go_manticron_cube(Player* player, GameObject* go)
410 {
411  // if current player is exhausted or last user is still channeling
412  if (player->HasAura(SPELL_MIND_EXHAUSTION))
413  return true;
414 
415  go_manticron_cubeAI* ai = static_cast<go_manticron_cubeAI*>(go->AI());
416  Player* lastUser = ai->GetManticronCubeLastUser();
417  if (lastUser && lastUser->HasAura(SPELL_SHADOW_GRASP))
418  return true;
419 
420  if (ScriptedInstance* pInstance = (ScriptedInstance*)go->GetInstanceData())
421  {
422  if (pInstance->GetData(TYPE_MAGTHERIDON_EVENT) != IN_PROGRESS)
423  return true;
424 
425  if (Creature* pMagtheridon = pInstance->GetSingleCreatureFromStorage(NPC_MAGTHERIDON))
426  {
427  if (!pMagtheridon->IsAlive())
428  return true;
429 
430  // the real spell is cast by player - casts SPELL_SHADOW_GRASP_VISUAL
431  player->CastSpell(nullptr, SPELL_SHADOW_GRASP, TRIGGERED_NONE);
432  }
433  }
434 
435  return true;
