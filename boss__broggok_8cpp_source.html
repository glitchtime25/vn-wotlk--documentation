101  void MovementInform(uint32 motionType, uint32 pointId) override

117  void ExecuteAction(uint32 action) override
118  {
119  switch (action)
120  {
121  case BROGGOK_SLIME_SPRAY:
122  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_SLIME_SPRAY : SPELL_SLIME_SPRAY_H) == CAST_OK)
123  ResetCombatAction(action, urand(4000, 12000));
124  break;
125  case BROGGOK_POISON_BOLT:
126  if (DoCastSpellIfCan(m_creature, m_isRegularMode ? SPELL_POISON_BOLT : SPELL_POISON_BOLT_H) == CAST_OK)
127  ResetCombatAction(action, urand(4000, 12000));
128  break;
129  case BROGGOK_POISON_CLOUD:
130  if (DoCastSpellIfCan(m_creature, SPELL_POISON_CLOUD) == CAST_OK)
131  ResetCombatAction(action, 20000);
132  break;
133  }
134  }
