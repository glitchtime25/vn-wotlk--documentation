93 void world_map_ebon_hold::OnCreatureDeath(Creature* pCreature)
94 {
95  switch (pCreature->GetEntry())
96  {
97  // summon an abomination if a behemoth or abomination dies
98  case NPC_FLESH_BEHEMOTH:
99  {
100  if (GetData(TYPE_BATTLE) != BATTLE_STATE_IN_PROGRESS && GetData(TYPE_BATTLE) != BATTLE_STATE_LOST)
101  return;
102 
103  ++m_uiAttackersDead;
104  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SCOURGE, MAX_FORCES_SCOURGE - m_uiAttackersDead);
105 
106  if (GetData(TYPE_BATTLE) == BATTLE_STATE_IN_PROGRESS)
107  {
108  if (Creature* pBehemoth = pCreature->SummonCreature(NPC_FLESH_BEHEMOTH, aAbominationLocations[0].m_fX, aAbominationLocations[0].m_fY, aAbominationLocations[0].m_fZ, aAbominationLocations[0].m_fO, TEMPSPAWN_CORPSE_DESPAWN, 0))
109  {
110  pBehemoth->SetWalk(false);
111  pBehemoth->GetMotionMaster()->MovePoint(0, 2281.6511f, -5282.5884f, 82.54389f);
112  }
113  }
114 
115  if (!pCreature->IsTemporarySummon())
116  pCreature->SetCorpseDelay(10);
117  break;
118  }
119  case NPC_RAMPAGING_ABOMINATION:
120  {
121  if (GetData(TYPE_BATTLE) != BATTLE_STATE_IN_PROGRESS && GetData(TYPE_BATTLE) != BATTLE_STATE_LOST)
122  return;
123 
124  ++m_uiAttackersDead;
125  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SCOURGE, MAX_FORCES_SCOURGE - m_uiAttackersDead);
126 
127  ++m_uiAbominationsDead;
128  if (m_uiAbominationsDead % MAX_ABOMINATIONS == 0 && GetData(TYPE_BATTLE) == BATTLE_STATE_IN_PROGRESS)
129  {
130  for (uint8 i = 0; i < MAX_ABOMINATIONS; ++i)
131  {
132  if (Creature* pAbomination = pCreature->SummonCreature(aAbominationLocations[i].m_uiEntry, aAbominationLocations[i].m_fX, aAbominationLocations[i].m_fY, aAbominationLocations[i].m_fZ, aAbominationLocations[i].m_fO, TEMPSPAWN_CORPSE_DESPAWN, 0))
133  {
134  pAbomination->SetWalk(false);
135  pAbomination->GetMotionMaster()->MovePoint(0, 2281.6511f, -5282.5884f, 82.54389f);
136  }
137  }
138  m_uiAbominationsDead = 0;
139  }
140  break;
141  }
142  case NPC_VOLATILE_GHOUL:
143  case NPC_WARRIOR_OF_THE_FROZEN_WASTES:
144  {
145  if (GetData(TYPE_BATTLE) != BATTLE_STATE_IN_PROGRESS && GetData(TYPE_BATTLE) != BATTLE_STATE_LOST)
146  return;
147 
148  ++m_uiAttackersDead;
149  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SCOURGE, MAX_FORCES_SCOURGE - m_uiAttackersDead);
150 
151  if (m_uiAttackersDead % MAX_UNDEAD_WARRIORS == 0 && GetData(TYPE_BATTLE) == BATTLE_STATE_IN_PROGRESS)
152  DoSpawnUndeadWave();
153  break;
154  }
155  case NPC_DEFENDER_OF_THE_LIGHT:
156  ++m_uiDefendersDead;
157  DoUpdateBattleWorldState(WORLD_STATE_FORCES_LIGHT, MAX_FORCES_LIGHT - m_uiDefendersDead);
158  // no break;
159  case NPC_KORFAX_CHAMPION_OF_THE_LIGHT:
160  case NPC_LORD_MAXWELL_TYROSUS:
161  case NPC_COMMANDER_ELIGOR_DAWNBRINGER:
162  case NPC_LEONID_BARTHALOMEW_THE_REVERED:
163  case NPC_DUKE_NICHOLAS_ZVERENHOFF:
164  case NPC_RIMBLAT_EARTHSHATTER:
165  case NPC_RAYNE:
166  pCreature->SummonCreature(pCreature->GetEntry(), pCreature->GetPositionX(), pCreature->GetPositionY(), pCreature->GetPositionZ(), pCreature->GetOrientation(), TEMPSPAWN_CORPSE_DESPAWN, 0);
167  break;
168  }

185 void world_map_ebon_hold::SetData(uint32 uiType, uint32 uiData)
186 {
187  if (uiType == TYPE_BATTLE)
188  {
189  m_uiBattleEncounter = uiData;
190 
191  switch (uiData)
192  {
193  case BATTLE_STATE_NOT_STARTED:
194  // no action during this state
195  break;
196  case BATTLE_STATE_PREPARE:
197  // display timer
198  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_TIMER_SHOW, 1);
199  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_TIMER_TIME, MAX_BATTLE_INTRO_TIMER);
200 
201  // update world states to also show the army
202  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SHOW, 1);
203  DoUpdateBattleWorldState(WORLD_STATE_FORCES_LIGHT, MAX_FORCES_LIGHT);
204  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SCOURGE, MAX_FORCES_SCOURGE);
205 
206  m_uiBattleProgressTimer = TIMER_BATTLE_PREPARE;
207  break;
208  case BATTLE_STATE_IN_PROGRESS:
209  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_TIMER_SHOW, 0);
210  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_BEGIN, 1);
211 
212  // force resapwn creatures just in case
213  DoResetBattleScene();
214 
215  m_uiBattleProgressTimer = TIMER_BATTLE_PROGRESS;
216  m_uiAttackersDead = 0;
217  m_uiDefendersDead = 0;
218  break;
219  case BATTLE_STATE_LOST:
220  // enable the holy traps
221  DoEnableHolyTraps();
222  m_uiTrapUsageTimer = 1000;
223  break;
224  case BATTLE_STATE_DIALOGUE:
225  // despawn light defenders; new ones will spawn for the dialogue
226  for (const auto& guid : m_lLightDefendersGuids)
227  if (Creature* pTemp = instance->GetCreature(guid))
228  pTemp->ForcedDespawn();
229 
230  // kill undead army if any left
231  DoDefeatUndeadArmy();
232  break;
233  case BATTLE_STATE_WAIT_QUEST:
234  // event is complete; wait for players to complete quest
235  m_uiBattleProgressTimer = TIMER_BATTLE_QUEST_WAIT;
236  break;
237  case BATTLE_STATE_RESET:
238  // update world states to default
239  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SHOW, 1);
240  DoUpdateBattleWorldState(WORLD_STATE_FORCES_LIGHT, MAX_FORCES_LIGHT);
241  DoUpdateBattleWorldState(WORLD_STATE_FORCES_SCOURGE, MAX_FORCES_SCOURGE);
242 
243  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_TIMER_SHOW, 0);
244  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_BEGIN, 0);
245 
246  // set timer to reset the scene
247  m_uiBattleProgressTimer = TIMER_BATTLE_RESET_DELAY;
248  break;
249  }
250  }
251 }

353 // Spawn new wave of undead warriors
354 void world_map_ebon_hold::DoSpawnUndeadWave()
355 {
356  GameObject* pSource = GetSingleGameObjectFromStorage(GO_LIGHT_OF_DAWN);
357  if (!pSource)
358  {
359  script_error_log("Ebon Hold: ERROR cannot find gameobject with entry: %u", GO_LIGHT_OF_DAWN);
360  return;
361  }
362 
363  float fX, fY, fZ;
364  for (uint8 i = 0; i < MAX_UNDEAD_WARRIORS; ++i)
365  {
366  uint32 entry = roll_chance_i(20) ? NPC_WARRIOR_OF_THE_FROZEN_WASTES : NPC_VOLATILE_GHOUL;
367 
368  pSource->GetRandomPoint(pSource->GetPositionX(), pSource->GetPositionY(), pSource->GetPositionZ(), 50.0f, fX, fY, fZ);
369  pSource->SummonCreature(entry, fX, fY, fZ, 0, TEMPSPAWN_CORPSE_DESPAWN, 0);
370  }
371 
372  // first warriors respawn will call the abominations as well
373  if (m_uiAttackersDead < MAX_UNDEAD_WARRIORS * 2)
374  {
375  for (uint8 i = 0; i < MAX_ABOMINATIONS; ++i)
376  {
377  if (Creature* pAbomination = pSource->SummonCreature(aAbominationLocations[i].m_uiEntry, aAbominationLocations[i].m_fX, aAbominationLocations[i].m_fY, aAbominationLocations[i].m_fZ, aAbominationLocations[i].m_fO, TEMPSPAWN_CORPSE_DESPAWN, 0))
378  {
379  pAbomination->SetWalk(false);
380  pAbomination->GetMotionMaster()->MovePoint(0, pSource->GetPositionX(), pSource->GetPositionY(), pSource->GetPositionZ());
381  }
382  }
383  }

386 void world_map_ebon_hold::Update(uint32 uiDiff)
387 {
388  if (m_uiGothikYellTimer)
389  {
390  if (m_uiGothikYellTimer <= uiDiff)
391  m_uiGothikYellTimer = 0;
392  else
393  m_uiGothikYellTimer -= uiDiff;
394  }
395 
396  // Light of Dawn battle preparation
397  if (GetData(TYPE_BATTLE) == BATTLE_STATE_PREPARE)
398  {
399  if (m_uiBattleProgressTimer)
400  {
401  if (m_uiBattleProgressTimer <= uiDiff)
402  {
403  Creature* pDarion = GetSingleCreatureFromStorage(NPC_HIGHLORD_DARION_MOGRAINE);
404  if (!pDarion)
405  {
406  script_error_log("Ebon Hold: ERROR cannot find creature with entry: %u", NPC_HIGHLORD_DARION_MOGRAINE);
407 
408  SetData(TYPE_BATTLE, BATTLE_STATE_RESET);
409  m_uiBattleProgressTimer = 0;
410  return;
411  }
412 
413  m_uiBattleProgressTimer = 0;
414  pDarion->AI()->SendAIEvent(AI_EVENT_CUSTOM_B, pDarion, pDarion);
415 
416  SetData(TYPE_BATTLE, BATTLE_STATE_IN_PROGRESS);
417  }
418  else
419  {
420  m_uiBattleProgressTimer -= uiDiff;
421 
422  if (m_uiBattleProgressTimer / IN_MILLISECONDS % 60 == 0)
423  DoUpdateBattleWorldState(WORLD_STATE_BATTLE_TIMER_TIME, m_uiBattleProgressTimer / (MINUTE * IN_MILLISECONDS));
424  }
425  }
426  }
427  // Light of Dawn battle execution
428  else if (GetData(TYPE_BATTLE) == BATTLE_STATE_IN_PROGRESS)
429  {
430  if (m_uiBattleProgressTimer)
431  {
432  if (m_uiBattleProgressTimer <= uiDiff || m_uiDefendersDead >= 100)
433  {
434  Creature* pDarion = GetSingleCreatureFromStorage(NPC_HIGHLORD_DARION_MOGRAINE);
435  if (!pDarion)
436  {
437  script_error_log("Ebon Hold: ERROR cannot find creature with entry: %u", NPC_HIGHLORD_DARION_MOGRAINE);
438 
439  SetData(TYPE_BATTLE, BATTLE_STATE_RESET);
440  m_uiBattleProgressTimer = 0;
441  return;
442  }
443 
444  // inform Darion about the battle lost
445  m_uiBattleProgressTimer = 0;
446  pDarion->AI()->SendAIEvent(AI_EVENT_CUSTOM_C, pDarion, pDarion);
447 
448  SetData(TYPE_BATTLE, BATTLE_STATE_LOST);
449  }
450  else
451  m_uiBattleProgressTimer -= uiDiff;
452  }
453  }
454  // Kill the undead army using the light traps
455  else if (GetData(TYPE_BATTLE) == BATTLE_STATE_LOST)
456  {
457  if (m_uiTrapUsageTimer < uiDiff)
458  {
459  if (Creature* pTirion = GetSingleCreatureFromStorage(NPC_HIGHLORD_TIRION_FORDRING))
460  for (const auto& guid : m_lLightTrapsGuids)
461  if (GameObject* pTrap = instance->GetGameObject(guid))
462  pTrap->Use(pTirion);
463 
464  DoSpawnUndeadWave();
465  m_uiTrapUsageTimer = urand(3000, 4000);
466  }
467  else
468  m_uiTrapUsageTimer -= uiDiff;
469  }
470  // Wait for players to complete quest; then clear the scene
471  else if (GetData(TYPE_BATTLE) == BATTLE_STATE_WAIT_QUEST)
472  {
473  if (m_uiBattleProgressTimer)
474  {
475  if (m_uiBattleProgressTimer <= uiDiff)
476  {
477  DoClearBattleScene();
478  m_uiBattleProgressTimer = 0;
479  SetData(TYPE_BATTLE, BATTLE_STATE_RESET);
480  }
481  else
482  m_uiBattleProgressTimer -= uiDiff;
483  }
484  }
485  // Light of Dawn reset
486  else if (GetData(TYPE_BATTLE) == BATTLE_STATE_RESET)
487  {
488  if (m_uiBattleProgressTimer)
489  {
490  if (m_uiBattleProgressTimer <= uiDiff)
491  {
492  DoResetBattleScene();
493  m_uiBattleProgressTimer = 0;
494  SetData(TYPE_BATTLE, BATTLE_STATE_NOT_STARTED);
495  }
496  else
497  m_uiBattleProgressTimer -= uiDiff;
498  }
499  }
