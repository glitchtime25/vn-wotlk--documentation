265  void HandleSplitPriests()
266  {
267  m_Phase = PHASE_NORMAL;
268  // Randomize the portals
269  std::shuffle(m_vSpotLightsGuidVector.begin(), m_vSpotLightsGuidVector.end(), *GetRandomGenerator());
270  // Summon 2 priests
271  for (uint8 i = 0; i < 2; ++i)
272  if (Creature* spotlight = m_creature->GetMap()->GetCreature(m_vSpotLightsGuidVector[i]))
273  spotlight->CastSpell(nullptr, SPELL_ASTROMANCER_PRIEST, TRIGGERED_OLD_TRIGGERED, nullptr, nullptr, m_creature->GetObjectGuid());
274  // Teleport the boss at the last portal
275  if (Creature* spotlight = m_creature->GetMap()->GetCreature(m_vSpotLightsGuidVector[2]))
276  m_creature->NearTeleportTo(spotlight->GetPositionX(), spotlight->GetPositionY(), spotlight->GetPositionZ(), spotlight->GetOrientation(), true);
277 
278  SetCombatScriptStatus(false);
279  SetCombatMovement(true, true);
280 
281  // Set as visible and reset spells timers
282  m_creature->SetVisibility(VISIBILITY_ON);
283  ResetCombatAction(SOLARIAN_ARCANE_MISSILES, 0);
284  ResetCombatAction(SOLARIAN_BLINDING_LIGHT, 30000u);
285  ResetCombatAction(SOLARIAN_WRATH_OF_THE_ASTROMANCER, urand(20000, 25000));
286  ResetCombatAction(SOLARIAN_SPLIT_PHASE, 70000);
287  }

289  void ExecuteAction(uint32 action) override
290  {
291  switch (action)
292  {
293  case SOLARIAN_PHASE_2:
294  {
295  if (m_creature->GetHealthPercent() < 20.0f)
296  {
297  if (DoCastSpellIfCan(nullptr, SPELL_SOLARIAN_TRANSFORM) == CAST_OK)
298  {
299  DoScriptText(SAY_VOIDA, m_creature);
300  ResetTimer(SOLARIAN_PHASE_2_DELAY, 2000);
301 
302  m_creature->SetArmor(WV_ARMOR);
303  m_Phase = PHASE_VOID;
304 
305  if (m_creature->GetVisibility() != VISIBILITY_ON)
306  m_creature->SetVisibility(VISIBILITY_ON);
307 
308  // Stop the combat for a small delay
309  SetCombatMovement(false);
310  SetCombatScriptStatus(true);
311  SetMeleeEnabled(false);
312  SetActionReadyStatus(action, false);
313  DisableCombatAction(SOLARIAN_SPLIT_PHASE);
314  DisableCombatAction(SOLARIAN_ARCANE_MISSILES);
315  }
316  }
317  break;
318  }
319  case SOLARIAN_SPLIT_PHASE:
320  {
321  m_Phase = PHASE_SPLIT;
322 
323  // After these 50 seconds she portals to the middle of the room and disappears, leaving 3 light portals behind.
324  // ToDo: check if there are some spells involved in this event!
325  SetCombatMovement(false);
326  SetMeleeEnabled(false);
327  SetCombatScriptStatus(true);
328  m_creature->NearTeleportTo(fRoomCenter[0], fRoomCenter[1], fRoomCenter[2], fRoomCenter[3], true);
329  DisableCombatAction(action);
330  ResetTimer(SOLARIAN_SPLIT_PHASE_DELAY, 1000);
331  break;
332  }
333  case SOLARIAN_WRATH_OF_THE_ASTROMANCER:
334  {
335 #ifdef PRENERF_2_0_3
336  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_WOTA_MAIN_CAST, SELECT_FLAG_PLAYER))
337  if (DoCastSpellIfCan(target, SPELL_WOTA_MAIN_CAST) == CAST_OK)
338  ResetCombatAction(action, urand(25000, 30000));
339 #else
340  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_WRATH_OF_THE_ASTROMANCER, SELECT_FLAG_PLAYER))
341  if (DoCastSpellIfCan(target, SPELL_WRATH_OF_THE_ASTROMANCER) == CAST_OK)
342  ResetCombatAction(action, urand(12000, 18000));
343 #endif
344  break;
345  }
346  case SOLARIAN_ARCANE_MISSILES:
347  {
348  // Solarian casts Arcane Missiles on on random targets in the raid.
349  std::vector<Unit*> units;
350  m_creature->SelectAttackingTargets(units, ATTACKING_TARGET_ALL_SUITABLE, 0, SPELL_ARCANE_MISSILES, SELECT_FLAG_PLAYER);
351  if (units.size() > 0)
352  {
353  std::shuffle(units.begin(), units.end(), *GetRandomGenerator());
354  for (Unit* unit : units)
355  {
356  if (!unit->HasAura(SPELL_WOTA_DOT))
357  {
358  if (DoCastSpellIfCan(unit, SPELL_ARCANE_MISSILES) == CAST_OK)
359  {
360  ResetCombatAction(action, urand(3000, 3500));
361  break;
362  }
363  }
364  }
365  }
366  break;
367  }
368  case SOLARIAN_BLINDING_LIGHT:
369  {
370 #ifdef PRENERF_2_0_3
371  // She casts this spell every 45 seconds. It is a kind of Moonfire spell, which she strikes down on the whole raid simultaneously. It hits everyone in the raid for 2280 to 2520 arcane damage.
372  if (DoCastSpellIfCan(nullptr, SPELL_BLINDING_LIGHT) == CAST_OK && DoCastSpellIfCan(nullptr, SPELL_MARK_OF_SOLARIAN) == CAST_OK)
373 #else
374  if (DoCastSpellIfCan(nullptr, SPELL_BLINDING_LIGHT) == CAST_OK)
375 #endif
376  ResetCombatAction(action, 30000);
377  break;
378  }
379  case SOLARIAN_FEAR:
380  {
381  if (DoCastSpellIfCan(nullptr, SPELL_PSYHIC_SCREAM) == CAST_OK)
382  ResetCombatAction(action, 20000);
383  break;
384  }
385  case SOLARIAN_VOID_BOLT:
386  {
387  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_VOID_BOLT) == CAST_OK)
388  ResetCombatAction(action, 10000);
389  break;
390  }
391  }
392  }
