423 bool Map::EnsureGridLoaded(const Cell& cell)
424 {
425  EnsureGridCreated(GridPair(cell.GridX(), cell.GridY()));
426  NGridType* grid = getNGrid(cell.GridX(), cell.GridY());
427 
428  MANGOS_ASSERT(grid != nullptr);
429  if (!isGridObjectDataLoaded(cell.GridX(), cell.GridY()))
430  {
431  // it's important to set it loaded before loading!
432  // otherwise there is a possibility of infinity chain (grid loading will be called many times for the same grid)
433  // possible scenario:
434  // active object A(loaded with loader.LoadN call and added to the map)
435  // summons some active object B, while B added to map grid loading called again and so on..
436  setGridObjectDataLoaded(true, cell.GridX(), cell.GridY());
437  ObjectGridLoader loader(*grid, this, cell);
438  loader.LoadN();
439 
440  // Add resurrectable corpses to world object list in grid
441  sObjectAccessor.AddCorpsesToGrid(GridPair(cell.GridX(), cell.GridY()), (*grid)(cell.CellX(), cell.CellY()), this);
442  return true;
443  }
444 
445  return false;
446 }
