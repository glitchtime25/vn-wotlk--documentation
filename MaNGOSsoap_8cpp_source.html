75 int ns1__executeCommand(soap* soap, char* command, char** result)
76 {
77  // security check
78  if (!soap->userid || !soap->passwd)
79  {
80  DEBUG_LOG("MaNGOSsoap: Client didn't provide login information");
81  return 401;
82  }
83 
84  auto const accountId = sAccountMgr.GetId(soap->userid);
85  if (!accountId)
86  {
87  DEBUG_LOG("MaNGOSsoap: Client used invalid username '%s'", soap->userid);
88  return 401;
89  }
90 
91  if (!sAccountMgr.CheckPassword(accountId, soap->passwd))
92  {
93  DEBUG_LOG("MaNGOSsoap: invalid password for account '%s'", soap->userid);
94  return 401;
95  }
96 
97  if (sAccountMgr.GetSecurity(accountId) < SOAPThread::MinLevel)
98  {
99  DEBUG_LOG("MaNGOSsoap: %s's gmlevel is too low", soap->userid);
100  return 403;
101  }
102 
103  if (!command || !*command)
104  return soap_sender_fault(soap, "Command mustn't be empty", "The supplied command was an empty string");
105 
106  DEBUG_LOG("MaNGOSsoap: got command '%s'", command);
107 
108  bool commandExecuted = false, commandSucceeded = false;
109  std::vector<char> buffer;
110  buffer.reserve(SOAPThread::CommandOutputBufferSize);
111 
112  // commands are executed in the world thread. We have to wait for them to be completed
113  sWorld.QueueCliCommand(new CliCommandHolder(accountId, SEC_CONSOLE, command,
114  [&buffer](const char* output)
115  {
116  assert(output);
117 
118  for (auto p = output; *p; ++p)
119  buffer.push_back(*p);
120  },
121  [soap, &commandExecuted, &commandSucceeded](bool success)
122  {
123  commandExecuted = true;
124  commandSucceeded = success;
125  }));
126 
127  while (!commandExecuted)
128  std::this_thread::sleep_for(std::chrono::milliseconds(50));
129 
130  buffer.push_back(0);
131  auto const printBuffer = soap_strdup(soap, &buffer[0]);
132 
133  if (!commandSucceeded)
134  {
135  auto ret = soap_sender_fault(soap, printBuffer, printBuffer);
136 
137  return ret;
138  }
139 
140  *result = printBuffer;
141  return SOAP_OK;
142 }
