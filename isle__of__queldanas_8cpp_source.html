271  AddCombatAction(DAWNBLADE_BLOOD_KNIGHT_SEAL_OF_WRATH, 5000, 10000);
272  AddCombatAction(DAWNBLADE_BLOOD_KNIGHT_JUDGEMENT_OF_WRATH, 10000, 20000);
273  AddCustomAction(DAWNBLADE_BLOOD_KNIGHT_START_EVENT, true, [&]() { StartDuel(); });

279  void ExecuteAction(uint32 action) override
280  {
281  switch (action)
282  {
283  case DAWNBLADE_BLOOD_KNIGHT_STOP_EVENT:
284  {
285  if (m_creature->IsInCombat())
286  {
287  if (m_sparringPartner)
288  {
289  if (Unit* attacking = m_creature->GetVictim())
290  if (attacking->GetObjectGuid() != m_sparringPartner || attacking->GetHealthPercent() <= 15.f)
291  StopDuel();
292  }
293  }
294  break;
295  }
296  case DAWNBLADE_BLOOD_KNIGHT_SEAL_OF_WRATH:
297  {
298  if (DoCastSpellIfCan(nullptr, SPELL_SEAL_OF_WRATH) == CAST_OK)
299  ResetCombatAction(action, urand(15000, 25000));
300  break;
301  }
302  case DAWNBLADE_BLOOD_KNIGHT_JUDGEMENT_OF_WRATH:
303  {
304  if (m_creature->HasAura(SPELL_SEAL_OF_WRATH))
305  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_JUDGEMENT_OF_WRATH) == CAST_OK)
306  {
307  m_creature->RemoveAurasDueToSpell(SPELL_SEAL_OF_WRATH);
308  ResetCombatAction(action, urand(25000, 30000));
309  }
310  break;
311  }
312  case DAWNBLADE_BLOOD_KNIGHT_HOLY_LIGHT:
313  {
314  if (Unit* target = DoSelectLowestHpFriendly(40.0f, 50.0, true, true))
315  if (target->GetObjectGuid() == m_creature->GetObjectGuid() || (target->GetFaction() != FACTION_SPARRING_DAWNBLADE && target->GetFaction() != FACTION_SPAR_BUDDY))
316  if (DoCastSpellIfCan(target, SPELL_HOLY_LIGHT) == CAST_OK)
317  ResetCombatAction(action, urand(16000, 24000));
318  break;
319  }
320  }
321  }
