146  static TypeID GetTypeId(HighGuid high)
147  {
148  switch (high)

151  // case HIGHGUID_CONTAINER: return TYPEID_CONTAINER; HIGHGUID_CONTAINER==HIGHGUID_ITEM currently
152  case HIGHGUID_UNIT: return TYPEID_UNIT;
153  case HIGHGUID_PET: return TYPEID_UNIT;
154  case HIGHGUID_PLAYER: return TYPEID_PLAYER;
155  case HIGHGUID_GAMEOBJECT: return TYPEID_GAMEOBJECT;
156  case HIGHGUID_DYNAMICOBJECT: return TYPEID_DYNAMICOBJECT;
157  case HIGHGUID_CORPSE: return TYPEID_CORPSE;
158  case HIGHGUID_MO_TRANSPORT: return TYPEID_GAMEOBJECT;
159  case HIGHGUID_VEHICLE: return TYPEID_UNIT;
160  // unknown
161  case HIGHGUID_INSTANCE:
162  case HIGHGUID_GROUP:
163  default: return TYPEID_OBJECT;
