88 void instance_uldaman::SetData(uint32 uiType, uint32 uiData)
89 {
90  switch (uiType)
91  {
92  case TYPE_ALTAR_EVENT:
93  if (uiData == DONE)
94  {
95  DoUseDoorOrButton(GO_TEMPLE_DOOR_UPPER);
96  DoUseDoorOrButton(GO_TEMPLE_DOOR_LOWER);
97  }
98  else if (uiData == IN_PROGRESS)
99  {
100  // Also do a reset before starting the event - this will respawn dead Keepers
101  DoResetKeeperEvent();
102  m_uiKeeperCooldown = 5000;
103  }
104  else if (uiData == NOT_STARTED)
105  {
106  DoResetKeeperEvent();
107  m_uiStoneKeepersFallen = 0;
108  }
109  m_auiEncounter[0] = uiData;
110  break;
111 
112  case TYPE_ARCHAEDAS:
113  if (uiData == DONE)
114  {
115  DoUseDoorOrButton(GO_ANCIENT_VAULT);
116  DoRespawnGameObject(GO_ANCIENT_TREASURE, HOUR);
117  }
118  m_auiEncounter[1] = uiData;
119  break;
120  }
121 
122  if (uiData == DONE)
123  {
124  OUT_SAVE_INST_DATA;
125 
126  std::ostringstream saveStream;
127 
128  saveStream << m_auiEncounter[0] << " " << m_auiEncounter[1];
129 
130  m_strInstData = saveStream.str();
131  SaveToDB();
132  OUT_SAVE_INST_DATA_COMPLETE;
133  }
134 }

207 void instance_uldaman::DoResetKeeperEvent()
208 {
209  if (m_lKeepers.empty())
210  {
211  script_error_log("Instance Uldaman: ERROR creature %u couldn't be found or something really bad happened.", NPC_STONE_KEEPER);
212  return;
213  }
214 
215  // Force reset all keepers to the original state
216  for (GuidList::const_iterator itr = m_lKeepers.begin(); itr != m_lKeepers.end(); ++itr)
217  {
218  if (Creature* pKeeper = instance->GetCreature(*itr))
219  {
220  if (!pKeeper->IsAlive())
221  pKeeper->Respawn();
222  }
223  }

265 void instance_uldaman::Update(uint32 uiDiff)
266 {
267  if (GetData(TYPE_ALTAR_EVENT) != IN_PROGRESS)
268  return;
269 
270  if (!m_uiKeeperCooldown)
271  return;
272 
273  if (m_uiKeeperCooldown <= uiDiff)
274  {
275  for (GuidList::const_iterator itr = m_lKeepers.begin(); itr != m_lKeepers.end(); ++itr)
276  {
277  // Get Keeper which is alive and out of combat
278  Creature* pKeeper = instance->GetCreature(*itr);
279  if (!pKeeper || !pKeeper->IsAlive() || pKeeper->GetVictim())
280  continue;
281 
282  // Get starter player for attack
283  Player* pPlayer = pKeeper->GetMap()->GetPlayer(m_playerGuid);
284  if (!pPlayer || !pPlayer->IsAlive())
285  {
286  // If he's not available, then get a random player, within a reasonamble distance in map
287  pPlayer = GetPlayerInMap(true, false);
288  if (!pPlayer || !pPlayer->IsWithinDistInMap(pKeeper, 50.0f))
289  {
290  SetData(TYPE_ALTAR_EVENT, NOT_STARTED);
291  return;
292  }
293  }
294 
295  // Attack the player
296  pKeeper->RemoveAurasDueToSpell(SPELL_STONED);
297  pKeeper->AI()->AttackStart(pPlayer);
298  break;
299  }
300 
301  m_uiKeeperCooldown = 0;
302  }
303  else
304  m_uiKeeperCooldown -= uiDiff;
305 }
