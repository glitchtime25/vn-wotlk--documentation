65  void Reset() override
66  {
67  // Check the map id because the same creature entry is involved in other scripted event in other instance
68  if (m_creature->GetMapId() != MAP_ID_MAGISTER)
69  return;
70 
71  m_creature->SetHover(true);
72 
73  // Move the dragon to landing point
74  m_creature->GetMotionMaster()->MovePath(1, PATH_FROM_ENTRY, FORCED_MOVEMENT_RUN, true);
75  }

77  void HandleTransform()
78  {
79  uint32 timer = 0;
80  switch (m_transformStage)
81  {
82  case 0:
83  {
84  m_creature->HandleEmote(EMOTE_ONESHOT_LAND);
85  m_creature->SetLevitate(false);
86  m_creature->SetHover(false);
87  timer = 1500;
88  break;
89  }
90  case 1:
91  {
92  DoCastSpellIfCan(nullptr, SPELL_CAMERA_SHAKE_MED);
93  m_creature->SetFacingTo(afKaelLandPoint[3]);
94  m_creature->SetFloatValue(OBJECT_FIELD_SCALE_X, 0.6f); // no spell for this
95  timer = 1000;
96  break;
97  }
98  case 2:
99  {
100  if (ScriptedInstance* instance = static_cast<ScriptedInstance*>(m_creature->GetInstanceData()))
101  if (Creature* creature = instance->GetSingleCreatureFromStorage(NPC_SCRYERS_BUNNY))
102  creature->CastSpell(nullptr, SPELL_ORB_KILL_CREDIT, TRIGGERED_OLD_TRIGGERED);
103  DoCastSpellIfCan(nullptr, SPELL_TRANSFORM_VISUAL);
104  m_creature->ForcedDespawn(1500);
105  timer = 500;
106  break;
107  }
108  case 3:
109  {
110  m_creature->SpawnCreature(5850236, m_creature->GetMap());
111  break;
112  }
113  }
114  ++m_transformStage;
115  if (timer)
116  ResetTimer(ACTION_TRANSFORM, timer);
117  }
118 
119  void JustRespawned() override
120  {
121  ScriptedAI::JustRespawned();
122  DoScriptText(SAY_SPAWN, m_creature);
123  }
