140  }
141  case 486:
142  {
143  str << "i486";
144 
145  const uint8 firstByte = revision >> 8;
146  const uint8 y = 0xF & (revision >> 4);

148

153 
154  break;
155  }
156  case 586:
157  {
158  str << "Pentium (i586)";
159 
160  const uint8 model = 0xFF & (revision >> 8);

163  str << " Model: " << static_cast<uint32>(model) << " Stepping: " << static_cast<uint32>(stepping);

165  break;

181 // this function assumes that the given code begins with a valid instruction. in other words, that
182 // it does not begin in random data or in the middle of an instruction.
183 void DeobfuscateAsm(std::vector<std::uint8_t> &code)
184 {
185 #define LSTRIP(c, l) do { if (c.size() <= l) { c.clear(); return; } else { c.erase(c.begin(), c.begin()+l); } } while(false)
186 
187  do
188  {
189  if (code.empty())
190  return;
191 
192  // At the start of this loop we re-assume the entry condition that the code begins at a valid instruction.
193 
194  // remove NOP
195  if (code[0] == 0x90)
196  {
197  LSTRIP(code, 1);
198  continue;
199  }
200 
201  // xchg statements which are either semantically equivalent to NOP or
202  // which will be inverted later on.
203  if (code[0] == 0x87)
204  {
205  // {"eax", "ebx", "ecx", "edx", "edi", "esi"};
206  if (code[1] == 0xC9 || // xchg ecx, ecx
207  code[1] == 0xCA || // xchg ecx, edx
208  code[1] == 0xCB || // xchg ebx, ecx
209  code[1] == 0xCE || // xchg esi, ecx
210  code[1] == 0xCF || // xchg ecx, edi
211  code[1] == 0xE4 || // xchg esp, esp
212  code[1] == 0xED || // xchg ebp, ebp
213  code[1] == 0xD1 || // xchg ecx, edx
214  code[1] == 0xD2 || // xchg edx, edx
215  code[1] == 0xD3 || // xchg ebx, edx
216  code[1] == 0xD6 || // xchg esi, edx
217  code[1] == 0xD7 || // xchg edx, edi
218  code[1] == 0xD9 || // xchg ecx, ebx
219  code[1] == 0xDA || // xchg ebx, edx
220  code[1] == 0xDB || // xchg ebx, ebx
221  code[1] == 0xDE || // xchg esi, ebx
222  code[1] == 0xDF || // xchg ebx, edi
223  code[1] == 0xF1 || // xchg ecx, esi
224  code[1] == 0xF2 || // xchg edx, esi
225  code[1] == 0xF3 || // xchg ebx, esi
226  code[1] == 0xF7 || // xchg edi, esi
227  code[1] == 0xF9 || // xchg ecx, edi
228  code[1] == 0xFA || // xchg edx, edi
229  code[1] == 0xFB || // xchg edi, esi
230  code[1] == 0xFE) // xchg ebx, edi
231  {
232  LSTRIP(code, 2);
233  continue;
234  }
235  }
236  // xchg one register with another. in wrobot this is always repeated/inverted later
237  else if (code[0] >= 0x91 && code[0] <= 0x97)
238  {
239  LSTRIP(code, 1);
240  continue;
241  }
242  // simple NOP-equivalent mov statements
243  else if (code[0] == 0x89)
244  {
245  if (code[1] == 0xC9 || // mov ecx, ecx
246  code[1] == 0xD2 || // mov edx, edx
247  code[1] == 0xDB || // mov ebx, ebx
248  code[1] == 0xE4 || // mov esp, esp
249  code[1] == 0xFF) // mov edi, edi
250  {
251  LSTRIP(code, 2);
252  continue;
253  }
254  }
255  // near JMP past random junk
256  else if (code[0] == 0xEB) // JMP +1-0xFF
257  {
258  // not enough room? clear out anything left and finish
259  if (code.size() < 2)
260  {
261  code.clear();
262  return;
263  }
264 
265  const unsigned int len = static_cast<unsigned int>(code[1]) + 2;
266 
267  LSTRIP(code, len);
268  continue;
269  }
270  // far JMP past random junk
271  else if (code[0] == 0xE9)
272  {
273  if (code.size() < 5)
274  {
275  code.clear();
276  return;
277  }
278 
279  const unsigned int len = *reinterpret_cast<unsigned int *>(&code[1]) + 5;
280 
281  LSTRIP(code, len);
282  continue;
283  }
284  // short conditional jumps that always are followed by far jumps to the same place:
285  // "JE", "JZ", "JNE", "JNZ", "JG", "JNLE", "JGE", "JNL", "JNGE", "JLE", "JNG", "JO", "JNO", "JP", "JPE", "JNP", "JPO", "JS"
286  else if (code[0] == 0x70 ||
287  code[0] == 0x71 ||
288  code[0] == 0x74 ||
289  code[0] == 0x75 ||
290  code[0] == 0x78 ||
291  code[0] == 0x7A ||
292  code[0] == 0x7B ||
293  code[0] == 0x7C ||
294  code[0] == 0x7D ||
295  code[0] == 0x7E ||
296  code[0] == 0x7F)
297  {
298  const unsigned int target = static_cast<unsigned int>(code[1]) + 2;
299 
300  unsigned int secondJumpTarget = 0;
301 
302  // currently the far jump will come within 12 bytes
303  for (auto i = 2u; i < 14; ++i)
304  {
305  if (code[i] == 0xE9)
306  {
307  secondJumpTarget = *reinterpret_cast<unsigned int *>(&code[i + 1]) + 5;
308  break;
309  }
310  else if (code[i] == 0xEB)
311  {
312  secondJumpTarget = static_cast<unsigned int>(code[i + 1]) + 2 + i;
313  break;
314  }
315  }
316 
317  if (secondJumpTarget && secondJumpTarget == target)
318  {
319  LSTRIP(code, target);
320  continue;
321  }
322  }
323  // these are instructions that are not part of the obfuscation. so for now we will skip them and
324  // deobfuscate whatever comes after them. this will let us build up deobfuscated code.
325  // pushfd. this is currently not part of the obfuscation of wrobot but rather it appears to always be
326  // the first non-obfuscated opcode. therefore, lets skip it and debofuscate what comes after it.
327  else if (code[0] == 0x9C || // pushfd
328  code[0] == 0x60) // pushad
329  {
330  std::vector<std::uint8_t> copy(code.begin() + 1, code.end());
331  DeobfuscateAsm(copy);
332 
333  code.resize(1);
334  code.insert(code.end(), copy.begin(), copy.end());
335 
336  // do nothing after this
337  return;
338  }
339 
340  // if we reach here, we haven't made any change. stop
341  break;
342  } while (true);
343 #undef LSTRIP
344 }

346 // returns true when the given hook code is suspicious
347 bool ValidateEndSceneHook(const std::vector<uint8> &code)
348 {
349  auto copy = code;
350 
351  // attempt asm deobfuscation to detect wrobot
352  DeobfuscateAsm(copy);
353 
354  std::stringstream str;
355 
356  for (auto i = 0u; i < code.size(); ++i)
357  {
358  str << "0x" << std::hex << std::uppercase << std::setw(2) << std::setfill('0') << static_cast<int>(code[i]);
359 
360  if (i < code.size() - 1)
361  str << " ";
362  }
363 
364  sLog.outBasic("WARDEN: Deobfuscate debug. Original code size: %lu deobfuscated size: %lu Code:\n%s",
365  uint64(code.size()), uint64(copy.size()), str.str().c_str());
366 
367  // wrobot's deobfuscated endscene hook begins with pushfd, pushad. if thats what this starts with,
368  // assume it is wrobot, regardless of what comes after it.
369  if (copy.size() >= 2 && copy[0] == 0x9C && copy[1] == 0x60)
370  return true;
371 
372  return code.size() == 200 && copy.size() < 15;
373 }

599 void WardenWin::ValidateEndScene(const std::vector<uint8> &code)
600 {
601  auto p = &code[0];
602 
603  // skip any NOPs
604  while (*p == 0x90) ++p;
605 
606  auto const nopCount = static_cast<int>(p - &code[0]);
607 
608  static constexpr size_t codeRequestLength = 200u;
609 
610  // int3 breakpoint
611  if (*p == 0xCC)
612  {
613  sLog.outBasic("WARDEN: Detected INT3 EndScene hook for account %u IP %s (NOP count = %d)",
614  _session->GetAccountId(), _session->GetRemoteAddress().c_str(), nopCount);
615  }
616  // JMP hook
617  else if (*p == 0xE9)
618  {
619  auto const dest = *reinterpret_cast<const uint32 *>(p + 1);
620 
621  auto const absoluteDest = _endSceneAddress + nopCount + dest + 5;
622  sLog.outBasic("WARDEN: Detected JMP EndScene hook for account %u IP %s (NOP count = %d)",
623  _session->GetAccountId(), _session->GetRemoteAddress().c_str(), nopCount);
624 
625  // request a custom scan just to check the JMP destination
626  EnqueueScans({ std::make_shared<WindowsMemoryScan>(absoluteDest, codeRequestLength,
627  // checker
628  [](const Warden *warden, ByteBuffer &buff)
629  {
630  auto const wardenWin = const_cast<WardenWin *>(reinterpret_cast<const WardenWin *>(warden));
631 
632  auto const result = buff.read<uint8>();
633 
634  if (!!result)
635  {
636  sLog.outBasic("WARDEN: Failed to read EndScene hook code from account %u ip %s",
637  wardenWin->_session->GetAccountId(), wardenWin->_session->GetRemoteAddress().c_str());
638 
639  return true;
640  }
641 
642  std::vector<uint8> code(codeRequestLength);
643 
644  buff.read(&code[0], code.size());
645 
646  if (ValidateEndSceneHook(code) && !!sAnticheatConfig.GetWardenSuspiciousEndSceneHookAction())
647  wardenWin->_anticheat->RecordCheat(sAnticheatConfig.GetWardenSuspiciousEndSceneHookAction(),
648  "Warden", "Suspicious client heuristic results. Probable bot.");
649 
650  return false;
651  }, "EndScene hook validate scan", None) });
652  }
