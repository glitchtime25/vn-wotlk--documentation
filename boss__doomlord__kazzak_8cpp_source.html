107  void UpdateAI(const uint32 uiDiff) override
108  {
109  // Return since we have no target
110  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
111  return;
112 
113  // Cleave_Timer
114  if (m_uiCleaveTimer < uiDiff)
115  {
116  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
117  m_uiCleaveTimer = urand(8000, 12000);
118  }
119  else
120  m_uiCleaveTimer -= uiDiff;
121 
122  // ThunderClap_Timer
123  if (m_uiThunderClapTimer < uiDiff)
124  {
125  if (DoCastSpellIfCan(nullptr, SPELL_THUNDERCLAP) == CAST_OK)
126  m_uiThunderClapTimer = urand(10000, 14000);
127  }
128  else
129  m_uiThunderClapTimer -= uiDiff;
130 
131  // VoidBolt_Timer
132  if (m_uiVoidBoltTimer < uiDiff)
133  {
134  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_VOID_BOLT) == CAST_OK)
135  m_uiVoidBoltTimer = urand(15000, 18000);
136  }
137  else
138  m_uiVoidBoltTimer -= uiDiff;
139 
140  // MarkOfKazzak_Timer
141  if (m_uiMarkOfKazzakTimer < uiDiff)
142  {
143  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_MARK_OF_KAZZAK, SELECT_FLAG_PLAYER | SELECT_FLAG_POWER_MANA))
144  {
145  if (DoCastSpellIfCan(pTarget, SPELL_MARK_OF_KAZZAK) == CAST_OK)
146  m_uiMarkOfKazzakTimer = 20000;
147  }
148  }
149  else
150  m_uiMarkOfKazzakTimer -= uiDiff;
151 
152  // Enrage_Timer
153  if (m_uiEnrageTimer < uiDiff)
154  {
155  if (DoCastSpellIfCan(nullptr, SPELL_FRENZY) == CAST_OK)
156  {
157  DoScriptText(EMOTE_GENERIC_FRENZY, m_creature);
158  m_uiEnrageTimer = 60000;
159  }
160  }
161  else
162  m_uiEnrageTimer -= uiDiff;
163 
164  // Great_Enrage_Timer
165  if (m_uiGreatEnrageTimer)
166  {
167  if (m_uiGreatEnrageTimer <= uiDiff)
168  {
169  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
170  m_uiGreatEnrageTimer = 0;
171  }
172  else
173  m_uiGreatEnrageTimer -= uiDiff;
174  }
175 
176  // Twisted Reflection
177  if (m_uiTwistedReflectionTimer < uiDiff)
178  {
179  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_TWISTED_REFLECTION, SELECT_FLAG_PLAYER))
180  {
181  if (DoCastSpellIfCan(pTarget, SPELL_TWISTED_REFLECTION) == CAST_OK)
182  m_uiTwistedReflectionTimer = 15000;
183  }
184  }
185  else
186  m_uiTwistedReflectionTimer -= uiDiff;
187 
188  DoMeleeAttackIfReady();
189  }
