29 std::shared_ptr<std::map<int32, UnitConditionEntry>> UnitConditionMgr::Load()
30 {
31  auto unitConditions = std::make_shared<std::map<int32, UnitConditionEntry>>();
32 
33  std::unique_ptr<QueryResult> result(WorldDatabase.Query("SELECT * FROM unit_condition"));
34  if (!result)
35  {
36  sLog.outString(">> Loaded 0 unit_condition definitions - table is empty.");
37  sLog.outString();
38  m_unitConditions = unitConditions;
39  return unitConditions;
40  }
41 
42  BarGoLink bar(result->GetRowCount());
43  uint32 Count = 0;
44 
45  do
46  {
47  bar.step();
48  Field* fields = result->Fetch();
49 
50  UnitConditionEntry conditionEntry;
51  conditionEntry.Id = fields[0].GetUInt32();
52  for (uint32 i = 0; i < 8; ++i)
53  {
54  conditionEntry.Variable[i] = fields[i + 2].GetUInt32();
55  if (conditionEntry.Variable[i] >= uint32(UnitCondition::MAX))
56  {
57  sLog.outErrorDb("Table unit_condition entry %d has invalid variable condition type %u. Setting to None.", conditionEntry.Id, conditionEntry.Variable[i]);
58  conditionEntry.Variable[i] = 0;
59  }
60  conditionEntry.Operation[i] = fields[i + 10].GetUInt32();
61  if (conditionEntry.Operation[i] >= uint32(ConditionOperation::MAX))
62  {
63  sLog.outErrorDb("Table unit_condition entry %d has invalid operation %u. Setting to None.", conditionEntry.Id, conditionEntry.Operation[i]);
64  conditionEntry.Operation[i] = 0;
65  }
66  conditionEntry.Value[i] = fields[i + 18].GetInt32();
67  }
68 
69  unitConditions->emplace(conditionEntry.Id, conditionEntry);
70  }
71  while (result->NextRow());
72 
73  sLog.outString(">> Loaded %u unit_condition definitions", Count);
74  sLog.outString();
75 
76  m_unitConditions = unitConditions;
77 
78  return unitConditions;
79 }

81 bool UnitConditionMgr::Meets(Unit const* source, Unit const* target, int32 Id)
82 {
83  auto unitConditions = source->GetMap()->GetMapDataContainer().GetUnitConditions();
84  auto itr = unitConditions->find(Id);
85  if (unitConditions->end() == itr)
86  return false;
87 
88  UnitConditionEntry& entry = itr->second;
89  for (uint32 i = 0; i < 8; i++)
90  {
91  bool result = meets(source, target, UnitCondition(entry.Variable[i]), entry.Operation[i], entry.Value[i]);
92  if (result && (entry.Flags & UNIT_CONDITION_FLAG_OR)) // return on first true or
93  return true;
94  else if (!result && (entry.Flags & UNIT_CONDITION_FLAG_OR) == 0) // return on first false and
95  return false;
96  }
97 
98  if ((entry.Flags & UNIT_CONDITION_FLAG_OR)) // no true reached - fail
99  return false;
100  else // and - success
101  return true;
102 }

115 int32 UnitConditionMgr::getConditionValue(Unit const* source, Unit const* target, UnitCondition conditionType, int32 value)
116 {
117  switch (conditionType)
118  {
119  case UnitCondition::NONE: return true;
120  case UnitCondition::RACE: return source->getRace();
121  case UnitCondition::CLASS: return source->getClass();
122  case UnitCondition::LEVEL: return source->GetLevel();
123  case UnitCondition::IS_SELF: return source == target;
124  case UnitCondition::IS_MY_PET: return source->GetObjectGuid() == target->GetMasterGuid();
125  case UnitCondition::IS_MASTER: return source->GetMasterGuid() == target->GetObjectGuid();
126  case UnitCondition::IS_TARGET: return target->GetTarget() == source;
127  case UnitCondition::CAN_ASSIST: return source->CanAssist(target);
128  case UnitCondition::CAN_ATTACK: return source->CanAttack(target);
129  case UnitCondition::HAS_PET: return source->HasAnyPet();
130  case UnitCondition::HAS_WEAPON: return source->hasWeapon(BASE_ATTACK) || source->hasWeapon(OFF_ATTACK);
131  case UnitCondition::HEALTH_PERCENT: return source->GetHealthPercent();
132  case UnitCondition::MANA_PERCENT: return source->GetPowerPercent(POWER_MANA);
133  case UnitCondition::RAGE_PERCENT: return source->GetPowerPercent(POWER_RAGE);
134  case UnitCondition::ENERGY_PERCENT: return source->GetPowerPercent(POWER_ENERGY);
135  case UnitCondition::COMBO_POINTS: return source->GetComboPoints();
136  case UnitCondition::HAS_HELPFUL_AURA_SPELL:
137  return source->HasAuraHolder([](SpellAuraHolder* holder)
138  {
139  return holder->IsPositive();
140  }) ? value : 0;
141  case UnitCondition::HAS_HELPFUL_AURA_DISPEL_TYPE:
142  return source->HasAuraHolder([value](SpellAuraHolder* holder)
143  {
144  return holder->IsPositive() && holder->GetSpellProto()->Dispel == value;
145  }) ? value : 0;
146  case UnitCondition::HAS_HELPFUL_AURA_MECHANIC:
147  return source->HasAuraHolder([value](SpellAuraHolder* holder)
148  {
149  return holder->IsPositive() && holder->HasMechanic(value);
150  }) ? value : 0;
151  case UnitCondition::HAS_HARMFUL_AURA_SPELL:
152  return source->HasAuraHolder([](SpellAuraHolder* holder)
153  {
154  return !holder->IsPositive();
155  }) ? value : 0;
156  case UnitCondition::HAS_HARMFUL_AURA_DISPEL_TYPE: break;
157  return source->HasAuraHolder([value](SpellAuraHolder* holder)
158  {
159  return !holder->IsPositive() && holder->GetSpellProto()->Dispel == value;
160  }) ? value : 0;
161  case UnitCondition::HAS_HARMFUL_AURA_MECHANIC:
162  return source->HasAuraHolder([value](SpellAuraHolder* holder)
163  {
164  return !holder->IsPositive() && holder->HasMechanic(value);
165  }) ? value : 0;
166  case UnitCondition::HAS_HARMFUL_AURA_SCHOOL:
167  return source->HasAuraHolder([value](SpellAuraHolder* holder)
168  {
169  return !holder->IsPositive() && holder->GetSpellProto()->SchoolMask & (1 << value);
170  }) ? value : 0;
171  case UnitCondition::DAMAGE_PHYSICAL: break;
172  case UnitCondition::DAMAGE_HOLY: break;
173  case UnitCondition::DAMAGE_FIRE: break;
174  case UnitCondition::DAMAGE_NATURE: break;
175  case UnitCondition::DAMAGE_FROST: break;
176  case UnitCondition::DAMAGE_SHADOW: break;
177  case UnitCondition::DAMAGE_ARCANE: break;
178  case UnitCondition::IN_COMBAT: return source->IsInCombat();
179  case UnitCondition::IS_MOVING: return source->IsMoving();
180  case UnitCondition::IS_CASTING: return source->GetCurrentSpell(CURRENT_GENERIC_SPELL) != nullptr;
181  case UnitCondition::IS_CASTING_SPELL: return source->GetCurrentSpell(CURRENT_GENERIC_SPELL) != nullptr/* && !source->GetCurrentSpell(CURRENT_GENERIC_SPELL)->m_spellInfo->HasAttribute(SPELL_ATTR_IS_ABILITY)*/;
182  case UnitCondition::IS_CHANNELING: return source->GetCurrentSpell(CURRENT_GENERIC_SPELL) != nullptr;
183  case UnitCondition::IS_CHANNELING_SPELL: return source->GetCurrentSpell(CURRENT_GENERIC_SPELL) != nullptr/* && !source->GetCurrentSpell(CURRENT_GENERIC_SPELL)->m_spellInfo->HasAttribute(SPELL_ATTR_IS_ABILITY)*/;
184  case UnitCondition::NUMBER_OF_MELEE_ATTACKERS:
185  return std::count_if(source->getAttackers().begin(), source->getAttackers().end(), [source](Unit* attacker)
186  {
187  return source->CanReachWithMeleeAttack(attacker);
188  });
189  case UnitCondition::IS_ATTACKING_ME: return source->GetTarget() == target;
190  case UnitCondition::RANGE: return sqrt(source->GetDistance(target, true, DIST_CALC_NONE));
191  case UnitCondition::IN_MELEE_RANGE: return source->CanReachWithMeleeAttack(target);
192  case UnitCondition::PURSUIT_TIME: break; // NYI - possible to implement from CombatManager timer
193  case UnitCondition::HAS_HARMFUL_AURA_CANCELLED_BY_DAMAGE: return source->HasNegativeAuraWithInterruptFlag(AURA_INTERRUPT_FLAG_DAMAGE);
194  case UnitCondition::HAS_HARMFUL_AURA_WITH_PERIODIC_DAMAGE: return source->HasAuraType(SPELL_AURA_PERIODIC_DAMAGE);
195  case UnitCondition::NUMBER_OF_ENEMIES: return source->getThreatManager().getThreatList().size();
196  case UnitCondition::NUMBER_OF_FRIENDS: break;
197  case UnitCondition::THREAT_PHYSICAL: break;
198  case UnitCondition::THREAT_HOLY: break;
199  case UnitCondition::THREAT_FIRE: break;
200  case UnitCondition::THREAT_NATURE: break;
201  case UnitCondition::THREAT_FROST: break;
202  case UnitCondition::THREAT_SHADOW: break;
203  case UnitCondition::THREAT_ARCANE: break;
204  case UnitCondition::IS_INTERRUPTIBLE: return source->IsInterruptible();
205  case UnitCondition::NUMBER_OF_ATTACKERS: return source->getAttackers().size();
206  case UnitCondition::NUMBER_OF_RANGED_ATTACKERS:
207  return std::count_if(source->getAttackers().begin(), source->getAttackers().end(), [source](Unit* attacker)
208  {
209  return !source->CanReachWithMeleeAttack(attacker);
210  });
211  case UnitCondition::CREATURE_TYPE: return source->GetCreatureType();
212  case UnitCondition::IS_MELEE_ATTACKING: return source->GetTarget() && source->CanReachWithMeleeAttack(source->GetTarget());
213  case UnitCondition::IS_RANGED_ATTACKING: return source->GetTarget() && !source->CanReachWithMeleeAttack(source->GetTarget());
214  case UnitCondition::HEALTH: return source->GetHealth();
215  case UnitCondition::SPELL_KNOWN: return source->HasSpell(value) ? value : 0;
216  case UnitCondition::HAS_HARMFUL_AURA_EFFECT: return value >= 0 && value < int32(TOTAL_AURAS) && source->HasAura([](Aura* aura)
217  {
218  return !aura->IsPositive();
219  }, AuraType(value)) ? value : 0;
220  case UnitCondition::IS_IMMUNE_TO_AREA_OF_EFFECT: return source->IsAOEImmune();
221  case UnitCondition::IS_PLAYER: return source->IsPlayer();
222  case UnitCondition::DAMAGE_MAGIC: break;
223  case UnitCondition::DAMAGE_TOTAL: break;
224  case UnitCondition::THREAT_MAGIC: break;
225  case UnitCondition::THREAT_TOTAL: break;
226  case UnitCondition::HAS_CRITTER: return source->GetMiniPet() != nullptr;
227  case UnitCondition::HAS_TOTEM_IN_SLOT_1: return source->GetTotem(TOTEM_SLOT_FIRE) != nullptr;
228  case UnitCondition::HAS_TOTEM_IN_SLOT_2: return source->GetTotem(TOTEM_SLOT_EARTH) != nullptr;
229  case UnitCondition::HAS_TOTEM_IN_SLOT_3: return source->GetTotem(TOTEM_SLOT_WATER) != nullptr;
230  case UnitCondition::HAS_TOTEM_IN_SLOT_4: return source->GetTotem(TOTEM_SLOT_AIR) != nullptr;
231  case UnitCondition::HAS_TOTEM_IN_SLOT_5: break;
232  case UnitCondition::CREATURE_ID: return source->GetEntry();
233  case UnitCondition::STRING_ID: return source->HasStringId(value) ? value : 0;
234  case UnitCondition::HAS_AURA: return source->HasAura(value) ? value : 0;
235  case UnitCondition::IS_ENEMY: return source->IsEnemy(target);
236  case UnitCondition::IS_SPEC_MELEE: break;
237  case UnitCondition::IS_SPEC_TANK: break;
238  case UnitCondition::IS_SPEC_RANGED: break;
239  case UnitCondition::IS_SPEC_HEALER: break;
240  case UnitCondition::IS_PLAYER_CONTROLLED_NPC: return source->IsCreature() && source->IsPlayerControlled();
241  case UnitCondition::IS_DYING: return source->GetHealth() == 0;
242  case UnitCondition::PATH_FAIL_COUNT: return source->GetMotionMaster()->GetCurrent()->IsReachable(); // TODO: Workaround
243  case UnitCondition::UNUSED: break;
244  case UnitCondition::LABEL: break;
245  }
246  return false;
247 }
