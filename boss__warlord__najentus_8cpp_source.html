159  void ExecuteAction(uint32 action) override
160  {
161  switch (action)
162  {
163  case NAJENTUS_ACTION_TIDAL_SHIELD:
164  if (DoCastSpellIfCan(nullptr, SPELL_TIDAL_SHIELD) == CAST_OK)
165  {
166  DisableCombatAction(NAJENTUS_ACTION_IMPALING_SPINE);
167  DisableCombatAction(NAJENTUS_ACTION_NEEDLE_SPINE);
168  ResetCombatAction(action, GetSubsequentActionTimer(NajentusActions(action)));
169  }
170  return;
171  case NAJENTUS_ACTION_ENRAGE:
172  if (DoCastSpellIfCan(nullptr, SPELL_BERSERK) == CAST_OK)
173  {
174  DoScriptText(SAY_ENRAGE, m_creature);
175  SetActionReadyStatus(action, false);
176  }
177  return;
178  case NAJENTUS_ACTION_IMPALING_SPINE:
179  {
180  if (Unit* target = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, SPELL_IMPALING_SPINE, SELECT_FLAG_PLAYER | SELECT_FLAG_SKIP_TANK))
181  {
182  if (DoCastSpellIfCan(target, SPELL_IMPALING_SPINE) == CAST_OK)
183  {
184  DoScriptText(urand(0, 1) ? SAY_NEEDLE1 : SAY_NEEDLE2, m_creature);
185  ResetCombatAction(action, GetSubsequentActionTimer(NajentusActions(action)));
186  }
187  }
188  return;
189  }
190  case NAJENTUS_ACTION_NEEDLE_SPINE:
191  {
192  if (DoCastSpellIfCan(nullptr, SPELL_NEEDLE_SPINE_TARGETING) == CAST_OK)
193  ResetCombatAction(action, GetSubsequentActionTimer(NajentusActions(action)));
194  return;
195  }
196  }
197  }
