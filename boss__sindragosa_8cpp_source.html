310  void UpdateAI(const uint32 uiDiff) override
311  {
312  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
313  return;
314 
315  // Berserk
316  if (m_uiBerserkTimer)
317  {
318  if (m_uiBerserkTimer <= uiDiff)
319  {
320  if (DoCastSpellIfCan(m_creature, SPELL_BERSERK) == CAST_OK)
321  {
322  DoScriptText(SAY_BERSERK, m_creature);
323  m_uiBerserkTimer = 0;
324  }
325  }
326  else
327  m_uiBerserkTimer -= uiDiff;
328  }
329 
330  switch (m_uiPhase)
331  {
332  case SINDRAGOSA_PHASE_THREE:
333  {
334  // Ice Tomb
335  if (m_uiIceTombSingleTimer <= uiDiff)
336  {
337  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 1, SPELL_ICE_TOMB_SINGLE, SELECT_FLAG_PLAYER))
338  {
339  if (DoCastSpellIfCan(pTarget, SPELL_ICE_TOMB) == CAST_OK)
340  m_uiIceTombSingleTimer = 15000;
341  }
342  }
343  else
344  m_uiIceTombSingleTimer -= uiDiff;
345 
346  // no break
347  }
348  case SINDRAGOSA_PHASE_GROUND:
349  {
350  // Phase 1 only
351  if (m_uiPhase == SINDRAGOSA_PHASE_GROUND)
352  {
353  // Health Check
354  if (m_creature->GetHealthPercent() <= 30.0f)
355  {
356  if (DoCastSpellIfCan(m_creature, SPELL_MYSTIC_BUFFET) == CAST_OK)
357  {
358  m_uiPhase = SINDRAGOSA_PHASE_THREE;
359  DoScriptText(SAY_PHASE_3, m_creature);
360  }
361  }
362 
363  // Phase 2 (air)
364  if (m_uiPhaseTimer <= uiDiff)
365  {
366  m_uiPhaseTimer = 33000;
367  DoScriptText(SAY_TAKEOFF, m_creature);
368  SetCombatMovement(false);
369  m_creature->GetMotionMaster()->MovePoint(SINDRAGOSA_POINT_GROUND_CENTER, SindragosaPosition[0][0], SindragosaPosition[0][1], SindragosaPosition[0][2]);
370  }
371  else
372  m_uiPhaseTimer -= uiDiff;
373  }
374 
375  // Cleave
376  if (m_uiCleaveTimer <= uiDiff)
377  {
378  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_CLEAVE) == CAST_OK)
379  m_uiCleaveTimer = urand(5000, 15000);
380  }
381  else
382  m_uiCleaveTimer -= uiDiff;
383 
384  // Tail Smash
385  if (m_uiTailSmashTimer <= uiDiff)
386  {
387  if (DoCastSpellIfCan(m_creature, SPELL_TAIL_SMASH) == CAST_OK)
388  m_uiTailSmashTimer = urand(10000, 20000);
389  }
390  else
391  m_uiTailSmashTimer -= uiDiff;
392 
393  // Frost Breath
394  if (m_uiFrostBreathTimer <= uiDiff)
395  {
396  if (DoCastSpellIfCan(m_creature->GetVictim(), SPELL_FROST_BREATH) == CAST_OK)
397  m_uiFrostBreathTimer = urand(15000, 20000);
398  }
399  else
400  m_uiFrostBreathTimer -= uiDiff;
401 
402  // Unchained Magic
403  if (m_uiUnchainedMagicTimer <= uiDiff)
404  {
405  if (DoCastSpellIfCan(m_creature, SPELL_UNCHAINED_MAGIC) == CAST_OK)
406  {
407  m_uiUnchainedMagicTimer = urand(40000, 60000);
408  DoScriptText(SAY_UNCHAINED_MAGIC, m_creature);
409  }
410  }
411  else
412  m_uiUnchainedMagicTimer -= uiDiff;
413 
414  // Icy Grip and Blistering Cold
415  if (m_uiIcyGripTimer <= uiDiff)
416  {
417  if (DoCastSpellIfCan(m_creature, SPELL_ICY_GRIP) == CAST_OK)
418  {
419  m_uiIcyGripTimer = 70000;
420  DoScriptText(SAY_BLISTERING_COLD, m_creature);
421  }
422  }
423  else
424  m_uiIcyGripTimer -= uiDiff;
425 
426  DoMeleeAttackIfReady();
427  break;
428  }
429  case SINDRAGOSA_PHASE_FLYING_TO_GROUND:
430  case SINDRAGOSA_PHASE_FLYING_TO_AIR:
431  break;
432  case SINDRAGOSA_PHASE_AIR:
433  {
434  // Phase One (ground)
435  if (m_uiPhaseTimer <= uiDiff)
436  {
437  m_uiPhase = SINDRAGOSA_PHASE_FLYING_TO_GROUND;
438  m_uiPhaseTimer = 42000;
439  m_creature->GetMotionMaster()->MovePoint(SINDRAGOSA_POINT_AIR_CENTER, SindragosaPosition[1][0], SindragosaPosition[1][1], SindragosaPosition[1][2]);
440  }
441  else
442  m_uiPhaseTimer -= uiDiff;
443 
444  // Frost Bomb
445  if (m_uiFrostBombTimer <= uiDiff)
446  {
447  DoFrostBomb();
448  m_uiFrostBombTimer = 6000;
449  }
450  else
451  m_uiFrostBombTimer -= uiDiff;
452 
453  break;
454  }
455  }
456  }
