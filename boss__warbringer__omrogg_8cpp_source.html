292  void UpdateAI(const uint32 uiDiff) override
293  {
294  if (m_uiDelayTimer < uiDiff)
295  {
296  m_uiDelayTimer = 3500;
297 
298  Creature* pLeftHead = m_creature->GetMap()->GetCreature(m_leftHeadGuid);
299  Creature* pRightHead = m_creature->GetMap()->GetCreature(m_rightHeadGuid);
300 
301  if (!pLeftHead || !pRightHead)
302  return;
303 
304  if (m_bAggroYell)
305  {
306  DoScriptText(GoCombatDelay[m_iAggro].id, pRightHead);
307  m_bAggroYell = false;
308  }
309 
310  if (m_bThreatYell2)
311  {
312  Creature* pSource = (pLeftHead->GetEntry() == ThreatDelay2[m_iThreat].creature ? pLeftHead : pRightHead);
313 
314  DoScriptText(ThreatDelay2[m_iThreat].id, pSource);
315  m_bThreatYell2 = false;
316  }
317 
318  if (m_bThreatYell)
319  {
320  Creature* pSource = (pLeftHead->GetEntry() == ThreatDelay1[m_iThreat].creature ? pLeftHead : pRightHead);
321 
322  DoScriptText(ThreatDelay1[m_iThreat].id, pSource);
323  m_bThreatYell = false;
324  m_bThreatYell2 = true;
325  }
326 
327  if (m_bKillingYell)
328  {
329  Creature* pSource = (pLeftHead->GetEntry() == KillingDelay[m_iKilling].creature ? pLeftHead : pRightHead);
330 
331  DoScriptText(KillingDelay[m_iKilling].id, pSource);
332  m_bKillingYell = false;
333  }
334  }
335  else
336  m_uiDelayTimer -= uiDiff;
337 
338  if (!m_creature->SelectHostileTarget() || !m_creature->GetVictim())
339  return;
340 
341  if (m_uiBlastCount && m_uiBlastWaveTimer)
342  {
343  if (m_uiBlastWaveTimer <= uiDiff)
344  {
345  if (DoCastSpellIfCan(m_creature, SPELL_BLAST_WAVE) == CAST_OK)
346  {
347  m_uiBlastWaveTimer = 5000;
348  ++m_uiBlastCount;
349 
350  if (m_uiBlastCount == 3)
351  m_uiBlastCount = 0;
352  }
353  }
354  else
355  m_uiBlastWaveTimer -= uiDiff;
356  }
357 
358  if (m_uiBurningMaulTimer < uiDiff)
359  {
360  if (DoCastSpellIfCan(m_creature, m_bIsRegularMode ? SPELL_BURNING_MAUL : SPELL_BURNING_MAUL_H) == CAST_OK)
361  {
362  DoScriptText(EMOTE_ENRAGE, m_creature);
363  m_uiBurningMaulTimer = 40000;
364  m_uiBlastWaveTimer = 16000;
365  m_uiBlastCount = 1;
366  }
367  }
368  else
369  m_uiBurningMaulTimer -= uiDiff;
370 
371  if (m_uiResetThreatTimer < uiDiff)
372  {
373  if (Unit* pTarget = m_creature->SelectAttackingTarget(ATTACKING_TARGET_RANDOM, 0, nullptr, SELECT_FLAG_PLAYER))
374  {
375  DoYellForThreat();
376  DoResetThreat();
377  AttackStart(pTarget);
378  }
379  m_uiResetThreatTimer = urand(25000, 40000);
380  }
381  else
382  m_uiResetThreatTimer -= uiDiff;
383 
384  if (m_uiFearTimer < uiDiff)
385  {
386  if (DoCastSpellIfCan(m_creature, SPELL_FEAR) == CAST_OK)
387  m_uiFearTimer = urand(15000, 35000);
388  }

390  m_uiFearTimer -= uiDiff;
391
