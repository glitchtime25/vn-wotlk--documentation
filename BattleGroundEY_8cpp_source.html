41 void BattleGroundEY::Update(uint32 diff)
42 {
43  BattleGround::Update(diff);
44 
45  if (GetStatus() != STATUS_IN_PROGRESS)
46  return;
47 
48  // resource counter
49  if (m_resourceUpdateTimer < diff)
50  {
51  UpdateResources();
52  m_resourceUpdateTimer = EY_RESOURCES_UPDATE_TIME;
53  }
54  else
55  m_resourceUpdateTimer -= diff;
56 
57  // flag respawn
58  if (m_flagState == EY_FLAG_STATE_WAIT_RESPAWN || m_flagState == EY_FLAG_STATE_ON_GROUND)
59  {
60  if (m_flagRespawnTimer < diff)
61  {
62  m_flagRespawnTimer = 0;
63 
64  if (m_flagState == EY_FLAG_STATE_WAIT_RESPAWN)
65  RespawnFlagAtCenter(true);
66  else
67  RespawnDroppedFlag();
68  }
69  else
70  m_flagRespawnTimer -= diff;
71  }
72 
73  // workaround for Fel Reaver Ruins flag capture needed on 3.3.5 only
74  // the original areatrigger (4514) is covered by a bigger one (4515) and is not triggered on client side
75  if (IsFlagPickedUp())
76  {
77  if (m_felReaverFlagTimer < diff)
78  {
79  Player* flagCarrier = sObjectMgr.GetPlayer(GetFlagCarrierGuid());
80  if (flagCarrier)
81  {
82  if (m_towerOwner[NODE_FEL_REAVER_RUINS] == flagCarrier->GetTeam())
83  {
84  // coords and range taken from DBC of areatrigger (4514)
85  if (flagCarrier->IsWithinDist3d(2044.0f, 1729.729f, 1190.03f, 3.0f))
86  ProcessPlayerFlagScoreEvent(flagCarrier, NODE_FEL_REAVER_RUINS);
87  }
88  }
89  m_felReaverFlagTimer = EY_FEL_REAVER_FLAG_UPDATE_TIME;
90  }
91  else
92  m_felReaverFlagTimer -= diff;
93  }
94 }

274 // process the capture events
275 bool BattleGroundEY::HandleEvent(uint32 eventId, Object* source, Object* target)
276 {
277  // event called when player picks up a dropped flag
278  if (eventId == EVENT_NETHERSTORM_FLAG_SPELL && target && target->IsPlayer() && source->IsGameObject())
279  {
280  DEBUG_LOG("BattleGroundEY: Handle flag pickup event id %u", eventId);
281 
282  HandlePlayerClickedOnFlag(static_cast<Player*>(target), static_cast<GameObject*>(source));
283  return true;
284  }
285 
286  GameObject* go = dynamic_cast<GameObject*>(source);
287  if (!go) // people can misuse various effects in bgs, need a whitelist not a blacklist - this fits the bill nicely
288  return true;
289 
290  // events called from the capture points
291  for (uint8 i = 0; i < EY_MAX_NODES; ++i)
292  {
293  if (eyTowers[i] == go->GetEntry())
294  {
295  for (uint8 j = 0; j < 4; ++j)
296  {
297  if (eyTowerEvents[i][j].eventEntry == eventId)
298  {
299  ProcessCaptureEvent(go, i, eyTowerEvents[i][j].team, eyTowerEvents[i][j].worldState, eyTowerEvents[i][j].message);
300 
301  // no need to iterate other events or towers
302  return false;
303  }
304  }
305  // no need to iterate other towers
306  return false;
307  }
308  }
309 
310  return false;
311 }

313 // Method that handles the capture point capture events
314 void BattleGroundEY::ProcessCaptureEvent(GameObject* go, uint32 towerId, Team team, uint32 /*newWorldState*/, uint32 message)
315 {
316  DEBUG_LOG("BattleGroundEY: Process capture point event from gameobject entry %u, captured by team %u", go->GetEntry(), team);
317 
318  if (team == ALLIANCE)
319  {
320  // update counter
321  GetBgMap()->GetVariableManager().SetVariable(WORLD_STATE_EY_TOWER_COUNT_ALLIANCE, GetBgMap()->GetVariableManager().GetVariable(WORLD_STATE_EY_TOWER_COUNT_ALLIANCE) + 1);
322 
323  SendMessageToAll(message, CHAT_MSG_BG_SYSTEM_ALLIANCE);
324 
325  // link graveyard
326  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(eyeGraveyardData[towerId].id, EY_ZONE_ID_MAIN, ALLIANCE);
327 
328  // spawn spirit healer and defender
329  if (Creature* healer = go->SummonCreature(BG_NPC_SPIRIT_GUIDE_ALLIANCE, eyeGraveyardData[towerId].x, eyeGraveyardData[towerId].y, eyeGraveyardData[towerId].z, eyeGraveyardData[towerId].o, TEMPSPAWN_DEAD_DESPAWN, 0))
330  m_spiritHealers[towerId] = healer->GetObjectGuid();
331  if (Creature* defender = go->SummonCreature(BG_NPC_HON_DEFENDER_TRIGGER_A, eyeTowerdData[towerId].x, eyeTowerdData[towerId].y, eyeTowerdData[towerId].z, eyeTowerdData[towerId].o, TEMPSPAWN_DEAD_DESPAWN, 0))
332  m_honorableDefender[towerId] = defender->GetObjectGuid();
333  }
334  else if (team == HORDE)
335  {
336  // update counter
337  GetBgMap()->GetVariableManager().SetVariable(WORLD_STATE_EY_TOWER_COUNT_HORDE, GetBgMap()->GetVariableManager().GetVariable(WORLD_STATE_EY_TOWER_COUNT_HORDE) + 1);
338 
339  SendMessageToAll(message, CHAT_MSG_BG_SYSTEM_HORDE);
340 
341  // link graveyard
342  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(eyeGraveyardData[towerId].id, EY_ZONE_ID_MAIN, HORDE);
343 
344  // spawn spirit healer and defender
345  if (Creature* healer = go->SummonCreature(BG_NPC_SPIRIT_GUIDE_HORDE, eyeGraveyardData[towerId].x, eyeGraveyardData[towerId].y, eyeGraveyardData[towerId].z, eyeGraveyardData[towerId].o, TEMPSPAWN_DEAD_DESPAWN, 0))
346  m_spiritHealers[towerId] = healer->GetObjectGuid();
347  if (Creature* defender = go->SummonCreature(BG_NPC_HON_DEFENDER_TRIGGER_H, eyeTowerdData[towerId].x, eyeTowerdData[towerId].y, eyeTowerdData[towerId].z, eyeTowerdData[towerId].o, TEMPSPAWN_DEAD_DESPAWN, 0))
348  m_honorableDefender[towerId] = defender->GetObjectGuid();
349  }
350  else
351  {
352  if (m_towerOwner[towerId] == ALLIANCE)
353  {
354  // update counter
355  GetBgMap()->GetVariableManager().SetVariable(WORLD_STATE_EY_TOWER_COUNT_ALLIANCE, GetBgMap()->GetVariableManager().GetVariable(WORLD_STATE_EY_TOWER_COUNT_ALLIANCE) - 1);
356 
357  SendMessageToAll(message, CHAT_MSG_BG_SYSTEM_ALLIANCE);
358  }
359  else
360  {
361  // update counter
362  GetBgMap()->GetVariableManager().SetVariable(WORLD_STATE_EY_TOWER_COUNT_HORDE, GetBgMap()->GetVariableManager().GetVariable(WORLD_STATE_EY_TOWER_COUNT_HORDE) - 1);
363 
364  SendMessageToAll(message, CHAT_MSG_BG_SYSTEM_HORDE);
365  }
366 
367  // unlink graveyard
368  GetBgMap()->GetGraveyardManager().SetGraveYardLinkTeam(eyeGraveyardData[towerId].id, EY_ZONE_ID_MAIN, TEAM_INVALID);
369 
370  // despawn spirit healer and defender
371  if (Creature* healer = GetBgMap()->GetCreature(m_spiritHealers[towerId]))
372  healer->ForcedDespawn();
373  if (Creature* defender = GetBgMap()->GetCreature(m_honorableDefender[towerId]))
374  defender->ForcedDespawn();
375  }
376 
377  // update capture point owner
378  Team oldTeam = m_towerOwner[towerId];
379  SetTowerOwner(EYNodes(towerId), team);
380 
381  if (oldTeam == ALLIANCE || oldTeam == HORDE) // only on going to grey
382  {
383  // teleport players off of GY
384  Creature* spiritHealer = GetClosestCreatureWithEntry(go, oldTeam == ALLIANCE ? BG_NPC_SPIRIT_GUIDE_ALLIANCE : BG_NPC_SPIRIT_GUIDE_HORDE, 100.f);
385 
386  if (spiritHealer)
387  spiritHealer->AI()->SendAIEvent(AI_EVENT_CUSTOM_A, spiritHealer, spiritHealer);
388  }
389 
390  // replace the visual flags
391  switch (team)
392  {
393  case ALLIANCE: SpawnEvent(towerId, TEAM_INDEX_ALLIANCE, true); break;
394  case HORDE: SpawnEvent(towerId, TEAM_INDEX_HORDE, true); break;
395  case TEAM_NONE: SpawnEvent(towerId, TEAM_INDEX_NEUTRAL, true); break;
396  default: break;
397  }
398 }

400 // Function that handles area triggers; used to check for flag carriers
401 bool BattleGroundEY::HandleAreaTrigger(Player* source, uint32 trigger)
402 {
403  if (GetStatus() != STATUS_IN_PROGRESS)
404  return false;
405 
406  if (!source->IsAlive()) // hack code, must be removed later
407  return false;
408 
409  switch (trigger)
410  {
411  case AREATRIGGER_BLOOD_ELF_TOWER_POINT:
412  if (m_towerOwner[NODE_BLOOD_ELF_TOWER] == source->GetTeam())
413  ProcessPlayerFlagScoreEvent(source, NODE_BLOOD_ELF_TOWER);
414  break;
415  case AREATRIGGER_FEL_REAVER_RUINS_POINT:
416  if (m_towerOwner[NODE_FEL_REAVER_RUINS] == source->GetTeam())
417  ProcessPlayerFlagScoreEvent(source, NODE_FEL_REAVER_RUINS);
418  break;
419  case AREATRIGGER_MAGE_TOWER_POINT:
420  if (m_towerOwner[NODE_MAGE_TOWER] == source->GetTeam())
421  ProcessPlayerFlagScoreEvent(source, NODE_MAGE_TOWER);
422  break;
423  case AREATRIGGER_DRAENEI_RUINS_POINT:
424  if (m_towerOwner[NODE_DRAENEI_RUINS] == source->GetTeam())
425  ProcessPlayerFlagScoreEvent(source, NODE_DRAENEI_RUINS);
426  break;
427  default:
428  return false;
429  }
430  return true;
431 }
